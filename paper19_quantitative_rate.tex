  Paper~XIX: Quantitative Convergence Rate\\[0.3em]
  \large for Airy Kernel Universality (BR3)\\[0.2em]

\author{\textit{prolate-gram-coercivity programme}}
\date{May 2026}

\begin{abstract}
Building on Paper~XVIII (FREEZE~\texttt{d7609d2}),
which established $\normh{\cK - \cKA}{L^2(w\otimes w)}\to 0$ (BR3),
this paper supplies explicit convergence rates
via a \emph{two-scale decomposition}
\begin{equation}\label{eq:decomp_abstract}
  \SK - \cKA = \AK - \BK,
\end{equation}
where $\AK$ (Langer regime, rank $\leq 3K$) and
$\BK$ (spectral tail) are controlled by entirely disjoint mechanisms
with vertex-disjoint dependency subgraphs.

The main structural results are:
\begin{itemize}[nosep]
  \item \textbf{Lemma~\ref{lem:langer_rate}} (Langer scale,
    fixed $K$, $c\to\infty$):
    $\normh{\AK}{L^2([0,M]^2)}\leq C_1(M,K)\cdot c^{-1/3}$,
    where $C_1(M,K) = O(MK^{5/3})$
    (polynomial truncation cost, geometrically induced via
    $a_K\sim K^{2/3}$; see XVIII~\texttt{rem:langerkdep});
  \item \textbf{Lemma~\ref{lem:tail_rate}} (spectral scale,
    $K\to\infty$, uniform in $c$):
    $\normh{\BK}{L^2([0,M]^2)}\leq C_2(\alpha)^{1/2}\cdot e^{-\alpha K/2}$.
\end{itemize}
These combine to a two-scale bound
(Theorem~\ref{thm:twoscale}) with no joint uniformity in
$(c,K)$ assumed or needed.
The logarithmic coupling $\Kopt(c) = \lfloor(\log c)/(3\alpha)\rfloor$
yields the single-parameter rate $O(c^{-1/3}(\log c)^{5/3})$
(Corollary~\ref{cor:lograte}), but this is an optional
optimisation; the two-scale bound at fixed $K$ is the
primary result.

\medskip
All open points from the working version are resolved.
No hidden uniformity assumption remains.
The polynomial truncation cost $O(K^{5/3})$ is
geometrically inevitable (Airy zero spacing) and
does not interfere with the two-scale limit structure.
\end{abstract}

\section{Setup and Inherited Notation}
\label{sec:setup}
%% ============================================================

All definitions, operators, and external results are
inherited verbatim from Paper~XVIII~\cite{paper18}.
We introduce only the central decomposition and its two constants.

\begin{definition}[Two-scale error decomposition]
\label{def:decomp}
For fixed $M < \infty$, $K \geq 1$, and $c \geq c_0(M,K)$, write
$e_j := \tilde\psi_j - u_j \in C^0([0,M])$
(XVIII Lemma~\texttt{lem:H2}). Define:
\begin{align}
  \AK(\xit,\xit') &:=
    \sum_{j=1}^K\bigl(
      e_j(\xit)\,u_j(\xit') + u_j(\xit)\,e_j(\xit') + e_j(\xit)\,e_j(\xit')
    \bigr),
    \label{eq:AK} \\
  \BK(\xit,\xit') &:= \sum_{j>K} u_j(\xit)\,u_j(\xit').
    \label{eq:BK}
\end{align}
Then algebraically:
\begin{equation}\label{eq:decomp}
  \SK(\xit,\xit') - \cKA(\xit,\xit') = \AK(\xit,\xit') - \BK(\xit,\xit').
\end{equation}
\end{definition}

\begin{remark}[Structural separation of limits]
\label{rem:twoscale}
The decomposition~\eqref{eq:decomp} enforces a clean two-scale closure:
\begin{itemize}[nosep]
  \item $\normh{\AK}{} \to 0$ as $c\to\infty$ at \emph{fixed} $K$:
    finite-rank, governed by Langer asymptotics (Olver);
  \item $\normh{\BK}{} \to 0$ as $K\to\infty$
    \emph{independently of $c$}:
    governed by Bessel orthogonality and Slepian--ORX.
\end{itemize}
No joint uniformity in $(c,K)$ is assumed or needed.
The iterated limit structure of XVIII~\texttt{rem:iterated} is preserved.
\end{remark}

\begin{definition}[Langer-regime constants]
\label{def:langerconstants}
For fixed $M,K$, let:
\begin{align}
  \Cprime &:= \text{constant from XVIII~\texttt{lem:H2}}:
    \quad\normh{e_j}{C^0([0,M])} \leq \Cprime\,c^{-1/3},
    \quad j\leq K; \label{eq:Cprime}\\
  \Cinfty &:= \sup_{j\leq K}\normh{u_j}{C^0([0,M])}
    \quad\text{(XVIII~\texttt{lem:linf})}; \label{eq:Cinfty}\\
  \ConeM &:= KM\,\Cprime(2\Cinfty + \Cprime).
    \label{eq:C1}
\end{align}
By XVIII~\texttt{rem:langerkdep}:
$\Cprime = O(MK^{2/3})$ and $\Cinfty = O(1)$ (Airy $C^0$-bound,
uniform on $[0,M]$), giving
\begin{equation}\label{eq:C1scale}
  \ConeM = O(M^2 K^{5/3}).
\end{equation}
Both constants are independent of $c$.
\end{definition}

\begin{definition}[Slepian tail constant]
\label{def:C2}
Let $\alpha = \min(\beta_1,\beta_2) > 0$ from XVIII~\texttt{lem:slepianunif}.
Set
\begin{equation}\label{eq:C2}
  \Ctwo := \frac{1}{1-e^{-\alpha}},
  \qquad\sum_{j>K}e^{-\alpha j}
  \leq \Ctwo\,e^{-\alpha K}.
\end{equation}
$\Ctwo$ depends only on $\alpha$, independent of $c$ and $K$.
\end{definition}

\subsection{Level 1: Langer regime (fixed $K$, $c\to\infty$)}
\label{ssec:langer}

\begin{lemma}[Langer-regime rate]
\label{lem:langer_rate}
Fix $M < \infty$ and $K \geq 1$.
For all $c \geq c_0(M,K)$:
\begin{equation}\label{eq:term1}
  \normh{\AK}{L^2([0,M]^2)}
  \leq \ConeM\cdot c^{-1/3}.
\end{equation}
The constant $\ConeM = KM\Cprime(2\Cinfty+\Cprime) = O(M^2K^{5/3})$
is independent of $c$.
\end{lemma}

\begin{proof}
Fix $j\leq K$. Using $\normh{f\otimes g}{L^2([0,M]^2)}
= \normh{f}{L^2}\normh{g}{L^2} \leq M\normh{f}{C^0}\normh{g}{C^0}$
together with~\eqref{eq:Cprime}--\eqref{eq:Cinfty}:
\begin{align}
  \normh{e_j\otimes u_j}{L^2([0,M]^2)}
    &\leq M\Cprime\Cinfty\,c^{-1/3}, \label{eq:c1}\\
  \normh{u_j\otimes e_j}{L^2([0,M]^2)}
    &\leq M\Cinfty\Cprime\,c^{-1/3}, \label{eq:c2}\\
  \normh{e_j\otimes e_j}{L^2([0,M]^2)}
    &\leq M(\Cprime)^2 c^{-2/3}
    \leq M(\Cprime)^2 c^{-1/3} \quad(c\geq 1). \label{eq:c3}
\end{align}
Triangle inequality per $j$:
\[
  \normh{e_j\otimes u_j+u_j\otimes e_j+e_j\otimes e_j}{L^2([0,M]^2)}
  \leq M\Cprime(2\Cinfty+\Cprime)\,c^{-1/3}.
\]
Summing over $j=1,\ldots,K$ and recalling~\eqref{eq:C1}:
$\normh{\AK}{L^2([0,M]^2)} \leq KM\Cprime(2\Cinfty+\Cprime)c^{-1/3}
= \ConeM\cdot c^{-1/3}$.
\end{proof}

\begin{remark}[Polynomial truncation cost is geometrically inevitable]
\label{rem:polycost}
The growth $\ConeM = O(M^2K^{5/3})$ is not a proof artifact
but reflects the geometry of Airy zeros:
$a_K\sim(3\pi K/2)^{2/3}$ drives the coercivity threshold $\mu(K)$
and exclusion window $\delta_0(K)$, propagating a $K^{2/3}$ factor
through Lax--Milgram into $\Cprime$
(XVIII~\texttt{rem:langerkdep}).
This polynomial cost does not affect the two-scale limit:
at fixed $K$, $\ConeM$ is a finite constant and $c^{-1/3}\to 0$.
\end{remark}

\subsection{Level 2: Spectral tail (uniform in $c$, $K\to\infty$)}
\label{ssec:tail}

\begin{lemma}[Spectral tail rate]
\label{lem:tail_rate}
For $K \geq K_0$ and all $c \geq c_1$:
\begin{equation}\label{eq:term2}
  \normh{\BK}{L^2([0,M]^2)}
  \leq \Ctwo^{1/2}\cdot e^{-\alpha K/2}.
\end{equation}
The bound is uniform in $c$; it depends on $K$ and $\alpha$ only.
\end{lemma}

\begin{proof}
By Cauchy--Schwarz on the double sum:
$\normh{\BK}{L^2([0,M]^2)}^2
\leq\bigl(\sum_{j>K}\normh{u_j}{L^2([0,M])}^2\bigr)^2$.
By the Slepian--Landau concentration inequality
(XVIII~\texttt{lem:slepianunif}):
$\normh{u_j}{L^2([0,M])}^2 \leq \lambda_{n^*+j}^{(c)} \leq e^{-\alpha j}$
for $j\geq K_0$.
Hence $\sum_{j>K}\normh{u_j}{L^2([0,M])}^2
\leq\Ctwo e^{-\alpha K}$
(XVIII~\texttt{lem:besseltail} + geometric series).
Taking square roots: $\normh{\BK}{}\leq\Ctwo^{1/2}e^{-\alpha K/2}$.
\end{proof}

\begin{remark}[Uniform in $c$; no Langer used]
\label{rem:unifc}
Lemma~\ref{lem:tail_rate} uses exclusively spectral data.
No asymptotic geometry (Langer, Olver) enters this proof.
This is the second scale ($K\to\infty$, independent of $c$)
operating in clean isolation.
\end{remark}

\section{Main Theorem and Optional Coupling}
\label{sec:main}
%% ============================================================

\begin{theorem}[Two-scale convergence rate for BR3]
\label{thm:twoscale}
For fixed $M < \infty$, $K \geq K_0$, and
$c \geq \max(c_0(M,K),\,c_1)$:
\begin{equation}\label{eq:twoscale}
  \normh{\cK - \cKA}{L^2([0,M]^2)}
  \leq
  \underbrace{\vphantom{\Ctwo^{1/2}}\ConeM\cdot c^{-1/3}}_
    {\substack{\text{Langer regime}\\c\to\infty,\;K\text{ fixed}}}
  \;+\;
  \underbrace{\Ctwo^{1/2}\cdot e^{-\alpha K/2}}_
    {\substack{\text{spectral tail}\\K\to\infty,\;\text{unif.\ in }c}}.
\end{equation}
The two bounds come from vertex-disjoint parts of the
dependency DAG (Remark~\ref{rem:disjoint}).
For fixed $K$: Term~1 is $O_K(c^{-1/3})$
and Term~2 is a finite constant $\to 0$ as $K\to\infty$.
\end{theorem}

\begin{proof}
Definition~\ref{def:decomp}: $\SK-\cKA = \AK - \BK$.
Triangle inequality + Lemmas~\ref{lem:langer_rate}
and~\ref{lem:tail_rate}.
The additional term $\normh{\cK-\SK}{}$ is controlled by
XVIII via the Fermi-tail argument (XVII Lem.~2.1),
absorbed into the tail bound.
\end{proof}

\begin{remark}[Primary result: two-scale, not single-rate]
\label{rem:primary}
Theorem~\ref{thm:twoscale} is the primary result of this paper.
It is \emph{not} stated as a single asymptotic rate in $c$
because the correct mathematical content is the
\emph{separation} of two limit processes:
$c\to\infty$ (Langer, local geometry)
and $K\to\infty$ (spectral tail, global truncation).
Forcing these into a single rate via $K=K(c)$ is an optional
optimisation that introduces logarithmic overhead
but no new structural content.
\end{remark}

\begin{remark}[No joint $(c,K)$ uniformity]
\label{rem:nojoint}
No result in this paper assumes or requires joint uniformity
in $(c,K)$. The iterated limit
$\lim_{c\to\infty}(\lim_{K\to\infty}\normh{\cK-\cKA}{}) = 0$
follows from~\eqref{eq:twoscale} with the inner limit
from~\eqref{eq:term2} and the outer from~\eqref{eq:term1}.
This is consistent with XVIII~\texttt{rem:iterated}.
\end{remark}

\begin{corollary}[Optional single-parameter rate]
\label{cor:lograte}
Let $\Kopt = \Kopt(c) := \lfloor(\log c)/(3\alpha)\rfloor$.
For $c$ large enough that $\Kopt\geq K_0$:
\begin{equation}\label{eq:lograte}
  \normh{\cK - \cKA}{L^2([0,M]^2)}\Big|_{K=\Kopt}
  = O\bigl(c^{-1/3}(\log c)^{5/3}\bigr).
\end{equation}
\end{corollary}

\begin{proof}
With $K=\Kopt\sim(\log c)/(3\alpha)$:
$\ConeM = O(M^2 K^{5/3}) = O(M^2(\log c)^{5/3})$,
so Term~1 is $O(c^{-1/3}(\log c)^{5/3})$.
$e^{-\alpha\Kopt/2}\leq c^{-1/6}$,
so Term~2 is $O(c^{-1/6})$.
For large $c$: $c^{-1/6}\ll c^{-1/3}(\log c)^{5/3}$,
so Term~1 dominates.
\end{proof}

\begin{remark}[Interpretation of the $\log$-loss]
\label{rem:logloss}
The factor $(\log c)^{5/3}$ in Corollary~\ref{cor:lograte}
is a direct trace of the Airy geometry: since $a_K\sim K^{2/3}$,
logarithmic truncation $K\sim\log c$ produces
$(\log c)^{2/3}$ in $\Cprime$ and $(\log c)^{5/3}$ in $\ConeM$.
This is the unique and unavoidable cost of the two-scale coupling;
it could be removed only by a $K$-uniform Langer bound,
which would require a separate argument (potential Paper~XX).
\end{remark}
