\title[Gap-S: Obstruction Analysis]{%
  Paper~XIII: Obstructions to Gap-S\\[0.3em]
  \large A Negative Reduction Analysis of the Spectral Gap Problem
  for Prolate Concentration Operators}

\author{\textit{prolate-gram-coercivity programme}}
\date{May 2026}

\begin{abstract}
This paper provides a \emph{negative reduction analysis} of the spectral
gap problem (Gap-S) for the prolate concentration operator
\[
  (\mathcal{K}_c f)(x) := \int_{-1}^1
  \frac{\sin(c(x-y))}{\pi(x-y)}\,f(y)\,dy
\]
in the transition regime $n \sim 2c/\pi$.
The primary contribution is not a proof of Gap-S,
but a precise identification of why all previous approaches fail.

\medskip\noindent
\textbf{Central new insight.}
All currently known approaches reduce Gap-S to a local spectral
counting problem in a $c^{-1/3}$-scale transition window:
the question of whether that window contains exactly two eigenvalues
of $\Kc$, no more and no fewer.
This counting problem --- condition~(Q5) --- is a Widom-type spectral
rigidity question, not a perturbative consequence of quasimode analysis.
It is the genuine obstruction.

\medskip\noindent
\textbf{Three structural results.}
\begin{itemize}
\item \emph{(Proposition~\ref{prop:gram-fails}.)} Gram coercivity
  does not imply spectral gaps:
  coordinate stability is not spectral rigidity.
\item \emph{(Proposition~\ref{prop:circular}.)} Every approach based
  on $\Kc$-invariant trial spaces is circularly dependent on Gap-S.
  This is a fixed-point obstruction and a
  \emph{no-go theorem for an entire class of methods}.
  It provides a unified explanation for the failure of Papers~X--XII.
\item \emph{(Remark~\ref{rem:Q5}.)} Condition~(Q5) is the primary
  obstruction: a local Widom-type counting problem, not a technical
  auxiliary condition.
\end{itemize}

\medskip\noindent
\textbf{Reduction to minimal inputs.}
Gap-S $\Longleftarrow$ microlocal spectral geometry $+$ local counting
rigidity, not Gap-S $\Longleftarrow$ linear algebra.
Three logically independent ingredients are isolated.
None suffices alone.

\medskip\noindent
\textbf{Conditional outputs} (value: precise problem formulation only).
\begin{itemize}
\item \emph{Theorem~A}: a lifting device; no genuine spectral
  information beyond its assumptions.
\item \emph{Theorem~B}: the more structurally sound route;
  avoids invariant spaces and Proposition~\ref{prop:circular}.
\item \emph{Corollary~C}: a roadmap consequence at scale $c^{-1/3}$.
\end{itemize}
\end{abstract}

\section{Operator Setup}
\label{sec:setup}
%% ============================================================

\begin{definition}[Prolate concentration operator]
$(\Kc f)(x) := \int_{-1}^1 \tfrac{\sin(c(x-y))}{\pi(x-y)} f(y)\,dy$;
eigenvalues $1 > \lam{c}{0} \geq \lam{c}{1} \geq \cdots > 0$.
\end{definition}

\begin{definition}
$\gap{c}{n} := \lam{c}{n} - \lam{c}{n+1}$,\quad
$V_N := \spann\{\pswf{0},\ldots,\pswf{N}\}$.
\end{definition}

\begin{lemma}[Ritz values and one-sided min-max bound]
\label{lem:compression}
Let $A: H \to H$ be compact self-adjoint with
eigenvalues $\lambda_0(A) \geq \lambda_1(A) \geq \cdots \to 0$,
and let $E \subset H$ be an $N$-dimensional subspace
with compression $T_E := P_E A|_E$, eigenvalues
$\sigma_0 \geq \cdots \geq \sigma_{N-1}$.
Then:
\begin{equation}\label{eq:ritz-upper}
  \lambda_j(A) \geq \sigma_j(T_E) \qquad (j = 0,\ldots,N-1).
\end{equation}
Ritz values $\sigma_j$ are not in general spectral points of $A$.
\end{lemma}

\begin{proof}
The min-max principle gives
$\lambda_j(A) = \max_{\dim S = j+1}\min_{u\in S,\|u\|=1}\ip{Au}{u}$
over all $(j{+}1)$-dimensional subspaces of $H$.
Restricting the maximum to subspaces of $E \subset H$ yields $\sigma_j$,
and the restricted maximum cannot exceed the unrestricted one.
\end{proof}

\section{Gram Coercivity as Stability Input}
\label{sec:gram}
%% ============================================================

\begin{assumption}[Gram coercivity --- Paper~I]
\label{ass:gram}
$\lambda_{\min}(G_{c,N}) \asymp c^{-1/2}$ for all large $c$.
(Proved unconditionally.)
\end{assumption}

\begin{proposition}[Gram coercivity does not imply spectral gaps]
\label{prop:gram-fails}
For any $\alpha > 0$ and $N \geq 2$, there exists a self-adjoint
operator on an $N$-dimensional space with $\lambda_{\min}(G) \geq \alpha$
but all consecutive eigenvalue gaps equal to zero.
\end{proposition}

\begin{proof}
$T := \mu \cdot \mathrm{Id}$ on any orthonormal basis:
$G = I$, gaps zero.
For general $\alpha$: scale basis by $\sqrt{\alpha}$;
$G = \alpha I$, gaps still zero.
\end{proof}

\begin{remark}[Coordinate stability $\neq$ spectral rigidity]
\label{rem:gram-limit}
Gram coercivity gives: coordinate stability
and Ritz-value upper bounds (Lemma~\ref{lem:compression}).
It does not give spectral gaps.
The additional required inputs are
Assumptions~\ref{ass:nondeg} and~\ref{ass:window}.
\end{remark}

\section{Conditional Transfer Principle (Theorem~A)}
\label{sec:transfer}
%% ============================================================

\begin{remark}[Status of Theorem~A]
\label{rem:thmA-status}
Theorem~A is a \emph{lifting device} containing no genuine spectral
information beyond its assumptions.
All content lies in Assumptions~\ref{ass:nondeg} and~\ref{ass:window}.
The transfer step is elementary.
The value of Theorem~A is solely in the logical separation of the
three required inputs.
\end{remark}

\subsection{Assumptions}

\begin{assumption}[Local spectral nondegeneracy]
\label{ass:nondeg}
For $T_{c,N} := P_{E_{c,N}}\Kc|_{E_{c,N}}$, there exists a basis of
$E_{c,N}$ in which
$\|T_{c,N} - \mathrm{diag}(\mu_j^{(c)})\|_{\op} \leq \delta(c)$,
with $\min_j|\mu_j^{(c)} - \mu_{j+1}^{(c)}| \geq \delta_{\mathrm{model}}(c)$
and $\delta(c) = o(\delta_{\mathrm{model}}(c))$.
\end{assumption}

\begin{remark}[Non-invariant spaces required]
For $E_{c,N} = V_N$ the assumption holds trivially with
$\mu_j = \lam{c}{j}$ but $\delta_{\mathrm{model}}$ equals the gap
(circular; see Proposition~\ref{prop:circular}).
For a non-invariant $\widetilde{V}_N$, verification requires:
(1) microlocal normal form; (2) off-diagonal control via
noncommutativity of $\Pi_{\mathrm{space}}, \Pi_{\mathrm{phase}}$;
(3) semiclassical symbol separation. Programme for Paper~XIV.
\end{remark}

\begin{assumption}[Spectral window isolation]
\label{ass:window}
$W_c := [\mu_{n+1}^{(c)} - \delta(c),\, \mu_n^{(c)} + \delta(c)]$
contains exactly two eigenvalues of $\Kc$:
$\lam{c}{n}$ and $\lam{c}{n+1}$.
\end{assumption}

\begin{remark}[Roles of the three assumptions]
\label{rem:three-roles}
\begin{center}
\renewcommand{\arraystretch}{1.2}
\begin{tabular}{lll}
\toprule
\textbf{Assumption} & \textbf{Type} & \textbf{Role} \\
\midrule
Gram coercivity (\ref{ass:gram}) & Stability & Coordinate non-collapse \\
Nondegeneracy (\ref{ass:nondeg}) & Local geometry & Forces compressed gap \\
Window isolation (\ref{ass:window}) & Index identification & Lifts to PSWF gap \\
\bottomrule
\end{tabular}
\end{center}
None suffices alone.
\end{remark}

\subsection{The compressed-to-global gap lemma}

\begin{lemma}[Compressed-to-global gap transfer]
\label{lem:comptoglobal}
Let $A$ compact self-adjoint, $E$ $N$-dimensional with Ritz values
$\sigma_0 \geq \cdots \geq \sigma_{N-1}$.
For index $n$, $\Delta > 0$, $\rho \geq 0$, suppose:
\begin{enumerate}[label=(C\arabic*),nosep]
  \item $\sigma_n - \sigma_{n+1} \geq \Delta$;\label{C1}
  \item $W := [\sigma_{n+1} - \rho,\, \sigma_n + \rho]$
    contains exactly two eigenvalues of $A$:
    $\lambda_p \geq \lambda_{p+1}$.\label{C2}
\end{enumerate}
Then $\lambda_p - \lambda_{p+1} \geq \Delta - 2\rho$.
\end{lemma}

\begin{proof}
By Lemma~\ref{lem:compression}, $\sigma_n \leq \lambda_n(A)$
and $\sigma_{n+1} \leq \lambda_{n+1}(A)$, so both Ritz values lie in $W$.
Since $W$ contains exactly $\lambda_p$ and $\lambda_{p+1}$,
the window bounds give
$\lambda_p \leq \sigma_n + \rho$ and $\lambda_{p+1} \geq \sigma_{n+1} - \rho$.
Subtracting and using~(C1): $\lambda_p - \lambda_{p+1} \geq \Delta - 2\rho$.
\end{proof}

\begin{remark}[On $\rho > 0$]
\label{rem:rho}
Ritz values are not spectral points of $A$ (Lemma~\ref{lem:compression}),
so $\rho = 0$ is unjustified in general.
In Theorem~A we set $\rho = \delta(c) > 0$,
which is $o(\delta_{\mathrm{model}})$ but not zero.
\end{remark}

\begin{remark}[Index identification]
\label{rem:index}
Lemma~\ref{lem:comptoglobal} establishes the gap between the two
eigenvalues of $A$ in $W$ but does not identify $p = n$.
This is the content of Assumption~\ref{ass:window}.
\end{remark}

\subsection{The theorem}

\begin{theorem}[Conditional Transfer Principle --- Theorem~A]
\label{thm:A}
Under Assumptions~\ref{ass:gram}, \ref{ass:nondeg}, \ref{ass:window}:
$\gap{c}{n} \geq \tfrac{1}{2}\delta_{\mathrm{model}}(c)$
for all large $c$ and transition-window $n$.
\end{theorem}

\begin{proof}
\textit{Step~1.} Weyl~$+$~Ass.~\ref{ass:nondeg}:
$\sigma_n^{(c)}-\sigma_{n+1}^{(c)} \geq \delta_{\mathrm{model}}-2\delta(c) =: \Delta$.
\textit{Step~2.} Apply Lem.~\ref{lem:comptoglobal} with $\rho=\delta(c)$
and Ass.~\ref{ass:window} as~(C2):
$\gap{c}{n} \geq \delta_{\mathrm{model}} - 4\delta(c)$.
\textit{Step~3.} $\delta = o(\delta_{\mathrm{model}})$.
\end{proof}

\begin{proposition}[Invariant-space circularity --- no-go theorem]
\label{prop:circular}
Let $E_{c,N} = V_N$ (a $\Kc$-invariant subspace).
Then:
\begin{enumerate}[nosep]
  \item Assumption~\ref{ass:nondeg} holds trivially with
    $\mu_j^{(c)} = \lam{c}{j}$ and $\delta(c)=0$,
    but $\delta_{\mathrm{model}}^{(c)} = \gap{c}{n}$.
  \item Assumption~\ref{ass:window} reduces to the statement that
    the window $[\lam{c}{n+1}, \lam{c}{n}]$ contains exactly
    $\lam{c}{n}$ and $\lam{c}{n+1}$ --- which is Gap-S.
\end{enumerate}
Hence for every $\Kc$-invariant trial space,
both non-trivial assumptions of Theorem~A reduce to Gap-S itself.
No argument based solely on $\Kc$-invariant trial spaces can
establish Gap-S without assuming it.
\end{proposition}

\begin{proof}
For $E_{c,N} = V_N$, the compression $T_{c,N} = \Kc|_{V_N}$
is already diagonal in the PSWF basis with eigenvalues $\lam{c}{j}$.
Point~(1): Assumption~\ref{ass:nondeg} is satisfied with
$\mu_j = \lam{c}{j}$, $\delta = 0$; the model gap is
$\min_j |\mu_j - \mu_{j+1}| = \gap{c}{n}$.
Point~(2): the window $W_c$ becomes
$[\lam{c}{n+1} - 0, \lam{c}{n} + 0] = [\lam{c}{n+1}, \lam{c}{n}]$;
the assertion that this window contains exactly two eigenvalues
of $\Kc$ is precisely the statement $\gap{c}{n} > 0$ with
no additional eigenvalue inside, i.e.\ Gap-S.
\end{proof}

\begin{remark}[Scope of Proposition~\ref{prop:circular}]
\label{rem:circular-scope}
Proposition~\ref{prop:circular} is a no-go theorem
for the following class of methods:
\begin{itemize}[nosep]
  \item exact diagonalisation in $V_N$;
  \item compression/Galerkin in any $\Kc$-invariant subspace;
  \item Ritz approximation from within $V_N$;
  \item direct coercivity in invariant settings (Papers~X--XII).
\end{itemize}
The non-circular route requires a non-$\Kc$-invariant trial space
$\widetilde{V}_N$ (phase-space truncation, Paper~XIV).
\end{remark}

\section{Conditional Quasimode Separation (Theorem~B)}
\label{sec:quasimode}
%% ============================================================

Theorem~B uses approximate eigenfunctions directly,
avoiding invariant spaces and Proposition~\ref{prop:circular}.
It is the most structurally sound conditional result in the paper.

\begin{definition}[Quasimode family]
$\|(\Kc - \mu_n^{(c)})f_n^{(c)}\| \leq \varepsilon(c)$,
$\|f_n^{(c)}\| = 1$.
\end{definition}

\begin{assumption}[Local quasimode package]
\label{ass:quasimode}
\begin{enumerate}[label=(Q\arabic*),nosep]
  \item $\|f_n^{(c)}\| = 1$;
  \item $\|(\Kc - \mu_n^{(c)})f_n^{(c)}\| \leq \varepsilon(c)$;
  \item $|\mu_n^{(c)} - \mu_{n+1}^{(c)}| \geq \delta_{\mathrm{model}}(c)$;
  \item $\|G_n^{(c)} - I\|_{\op} \leq \eta(c) < 1$
    (linear independence only; much weaker than Gram coercivity);
  \item \textbf{[Primary difficulty]}
    $[\mu_{n+1}^{(c)} - \varepsilon(c),\, \mu_n^{(c)} + \varepsilon(c)]$
    contains exactly two eigenvalues of $\Kc$.
\end{enumerate}
\end{assumption}

\begin{remark}[On the depth of (Q5)]
\label{rem:Q5}
(Q5) is the local spectral counting statement $N(W) = 2$
in the transition window.
It is not a perturbative consequence of quasimode analysis.
It requires simultaneously an upper counting bound
(no extra eigenvalue crowds the window)
and a lower counting bound (no eigenvalue escapes).
This is a local spectral rigidity problem in the regime
$n - 2c/\pi \sim c^{1/3}$, of Widom-type character.
All currently known approaches to Gap-S reduce to this condition.
It is the deepest open problem identified in this paper.
\end{remark}

\begin{lemma}[Quasimode window separation]
\label{lem:qmsep}
Let $A$ compact self-adjoint, $\mu_n > \mu_{n+1}$,
$\varepsilon$-quasimodes $f_n, f_{n+1}$ with $\|G-I\|_{\op} < 1$.
If $[\mu_{n+1}-\varepsilon,\, \mu_n+\varepsilon]$ contains exactly two
eigenvalues of $A$ (condition~(Q5)), then
$\lambda_n(A) - \lambda_{n+1}(A) \geq (\mu_n-\mu_{n+1}) - 2\varepsilon$.
\end{lemma}

\begin{proof}
For $j \in \{n,n+1\}$: $\|(A-\mu_j)f_j\| \leq \varepsilon$, $\|f_j\|=1$
imply $\mathrm{dist}(\mu_j,\sigma(A)) \leq \varepsilon$.
$\|G-I\|<1$ implies linear independence.
(Q5) gives exact count and index identification.
Ordering and subtracting gives the bound.
\end{proof}

\begin{theorem}[Conditional Quasimode Separation --- Theorem~B]
\label{thm:B}
Under Assumption~\ref{ass:quasimode}:
$\gap{c}{n} \geq \delta_{\mathrm{model}}(c) - 2\varepsilon(c)$.
If $\varepsilon = o(\delta_{\mathrm{model}})$:
$\gap{c}{n} \geq \tfrac{1}{2}\delta_{\mathrm{model}}(c)$.
\end{theorem}

\begin{proof}
Lemma~\ref{lem:qmsep} with $A = \Kc$.
\end{proof}

%% ============================================================
\section{Airy-Scale Roadmap Consequence}
\label{sec:airy}
%% ============================================================

\begin{assumption}[Airy transition asymptotics]
\label{ass:airy}
Assumption~\ref{ass:quasimode} holds with
$\delta_{\mathrm{model}} \asymp c^{-1/3}$ and $\varepsilon = o(c^{-1/3})$
in the regime $n \sim 2c/\pi$.
\end{assumption}

\begin{remark}[Status]
The scale $c^{-1/3}$ is expected from semiclassical
turning-point heuristics.
Proving it requires: (i) asymptotic analysis of the PSWF kernel;
(ii) Airy quasimodes with residual $o(c^{-1/3})$;
(iii) (Q5) in this regime. None is established.
\end{remark}

\begin{corollary}[Roadmap consequence --- Corollary~C]
\label{cor:C}
Under Assumption~\ref{ass:airy}:
$\gap{c}{n} \geq Cc^{-1/3}$ for all large $c$.
\end{corollary}

\begin{remark}
Corollary~C is a roadmap consequence, not a theorem.
The originality of this paper lies in
Propositions~\ref{prop:gram-fails} and~\ref{prop:circular}
and Remark~\ref{rem:Q5}.
\end{remark}
