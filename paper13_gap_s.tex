\title[Gap-S: Reduction Framework]{%
  Paper~XIII: A Reduction Framework for Gap-S\\[0.3em]
  \large Gram Coercivity as Stability Input,
  Quasimode Transfer as Gap Mechanism}

\author{\textit{prolate-gram-coercivity programme}}
\date{May 2026}

\begin{abstract}
This paper establishes a reduction framework for the spectral gap problem
(Gap-S) for the prolate concentration operator
\[
  (\mathcal{K}_c f)(x) := \int_{-1}^1
  \frac{\sin(c(x-y))}{\pi(x-y)}\,f(y)\,dy
\]
in the transition regime $n \sim 2c/\pi$.

\medskip
\textbf{Central conceptual clarification.}
Gram coercivity ($\lambda_{\min}(G_N) \asymp c^{-1/2}$, Paper~I)
is a \emph{stability input}, not a direct source of eigenvalue repulsion.
It controls the geometry of the trial basis and prevents coordinate collapse,
but does not by itself force spectral gaps.
A self-adjoint operator can have perfectly conditioned Gram matrix
and still possess degenerate (or near-degenerate) spectrum;
the trivial example is $T_E = \lambda I$ on any orthonormal space.

\medskip\noindent
\textbf{Main outputs.}
\begin{itemize}
\item \emph{(Theorem~A, Conditional Transfer.)} Under Gram coercivity
  (Paper~I) \emph{and} an explicit compressed non-degeneracy hypothesis
  (Assumption~\ref{ass:nondeg}), one obtains
  $\gap{c}{n} \geq C c^{-1/2}$ for transition-window indices.
  Gram coercivity is the stability input; compressed non-degeneracy
  is the dynamical input that actually forces the gap.
\item \emph{(Theorem~B, Quasimode Transfer.)} Under local quasimode
  hypotheses, a sharp conditional bound at model scale follows.
  This is the most solid result in the paper.
\item \emph{(Corollary~C.)} Under Airy transition asymptotics,
  the conditional gap scale is $c^{-1/3}$.
\item \emph{(Open Problem~\ref{prob:nondeg})} identifies compressed
  non-degeneracy as the precise missing input needed to make
  Theorem~A fully unconditional.
\end{itemize}

\medskip\noindent
\textbf{Honest assessment.}
Theorem~A requires a compressed non-degeneracy hypothesis that is not
deducible from Gram coercivity alone.
Theorem~B is mathematically solid.
The paper's primary contribution is a precise structural reduction:
Gap-S reduces to a local non-degenerate compressed spectral model,
and Gram coercivity enters as stability, not as gap-forcing mechanism.
\end{abstract}

\subsection{Programme position}

This paper occupies the following position in the
\texttt{prolate-gram-coercivity} series:
\begin{itemize}[nosep]
  \item \textbf{Paper~I} (\texttt{paper1.tex}): Gram coercivity
    $\lambda_{\min}(G_N^{(c)}) \geq \alpha_N(c) \asymp c^{-1/2}$ --- proved.
  \item \textbf{Paper~IX} (\texttt{paper9\_superconcentration.tex}):
    B-strong $\Leftrightarrow$ spectral gap
    $\varepsilon_N^{(c)} \geq Cc^{-1/2}$.
  \item \textbf{Papers~X--XII}: All direct routes to B-strong
    (WKB, Fredholm, direct coercivity) fail; the obstruction is
    structural (\texttt{paper10--12.tex}).
  \item \textbf{This paper}: Gram coercivity from Paper~I is
    a stability input. Combined with compressed non-degeneracy,
    it transfers to a genuine spectral gap bound.
\end{itemize}

\subsection{The Gap-S problem}

Gap-S (companion series \texttt{prolate-primes-paper}, Paper~XII Hyp.~1)
asks for $g(c) > 0$ with
\begin{equation}\label{eq:gapS}
  |\lam{c}{n} - \lam{c}{n+1}| \geq g(c) \gg c^{-1/4},
  \quad n \sim 2c/\pi.
\end{equation}

\begin{remark}[Relation to B-strong]
\label{rem:bstrong}
By Paper~IX, B-strong is $\varepsilon_N^{(c)} \geq Cc^{-1/2}$
for $N = \lfloor Ac^{1/3} \rfloor \ll 2c/\pi$.
This is a specific instance of Gap-S at an index deep below the
transition window.
Theorem~A (under its two hypotheses) closes this instance.
\end{remark}

\section{Operator Setup}
\label{sec:setup}
%% ============================================================

\begin{definition}[Prolate concentration operator]
For $c > 0$, let $\Kc : L^2([-1,1]) \to L^2([-1,1])$ be the sinc-kernel operator.
Its eigenvalues $1 > \lam{c}{0} \geq \lam{c}{1} \geq \cdots > 0$
are the PSWF concentration eigenvalues; the eigenfunctions are $\pswf{n}$.
\end{definition}

\begin{definition}[Gap, residual, trial space]
\[
  \gap{c}{n} := \lam{c}{n} - \lam{c}{n+1},
  \qquad
  \varepsilon_n^{(c)} := 1 - \lam{c}{n},
  \qquad
  V_N := \spann\{\pswf{0},\ldots,\pswf{N}\}.
\]
\end{definition}

\begin{lemma}[Classical properties]
$\Kc$ is compact, self-adjoint, positive, $\|\Kc\|_{\op} \leq 1$,
Hilbert--Schmidt, with spectrum in $(0,1)$ accumulating only at $0$.
\end{lemma}

%% ============================================================
\section{Gram Coercivity as Stability Input}
\label{sec:gram}
%% ============================================================

\begin{assumption}[Gram coercivity --- Paper~I import]
\label{ass:gram}
There exist trial spaces $E_{c,N} \subset L^2([-1,1])$ with Gram matrices
$G_{c,N}$ such that
\[
  \lambda_{\min}(G_{c,N}) \geq \alpha_N(c) \asymp c^{-1/2}
\]
for all sufficiently large $c$.
Established unconditionally in \textbf{Paper~I} (\texttt{paper1.tex}).
\end{assumption}

\begin{lemma}[Gram invertibility --- coordinate stability]
\label{lem:gram-norm}
If $E = \spann\{\phi_1,\ldots,\phi_N\}$ with $\lambda_{\min}(G) \geq \alpha > 0$,
then for all $a \in \mathbb{C}^N$:
\[
  \Bigl\|\sum_j a_j \phi_j\Bigr\|^2 \geq \alpha\,\|a\|_{\ell^2}^2.
\]
The coordinate map is injective with stable inverse (norm $\leq \alpha^{-1/2}$).
\end{lemma}

\begin{proof}
$\|\sum a_j \phi_j\|^2 = a^* G a \geq \alpha \|a\|^2$.
\end{proof}

\begin{lemma}[Compression --- Rayleigh quotient identity]
\label{lem:compression}
For finite-dimensional $E \subset L^2([-1,1])$, the compression
$T_E := P_E \Kc|_E$ is self-adjoint on $E$ with
\[
  \frac{\ip{T_E u}{u}}{\|u\|^2} = \frac{\ip{\Kc u}{u}}{\|u\|^2}
  \qquad (u \in E \setminus \{0\}).
\]
\end{lemma}

\begin{remark}[What Gram coercivity does and does not give]
\label{rem:gram-limit}
Lemma~\ref{lem:gram-norm} gives \emph{coordinate stability}:
the trial vectors do not collapse linearly.
Lemma~\ref{lem:compression} identifies the compressed eigenvalues
with bona fide Rayleigh quotients of $\Kc$.

\medskip\noindent
However, Gram coercivity does \emph{not} force spectral gaps.
For any $\lambda \in (0,1)$ and any orthonormal basis $\{e_j\}$ of $E$,
the operator $T_E = \lambda \cdot \mathrm{Id}_E$ has
$\lambda_{\min}(G) = 1$ (perfect conditioning)
but all eigenvalues equal $\lambda$ (zero gaps).

\medskip\noindent
Conclusion: to obtain a spectral gap from Gram coercivity, one needs
an additional \emph{dynamical} or \emph{structural} input about
the compressed operator $T_{c,N}$.
This is made explicit in Assumption~\ref{ass:nondeg} below.
\end{remark}

%% ============================================================
\section{Conditional Transfer Theorem}
\label{sec:transfer}
%% ============================================================

\begin{assumption}[Compressed non-degeneracy]
\label{ass:nondeg}
For the compressed operator $T_{c,N} = P_{E_{c,N}}\Kc|_{E_{c,N}}$,
assume that in an appropriate basis the compressed symbol satisfies:
\begin{equation}\label{eq:nondeg}
  \|T_{c,N} - \mathrm{diag}(\mu_j^{(c)})\|_{\op} \leq \delta(c),
\end{equation}
where $\mu_j^{(c)}$ are approximate diagonal values with
\begin{equation}\label{eq:model-sep}
  \min_j |\mu_j^{(c)} - \mu_{j+1}^{(c)}| \geq \delta_{\mathrm{model}}(c),
  \qquad
  \delta(c) = o(\delta_{\mathrm{model}}(c)).
\end{equation}
\end{assumption}

\begin{remark}[Role of the two assumptions]
\label{rem:two-assumptions}
Assumption~\ref{ass:gram} (Gram coercivity) and
Assumption~\ref{ass:nondeg} (compressed non-degeneracy) play distinct roles:
\begin{itemize}[nosep]
  \item Gram coercivity ensures the trial basis is geometrically stable:
    coordinates do not collapse, and the compressed spectral problem
    is well-conditioned as a \emph{linear algebra problem}.
  \item Compressed non-degeneracy is the \emph{dynamical input}:
    it asserts that the compressed operator is approximately diagonal
    with separated approximate eigenvalues.
    This is what actually forces the spectral gap.
\end{itemize}
Neither assumption alone suffices.
Together they form a complete gap-forcing mechanism.
\end{remark}

\begin{theorem}[Conditional Transfer --- Theorem~A]
\label{thm:A}
Under Assumptions~\ref{ass:gram} and~\ref{ass:nondeg}, for all sufficiently
large $c$ and all relevant transition-window indices $n$:
\begin{equation}\label{eq:thmA}
  \gap{c}{n}
  \geq \delta_{\mathrm{model}}(c) - 2\delta(c)
  \geq \tfrac{1}{2}\delta_{\mathrm{model}}(c).
\end{equation}
In particular, if $\delta_{\mathrm{model}}(c) \asymp c^{-1/2}$, then
$\gap{c}{n} \geq Cc^{-1/2}$, which closes Gap-S and B-strong.
\end{theorem}

\begin{proof}
By Assumption~\ref{ass:nondeg}, the compressed operator
$T_{c,N}$ is $\delta(c)$-close in operator norm to
$\mathrm{diag}(\mu_j^{(c)})$.
By eigenvalue perturbation (Weyl's inequality), the eigenvalues of
$T_{c,N}$ are within $\delta(c)$ of $\mu_j^{(c)}$.
Since $|\mu_j^{(c)} - \mu_{j+1}^{(c)}| \geq \delta_{\mathrm{model}}(c)$
and $\delta(c) = o(\delta_{\mathrm{model}}(c))$,
adjacent eigenvalues of $T_{c,N}$ are separated by at least
$\delta_{\mathrm{model}}(c) - 2\delta(c) \geq \tfrac{1}{2}\delta_{\mathrm{model}}(c)$.
By Lemma~\ref{lem:compression}, these eigenvalues are Rayleigh quotients
of $\Kc$; the Courant--Fischer principle then gives~\eqref{eq:thmA}.
\end{proof}

\begin{remark}[Gram coercivity's role in Theorem~A]
Note that Gram coercivity (Assumption~\ref{ass:gram})
does not appear explicitly in the proof of Theorem~A.
Its role is structural:
it guarantees that the trial space $E_{c,N}$ is geometrically stable,
so that Assumption~\ref{ass:nondeg} (compressed non-degeneracy)
can be stated and verified without coordinate collapse artefacts.
Without Gram coercivity, the compressed symbol
$T_{c,N}$ is ill-conditioned as a finite-dimensional operator,
and \eqref{eq:nondeg} would not be a meaningful statement.
\end{remark}

\begin{remark}[Why the earlier version was invalid]
A previous version of this paper claimed that Gram coercivity alone
implied spectral gaps, via the argument:
``a collapse of adjacent eigenvalues of $T_{c,N}$ would force the
quadratic form to collapse, contradicting Gram coercivity.''
This argument is \emph{incorrect}.
The operator $T_E = \lambda I$ on any orthonormal basis has
$\lambda_{\min}(G) = 1$ (no collapse) but all gaps zero.
The error was conflating coordinate stability with spectral separation.
Theorem~A corrects this by making the dynamical input (Assumption~\ref{ass:nondeg})
explicit.
\end{remark}

\begin{problem}[Compressed non-degeneracy for PSWF]
\label{prob:nondeg}
Verify Assumption~\ref{ass:nondeg} for the PSWF trial spaces $E_{c,N}$
aligned with $V_N = \spann\{\pswf{0},\ldots,\pswf{N}\}$.
Specifically, show that the compressed operator $T_{c,N}$ is
approximately diagonal in an appropriate basis with model separation
$\delta_{\mathrm{model}}(c) \asymp c^{-1/2}$.

\medskip\noindent
Note: since $V_N$ is $\Kc$-invariant (Paper~XII), the compressed operator
on $V_N$ \emph{is} diagonal with eigenvalues $\lam{c}{0},\ldots,\lam{c}{N}$.
The compressed non-degeneracy assumption then reduces to:
$\min_{k \leq N} \gap{c}{k} \geq Cc^{-1/2}$,
i.e.\ to Gap-S itself.
This circularity confirms that Gram coercivity alone cannot close the
argument: a genuinely new input is required, either from quasimode
construction (Section~\ref{sec:quasimode}) or from a non-invariant
trial space (Option~B of Paper~XII).
\end{problem}

%% ============================================================
\section{Quasimode Transfer (the Solid Route)}
\label{sec:quasimode}
%% ============================================================

Theorem~B does not rely on compressed non-degeneracy.
Instead it uses local quasimodes that are approximately eigenfunctions
of $\Kc$ with explicitly separated approximate eigenvalues.
This is the most mathematically solid route in the paper.

\begin{definition}[Quasimode family]
A normalised family $f_n^{(c)} \in L^2([-1,1])$ is an $\varepsilon(c)$-quasimode
for $\Kc$ with parameters $\mu_n(c)$ if
$\|(\Kc - \mu_n(c))f_n^{(c)}\| \leq \varepsilon(c)$.
\end{definition}

\begin{assumption}[Local quasimode package]
\label{ass:quasimode}
There exists a transition-window family $\{f_n^{(c)}\}$ and $\mu_n(c)$ such that:
\begin{enumerate}[label=(Q\arabic*),nosep]
  \item $\|f_n^{(c)}\| = 1$;
  \item $\|(\Kc - \mu_n(c))f_n^{(c)}\| \leq \varepsilon(c)$;
  \item $|\mu_n(c) - \mu_{n+1}(c)| \geq \delta_{\mathrm{model}}(c)$;
  \item $\|G_n^{(c)} - I\|_{\op} \leq \eta(c) \to 0$
    (consecutive quasimodes nearly orthogonal);
  \item The window $[\mu_{n+1}(c) - \varepsilon,\, \mu_n(c) + \varepsilon]$
    contains exactly two eigenvalues of $\Kc$ (isolation condition).
\end{enumerate}
\end{assumption}

\begin{lemma}[Spectral index stability --- Kato--Temple type]
\label{lem:index}
Let $A$ be compact self-adjoint.
Suppose $\mu_n > \mu_{n+1}$ with $\varepsilon(c)$-quasimodes $f_n, f_{n+1}$
satisfying $\|G - I\|_{\op} \leq \eta < 1$.
If $[\mu_{n+1} - \varepsilon,\, \mu_n + \varepsilon]$ contains exactly two
eigenvalues of $A$, then:
\[
  \lambda_n(A) - \lambda_{n+1}(A)
  \geq (\mu_n - \mu_{n+1}) - 2\varepsilon.
\]
\end{lemma}

\begin{proof}
The spectral theorem gives $\mathrm{dist}(\mu_j, \sigma(A)) \leq \varepsilon$,
so the interval contains at least two eigenvalues.
The Gram condition implies $f_n, f_{n+1}$ are linearly independent,
giving an approximate spectral subspace of dimension two.
By the isolation hypothesis these are exactly $\lambda_n, \lambda_{n+1}$.
Ordering: $\lambda_n \geq \mu_n - \varepsilon$,
$\lambda_{n+1} \leq \mu_{n+1} + \varepsilon$; subtract.
\end{proof}

\begin{theorem}[Quasimode Transfer --- Theorem~B]
\label{thm:B}
Under Assumption~\ref{ass:quasimode}, for all sufficiently large $c$:
\[
  \gap{c}{n} \geq \delta_{\mathrm{model}}(c) - 2\varepsilon(c).
\]
If $\varepsilon(c) = o(\delta_{\mathrm{model}}(c))$,
then $\gap{c}{n} \geq \frac{1}{2}\delta_{\mathrm{model}}(c)$.
\end{theorem}

\begin{proof}
Apply Lemma~\ref{lem:index} with $A = \Kc$.
\end{proof}

\begin{remark}[Why Theorem~B is more solid than Theorem~A]
Theorem~B uses quasimodes, which are approximate eigenfunctions of $\Kc$
itself.
The gap is forced directly by the model separation $\delta_{\mathrm{model}}$
and the residual $\varepsilon$, with no intermediate compressed operator.
The Gram condition (Q4) enters only to ensure linear independence,
not to force separation.
This is a standard Kato-/Temple-type argument and is mathematically
clean.
\end{remark}

\section{Airy-Scale Conditional Consequence}
\label{sec:airy}
%% ============================================================

\begin{assumption}[Airy transition asymptotics]
\label{ass:airy}
In the transition regime $n \sim 2c/\pi$,
Assumption~\ref{ass:quasimode} holds with
$\delta_{\mathrm{model}}(c) \asymp c^{-1/3}$
and $\varepsilon(c) = o(c^{-1/3})$.
\end{assumption}

\begin{corollary}[Conditional Airy-scale gap --- Corollary~C]
\label{cor:C}
Under Assumption~\ref{ass:airy}:
$\gap{c}{n} \geq Cc^{-1/3}$
for all large $c$ and transition-window $n$.
\end{corollary}

\begin{proof}
$\varepsilon(c) = o(\delta_{\mathrm{model}}(c))$, so Theorem~B applies.
\end{proof}

\section{Open Problems}
\label{sec:open}
%% ============================================================

\begin{problem}[Compressed non-degeneracy --- the critical gap]
\label{prob:nondeg-full}
Provide any of the following sufficient conditions for
Assumption~\ref{ass:nondeg}:
\begin{enumerate}[nosep]
  \item Approximate diagonalisation of $T_{c,N}$ in a non-PSWF basis
    with separation $\asymp c^{-1/2}$;
  \item Monotonicity or Sturm-oscillation argument showing
    compressed eigenvalues are distinct and separated;
  \item Local model spacing via a non-invariant trial space
    (Option~B of Paper~XII: replace $V_N$ by a phase-space cutoff
    $\widetilde{V}_N$ that is not $\Kc$-invariant).
\end{enumerate}
Note: for $V_N = \spann\{\pswf{0},\ldots,\pswf{N}\}$
(which is $\Kc$-invariant),
the compressed non-degeneracy reduces circularly to Gap-S itself
(see Open Problem~\ref{prob:nondeg}).
A non-invariant trial space is therefore the most promising route.
\end{problem}

\begin{problem}[Transition-window quasimodes]
\label{prob:quasimodes}
Construct a quasimode family satisfying Assumption~\ref{ass:airy}
(residual $o(c^{-1/3})$, model spacing $\asymp c^{-1/3}$,
isolation condition).
This is the primary missing semiclassical ingredient for Corollary~C.
\end{problem}

\begin{problem}[Local spectral counting]
\label{prob:counting}
Establish the isolation condition (Q5) in the PSWF transition regime.
This is a Widom-type local eigenvalue counting input.
\end{problem}

\begin{problem}[Fold amplitude regime identification]
\label{prob:fold}
Determine the scaling parameter $\alpha$ for the PSWF fold integral
$\Phi_\alpha(u)$
(\texttt{fold\_model.tex}, companion repo \texttt{prolate-primes-paper}).
This governs amplitude regularity for Paper~XV of the companion series.
\end{problem}

\textbf{Conceptual summary.}
The central contribution of Paper~XIII is the precise structural reduction:
\begin{quote}\itshape
Gram coercivity reduces Gap-S to a local non-degenerate compressed spectral
model problem. Gram coercivity is the stability input;
