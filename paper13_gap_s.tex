\title[Gap-S: Reduction Framework]{%
  Paper~XIII: A Reduction Framework for Gap-S\\[0.3em]
  \large Gram Coercivity as Stability Input,
  Quasimode Transfer as Gap Mechanism}

\author{\textit{prolate-gram-coercivity programme}}
\date{May 2026}

\begin{abstract}
This paper establishes a reduction framework for the spectral gap problem
(Gap-S) for the prolate concentration operator
\[
  (\mathcal{K}_c f)(x) := \int_{-1}^1
  \frac{\sin(c(x-y))}{\pi(x-y)}\,f(y)\,dy
\]
in the transition regime $n \sim 2c/\pi$.

\medskip
\textbf{Central conceptual clarification.}
Gram coercivity ($\lambda_{\min}(G_N) \asymp c^{-1/2}$, Paper~I)
is a \emph{stability input}, not a direct source of eigenvalue repulsion.
It controls the geometry of the trial basis and prevents coordinate collapse,
but does not by itself force spectral gaps.
The trivial example $T_E = \lambda I$ on any orthonormal space shows this:
$\lambda_{\min}(G) = 1$ yet all gaps vanish.

\medskip\noindent
\textbf{Main outputs.}
\begin{itemize}
\item \emph{(Theorem~A, Conditional Compressed Transfer.)}
  Under Gram coercivity (Paper~I), local spectral nondegeneracy
  (Assumption~\ref{ass:nondeg}), and compressed spectral window isolation
  (Assumption~\ref{ass:window}), one obtains
  $\gap{c}{n} \geq Cc^{-1/2}$.
  Theorem~A is a transfer lemma, not a deep result:
  Assumption~\ref{ass:nondeg} already encodes most of the gap information.
  Its value is the precise identification of what is needed.
\item \emph{(Theorem~B, Quasimode Transfer.)}
  Under local quasimode hypotheses (Assumption~\ref{ass:quasimode}),
  a sharp conditional bound at model scale follows.
  This is the most solid result in the paper.
  Condition~(Q5) is the genuinely nontrivial spectral counting assumption;
  it is expected to encode most of the transition-window spectral rigidity.
\item \emph{(Corollary~C.)} Under Airy transition asymptotics,
  the conditional gap scale is $c^{-1/3}$.
  Corollary~C should be interpreted as a semiclassical prediction,
  conditional on a fully justified Airy quasimode package
  that has not yet been constructed.
\end{itemize}

\medskip\noindent
\textbf{Primary contribution.}
The paper's central result is conceptual:
the identification of the exact obstruction to Gap-S.
For $\Kc$-invariant trial spaces,
all three ingredients of Theorem~A reduce circularly to Gap-S itself,
explaining why Papers~X--XII could not close the argument.
The non-circular route requires a non-$\Kc$-invariant trial space
(Option~B of Paper~XII) and a semiclassical symbol analysis.
\end{abstract}

\subsection{Gap-S}

Gap-S (companion series \texttt{prolate-primes-paper}, Paper~XII Hyp.~1)
asks for $g(c) > 0$ with
\begin{equation}\label{eq:gapS}
  |\lam{c}{n} - \lam{c}{n+1}| \geq g(c) \gg c^{-1/4},
  \quad n \sim 2c/\pi.
\end{equation}

\begin{remark}[Relation to B-strong]
\label{rem:bstrong}
B-strong (Paper~IX) is $\varepsilon_N^{(c)} \geq Cc^{-1/2}$
for $N = \lfloor Ac^{1/3} \rfloor \ll 2c/\pi$:
a specific instance of Gap-S at the Galerkin truncation index.
\end{remark}

\section{Operator Setup}
\label{sec:setup}
%% ============================================================

\begin{definition}[Prolate concentration operator]
For $c > 0$,
$(\Kc f)(x) := \int_{-1}^1 \frac{\sin(c(x-y))}{\pi(x-y)} f(y)\,dy$.
Eigenvalues $1 > \lam{c}{0} \geq \lam{c}{1} \geq \cdots > 0$
with eigenfunctions $\pswf{n}$.
\end{definition}

\begin{definition}[Gap, residual, Galerkin subspace]
$\gap{c}{n} := \lam{c}{n} - \lam{c}{n+1}$,\quad
$\varepsilon_n^{(c)} := 1 - \lam{c}{n}$,\quad
$V_N := \spann\{\pswf{0},\ldots,\pswf{N}\}$.
\end{definition}

\begin{lemma}[Classical properties]
$\Kc$ is compact, self-adjoint, positive, $\|\Kc\|_{\op} \leq 1$,
Hilbert--Schmidt, spectrum in $(0,1)$.
\end{lemma}

\section{Gram Coercivity as Stability Input}
\label{sec:gram}
%% ============================================================

\begin{assumption}[Gram coercivity --- Paper~I]
\label{ass:gram}
There exist trial spaces $E_{c,N} \subset L^2([-1,1])$ with Gram matrices
$G_{c,N}$ satisfying
$\lambda_{\min}(G_{c,N}) \geq \alpha_N(c) \asymp c^{-1/2}$
for all sufficiently large $c$.
(Proved unconditionally in \textbf{Paper~I}.)
\end{assumption}

\begin{lemma}[Gram invertibility --- coordinate stability]
\label{lem:gram-norm}
If $\lambda_{\min}(G) \geq \alpha > 0$, then
$\|\sum_j a_j \phi_j\|^2 \geq \alpha\|a\|_{\ell^2}^2$.
\end{lemma}

\begin{lemma}[Compression --- Rayleigh quotient identity]
\label{lem:compression}
For finite-dimensional $E$, the compression $T_E := P_E \Kc|_E$
is self-adjoint with $\ip{T_E u}{u}/\|u\|^2 = \ip{\Kc u}{u}/\|u\|^2$
for $u \in E\setminus\{0\}$.
\end{lemma}

\begin{remark}[What Gram coercivity does and does not give]
\label{rem:gram-limit}
Lemma~\ref{lem:gram-norm} gives coordinate stability.
Lemma~\ref{lem:compression} identifies compressed eigenvalues
as bona fide Rayleigh quotients of $\Kc$.
However, Gram coercivity does \emph{not} force spectral gaps.
Counterexample: $T_E := \lambda \cdot \mathrm{Id}_E$ on any orthonormal
basis gives $\lambda_{\min}(G) = 1$ but zero gaps.
Two additional inputs are required:
\begin{itemize}[nosep]
  \item \emph{Local spectral nondegeneracy
    (Assumption~\ref{ass:nondeg}):} the compressed operator must have
    approximately separated spectrum.
  \item \emph{Spectral window isolation
    (Assumption~\ref{ass:window}):} compressed eigenvalues must
    be identifiable as specific consecutive PSWF eigenvalues.
\end{itemize}
\end{remark}

\section{Conditional Transfer Theorem}
\label{sec:transfer}

\subsection{Assumptions and roles}

\begin{assumption}[Local spectral nondegeneracy]
\label{ass:nondeg}
For the compressed operator $T_{c,N} := P_{E_{c,N}}\Kc|_{E_{c,N}}$,
there exists a basis of $E_{c,N}$ in which
\begin{equation}\label{eq:nondeg}
  \|T_{c,N} - \mathrm{diag}(\mu_j^{(c)})\|_{\op} \leq \delta(c),
\end{equation}
where $\min_j|\mu_j^{(c)} - \mu_{j+1}^{(c)}| \geq \delta_{\mathrm{model}}(c)$
and $\delta(c) = o(\delta_{\mathrm{model}}(c))$.
\end{assumption}

\begin{remark}[Operationalising Assumption~\ref{ass:nondeg}]
\label{rem:nondeg-operationalize}
For $\Kc$-invariant spaces (e.g.\ $V_N$), \eqref{eq:nondeg} is trivially
satisfied with $\mu_j = \lam{c}{j}$ --- but then $\delta_{\mathrm{model}}$
is the gap itself (circular).
For a \emph{non-invariant} trial space $\widetilde{V}_N$,
$T_{c,N}$ is genuinely non-diagonal and \eqref{eq:nondeg} becomes
a non-trivial statement about the compressed symbol.
Verifying it requires:
\begin{enumerate}[nosep]
  \item Canonical coordinates for approximate diagonalisation
    (microlocal normal form);
  \item Off-diagonal control via noncommutativity of
    $\Pi_{\mathrm{space}}$ and $\Pi_{\mathrm{phase}}$;
  \item A semiclassical scale at which symbol separation
    $\delta_{\mathrm{model}}(c) \asymp c^{-1/2}$ is visible.
\end{enumerate}
This is the concrete programme for Paper~XIV.
\end{remark}

\begin{assumption}[Compressed spectral window isolation]
\label{ass:window}
For the relevant pair $(n, n+1)$, the spectral window
$W_{c} := [\mu_{n+1}^{(c)} - \delta(c),\; \mu_n^{(c)} + \delta(c)]$
contains exactly two eigenvalues of $\Kc$, counted with multiplicity,
and these are $\lam{c}{n}$ and $\lam{c}{n+1}$.
\end{assumption}

\begin{remark}[Roles of the three assumptions]
\label{rem:three-roles}
\begin{center}
\renewcommand{\arraystretch}{1.2}
\begin{tabular}{lll}
\toprule
\textbf{Assumption} & \textbf{Type} & \textbf{Role} \\
\midrule
Gram coercivity (\ref{ass:gram}) & Stability & Coordinate non-collapse \\
Spectral nondegeneracy (\ref{ass:nondeg}) & Local spectral geometry & Forces compressed gap \\
Window isolation (\ref{ass:window}) & Index identification & Lifts to PSWF gap \\
\bottomrule
\end{tabular}
\end{center}
None of the three suffices alone; together they form a minimal gap-forcing package.
Theorem~A is a \emph{transfer lemma}: the depth lies entirely in
verifying these assumptions.
\end{remark}

\subsection{Interlacing lemma for window-isolated eigenvalues}

\begin{lemma}[Window interlacing]
\label{lem:interlacing}
Let $A: H \to H$ be a compact self-adjoint operator with decreasing
eigenvalues $\lambda_0(A) \geq \lambda_1(A) \geq \cdots$,
and let $E \subset H$ be an $N$-dimensional subspace with
compression $T_E := P_E A|_E$, eigenvalues
$\sigma_0(T_E) \geq \cdots \geq \sigma_{N-1}(T_E)$.

Suppose that, for some index $k \in \{0,\ldots,N-1\}$,
the open interval $I := (\sigma_{k}(T_E) - \rho,\, \sigma_{k-1}(T_E) + \rho)$
(with $\rho \geq 0$) contains exactly $m$ eigenvalues of $A$,
counted with multiplicity.
Denote them $\lambda_{p}(A) \geq \cdots \geq \lambda_{p+m-1}(A)$.

If in addition the spectral projector of $T_E$ onto the subspace
spanned by the eigenvectors with eigenvalues inside $I$ has rank $m$,
then $\{\sigma_{k-m+1}(T_E),\ldots,\sigma_k(T_E)\} \subset I$
and the $m$ eigenvalues of $A$ inside $I$ satisfy
\[
  \lambda_{p+j}(A) \in [\sigma_{k-j}(T_E) - \rho,\;\sigma_{k-j}(T_E) + \rho]
  \qquad (j = 0,\ldots,m-1).
\]
In particular, with $m = 2$, $k = n+1$:
\begin{equation}\label{eq:interlace-gap}
  \lam{c}{n} - \lam{c}{n+1}
  \geq (\sigma_n(T_E) - \sigma_{n+1}(T_E)) - 2\rho.
\end{equation}
\end{lemma}

\begin{proof}
The min--max principle for $A$ gives
$\lambda_j(A) = \max_{\dim S = j+1} \min_{u \in S,\, \|u\|=1} \ip{Au}{u}$.
Since every unit vector $u \in E$ satisfies
$\ip{Au}{u} = \ip{T_E u}{u}$ (Lemma~\ref{lem:compression}),
the subspaces of $E$ are valid test spaces for $A$, giving
\begin{equation}\label{eq:minmax-bound}
  \lambda_j(A) \geq \sigma_j(T_E)
  \quad \text{for } j = 0,\ldots,N-1.
\end{equation}
Now let $I = (a,b)$ with $b - a = \sigma_{k-1}(T_E) - \sigma_k(T_E) + 2\rho$.
By hypothesis, $I$ contains exactly $m$ eigenvalues of $A$ and
exactly $m$ eigenvalues of $T_E$ (spectral projector rank condition).
Denote the $A$-eigenvalues in $I$ as $\lambda_p \geq \cdots \geq \lambda_{p+m-1}$
and the $T_E$-eigenvalues as $\sigma_{k-m+1} \geq \cdots \geq \sigma_k$.
From \eqref{eq:minmax-bound}: $\lambda_{p+j} \geq \sigma_{k-j} > a$ for each $j$.
The upper bound $\lambda_{p+j} < b + \rho = \sigma_{k-j} + \rho$
follows from the assumption that no eigenvalue of $A$ lies above $b + \rho$
(guaranteed by the exact count in $I$ together with the isolation $\rho$ gap
to the next compressed eigenvalue).
Subtracting the $j=0$ and $j=1$ bounds gives~\eqref{eq:interlace-gap}.
\end{proof}

\begin{remark}[Relation to Assumption~\ref{ass:window}]
Assumption~\ref{ass:window} posits exactly the ``exact count $m = 2$''
condition in Lemma~\ref{lem:interlacing}, with $\rho = \delta(c)$.
The spectral projector rank condition is an additional technical requirement;
it is automatically satisfied when the compressed eigenvalues inside $I$
are simple, which holds generically and follows here from
Assumption~\ref{ass:nondeg} (the approximate diagonal structure implies
the $T_{c,N}$-eigenvalues inside $W_c$ are distinct for $\delta(c)$
sufficiently small relative to $\delta_{\mathrm{model}}(c)$).
\end{remark}

\subsection{The theorem}

\begin{theorem}[Conditional Compressed Transfer --- Theorem~A]
\label{thm:A}
Under Assumptions~\ref{ass:gram}, \ref{ass:nondeg}, and~\ref{ass:window},
for all sufficiently large $c$ and relevant transition-window indices $n$:
\begin{equation}\label{eq:thmA}
  \gap{c}{n} \geq \tfrac{1}{2}\delta_{\mathrm{model}}(c).
\end{equation}
In particular, if $\delta_{\mathrm{model}}(c) \asymp c^{-1/2}$,
then $\gap{c}{n} \geq Cc^{-1/2}$.
\end{theorem}

\begin{proof}
\textit{Step 1 (compressed gap via Weyl).}
By Assumption~\ref{ass:nondeg} and Weyl's inequality applied to
$T_{c,N} = \mathrm{diag}(\mu_j^{(c)}) + R$ with $\|R\|_{\op} \leq \delta(c)$,
the eigenvalues $\sigma_j^{(c)}$ of $T_{c,N}$ satisfy
$|\sigma_j^{(c)} - \mu_j^{(c)}| \leq \delta(c)$.
Hence
\[
  \sigma_n^{(c)} - \sigma_{n+1}^{(c)}
  \geq (\mu_n^{(c)} - \mu_{n+1}^{(c)}) - 2\delta(c)
  \geq \delta_{\mathrm{model}}(c) - 2\delta(c).
\]
The eigenvalues $\sigma_n^{(c)}$ and $\sigma_{n+1}^{(c)}$ are both
contained in the window $W_c = [\mu_{n+1}^{(c)} - \delta(c),\, \mu_n^{(c)} + \delta(c)]$.

\textit{Step 2 (index identification via Lemma~\ref{lem:interlacing}).}
By Assumption~\ref{ass:window}, $W_c$ contains exactly two eigenvalues
of $\Kc$, namely $\lam{c}{n}$ and $\lam{c}{n+1}$.
Assumption~\ref{ass:nondeg} implies the two $T_{c,N}$-eigenvalues
inside $W_c$ are distinct (for $\delta(c)$ small), so the spectral
projector rank condition of Lemma~\ref{lem:interlacing} holds.
Applying Lemma~\ref{lem:interlacing} with $A = \Kc$,
$E = E_{c,N}$, $m = 2$, $\rho = \delta(c)$, and window $W_c$:
\[
  \gap{c}{n} = \lam{c}{n} - \lam{c}{n+1}
  \geq (\sigma_n^{(c)} - \sigma_{n+1}^{(c)}) - 2\delta(c).
\]

\textit{Step 3 (conclusion).}
Combining Steps~1 and~2:
\[
  \gap{c}{n}
  \geq \delta_{\mathrm{model}}(c) - 4\delta(c)
  \geq \tfrac{1}{2}\delta_{\mathrm{model}}(c),
\]
where the last inequality holds since $\delta(c) = o(\delta_{\mathrm{model}}(c))$
implies $4\delta(c) \leq \tfrac{1}{2}\delta_{\mathrm{model}}(c)$ for large $c$.
\end{proof}

\begin{remark}[On the circularity for invariant trial spaces]
\label{rem:circular}
If $E_{c,N} = V_N$ ($\Kc$-invariant), then $T_{c,N}$ is exactly diagonal
with eigenvalues $\lam{c}{0},\ldots,\lam{c}{N}$, and both
Assumption~\ref{ass:nondeg} and Assumption~\ref{ass:window}
reduce to Gap-S itself.
This is precisely why Papers~X--XII could not close the argument:
any approach built on $\Kc$-invariant spaces inherits the circularity.
The non-circular route is Option~B of Paper~XII:
a non-invariant phase-space trial space $\widetilde{V}_N$,
for which the compressed symbol structure is governed by the
noncommutativity of $\Pi_{\mathrm{space}}$ and $\Pi_{\mathrm{frequency}}$,
not by the PSWF eigenvalues themselves.
\end{remark}

%% ============================================================
\section{Quasimode Transfer (the Solid Route)}
\label{sec:quasimode}
%% ============================================================

Theorem~B does not compress $\Kc$ to a finite-dimensional model;
it uses approximate eigenfunctions of $\Kc$ directly
and is free of the circularity of Theorem~A.
The argument is standard Kato--Temple type.

\begin{definition}[Quasimode family]
A normalised family $f_n^{(c)} \in L^2([-1,1])$ is an $\varepsilon(c)$-quasimode
for $\Kc$ with parameters $\mu_n(c)$ if
$\|(\Kc - \mu_n(c))f_n^{(c)}\| \leq \varepsilon(c)$.
\end{definition}

\begin{assumption}[Local quasimode package]
\label{ass:quasimode}
There exists a transition-window family $\{f_n^{(c)}\}$ and $\mu_n(c)$ such that:
\begin{enumerate}[label=(Q\arabic*),nosep]
  \item $\|f_n^{(c)}\| = 1$;
  \item $\|(\Kc - \mu_n(c))f_n^{(c)}\| \leq \varepsilon(c)$;
  \item $|\mu_n(c) - \mu_{n+1}(c)| \geq \delta_{\mathrm{model}}(c)$;
  \item $\|G_n^{(c)} - I\|_{\op} \leq \eta(c) \to 0$
    (consecutive quasimodes nearly orthogonal);
  \item \textbf{[Nontrivial]} The spectral window
    $[\mu_{n+1}(c) - \varepsilon(c),\, \mu_n(c) + \varepsilon(c)]$
    contains exactly two eigenvalues of $\Kc$ counted with multiplicity.
\end{enumerate}
\end{assumption}

\begin{remark}[On the depth of condition~(Q5)]
\label{rem:Q5}
Condition~(Q5) is a genuinely nontrivial local spectral counting assumption,
not a minor technical auxiliary condition.
It asserts that the spectral window around two quasimode parameters
contains \emph{exactly} two eigenvalues of $\Kc$ --- neither more nor fewer.
This is essentially a local Weyl law / Widom-type counting control
in the transition regime $n \sim 2c/\pi$, and requires:
\begin{itemize}[nosep]
  \item An upper counting bound: no extra eigenvalue crowds the window;
  \item A lower counting bound: no eigenvalue escapes the window.
\end{itemize}
(Q5) is expected to encode most of the transition-window spectral rigidity
and is the deepest open problem identified in this paper.
See Open Problem~\ref{prob:counting}.
\end{remark}

\begin{lemma}[Spectral index stability --- Kato--Temple type]
\label{lem:index}
Let $A$ compact self-adjoint, $\mu_n > \mu_{n+1}$,
$\varepsilon$-quasimodes $f_n, f_{n+1}$, $\|G - I\|_{\op} \leq \eta < 1$.
If the window $[\mu_{n+1} - \varepsilon,\, \mu_n + \varepsilon]$ contains
exactly two eigenvalues of $A$ (condition (Q5)), then those are
$\lambda_n(A)$ and $\lambda_{n+1}(A)$, and
\[
  \lambda_n(A) - \lambda_{n+1}(A)
  \geq (\mu_n - \mu_{n+1}) - 2\varepsilon.
\]
\end{lemma}

\begin{proof}
$\mathrm{dist}(\mu_j,\sigma(A))\leq\varepsilon$ (spectral theorem);
$f_n,f_{n+1}$ linearly independent (Gram condition);
exact count (Q5) identifies $\lambda_n(A), \lambda_{n+1}(A)$;
order and subtract.
\end{proof}

\begin{theorem}[Quasimode Transfer --- Theorem~B]
\label{thm:B}
Under Assumption~\ref{ass:quasimode}, for all sufficiently large $c$:
\[
  \gap{c}{n} \geq \delta_{\mathrm{model}}(c) - 2\varepsilon(c).
\]
If $\varepsilon(c) = o(\delta_{\mathrm{model}}(c))$,
then $\gap{c}{n} \geq \frac{1}{2}\delta_{\mathrm{model}}(c)$.
\end{theorem}

\begin{proof}
Apply Lemma~\ref{lem:index} with $A = \Kc$.
\end{proof}

\section{Airy-Scale Conditional Consequence}
\label{sec:airy}
%% ============================================================

\begin{assumption}[Airy transition asymptotics]
\label{ass:airy}
In the transition regime $n \sim 2c/\pi$,
Assumption~\ref{ass:quasimode} holds with
$\delta_{\mathrm{model}}(c) \asymp c^{-1/3}$
and $\varepsilon(c) = o(c^{-1/3})$.
\end{assumption}

\begin{remark}[Status of Assumption~\ref{ass:airy}]
\label{rem:airy-status}
The scale $c^{-1/3}$ is the expected turning-point eigenvalue spacing
in the Airy quantization scheme, consistent with Widom--Deift
transition asymptotics and numerical evidence.
Proving Assumption~\ref{ass:airy} from first principles requires:
\begin{itemize}[nosep]
  \item Steepest-descent analysis of the PSWF transition kernel;
  \item Explicit Airy quasimodes with residual $o(c^{-1/3})$;
  \item Local Widom-type counting control (condition~(Q5) in this regime).
\end{itemize}
\end{remark}

\begin{corollary}[Conditional Airy-scale gap --- Corollary~C]
\label{cor:C}
Under Assumption~\ref{ass:airy},
$\gap{c}{n} \geq Cc^{-1/3}$ for all large $c$ and transition-window $n$.
\end{corollary}

\begin{proof}
Theorem~B with $\varepsilon(c) = o(\delta_{\mathrm{model}}(c))$.
\end{proof}

\begin{remark}[Corollary~C as a semiclassical prediction]
\label{rem:corC-status}
Corollary~C should currently be interpreted as a
\emph{semiclassical prediction}, not a theorem.
Its value is conditional on a fully justified Airy quasimode package
that remains to be constructed.
It records what one would obtain \emph{if} such a package were available,
and identifies precisely what needs to be built.
It is a target theorem rather than a current achievement.
\end{remark}

\begin{remark}[Hierarchy of bounds]
$c^{-1/3} \gg c^{-1/2} \gg c^{-1/4}$.
Both Theorem~A ($c^{-1/2}$, conditional) and Corollary~C
($c^{-1/3}$, prediction) would close Gap-S if their assumptions
could be verified. The difficulty lies entirely in those assumptions.
\end{remark}

\section{Open Problems}

\begin{problem}[Non-invariant trial space: compressed symbol analysis]
\label{prob:nondeg}
Construct $\widetilde{V}_N = \Pi_{\mathrm{phase}} \Pi_{\mathrm{space}} L^2([-1,1])$
and verify Assumption~\ref{ass:nondeg} with
$\delta_{\mathrm{model}}(c) \asymp c^{-1/2}$: (1) microlocal normal form;
(2) off-diagonal control via noncommutativity of $\Pi_{\mathrm{space}},\Pi_{\mathrm{phase}}$;
(3) semiclassical scale of symbol separation.
This is the programme for Paper~XIV.
\end{problem}

\begin{problem}[Spectral window isolation for compressed operator]
\label{prob:window}

of OP~\ref{prob:nondeg}: local interlacing / counting for
the compressed spectrum of $T_{c,N}$.
\end{problem}

\begin{problem}[(Q5): Local spectral counting in the transition regime]
\label{prob:counting}
Verify condition~(Q5) in the PSWF transition regime $n \sim 2c/\pi$:
the spectral window contains exactly two eigenvalues of $\Kc$.
This is a local Widom-type counting problem and is expected to encode
most of the transition-window spectral rigidity.
It is the deepest open problem identified in this paper.
\end{problem}

\begin{problem}[Airy quasimode package]

(residual $o(c^{-1/3})$, spacing $\asymp c^{-1/3}$,
condition~(Q5) in the Airy regime).
\end{problem}

\end{problem}
