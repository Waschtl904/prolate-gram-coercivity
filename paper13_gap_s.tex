\title[Gap-S: Reduction Framework]{%
  Paper~XIII: A Reduction Framework for Gap-S\\[0.3em]
  \large Gram Coercivity as Stability Input,
  Quasimode Transfer as Gap Mechanism}

\author{\textit{prolate-gram-coercivity programme}}
\date{May 2026}

\begin{abstract}
This paper establishes a reduction framework for the spectral gap problem
(Gap-S) for the prolate concentration operator
\[
  (\mathcal{K}_c f)(x) := \int_{-1}^1
  \frac{\sin(c(x-y))}{\pi(x-y)}\,f(y)\,dy
\]
in the transition regime $n \sim 2c/\pi$.

\medskip
\textbf{Central conceptual clarification.}
Gram coercivity ($\lambda_{\min}(G_N) \asymp c^{-1/2}$, Paper~I)
is a \emph{stability input}, not a direct source of eigenvalue repulsion.
It controls the geometry of the trial basis and prevents coordinate collapse,
but does not by itself force spectral gaps.
The trivial example $T_E = \lambda I$ on any orthonormal space shows this:
$\lambda_{\min}(G) = 1$ yet all gaps vanish.

\medskip\noindent
\textbf{Main outputs.}
\begin{itemize}
\item \emph{(Theorem~A, Conditional Compressed Transfer.)}
  Under Gram coercivity (Paper~I), compressed spectral nondegeneracy
  (Assumption~\ref{ass:nondeg}), and compressed spectral window isolation
  (Assumption~\ref{ass:window}), one obtains
  $\gap{c}{n} \geq C c^{-1/2}$ for transition-window indices.
  The three assumptions play distinct roles: stability, local spectral
  geometry, and index identification respectively.
\item \emph{(Theorem~B, Quasimode Transfer.)}
  Under local quasimode hypotheses (Assumption~\ref{ass:quasimode}),
  a sharp conditional bound at model scale follows directly.
  This is the most solid result in the paper.
\item \emph{(Corollary~C.)} Under Airy transition asymptotics,
  the conditional gap scale is $c^{-1/3}$.
\end{itemize}

\medskip\noindent
\textbf{Honest assessment.}
Theorem~A requires three assumptions, none of which is deducible from
Gram coercivity alone.
For $\Kc$-invariant trial spaces (e.g.\ $V_N = \spann\{\pswf{0},\ldots,\pswf{N}\}$),
all three reduce circularly to Gap-S itself.
The non-circular route requires a non-invariant trial space
(Option~B of Paper~XII).
Theorem~B is mathematically solid and does not have this circularity.
The paper's primary contribution is the precise identification of three
distinct ingredients required for Gap-S:
\emph{coordinate stability}, \emph{local spectral nondegeneracy},
and \emph{spectral window isolation}.
\end{abstract}

\subsection{Gap-S}

Gap-S (companion series \texttt{prolate-primes-paper}, Paper~XII Hyp.~1)
asks for $g(c) > 0$ with
\begin{equation}\label{eq:gapS}
  |\lam{c}{n} - \lam{c}{n+1}| \geq g(c) \gg c^{-1/4},
  \quad n \sim 2c/\pi.
\end{equation}

\begin{remark}[Relation to B-strong]
\label{rem:bstrong}
B-strong (Paper~IX) is $\varepsilon_N^{(c)} \geq Cc^{-1/2}$ for
$N = \lfloor Ac^{1/3} \rfloor \ll 2c/\pi$.
This is a specific instance of Gap-S at the Galerkin truncation index,
deep below the transition window.
\end{remark}

\section{Operator Setup}
\label{sec:setup}
%% ============================================================

\begin{definition}[Prolate concentration operator]
For $c > 0$, let $\Kc : L^2([-1,1]) \to L^2([-1,1])$ be
$(\Kc f)(x) := \int_{-1}^1 \frac{\sin(c(x-y))}{\pi(x-y)} f(y)\,dy$.
Its eigenvalues $1 > \lam{c}{0} \geq \lam{c}{1} \geq \cdots > 0$
are the PSWF concentration eigenvalues with eigenfunctions $\pswf{n}$.
\end{definition}

\begin{definition}[Gap, residual, Galerkin subspace]
$\gap{c}{n} := \lam{c}{n} - \lam{c}{n+1}$,\quad
$\varepsilon_n^{(c)} := 1 - \lam{c}{n}$,\quad
$V_N := \spann\{\pswf{0},\ldots,\pswf{N}\}$.
\end{definition}

\begin{lemma}[Classical properties]
$\Kc$ is compact, self-adjoint, positive, $\|\Kc\|_{\op} \leq 1$,
Hilbert--Schmidt, with spectrum in $(0,1)$ accumulating only at $0$.
\end{lemma}

%% ============================================================
\section{Gram Coercivity as Stability Input}
\label{sec:gram}
%% ============================================================

\begin{assumption}[Gram coercivity --- Paper~I]
\label{ass:gram}
There exist trial spaces $E_{c,N} \subset L^2([-1,1])$ with Gram matrices
$G_{c,N}$ satisfying
$\lambda_{\min}(G_{c,N}) \geq \alpha_N(c) \asymp c^{-1/2}$
for all sufficiently large $c$.
(Proved unconditionally in \textbf{Paper~I}.)
\end{assumption}

\begin{lemma}[Gram invertibility --- coordinate stability]
\label{lem:gram-norm}
If $E = \spann\{\phi_1,\ldots,\phi_N\}$ with
$\lambda_{\min}(G) \geq \alpha > 0$, then
$\|\sum_j a_j \phi_j\|^2 \geq \alpha\,\|a\|_{\ell^2}^2$
for all $a \in \mathbb{C}^N$.
\end{lemma}

\begin{proof}
$\|\sum a_j\phi_j\|^2 = a^*Ga \geq \alpha\|a\|^2$.
\end{proof}

\begin{lemma}[Compression --- Rayleigh quotient identity]
\label{lem:compression}
For finite-dimensional $E \subset L^2([-1,1])$, the compression
$T_E := P_E \Kc|_E$ is self-adjoint on $E$ with
$\ip{T_E u}{u}/\|u\|^2 = \ip{\Kc u}{u}/\|u\|^2$
for all $u \in E \setminus\{0\}$.
\end{lemma}

\begin{remark}[What Gram coercivity does and does not give]
\label{rem:gram-limit}
Lemma~\ref{lem:gram-norm} gives \emph{coordinate stability}:
the trial vectors do not collapse linearly.
Lemma~\ref{lem:compression} shows that compressed eigenvalues
are bona fide Rayleigh quotients of $\Kc$.

\medskip
However, Gram coercivity does \emph{not} force spectral gaps.
Counterexample: take $\{e_j\}$ any orthonormal basis of $E$ and
set $T_E := \lambda \cdot \mathrm{Id}_E$.
Then $\lambda_{\min}(G) = 1$ (perfect conditioning)
but all consecutive eigenvalues of $T_E$ coincide (zero gaps).

\medskip
Conclusion: two additional inputs are required beyond Gram coercivity.
\begin{itemize}[nosep]
  \item \emph{Local spectral nondegeneracy:} the compressed operator
    must have approximately separated spectrum
    (Assumption~\ref{ass:nondeg}).
  \item \emph{Spectral window isolation:} compressed eigenvalues must
    be identifiable as specific consecutive PSWF eigenvalues
    (Assumption~\ref{ass:window}).
\end{itemize}
These three assumptions together form a complete gap-forcing package.
\end{remark}

%% ============================================================
\section{Conditional Transfer Theorem}
\label{sec:transfer}
%% ============================================================

\subsection{The three assumptions and their roles}

\begin{assumption}[Local spectral nondegeneracy]
\label{ass:nondeg}
For the compressed operator $T_{c,N} := P_{E_{c,N}}\Kc|_{E_{c,N}}$,
there exists a basis of $E_{c,N}$ in which
\begin{equation}\label{eq:nondeg}
  \|T_{c,N} - \mathrm{diag}(\mu_j^{(c)})\|_{\op} \leq \delta(c),
\end{equation}
where the $\mu_j^{(c)}$ are approximate diagonal values with
$\min_j |\mu_j^{(c)} - \mu_{j+1}^{(c)}| \geq \delta_{\mathrm{model}}(c)$
and $\delta(c) = o(\delta_{\mathrm{model}}(c))$.
\end{assumption}

\begin{assumption}[Compressed spectral window isolation]
\label{ass:window}
For the relevant transition-window pair of indices $(n, n+1)$,
the spectral window
$[\mu_{n+1}^{(c)} - \delta(c),\; \mu_n^{(c)} + \delta(c)]$
contains exactly two eigenvalues of $\Kc$ counted with multiplicity,
and these eigenvalues are $\lam{c}{n}$ and $\lam{c}{n+1}$.
\end{assumption}

\begin{remark}[Roles of the three assumptions]
\label{rem:three-roles}
\begin{center}
\renewcommand{\arraystretch}{1.25}
\begin{tabular}{lll}
\toprule
\textbf{Assumption} & \textbf{Type} & \textbf{Role} \\
\midrule
Gram coercivity (\ref{ass:gram}) & Stability & Coordinate non-collapse \\
Spectral nondegeneracy (\ref{ass:nondeg}) & Local spectral geometry & Forces compressed gap \\
Window isolation (\ref{ass:window}) & Index identification & Lifts compressed gap to PSWF gap \\
\bottomrule
\end{tabular}
\end{center}
None of the three suffices alone.
All three together form a minimal gap-forcing package.
\end{remark}

\subsection{The theorem}

\begin{theorem}[Conditional Compressed Transfer --- Theorem~A]
\label{thm:A}
Under Assumptions~\ref{ass:gram}, \ref{ass:nondeg}, and~\ref{ass:window},
for all sufficiently large $c$ and all relevant transition-window
indices $n$:
\begin{equation}\label{eq:thmA}
  \gap{c}{n}
  \geq \delta_{\mathrm{model}}(c) - 2\delta(c)
  \geq \tfrac{1}{2}\delta_{\mathrm{model}}(c).
\end{equation}
In particular, if $\delta_{\mathrm{model}}(c) \asymp c^{-1/2}$,
then $\gap{c}{n} \geq Cc^{-1/2}$, closing Gap-S and B-strong.
\end{theorem}

\begin{proof}
\textit{Step 1: compressed gap.}
By Assumption~\ref{ass:nondeg} and Weyl's inequality,
the eigenvalues $\sigma_j^{(c)}$ of $T_{c,N}$ satisfy
$|\sigma_j^{(c)} - \mu_j^{(c)}| \leq \delta(c)$.
Hence
\[
  \sigma_n^{(c)} - \sigma_{n+1}^{(c)}
  \geq (\mu_n^{(c)} - \mu_{n+1}^{(c)}) - 2\delta(c)
  \geq \delta_{\mathrm{model}}(c) - 2\delta(c).
\]

\textit{Step 2: index identification.}
By Assumption~\ref{ass:window}, the window
$[\mu_{n+1}^{(c)} - \delta(c),\; \mu_n^{(c)} + \delta(c)]$
contains exactly two eigenvalues of $\Kc$, namely
$\lam{c}{n}$ and $\lam{c}{n+1}$.
Since $\sigma_n^{(c)}, \sigma_{n+1}^{(c)}$ lie in this window
(by Step~1 and the perturbation bound),
the min--max principle via Lemma~\ref{lem:compression}
gives $\lam{c}{n} \geq \sigma_n^{(c)}$ and
$\lam{c}{n+1} \leq \sigma_{n+1}^{(c)} + 2\delta(c)$.

\textit{Step 3: conclusion.}
Subtracting and using Step~1:
$\gap{c}{n} \geq \sigma_n^{(c)} - \sigma_{n+1}^{(c)} - 2\delta(c)
\geq \delta_{\mathrm{model}}(c) - 4\delta(c)
\geq \frac{1}{2}\delta_{\mathrm{model}}(c)$
for $\delta(c) = o(\delta_{\mathrm{model}}(c))$.
\end{proof}

\begin{remark}[Gram coercivity's role in Theorem~A]
Gram coercivity (Assumption~\ref{ass:gram}) does not appear in the
algebraic steps of the proof.
Its role is structural: it ensures the trial space $E_{c,N}$ is
geometrically stable, so that Assumptions~\ref{ass:nondeg}
and~\ref{ass:window} can be meaningfully stated and verified
without coordinate collapse artefacts.
An ill-conditioned trial space would make the compressed operator
unstable, rendering the approximate diagonalization in
\eqref{eq:nondeg} numerically meaningless.
\end{remark}

\begin{remark}[On the circularity for invariant trial spaces]
\label{rem:circular}
If $E_{c,N} = V_N = \spann\{\pswf{0},\ldots,\pswf{N}\}$,
then $V_N$ is $\Kc$-invariant and $T_{c,N} = \Kc|_{V_N}$
is already exactly diagonal with eigenvalues
$\lam{c}{0} \geq \cdots \geq \lam{c}{N}$.
In this case:
\begin{itemize}[nosep]
  \item Assumption~\ref{ass:nondeg} reduces to: the eigenvalues
    $\lam{c}{n}$ are separated by $\delta_{\mathrm{model}}(c)$.
  \item Assumption~\ref{ass:window} is automatically satisfied.
  \item Both reduce circularly to Gap-S itself.
\end{itemize}
Conclusion: a non-invariant trial space is essential.
Option~B of Paper~XII (phase-space cutoff $\widetilde{V}_N$
not invariant under $\Kc$) is the natural candidate.
For such a space, $T_{c,N}$ is genuinely non-diagonal, and
Assumption~\ref{ass:nondeg} becomes a non-trivial statement
about the symbol structure of the compressed operator.
\end{remark}

%% ============================================================
\section{Quasimode Transfer (the Solid Route)}
\label{sec:quasimode}
%% ============================================================

Theorem~B is the most mathematically solid result in the paper.
It does not require a compressed non-degeneracy assumption;
it uses quasimodes that are approximate eigenfunctions of $\Kc$ directly.
The structure is standard Kato--Temple type and does not suffer from
the circularity identified in Remark~\ref{rem:circular}.

\begin{definition}[Quasimode family]
A normalised family $f_n^{(c)} \in L^2([-1,1])$ is an $\varepsilon(c)$-quasimode
for $\Kc$ with reference parameters $\mu_n(c)$ if
$\|(\Kc - \mu_n(c))f_n^{(c)}\| \leq \varepsilon(c)$.
\end{definition}

\begin{assumption}[Local quasimode package]
\label{ass:quasimode}
There exists a transition-window family $\{f_n^{(c)}\}$ and $\mu_n(c)$ such that:
\begin{enumerate}[label=(Q\arabic*),nosep]
  \item $\|f_n^{(c)}\| = 1$;
  \item $\|(\Kc - \mu_n(c))f_n^{(c)}\| \leq \varepsilon(c)$;
  \item $|\mu_n(c) - \mu_{n+1}(c)| \geq \delta_{\mathrm{model}}(c)$;
  \item $\|G_n^{(c)} - I\|_{\op} \leq \eta(c) \to 0$
    (consecutive quasimodes are nearly orthogonal);
  \item The spectral window
    $[\mu_{n+1}(c) - \varepsilon(c),\, \mu_n(c) + \varepsilon(c)]$
    contains exactly two eigenvalues of $\Kc$ counted with multiplicity.
\end{enumerate}
\end{assumption}

\begin{lemma}[Spectral index stability --- Kato--Temple type]
\label{lem:index}
Let $A$ be compact self-adjoint with decreasing eigenvalues
$\lambda_0 \geq \lambda_1 \geq \cdots$
Suppose $\mu_n > \mu_{n+1}$ with $\varepsilon(c)$-quasimodes $f_n, f_{n+1}$
satisfying $\|G - I\|_{\op} \leq \eta < 1$ for their $2\times 2$ Gram
matrix.
If $[\mu_{n+1} - \varepsilon,\, \mu_n + \varepsilon]$ contains exactly two
eigenvalues of $A$, then those are $\lambda_n(A)$ and $\lambda_{n+1}(A)$,
and
\[
  \lambda_n(A) - \lambda_{n+1}(A)
  \geq (\mu_n - \mu_{n+1}) - 2\varepsilon.
\]
\end{lemma}

\begin{proof}
The spectral theorem gives $\mathrm{dist}(\mu_j, \sigma(A)) \leq \varepsilon$,
so the window contains $\geq 2$ eigenvalues.
The Gram condition $\|G-I\|<1$ implies $f_n, f_{n+1}$ are linearly
independent, giving an approximate spectral subspace of dimension $2$.
By the isolation hypothesis (Q5) these are exactly
$\lambda_n(A)$ and $\lambda_{n+1}(A)$.
Ordering: $\lambda_n \geq \mu_n - \varepsilon$ and
$\lambda_{n+1} \leq \mu_{n+1} + \varepsilon$; subtract.
\end{proof}

\begin{theorem}[Quasimode Transfer --- Theorem~B]
\label{thm:B}
Under Assumption~\ref{ass:quasimode}, for all sufficiently large $c$
and all relevant transition-window indices $n$:
\[
  \gap{c}{n}
  \geq \delta_{\mathrm{model}}(c) - 2\varepsilon(c).
\]
If $\varepsilon(c) = o(\delta_{\mathrm{model}}(c))$,
then $\gap{c}{n} \geq \frac{1}{2}\delta_{\mathrm{model}}(c)$.
\end{theorem}

\begin{proof}
Apply Lemma~\ref{lem:index} with $A = \Kc$ under
Assumption~\ref{ass:quasimode}.
\end{proof}

\begin{remark}[Why Theorem~B avoids the circularity of Theorem~A]
Theorem~B does not compress $\Kc$ to a finite-dimensional operator;
it uses approximate eigenfunctions of $\Kc$ itself.
The gap is forced by the model separation $\delta_{\mathrm{model}}$ and
the residual $\varepsilon$, with no intermediate compressed operator.
The Gram condition (Q4) enters only to certify linear independence,
not to force separation.
The index identification is handled by the isolation condition (Q5),
which is the direct analogue of Assumption~\ref{ass:window} in Theorem~A
but now stated as a hypothesis on $\Kc$ directly.
\end{remark}

\section{Airy-Scale Conditional Consequence}
\label{sec:airy}
%% ============================================================

\begin{assumption}[Airy transition asymptotics]
\label{ass:airy}
In the transition regime $n \sim 2c/\pi$,
Assumption~\ref{ass:quasimode} holds with
$\delta_{\mathrm{model}}(c) \asymp c^{-1/3}$
and $\varepsilon(c) = o(c^{-1/3})$.
\end{assumption}

\begin{remark}[Status]
This is a non-trivial semiclassical input requiring
steepest-descent analysis of the PSWF transition kernel
together with local Widom-type counting control.
The paper isolates this input; it does not claim to prove it.
\end{remark}

\begin{corollary}[Conditional Airy-scale gap --- Corollary~C]
\label{cor:C}
Under Assumption~\ref{ass:airy},
$\gap{c}{n} \geq Cc^{-1/3}$ for all large $c$ and transition-window $n$.
\end{corollary}

\begin{proof}
$\varepsilon(c) = o(\delta_{\mathrm{model}}(c))$, so Theorem~B gives
$\gap{c}{n} \geq \frac{1}{2}\delta_{\mathrm{model}}(c) \asymp c^{-1/3}$.
\end{proof}

\begin{remark}[Hierarchy of bounds]
Both bounds exceed the Gap-S threshold:
$c^{-1/3} \gg c^{-1/2} \gg c^{-1/4}$.
Theorem~A (conditional, $c^{-1/2}$) closes Gap-S if its three
assumptions can be verified for a non-invariant trial space.
Corollary~C (conditional, $c^{-1/3}$) gives the sharper Airy-scale
bound once transition quasimodes are constructed.
\end{remark}

\section{Open Problems}
\label{sec:open}
%% ============================================================

\begin{problem}[Local spectral nondegeneracy via non-invariant trial space]
\label{prob:nondeg}
Construct a non-$\Kc$-invariant trial space $\widetilde{V}_N$
(Option~B of Paper~XII, e.g.\ a phase-space cutoff)
for which the compressed operator $\widetilde{T}_{c,N}$
satisfies Assumption~\ref{ass:nondeg} with
$\delta_{\mathrm{model}}(c) \asymp c^{-1/2}$.
This is non-circular: since $\widetilde{V}_N$ is not $\Kc$-invariant,
$\widetilde{T}_{c,N}$ is genuinely non-diagonal and
\eqref{eq:nondeg} becomes a non-trivial statement.
\end{problem}

\begin{problem}[Spectral window isolation for the compressed operator]
\label{prob:window}
Under the non-invariant trial space of Open Problem~\ref{prob:nondeg},
verify Assumption~\ref{ass:window}: the compressed spectral window
contains exactly the relevant consecutive PSWF eigenvalues.
This is a local counting / interlacing problem for the compressed spectrum.
\end{problem}

\begin{problem}[Transition-window quasimodes]
\label{prob:quasimodes}
Construct a quasimode family satisfying Assumption~\ref{ass:airy}
(residual $o(c^{-1/3})$, model spacing $\asymp c^{-1/3}$,
isolation condition).
This is the primary missing semiclassical ingredient for Corollary~C.
\end{problem}

\begin{problem}[Local spectral counting]
\label{prob:counting}
Establish the isolation condition~(Q5) in the PSWF transition regime.
This is the Widom-type local eigenvalue counting input needed for
Lemma~\ref{lem:index}.
\end{problem}

\begin{problem}[Fold amplitude regime]
\label{prob:fold}
Determine the scaling parameter $\alpha$ for the PSWF fold integral
$\Phi_\alpha(u)$
(\texttt{fold\_model.tex}, \texttt{Waschtl904/prolate-primes-paper}).
This governs amplitude regularity for Paper~XV of the companion series.
\end{problem}
