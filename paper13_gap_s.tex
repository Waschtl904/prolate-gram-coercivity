\title[Gap-S: Obstruction Analysis]{%
  Paper~XIII: Obstructions to Gap-S\\[0.3em]
  \large A Negative Reduction Analysis of the Spectral Gap Problem
  for Prolate Concentration Operators}

\author{\textit{prolate-gram-coercivity programme}}
\date{May 2026}

\begin{abstract}
This paper provides a \emph{negative reduction analysis} of the spectral
gap problem (Gap-S) for the prolate concentration operator
\[
  (\mathcal{K}_c f)(x) := \int_{-1}^1
  \frac{\sin(c(x-y))}{\pi(x-y)}\,f(y)\,dy
\]
in the transition regime $n \sim 2c/\pi$.
The primary contribution is not a proof of Gap-S,
but a precise identification of why all previous approaches fail
and what minimal inputs would be logically sufficient.

\medskip\noindent
\textbf{Central negative results.}
\begin{itemize}
\item Gram coercivity ($\lambda_{\min}(G_N) \asymp c^{-1/2}$, Paper~I)
  does \emph{not} imply nontrivial spectral gaps
  (Proposition~\ref{prop:gram-fails}).
  Gram coercivity controls coordinates, not spectral separation.
\item Every approach based on $\Kc$-invariant trial spaces is
  \emph{circularly dependent on Gap-S} (Remark~\ref{rem:circular}).
  This provides a unified explanation for the failure of all
  strategies in Papers~X--XII.
\end{itemize}

\medskip\noindent
\textbf{Reduction to minimal inputs.}
Logically sufficient conditions for Gap-S are identified and
decomposed into three independent ingredients:
Gram coercivity (stability; unconditional),
spectral nondegeneracy (local spectral geometry; open),
and spectral window isolation (index identification; open).
None of the three suffices alone.
The primary open difficulty is condition~(Q5):
a local spectral counting statement $N(W) = 2$
in the transition window, which is a Widom-type problem
in the regime $n - 2c/\pi \sim c^{1/3}$.

\medskip\noindent
\textbf{Conditional outputs.}
\begin{itemize}
\item \emph{(Theorem~A, Conditional Transfer Principle.)}
  Under the three ingredients above,
  $\gap{c}{n} \geq Cc^{-1/2}$.
  Theorem~A is a \emph{lifting device}, not a deep result;
  the depth lies entirely in its assumptions.
\item \emph{(Theorem~B, Conditional Quasimode Separation.)}  
  Under local quasimode hypotheses, a conditional gap bound follows.
  Theorem~B is the most structurally sound conditional result;
  it does not use invariant spaces and avoids the circularity of Theorem~A.
\item \emph{(Corollary~C.)} A roadmap consequence: under Airy transition
  asymptotics, the gap scale would be $c^{-1/3}$.
  This is not a theorem; it defines what needs to be built.
\end{itemize}
\end{abstract}

\subsection{Gap-S}

Gap-S asks for $g(c) > 0$ with
\begin{equation}\label{eq:gapS}
  |\lam{c}{n} - \lam{c}{n+1}| \geq g(c) \gg c^{-1/4},
  \quad n \sim 2c/\pi.
\end{equation}

\section{Operator Setup}
\label{sec:setup}
%% ============================================================

\begin{definition}[Prolate concentration operator]
$(\Kc f)(x) := \int_{-1}^1 \tfrac{\sin(c(x-y))}{\pi(x-y)} f(y)\,dy$;
eigenvalues $1 > \lam{c}{0} \geq \lam{c}{1} \geq \cdots > 0$.
\end{definition}

\begin{definition}
$\gap{c}{n} := \lam{c}{n} - \lam{c}{n+1}$,\quad
$V_N := \spann\{\pswf{0},\ldots,\pswf{N}\}$.
\end{definition}

\begin{lemma}[Compression and Cauchy interlacing]
\label{lem:compression}
For finite-dimensional $E \subset L^2([-1,1])$ with $\dim E = N$,
the compression $T_E := P_E \Kc|_E$ is self-adjoint
and its eigenvalues $\sigma_0(T_E) \geq \cdots \geq \sigma_{N-1}(T_E)$
are the Ritz values of $\Kc$ with respect to $E$.
They satisfy the Cauchy interlacing inequalities:
\begin{equation}\label{eq:cauchy}
  \lam{c}{j} \geq \sigma_j(T_E) \geq \lam{c}{j+N'}
  \qquad (j = 0,\ldots,N-1),
\end{equation}
where $N' = \dim L^2 - N$ in the finite-rank sense
(for compact operators: the appropriate complement dimension).
In particular, Ritz values are not in general eigenvalues
or even spectral points of $\Kc$.
\end{lemma}

\begin{proof}
Ritz-value characterisation: standard min-max principle restricted to $E$.
Cauchy interlacing: Kato~\cite{Kato1966}, Chapter~V.\S4.
\end{proof}

\section{Gram Coercivity as Stability Input}
\label{sec:gram}
%% ============================================================

\begin{assumption}[Gram coercivity --- Paper~I]
\label{ass:gram}
$\lambda_{\min}(G_{c,N}) \geq \alpha_N(c) \asymp c^{-1/2}$
for all large $c$. (Proved unconditionally in Paper~I.)
\end{assumption}

\begin{proposition}[Gram coercivity does not imply spectral gaps]
\label{prop:gram-fails}
For any $\alpha > 0$ and any $N \geq 2$, there exists a self-adjoint
operator on an $N$-dimensional space with $\lambda_{\min}(G) \geq \alpha$
but all consecutive eigenvalue gaps equal to zero.
\end{proposition}

\begin{proof}
$T := \mu \cdot \mathrm{Id}$ on any orthonormal basis $\{e_j\}$:
$G = I$, all gaps zero.
For general $\alpha$: replace $e_j$ by $\sqrt{\alpha}\,e_j$;
$G = \alpha I$, $\lambda_{\min}(G) = \alpha$, gaps still zero.
\end{proof}

\begin{remark}[What Gram coercivity does and does not give]
\label{rem:gram-limit}
\emph{Coordinate stability $\neq$ spectral rigidity.}
Gram coercivity gives: coordinate stability (Gram invertibility)
and Ritz-value reliability (Cauchy sandwiching, Lemma~\ref{lem:compression}).
It does not give any spectral gap, for any value of $\lambda_{\min}(G)$.
The additional required inputs are local spectral nondegeneracy
(Assumption~\ref{ass:nondeg}) and window isolation
(Assumption~\ref{ass:window}).
\end{remark}

\section{Conditional Transfer Principle (Theorem~A)}

\begin{remark}[Status of Theorem~A]
\label{rem:thmA-status}
Theorem~A is a \emph{lifting device}.
Its value is not the transfer step itself, which is elementary,
but the identification and separation of the three logically
independent inputs required for Gap-S.
The depth lies entirely in verifying
Assumptions~\ref{ass:nondeg} and~\ref{ass:window};
both of these, for $\Kc$-invariant trial spaces,
reduce to Gap-S itself (Remark~\ref{rem:circular}).
\end{remark}

\subsection{Assumptions}

\begin{assumption}[Local spectral nondegeneracy]
\label{ass:nondeg}
For the compression $T_{c,N} := P_{E_{c,N}}\Kc|_{E_{c,N}}$,
there exists a basis of $E_{c,N}$ in which
\begin{equation}\label{eq:nondeg}
  \|T_{c,N} - \mathrm{diag}(\mu_j^{(c)})\|_{\op} \leq \delta(c),
\end{equation}
with $\min_j|\mu_j^{(c)} - \mu_{j+1}^{(c)}| \geq \delta_{\mathrm{model}}(c)$
and $\delta(c) = o(\delta_{\mathrm{model}}(c))$.
\end{assumption}

\begin{remark}[Operationalising Assumption~\ref{ass:nondeg}]
For $\Kc$-invariant spaces (e.g.\ $V_N$), \eqref{eq:nondeg} holds with
$\mu_j = \lam{c}{j}$, $\delta = 0$, but then $\delta_{\mathrm{model}}$
is the gap itself (circular).
For a non-invariant trial space $\widetilde{V}_N$,
verifying \eqref{eq:nondeg} requires:
(1) a microlocal normal form;
(2) off-diagonal control via noncommutativity of $\Pi_{\mathrm{space}}$
and $\Pi_{\mathrm{phase}}$;
(3) identification of the semiclassical scale of symbol separation.
Programme for Paper~XIV.
\end{remark}

\begin{assumption}[Spectral window isolation]
\label{ass:window}
The window $W_c := [\mu_{n+1}^{(c)} - \delta(c),\, \mu_n^{(c)} + \delta(c)]$
contains exactly two eigenvalues of $\Kc$,
namely $\lam{c}{n}$ and $\lam{c}{n+1}$.
\end{assumption}

\begin{remark}[Roles of the three assumptions]
\label{rem:three-roles}
\begin{center}
\renewcommand{\arraystretch}{1.2}
\begin{tabular}{lll}
\toprule
\textbf{Assumption} & \textbf{Type} & \textbf{Role} \\
\midrule
Gram coercivity (\ref{ass:gram}) & Stability & Coordinate non-collapse \\
Spectral nondeg.\ (\ref{ass:nondeg}) & Local spectral geometry & Forces compressed gap \\
Window isolation (\ref{ass:window}) & Index identification & Lifts to PSWF gap \\
\bottomrule
\end{tabular}
\end{center}
None of the three suffices alone;
Theorem~A is a lifting device whose depth lies entirely in the assumptions.
\end{remark}

\subsection{The compressed-to-global gap lemma}

\begin{lemma}[Compressed-to-global gap transfer]
\label{lem:comptoglobal}
Let $A: H \to H$ be compact self-adjoint,
$E \subset H$ $N$-dimensional with compression $T_E$
having Ritz values $\sigma_0 \geq \cdots \geq \sigma_{N-1}$.
For some index $n$ and parameters $\Delta > 0$, $\rho \geq 0$, suppose:
\begin{enumerate}[label=(C\arabic*),nosep]
  \item $\sigma_n - \sigma_{n+1} \geq \Delta$;\label{C1}
  \item The window $W := [\sigma_{n+1} - \rho,\, \sigma_n + \rho]$
    contains exactly two eigenvalues of $A$,
    denoted $\lambda_p \geq \lambda_{p+1}$.\label{C2}
\end{enumerate}
Then $\lambda_p - \lambda_{p+1} \geq \Delta - 2\rho$.
\end{lemma}

\begin{proof}
By the Cauchy interlacing inequalities (Lemma~\ref{lem:compression}),
$\sigma_j$ lies between two consecutive eigenvalues of $A$,
so $\mathrm{dist}(\sigma_j, \sigma(A)) \leq \max_k(\lam{}{k}-\lam{}{k+1})$.
Condition~(C2) implies $\sigma_n, \sigma_{n+1} \in W$
(by definition of $W$).
Since $W$ contains exactly two eigenvalues $\lambda_p \geq \lambda_{p+1}$
of $A$, and each of $\sigma_n, \sigma_{n+1}$ is within $\rho$ of
some eigenvalue of $A$ (by~(C2) and the Cauchy trap),
we obtain $\lambda_p \leq \sigma_n + \rho$ and
$\lambda_{p+1} \geq \sigma_{n+1} - \rho$.
Subtracting: $\lambda_p - \lambda_{p+1} \geq (\sigma_n - \sigma_{n+1}) - 2\rho \geq \Delta - 2\rho$.
\end{proof}

\begin{remark}[On condition~(C1) and the role of $\rho$]
\label{rem:rho}
Note that Ritz values $\sigma_j(T_E)$ are \emph{not} in general
spectral points of $A$: a Ritz value lies between
consecutive eigenvalues of $A$ (Cauchy interlacing), but is
not itself an eigenvalue or even in $\sigma(A)$ in general.
Accordingly, $\rho > 0$ is a genuine additional parameter
in Lemma~\ref{lem:comptoglobal};
setting $\rho = 0$ is not justified by Ritz-value theory alone.
In the proof of Theorem~A, $\rho$ is taken equal to $\delta(c)$
(the perturbation in Assumption~\ref{ass:nondeg}),
which is $o(\delta_{\mathrm{model}}(c))$ and hence asymptotically negligible;
but it is not zero.
\end{remark}

\begin{remark}[On index identification]
\label{rem:index}
Lemma~\ref{lem:comptoglobal} establishes the gap between the two
eigenvalues of $A$ inside the window $W$, but does not identify
their global indices as $p = n$.
This index identification is the content of
Assumption~\ref{ass:window}: an external spectral counting input,
not derivable from the compression.
\end{remark}

\subsection{The theorem}

\begin{theorem}[Conditional Transfer Principle --- Theorem~A]
\label{thm:A}
Under Assumptions~\ref{ass:gram}, \ref{ass:nondeg}, and~\ref{ass:window},
for all sufficiently large $c$ and transition-window indices $n$:
\begin{equation}\label{eq:thmA}
  \gap{c}{n} \geq \tfrac{1}{2}\delta_{\mathrm{model}}(c).
\end{equation}
In particular, $\delta_{\mathrm{model}}(c) \asymp c^{-1/2}$ closes Gap-S.
\end{theorem}

\begin{proof}
\textit{Step~1 (compressed gap).}
By Weyl and Assumption~\ref{ass:nondeg}:
$\sigma_n^{(c)} - \sigma_{n+1}^{(c)} \geq \delta_{\mathrm{model}}(c) - 2\delta(c) =: \Delta$.

\textit{Step~2 (global gap via Lemma~\ref{lem:comptoglobal}).}
Set $\rho := \delta(c) > 0$ (not zero; see Remark~\ref{rem:rho}).
Condition~(C1) holds with $\Delta$ from Step~1.
Condition~(C2) is Assumption~\ref{ass:window} with $\rho = \delta(c)$.
Applying Lemma~\ref{lem:comptoglobal}:
$\gap{c}{n} \geq \Delta - 2\delta(c) = \delta_{\mathrm{model}}(c) - 4\delta(c)$.

\textit{Step~3.}
$\delta(c) = o(\delta_{\mathrm{model}}(c))$ gives the conclusion.
\end{proof}

\begin{remark}[On the circularity for invariant trial spaces]
\label{rem:circular}
If $E_{c,N} = V_N$ ($\Kc$-invariant), $T_{c,N}$ is exactly diagonal
with eigenvalues $\lam{c}{0},\ldots,\lam{c}{N}$, so
Assumption~\ref{ass:nondeg} holds trivially with $\mu_j = \lam{c}{j}$
but then $\delta_{\mathrm{model}}$ is the gap itself;
and Assumption~\ref{ass:window} reduces to the assertion that
the window around two consecutive PSWF eigenvalues contains
exactly those two eigenvalues --- which is Gap-S.
Every invariant-space approach inherits this circularity.
This is the precise mechanism of failure for Papers~X--XII.
The only non-circular route is a non-$\Kc$-invariant trial space
$\widetilde{V}_N$ whose compressed symbol is governed by the
noncommutativity of $\Pi_{\mathrm{space}}$ and $\Pi_{\mathrm{frequency}}$.
\end{remark}

\section{Conditional Quasimode Separation (Theorem~B)}
\label{sec:quasimode}
%% ============================================================

Theorem~B works directly with approximate eigenfunctions of $\Kc$,
avoiding invariant spaces and the circularity of Theorem~A.
It is the most structurally sound conditional result in the paper.

\begin{definition}[Quasimode family]
$\{f_n^{(c)}\}$ is an $\varepsilon(c)$-quasimode family if
$\|(\Kc - \mu_n(c))f_n^{(c)}\| \leq \varepsilon(c)$ and $\|f_n^{(c)}\| = 1$.
\end{definition}

\begin{assumption}[Local quasimode package]
\label{ass:quasimode}
\begin{enumerate}[label=(Q\arabic*),nosep]
  \item $\|f_n^{(c)}\| = 1$;
  \item $\|(\Kc - \mu_n(c))f_n^{(c)}\| \leq \varepsilon(c)$;
  \item $|\mu_n(c) - \mu_{n+1}(c)| \geq \delta_{\mathrm{model}}(c)$;
  \item $\|G_n^{(c)} - I\|_{\op} \leq \eta(c) < 1$
    (quasimodes linearly independent; conceptually much weaker
    than Gram coercivity of Paper~I);
  \item \textbf{[Primary difficulty]}
    $[\mu_{n+1}(c) - \varepsilon(c),\, \mu_n(c) + \varepsilon(c)]$
    contains exactly two eigenvalues of $\Kc$.
\end{enumerate}
\end{assumption}

\begin{remark}[On the depth of (Q5)]
\label{rem:Q5}
(Q5) is a local spectral counting statement $N(W) = 2$
in the transition window.
It requires simultaneously an upper counting bound
(no extra eigenvalue crowds the window)
and a lower counting bound (no eigenvalue escapes).
This is a Widom-type local law in the regime
$n - 2c/\pi \sim c^{1/3}$
and is expected to encode most of the spectral rigidity
of the transition window.
It is the deepest open problem in this paper.
\end{remark}

\begin{remark}[Role of (Q4) in Theorem~B]
\label{rem:Q4role}
Condition~(Q4) asserts $\|G - I\|_{\op} < 1$,
which guarantees linear independence of the quasimode pair.
This is conceptually much weaker than the Gram coercivity
$\lambda_{\min}(G) \asymp c^{-1/2}$ of Paper~I:
it is a non-degeneracy condition, not a quantitative lower bound.
The connection to Gram coercivity is structural
(it motivates that suitable nearly-orthogonal quasimodes exist)
but (Q4) as used in Theorem~B requires only $\eta(c) < 1$.
\end{remark}

\begin{lemma}[Quasimode window separation]
\label{lem:qmsep}
Let $A$ compact self-adjoint, $\mu_n > \mu_{n+1}$,
$\varepsilon$-quasimodes $f_n, f_{n+1}$ with $\|G - I\|_{\op} < 1$.
If the window $[\mu_{n+1} - \varepsilon,\, \mu_n + \varepsilon]$
contains exactly two eigenvalues of $A$ (condition~(Q5)),
these are $\lambda_n(A)$ and $\lambda_{n+1}(A)$, and
\begin{equation}\label{eq:qmsep}
  \lambda_n(A) - \lambda_{n+1}(A) \geq (\mu_n - \mu_{n+1}) - 2\varepsilon.
\end{equation}
\end{lemma}

\begin{proof}
For each $j \in \{n, n+1\}$:
$\|(A - \mu_j)f_j\| \leq \varepsilon$ and $\|f_j\| = 1$
imply $\mathrm{dist}(\mu_j, \sigma(A)) \leq \varepsilon$
(standard spectral inclusion for approximate eigenvectors).
Linear independence of $f_n, f_{n+1}$ (from $\|G - I\| < 1$)
excludes $\mu_n$ and $\mu_{n+1}$ from mapping to the same eigenvalue.
Condition~(Q5) provides exact count and index identification.
Ordering and subtracting gives~\eqref{eq:qmsep}.
\end{proof}

\begin{theorem}[Conditional Quasimode Separation --- Theorem~B]
\label{thm:B}
Under Assumption~\ref{ass:quasimode}:
$\gap{c}{n} \geq \delta_{\mathrm{model}}(c) - 2\varepsilon(c)$.
If $\varepsilon(c) = o(\delta_{\mathrm{model}}(c))$,
then $\gap{c}{n} \geq \frac{1}{2}\delta_{\mathrm{model}}(c)$.
\end{theorem}

\begin{proof}
Apply Lemma~\ref{lem:qmsep} with $A = \Kc$.
\end{proof}

\section{Airy-Scale Roadmap Consequence}
\label{sec:airy}
%% ============================================================

\begin{assumption}[Airy transition asymptotics]
\label{ass:airy}
Assumption~\ref{ass:quasimode} holds in the regime $n \sim 2c/\pi$
with $\delta_{\mathrm{model}}(c) \asymp c^{-1/3}$
and $\varepsilon(c) = o(c^{-1/3})$.
\end{assumption}

\begin{remark}[Status of Assumption~\ref{ass:airy}]
The scale $c^{-1/3}$ is expected from Airy quantization and
Widom--Deift transition asymptotics.
Proving it requires: (i) steepest-descent analysis of the PSWF kernel;
(ii) Airy quasimodes with residual $o(c^{-1/3})$;
(iii) condition~(Q5) in this regime.
None is established from first principles.
\end{remark}

\begin{corollary}[Roadmap at Airy scale --- Corollary~C]
\label{cor:C}
Under Assumption~\ref{ass:airy},
$\gap{c}{n} \geq Cc^{-1/3}$ for all large $c$.
\end{corollary}

\begin{remark}[Corollary~C is not a theorem]
\label{rem:corC-status}
Corollary~C records what follows if Assumption~\ref{ass:airy} is verified.
It is a \emph{roadmap consequence}, not a current achievement.
The originality of this paper lies not in Corollary~C
but in Proposition~\ref{prop:gram-fails},
Remark~\ref{rem:circular}, and Remark~\ref{rem:Q5}.
\end{remark}

Construct $\widetilde{V}_N = \Pi_{\mathrm{phase}}\Pi_{\mathrm{space}} L^2$
and verify Assumption~\ref{ass:nondeg} with $\delta_{\mathrm{model}} \asymp c^{-1/2}$

\label{prob:window}
Verify Assumption~\ref{ass:window} for the trial space

\label{prob:counting}
Prove $N(W) = 2$ for the transition window in the regime
$n - 2c/\pi \sim c^{1/3}$.

\emph{Deepest open problem in this paper.}
\end{problem}

\begin{problem}[Airy quasimode package]
\label{prob:quasimodes}
Construct quasimodes satisfying Assumption~\ref{ass:airy}
(residual $o(c^{-1/3})$, spacing $\asymp c^{-1/3}$, (Q5) in the Airy regime).
\end{problem}

\begin{problem}[Fold amplitude regime]
\label{prob:fold}

(\texttt{fold\_model.tex}, \texttt{Waschtl904/prolate-primes-paper}).
\end{problem}

The originality of this paper lies in three results, in order of importance:
