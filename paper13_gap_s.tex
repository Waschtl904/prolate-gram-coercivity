\title[Gap-S: Obstruction Analysis]{%
  Paper~XIII: Obstructions to Gap-S\\[0.3em]
  \large A Negative Reduction Analysis of the Spectral Gap Problem
  for Prolate Concentration Operators}

\author{\textit{prolate-gram-coercivity programme}}
\date{May 2026}

\begin{abstract}
This paper provides a \emph{negative reduction analysis} of the spectral
gap problem (Gap-S) for the prolate concentration operator
\[
  (\mathcal{K}_c f)(x) := \int_{-1}^1
  \frac{\sin(c(x-y))}{\pi(x-y)}\,f(y)\,dy
\]
in the transition regime $n \sim 2c/\pi$.
The primary contribution is not a proof of Gap-S,
but a precise identification of why all previous approaches fail.

\medskip\noindent
\textbf{Central new insight.}
All currently known approaches reduce Gap-S to a local spectral
counting problem in a $c^{-1/3}$-scale transition window:
the question of whether that window contains exactly two eigenvalues
of $\Kc$, no more and no fewer.
This counting problem --- condition~(Q5) --- is a Widom-type spectral
rigidity question, not a perturbative consequence of quasimode analysis.
It is the genuine obstruction.

\medskip\noindent
\textbf{Central negative results.}
\begin{itemize}
\item Gram coercivity ($\lambda_{\min}(G_N) \asymp c^{-1/2}$, Paper~I)
  does \emph{not} imply nontrivial spectral gaps
  (Proposition~\ref{prop:gram-fails}):
  coordinate stability is not spectral rigidity.
\item Every approach based on $\Kc$-invariant trial spaces
  is circularly dependent on Gap-S (Remark~\ref{rem:circular}).
  This is a fixed-point obstruction: for invariant spaces,
  the model gap is identical to the sought gap.
  This explains the failure of all strategies in Papers~X--XII.
\end{itemize}

\medskip\noindent
\textbf{Reduction to minimal inputs.}
Gap-S $\Longleftarrow$ microlocal spectral geometry $+$ local counting
rigidity, not Gap-S $\Longleftarrow$ linear algebra.
Three logically independent ingredients are isolated:
Gram coercivity (stability, unconditional),
spectral nondegeneracy (local geometry, open),
and spectral window isolation (index counting, open).
None suffices alone.

\medskip\noindent
\textbf{Conditional outputs} (value: precise problem formulation,
not independent results).
\begin{itemize}
\item \emph{Theorem~A} (Conditional Transfer Principle):
  a lifting device whose entire content lies in its assumptions.
  It contains no genuine spectral information beyond the inputs.
\item \emph{Theorem~B} (Conditional Quasimode Separation):
  the more structurally sound route; avoids invariant spaces
  and the circularity of Theorem~A.
\item \emph{Corollary~C}: a roadmap consequence
  identifying the $c^{-1/3}$ target scale.
\end{itemize}
\end{abstract}

\subsection{Gap-S}

Gap-S asks for $g(c) > 0$ with
\begin{equation}\label{eq:gapS}
  |\lam{c}{n} - \lam{c}{n+1}| \geq g(c) \gg c^{-1/4},
  \quad n \sim 2c/\pi.
\end{equation}

\section{Operator Setup}
\label{sec:setup}
%% ============================================================

\begin{definition}[Prolate concentration operator]
$(\Kc f)(x) := \int_{-1}^1 \tfrac{\sin(c(x-y))}{\pi(x-y)} f(y)\,dy$;
eigenvalues $1 > \lam{c}{0} \geq \lam{c}{1} \geq \cdots > 0$.
\end{definition}

\begin{definition}
$\gap{c}{n} := \lam{c}{n} - \lam{c}{n+1}$,\quad
$V_N := \spann\{\pswf{0},\ldots,\pswf{N}\}$.
\end{definition}

\begin{lemma}[Compression, Ritz values, and min-max principle]
\label{lem:compression}
Let $A: H \to H$ be a compact self-adjoint operator
on a separable Hilbert space $H$,
with eigenvalues $\lambda_0(A) \geq \lambda_1(A) \geq \cdots \to 0$.
Let $E \subset H$ be an $N$-dimensional subspace
with compression $T_E := P_E A|_E$,
having eigenvalues $\sigma_0 \geq \cdots \geq \sigma_{N-1}$.

The eigenvalues $\sigma_j$ are the Ritz values of $A$ in $E$:
\begin{equation}\label{eq:ritz}
  \sigma_j = \max_{\substack{S \subset E\\ \dim S = j+1}}
  \min_{u \in S,\, \|u\|=1} \ip{Au}{u}.
\end{equation}
In particular, by the min-max principle applied simultaneously
to $A$ (with all of $H$ as test space) and to $A|_E$:
\begin{equation}\label{eq:interlace}
  \lambda_j(A) \geq \sigma_j(T_E)
  \qquad \text{for all } j = 0,\ldots,N-1.
\end{equation}
Ritz values are not in general eigenvalues or spectral points of $A$.
\end{lemma}

\begin{proof}
\eqref{eq:ritz} is the min-max principle restricted to $E$.
\eqref{eq:interlace} holds because every $j$-dimensional subspace
of $E$ is a valid test space for the min-max characterisation of
$\lambda_j(A)$ over all of $H$, so the constrained maximum
$\sigma_j$ cannot exceed the unconstrained maximum $\lambda_j(A)$.
The claim that $\sigma_j \notin \sigma(A)$ in general is illustrated
by any compression to a subspace not spanned by eigenvectors.
\end{proof}

\begin{remark}[What interlacing gives and does not give]
\label{rem:interlace}
\eqref{eq:interlace} gives the one-sided bound $\sigma_j \leq \lambda_j(A)$.
For the lower bound in finite-rank settings one sometimes writes
$\sigma_j \geq \lambda_{j+N'}(A)$ for an appropriate $N'$;
however, for the arguments of this paper only the upper bound
\eqref{eq:interlace} is used, and we do not claim a specific
codimension-dependent lower bound.
\end{remark}

\section{Gram Coercivity as Stability Input}
\label{sec:gram}
%% ============================================================

\begin{assumption}[Gram coercivity --- Paper~I]
\label{ass:gram}
$\lambda_{\min}(G_{c,N}) \geq \alpha_N(c) \asymp c^{-1/2}$
for all large $c$. (Proved unconditionally in Paper~I.)
\end{assumption}

\begin{proposition}[Gram coercivity does not imply spectral gaps]
\label{prop:gram-fails}
For any $\alpha > 0$ and any $N \geq 2$, there exists a self-adjoint
operator on an $N$-dimensional space with $\lambda_{\min}(G) \geq \alpha$
but all consecutive eigenvalue gaps equal to zero.
\end{proposition}

\begin{proof}
$T := \mu \cdot \mathrm{Id}$ on any orthonormal basis:
$G = I$, gaps zero.
For general $\alpha$: replace basis vectors by $\sqrt{\alpha}$ multiples;
$G = \alpha I$, $\lambda_{\min}(G) = \alpha$, gaps still zero.
\end{proof}

\begin{remark}[Coordinate stability $\neq$ spectral rigidity]
\label{rem:gram-limit}
Proposition~\ref{prop:gram-fails} is the key negative result.
Gram coercivity gives coordinate stability (Gram invertibility)
and Ritz-value upper bounds (Lemma~\ref{lem:compression}),
but no spectral gap, for any value of $\lambda_{\min}(G)$.
The additional required inputs are local spectral nondegeneracy
(Assumption~\ref{ass:nondeg}) and window isolation
(Assumption~\ref{ass:window}).
\end{remark}

\section{Conditional Transfer Principle (Theorem~A)}
\label{sec:transfer}
%% ============================================================

\begin{remark}[Status of Theorem~A]
\label{rem:thmA-status}
Theorem~A is a \emph{lifting device}.
It contains \emph{no genuine spectral information beyond its assumptions}.
The information content lies entirely in
Assumptions~\ref{ass:nondeg} (local spectral geometry)
and~\ref{ass:window} (index counting).
The transfer step itself is elementary.
The value of Theorem~A is exclusively in the identification
and logical separation of the three required inputs;
it is not a theorem about spectral gaps per se.
\end{remark}

\subsection{Assumptions}

\begin{assumption}[Local spectral nondegeneracy]
\label{ass:nondeg}
For the compression $T_{c,N} := P_{E_{c,N}}\Kc|_{E_{c,N}}$,
there exists a basis of $E_{c,N}$ in which
\begin{equation}\label{eq:nondeg}
  \|T_{c,N} - \mathrm{diag}(\mu_j^{(c)})\|_{\op} \leq \delta(c),
\end{equation}
with $\min_j|\mu_j^{(c)} - \mu_{j+1}^{(c)}| \geq \delta_{\mathrm{model}}(c)$
and $\delta(c) = o(\delta_{\mathrm{model}}(c))$.
\end{assumption}

\begin{remark}[Operationalising Assumption~\ref{ass:nondeg}]
For $\Kc$-invariant spaces (e.g.\ $V_N$), \eqref{eq:nondeg} holds trivially
with $\mu_j = \lam{c}{j}$, $\delta = 0$, but $\delta_{\mathrm{model}}$
then equals the gap itself (circular).
For a non-invariant trial space $\widetilde{V}_N$, verifying
\eqref{eq:nondeg} requires: (1) a microlocal normal form;
(2) off-diagonal control via noncommutativity of $\Pi_{\mathrm{space}}$
and $\Pi_{\mathrm{phase}}$; (3) identification of the semiclassical
scale of symbol separation. Programme for Paper~XIV.
\end{remark}

\begin{assumption}[Spectral window isolation]
\label{ass:window}
The window $W_c := [\mu_{n+1}^{(c)} - \delta(c),\, \mu_n^{(c)} + \delta(c)]$
contains exactly two eigenvalues of $\Kc$,
namely $\lam{c}{n}$ and $\lam{c}{n+1}$.
\end{assumption}

\begin{remark}[Roles of the three assumptions]
\label{rem:three-roles}
\begin{center}
\renewcommand{\arraystretch}{1.2}
\begin{tabular}{lll}
\toprule
\textbf{Assumption} & \textbf{Type} & \textbf{Role} \\
\midrule
Gram coercivity (\ref{ass:gram}) & Stability & Coordinate non-collapse \\
Spectral nondeg.\ (\ref{ass:nondeg}) & Local spectral geometry & Forces compressed gap \\
Window isolation (\ref{ass:window}) & Index identification & Lifts to PSWF gap \\
\bottomrule
\end{tabular}
\end{center}
None of the three suffices alone.
Theorem~A is a lifting device; its content is the identification
of these inputs, not the transfer step.
\end{remark}

\subsection{The compressed-to-global gap lemma}

\begin{lemma}[Compressed-to-global gap transfer]
\label{lem:comptoglobal}
Let $A: H\to H$ compact self-adjoint,
$E \subset H$ $N$-dimensional with compression $T_E$
having Ritz values $\sigma_0 \geq \cdots \geq \sigma_{N-1}$.
For some index $n$ and $\Delta > 0$, $\rho \geq 0$, suppose:
\begin{enumerate}[label=(C\arabic*),nosep]
  \item $\sigma_n - \sigma_{n+1} \geq \Delta$;\label{C1}
  \item The window $W := [\sigma_{n+1} - \rho,\, \sigma_n + \rho]$
    contains exactly two eigenvalues of $A$,
    denoted $\lambda_p \geq \lambda_{p+1}$.\label{C2}
\end{enumerate}
Then $\lambda_p - \lambda_{p+1} \geq \Delta - 2\rho$.
\end{lemma}

\begin{proof}
By Lemma~\ref{lem:compression}~\eqref{eq:interlace},
$\lambda_j(A) \geq \sigma_j$, so both $\sigma_n$ and $\sigma_{n+1}$
are bounded above by $\lambda_n(A)$ and $\lambda_{n+1}(A)$ respectively.
By~(C2), the only eigenvalues of $A$ in $W$ are $\lambda_p$ and
$\lambda_{p+1}$; combined with~(C2) and the Ritz-value bound,
these eigenvalues satisfy
$\lambda_p \leq \sigma_n + \rho$ and $\lambda_{p+1} \geq \sigma_{n+1} - \rho$.
Subtracting and using~(C1): $\lambda_p - \lambda_{p+1} \geq \Delta - 2\rho$.
\end{proof}

\begin{remark}[On $\rho$ and the Ritz-value bound]
\label{rem:rho}
Ritz values are not spectral points of $A$ in general
(Lemma~\ref{lem:compression}).
Accordingly $\rho = 0$ is not justified by Ritz-value theory alone;
in Theorem~A we take $\rho = \delta(c) > 0$,
which is $o(\delta_{\mathrm{model}}(c))$ and hence asymptotically negligible
but not zero.
\end{remark}

\begin{remark}[On index identification]
\label{rem:index}
Lemma~\ref{lem:comptoglobal} establishes the gap between the
two eigenvalues of $A$ inside $W$, but does not identify
their global indices as $p = n$.
Index identification is the content of Assumption~\ref{ass:window}.
\end{remark}

\subsection{The theorem}

\begin{theorem}[Conditional Transfer Principle --- Theorem~A]
\label{thm:A}
Under Assumptions~\ref{ass:gram}, \ref{ass:nondeg}, and~\ref{ass:window},
for all sufficiently large $c$ and transition-window indices $n$:
$\gap{c}{n} \geq \tfrac{1}{2}\delta_{\mathrm{model}}(c)$.
In particular, $\delta_{\mathrm{model}}(c) \asymp c^{-1/2}$ closes Gap-S.
\end{theorem}

\begin{proof}
\textit{Step~1.}
Weyl~$+$~Assumption~\ref{ass:nondeg}:
$\sigma_n^{(c)} - \sigma_{n+1}^{(c)} \geq \delta_{\mathrm{model}} - 2\delta(c) =: \Delta$.
\textit{Step~2.}
Apply Lemma~\ref{lem:comptoglobal} with $\rho = \delta(c)$
and Assumption~\ref{ass:window} as~(C2):
$\gap{c}{n} \geq \Delta - 2\delta(c) = \delta_{\mathrm{model}} - 4\delta(c)$.
\textit{Step~3.}
$\delta = o(\delta_{\mathrm{model}})$ gives the conclusion.
\end{proof}

\begin{remark}[Circularity for invariant trial spaces]
\label{rem:circular}
If $E_{c,N} = V_N$ ($\Kc$-invariant), $T_{c,N}$ is exactly diagonal
with eigenvalues $\lam{c}{0},\ldots,\lam{c}{N}$.
Then Assumption~\ref{ass:nondeg} holds trivially with $\mu_j = \lam{c}{j}$
but $\delta_{\mathrm{model}} =$ the gap, and
Assumption~\ref{ass:window} reduces to Gap-S itself.
This is a fixed-point obstruction:
the model gap is identical to the sought gap.
Every invariant-space approach inherits this circularity;
this is the precise failure mechanism of Papers~X--XII.
The only non-circular route is a non-$\Kc$-invariant trial space
$\widetilde{V}_N$ whose compressed symbol is governed by the
noncommutativity of $\Pi_{\mathrm{space}}$ and $\Pi_{\mathrm{frequency}}$.
\end{remark}

\section{Conditional Quasimode Separation (Theorem~B)}
\label{sec:quasimode}
%% ============================================================

Theorem~B works directly with approximate eigenfunctions of $\Kc$,
avoiding invariant spaces and the fixed-point obstruction of Theorem~A.
It is the more structurally sound conditional result.

\begin{definition}[Quasimode family]
$\{f_n^{(c)}\}$ is an $\varepsilon(c)$-quasimode family if
$\|(\Kc - \mu_n(c))f_n^{(c)}\| \leq \varepsilon(c)$ and $\|f_n^{(c)}\| = 1$.
\end{definition}

\begin{assumption}[Local quasimode package]
\label{ass:quasimode}
\begin{enumerate}[label=(Q\arabic*),nosep]
  \item $\|f_n^{(c)}\| = 1$;
  \item $\|(\Kc - \mu_n(c))f_n^{(c)}\| \leq \varepsilon(c)$;
  \item $|\mu_n(c) - \mu_{n+1}(c)| \geq \delta_{\mathrm{model}}(c)$;
  \item $\|G_n^{(c)} - I\|_{\op} \leq \eta(c) < 1$
    (quasimodes linearly independent; much weaker than Gram coercivity);
  \item \textbf{[Primary difficulty]}
    $[\mu_{n+1}(c) - \varepsilon(c),\, \mu_n(c) + \varepsilon(c)]$
    contains exactly two eigenvalues of $\Kc$.
\end{enumerate}
\end{assumption}

\begin{remark}[On the depth of (Q5)]
\label{rem:Q5}
(Q5) is a local spectral counting statement $N(W) = 2$.
It requires simultaneously:
an upper counting bound (no extra eigenvalue crowds the window)
and a lower counting bound (no eigenvalue escapes).
This is not a perturbative consequence of quasimode analysis;
it is a local spectral rigidity problem in the transition regime
$n - 2c/\pi \sim c^{1/3}$, of the same character as
Widom-type counting problems for integral operators
with sharp transitions.
It is the deepest open problem identified in this paper.
\end{remark}

\begin{remark}[Role of (Q4)]
\label{rem:Q4role}
(Q4) requires only $\|G - I\|_{\op} < 1$ (linear independence);
this is conceptually much weaker than the $c^{-1/2}$ Gram coercivity
of Paper~I. The connection is structural (Paper~I motivates that
nearly-orthogonal quasimodes exist) but (Q4) as used in Theorem~B
is a non-degeneracy condition only.
\end{remark}

\begin{lemma}[Quasimode window separation]
\label{lem:qmsep}
Let $A$ compact self-adjoint, $\mu_n > \mu_{n+1}$,
$\varepsilon$-quasimodes $f_n, f_{n+1}$ with $\|G - I\|_{\op} < 1$.
If the window $[\mu_{n+1} - \varepsilon,\, \mu_n + \varepsilon]$
contains exactly two eigenvalues of $A$ (condition~(Q5)),
then
$\lambda_n(A) - \lambda_{n+1}(A) \geq (\mu_n - \mu_{n+1}) - 2\varepsilon$.
\end{lemma}

\begin{proof}
For $j \in \{n, n+1\}$:
$\|(A-\mu_j)f_j\| \leq \varepsilon$, $\|f_j\|=1$
imply $\mathrm{dist}(\mu_j, \sigma(A)) \leq \varepsilon$
(spectral inclusion for approximate eigenvectors).
Linear independence (from $\|G-I\|<1$) excludes collision.
(Q5) provides exact count and index identification.
Ordering and subtracting gives the bound.
\end{proof}

\begin{theorem}[Conditional Quasimode Separation --- Theorem~B]
\label{thm:B}
Under Assumption~\ref{ass:quasimode}:
$\gap{c}{n} \geq \delta_{\mathrm{model}}(c) - 2\varepsilon(c)$.
If $\varepsilon(c) = o(\delta_{\mathrm{model}}(c))$,
then $\gap{c}{n} \geq \frac{1}{2}\delta_{\mathrm{model}}(c)$.
\end{theorem}

\begin{proof}
Apply Lemma~\ref{lem:qmsep} with $A = \Kc$.
\end{proof}

\section{Airy-Scale Roadmap Consequence}
\label{sec:airy}
%% ============================================================

\begin{assumption}[Airy transition asymptotics]
\label{ass:airy}
Assumption~\ref{ass:quasimode} holds in the regime $n \sim 2c/\pi$
with $\delta_{\mathrm{model}}(c) \asymp c^{-1/3}$
and $\varepsilon(c) = o(c^{-1/3})$.
\end{assumption}

\begin{remark}[Status of Assumption~\ref{ass:airy}]
The scale $c^{-1/3}$ is expected from semiclassical
turning-point heuristics in the transition regime.
Proving it requires: (i) asymptotic analysis of the PSWF
kernel near the turning point; (ii) construction of Airy
quasimodes with residual $o(c^{-1/3})$;
(iii) condition~(Q5) in this regime.
None of these is established from first principles.
\end{remark}

\begin{corollary}[Roadmap at Airy scale --- Corollary~C]
\label{cor:C}
Under Assumption~\ref{ass:airy},
$\gap{c}{n} \geq Cc^{-1/3}$ for all large $c$.
\end{corollary}

\begin{remark}[Corollary~C is a roadmap consequence, not a theorem]
\label{rem:corC-status}
Corollary~C records what follows if Assumption~\ref{ass:airy} is verified.
The originality of this paper lies not in Corollary~C
but in Proposition~\ref{prop:gram-fails},
Remark~\ref{rem:circular}, and Remark~\ref{rem:Q5}.
\end{remark}
