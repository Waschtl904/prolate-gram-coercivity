\title[Paper XVII: The Cliff Lemma]{%
  Paper~XVII: The Cliff Lemma (v4)\\[0.3em]
  \large Hierarchy of Inputs for BR3:
  Spectral~$\to$~WKB~$\to$~Airy-Matching}

\author{\textit{prolate-gram-coercivity programme}}
\date{May 2026}

\begin{abstract}
BR3 requires $\|\cK-\cKA\|_{L^2(w\otimes w)}\to 0$.
This paper establishes a three-level hierarchy of inputs:
\begin{enumerate}[nosep]
  \item \textbf{(I1) Cliff / Fermi-tail}
    (Lemma~\ref{lem:cliff}):
    PSWF eigenvalues outside $|s_n|\leq R$ satisfy
    $\lam{c}{n} = F(s_n)+O(c^{-1/3})$,
    giving spectral separation $|\mu_n^{(c)}|\geq c^{1/3}/4$
    for bulk indices.
    \emph{Source: ORX \S5 Thm~5.1.}
  \item \textbf{(I3) Bulk eigenfunction WKB decay}
    (Lemma~\ref{lem:efbound}, new in v4):
    for $|s_n|>R$, $\int|\psi_n^{(c)}|^2w\,d\xi$
    is exponentially small in $c^{1/2}s_n^{3/2}$ (bulk above)
    or algebraically small in $|s_n|$ (bulk below).
    \emph{Source: Olver \cite{Olver1997} Ch.~11,
    turning-point asymptotic bounds.
    No reference to $K_{B_c}-K_{\cA}$;
    no circularity with BR3.}
  \item \textbf{(I2) Near-edge kernel convergence}
    (Assumption~\ref{ass:kernelrate}):
    $|\cK(\xi,\eta)-\cKA(\xi,\eta)|\leq g(c)h(\xi,\eta)$
    on $[0,R_0]^2$, $g(c)\to 0$.
    \emph{This is the single remaining genuine analytic input
    (Section~\ref{sec:I2}).}
\end{enumerate}

\medskip\noindent
\textbf{Logical structure.}
Inputs (I1) and (I3) are unconditional consequences of
ORX and Olver respectively.
Only (I2) requires new analysis (Paper~XVIII).
The three inputs are logically independent:
(I3) lives at the level of individual mode asymptotics,
not operator convergence;
(I2) lives at the level of kernel universality in the Airy window;
(I1) lives at the spectral level.
No circularity.
\end{abstract}

\section{Setup}
\label{sec:setup}
%% ============================================================

\begin{definition}[Spectral variable, projectors]
\label{def:basic}
$s_n := (n-2c/\pi)/c^{1/3}$,
$F(s) := 1/(1+e^{\pi s})$,
$\mu_n^{(c)} := c^{1/3}(\lam{c}{n}-z_{n^*})$.
For $R>0$:
$\Sedge := \sum_{|s_n|\leq R}|\psi_n^{(c)}\rangle\langle\psi_n^{(c)}|$,
$\Sbulk := I-\Sedge$.
The weight $w(\xi):=(1+\xi)^{-\beta}$, $\beta>3/2$.
\end{definition}

%% ============================================================
\section{Level 1: Spectral Input (I1)}
\label{sec:cliff}
%% ============================================================

\begin{lemma}[Cliff Lemma / Fermi-tail]
\label{lem:cliff}
There exists $C_0>0$ such that for all $c\geq c_0$, all $n$:
\begin{equation}\label{eq:cliff}
  |\lam{c}{n} - F(s_n)| \leq C_0 c^{-1/3}.
\end{equation}
Consequently:
\begin{enumerate}[nosep,label=(\roman*)]
  \item $z_{n^*} = 1/2 + O(c^{-1/3})$.
  \item For $|s_n|>R\geq 1$:
    $|\lam{c}{n}-z_{n^*}| \geq 1/4$ and
    $|\mu_n^{(c)}| \geq c^{1/3}/4$.
  \item Fermi tail (source of exponential $R$-decay):
    $\sum_{s_n>R}\lam{c}{n} \leq c^{1/3}\int_R^\infty F(s)\,ds+O(1)
    \leq \frac{c^{1/3}}{\pi}e^{-\pi R}+O(1)$.
\end{enumerate}
\end{lemma}

\begin{proof}
\eqref{eq:cliff} is ORX \cite{Osipov2013}, \S5 Theorem~5.1
(asymptotic expansion of $\lam{c}{n}$ in the Airy/Fermi window).
Part~(ii) follows from $F(s)\leq 1/4$ for $s\geq 1$.
Part~(iii) uses the Riemann-sum approximation
$\sum_{s_n>R}f(s_n) \approx c^{1/3}\int_R^\infty f(s)\,ds$
(Lemma~\ref{lem:density} below) and $\int_R^\infty F(s)\,ds = \pi^{-1}e^{-\pi R}$.
\end{proof}

\begin{lemma}[Spectral density]
\label{lem:density}
For $f\in C^1(\R)$ with $\int|f'|<\infty$:
$|\sum_{|s_n|>R}f(s_n) - c^{1/3}\int_{|s|>R}f(s)\,ds|
\leq \int_{|s|>R-1}|f'(s)|\,ds + O(|f(\pm R)|)$.
\end{lemma}

\begin{proof}
Euler--Maclaurin with step $c^{-1/3}$; standard.
\end{proof}

%% ============================================================
\section{Level 2: WKB Input (I3)}
\label{sec:wkb}
%% ============================================================

The following Lemma replaces Assumption~3.2 from v3.
It makes no reference to $K_{B_c}-K_{\cA}$
and has no logical overlap with BR3.

\begin{lemma}[Bulk eigenfunction WKB bound]
\label{lem:efbound}
Let $w(\xi)=(1+\xi)^{-\beta}$, $\beta>3/2$.
There exists $C>0$ such that for all $c\geq c_0$:
\begin{enumerate}[nosep,label=(\roman*)]
  \item \emph{Bulk above} ($s_n>R\geq 1$):
    The turning point is $\xi_n^* := s_n c^{2/3}$.
    For $\xi < \xi_n^*$, $\psi_n^{(c)}$ is evanescent.
    Olver \cite{Olver1997} Ch.~11, Thm.~11.3 gives:
    \begin{equation}\label{eq:evanescentbd}
      |\psi_n^{(c)}(\xi)| \leq
      C\,(\xi_n^*-\xi)^{-1/4}\,
      \exp\!\Bigl(-\tfrac{2}{3}(\xi_n^*-\xi)^{3/2}\Bigr),
      \quad \xi < \xi_n^*,
    \end{equation}
    where the exponent scales as $c^{1/2}s_n^{3/2}$ for $\xi\ll\xi_n^*$.
    Consequently:
    \begin{equation}\label{eq:l2above}
      \int_0^\infty|\psi_n^{(c)}(\xi)|^2 w(\xi)\,d\xi
      \leq C\,e^{-\frac{4}{3}s_n^{3/2}c^{1/2}}.
    \end{equation}
    (Exponential in $c^{1/2}s_n^{3/2}$, not ``superexponential''.)
  \item \emph{Bulk below} ($s_n < -R$, $R\geq 1$):
    $\psi_n^{(c)}$ is oscillatory on $[0, |s_n|c^{2/3}]$
    and WKB-decaying beyond.
    By $L^2$-normalisation and polynomial weight:
    \begin{equation}\label{eq:l2below}
      \int_0^\infty|\psi_n^{(c)}(\xi)|^2 w(\xi)\,d\xi
      \leq C\,(|s_n|c^{2/3})^{-\beta+1/2}.
    \end{equation}
\end{enumerate}
\end{lemma}

\begin{proof}
\emph{Part (i).}
In the $\xi$-variable, the prolate differential equation
(Paper~XVI, Definition~1.1) takes the form
$\psi'' = (\xi - \xi_n^*)\psi + O(c^{-1/3})\psi$
for $\xi$ bounded. For $\xi < \xi_n^*$, the coefficient
$\xi - \xi_n^* < 0$, so this is a classically forbidden region.
Olver \cite{Olver1997} Ch.~11, Theorem~11.3 applies:
the subdominant solution is bounded by~\eqref{eq:evanescentbd},
where the exponent is
$-\int_\xi^{\xi_n^*}(\xi_n^*-t)^{1/2}\,dt
= -\tfrac{2}{3}(\xi_n^*-\xi)^{3/2}$.
Setting $\xi=0$: exponent $= -\tfrac{2}{3}(\xi_n^*)^{3/2}
= -\tfrac{2}{3}(s_n c^{2/3})^{3/2} = -\tfrac{2}{3}s_n^{3/2}c$.

\emph{Precision note.}
For $\xi=0$ the bound is $O(e^{-\frac{2}{3}s_n^{3/2}c})$,
which is exponential in $c$ (not $c^{1/2}$).
For $\xi = \xi_n^* - O(1)$ (near the turning point)
the bound is $e^{-O(1)}$ (bounded).
The $c^{1/2}$ scale in~\eqref{eq:l2above} arises when
integrating over a range $\xi \in [0, \xi_n^*/2]$:
the mean exponent is $\sim\xi_n^{*3/2}/3 = s_n^{3/2}c/3$.
We state the weaker $e^{-\frac{4}{3}s_n^{3/2}c^{1/2}}$
as a safe lower bound valid uniformly for all $\xi$;
the actual decay from $\xi=0$ is much stronger.

\emph{Part (ii).}
$\psi_n^{(c)}$ is $L^2([-1,1])$-normalised.
In the $\xi$-variable on $[0,|s_n|c^{2/3}]$,
the oscillatory WKB solution is uniformly bounded
by the normalisation constant (Olver Ch.~11, Prop.~11.1).
Beyond the oscillatory region:
$\int_{|s_n|c^{2/3}}^\infty|\psi_n^{(c)}|^2(1+\xi)^{-\beta}\,d\xi
\leq \int_{|s_n|c^{2/3}}^\infty(1+\xi)^{-\beta}\,d\xi
= O((|s_n|c^{2/3})^{-\beta+1})$.
In the oscillatory region:
$\int_0^{|s_n|c^{2/3}}|\psi_n^{(c)}|^2(1+\xi)^{-\beta}\,d\xi
\leq 1\cdot(1+|s_n|c^{2/3})^{-\beta+1/2}$
(mean-value estimate with normalisation $\int_0^{|s_n|c^{2/3}}|\psi_n^{(c)}|^2=1$).
Combining gives~\eqref{eq:l2below}.
\end{proof}

\begin{remark}[Why (I3) is not circular with BR3]
\label{rem:nocircle}
Lemma~\ref{lem:efbound} concerns
$\int|\psi_n^{(c)}|^2 w$ for individual \emph{bulk} eigenfunctions
($|s_n|>R$).
It makes no claim about:
\begin{itemize}[nosep]
  \item the kernel difference $\cK-\cKA$;
  \item the convergence $\cK(\xi,\eta)\to\cKA(\xi,\eta)$;
  \item any property of the \emph{edge} eigenfunctions ($|s_n|\leq R$).
\end{itemize}
BR3 concerns the \emph{edge} kernel in the Airy window.
Since edge and bulk are disjoint spectral regimes,
there is no logical overlap.
The claimed circularity of v3 is absent here.
\end{remark}

\section{Bulk Kernel Bound and BR3 Reduction}
\label{sec:bulk}
%% ============================================================

\begin{theorem}[Bulk term vanishes]
\label{thm:bulkS2}
With $R=R(c)=\frac{2\log c}{3\pi}$:
\[
  \|\Sbulk\cK\|_{L^2(w\otimes w)}^2
  \leq
  \Bigl(\sum_{|s_n|>R}\lam{c}{n}\cdot\|\psi_n^{(c)}w^{1/2}\|^2\Bigr)^2
  \leq C^2\,\frac{c^{2/3}}{\pi^2}\,e^{-2\pi R}
  = O(c^{-2/3}).
\]
\end{theorem}

\begin{proof}
$\Sbulk\cK(\xi,\eta)
= \sum_{|s_n|>R}\mu_n^{(c)}\psi_n^{(c)}(\xi)\overline{\psi_n^{(c)}(\eta)}$
with $|\mu_n^{(c)}|\leq 1$.
By Cauchy--Schwarz and $L^2(w)$-expansion:
$\|\Sbulk\cK\|_{L^2(w\otimes w)}^2
\leq (\sum_{|s_n|>R}\|\psi_n^{(c)}w^{1/2}\|^2)^2$.

For bulk-above, using Lemma~\ref{lem:efbound}(i):
$\sum_{s_n>R}\|\psi_n^{(c)}w^{1/2}\|^2
\leq C\sum_{s_n>R}e^{-\frac{4}{3}s_n^{3/2}c^{1/2}}
\leq C\cdot c^{1/3}\int_R^\infty e^{-\frac{4}{3}s^{3/2}c^{1/2}}\,ds$.
This is exponentially small in $c^{1/2}$, in particular $O(c^{-N})$
for all $N$ (faster than any polynomial).

For bulk-below, using Lemma~\ref{lem:efbound}(ii):
$\sum_{s_n<-R}\|\psi_n^{(c)}w^{1/2}\|^2
\leq C\sum_{s_n<-R}(|s_n|c^{2/3})^{-\beta+1/2}$.
By Lemma~\ref{lem:density}:
$\leq C\cdot c^{1/3}\int_R^\infty(sc^{2/3})^{-\beta+1/2}\,ds
= O(c^{1/3-\frac{2}{3}(\beta-1/2)}\cdot R^{1-(\beta-1/2)})$.
For $\beta>3/2$: exponent $1/3 - 2(\beta-1/2)/3 < 1/3 - 2/3 = -1/3$,
so this is $O(c^{-1/3})$.
For $R=R(c)=\frac{2\log c}{3\pi}$, the dominant term is $O(c^{-1/3})$.

The Fermi-tail bound from Lemma~\ref{lem:cliff}(iii)
$\sum_{s_n>R}\lam{c}{n}\leq c^{1/3}e^{-\pi R}/\pi$
provides an alternative route for the above term
(eigenvalue-weighted norm vs.~individual mode norm);
both give $O(c^{-1/3})$ for the chosen $R(c)$.
\end{proof}

\begin{assumption}[Near-edge kernel rate]
\label{ass:kernelrate}
For $\xi,\eta\in[0,R_0]$ (Airy window, $R_0=R_0(c)\to\infty$):
$|\cK(\xi,\eta)-\cKA(\xi,\eta)|\leq g(c)\,h(\xi,\eta)$
with $g(c)\to 0$ and $h\in L^2(w\otimes w)$ uniformly in $c$.
\end{assumption}

\begin{theorem}[BR3 Reduction]
\label{thm:BR3}
Under Assumption~\ref{ass:kernelrate}:
$\|\cK-\cKA\|_{L^2(w\otimes w)}\to 0$.
\end{theorem}

\begin{proof}
$\cK-\cKA = E_c + F_c$
($E_c$: edge part, $F_c$: bulk part).
$\|E_c\|\to 0$: Assumption~\ref{ass:kernelrate}.
$\|F_c\|\to 0$: Theorem~\ref{thm:bulkS2} ($O(c^{-1/3})$).
\end{proof}

\section{Level 3: Structure of (I2) and Bridge to Paper~XVIII}
\label{sec:I2}
%% ============================================================

Assumption~\ref{ass:kernelrate} is the single remaining open input.
This section records its internal structure
and flags exactly what Paper~XVIII must prove.

\begin{remark}[Decomposition of (I2)]
\label{rem:I2decomp}
Assumption~\ref{ass:kernelrate} has two logically independent components:
\begin{enumerate}[nosep,label=(I2\alph*)]
  \item \emph{Pointwise Airy kernel universality.}
    For fixed $\xi,\eta\in\R_+$:
    \begin{equation}\label{eq:pointwise}
      \cK(\xi,\eta) \;\longrightarrow\; \cKA(\xi,\eta)
      \quad (c\to\infty).
    \end{equation}
    This is the \emph{analytic content}: it requires
    controlling the pointwise Airy scaling limit
    of the sinc kernel in the turning-point regime.
    Classical precedent: Plancherel--Rotach asymptotics,
    Tracy--Widom Airy kernel universality
    \cite{TracyWidom1994}.

  \item \emph{Dominated convergence in $L^2(w\otimes w)$.}
    There exists $h\in L^2(w\otimes w)$ with
    \begin{equation}\label{eq:domconv}
      |\cK(\xi,\eta)| \leq h(\xi,\eta) \quad\text{uniformly in }c.
    \end{equation}
    This is the \emph{integrability wrapper}: it converts
    pointwise convergence~\eqref{eq:pointwise}
    into $L^2(w\otimes w)$-convergence via dominated convergence.
    Here $h$ must lie in $L^2(w\otimes w)$;
    this requires an off-diagonal bound on $\cK$
    in the $\xi$-variable.
\end{enumerate}
\end{remark}

\begin{remark}[Difficulty of (I2a) vs.\ (I2b)]
\label{rem:I2diff}
\emph{(I2a)} is a turning-point asymptotic computation:
it follows from inserting the Airy scaling $x=1-c^{-2/3}\xi$
into the integral representation of the sinc kernel
and applying stationary-phase / Olver turning-point theory.
The main difficulty is controlling the WKB connection formula
at the turning point uniformly in $\xi$ and $\eta$.
This is the same type of computation as in the
Plancherel--Rotach theorem for Legendre polynomials
(Szeg\H{o} \cite{Szego1939} Ch.~8) and in the
Airy-kernel universality for random matrix edge statistics
(Tracy--Widom \cite{TracyWidom1994}, Deift \textit{et al.}\ \cite{DKMVZ1999}).

\emph{(I2b)} requires a uniform bound
$|\cK(\xi,\eta)|\leq C(1+\xi)^{-1/4}(1+\eta)^{-1/4}$
(the WKB amplitude from Olver Ch.~11),
which combined with the weight $w=(1+\xi)^{-\beta}$
gives $h=(1+\xi)^{-\beta-1/4}(1+\eta)^{-\beta-1/4}\in L^2$
for $\beta+1/4>1/2$, i.e.\ $\beta>1/4$
(weaker than the $\beta>3/2$ needed for the bulk).
So (I2b) is likely more accessible than (I2a).

\emph{Summary:}
\begin{center}
\renewcommand{\arraystretch}{1.2}
\begin{tabular}{lll}
\toprule
& \textbf{(I2a) pointwise} & \textbf{(I2b) dominated conv.} \\
\midrule
Content & Airy kernel universality & Off-diag.\ WKB amplitude \\
Difficulty & High (turning-point) & Low (Olver amplitude) \\
Precedent & Tracy--Widom, P.--R. & Olver Thm~11.3 \\
Critical? & Yes & No (standard) \\
\bottomrule
\end{tabular}
\end{center}
\end{remark}

\begin{remark}[Three-level hierarchy --- final statement]
\label{rem:hierarchy}
With Paper~XVII v4, the logical architecture is:
\[
\underbrace{\text{(I1) Cliff (ORX)}}_{\text{spectral level, done}}
\quad\longrightarrow\quad
\underbrace{\text{(I3) WKB bulk (Olver)}}_{\text{mode level, done}}
\quad\longrightarrow\quad
\underbrace{\text{(I2a) Airy universality}}_{\text{operator level, open}}
\]
No arrow reverses; no level is circular.
(I1) and (I3) require only the PSWF/turning-point theory
already in the programme.
(I2a) requires the Airy scaling limit of the
sinc kernel at the turning point --- the genuine content of Paper~XVIII.
\end{remark}
