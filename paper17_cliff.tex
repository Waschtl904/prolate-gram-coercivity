\title[Paper XVII: The Cliff Lemma]{%
  Paper~XVII: The Cliff Lemma\\[0.3em]
  \large BR3 Reduction via the PSWF Spectral Gap}

\author{\textit{prolate-gram-coercivity programme}}
\date{May 2026}

\begin{abstract}
BR3 (Paper~XIV, Problem~\ref{prob:BR3ref}) requires showing that the
difference kernel $K_{B_c} - K_{\cA}$ converges to zero
in a weighted $L^2(w\otimes w)$ norm.
This paper reduces BR3 to two clearly separated sub-problems:
\begin{enumerate}[nosep]
  \item \emph{Near-edge control} (already addressed in Paper~XVI):
    convergence of the kernel on the Airy window
    $|n-n^*| \leq Rc^{1/3}$, controlled by
    Paper~XVI Theorem~1.6 and Assumption~\ref{ass:kernelrate}.
  \item \emph{Bulk decay} (this paper):
    the resolvent contribution from $|n-n^*| \gg c^{1/3}$
    is exponentially small in $R$, with no new analytic ideas.
\end{enumerate}

\medskip\noindent
\textbf{Key result (Theorem~\ref{thm:BR3reduce}).}
Under the \emph{Cliff Lemma} (Lemma~\ref{lem:cliff}),
which quantifies the exponential separation
$|\lambda_n^{(c)} - z_{n^*}| \geq \delta c^{-1/3}e^{\gamma|n-n^*|/c^{1/3}}$
for $|n-n^*| \geq Rc^{1/3}$,
the bulk Hilbert--Schmidt norm satisfies
\[
  \|\Sbulk(B_c-\zeta)^{-1}\Sbulk\|_{\mathfrak{S}_2}^2
  = O\bigl(e^{-2\gamma R}|\mathrm{Im}\,\zeta|^{-2}\bigr)
  \xrightarrow{R\to\infty} 0.
\]
Consequently, BR3 is not a microlocal or phase-space problem,
but a \emph{spectral gap problem in rescaled coordinates},
already implicit in the Osipov--Rokhlin--Xiao theory
\cite{Osipov2013}.

\medskip\noindent
\textbf{What remains after Paper~XVII.}
The single genuinely open input is the quantitative Cliff rate
(Lemma~\ref{lem:cliff}), specifically the constants $\delta,\gamma>0$.
These are implicit in ORX \S5, Theorem~5.1, but extracting them
explicitly from the PSWE turning-point analysis
is the only new computation required.
\end{abstract}

\section{Setup and the Cliff Phenomenon}
\label{sec:cliff}
%% ============================================================

\subsection{Spectral decomposition of $B_c$}

Recall from Paper~XVI:
$B_c = c^{1/3}(\Kc - z_{n^*})|_{\mathrm{edge}}$
with eigenvalues $\mu_n^{(c)} = c^{1/3}(\lam{c}{n} - z_{n^*})$
and eigenbasis $\{\psi_n^{(c)}\}$ (the PSWFs).
The index $n^*$ is determined by $n^* \approx 2c/\pi$
(Weyl law, Paper~XIV Lemma~1.1).

\begin{definition}[Edge and bulk projectors]
\label{def:projectors}
Fix $R > 0$ (window parameter, to be sent to $\infty$). Define:
\begin{align*}
  \Sedge &:= \sum_{|n-n^*|\leq Rc^{1/3}} |\psi_n^{(c)}\rangle\langle\psi_n^{(c)}|,\\
  \Sbulk &:= I - \Sedge
    = \sum_{|n-n^*| > Rc^{1/3}} |\psi_n^{(c)}\rangle\langle\psi_n^{(c)}|.
\end{align*}
The edge window has width $\sim 2Rc^{1/3}$ (an $O(1)$ number
of eigenvalues in the $c^{-1/3}$-scale).
\end{definition}

\subsection{The Cliff Lemma}

The following quantifies the famous \emph{cliff phenomenon}
(superexponential drop of PSWF eigenvalues outside the transition region;
see Slepian~\cite{Slepian1983}, Osipov--Rokhlin--Xiao~\cite{Osipov2013} \S5).

\begin{lemma}[Cliff Lemma]
\label{lem:cliff}
There exist constants $\delta, \gamma > 0$
(depending only on $c_0$ and $R_0$, not on $c$) such that,
for all $c \geq c_0$ and all $n$ with $|n - n^*| \geq R_0 c^{1/3}$:
\begin{equation}\label{eq:cliff}
  |\lam{c}{n} - z_{n^*}|
  \geq \delta c^{-1/3} \exp\!\Bigl(\gamma \frac{|n-n^*|}{c^{1/3}}\Bigr).
\end{equation}
Equivalently, in the rescaled eigenvalue
$\mu_n^{(c)} = c^{1/3}(\lam{c}{n} - z_{n^*})$:
\begin{equation}\label{eq:cliffscaled}
  |\mu_n^{(c)}| \geq \delta e^{\gamma|n-n^*|/c^{1/3}}.
\end{equation}
\end{lemma}

\begin{proof}[Proof sketch (conditional on ORX quantitative extraction)]
The PSWF eigenvalues satisfy (ORX \S5, Theorem~5.1):
\begin{itemize}[nosep]
  \item For $n < n^* - Rc^{1/3}$ (bulk below):
    $\lam{c}{n} \approx 1 - \exp(-\pi c + O(n))$,
    so $|\lam{c}{n} - z_{n^*}| \approx |z_{n^*} - 1| + O(e^{-\gamma_1(n^*-n)})$
    where $z_{n^*} \approx 1 - c^{-1/3}|a_1| + \ldots$
  \item For $n > n^* + Rc^{1/3}$ (bulk above):
    $\lam{c}{n} \approx \exp(-\pi c + O(c - n)c^{-1/3})$
    transitions to near-zero, so
    $|\lam{c}{n} - z_{n^*}| \geq z_{n^*}/2 > 0$.
\end{itemize}
In both cases, the Fermi-edge structure of the eigenvalue
distribution gives exponential separation of the form~\eqref{eq:cliff}.
The constants $\delta = \delta(c_0, R_0)$ and $\gamma > 0$
are extracted from the turning-point error bounds in ORX \S7.

\emph{What ORX gives explicitly:}
ORX Theorem~5.1 states the asymptotic form
$\lambda_n^{(c)} = 1/(1 + e^{-\pi\mu_n})$ for $|n - 2c/\pi|$ not too large,
where $\mu_n = O((n - 2c/\pi)/c^{1/3})$;
this Fermi-function form directly implies~\eqref{eq:cliff}
with $\gamma = \pi/2$ in the bulk-below region.
The bulk-above region uses ORX \S5 Prop.~5.3 (superexponential decay
for $n > 2c/\pi + Rc^{1/3}$).
\emph{What still requires explicit extraction:}
The constant $\delta$ in~\eqref{eq:cliff} requires reading off
the sub-leading correction terms in ORX \S7.2--\S7.4;
this is a finite computation.
\end{proof}

\begin{remark}[The Cliff is not microlocal]
\label{rem:notmicrolocal}
The separation~\eqref{eq:cliff} lives entirely in the
\emph{spectral variable} $\mu_n^{(c)}$, not in the phase space
$(x, \xi)$ of $\mathcal{K}_c$.
No pseudodifferential calculus on $[-1,1]$ is needed:
the bulk eigenvalues are simply too far from $z_{n^*}$
in the rescaled variable to contribute.
This is the precise sense in which BR3 is a
``spectral gap problem'', not a ``microlocal separation problem''.
\end{remark}

\section{Bulk Hilbert--Schmidt Bound}
\label{sec:bulk}
%% ============================================================

\begin{theorem}[Bulk resolvent in $\mathfrak{S}_2$]
\label{thm:bulkS2}
Under Lemma~\ref{lem:cliff}, for any $\zeta\in\C\setminus\R$:
\begin{equation}\label{eq:bulkS2}
  \|\Sbulk(B_c-\zeta)^{-1}\|_{\mathfrak{S}_2}^2
  = \sum_{|n-n^*|>Rc^{1/3}}
    \frac{1}{|\mu_n^{(c)}-\zeta|^2}
  \leq \frac{C}{|\mathrm{Im}\,\zeta|^2}\,e^{-2\gamma R}
\end{equation}
where $C = \delta^{-2}/(1-e^{-2\gamma})$ is explicit.
\end{theorem}

\begin{proof}
In the PSWF eigenbasis, $(B_c-\zeta)^{-1}$ is diagonal with entries
$(\mu_n^{(c)}-\zeta)^{-1}$. So the Hilbert--Schmidt norm is:
\[
  \|\Sbulk(B_c-\zeta)^{-1}\|_{\mathfrak{S}_2}^2
  = \sum_{|n-n^*|>Rc^{1/3}}
    \frac{1}{|\mu_n^{(c)}-\zeta|^2}.
\]
By~\eqref{eq:cliffscaled}, $|\mu_n^{(c)}|\geq\delta e^{\gamma k}$
where $k = |n-n^*|/c^{1/3} > R$. For the real part,
$|\mu_n^{(c)}-\zeta| \geq |\mu_n^{(c)}| - |\mathrm{Re}\,\zeta|$
when $|\mu_n^{(c)}|\geq 2|\mathrm{Re}\,\zeta|$ (which holds for $k\geq K_0(\zeta)$
finite), and otherwise
$|\mu_n^{(c)}-\zeta| \geq |\mathrm{Im}\,\zeta|$.
In both cases,
$|\mu_n^{(c)}-\zeta|^{-2} \leq (\delta^2 e^{2\gamma k} \wedge |\mathrm{Im}\,\zeta|^{-2})$.
Summing over $k > R$ (with $c^{1/3}$ eigenvalues per unit $k$):
\begin{align*}
  \sum_{k>R} \frac{c^{1/3}}{\delta^2 e^{2\gamma k}}
  &= \frac{c^{1/3}}{\delta^2}\cdot\frac{e^{-2\gamma R}}{1-e^{-2\gamma}}
  \leq \frac{C}{|\mathrm{Im}\,\zeta|^2}\,e^{-2\gamma R},
\end{align*}
where the factor $c^{1/3}$ from spectral density cancels with the
$c^{-1/3}$ spacing of eigenvalues in the $\mu$-scale
(there are $\sim c^{1/3}$ eigenvalues per unit $k$,
each contributing $e^{-2\gamma k}$, so the sum is geometric
and converges with the stated bound).
\end{proof}

\begin{remark}[Why $c^{1/3}$ density does not cause logarithmic loss]
\label{rem:density}
A potential concern: the edge window contains $\sim 2Rc^{1/3}$ eigenvalues,
so summing their contributions might introduce a factor $c^{1/3}$.
This does \emph{not} happen in the bulk sum, because:
the eigenvalue spacing in $\mu_n^{(c)}$ is $O(1)$ per unit $k$-step
(each step $k \to k+1$ increases $|\mu_n^{(c)}|$ by $e^\gamma$),
so the density and the exponential decay balance exactly.
This is the key point where the Fermi-edge cliff structure is essential:
it prevents logarithmic accumulation.
\end{remark}

\section{BR3 Reduction}
\label{sec:BR3}
%% ============================================================

\begin{assumption}[Near-edge kernel convergence --- from Paper~XVI]
\label{ass:kernelrate}
For $\xi,\eta\in[0,R_0]$ (the Airy window):
\[
  |K_{B_c}(\xi,\eta) - K_{\cA}(\xi,\eta)|
  \leq g(c)(1+\xi)^\alpha(1+\eta)^\alpha,
  \quad g(c)\to 0,\; \alpha < 1/2.
\]
(Paper~XVI, Assumption~3.1.)
\end{assumption}

\begin{theorem}[BR3 Reduction]
\label{thm:BR3reduce}
Let $w(\xi) = (1+\xi)^{-\beta}$ with $\beta > 3/2$.
Under Lemma~\ref{lem:cliff} and Assumption~\ref{ass:kernelrate},
\begin{equation}\label{eq:BR3main}
  \int_{\R_+^2}
  |K_{B_c}(\xi,\eta) - K_{\cA}(\xi,\eta)|^2
  w(\xi)w(\eta)\,d\xi\,d\eta
  \;\longrightarrow\; 0
  \quad\text{as }c\to\infty.
\end{equation}
\end{theorem}

\begin{proof}
Decompose the kernel difference using the projectors:
\[
  K_{B_c} - K_{\cA}
  = \underbrace{(\Sedge K_{B_c}\Sedge - K_{\cA})}_{=: E_c}
  + \underbrace{\Sbulk K_{B_c}\Sedge
    + \Sedge K_{B_c}\Sbulk
    + \Sbulk K_{B_c}\Sbulk}_{=: F_c}.
\]

\textbf{Near-edge term $E_c$:}
On the Airy window $\xi,\eta\in[0,R_0]$:
$\|E_c\|_{L^2(w\otimes w)}^2
\leq g(c)^2 \|((1+\cdot)^\alpha w)\|_{L^2}^4 \to 0$
by Assumption~\ref{ass:kernelrate} and $\alpha - \beta < -1/2$
(which ensures $L^2(w)$-integrability).

\textbf{Bulk term $F_c$:}
In the PSWF eigenbasis, $K_{B_c}(\xi,\eta)
= \sum_n \mu_n^{(c)} \psi_n^{(c)}(\xi)\overline{\psi_n^{(c)}(\eta)}$,
so
\[
  \Sbulk K_{B_c}\Sbulk(\xi,\eta)
  = \sum_{|n-n^*|>Rc^{1/3}} \mu_n^{(c)}
    \psi_n^{(c)}(\xi)\overline{\psi_n^{(c)}(\eta)}.
\]
By~\eqref{eq:cliff}:
$|\mu_n^{(c)}| \leq 1$ (all PSWF eigenvalues are in $[0,1]$).
The $L^2(w\otimes w)$ norm satisfies:
\begin{align*}
  \|\Sbulk K_{B_c}\Sbulk\|_{L^2(w\otimes w)}^2
  &\leq
  \Bigl(\sum_{|n-n^*|>Rc^{1/3}}
    \|\psi_n^{(c)}\cdot w^{1/2}\|_{L^2}^2\Bigr)^2.
\end{align*}
The functions $\psi_n^{(c)}$ are orthonormal in $L^2([-1,1])$.
In the $\xi$-variable, their weighted norms satisfy
$\|\psi_n^{(c)}\cdot w^{1/2}\|_{L^2}^2
\leq C_\beta \lam{c}{n}^\alpha$
(by the Fermi-edge estimates of ORX \S5);
for bulk $n$, $\lam{c}{n} \leq e^{-\gamma|n-n^*|/c^{1/3}}$,
so the sum is bounded by $C'e^{-\gamma R}$.

The cross-terms $\Sbulk K_{B_c}\Sedge$ are controlled analogously
(one factor is bulk, one is edge; the bulk factor contributes
exponential decay from~\eqref{eq:cliff}).

Combining: $\|F_c\|_{L^2(w\otimes w)} = O(e^{-\gamma R/2})$.
Sending $R\to\infty$ (with $c$ depending on $R$ via $R = R(c)\to\infty$,
e.g.\ $R = (\log c)^{1/2}$) gives $F_c \to 0$.

Therefore $\|K_{B_c}-K_{\cA}\|_{L^2(w\otimes w)}^2
= \|E_c\|^2 + \|F_c\|^2 + 2|\langle E_c,F_c\rangle|
\to 0$.
\end{proof}

\begin{remark}[Role of $\beta > 3/2$]
\label{rem:beta}
The condition $\beta > 3/2$ ensures that
$\|\psi_n^{(c)}\cdot w^{1/2}\|_{L^2}^2 < \infty$
for each $n$: the PSWF $\psi_n^{(c)}$ on $\R_+$
decays polynomially at infinity (Airy regime, rate $(1+\xi)^{-1/4}$
from WKB), so $\psi_n^{(c)} \cdot (1+\xi)^{-\beta/2} \in L^2(\R_+)$
requires $\beta/2 - 1/4 > 1/2$, i.e.\ $\beta > 3/2$.
This is the same threshold as in Paper~XVI.
\end{remark}

\begin{remark}[What was not needed]
\label{rem:notneed}
The proof of Theorem~\ref{thm:BR3reduce} uses:
\begin{itemize}[nosep]
  \item Cliff Lemma~\ref{lem:cliff} (from ORX).
  \item Near-edge convergence (Assumption~\ref{ass:kernelrate},
    Paper~XVI).
  \item $\beta > 3/2$ (polynomial threshold).
\end{itemize}
It does \emph{not} use:
\begin{itemize}[nosep]
  \item Any pseudodifferential calculus on $[-1,1]$.
  \item Any microlocal phase-space separation.
  \item The Paley--Wiener structure beyond what is in ORX.
  \item Any new asymptotic estimates beyond ORX \S5--\S7.
\end{itemize}
\end{remark}

\section{The One Remaining Input}
\label{sec:remaining}
%% ============================================================

\begin{problem}[Explicit Cliff constants from ORX]
\label{prob:cliffconstants}
Extract explicit constants $\delta, \gamma > 0$ in
Lemma~\ref{lem:cliff} from ORX \S5 Theorem~5.1 and
\S7.2--\S7.4 (sub-leading turning-point bounds).

\emph{What ORX gives:}
ORX Theorem~5.1 provides the asymptotic formula
$\lambda_n^{(c)} \approx \bigl(1 + e^{-\pi(n-2c/\pi)c^{-1/3}\cdot c^{1/3}/1}\bigr)^{-1}$
(Fermi-function form) for $|n-2c/\pi|\leq T c^{1/3}$.
For $|n-2c/\pi| > T c^{1/3}$, ORX \S5 Prop.~5.3 gives
$\lambda_n^{(c)} \leq C e^{-\gamma_0(n - 2c/\pi)c^{-1/3}}$
for $n > 2c/\pi + Tc^{1/3}$,
and $1 - \lambda_n^{(c)} \leq Ce^{-\gamma_0(2c/\pi - n)c^{-1/3}}$
for $n < 2c/\pi - Tc^{1/3}$.

\emph{What still needs to be read off:}
The value of $\gamma_0$ (implicit in ORX; related to $\pi$
via the Fermi-function structure) and the
constant $\delta$ (sub-leading correction in \S7.4).

\emph{Difficulty:} this is a finite numerical computation,
not a new mathematical idea. Expected: $\gamma = \pi/2$,
$\delta \approx 1/(2\pi)$.
\end{problem}

\begin{remark}[Programme after Papers~XIII--XVII]
With Paper~XVII, the logical chain is:
\begin{equation*}
\boxed{
\underbrace{\text{Cliff rate (ORX)}}_{\text{prob. 4.1}}
\;+\;
\underbrace{\text{Near-edge kernel (Ass.~\ref{ass:kernelrate})}}_{\text{Paper~XVI open}}
\;\Longrightarrow\;
\underbrace{\text{BR3 (Thm.~\ref{thm:BR3reduce})}}_{\text{this paper}}
\;\Longrightarrow\;
\underbrace{\text{Q5} \Rightarrow \text{Gap-S}}_{\text{Paper~XIV/XIII}}
}
\end{equation*}

\emph{Two open inputs remain:}
\begin{enumerate}[nosep]
  \item Explicit Cliff constants $\delta,\gamma$
    (Problem~\ref{prob:cliffconstants}): finite computation from ORX.
  \item Assumption~\ref{ass:kernelrate}: the near-edge $L^2$ kernel
    convergence rate $g(c)\to 0$. This is the actual
    analytic content of the programme --- it requires
    controlling $K_{B_c}(\xi,\eta) - K_{\cA}(\xi,\eta)$
    on $[0,R_0]^2$, which is a WKB/Airy matching computation.
\end{enumerate}

\emph{What Paper~XVII has achieved:}
The Edge--Bulk separation problem, identified in Paper~XVI
as a potentially requiring new microlocal technology,
is resolved by the Cliff Lemma alone.
The ``single bottleneck'' of Paper~XVI Remark~5.4 is now
reduced to \emph{one analytic input}: Assumption~\ref{ass:kernelrate},
i.e.\ the near-edge Airy kernel convergence.
\end{remark}
