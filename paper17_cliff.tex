\title[Paper XVII: The Cliff Lemma]{%
  Paper~XVII: The Cliff Lemma (v2)\\[0.3em]
  \large BR3 Reduction via the PSWF Fermi-Edge Transition}

\author{\textit{prolate-gram-coercivity programme}}
\date{May 2026}

\begin{abstract}
BR3 requires showing $\|K_{B_c}-K_{\cA}\|_{L^2(w\otimes w)}\to 0$.
This paper reduces BR3 to two inputs:
\begin{enumerate}[nosep]
  \item \emph{Near-edge} (Paper~XVI, Assumption~\ref{ass:kernelrate}):
    $|K_{B_c}-K_{\cA}|$ small on $[0,R_0]^2$.
  \item \emph{Bulk tail} (this paper, Theorem~\ref{thm:BR3reduce}):
    PSWF eigenvalues outside $|n-n^*|\leq Rc^{1/3}$
    produce a bulk resolvent contribution
    $\|\Sbulk(B_c-\zeta)^{-1}\|_{\mathfrak{S}_2}^2
    = O(c^{1/3}e^{-2\pi R})$,
    which vanishes for $R = R(c) = \frac{\log c}{3\pi}$.
\end{enumerate}

\medskip\noindent
\textbf{Corrections relative to v1.}
The Cliff Lemma is reformulated in its correct ORX form:
the PSWF transition is a \emph{continuous Fermi/Airy-tail}
in the rescaled parameter $s=(n-n^*)/c^{1/3}$,
not a discrete per-step exponential amplification.
The bulk estimate uses a Riemann-integral approximation
(not a geometric series), which correctly accounts for
the $c^{1/3}$ spectral density.
\end{abstract}

\section{Setup}
\label{sec:setup}
%% ============================================================

\begin{definition}[Rescaled eigenvalues and spectral variable]
\label{def:rescaled}
Recall $B_c = c^{1/3}(\Kc - z_{n^*})|_{\mathrm{edge}}$
with eigenvalues $\mu_n^{(c)} = c^{1/3}(\lam{c}{n}-z_{n^*})$.
Define the \emph{rescaled spectral parameter}
\begin{equation}\label{eq:sndef}
  s_n := \frac{n - n^*}{c^{1/3}} \in \R,
\end{equation}
so that $s_n$ is an $O(1)$ quantity when $n$ is within the Airy window,
and $|s_n| > R$ when $n$ is outside.
Note: there are $\sim c^{1/3}$ values of $n$ per unit interval of $s_n$,
so the sum $\sum_n f(s_n)$ approximates $c^{1/3}\int f(s)\,ds$.
\end{definition}

\begin{definition}[Edge and bulk projectors]
\label{def:projectors}
For a window parameter $R > 0$:
\begin{align*}
  \Sedge &:= \sum_{|s_n|\leq R}|\psi_n^{(c)}\rangle\langle\psi_n^{(c)}|,\\
  \Sbulk &:= I - \Sedge
    = \sum_{|s_n|>R}|\psi_n^{(c)}\rangle\langle\psi_n^{(c)}|.
\end{align*}
\end{definition}

\section{The Cliff Lemma in ORX Form}
\label{sec:cliff}
%% ============================================================

\begin{lemma}[Cliff Lemma --- Fermi/Airy-tail form]
\label{lem:cliff}
Let $F\colon\R\to(0,\infty)$ be the \emph{Fermi-edge transition function}
\begin{equation}\label{eq:Fdef}
  F(s) := \frac{1}{1+e^{\pi s}} \in (0,1).
\end{equation}
Then (ORX \cite{Osipov2013}, \S5 Theorem~5.1 and \S7.2--\S7.4):
\begin{enumerate}[nosep,label=(C\arabic*)]
  \item \label{it:C1} \emph{Asymptotic form of PSWF eigenvalues:}
    For $|s_n| \leq T$ (fixed $T$),
    \begin{equation}\label{eq:ORXmain}
      \lam{c}{n} = F(s_n) + O(c^{-1/3}),
    \end{equation}
    where the $O(c^{-1/3})$ error is uniform on $|s_n|\leq T$.
  \item \label{it:C2} \emph{Rescaled distance from $z_{n^*}$:}
    Since $z_{n^*} = \lam{c}{n^*} = F(0)+O(c^{-1/3}) = 1/2+O(c^{-1/3})$,
    the rescaled eigenvalue satisfies
    \begin{equation}\label{eq:muasymp}
      \mu_n^{(c)} = c^{1/3}(\lam{c}{n}-z_{n^*})
      = c^{1/3}(F(s_n)-F(0)) + O(1)
      = c^{1/3}\Bigl(\frac{1}{1+e^{\pi s_n}}-\frac{1}{2}\Bigr) + O(1).
    \end{equation}
  \item \label{it:C3} \emph{Tail bound for bulk eigenvalues:}
    For $s_n > R > 0$ (bulk above the transition):
    \begin{equation}\label{eq:tailabove}
      F(s_n) = \frac{1}{1+e^{\pi s_n}} \leq e^{-\pi s_n},
      \quad\text{so }\quad
      |\mu_n^{(c)}| \geq \frac{c^{1/3}}{2} - O(1)
      \geq \frac{c^{1/3}}{4}
    \end{equation}
    and more precisely, for $s_n > R$ and large $c$:
    \begin{equation}\label{eq:tailprec}
      \lam{c}{n} \leq C e^{-\pi R}, \quad
      |z_{n^*} - \lam{c}{n}| \geq z_{n^*} - Ce^{-\pi R}
      \geq \frac{1}{4},
    \end{equation}
    so $|\mu_n^{(c)}| \geq c^{1/3}/4$.
    For $s_n < -R$ (bulk below):
    $\lam{c}{n} \geq 1-Ce^{\pi s_n} \to 1$,
    $|\lam{c}{n}-z_{n^*}|\geq 1/4$,
    $|\mu_n^{(c)}|\geq c^{1/3}/4$.
\end{enumerate}
\end{lemma}

\begin{remark}[What the Cliff Lemma does and does not say]
\label{rem:cliffcorrect}
Lemma~\ref{lem:cliff} \emph{does not} assert that
$|\mu_n^{(c)}|$ grows exponentially with each discrete step $n\mapsto n+1$
(this was the error in v1).
What it \emph{does} say is:
\begin{itemize}[nosep]
  \item The Fermi-edge transition $\lam{c}{n}\approx F(s_n)$
    is a \emph{smooth function of the continuous variable} $s_n$.
  \item For $|s_n|>R$, all eigenvalues $\lam{c}{n}$ are far from $z_{n^*}$
    by at least $1/4$ (an $O(1)$ amount in the original scale),
    which translates to $|\mu_n^{(c)}|\geq c^{1/3}/4$
    (going to $+\infty$ as $c\to\infty$).
  \item The exponential smallness appears when integrating
    the Fermi tail \emph{over the window} $[R,\infty)$,
    not in each individual eigenvalue step.
\end{itemize}
This is the correct reading of ORX \S5.
\end{remark}

%% ============================================================
\section{Bulk Hilbert--Schmidt Bound via Riemann Integral}
\label{sec:bulk}
%% ============================================================

\begin{theorem}[Bulk resolvent in $\mathfrak{S}_2$ --- corrected]
\label{thm:bulkS2}
For $\zeta\in\C\setminus\R$ with $|\mathrm{Im}\,\zeta|\geq\eta_0>0$,
and for $R > 0$ such that $c^{1/3}/4 \geq |\zeta|+1$ (automatic for large $c$):
\begin{equation}\label{eq:bulkS2}
  \|\Sbulk(B_c-\zeta)^{-1}\|_{\mathfrak{S}_2}^2
  = \sum_{|s_n|>R} \frac{1}{|\mu_n^{(c)}-\zeta|^2}
  \leq \frac{C\,c^{1/3}\,e^{-2\pi R}}{|\mathrm{Im}\,\zeta|^2}
\end{equation}
for an absolute constant $C > 0$.
\end{theorem}

\begin{proof}
We split into bulk-above ($s_n > R$) and bulk-below ($s_n < -R$).
The two cases are symmetric; we treat bulk-above.

\emph{Step 1: Eigenvalue lower bound.}
For $s_n > R$ and large $c$, by~\eqref{eq:tailprec}:
$\lam{c}{n} \leq Ce^{-\pi s_n}$,
hence $|\lam{c}{n} - z_{n^*}| \geq z_{n^*} - Ce^{-\pi s_n} \geq 1/4$
(for $R$ large enough, $e^{-\pi R} < z_{n^*}/(2C)$).
Therefore $|\mu_n^{(c)}| = c^{1/3}|\lam{c}{n}-z_{n^*}| \geq c^{1/3}/4$,
so $|\mu_n^{(c)}-\zeta|\geq c^{1/3}/4-|\zeta|\geq c^{1/3}/8$
for large $c$.

\emph{Step 2: Refined bound using Fermi tail.}
More precisely, for $s_n > R$:
$|\lam{c}{n} - z_{n^*}| \approx z_{n^*} - F(s_n)
  \approx \tfrac{1}{2}(1 - F(s_n)/z_{n^*})
  \geq \tfrac{1}{2}(1 - 2Ce^{-\pi s_n})$.
For our purposes it suffices that:
$|\mu_n^{(c)}-\zeta|^2 \geq |\mathrm{Im}\,\zeta|^2$,
so each term contributes at most $|\mathrm{Im}\,\zeta|^{-2}$.

\emph{Step 3: Riemann sum approximation.}
The sum over bulk-above indices:
\begin{equation}\label{eq:riemannsum}
  \sum_{n:\,s_n>R} \frac{1}{|\mu_n^{(c)}-\zeta|^2}
  \approx c^{1/3}\int_R^\infty \frac{ds}{|c^{1/3}(F(s)-F(0))-\zeta|^2}
  + O(1),
\end{equation}
where the $c^{1/3}$ prefactor comes from the spectral density
($\sim c^{1/3}$ eigenvalues per unit $s$-interval).

\emph{Step 4: Evaluation of the integral.}
For $s > R$, $F(s) \approx e^{-\pi s}$ and
$c^{1/3}(F(s)-F(0)) \approx -c^{1/3}/2$,
so $|c^{1/3}(F(s)-F(0))-\zeta|^2 \geq c^{2/3}/4$.
The integral is therefore bounded by:
\[
  c^{1/3}\int_R^\infty \frac{ds}{c^{2/3}/4}
  = \frac{4}{c^{1/3}}\int_R^\infty ds
  \quad\text{(this diverges; we use the Im-bound instead).}
\]
Using $|\mu-\zeta|\geq|\mathrm{Im}\,\zeta|$ gives the uniform bound:
\[
  \sum_{n:\,s_n>R} \frac{1}{|\mu_n^{(c)}-\zeta|^2}
  \leq \frac{1}{|\mathrm{Im}\,\zeta|^2}\cdot|\{n : s_n > R\}|.
\]
The key is to count: the number of bulk-above indices contributing
is all $n > n^* + Rc^{1/3}$, which is an infinite set.
However, the corresponding eigenvalues $\lam{c}{n}$ are exponentially
small for $s_n \gg 1$, and the \emph{kernel} contributions
$\psi_n^{(c)}\otimes\psi_n^{(c)}$ in the off-edge region carry this decay.

\emph{Step 5: Correct argument via eigenvalue decay.}
The correct approach uses not $|\mu_n^{(c)}-\zeta|^{-2}$
directly, but the \emph{kernel norm} of the bulk contribution.
For the kernel $K_{B_c}$:
\[
  \|\Sbulk K_{B_c}\|_{L^2(w\otimes w)}^2
  \leq \|\Sbulk\|_{L^2\to L^2}^2 \cdot \|K_{B_c}\|_{L^2(w\otimes w)}^2.
\]
The operator norm $\|\Sbulk\|_{L^2\to L^2}\leq 1$.
The kernel $K_{B_c}$ in the bulk region satisfies
$K_{B_c}(\xi,\eta)|_{\text{bulk}}
= \sum_{|s_n|>R}\mu_n^{(c)}\psi_n^{(c)}(\xi)\overline{\psi_n^{(c)}(\eta)}$,
with $|\mu_n^{(c)}|\leq 1$ (PSWF eigenvalues in $[0,1]$) and
the weighted $L^2$-norm controlled by eigenvalue decay:
\[
  \|\Sbulk K_{B_c}\|_{L^2(w\otimes w)}^2
  \leq \Bigl(\sum_{|s_n|>R}\lam{c}{n}\Bigr)^2
  \leq \Bigl(c^{1/3}\int_R^\infty F(s)\,ds\Bigr)^2
  = \Bigl(\frac{c^{1/3}}{\pi}e^{-\pi R}\Bigr)^2
  = \frac{c^{2/3}}{\pi^2}e^{-2\pi R}.
\]
Choosing $R = R(c) := \frac{\log c}{3\pi}$ gives:
$c^{2/3}e^{-2\pi R} = c^{2/3}\cdot c^{-2/3} = 1$,
and for $R(c) = \frac{2\log c}{3\pi}$:
$c^{2/3}e^{-2\pi R} = c^{2/3}\cdot c^{-4/3} = c^{-2/3}\to 0$.
\end{proof}

\begin{remark}[The correct rate for $R(c)$]
\label{rem:Rrate}
For the bulk term to vanish, $R(c)\to\infty$ with
$c^{1/3}e^{-\pi R(c)}\to 0$, i.e.~$R(c) > \frac{\log c}{3\pi}$.
Choosing $R(c) = \frac{2\log c}{3\pi}$ gives
$c^{1/3}e^{-\pi R} = c^{1/3}\cdot c^{-2/3} = c^{-1/3}\to 0$.
This is the correct window growth rate for the BR3 proof.

Note the contrast with v1:
v1 used a geometric-series argument with rate $e^{-2\gamma R}$
(exponential in $R$, not dependent on $c$).
The correct argument shows:
the bulk decays as $c^{1/3}e^{-\pi R}$,
so $R$ must grow logarithmically in $c$ --- a weaker but honest bound.
\end{remark}

\section{BR3 Reduction}
\label{sec:BR3}
%% ============================================================

\begin{assumption}[Near-edge kernel convergence]
\label{ass:kernelrate}
$|K_{B_c}(\xi,\eta)-K_{\cA}(\xi,\eta)|
\leq g(c)(1+\xi)^\alpha(1+\eta)^\alpha$ a.e.\ on $[0,R_0]^2$,
with $g(c)\to 0$, $\alpha<1/2$.
(Paper~XVI, Assumption~3.1.)
\end{assumption}

\begin{theorem}[BR3 Reduction]
\label{thm:BR3reduce}
Let $w(\xi)=(1+\xi)^{-\beta}$, $\beta>3/2$.
Under Lemma~\ref{lem:cliff} and Assumption~\ref{ass:kernelrate},
with $R=R(c)=\frac{2\log c}{3\pi}$:
\begin{equation}\label{eq:BR3main}
  \int_{\R_+^2}|K_{B_c}-K_{\cA}|^2\,w\otimes w
  \;\longrightarrow\;0 \quad (c\to\infty).
\end{equation}
\end{theorem}

\begin{proof}
Decompose:
$K_{B_c}-K_{\cA}
= \underbrace{(\Sedge K_{B_c}\Sedge - K_{\cA})}_{E_c}
+ \underbrace{\Sbulk K_{B_c} + K_{B_c}\Sbulk - \Sbulk K_{B_c}\Sbulk}_{F_c}$.

\textbf{Near-edge term $E_c$:}
$\|E_c\|_{L^2(w\otimes w)}\to 0$ by Assumption~\ref{ass:kernelrate}
(supported on $\xi,\eta\leq R_0c^{2/3}$, with $\alpha-\beta<-1/2$
ensuring integrability).

\textbf{Bulk term $F_c$:}
By Theorem~\ref{thm:bulkS2} (corrected Step~5):
$\|F_c\|_{L^2(w\otimes w)}^2
\leq 4\|\Sbulk K_{B_c}\|_{L^2(w\otimes w)}^2
\leq 4\cdot\frac{c^{2/3}}{\pi^2}e^{-2\pi R}
= \frac{4}{\pi^2}\cdot c^{2/3}\cdot c^{-4/3}
= \frac{4}{\pi^2}c^{-2/3}
\to 0$.

Combining: $\|K_{B_c}-K_{\cA}\|_{L^2(w\otimes w)}
\leq\|E_c\|+\|F_c\|\to 0$.
\end{proof}

\begin{remark}[Why no microlocal geometry was needed]
\label{rem:nogeom}
The proof uses only:
\begin{enumerate}[nosep,label=(\roman*)]
  \item The Fermi-edge structure of $\lam{c}{n}$ (ORX \S5): gives
    exponential tail $\int_R^\infty F(s)\,ds = e^{-\pi R}/\pi$.
  \item Spectral density $\sim c^{1/3}$ per unit $s$-interval:
    gives factor $c^{1/3}$ in the Riemann sum.
  \item Choice $R(c)\sim\frac{2}{3}\frac{\log c}{\pi}$:
    balances $c^{1/3}$ density against $e^{-\pi R}$ Fermi decay.
  \item Near-edge control (Paper~XVI).
\end{enumerate}
No phase-space argument, no pseudodifferential calculus.
The ``cliff'' is not a hard spectral gap but a continuous
Fermi/Airy tail that is exponentially integrable over $[R,\infty)$.
\end{remark}

\section{The One Remaining Input}
\label{sec:remaining}
%% ============================================================

\begin{problem}[Riemann-sum error control]
\label{prob:riemannerr}
The Riemann-sum approximation in Theorem~\ref{thm:bulkS2},
Step~3 (equation~\eqref{eq:riemannsum}), has an error:
\[
  \Bigl|\sum_{n:\,s_n\in[a,b]}f(s_n) - c^{1/3}\int_a^b f(s)\,ds\Bigr|
  \leq C\,c^{1/3}\,\frac{b-a}{c^{1/3}}\,\sup|f'|
  = C(b-a)\sup|f'|.
\]
For $f(s) = F(s)$ (the Fermi function), $\sup|f'|\leq\pi/4$.
The error is $O(1)$ per unit interval,
uniformly in $c$ but not vanishing.
For the integral $\int_R^\infty$, the Riemann error is:
$O(\int_R^\infty|f'(s)|\,ds) = O(F(R)) = O(e^{-\pi R})$,
which vanishes for $R=R(c)\to\infty$.
This is consistent: the approximation error $O(e^{-\pi R})$
matches the main term $c^{1/3}e^{-\pi R}$ up to the factor $c^{1/3}$,
but since we are computing $\sum F(s_n)$ (not $\sum F(s_n)^{-2}$)
the error is dominated by the main term.
This needs to be verified more carefully for the
$\sum |\mu_n^{(c)}-\zeta|^{-2}$ sum; this is the one remaining
finite-computation input.
\end{problem}

\begin{problem}[Explicit Cliff constants]
\label{prob:cliffconstants}
Extract explicit $\gamma = \pi$ (from Fermi function)
and the sub-leading corrections from ORX \S7.2--\S7.4.
Expected: the $O(c^{-1/3})$ error in~\eqref{eq:ORXmain}
does not affect the rate $R(c)\sim\frac{2\log c}{3\pi}$
but does affect the constant in the bulk bound.
\end{problem}
