\begin{abstract}
We reduce the open amplitude hypothesis~(H2) of
Paper~VII \cite{paperVII} to a single uniform Airy
approximation over the transition window
(Proposition~\ref{prop:U}): a uniform $O(c^{-2/3})$
error for PSWF eigenvalues $\lambda_l$ relative to
$\FAi(x_l)$ over $O(c^{1/3})$ indices simultaneously.
All results are conditional on Proposition~\ref{prop:U}
(stated as an assumption; see Remark~\ref{rem:propU_route}
for a proof strategy) and on Assumption~B-strong
\cite{paperVI}.

The main contributions are:
\begin{itemize}
\item Lemma~\ref{lem:A} (Uniform Airy Increment): a
  discrete-to-continuous stability result for PSWF eigenvalue
  differences $\lambda_l - \lambda_{l+1}$ in the Airy
  transition regime.
\item Lemma~\ref{lem:B} (Airy Ratio Bound): uniform control
  of the ratio $\mathrm{Ai}(x)^2 / \FAi(x)$
  on a compact interval containing no zero of $\mathrm{Ai}$.
\item Lemma~\ref{lem:F} (Airy Rigidity / Dyadic Separation
  Principle): a quantitative bi-Lipschitz bound on $\FAi$
  and a corresponding spectral separation estimate
  $|\lambda_i - \lambda_j| \gtrsim c^{-1/3} \cdot |i-j|$,
  with explicit threshold $c_1 = c_1(A_0, C_1, c_{\min})$.
\item Corollary~\ref{cor:C} (H2 in truncated dyadic sense):
  for any fixed dyadic cutoff $m_0 \ge 1$,
  $\delta_m = O(2^{-m}) + O(c^{-1/3})$ via a
  two-parameter argument.
\item Lemma~\ref{lem:E} (Core--WKB Split): the Core regime
  $m < m_0$ is controlled directly by H3;
  Theorem~2.1 of \cite{paperVII} is applied to the
  truncated operator with $m \ge m_0$ only.
\end{itemize}
The central structural mechanism is the
\emph{Dyadic Separation Principle}
(Lemma~\ref{lem:F}, Remark~\ref{rem:dyadic}):
\[
  \frac{|\lambda_j - \lambda_{j+1}|}{|\lambda_i - \lambda_j|}
  \;\sim\; 2^{-m}
  \quad \text{for } j \in \mathcal{B}_m(i),
\]
which is the structural mechanism behind H2 and is
not contained in ORX \cite{OsipovRokhlinXiao}.
\end{abstract}

% ============================================================
\section{Introduction and Context}
\label{sec:intro}
% ============================================================

Paper~VII \cite{paperVII} established an abstract dyadic
cancellation theorem (Theorem~2.1) and reduced the
contraction problem for the PSWF Galerkin operator
$\mathcal{T}_c^{(N)}$ to the verification of three
hypotheses:
\begin{itemize}
\item[(H1)] Non-degenerate oscillatory phase: verified
  unconditionally from Widom asymptotics.
\item[(H2)] Blockwise amplitude regularity: the relative
  variation of $A_{ij}$ within each dyadic block
  $\mathcal{B}_m(i)$ is $o(1)$ as $c \to \infty$,
  for each fixed $m \ge m_0$.
\item[(H3)] Uniform amplitude bound: $A_{ij} \le C$
  for all $i \ne j$; equivalent to Assumption~B-strong
  \cite{paperVI}.
\end{itemize}

H2 was stated in \cite{paperVII} as Conjecture~6.2.
The present paper does not unconditionally close H2,
but reduces it to a single analytic input:

\begin{quote}
\textit{Proposition~\ref{prop:U}: a uniform
$O(c^{-2/3})$ Airy approximation for PSWF eigenvalues
over the entire transition window of size $O(c^{1/3})$.}
\end{quote}

This is the missing ingredient that ORX
\cite{OsipovRokhlinXiao} provides only pointwise.
The present paper is therefore a
\emph{conditional reduction theorem}:
under Proposition~\ref{prop:U} and B-strong,
all hypotheses (H1)--(H3) are verified and the
contraction estimate follows.

\medskip\noindent
\textbf{The Dyadic Separation Principle.}
The new structural insight of this paper is that
the ratio
\[
  \frac{|\lambda_j - \lambda_{j+1}|}{|\lambda_i - \lambda_j|}
  \;\sim\; 2^{-m}
  \quad (j \in \mathcal{B}_m(i))
\]
decays purely combinatorially in $m$, independently of $c$.
This is the operative content of Lemma~\ref{lem:F}:
both numerator and denominator scale as $c^{-1/3}$,
but the denominator carries an extra $2^m$
from $|i-j| \ge 2^m$, so the $c$-dependence cancels.
This mechanism does not appear in ORX.

\medskip\noindent
\textbf{H2 in truncated dyadic sense.}
H2 does not hold at all dyadic scales:
for $m = 0$, the relative amplitude variation is $O(1)$.
The correct statement is two-regime:
\[
  \text{H2 holds for } m \ge m_0;
  \quad
  m < m_0 \text{ controlled by H3.}
\]
Theorem~2.1 of \cite{paperVII} is applied to the
truncated operator (blocks $m \ge m_0$ only).

\medskip\noindent
\textbf{What is not proved here.}
Proposition~\ref{prop:U} is stated as an assumption;
see Remark~\ref{rem:propU_route} for a proof strategy.
B-strong remains open; see Remark~\ref{rem:open}.

\medskip\noindent
\textbf{Structure.}
Section~\ref{sec:setup} fixes notation.
Section~\ref{sec:propU} states Proposition~U.
Sections~\ref{sec:lemmaA}--\ref{sec:lemmaF} prove
the supporting lemmas.
Section~\ref{sec:H2} proves Corollary~C.
Section~\ref{sec:split} proves Lemma~E.
Section~\ref{sec:closure} collects the reduction.

\section{Setup and Notation}
\label{sec:setup}
% ============================================================

Throughout, $c \ge 1$ is the bandwidth parameter and
$\lambda_k = \lambda_k^{(c)} \in (0,1)$ are the PSWF
eigenvalues, ordered decreasingly.

\medskip\noindent
\textbf{Airy zeros and the parameter $A_0$.}
All real zeros of $\mathrm{Ai}$ are negative
(since $\mathrm{Ai}(x) > 0$ for all $x \ge 0$
and the zeros are the negative values
$a_1 > a_2 > \cdots$, $a_1 \approx -2.3381$
\cite[Sec.~9.9]{DLMF}).
We fix once and for all:
\[
  \boxed{A_0 \in (0,\, |a_1|) \approx (0,\, 2.3381).}
\]
Then $[-A_0, A_0]$ contains no zero of $\mathrm{Ai}$;
all quantities involving $\mathrm{Ai}$ on this interval
are uniformly bounded above and below.

\medskip\noindent
\textbf{Compactness constants.}
We introduce three constants, depending only on $A_0$:
\begin{align*}
  c_* &:= \min_{x\in[-A_0,\,0]}\FAi(x) > 0,\\
  c_{\min} &:= \min_{x\in[-A_0,\,A_0]}\mathrm{Ai}(x)^2 > 0,
  \qquad
  C_{\max} := \max_{x\in[-A_0,\,A_0]}\mathrm{Ai}(x)^2 < \infty.
\end{align*}
Positivity and finiteness follow from continuity on a
compact zero-free set.

\medskip\noindent
\textbf{Lipschitz constant of $\mathrm{Ai}^2$.}
Since $\mathrm{Ai} \in C^\infty(\mathbb{R})$, the
function $\mathrm{Ai}(x)^2$ is $C^1$ on $[-A_0, A_0]$
with bounded derivative. We define:
\[
  L_{A_0} := \sup_{x \in [-A_0,\,A_0]}
  \bigl|\tfrac{d}{dx}\mathrm{Ai}(x)^2\bigr|
  = 2\sup_{x \in [-A_0,A_0]}|\mathrm{Ai}(x)\,\mathrm{Ai}'(x)|
  < \infty.
\]
This is the Lipschitz constant of $\mathrm{Ai}^2$
on $[-A_0, A_0]$, used in the error bound
$|\mathrm{Ai}(\xi)^2 - \mathrm{Ai}(x)^2| \le L_{A_0}|\xi - x|$
in the proof of Lemma~\ref{lem:A}.

\medskip\noindent
\textbf{Uniformity of constants.}
All constants $c_*, c_{\min}, C_{\max}, L_{A_0}$ depend
only on $A_0$ and are uniform in $c$, $i$, $j$, $m$.

\medskip\noindent
\textbf{Rescaled index.}
\[
  x_l := (l - \NSh)\,(c/2)^{-1/3},
  \qquad \NSh := \frac{2c}{\pi}.
\]
This is a definition; the spacing
$x_{l+1} - x_l = (c/2)^{-1/3}$ is exactly constant.
All Galerkin indices ($N = \lfloor Ac^{1/3}\rfloor$,
$A < |a_1|$) satisfy $x_l \in [-A,0] \subset [-A_0,A_0]$.

\medskip\noindent
\textbf{Further definitions:}
\begin{align*}
  \FAi(x) &:= \int_x^\infty \mathrm{Ai}(t)^2\,dt,\\
  A_{ij} &:= P_{ij}\cdot c^{1/2},
  \quad
  P_{ij} := c\cdot
  \frac{|\lambda_i-\lambda_j|}{(1-\lambda_i)(1-\lambda_j)}
  \quad (i\ne j).
\end{align*}

Dyadic blocks: $\Bm(i) := \{j : 2^m \le |i-j| < 2^{m+1}\}$.

\medskip\noindent
\textbf{Explicit bound on $1-\lambda_j$.}
Under Proposition~\ref{prop:U} with $|E_j| \le C_1 c^{-2/3}$
and $\FAi(x_j) \ge c_* > 0$,
for $c \ge c_0$ such that $C_1 c_0^{-2/3} \le c_*/2$:
\begin{equation}\label{eq:onemlam}
  \tfrac{1}{2}\,\FAi(x_j)
  \;\le\; 1-\lambda_j
  \;\le\; 2\,\FAi(x_j),
  \qquad \text{uniform in all transition-window indices.}
\end{equation}

\section{ORX Uniformity in the Transition Window}
\label{sec:propU}
% ============================================================

\begin{proposition}[ORX Uniformity]
\label{prop:U}
There exist $C_1, c_0 > 0$ such that for all $c \ge c_0$
and all $l$ with $|l - \NSh| \le A_0 c^{1/3}$:
\[
  \abs{\lambda_l - \FAi(x_l)} \le C_1\, c^{-2/3},
\]
uniform in $l$ throughout the transition window.
\end{proposition}

\begin{remark}[Status and context of Proposition~\ref{prop:U}]
\label{rem:propU_status}
ORX \cite{OsipovRokhlinXiao} establishes pointwise
convergence $\lambda_l \to \FAi(x_l)$ for fixed
rescaled $x$. Proposition~\ref{prop:U} requires a
uniform $O(c^{-2/3})$ error over $O(c^{1/3})$ indices
simultaneously — this is the analytic content missing
from ORX and is the genuine hard core of the whole
argument.
All results below are conditional on it.
\end{remark}

\begin{remark}[Route to a proof of Proposition~\ref{prop:U}]
\label{rem:propU_route}
A proof is expected from three steps within the ORX
framework:
\begin{enumerate}[(i)]
\item \textbf{Uniform Liouville--Green (WKB) control:}
  the WKB remainder is $O(c^{-2/3})$, uniform over
  the transition window.
\item \textbf{Airy matching with explicit remainder:}
  the connection formula between the WKB solution and
  the Airy function holds with a $O(c^{-2/3})$ remainder,
  uniform in $x_l \in [-A_0, A_0]$.
\item \textbf{Stability of the inverse spectral map:}
  $O(c^{-2/3})$ perturbations of $\FAi(x_l)$
  propagate to $O(c^{-2/3})$ perturbations of $\lambda_l$
  uniformly in $l$.
\end{enumerate}
Each step is expected to require careful bookkeeping
of the uniformity constants already implicit in the
ORX analysis.
\end{remark}

\section{Lemma A: Uniform Airy Increment}
\label{sec:lemmaA}
% ============================================================

\begin{lemma}[Uniform Airy Increment]
\label{lem:A}
Under Proposition~\ref{prop:U}, for all $c \ge c_0$ and all
$l$ with $|l - \NSh| \le A_0 c^{1/3}$:
\[
  \lambda_l - \lambda_{l+1}
  = (c/2)^{-1/3}\,\mathrm{Ai}(x_l)^2 + R_l,
\]
where $|R_l| \le C\,c^{-2/3}$ with
$C = 2C_1 + L_{A_0}$, uniform in $l$.
In particular,
$|\lambda_l - \lambda_{l+1}| \le C_A\,(c/2)^{-1/3}$
with $C_A := C_{\max} + C$.
\end{lemma}

\begin{proof}
From Proposition~\ref{prop:U}:
\[
  \lambda_l - \lambda_{l+1}
  = [\FAi(x_l) - \FAi(x_{l+1})] + (E_l - E_{l+1}),
\]
$|E_l - E_{l+1}| \le 2C_1 c^{-2/3}$.

\textbf{Main term.}
Since $\FAi' = -\mathrm{Ai}^2$ and
$x_{l+1} - x_l = (c/2)^{-1/3}$, the MVT gives
$\xi_l \in [x_l, x_{l+1}]$ with
$\FAi(x_l) - \FAi(x_{l+1})
= \mathrm{Ai}(\xi_l)^2 (c/2)^{-1/3}$.

\textbf{Replacing $\xi_l$ by $x_l$.}
Since $\mathrm{Ai}^2 \in C^1([-A_0,A_0])$
with Lipschitz constant $L_{A_0}$
(Section~\ref{sec:setup}),
and $|\xi_l - x_l| \le (c/2)^{-1/3}$:
\[
  |\mathrm{Ai}(\xi_l)^2 - \mathrm{Ai}(x_l)^2|
  \le L_{A_0}\,(c/2)^{-1/3},
\]
giving a contribution $L_{A_0}(c/2)^{-2/3}$ to $R_l$.

Combining: $|R_l| \le 2C_1 c^{-2/3} + L_{A_0}(c/2)^{-2/3}
\le (2C_1 + L_{A_0})c^{-2/3}$.
\end{proof}

\section{Lemma B: Airy Ratio Bound}
\label{sec:lemmaB}
% ============================================================

\begin{lemma}[Airy Ratio Bound]
\label{lem:B}
\[
  \frac{\mathrm{Ai}(x)^2}{\FAi(x)}
  \le \frac{C_{\max}}{c_*}
  \qquad \text{for all } x \in [-A_0, A_0].
\]
\end{lemma}

\begin{proof}
Numerator $\le C_{\max}$, denominator $\ge c_* > 0$
(Section~\ref{sec:setup}).
\end{proof}

\section{Lemma F: Airy Rigidity and Dyadic Separation}
\label{sec:lemmaF}
% ============================================================

\begin{lemma}[Airy Rigidity / Dyadic Separation Principle]
\label{lem:F}
$\FAi$ satisfies the bi-Lipschitz bound
\[
  c_{\min}\,|x_i - x_j|
  \le |\FAi(x_i) - \FAi(x_j)|
  \le C_{\max}\,|x_i - x_j|
  \quad \text{for all } x_i, x_j \in [-A_0, A_0].
\]
Under Proposition~\ref{prop:U}, with threshold
$c_1 := 2(4C_1/c_{\min})^3$,
for all $c \ge c_1$ and all $i \ne j$ with
$x_i, x_j \in [-A_0, A_0]$:
\begin{equation}\label{eq:specsep}
  |\lambda_i - \lambda_j|
  \ge \tfrac{1}{2}\,c_{\min}\,(c/2)^{-1/3}\,|i-j|.
\end{equation}
For $j \in \Bm(i)$:
$|\lambda_i - \lambda_j|
\ge \tfrac{1}{2}\,c_{\min}\,(c/2)^{-1/3}\cdot 2^m$.
\end{lemma}

\begin{proof}
\textbf{Bi-Lipschitz.}
$|\FAi(x_i)-\FAi(x_j)| = |\int_{x_j}^{x_i}\mathrm{Ai}(t)^2\,dt|$;
the integrand satisfies $c_{\min} \le \mathrm{Ai}(t)^2
\le C_{\max}$ on $[-A_0,A_0]$, giving both bounds.
(The strict positivity $c_{\min} > 0$ is the key
rigidity mechanism: it holds because $A_0 < |a_1|$,
so $[-A_0,A_0]$ contains no zero of $\mathrm{Ai}$,
and the lower bi-Lipschitz bound is what drives the
Dyadic Separation Principle.)

\textbf{Spectral separation.}
$|\lambda_i-\lambda_j|
\ge c_{\min}(c/2)^{-1/3}|i-j| - 2C_1 c^{-2/3}$.
For $c \ge c_1 = 2(4C_1/c_{\min})^3$:
$2C_1 c^{-2/3} \le \frac{1}{2}c_{\min}(c/2)^{-1/3}$,
giving~\eqref{eq:specsep}.
\end{proof}

\begin{remark}[Microscopic scale and dyadic truncation]
\label{rem:microscopic}
At $|i-j| = O(1)$, both
$c_{\min}(c/2)^{-1/3}|i-j|$ and $2C_1 c^{-2/3}$
are $O(c^{-1/3})$, so the absorption fails
and~\eqref{eq:specsep} degenerates.
This is why $m \ge m_0$ (i.e.\ $|i-j| \ge 2^{m_0}$
with $m_0$ fixed, independent of $c$) is necessary:
the separation $2^{m_0}$ is a fixed positive integer,
guaranteeing absorption for all $c \ge c_1$.
The Core regime $m < m_0$ is handled by H3.
\end{remark}

\section{Corollary C: H2 in Truncated Dyadic Sense}
\label{sec:H2}
% ============================================================

\begin{corollary}[H2 in truncated dyadic sense]
\label{cor:C}
Fix $m_0 \ge 1$. Let $c_2 := \max(c_0,c_1)$.
Under Proposition~\ref{prop:U}, Lemmas~\ref{lem:B},
\ref{lem:F}, define
\[
  \delta_m
  := \frac{2C_A}{c_{\min}}\cdot 2^{-m}
   + 2\,\frac{C_{\max}}{c_*}\cdot(c/2)^{-1/3}.
\]
All indices $i,j$ below lie strictly inside the
transition window, away from the bulk plateau
(i.e.\ $x_j \in [-A_0,0]$ and $\FAi(x_j) \ge c_* > 0$)
so that~\eqref{eq:onemlam} applies uniformly.
For all $c \ge c_2$, all $i \le N$, $m \ge m_0$,
and $j, j+1 \in \Bm(i)$:
\[
  \frac{|A_{ij} - A_{i,j+1}|}{A_{ij}} \le \delta_m.
\]
For any $\varepsilon > 0$: choose $m_0$ so that
$\frac{2C_A}{c_{\min}} 2^{-m_0} < \varepsilon/2$,
then $c$ so that
$2\frac{C_{\max}}{c_*}(c/2)^{-1/3} < \varepsilon/2$;
then $\sup_{m \ge m_0}\delta_m < \varepsilon$.
This is H2 (in truncated dyadic sense) for
Theorem~2.1 in \cite{paperVII}.
\end{corollary}

\begin{proof}
\[
  \frac{|A_{ij} - A_{i,j+1}|}{A_{ij}}
  \lesssim
  \underbrace{\frac{|\lambda_j - \lambda_{j+1}|}
              {|\lambda_i - \lambda_j|}}_{\text{Term 1}}
  +
  \underbrace{\frac{|\lambda_j - \lambda_{j+1}|}
              {1 - \lambda_j}}_{\text{Term 2}}.
\]

\textbf{Term 1.}
Lemma~\ref{lem:A}: numerator $\le C_A(c/2)^{-1/3}$.
Lemma~\ref{lem:F}: denominator
$\ge \frac{1}{2}c_{\min}(c/2)^{-1/3}\cdot 2^m$.
The $(c/2)^{-1/3}$ factors cancel:
Term~1 $\le \frac{2C_A}{c_{\min}} 2^{-m} = O(2^{-m})$,
independent of $c$.

\textbf{Term 2.}
Lemma~\ref{lem:A} + Lemma~\ref{lem:B}
+ lower bound from~\eqref{eq:onemlam}:
\[
  \frac{|\lambda_j-\lambda_{j+1}|}{1-\lambda_j}
  \le
  \frac{(c/2)^{-1/3}\mathrm{Ai}(x_j)^2 + O(c^{-2/3})}
       {\tfrac{1}{2}\FAi(x_j)}
  \le 2\cdot\frac{C_{\max}}{c_*}\cdot(c/2)^{-1/3}
  = O(c^{-1/3}).
\]
(Step: $|\lambda_j-\lambda_{j+1}| \approx
(c/2)^{-1/3}\mathrm{Ai}(x_j)^2$ by Lemma~\ref{lem:A};
then $\mathrm{Ai}(x_j)^2/\FAi(x_j) \le C_{\max}/c_*$
by Lemma~\ref{lem:B}; denominator
$\ge c_*/2$ by~\eqref{eq:onemlam}.
Note: indices are strictly inside the transition window,
not at the bulk plateau, so $\FAi(x_j) \ge c_* > 0$.)

\textbf{Combining:} $\delta_m = \text{Term 1} + \text{Term 2}$.
\end{proof}

\begin{remark}[The Dyadic Separation Principle in action]
\label{rem:dyadic}
The cancellation of $(c/2)^{-1/3}$ in Term~1 is
the operative content of the Dyadic Separation Principle:
numerator and denominator share the same $c^{-1/3}$ scale,
but the denominator carries an extra $2^m$ from $|i-j|\ge 2^m$.
The result is a purely combinatorial $O(2^{-m})$,
independent of $c$.
The $O(c^{-1/3})$ in $\delta_m$ comes from Term~2 alone.
\end{remark}

% ============================================================
\section{Lemma E: Core--WKB Split}
\label{sec:split}
% ============================================================

\begin{lemma}[Core--WKB Split]
\label{lem:E}
Fix $m_0 \ge 1$.
Assume Proposition~\ref{prop:U},
Lemmas~\ref{lem:A}, \ref{lem:B}, \ref{lem:F},
and Assumption~B-strong \cite{paperVI}
(H3: $A_{ij} \le C_2$ uniformly).
Then $\sum_{j \ne i} |T_{ij}| \le C(m_0, A, C_2)$
uniformly in $i$, $N$, $c \ge c_2$.
\end{lemma}

\begin{proof}
Split at $m_0$:
\[
  \sum_{j\ne i}|T_{ij}|
  = \sum_{m < m_0}\sum_{j\in\Bm(i)}|T_{ij}|
  + \sum_{m \ge m_0}\sum_{j\in\Bm(i)}|T_{ij}|.
\]

\textbf{Core ($m < m_0$).}
The kernel satisfies $|T_{ij}| \le C_2/|i-j|$,
a direct consequence of B-strong ($A_{ij} \le C_2$)
and the off-diagonal structure of $T_{ij}$
(see \cite[Lemma~5.2]{paperVII}).
Each block $\Bm(i)$ has $\le 2^{m+1}$ terms with
$|i-j| \ge 2^m$, giving $\sum_{\Bm}|T_{ij}| \le 2C_2$;
summing over $m_0$ blocks yields $\le 2C_2 m_0$.

\textbf{WKB ($m \ge m_0$).}
Corollary~\ref{cor:C} gives H2 for all $m \ge m_0$.
Theorem~2.1 of \cite{paperVII} is applied to the
\emph{truncated operator} (blocks $m \ge m_0$ only),
using H1 (unconditional) and H3 (B-strong):
this gives $O(1)$ uniformly.
\end{proof}

\begin{remark}[Role of H3]
\label{rem:H3role}
H3 is not used in proving H2 (Corollary~\ref{cor:C}).
It enters in two places in Lemma~\ref{lem:E}:
(i) directly in the Core estimate
($|T_{ij}| \le C_2/|i-j|$),
and (ii) as input to Theorem~2.1 of \cite{paperVII}
in the WKB part.
In particular, a proof of H3 alone would not simplify
or strengthen the H2 argument.
\end{remark}

\section{Conditional Reduction: Hypothesis Verification}
\label{sec:closure}
% ============================================================

\begin{corollary}[Conditional Reduction]
\label{cor:closure}
Under Proposition~\ref{prop:U} and B-strong
\cite{paperVI}, hypotheses \textup{(H1)--(H3)} of
Theorem~2.1 in \cite{paperVII} are verified
(H2 in truncated dyadic sense).
Consequently:
\[
  \sup_{i\le N}\,\sum_{j\ne i}|P_{ij}|
  \le C\,c^{-1/2}\log c
  \qquad (\text{Assumption }B'\text{ of }\cite{paperVI}),
\]
and the contraction estimate
$\norm{D\mathcal{T}_c^{(N)}} \le C(A)\,c^{-1/2}\log c < 1$
holds for $c \ge c_2$, conditional on
Proposition~\ref{prop:U} and B-strong.
\end{corollary}

\begin{remark}[Logical dependency chain]
\[
  \text{Prop.~U}
  \Rightarrow \text{Lem.~A, F}
  \Rightarrow \text{Cor.~C (H2)}
  \xrightarrow{+\text{H3}} \text{Lem.~E}
  \Rightarrow B' \Rightarrow \text{Assumption~A.}
\]
H1: unconditional.
Lemma~B: used in Cor.~C Term~2.
H3: enters only in Lemma~E (not in H2).
All thresholds $c_0, c_1, c_2$ depend only on
$A_0, C_1, c_{\min}$.
\end{remark}
