\begin{abstract}
We close the open amplitude hypothesis~(H2) of
Paper~VII \cite{paperVII} under a single additional analytic
input: a uniform Airy approximation for PSWF eigenvalues in
the transition window (Proposition~\ref{prop:U},
stated as a conditional assumption pending full extraction
from ORX \cite{OsipovRokhlinXiao}).

The main contributions are:
\begin{itemize}
\item Lemma~\ref{lem:A} (Uniform Airy Increment): a
  discrete-to-continuous stability result for PSWF eigenvalue
  differences $\lambda_l - \lambda_{l+1}$ in the Airy
  transition regime.
\item Lemma~\ref{lem:B} (Airy Ratio Bound): uniform control
  of the ratio $\mathrm{Ai}(x)^2 / \FAi(x)$
  on the Airy-zero-free window $[-A_0, A_0]$ with
  $A_0 < |a_1|$, where $a_1 \approx -2.338$ is the
  largest (least negative) real zero of $\mathrm{Ai}$.
\item Lemma~\ref{lem:F} (Airy Rigidity): stability of
  spectral separations $|\lambda_i - \lambda_j|$ under the
  Airy approximation, with quantitative Lipschitz lower
  bound on $|\FAi(x_i) - \FAi(x_j)|$ from strict positivity
  of $\mathrm{Ai}(x)^2$ on the zero-free window.
\item Corollary~\ref{cor:C} (H2 for $m \ge m_0$): hypothesis
  (H2) of Theorem~2.1 in \cite{paperVII} holds for all
  dyadic blocks of scale $m \ge m_0$ (fixed), with
  $\delta_m = Cc^{-1/3} \to 0$ uniform in $c$.
\item Lemma~\ref{lem:E} (Core--WKB Split): the Core regime
  $m < m_0$ is controlled directly by Assumption~B-strong
  (H3), contributing $O(1)$ to row sums.
\end{itemize}
Together, these results show that hypotheses (H1)--(H3) of
Paper~VII are fully verified conditional on
Assumption~B-strong and Proposition~\ref{prop:U}.
\end{abstract}

% ============================================================
\section{Introduction and Context}
\label{sec:intro}
% ============================================================

Paper~VII \cite{paperVII} established an abstract dyadic
cancellation theorem (Theorem~2.1) and reduced the
contraction problem for the PSWF Galerkin operator
$\mathcal{T}_c^{(N)}$ to the verification of three
hypotheses:
\begin{itemize}
\item[(H1)] Non-degenerate oscillatory phase: verified
  unconditionally from Widom asymptotics.
\item[(H2)] Blockwise amplitude regularity: the relative
  variation of the amplitude $A_{ij}$ within each dyadic
  block $\mathcal{B}_m(i)$ is $o(1)$ uniformly in $c$.
\item[(H3)] Uniform amplitude bound: $A_{ij} \le C$ for
  all $i \ne j$, equivalent to Assumption~B-strong
  (\cite{paperVI}).
\end{itemize}

H2 was stated in \cite{paperVII} as Conjecture~6.2.
The present paper closes this conjecture under
Proposition~\ref{prop:U}, which is a uniformization
of existing ORX asymptotics.

\medskip\noindent
\textbf{Central observation.}
H2 does not hold uniformly at all dyadic scales.
For the smallest blocks ($m = 0$, i.e., $|i-j| = 1$),
the relative amplitude variation can be $O(1)$, not $o(1)$.
The correct statement is a \emph{two-regime} one:
\[
  \text{H2 holds for } m \ge m_0 \text{ (WKB regime)};
  \qquad
  m < m_0 \text{ controlled by H3 (Core regime).}
\]

\medskip\noindent
\textbf{Airy zero geometry.}
All real zeros of $\mathrm{Ai}$ are strictly negative.
We denote by $a_1 \approx -2.338$ the largest (least
negative) real zero, and by $a_k$ the $k$-th zero in
decreasing order ($a_1 > a_2 > \cdots$).
In particular:
\[
  \mathrm{Ai}(x) > 0 \quad \text{for all } x > a_1,
\]
which includes all of $[a_1, +\infty)$ and in particular
all of $[0, +\infty)$.
For a window $[-A_0, A_0]$ with $A_0 < |a_1| \approx 2.338$,
we have $[-A_0, A_0] \subset (a_1, +\infty)$,
so $\mathrm{Ai}$ is strictly positive and bounded away
from zero on this interval.

\medskip\noindent
\textbf{Restriction on the phase window.}
Throughout, we fix $A_0 < |a_1| \approx 2.338$.
The Galerkin regime $l \le N = \lfloor Ac^{1/3}\rfloor$
corresponds to rescaled indices $x_l \in [-A, 0]$,
which lies within $(a_1, 0] \subset (a_1, +\infty)$
provided $A < |a_1|$.
All positivity and Lipschitz estimates below are restricted
to this zero-free window.

\medskip\noindent
\textbf{What is not proved here.}
Proposition~\ref{prop:U} (ORX Uniformity) is stated as
an assumption. B-strong (H3) remains open.

\medskip\noindent
\textbf{Structure.}
Section~\ref{sec:setup} fixes notation.
Section~\ref{sec:propU} states Proposition~U.
Sections~\ref{sec:lemmaA}--\ref{sec:lemmaF} prove the
four supporting lemmas.
Section~\ref{sec:H2} proves Corollary~C.
Section~\ref{sec:split} proves Lemma~E.
Section~\ref{sec:closure} collects the closure.

\section{Setup and Notation}
\label{sec:setup}
% ============================================================

Throughout, $c \ge 1$ is the bandwidth parameter,
$N = \lfloor A c^{1/3}\rfloor$ is the Galerkin truncation
with $A < |a_1|$, and $\lambda_k = \lambda_k^{(c)} \in (0,1)$
are the PSWF eigenvalues, ordered decreasingly.

\medskip\noindent
\textbf{Rescaled index.}
We define $x_l$ as a \emph{deterministic linear rescaling}:
\[
  x_l := (l - \NSh)\,(c/2)^{-1/3},
  \qquad \NSh := \frac{2c}{\pi}.
\]
This is a definition, not an asymptotic formula.
The spacing $x_{l+1} - x_l = (c/2)^{-1/3}$ is exactly
constant in $l$.

For $l \le N_\mathrm{Sh}$ (bulk modes below the Shannon
number), $x_l \le 0$; for $l = N$, $x_l = -A(1 + o(1))$.
All indices in the Galerkin range therefore satisfy
$x_l \in [-A, 0] \subset (a_1, 0]$,
remaining in the zero-free region of $\mathrm{Ai}$.

Further definitions:
\begin{align*}
  \FAi(x) &:= \int_x^\infty \mathrm{Ai}(t)^2\,dt
           \qquad \text{(complementary Airy CDF)},\\
  A_{ij} &:= P_{ij} \cdot c^{1/2},
  \qquad
  P_{ij} := c \cdot
  \frac{\abs{\lambda_i - \lambda_j}}
       {(1-\lambda_i)(1-\lambda_j)}
  \qquad (i\ne j).
\end{align*}

Dyadic blocks: for fixed $i$ and $m = 0,\ldots,M$ with
$M = \lfloor \log_2 N\rfloor$,
\[
  \Bm(i) := \{j : 2^m \le |i-j| < 2^{m+1}\}.
\]

\section{ORX Uniformity in the Transition Window}
\label{sec:propU}
% ============================================================

\begin{proposition}[ORX Uniformity]
\label{prop:U}
There exist constants $A_0 < |a_1|$, $C_1, c_0 > 0$
such that for all $c \ge c_0$ and all $l$ with
$|l - \NSh| \le A_0 c^{1/3}$:
\[
  \abs{\lambda_l - \FAi(x_l)} \le C_1\, c^{-2/3},
\]
uniform in $l$ throughout the transition window.
\end{proposition}

\begin{remark}[Status of Proposition~\ref{prop:U}]
\label{rem:propU_status}
This is a \emph{uniform} version of the pointwise Airy
approximation in \cite[Theorem~4.6]{OsipovRokhlinXiao}.
ORX establishes pointwise convergence $\lambda_l \to \FAi(x_l)$
for fixed rescaled $x$; Proposition~\ref{prop:U} requires
this with a uniform $O(c^{-2/3})$ error over
$O(c^{1/3})$ indices simultaneously.
This is expected from the WKB matching structure of the ORX
proof (turning point $s_* = 1-c^{-2/3}$, uniform matching
error across the transition zone), but the uniform bound
is not explicitly stated in ORX. Proving it requires
tracking uniformity constants in ORX Sections~4.4--4.7;
no Riemann--Hilbert methods are needed.
\end{remark}

\section{Lemma A: Uniform Airy Increment}
\label{sec:lemmaA}
% ============================================================

\begin{lemma}[Uniform Airy Increment]
\label{lem:A}
Under Proposition~\ref{prop:U}, for all $c \ge c_0$ and all
$l$ with $|l - \NSh| \le A_0 c^{1/3}$:
\[
  \lambda_l - \lambda_{l+1}
  = (c/2)^{-1/3}\,\mathrm{Ai}(x_l)^2 + R_l,
\]
where $|R_l| \le C\,c^{-2/3}$, uniform in $l$.
\end{lemma}

\begin{proof}
From Proposition~\ref{prop:U}:
\[
  \lambda_l - \lambda_{l+1}
  = \bigl[\FAi(x_l) - \FAi(x_{l+1})\bigr]
    + (E_l - E_{l+1}),
\]
where $|E_k| \le C_1 c^{-2/3}$, so
$|E_l - E_{l+1}| \le 2C_1 c^{-2/3}$.

\textbf{Main term.}
Since $\FAi' = -\mathrm{Ai}^2$ and
$x_{l+1} - x_l = (c/2)^{-1/3}$ (exactly, by definition),
the mean value theorem gives $\xi_l \in [x_l, x_{l+1}]$:
\[
  \FAi(x_l) - \FAi(x_{l+1})
  = \mathrm{Ai}(\xi_l)^2 \cdot (c/2)^{-1/3}.
\]

\textbf{Replacing $\xi_l$ by $x_l$.}
On $[-A_0, A_0]$ with $A_0 < |a_1|$, the function
$\mathrm{Ai}(x)^2$ is $C^1$ with Lipschitz constant
$L := \sup_{|x|\le A_0} 2|\mathrm{Ai}(x)||\mathrm{Ai}'(x)|
< \infty$.
Since $|\xi_l - x_l| \le (c/2)^{-1/3}$:
\[
  \abs{\mathrm{Ai}(\xi_l)^2 - \mathrm{Ai}(x_l)^2}
  \le L\,(c/2)^{-1/3},
\]
contributing an error $L(c/2)^{-2/3} = O(c^{-2/3})$.

Combining: $|R_l| \le (2C_1 + L)c^{-2/3}$.
\end{proof}

\begin{remark}
The spacing $x_{l+1} - x_l = (c/2)^{-1/3}$ is
\emph{exactly} constant by definition of $x_l$,
avoiding any secondary drift correction.
\end{remark}

\section{Lemma B: Airy Ratio Bound}
\label{sec:lemmaB}
% ============================================================

\begin{lemma}[Airy Ratio Bound]
\label{lem:B}
Let $a_1 \approx -2.338$ be the largest real zero of
$\mathrm{Ai}$. For each $A_0$ with $0 < A_0 < |a_1|$
there exists $C_{A_0} < \infty$ such that
\[
  \frac{\mathrm{Ai}(x)^2}{\FAi(x)}
  \le C_{A_0}
  \qquad \text{for all } x \in [-A_0, A_0].
\]
\end{lemma}

\begin{proof}
Since $A_0 < |a_1|$, the interval $[-A_0, A_0]$ contains
no real zero of $\mathrm{Ai}$ (all zeros satisfy
$x \le a_1 < -A_0$). Hence both $\mathrm{Ai}(x)^2$ and
$\FAi(x) = \int_x^\infty \mathrm{Ai}(t)^2\,dt$ are
continuous and strictly positive on $[-A_0, A_0]$.
By compactness, the ratio is bounded above.
\end{proof}

\begin{remark}
For $x > 0$: $\mathrm{Ai}(x) > 0$ holds trivially since
all zeros are negative. The binding constraint is on the
negative side: $x \ge -A_0$ with $A_0 < |a_1|$.
\end{remark}

\section{Lemma F: Airy Rigidity}
\label{sec:lemmaF}
% ============================================================

\begin{lemma}[Airy Rigidity]
\label{lem:F}
Let $a_1 \approx -2.338$ be the largest real zero of
$\mathrm{Ai}$, and fix $A_0$ with $0 < A_0 < |a_1|$.
Then:
\[
  c_{\min}(A_0)
  := \min_{x \in [-A_0,\, A_0]} \mathrm{Ai}(x)^2 > 0,
\]
since $[-A_0, A_0] \subset (a_1, +\infty)$
contains no zero of $\mathrm{Ai}$.
In particular, $\FAi$ satisfies the Lipschitz lower bound:
\[
  |\FAi(x_i) - \FAi(x_j)|
  = \left|\int_{x_j}^{x_i} \mathrm{Ai}(t)^2\,dt\right|
  \ge c_{\min}(A_0)\,|x_i - x_j|
  \qquad \text{for } x_i, x_j \in [-A_0, A_0].
\]
Under Proposition~\ref{prop:U}, for all $i,j$ with
$x_i, x_j \in [-A_0, A_0]$ and $|i-j| \ge 2^{m_0}$
(fixed $m_0$), for $c$ sufficiently large:
\[
  |\lambda_i - \lambda_j|
  \ge \tfrac{1}{2}\,c_{\min}(A_0)\,(c/2)^{-1/3}\,|i-j|.
\]
\end{lemma}

\begin{proof}
\textbf{Strict positivity.}
Since $A_0 < |a_1|$, the interval $[-A_0, A_0]$ is
contained in $(a_1, +\infty)$, where $\mathrm{Ai}$ has
no zeros. By continuity and compactness,
$c_{\min}(A_0) = \min_{[-A_0,A_0]}\mathrm{Ai}(x)^2 > 0$.
Note: for $x \ge 0$, $\mathrm{Ai}(x) > 0$ is automatic
(no positive zeros exist); the constraint $A_0 < |a_1|$
is needed only to exclude negative zeros.

\textbf{Lipschitz lower bound.}
By the fundamental theorem of calculus and positivity:
\[
  |\FAi(x_i) - \FAi(x_j)|
  \ge c_{\min}(A_0)\,|x_i - x_j|
  = c_{\min}(A_0)\,(c/2)^{-1/3}\,|i-j|.
\]

\textbf{Spectral separation.}
From Proposition~\ref{prop:U}, $\lambda_k = \FAi(x_k)+E_k$
with $|E_k| \le C_1 c^{-2/3}$, so:
\[
  |\lambda_i - \lambda_j|
  \ge |\FAi(x_i)-\FAi(x_j)| - 2C_1 c^{-2/3}
  \ge c_{\min}(A_0)(c/2)^{-1/3}|i-j| - 2C_1 c^{-2/3}.
\]
For $|i-j| \ge 2^{m_0}$ (fixed) and $c$ large enough,
the first term dominates, giving the stated lower bound.
\end{proof}

\begin{remark}[Role in Corollary~\ref{cor:C}]
Lemma~\ref{lem:F} closes the logical gap in
Corollary~\ref{cor:C}: Term~1 of the relative variation
($|\lambda_j - \lambda_{j+1}|/|\lambda_i - \lambda_j|$)
is bounded by $O((c/2)^{-1/3}) / O((c/2)^{-1/3} \cdot 2^{m_0})
= O(2^{-m_0}) = O(1)$, which combined with
the $O(c^{-1/3})$ bound from Lemma~\ref{lem:B} on Term~2
yields the total relative variation $O(c^{-1/3})$.
\end{remark}

\section{Corollary C: H2 for Large Dyadic Scales}
\label{sec:H2}
% ============================================================

\begin{corollary}[H2 for $m \ge m_0$]
\label{cor:C}
Fix $m_0 \ge 1$ (independent of $c$). Under
Proposition~\ref{prop:U}, Lemma~\ref{lem:B},
and Lemma~\ref{lem:F},
for all $c \ge c_0$, all $i \le N$, all $m \ge m_0$,
and all $j, j+1 \in \Bm(i)$:
\[
  \frac{|A_{ij} - A_{i,j+1}|}{A_{ij}}
  \le C\,c^{-1/3},
\]
uniform in $i, j, m, c$.
In particular, hypothesis~\textup{(H2)} of
Theorem~2.1 in \cite{paperVII} holds for all
$m \ge m_0$ with $\delta_m = Cc^{-1/3} \to 0$.
\end{corollary}

\begin{proof}
At fixed $i$, the variation of
$A_{ij} \propto |\lambda_i-\lambda_j|/
((1-\lambda_i)(1-\lambda_j))$ in $j$ yields:
\[
  \frac{|A_{ij} - A_{i,j+1}|}{A_{ij}}
  \lesssim
  \underbrace{\frac{|\lambda_j - \lambda_{j+1}|}{|\lambda_i - \lambda_j|}}_{\text{Term 1}}
  +
  \underbrace{\frac{|\lambda_j - \lambda_{j+1}|}{1 - \lambda_j}}_{\text{Term 2}}.
\]

\textbf{Term 2.}
By Lemma~\ref{lem:A}, $|\lambda_j-\lambda_{j+1}| \le C(c/2)^{-1/3}$.
By Proposition~\ref{prop:U}, $1-\lambda_j \approx \FAi(x_j)$.
Lemma~\ref{lem:B} then gives:
\[
  \frac{|\lambda_j - \lambda_{j+1}|}{1 - \lambda_j}
  \lesssim \frac{(c/2)^{-1/3}\mathrm{Ai}(x_j)^2}{\FAi(x_j)}
  \le C_{A_0}(c/2)^{-1/3} = O(c^{-1/3}).
\]

\textbf{Term 1.}
For $m \ge m_0$, $|i-j| \ge 2^{m_0}$.
By Lemma~\ref{lem:F}:
$|\lambda_i-\lambda_j| \ge \frac{1}{2}c_{\min}(A_0)
(c/2)^{-1/3} \cdot 2^{m_0}$.
Hence:
\[
  \frac{|\lambda_j - \lambda_{j+1}|}{|\lambda_i - \lambda_j|}
  \le \frac{C(c/2)^{-1/3}}
           {\frac{1}{2}c_{\min}(A_0)(c/2)^{-1/3} \cdot 2^{m_0}}
  = \frac{2C}{c_{\min}(A_0) \cdot 2^{m_0}} = O(1).
\]

\textbf{Conclusion.}
Term~1 = $O(1)$, Term~2 = $O(c^{-1/3})$;
total relative variation is $O(c^{-1/3}) \to 0$.
\end{proof}

\begin{remark}[Why H2 fails for $m = 0$]
\label{rem:H2_m0}
For $|i-j| = 1$: Lemma~\ref{lem:F} gives
$|\lambda_i - \lambda_j| \sim (c/2)^{-1/3}$,
of the same order as $|\lambda_j-\lambda_{j+1}|$,
so Term~1 = $O(1)$.
H2 therefore fails ($\delta_0 \not\to 0$) at microscopic
scale; this requires H3/B-strong control.
\end{remark}

\section{Lemma E: Core--WKB Split}
\label{sec:split}
% ============================================================

\begin{lemma}[Core--WKB Split]
\label{lem:E}
Fix $m_0 \ge 1$ independent of $c$.
Assume Proposition~\ref{prop:U}, Lemmas~\ref{lem:A},
\ref{lem:B}, \ref{lem:F}, and Assumption~B-strong
\cite{paperVI} (H3: $A_{ij} \le C_2$ for all
$i \ne j$, $i,j \le N$). Then for all $i \le N$:
\[
  \sum_{j \ne i} |T_{ij}| \le C(m_0, A, C_2)
\]
uniform in $i$, $N$, and $c \ge c_0$.
\end{lemma}

\begin{proof}
Split at scale $m_0$:
\[
  \sum_{j \ne i} |T_{ij}|
  = \underbrace{\sum_{m=0}^{m_0-1} \sum_{j\in\Bm(i)}
    |T_{ij}|}_{\text{Core}}
  + \underbrace{\sum_{m=m_0}^{M} \sum_{j\in\Bm(i)}
    |T_{ij}|}_{\text{WKB}}.
\]

\textbf{Core ($m < m_0$).}
By H3: $|T_{ij}| \le C_2/|i-j|$.
Each $\Bm(i)$ has $\le 2^{m+1}$ elements with
$|i-j| \ge 2^m$, giving $\sum_{j\in\Bm}|T_{ij}| \le 2C_2$.
Summing over $m_0$ blocks: $\le 2C_2 m_0 = O(1)$.

\textbf{WKB ($m \ge m_0$).}
Corollary~\ref{cor:C} gives H2 for all $m \ge m_0$.
With H1 (unconditional) and H3 (B-strong), all hypotheses
of Theorem~2.1 in \cite{paperVII} hold for $m \ge m_0$,
giving $\sum_{m \ge m_0}\sum_{j\in\Bm}|T_{ij}| = O(1)$.

\textbf{Total:} $O(1) + O(1) = O(1)$.
\end{proof}

\section{Closure: Full Hypothesis Verification}
\label{sec:closure}
% ============================================================

\begin{corollary}[Full Hypothesis Closure]
\label{cor:closure}
Under Proposition~\ref{prop:U} and
Assumption~B-strong \cite{paperVI},
all three hypotheses \textup{(H1)--(H3)} of
Theorem~2.1 in \cite{paperVII} are verified.
Consequently:
\[
  \sup_{i\le N}\,\sum_{j\ne i}|P_{ij}|
  \le C\,c^{-1/2}\log c,
\]
which is Assumption~$B'$ of \cite{paperVI}.
The contraction estimate
$\norm{D\mathcal{T}_c^{(N)}} \le C(A)\,c^{-1/2}\log c < 1$
for $c \ge c_0$ follows, establishing Assumption~A
conditional on Proposition~\ref{prop:U} and B-strong.
\end{corollary}

\begin{remark}[Logical dependency chain]
The complete implication structure is:
\[
  \text{Prop.~U}
  \;\Rightarrow\; \text{Lem.~A, F}
  \;\Rightarrow\; \text{Cor.~C (H2, } m \ge m_0\text{)}
  \;\xrightarrow{+\,\text{H3}}\; \text{Lem.~E}
  \;\Rightarrow\; B' \;\Rightarrow\; \text{Assumption~A}.
\]
H1 enters unconditionally via \cite{paperVII}.
Lemma~B is used in the proof of Cor.~C (Term~2 control).
\end{remark}

\begin{remark}[What remains open]
\label{rem:open}
Two open problems remain:
\begin{itemize}
\item \textbf{Proposition~\ref{prop:U}:}
  Uniform Airy approximation over the transition window.
  Expected from ORX; requires explicit constant tracking
  in ORX Sections~4.4--4.7.
\item \textbf{Assumption~B-strong:}
  $P_{kl} \le C_2 c^{1/2}$ in the transition zone;
  the deepest remaining open problem in the program.
\end{itemize}
All other hypotheses are closed.
\end{remark}
