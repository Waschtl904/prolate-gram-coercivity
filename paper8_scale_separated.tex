\title[Scale-Separated Dyadic Cancellation]
      {Scale-Separated Dyadic Cancellation\\
       for PSWF Galerkin Operators:\\
       Reduction of H2 to Uniform Airy Asymptotics}

\author{[Author]}
\date{\today}
\maketitle

% ------------------------------------------------------------
\begin{abstract}
We reduce the open amplitude hypothesis~(H2) of
Paper~VII \cite{paperVII} to two uniform analytic inputs
about PSWF eigenvalues in the transition window
(Proposition~\ref{prop:U}): a pointwise $O(c^{-2/3})$
Airy approximation and a discrete Lipschitz control on
consecutive eigenvalue differences.
All results are conditional on Proposition~\ref{prop:U}
and on Assumption~B-strong \cite{paperVI}.

The main contributions are:
\begin{itemize}
\item Lemma~\ref{lem:A} (Uniform Airy Increment):
  under Proposition~\ref{prop:U}~(U'),
  $\lambda_l - \lambda_{l+1}
  = (c/2)^{-1/3}\mathrm{Ai}(x_l)^2 + O(c^{-2/3})$
  uniformly in the transition window.
\item Lemma~\ref{lem:B} (Airy Ratio Bound):
  uniform control of $\mathrm{Ai}(x)^2/\FAi(x)$.
\item Lemma~\ref{lem:F} (Dyadic Separation Principle):
  $|\lambda_i - \lambda_j| \ge
  \tfrac{1}{2}c_{\min}(c/2)^{-1/3}|i-j|$
  for $c \ge c_1$.
\item Corollary~\ref{cor:C} (H2 in truncated dyadic sense):
  $\delta_m = O(2^{-m}) + O(c^{-1/3})$.
\item Lemma~\ref{lem:E} (Core--WKB Split).
\end{itemize}
The central mechanism is the
\emph{Dyadic Separation Principle}: for
$j \in \mathcal{B}_m(i)$,
$|\lambda_j-\lambda_{j+1}|/|\lambda_i-\lambda_j|
\sim 2^{-m}$, which is not contained in ORX
\cite{OsipovRokhlinXiao}.
\end{abstract}

% ============================================================
\section{Introduction and Context}
\label{sec:intro}
% ============================================================

Paper~VII \cite{paperVII} established an abstract dyadic
cancellation theorem (Theorem~2.1) and reduced the
contraction problem to three hypotheses (H1)--(H3).
H2 was stated as Conjecture~6.2 and is reduced here
to Proposition~\ref{prop:U}, a uniform Airy approximation
not explicitly provided by ORX \cite{OsipovRokhlinXiao}.
This paper is a \emph{conditional reduction theorem}:
under Proposition~\ref{prop:U} and B-strong,
hypotheses (H1)--(H3) are verified and the contraction
estimate follows.

\medskip\noindent
\textbf{The Dyadic Separation Principle.}
$|\lambda_j-\lambda_{j+1}|/|\lambda_i-\lambda_j|
\sim 2^{-m}$ for $j \in \mathcal{B}_m(i)$:
both scales are $c^{-1/3}$, but the denominator carries
an extra $2^m$ from $|i-j| \ge 2^m$, so the
$c$-dependence cancels.

\medskip\noindent
\textbf{H2 in truncated dyadic sense.}
H2 fails at $m=0$ (variation is $O(1)$).
For $m \ge m_0$ (WKB): H2 holds via Corollary~\ref{cor:C}.
For $m < m_0$ (Core): bounded by H3.
Theorem~2.1 of \cite{paperVII} is applied to the
truncated operator (blocks $m \ge m_0$ only).

\medskip\noindent
\textbf{What is not proved here.}
Proposition~\ref{prop:U}~(U') is stated as an assumption;
see Remark~\ref{rem:propU_route} for a proof strategy.
B-strong remains open; see Remark~\ref{rem:open}.

\medskip\noindent
\textbf{Structure.}
Sec.~\ref{sec:setup}: notation.
Sec.~\ref{sec:propU}: Proposition~U.
Secs.~\ref{sec:lemmaA}--\ref{sec:lemmaF}: lemmas.
Sec.~\ref{sec:H2}: Corollary~C.
Sec.~\ref{sec:split}: Lemma~E.
Sec.~\ref{sec:closure}: reduction.

\section{Setup and Notation}
\label{sec:setup}
% ============================================================

Throughout, $c \ge 1$; $\lambda_k = \lambda_k^{(c)} \in (0,1)$
ordered decreasingly.

\medskip\noindent
\textbf{Airy zeros and $A_0$.}
All zeros of $\mathrm{Ai}$ are real and negative;
$a_1 \approx -2.3381$ is the largest
\cite[Sec.~9.9]{DLMF}. We fix:
\[
  \boxed{A_0 \in (0,|a_1|) \approx (0,2.3381).}
\]
$[-A_0,A_0]$ contains no zero of $\mathrm{Ai}$.

\medskip\noindent
\textbf{Compactness constants (depend only on $A_0$):}
\begin{align*}
  c_* &:= \min_{x\in[-A_0,0]}\FAi(x) > 0,\\
  c_{\min} &:= \min_{x\in[-A_0,A_0]}\mathrm{Ai}(x)^2 > 0,
  \quad
  C_{\max} := \max_{x\in[-A_0,A_0]}\mathrm{Ai}(x)^2 < \infty.
\end{align*}

\medskip\noindent
\textbf{Lipschitz constant of $\mathrm{Ai}^2$:}
\[
  L_{A_0} :=
  \sup_{x \in [-A_0,A_0]}\abs{\tfrac{d}{dx}\mathrm{Ai}(x)^2}
  = 2\sup_{[-A_0,A_0]}|\mathrm{Ai}\,\mathrm{Ai}'| < \infty.
\]

\medskip\noindent
\textbf{Monotonicity fact.}
$\FAi(x) = \int_x^\infty\mathrm{Ai}(t)^2\,dt$ is strictly
decreasing with $\FAi'(x) = -\mathrm{Ai}(x)^2 \le -c_{\min}<0$
on $[-A_0,A_0]$.
In particular, $\FAi(x) \ge \FAi(0) > 0$ for all
$x \le 0$, so $c_* = \min_{[-A_0,0]}\FAi \ge \FAi(0) > 0$.
This is why~\eqref{eq:onemlam} holds uniformly for all
transition-window indices with $x_j \le 0$.

\medskip\noindent
\textbf{Uniformity.}
All constants depend only on $A_0$; none depend on
$c$, $i$, $j$, $m$.

\medskip\noindent
\textbf{Rescaled index:}
$x_l := (l-\NSh)(c/2)^{-1/3}$, $\NSh := 2c/\pi$.
Spacing $x_{l+1}-x_l = (c/2)^{-1/3}$ exactly.
All Galerkin indices satisfy $x_l \in [-A,0]
\subset [-A_0,0] \subset [-A_0,A_0]$.

\medskip\noindent
\textbf{Further definitions:}
\begin{align*}
  A_{ij} &:= P_{ij}\cdot c^{1/2},\\
  P_{ij} &:= c\cdot
  \frac{|\lambda_i-\lambda_j|}{(1-\lambda_i)(1-\lambda_j)}
  \quad(i\ne j).
\end{align*}
$\Bm(i) := \{j : 2^m \le |i-j| < 2^{m+1}\}$.

\medskip\noindent
\textbf{Explicit bound on $1-\lambda_j$.}
Under (U) of Proposition~\ref{prop:U},
$\lambda_j = \FAi(x_j)+E_j$, $|E_j| \le C_1 c^{-2/3}$.
For $c \ge c_0$ with $C_1 c_0^{-2/3} \le c_*/2$:
\begin{equation}\label{eq:onemlam}
  \tfrac{1}{2}\FAi(x_j) \le 1-\lambda_j \le 2\FAi(x_j)
\end{equation}
uniformly for all transition-window indices
(using $\FAi(x_j) \ge c_* > 0$, the monotonicity fact).

\section{ORX Uniformity in the Transition Window}
\label{sec:propU}
% ============================================================

\begin{proposition}[ORX Uniformity]
\label{prop:U}
There exist $C_1, C_2, c_0 > 0$ (depending only on $A_0$)
such that for all $c \ge c_0$ and all $l$ with
$|l - \NSh| \le A_0 c^{1/3}$, the following two
estimates hold:
\begin{itemize}
\item[\textup{(U)}] \textbf{Pointwise:}
  $\abs{\lambda_l - \FAi(x_l)} \le C_1\, c^{-2/3}$.
\item[\textup{(U')}] \textbf{Discrete increment:}
  $\abs{(\lambda_l - \lambda_{l+1})
  - (\FAi(x_l) - \FAi(x_{l+1}))}
  \le C_2\, c^{-2/3}$.
\end{itemize}
Both bounds are uniform in $l$ throughout the window.
\end{proposition}

\begin{remark}[Why (U') is the correct condition]
\label{rem:propU_uprime}
Lemma~\ref{lem:A} requires controlling
$E_l - E_{l+1} := (\lambda_l - \FAi(x_l))
- (\lambda_{l+1} - \FAi(x_{l+1}))$.
Condition~(U) alone gives only $|E_l|, |E_{l+1}| \le C_1 c^{-2/3}$,
but does \emph{not} prevent the errors from oscillating
at scale $c^{-2/3}$, in which case $|E_l - E_{l+1}|$
could be $O(c^{-2/3})$ but not necessarily with
the right constant for a useful bound.
Condition~(U') directly asserts
$|E_l - E_{l+1}| \le C_2 c^{-2/3}$ as a separate,
explicit requirement.
This is a discrete Lipschitz (or equicontinuity)
condition on the error sequence $\{E_l\}$ and is the
minimal additional hypothesis needed for Lemma~\ref{lem:A}.
\end{remark}

\begin{remark}[Status and context]
\label{rem:propU_status}
ORX \cite{OsipovRokhlinXiao} provides~(U) pointwise
for fixed rescaled $x$.
Condition~(U') is a uniform discrete increment bound;
it is the genuine analytic hard core that ORX does
not explicitly address.
All results below are conditional on
Proposition~\ref{prop:U}.
\end{remark}

\begin{remark}[Route to a proof]
\label{rem:propU_route}
A proof of Proposition~\ref{prop:U} is expected from:
\begin{enumerate}[(i)]
\item Uniform Liouville--Green (WKB) control with
  $O(c^{-2/3})$ remainder over the transition window.
\item Airy matching with explicit uniform remainder
  $O(c^{-2/3})$ in $x_l \in [-A_0,A_0]$;
  this yields~(U).
\item Stability of the spectral map under $O(c^{-2/3})$
  perturbations, \emph{including its discrete derivative}
  (i.e.\ stability of $\lambda_l - \lambda_{l+1}$);
  this yields~(U').
\end{enumerate}
Step~(iii) is the additional content beyond~(U)
and requires tracking how the WKB matching propagates
to consecutive eigenvalue differences.
\end{remark}

\section{Lemma A: Uniform Airy Increment}
\label{sec:lemmaA}
% ============================================================

\begin{lemma}[Uniform Airy Increment]
\label{lem:A}
Under Proposition~\ref{prop:U}~\textup{(U')},
for all $c \ge c_0$ and all $l$ with
$|l-\NSh| \le A_0 c^{1/3}$:
\[
  \lambda_l - \lambda_{l+1}
  = (c/2)^{-1/3}\,\mathrm{Ai}(x_l)^2 + R_l,
\]
where $|R_l| \le (L_{A_0}\,(c/2)^{-1/3} + C_2)\,c^{-1/3}$,
uniform in $l$.
In particular,
$|\lambda_l - \lambda_{l+1}| \le C_A\,(c/2)^{-1/3}$
with $C_A := C_{\max} + L_{A_0} + C_2$.
\end{lemma}

\begin{proof}
Write:
\begin{align*}
  \lambda_l - \lambda_{l+1}
  &= [\FAi(x_l) - \FAi(x_{l+1})]
  + [(\lambda_l - \lambda_{l+1}) -
     (\FAi(x_l) - \FAi(x_{l+1}))].
\end{align*}

\textbf{Main term.}
$\FAi' = -\mathrm{Ai}^2$ and $x_{l+1}-x_l = (c/2)^{-1/3}$,
so by MVT: $\FAi(x_l)-\FAi(x_{l+1})
= \mathrm{Ai}(\xi_l)^2(c/2)^{-1/3}$
for some $\xi_l \in [x_l,x_{l+1}]$.
By Lipschitz continuity of $\mathrm{Ai}^2$ with
constant $L_{A_0}$:
$|\mathrm{Ai}(\xi_l)^2-\mathrm{Ai}(x_l)^2|
\le L_{A_0}(c/2)^{-1/3}$,
so the MVT contribution to $R_l$ is
$\le L_{A_0}(c/2)^{-2/3}$.

\textbf{Error term.}
By (U') directly:
$|(\lambda_l-\lambda_{l+1})-(\FAi(x_l)-\FAi(x_{l+1}))|
\le C_2 c^{-2/3}$.

\textbf{Combining:}
$|R_l| \le L_{A_0}(c/2)^{-2/3} + C_2 c^{-2/3}
\le (L_{A_0}+C_2 2^{2/3})c^{-2/3}$.
This is $O(c^{-2/3})$ uniformly in $l$.
\end{proof}

\begin{remark}
The use of~(U') in the error term is essential:
without it, $|E_l - E_{l+1}|$ is not directly bounded
by $C c^{-2/3}$ from~(U) alone (the errors could
oscillate at scale $c^{-2/3}$). See
Remark~\ref{rem:propU_uprime}.
\end{remark}

\section{Lemma B: Airy Ratio Bound}
\label{sec:lemmaB}
% ============================================================

\begin{lemma}[Airy Ratio Bound]
\label{lem:B}
$\mathrm{Ai}(x)^2/\FAi(x) \le C_{\max}/c_*$
for all $x \in [-A_0,A_0]$.
\end{lemma}

\begin{proof}
Numerator $\le C_{\max}$, denominator
$\ge c_* > 0$.
\end{proof}

\section{Lemma F: Airy Rigidity and Dyadic Separation}
\label{sec:lemmaF}
% ============================================================

\begin{lemma}[Airy Rigidity / Dyadic Separation Principle]
\label{lem:F}
$\FAi$ is bi-Lipschitz on $[-A_0,A_0]$:
\[
  c_{\min}|x_i-x_j|
  \le |\FAi(x_i)-\FAi(x_j)|
  \le C_{\max}|x_i-x_j|.
\]
Under~\textup{(U)}, with $c_1 := 2(4C_1/c_{\min})^3$,
for $c \ge c_1$ and all $i \ne j$ with
$x_i,x_j \in [-A_0,A_0]$:
\begin{equation}\label{eq:specsep}
  |\lambda_i-\lambda_j|
  \ge \tfrac{1}{2}c_{\min}(c/2)^{-1/3}|i-j|.
\end{equation}
\end{lemma}

\begin{proof}
\textbf{Bi-Lipschitz.}
$|\FAi(x_i)-\FAi(x_j)| = |\int_{x_j}^{x_i}\mathrm{Ai}^2\,dt|$;
$c_{\min} \le \mathrm{Ai}^2 \le C_{\max}$ on $[-A_0,A_0]$.
(Strict positivity $c_{\min}>0$: $[-A_0,A_0]$ contains
no zero of $\mathrm{Ai}$ since $A_0 < |a_1|$; this is
the key rigidity fact driving the Dyadic Separation
Principle.)

\textbf{Spectral separation.}
By~(U): $|\lambda_i-\lambda_j|
\ge c_{\min}(c/2)^{-1/3}|i-j| - 2C_1 c^{-2/3}$.
For $c \ge c_1$: $2C_1 c^{-2/3}
\le \frac{1}{2}c_{\min}(c/2)^{-1/3}$,
absorbing the error.
\end{proof}

\begin{remark}[Microscopic scale and dyadic truncation]
\label{rem:microscopic}
At $|i-j|=O(1)$, both main term and error
are $O(c^{-1/3})$; absorption fails and
\eqref{eq:specsep} degenerates.
This is precisely why $m \ge m_0$ (fixed $2^{m_0}$
independent of $c$) is necessary.
The Core regime $m < m_0$ is handled by H3.
\end{remark}

\section{Corollary C: H2 in Truncated Dyadic Sense}
\label{sec:H2}
% ============================================================

\begin{corollary}[H2 in truncated dyadic sense]
\label{cor:C}
Fix $m_0 \ge 1$; let $c_2 := \max(c_0,c_1)$.
Under Proposition~\ref{prop:U}, define
\[
  \delta_m
  := \frac{2C_A}{c_{\min}} 2^{-m}
  + 2\,\frac{C_{\max}}{c_*}(c/2)^{-1/3}.
\]
All indices $i,j$ lie strictly inside the transition
window ($x_j \in [-A_0,0]$, $\FAi(x_j) \ge c_* > 0$,
away from the bulk plateau), so~\eqref{eq:onemlam}
applies uniformly.
For all $c \ge c_2$, $i \le N$, $m \ge m_0$,
$j,j+1 \in \Bm(i)$:
\[
  \frac{|A_{ij}-A_{i,j+1}|}{A_{ij}} \le \delta_m
  = O(2^{-m})+O(c^{-1/3}).
\]
For any $\varepsilon>0$: first choose $m_0$ so
$\frac{2C_A}{c_{\min}}2^{-m_0} < \varepsilon/2$,
then $c$ so
$2\frac{C_{\max}}{c_*}(c/2)^{-1/3} < \varepsilon/2$;
then $\sup_{m\ge m_0}\delta_m < \varepsilon$.
\end{corollary}

\begin{proof}
$\frac{|A_{ij}-A_{i,j+1}|}{A_{ij}}
\lesssim
\underbrace{\frac{|\lambda_j-\lambda_{j+1}|}
            {|\lambda_i-\lambda_j|}}_{\text{Term 1}}
+
\underbrace{\frac{|\lambda_j-\lambda_{j+1}|}
            {1-\lambda_j}}_{\text{Term 2}}.$

\textbf{Term 1 (Dyadic Separation).}
Lemma~\ref{lem:A}: numerator $\le C_A(c/2)^{-1/3}$.
Lemma~\ref{lem:F}: denominator
$\ge \frac{1}{2}c_{\min}(c/2)^{-1/3}\cdot 2^m$.
The $(c/2)^{-1/3}$ factors cancel:
Term~1 $\le \frac{2C_A}{c_{\min}}2^{-m}$, independent of $c$.

\textbf{Term 2 (Global error).}
Lemma~\ref{lem:A} + Lemma~\ref{lem:B}
+ lower bound of~\eqref{eq:onemlam}:
\[
  \frac{|\lambda_j-\lambda_{j+1}|}{1-\lambda_j}
  \le
  \frac{(c/2)^{-1/3}\mathrm{Ai}(x_j)^2 + O(c^{-2/3})}
       {\frac{1}{2}\FAi(x_j)}
  \le 2\frac{C_{\max}}{c_*}(c/2)^{-1/3}.
\]
(Indices strictly inside transition window:
$\FAi(x_j) \ge c_* > 0$; not near bulk plateau.)
\end{proof}

\begin{remark}[The Dyadic Separation Principle]
\label{rem:dyadic}
The cancellation of $(c/2)^{-1/3}$ in Term~1 is
the operative content:
both $|\lambda_j-\lambda_{j+1}|$ and
$|\lambda_i-\lambda_j|$ scale as $c^{-1/3}$,
but $|\lambda_i-\lambda_j|$ carries $2^m$.
Result: purely combinatorial $O(2^{-m})$,
independent of $c$.
The $O(c^{-1/3})$ in $\delta_m$ comes from Term~2 only.
\end{remark}

\section{Lemma E: Core--WKB Split}
\label{sec:split}
% ============================================================

\begin{lemma}[Core--WKB Split]
\label{lem:E}
Fix $m_0 \ge 1$; assume Proposition~\ref{prop:U}
and B-strong \cite{paperVI} ($A_{ij} \le C_2$).
$\sum_{j\ne i}|T_{ij}| \le C(m_0,A,C_2)$
uniformly in $i$, $N$, $c \ge c_2$.
\end{lemma}

\begin{proof}
Split at $m_0$.

\textbf{Core ($m<m_0$).}
$|T_{ij}| \le C_2/|i-j|$, a consequence of B-strong
and the off-diagonal structure of $T_{ij}$
(\cite[Lemma~5.2]{paperVII}).
Each $\Bm(i)$ contributes $\le 2C_2$;
sum over $m_0$ blocks: $\le 2C_2 m_0$.

\textbf{WKB ($m \ge m_0$).}
Corollary~\ref{cor:C} gives H2 for all $m\ge m_0$.
Apply Theorem~2.1 of \cite{paperVII} to the
\emph{truncated operator}, using H1 and H3: $O(1)$.
\end{proof}

\begin{remark}[Role of H3]
\label{rem:H3role}
H3 is not used in proving H2 (Corollary~\ref{cor:C}).
It enters in Lemma~\ref{lem:E} in two places:
(i) directly in the Core estimate
($|T_{ij}| \le C_2/|i-j|$),
and (ii) as input to Theorem~2.1 of \cite{paperVII}.
A proof of H3 would not simplify the H2 argument.
\end{remark}

\section{Conditional Reduction: Hypothesis Verification}
\label{sec:closure}
% ============================================================

\begin{corollary}[Conditional Reduction]
\label{cor:closure}
Under Proposition~\ref{prop:U} and B-strong,
hypotheses (H1)--(H3) of Theorem~2.1 in
\cite{paperVII} are verified (H2 in truncated
dyadic sense), giving:
\[
  \sup_{i\le N}\sum_{j\ne i}|P_{ij}|
  \le C c^{-1/2}\log c
  \qquad(\text{Assumption }B').
\]
The contraction estimate
$\norm{D\mathcal{T}_c^{(N)}} \le C(A)c^{-1/2}\log c < 1$
holds for $c \ge c_2$, conditional on
Proposition~\ref{prop:U} and B-strong.
\end{corollary}

\begin{remark}[Logical dependency chain]
\[
  \text{(U')}\Rightarrow\text{Lem.~A};
  \quad
  \text{(U)}\Rightarrow\text{Lem.~F};
  \quad
  \text{Lem.~A}+\text{Lem.~F}\Rightarrow\text{Cor.~C (H2)};
  \quad
  \text{H2}+\text{H3}\Rightarrow\text{Lem.~E}\Rightarrow B'\Rightarrow A.
\]
H1: unconditional.
All thresholds $c_0,c_1,c_2$ depend only on
$A_0,C_1,c_{\min}$.
\end{remark}

\begin{remark}[Open problems]
\label{rem:open}
\begin{itemize}
\item \textbf{Proposition~\ref{prop:U}~(U'):}
  the genuine analytic hard core;
  (U) is the pointwise bound from ORX,
  (U') is the discrete increment bound.
  See Remark~\ref{rem:propU_route} for strategy.
\item \textbf{B-strong:}
  $P_{kl} \le C_2 c^{1/2}$ in the transition zone.
  Plausible from Widom + WKB-to-Airy matching,
  consistent with numerical evidence;
  no proof currently available.
\end{itemize}
\end{remark}
