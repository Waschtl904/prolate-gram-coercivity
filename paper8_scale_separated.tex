\title[Scale-Separated Dyadic Cancellation]
      {Scale-Separated Dyadic Cancellation\\
       for PSWF Galerkin Operators:\\
       Reduction of H2 to Uniform Airy Asymptotics}

\author{[Author]}
\date{\today}
\maketitle

% ------------------------------------------------------------
\begin{abstract}
We reduce the open amplitude hypothesis~(H2) of
Paper~VII \cite{paperVII} to two uniform analytic inputs
about PSWF eigenvalues in the transition window
(Proposition~\ref{prop:U}): a pointwise $O(c^{-2/3})$
Airy approximation~(U) and a discrete increment
bound~(U') controlling the discrete derivative of
the spectral curve.
All results are conditional on Proposition~\ref{prop:U}
and on Assumption~B-strong \cite{paperVI}.

The main contributions are:
\begin{itemize}
\item Lemma~\ref{lem:A} (Uniform Airy Increment):
  under~(U'),
  $\lambda_l - \lambda_{l+1}
  = (c/2)^{-1/3}\mathrm{Ai}(x_l)^2 + R_l$
  with $|R_l| \le C_A' c^{-2/3}$ uniformly.
\item Lemma~\ref{lem:B} (Airy Ratio Bound).
\item Lemma~\ref{lem:F} (Dyadic Separation Principle):
  $|\lambda_i-\lambda_j| \ge
  \tfrac{1}{2}c_{\min}(c/2)^{-1/3}|i-j|$
  for $c \ge c_1$.
\item Corollary~\ref{cor:C} (H2 in truncated dyadic sense):
  $\delta_m = O(2^{-m})+O(c^{-1/3})$.
\item Lemma~\ref{lem:E} (Core--WKB Split).
\end{itemize}
The central mechanism is the
\emph{Dyadic Separation Principle}: for
$j \in \mathcal{B}_m(i)$,
$|\lambda_j-\lambda_{j+1}|/|\lambda_i-\lambda_j|
\sim 2^{-m}$, not contained in ORX
\cite{OsipovRokhlinXiao}.
\end{abstract}

% ============================================================
\section{Introduction and Context}
\label{sec:intro}
% ============================================================

Paper~VII \cite{paperVII} established an abstract dyadic
cancellation theorem (Theorem~2.1) and reduced the
contraction problem to three hypotheses (H1)--(H3).
H2 was stated as Conjecture~6.2 and is reduced here
to Proposition~\ref{prop:U}.
This paper is a \emph{conditional reduction theorem}:
under Proposition~\ref{prop:U} and B-strong,
hypotheses (H1)--(H3) are verified and the contraction
estimate follows.

\medskip\noindent
\textbf{The Dyadic Separation Principle.}
For $j \in \mathcal{B}_m(i)$:
both $|\lambda_j-\lambda_{j+1}|$ and $|\lambda_i-\lambda_j|$
scale as $c^{-1/3}$, but $|\lambda_i-\lambda_j|$ carries
an extra $2^m$ from $|i-j| \ge 2^m$, so the ratio is
$O(2^{-m})$ independently of $c$.

\medskip\noindent
\textbf{H2 in truncated dyadic sense.}
For $m \ge m_0$: H2 via Corollary~\ref{cor:C}.
For $m < m_0$: bounded by H3.
Theorem~2.1 of \cite{paperVII} applied to truncated
operator (blocks $m \ge m_0$ only).

\medskip\noindent
\textbf{What is not proved here.}
Proposition~\ref{prop:U} and B-strong remain open;
see Remarks~\ref{rem:propU_route} and~\ref{rem:open}.

\section{Setup and Notation}
\label{sec:setup}
% ============================================================

Throughout, $c \ge 1$; $\lambda_k \in (0,1)$ ordered
decreasing.

\medskip\noindent
\textbf{Airy zeros and $A_0$.}
All zeros of $\mathrm{Ai}$ are negative;
$a_1 \approx -2.3381$ is the largest \cite[Sec.~9.9]{DLMF}.
Fix $A_0 \in (0,|a_1|)$; then $[-A_0,A_0]$ is zero-free.

\medskip\noindent
\textbf{Compactness constants (depend only on $A_0$):}
\begin{align*}
  c_* &:= \min_{[-A_0,0]}\FAi > 0,\\
  c_{\min} &:= \min_{[-A_0,A_0]}\mathrm{Ai}^2 > 0,\quad
  C_{\max} := \max_{[-A_0,A_0]}\mathrm{Ai}^2 < \infty.
\end{align*}

\medskip\noindent
\textbf{Lipschitz constant:}
$L_{A_0} := \sup_{[-A_0,A_0]}|\frac{d}{dx}\mathrm{Ai}^2|
= 2\sup|\mathrm{Ai}\,\mathrm{Ai}'| < \infty.$

\medskip\noindent
\textbf{Monotonicity fact.}
$\FAi$ is strictly decreasing on $\mathbb{R}$
($\FAi' = -\mathrm{Ai}^2 \le -c_{\min} < 0$ on $[-A_0,A_0]$),
so $\FAi(x) \ge \FAi(0) > 0$ for all $x \le 0$.
Hence $c_* \ge \FAi(0) > 0$, and~\eqref{eq:onemlam}
holds uniformly for all transition-window indices
with $x_j \le 0$.

\medskip\noindent
\textbf{All constants} depend only on $A_0$; none
depend on $c$, $i$, $j$, $m$.

\medskip\noindent
\textbf{Rescaled index:}
$x_l := (l-\NSh)(c/2)^{-1/3}$, $\NSh = 2c/\pi$;
spacing $x_{l+1}-x_l = (c/2)^{-1/3}$ exactly.
All Galerkin indices: $x_l \in [-A,0] \subset [-A_0,0]$.

\medskip\noindent
\textbf{Notation:}
$A_{ij} := P_{ij}c^{1/2}$,\;
$P_{ij} := c|\lambda_i-\lambda_j|/
((1-\lambda_i)(1-\lambda_j))$;\;
$\Bm(i) := \{j: 2^m \le |i-j| < 2^{m+1}\}$.

\medskip\noindent
\textbf{Explicit bound on $1-\lambda_j$.}
For $c \ge c_0$ (with $C_1 c_0^{-2/3} \le c_*/2$):
\begin{equation}\label{eq:onemlam}
  \tfrac{1}{2}\FAi(x_j) \le 1-\lambda_j \le 2\FAi(x_j)
\end{equation}
uniformly in all transition-window indices.

\section{ORX Uniformity in the Transition Window}
\label{sec:propU}
% ============================================================

\begin{proposition}[ORX Uniformity]
\label{prop:U}
There exist $C_1, C_2, c_0 > 0$ (depending only on $A_0$)
such that for all $c \ge c_0$ and all $l$ with
$|l-\NSh| \le A_0 c^{1/3}$:
\begin{itemize}
\item[\textup{(U)}] $\abs{\lambda_l - \FAi(x_l)}
  \le C_1 c^{-2/3}$.
\item[\textup{(U')}] $\abs{(\lambda_l-\lambda_{l+1})
  -(\FAi(x_l)-\FAi(x_{l+1}))}
  \le C_2 c^{-2/3}$.
\end{itemize}
Both bounds are uniform in $l$.
\end{proposition}

\begin{remark}[Why (U') is necessary]
\label{rem:propU_uprime}
Lemma~\ref{lem:A} requires bounding
$E_l - E_{l+1}$ where $E_l := \lambda_l - \FAi(x_l)$.
Condition~(U) gives only $|E_l|, |E_{l+1}| \le C_1 c^{-2/3}$
individually; it does not prevent the errors from
oscillating at scale $c^{-2/3}$, in which case
$|E_l - E_{l+1}|$ need not be bounded by a useful
constant.
Condition~(U') directly supplies this bound.
In particular, (U') is not a technical refinement of~(U),
but a genuinely stronger condition: it controls the
discrete derivative of the spectral curve.
\end{remark}

\begin{remark}[Heuristic interpretation of (U')]
\label{rem:Uprime_heuristic}
In the transition window, the eigenvalues $\lambda_l$
approximate the smooth Airy profile $\FAi(x)$ sampled
at the discrete points $x_l$.
Condition~(U) asserts pointwise closeness;
condition~(U') asserts that the approximation is
stable under discrete differentiation.
Equivalently: (U') is a \emph{bounded discrete curvature}
condition on the spectral curve $l \mapsto \lambda_l$,
ensuring no oscillations of the error sequence $\{E_l\}$
at scale $c^{-2/3}$.
\end{remark}

\begin{remark}[Status and route to proof]
\label{rem:propU_route}
ORX \cite{OsipovRokhlinXiao} provides~(U) pointwise.
A proof of (U') is expected from:
\begin{enumerate}[(i)]
\item Uniform WKB control with $O(c^{-2/3})$ remainder.
\item Airy matching with uniform remainder in
  $x_l \in [-A_0,A_0]$; yields~(U).
\item Stability of the spectral map including its
  discrete derivative; yields~(U').
\end{enumerate}
Step~(iii) is the additional content beyond~(U).
\end{remark}

\section{Lemma A: Uniform Airy Increment}
\label{sec:lemmaA}
% ============================================================

\begin{lemma}[Uniform Airy Increment]
\label{lem:A}
Under~\textup{(U')}, for all $c \ge c_0$ and all $l$
with $|l-\NSh| \le A_0 c^{1/3}$:
\[
  \lambda_l - \lambda_{l+1}
  = (c/2)^{-1/3}\,\mathrm{Ai}(x_l)^2 + R_l,
\]
where
\[
  |R_l| \le C_A'\,c^{-2/3},
  \qquad
  C_A' := L_{A_0}\cdot 2^{2/3} + C_2,
\]
depending only on $A_0, L_{A_0}, C_2$, uniform in $l$.
In particular, the remainder is $O(c^{-2/3})$ uniformly
in $l$, and $|\lambda_l-\lambda_{l+1}| \le C_A(c/2)^{-1/3}$
with $C_A := C_{\max} + C_A'\cdot 2^{-1/3}$.
\end{lemma}

\begin{proof}
Write:
\[
  \lambda_l-\lambda_{l+1}
  = [\FAi(x_l)-\FAi(x_{l+1})]
  + [(\lambda_l-\lambda_{l+1})-(\FAi(x_l)-\FAi(x_{l+1}))].
\]

\textbf{Main term.}
$\FAi'=-\mathrm{Ai}^2$, spacing $(c/2)^{-1/3}$ exact;
MVT gives $\xi_l \in [x_l,x_{l+1}]$ with
$\FAi(x_l)-\FAi(x_{l+1})
= \mathrm{Ai}(\xi_l)^2(c/2)^{-1/3}$.
By Lipschitz continuity of $\mathrm{Ai}^2$ with
constant $L_{A_0}$:
$|\mathrm{Ai}(\xi_l)^2 - \mathrm{Ai}(x_l)^2|
\le L_{A_0}(c/2)^{-1/3}$,
contributing $L_{A_0}(c/2)^{-2/3}$ to $R_l$.

\textbf{Increment error.}
By~(U') directly:
$|(\lambda_l-\lambda_{l+1})-(\FAi(x_l)-\FAi(x_{l+1}))|
\le C_2 c^{-2/3}$.

\textbf{Combining:}
$|R_l|
\le L_{A_0}(c/2)^{-2/3} + C_2 c^{-2/3}
= (L_{A_0}\cdot 2^{2/3}+C_2)\,c^{-2/3}
= C_A'\,c^{-2/3}$.
\end{proof}

\begin{remark}
The use of~(U') in the increment error is essential:
without it, $|E_l-E_{l+1}|$ is not controlled
by $Cc^{-2/3}$ from~(U) alone.
\end{remark}

\section{Lemma B: Airy Ratio Bound}
\label{sec:lemmaB}
% ============================================================

\begin{lemma}[Airy Ratio Bound]
\label{lem:B}
$\mathrm{Ai}(x)^2/\FAi(x) \le C_{\max}/c_*$
for all $x \in [-A_0,A_0]$.
\end{lemma}

\begin{proof}
Numerator $\le C_{\max}$, denominator $\ge c_*>0$.
\end{proof}

\section{Lemma F: Airy Rigidity and Dyadic Separation}
\label{sec:lemmaF}
% ============================================================

\begin{lemma}[Airy Rigidity / Dyadic Separation Principle]
\label{lem:F}
$\FAi$ is bi-Lipschitz on $[-A_0,A_0]$:
$c_{\min}|x_i-x_j|
\le |\FAi(x_i)-\FAi(x_j)|
\le C_{\max}|x_i-x_j|$.
Under~\textup{(U)}, with $c_1 := 2(4C_1/c_{\min})^3$,
for $c \ge c_1$:
\begin{equation}\label{eq:specsep}
  |\lambda_i-\lambda_j|
  \ge \tfrac{1}{2}c_{\min}(c/2)^{-1/3}|i-j|.
\end{equation}
\end{lemma}

\begin{proof}
\textbf{Bi-Lipschitz.}
$|\FAi(x_i)-\FAi(x_j)|
= |\int_{x_j}^{x_i}\mathrm{Ai}^2\,dt|$;
$c_{\min} \le \mathrm{Ai}^2 \le C_{\max}$ on $[-A_0,A_0]$
(strict positivity: $A_0 < |a_1|$, no zero;
this is the key rigidity fact).

\textbf{Spectral separation.}
By~(U):
$|\lambda_i-\lambda_j|
\ge c_{\min}(c/2)^{-1/3}|i-j|-2C_1c^{-2/3}$.
For $c \ge c_1$: $2C_1c^{-2/3}
\le \frac{1}{2}c_{\min}(c/2)^{-1/3}$,
giving~\eqref{eq:specsep}.
\end{proof}

\begin{remark}[Microscopic scale and dyadic truncation]
\label{rem:microscopic}
At $|i-j|=O(1)$, absorption fails and~\eqref{eq:specsep}
degenerates (both main term and error are $O(c^{-1/3})$).
This is why $m \ge m_0$ is necessary.
\end{remark}

\section{Corollary C: H2 in Truncated Dyadic Sense}
\label{sec:H2}
% ============================================================

\begin{corollary}[H2 in truncated dyadic sense]
\label{cor:C}
Fix $m_0\ge 1$; let $c_2:=\max(c_0,c_1)$.
Under Proposition~\ref{prop:U}, define
\[
  \delta_m
  := \frac{2C_A}{c_{\min}}2^{-m}
  + 2\frac{C_{\max}}{c_*}(c/2)^{-1/3}.
\]
All indices $i,j$ lie in the transition window
($x_j \in [-A_0,0]$, $\FAi(x_j) \ge c_*>0$),
so~\eqref{eq:onemlam} applies uniformly.
For $c \ge c_2$, $i \le N$, $m \ge m_0$,
$j,j+1 \in \Bm(i)$:
\[
  \frac{|A_{ij}-A_{i,j+1}|}{A_{ij}}
  \le \delta_m = O(2^{-m})+O(c^{-1/3}).
\]
For any $\varepsilon>0$: choose $m_0$ so
$\frac{2C_A}{c_{\min}}2^{-m_0}<\varepsilon/2$,
then $c$ so
$2\frac{C_{\max}}{c_*}(c/2)^{-1/3}<\varepsilon/2$;
then $\sup_{m\ge m_0}\delta_m < \varepsilon$.
\end{corollary}

\begin{proof}
$\frac{|A_{ij}-A_{i,j+1}|}{A_{ij}}
\lesssim
\underbrace{\frac{|\lambda_j-\lambda_{j+1}|}
            {|\lambda_i-\lambda_j|}}_{\text{Term 1}}
+
\underbrace{\frac{|\lambda_j-\lambda_{j+1}|}
            {1-\lambda_j}}_{\text{Term 2}}.$

\textbf{Term 1.}
Lemma~\ref{lem:A}: numerator $\le C_A(c/2)^{-1/3}$.
Lemma~\ref{lem:F}: denominator
$\ge \frac{1}{2}c_{\min}(c/2)^{-1/3}\cdot 2^m$.
The $(c/2)^{-1/3}$ cancels:
Term~1 $\le \frac{2C_A}{c_{\min}}2^{-m}$, \emph{independent of $c$}.

\textbf{Term 2.}
Lemma~\ref{lem:A} + Lemma~\ref{lem:B}
+ lower bound~\eqref{eq:onemlam}:
$\frac{|\lambda_j-\lambda_{j+1}|}{1-\lambda_j}
\le \frac{(c/2)^{-1/3}\mathrm{Ai}(x_j)^2+O(c^{-2/3})}
         {\frac{1}{2}\FAi(x_j)}
\le 2\frac{C_{\max}}{c_*}(c/2)^{-1/3}$.
\end{proof}

\begin{remark}[The Dyadic Separation Principle]
\label{rem:dyadic}
The cancellation of $(c/2)^{-1/3}$ in Term~1 is
the operative content: both scales are $c^{-1/3}$,
but the denominator carries $2^m$.
Result: purely combinatorial $O(2^{-m})$, independent of $c$.
The $O(c^{-1/3})$ in $\delta_m$ comes from Term~2 only.
\end{remark}

\section{Lemma E: Core--WKB Split}
\label{sec:split}
% ============================================================

\begin{lemma}[Core--WKB Split]
\label{lem:E}
Fix $m_0 \ge 1$; assume Proposition~\ref{prop:U}
and B-strong \cite{paperVI} ($A_{ij} \le C_2$).
Then $\sum_{j\ne i}|T_{ij}| \le C(m_0,A,C_2)$
uniformly in $i$, $N$, $c \ge c_2$.
\end{lemma}

\begin{proof}
\textbf{Core ($m<m_0$).}
$|T_{ij}| \le C_2/|i-j|$ (B-strong +
off-diagonal structure, \cite[Lemma~5.2]{paperVII});
sum: $\le 2C_2 m_0$.

\textbf{WKB ($m\ge m_0$).}
Corollary~\ref{cor:C} gives H2.
Apply Theorem~2.1 of \cite{paperVII} to truncated
operator, using H1 and H3: $O(1)$.
\end{proof}

\begin{remark}[Role of H3]
\label{rem:H3role}
H3 is not used in proving H2.
It enters Lemma~\ref{lem:E} in two places:
(i) Core estimate; (ii) input to Theorem~2.1.
\end{remark}

% ============================================================
\section{Conditional Reduction}
\label{sec:closure}
% ============================================================

\begin{corollary}[Conditional Reduction]
\label{cor:closure}
Under Proposition~\ref{prop:U} and B-strong,
hypotheses (H1)--(H3) of Theorem~2.1 in \cite{paperVII}
are verified, giving
$\sup_{i\le N}\sum_{j\ne i}|P_{ij}|
\le Cc^{-1/2}\log c$
and $\norm{D\mathcal{T}_c^{(N)}} < 1$ for $c\ge c_2$.
\end{corollary}

\begin{remark}[Dependency chain]
\[
  \text{(U')}\Rightarrow\text{Lem.~A};\quad
  \text{(U)}\Rightarrow\text{Lem.~F};\quad
  \text{Lem.~A+F}\Rightarrow\text{Cor.~C};\quad
  \text{Cor.~C+H3}\Rightarrow\text{Lem.~E}
  \Rightarrow B'\Rightarrow A.
\]
All thresholds depend only on $A_0,C_1,c_{\min}$.
\end{remark}

\begin{remark}[Open problems]
\label{rem:open}
\begin{itemize}
\item \textbf{(U'):} the genuine hard core;
  see Remark~\ref{rem:propU_route}.
\item \textbf{B-strong:} $P_{kl}\le C_2 c^{1/2}$
  in the transition zone; plausible from Widom +
  WKB-to-Airy matching; no proof available.
\end{itemize}
\end{remark}
