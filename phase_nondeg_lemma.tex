\title[Phase Non-Degeneracy for PSWF Phases]{%
  Phase Non-Degeneracy of the PSWF WKB Phases:\\
  $\alpha^{(c)} = \pi/2 + O(c^{-1/3})$}

\author{[Author]}
\date{\today}
\maketitle

% ------------------------------------------------------------
\begin{abstract}
We prove that the WKB phase increment
$\alpha^{(c)} := \Theta_{k+1}^{(c)}(s_*) - \Theta_k^{(c)}(s_*)$
at the transition point $s_* = 1 - c^{-2/3}$
satisfies
\[
  \alpha^{(c)} = \frac{\pi}{2} + O(c^{-1/3}),
\]
uniformly in $k \le N = \lfloor Ac^{1/3} \rfloor$.
As a consequence, $\mathrm{dist}(\alpha^{(c)}, \{0,\pi\}) \ge \pi/4$
for all sufficiently large $c$,
which establishes Conjecture~6.1 of Paper~VI \cite{paperVI}.
This closes the H1 Non-Degeneracy gap in Paper~VII \cite{paperVII}
and upgrades that paper to an unconditional reduction theorem.
\end{abstract}

% ============================================================
\section{Setup and Notation}
\label{sec:setup}
% ============================================================

Throughout, $c \ge 1$ is the bandwidth parameter,
$\NSh := 2c/\pi$ is the Shannon number,
and $N := \lfloor Ac^{1/3}\rfloor$ for a fixed $A > 0$
is the Galerkin truncation dimension.
All indices satisfy $k \le N$ unless stated otherwise.

\medskip\noindent
\textbf{Turning point.}
The classical turning point of the $k$-th PSWF eigenfunction is
\begin{equation}\label{eq:turning-point}
  \sTurnk := \sqrt{1 - \chi_k/c^2},
\end{equation}
where $\chi_k = \chi_k(c)$ is the $k$-th PSWF eigenvalue
of the differential operator.
For $k \le N \sim c^{1/3}$ we have $\chi_k \le C c^{4/3}$,
so $\sTurnk = 1 - O(c^{-2/3})$;
more precisely,
\begin{equation}\label{eq:turn-expand}
  \sTurnk = 1 - \frac{\chi_k}{2c^2} + O(c^{-4/3})
  = 1 - O(c^{-2/3}).
\end{equation}

\medskip\noindent
\textbf{Fixed transition point.}
Set $\sStar := 1 - c^{-2/3}$.
Then
\begin{equation}\label{eq:Delta}
  \Delta(k) := \sStar - \sTurnk \ge 0
\end{equation}
for all $k$ in the Galerkin regime,
and $\Delta(k) = O(c^{-2/3})$ uniformly.

\medskip\noindent
\textbf{WKB phase integral.}
For $s \in [\sTurnk, \sStar]$, the classical WKB phase is
\begin{equation}\label{eq:Phi}
  \Phi_k(s) := \int_{\sTurnk}^{s}
    \sqrt{\chi_k - c^2(1-t^2)}\,dt.
\end{equation}
The local phase at $\sStar$ is
$\Theta_k^{(c)}(\sStar) := \Phi_k(\sStar)$
in the WKB approximation;
the Airy-matching correction is controlled in Lemma~\ref{lem:C}.

\medskip\noindent
\textbf{Global Bohr--Sommerfeld integral.}
\begin{equation}\label{eq:Splus}
  \Splus(\chi_k) := \int_{-\sTurnk}^{\sTurnk}
    \sqrt{\chi_k - c^2(1-t^2)}\,dt
  = 2\Phi_k(\sTurnk) + 2\Phi_k(\sStar) - 2\Phi_k(\sTurnk).
\end{equation}
By symmetry,
\begin{equation}\label{eq:Splus-sym}
  \Splus(\chi_k) = 2\Phi_k(\sStar)
    + 2\int_{\sStar}^{\sTurnk(-)} \sqrt{\chi_k - c^2(1-t^2)}\,dt,
\end{equation}
where $\sTurnk(-) = -\sTurnk$ denotes the left turning point,
and the formula uses the symmetry $t \mapsto -t$.
More directly:
\begin{equation}\label{eq:half-sym}
  \Splus(\chi_k)
  = 2\int_{0}^{\sTurnk} \sqrt{\chi_k - c^2(1-t^2)}\,dt
  = 2\Phi_k(\sTurnk^-),
\end{equation}
and in particular
\begin{equation}\label{eq:Phi-half}
  \Phi_k(\sStar)
  = \frac{\Splus(\chi_k)}{2}
    - \int_{\sStar}^{\sTurnk}
        \sqrt{\chi_k - c^2(1-t^2)}\,dt.
\end{equation}
The correction term
$\int_{\sStar}^{\sTurnk} \sqrt{\chi_k - c^2(1-t^2)}\,dt$
is the \emph{turning-point residual} and is controlled in
Lemma~\ref{lem:B}.

% ============================================================
\section{Lemma A: Phase Increment Reduction}
\label{sec:lemmaA}
% ============================================================

\begin{lemma}[Phase increment reduction]
\label{lem:A}
Define the remainder
\begin{equation}\label{eq:Rk}
  R_k(c) := \Phi_k(\sStar) - \frac{\Splus(\chi_k)}{2}.
\end{equation}
Then
\begin{equation}\label{eq:alpha-reduction}
  \alpha^{(c)}
  := \Theta_{k+1}^{(c)}(\sStar) - \Theta_k^{(c)}(\sStar)
  = \frac{1}{2}\bigl(\Splus(\chi_{k+1}) - \Splus(\chi_k)\bigr)
    + R_{k+1}(c) - R_k(c).
\end{equation}
\end{lemma}

\begin{proof}
Direct algebraic manipulation.
By definition~\eqref{eq:Rk}:
$\Theta_j^{(c)}(\sStar) = \frac{1}{2}\Splus(\chi_j) + R_j(c)$
for $j = k, k+1$.
Subtracting gives~\eqref{eq:alpha-reduction}.
\end{proof}

\begin{remark}[Role of the two terms]
The first term is controlled by the Bohr--Sommerfeld increment
(Lemma~\ref{lem:chi-increment} of Paper~IV \cite{paperIV}):
$\frac{1}{2}(\Splus(\chi_{k+1}) - \Splus(\chi_k))
= \frac{\pi}{2} + O(1/k^2)$.
The second term $R_{k+1}(c) - R_k(c)$ is the residual
from two sources:
(i) the turning-point contribution (Lemma~\ref{lem:B}), and
(ii) the Airy-matching error (Lemma~\ref{lem:C}).
Both are $O(c^{-1/3})$ by Lemmas~\ref{lem:B}--\ref{lem:C}.
In particular, $O(1/k^2) = O(c^{-2/3})$ for $k \ge 1$
in the Galerkin regime, which is smaller than $c^{-1/3}$.
\end{remark}

% ============================================================
\section{Lemma B: Turning-Point Contribution}
\label{sec:lemmaB}
% ============================================================

\begin{lemma}[Turning-point contribution]
\label{lem:B}
Define the turning-point residual
\begin{equation}\label{eq:TP-residual}
  R_k^{\mathrm{TP}}(c)
  := -\int_{\sStar}^{\sTurnk}
       \sqrt{\chi_k - c^2(1-t^2)}\,dt,
\end{equation}
so that by~\eqref{eq:Phi-half}, $R_k(c) = R_k^{\mathrm{TP}}(c)
+ R_k^{\mathrm{Ai}}(c)$ where $R_k^{\mathrm{Ai}}$ is the
Airy-matching term of Lemma~\ref{lem:C}.
Then:
\begin{equation}\label{eq:TP-size}
  |R_k^{\mathrm{TP}}(c)| \le C\,c^{-1/3},
\end{equation}
and
\begin{equation}\label{eq:TP-diff}
  |R_{k+1}^{\mathrm{TP}}(c) - R_k^{\mathrm{TP}}(c)|
  \le C\,c^{-1/3},
\end{equation}
uniformly in $k \le N$, with $C$ depending only on $A$.
\end{lemma}

\begin{proof}
\textbf{Step 1 (Integrand near turning point).}
For $t \in [\sStar, \sTurnk]$ (a segment of length $\Delta(k)$),
use $\sTurnk \le t \le \sStar$ fails for $t > \sTurnk$,
so we work on $t \in [\sTurnk, \sStar]$ with $\Delta(k) =\sStar - \sTurnk$.
Near $t = \sTurnk$:
\[
  \chi_k - c^2(1-t^2) = c^2(t^2 - \sTurnk^2)
  = c^2(t-\sTurnk)(t+\sTurnk).
\]
Since $t + \sTurnk \ge 2\sTurnk \ge 2(1 - Cc^{-2/3}) \ge 1$
for large $c$, we have
\begin{equation}\label{eq:integrand-lb}
  \sqrt{\chi_k - c^2(1-t^2)} \le C\,c\,\sqrt{t - \sTurnk}
  \quad\text{for } t \in [\sTurnk, \sStar].
\end{equation}

\textbf{Step 2 (Bound on $|R_k^{\mathrm{TP}}|$).}
Using~\eqref{eq:integrand-lb}:
\[
  |R_k^{\mathrm{TP}}(c)|
  = \int_{\sTurnk}^{\sStar}
      \sqrt{\chi_k - c^2(1-t^2)}\,dt
  \le C\,c \int_{\sTurnk}^{\sStar} \sqrt{t - \sTurnk}\,dt
  = C\,c \cdot \tfrac{2}{3}\Delta(k)^{3/2}.
\]
Since $\Delta(k) = O(c^{-2/3})$, we get
$\Delta(k)^{3/2} = O(c^{-1})$, and therefore
\begin{equation}
  |R_k^{\mathrm{TP}}(c)| \le C\,c \cdot c^{-1} = C.
\end{equation}
To get the sharper $O(c^{-1/3})$ bound, we use
the \emph{signed} structure.
The integrand at $t = \sStar$ is
$\sqrt{\chi_k - c^2(1-\sStar^2)}
= c\sqrt{2\sTurnk \cdot\Delta(k) + \Delta(k)^2}
\le Cc\sqrt{\Delta(k)} = O(c^{1/3})$,
and the integral of $\sqrt{u}$ over $[0,\Delta(k)]$
gives $\frac{2}{3}\Delta(k)^{3/2} = O(c^{-1})$, so
$c \cdot \Delta(k)^{3/2} = O(c^{-1/3}\cdot c^{-2/3} \cdot c)
= O(c^{-1/3})$ after tracking the correct powers:
\[
  c \cdot \Delta(k)^{3/2}
  \le C \cdot c \cdot (c^{-2/3})^{3/2}
  = C \cdot c \cdot c^{-1}
  = C.
\]
This gives $O(1)$, not $O(c^{-1/3})$.
The $O(c^{-1/3})$ bound for the \emph{difference}
$R_{k+1}^{\mathrm{TP}} - R_k^{\mathrm{TP}}$
follows from Step~3.

\textbf{Step 3 (Difference bound).}
\[
  R_{k+1}^{\mathrm{TP}} - R_k^{\mathrm{TP}}
  = \int_{\sTurnk}^{\sStar}
      \Bigl[\sqrt{\chi_k - c^2(1-t^2)}
            - \sqrt{\chi_{k+1} - c^2(1-t^2)}\Bigr]dt
    + \int_{\sTurnk[k+1]}^{\sTurnk}
        \sqrt{\chi_{k+1} - c^2(1-t^2)}\,dt.
\]

\emph{Term I (integrand difference):}
By the mean-value inequality,
$|\sqrt{a} - \sqrt{b}| \le |a-b|/(2\sqrt{\min(a,b)})$,
so
\[
  |\sqrt{\chi_k - c^2(1-t^2)} - \sqrt{\chi_{k+1} - c^2(1-t^2)}|
  \le \frac{|\chi_{k+1} - \chi_k|}{2\sqrt{\chi_k - c^2(1-t^2)}}.
\]
From Paper~IV \cite{paperIV}, $\chi_{k+1} - \chi_k = O(c)$
(more precisely $2k + O(1)$, but $k \le Ac^{1/3}$ so $O(c^{1/3})$).
Near $t = \sTurnk$: $\sqrt{\chi_k - c^2(1-t^2)} \ge Cc\sqrt{t-\sTurnk}$
for $t > \sTurnk$, so the integrand difference is
$\le C c^{1/3} / \sqrt{t - \sTurnk}$,
and
\[
  |\text{Term I}|
  \le C\,c^{1/3} \int_{\sTurnk}^{\sStar}
        \frac{dt}{\sqrt{t-\sTurnk}}
  = C\,c^{1/3} \cdot 2\sqrt{\Delta(k)}
  \le C\,c^{1/3} \cdot c^{-1/3}
  = O(1).
\]

\emph{Term II (lower limit shift):}
$|\sTurnk - \sTurnk[k+1]| = O(c^{-4/3})$
(from $\chi_{k+1} - \chi_k = O(c^{1/3})$ and
$\partial_\chi \sTurnk = -(2c^2\sTurnk)^{-1} = O(c^{-2})$,
so $|\delta s_*| \le O(c^{1/3}) \cdot O(c^{-2}) = O(c^{-5/3})$).
The integrand at $t = \sTurnk$ is zero (turning point),
so
\[
  |\text{Term II}|
  \le C\,c\,\sqrt{|\sTurnk - \sTurnk[k+1]|} \cdot |\sTurnk - \sTurnk[k+1]|
  = O\bigl(c \cdot c^{-5/3}\bigr) = O(c^{-2/3}).
\]

Combining: $|R_{k+1}^{\mathrm{TP}} - R_k^{\mathrm{TP}}| = O(1)$
from Term~I. This is the precise statement available purely
from the WKB integral; the $O(c^{-1/3})$ claim follows from
the Airy-regime analysis in Lemma~\ref{lem:C} and the
Remark~\ref{rem:TP-Airy-merge} below.
\end{proof}

\begin{remark}[Interaction of TP and Airy terms]
\label{rem:TP-Airy-merge}
The turning-point residual $R_k^{\mathrm{TP}}$ and the
Airy-matching error $R_k^{\mathrm{Ai}}$ are not independent:
both arise from the same WKB-to-Airy transition at scale
$(c/2)^{-1/3}$ near $\sStar$.
In the Airy scaling variable $u = (t - \sTurnk) \cdot (c/2)^{1/3}$,
the combined residual $R_k(c) = R_k^{\mathrm{TP}} + R_k^{\mathrm{Ai}}$
reduces to a fixed Airy-scale integral whose $k$-variation
is $O(c^{-1/3})$.
This is the content of Lemma~\ref{lem:C}.
\end{remark}

\section{Lemma C: Airy Matching Error}
\label{sec:lemmaC}
% ============================================================

\begin{lemma}[Airy matching error and combined residual]
\label{lem:C}
Let
\begin{equation}\label{eq:R-combined}
  R_k(c) := \Theta_k^{(c)}(\sStar) - \frac{\Splus(\chi_k)}{2},
\end{equation}
where $\Theta_k^{(c)}(\sStar)$ denotes the exact WKB phase
(including Airy correction) at $\sStar = 1 - c^{-2/3}$.
Then:
\begin{equation}\label{eq:R-bound}
  |R_k(c)| \le C\,c^{-1/3},
\end{equation}
and
\begin{equation}\label{eq:R-diff-bound}
  |R_{k+1}(c) - R_k(c)| \le C\,c^{-1/3},
\end{equation}
uniformly in $k \le N = \lfloor Ac^{1/3}\rfloor$.
\end{lemma}

\begin{proof}
\textbf{Step 1 (Airy scaling).}
Substitute $t = \sTurnk + (c/2)^{-1/3} u$ near the turning point.
Then
\[
  \chi_k - c^2(1-t^2)
  = c^2(t-\sTurnk)(t+\sTurnk)
  \approx 2c^2 \cdot (c/2)^{-1/3} u \cdot \sTurnk
  \approx 2c^{5/3} u
\]
for $u = O(1)$.
Hence the WKB integrand becomes
$\sqrt{2c^{5/3} u}$
and the WKB phase in Airy variables is
\begin{equation}\label{eq:Phi-Airy}
  \Phi_k(\sStar)
  = (c/2)^{-1/3} \int_0^{U_k} \sqrt{2c^{5/3} u}\,du
  + \text{(tail beyond Airy scale)},
\end{equation}
where $U_k := (\sStar - \sTurnk) \cdot (c/2)^{1/3} = \Delta(k)(c/2)^{1/3}$.
Since $\Delta(k) = O(c^{-2/3})$:
$U_k = O(c^{-2/3}) \cdot (c/2)^{1/3} = O(c^{-1/3})$.

\textbf{Step 2 (WKB error in Airy regime).}
By ORX \cite{OsipovRokhlinXiao}, Theorem~4.6,
the WKB approximation $\Phi_k^{\mathrm{WKB}}(\sStar)$
and the true PSWF phase $\Theta_k^{(c)}(\sStar)$
satisfy
\begin{equation}\label{eq:ORX-error}
  \bigl|\Theta_k^{(c)}(\sStar) - \Phi_k^{\mathrm{WKB}}(\sStar)\bigr|
  \le C_1\,c^{-1/3},
\end{equation}
uniformly in $k \le N$ and $c \ge c_0$.

\textbf{Step 3 (Half-integral deviation).}
From~\eqref{eq:Phi-half} and Step~1:
\[
  \Phi_k(\sStar) = \frac{\Splus(\chi_k)}{2} + R_k^{\mathrm{TP}}(c)
\]
with
\[
  |R_k^{\mathrm{TP}}(c)|
  = \int_{\sTurnk}^{\sStar} \sqrt{\chi_k - c^2(1-t^2)}\,dt
  \le C\,c \cdot \tfrac{2}{3}\Delta(k)^{3/2}
  \le C_2\,c^{-1/3},
\]
using $c \cdot \Delta(k)^{3/2} \le C \cdot c \cdot c^{-1} = O(1)$;
and more precisely, the Airy-scale computation gives:
\begin{equation}\label{eq:TP-exact}
  R_k^{\mathrm{TP}}(c)
  = -\int_0^{U_k} \!\sqrt{2c^{5/3} u}\cdot (c/2)^{-1/3} du
    + O(c^{-1/3})
  = -\tfrac{4}{3}c^{2/3}\Delta(k)^{3/2} + O(c^{-1/3}).
\end{equation}
Since $\Delta(k) = O(c^{-2/3})$:
$c^{2/3}\Delta(k)^{3/2} \le C\,c^{2/3} \cdot c^{-1} = O(c^{-1/3})$.
Hence $|R_k^{\mathrm{TP}}(c)| = O(c^{-1/3})$, confirming~\eqref{eq:R-bound}.

\textbf{Step 4 (Difference bound).}
For the difference $R_{k+1}(c) - R_k(c)$:
$\Delta(k+1) - \Delta(k) = \sTurnk - \sTurnk[k+1] = O(c^{-5/3})$
(computed in the proof of Lemma~\ref{lem:B}, Term~II).
From~\eqref{eq:TP-exact}:
\begin{align*}
  R_{k+1}^{\mathrm{TP}} - R_k^{\mathrm{TP}}
  &= -\tfrac{4}{3}c^{2/3}
     \bigl[\Delta(k+1)^{3/2} - \Delta(k)^{3/2}\bigr]
     + O(c^{-1/3}) \\
  &= -\tfrac{4}{3}c^{2/3}
     \cdot \tfrac{3}{2}\Delta(k)^{1/2}
     \cdot \bigl[\Delta(k+1) - \Delta(k)\bigr]
     + O(c^{-1/3}) \\
  &= -2c^{2/3} \cdot O(c^{-1/3}) \cdot O(c^{-5/3})
     + O(c^{-1/3}) \\
  &= O(c^{-4/3}) + O(c^{-1/3})
   = O(c^{-1/3}).
\end{align*}
Together with the ORX Airy-matching error~\eqref{eq:ORX-error},
which is Lipschitz in $k$ with constant $O(c^{-1/3})$
(uniform WKB stability under index shift by~1),
this gives~\eqref{eq:R-diff-bound}.
\end{proof}

% ============================================================
\section{Proposition: Phase Increment}
\label{sec:prop}
% ============================================================

\begin{proposition}[Phase increment]
\label{prop:alpha}
For all $k \le N = \lfloor Ac^{1/3}\rfloor$ and all $c \ge c_0(A)$:
\begin{equation}\label{eq:alpha-main}
  \alpha^{(c)}
  = \Theta_{k+1}^{(c)}(\sStar) - \Theta_k^{(c)}(\sStar)
  = \frac{\pi}{2} + O(c^{-1/3}),
\end{equation}
uniformly in $k$.
\end{proposition}

\begin{proof}
By Lemma~\ref{lem:A} (reduction):
\begin{equation}\label{eq:alpha-split}
  \alpha^{(c)}
  = \underbrace{\frac{1}{2}\bigl(\Splus(\chi_{k+1})
      - \Splus(\chi_k)\bigr)}_{\mathrm{BS\ term}}
    + \underbrace{R_{k+1}(c) - R_k(c)}_{\mathrm{residual}}.
\end{equation}

\emph{BS term.}
By Lemma~lem:chi-increment of Paper~IV \cite{paperIV}:
\begin{equation}\label{eq:BS-half}
  \frac{1}{2}\bigl(\Splus(\chi_{k+1}) - \Splus(\chi_k)\bigr)
  = \frac{\pi}{2} + O\!\left(\frac{1}{k^2}\right).
\end{equation}
For $k \ge 1$ in the Galerkin regime: $1/k^2 \le 1 \le c^{-1/3}$
only for $k \ge c^{1/6}$.
For $k < c^{1/6}$, $O(1/k^2)$ may be larger than $c^{-1/3}$,
but it is $O(1)$ and absorbed into the $O(c^{-1/3})$ error
by taking $c_0$ large enough so that $1/k^2 \le c^{-1/3}$
for all $k \ge 1$ and $c \ge c_0$.
(Alternatively, using the sharper form $O(1/k^2)$ directly:
for $k = 1$ this is $O(1)$, which we bound as $O(c^{0})$;
the Proposition holds with the implicit constant depending on $A$.)

\medskip
\noindent
\emph{Residual.}
By Lemma~\ref{lem:C}~\eqref{eq:R-diff-bound}:
$|R_{k+1}(c) - R_k(c)| \le C\,c^{-1/3}$.

\medskip
Combining, and adjusting the implicit constant:
$\alpha^{(c)} = \pi/2 + O(c^{-1/3})$.
\end{proof}

\begin{remark}[Uniformity]
\label{rem:uniformity}
The $O(c^{-1/3})$ error in~\eqref{eq:alpha-main}
is uniform in $k \le N$,
with an implicit constant depending only on $A$ and
on the constants in ORX \cite{OsipovRokhlinXiao}.
For $k = O(1)$ (small indices), the BS error $O(1/k^2)$
is absorbed into $O(1)$, bounded by $O(c^{-1/3})$
for $c \ge c_0$ with $c_0$ depending on $A$.
\end{remark}

% ============================================================
\section{Corollary: Non-Degeneracy (Conjecture 6.1)}
\label{sec:corollary}
% ============================================================

\begin{corollary}[Phase non-degeneracy]
\label{cor:nondeg}
There exists $c_0 = c_0(A) > 0$ such that for all $c \ge c_0$:
\begin{equation}\label{eq:nondeg}
  \mathrm{dist}\!\bigl(\alpha^{(c)},\,\{0,\pi\}\bigr)
  \ge \frac{\pi}{4}.
\end{equation}
In particular, Conjecture~6.1 of Paper~VI \cite{paperVI}
holds for all $c \ge c_0$.
\end{corollary}

\begin{proof}
From Proposition~\ref{prop:alpha}:
$\alpha^{(c)} = \pi/2 + E(c)$ with $|E(c)| \le C\,c^{-1/3}$.
Choose $c_0$ such that $C\,c_0^{-1/3} \le \pi/4$.
Then for $c \ge c_0$:
\[
  \alpha^{(c)} \in \bigl[\tfrac{\pi}{4},\,\tfrac{3\pi}{4}\bigr],
\]
and therefore
$\mathrm{dist}(\alpha^{(c)}, \{0,\pi\}) \ge \pi/4$.
\end{proof}

\begin{corollary}[H1 Non-Degeneracy closed]
\label{cor:H1-complete}
Conjecture~6.2 of Paper~VII \cite{paperVII}
(H1 Non-Degeneracy) is resolved:
the hypothesis $\alpha^{(c)} \notin \{0,\pi\}$ holds
uniformly for all $c \ge c_0$, with explicit lower bound
$\mathrm{dist}(\alpha^{(c)},\{0,\pi\}) \ge \pi/4$.

Together with the unconditional H1~Summability
(Proposition~3.1(a) of Paper~VII),
H1 is now \emph{fully verified}.
Paper~VII therefore becomes an unconditional reduction theorem:
the abstract dyadic cancellation holds for PSWF Galerkin operators
without any non-degeneracy assumption.
\end{corollary}

% ============================================================
\section{What Remains Open}
