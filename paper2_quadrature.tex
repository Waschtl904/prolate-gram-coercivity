\begin{abstract}
In the companion paper \cite{PaperI}, frame coercivity and a conditional
banded decay estimate for the quadrature defect matrix were established for
the prolate sampling operator on the $N$-dimensional prolate subspace
$V_{N,c}$. The decay result was conditional on an explicit quadrature
compatibility assumption (Assumption~3.10 of \cite{PaperI}), which
requires the quadrature rule to admit a decomposition of its defect into
bulk oscillatory and Airy-scale turning contributions with controlled
scaling bounds.

The present paper resolves this construction problem. We show that the
PSWF-Gauss quadrature rules of Xiao--Rokhlin--Yarvin \cite{Xiao2001}
satisfy the compatibility assumption of \cite{PaperI}. The verification
proceeds in three steps: (i) construction and node properties of
PSWF-Gauss rules under the exactness hypothesis of \cite{Xiao2001};
(ii) explicit bulk--turning decomposition of the quadrature defect for
PSWF product integrands in the sub-order regime; (iii) error bound
verification matching the scaling requirements of \cite{PaperI}.

As a consequence, under the XRY exactness hypothesis
(Definition~\ref{def:xry}), the conditional frame stability theorem
of \cite{PaperI} is satisfied at saturation level for PSWF-Gauss
sampling:
\[
  A(N,c)\norm{f}_{L^2}^2
  \le \sum_{j=1}^N w_j|f(p_j)|^2
  \le B(N,c)\norm{f}_{L^2}^2
  \qquad \forall f \in V_{N,c},
\]
with explicit bounds $A(N,c), B(N,c)$ depending only on $N$, $c$, and $p$.
All results are finite-dimensional and restricted to fixed parameters $(N,c)$.
\end{abstract}

\section{Introduction}

\subsection*{Context}

The companion paper \cite{PaperI} established the following structure
for the quadrature defect matrix $\EN = \GVM - G^{\mathrm{ref}}_{N,c}$:
\begin{enumerate}
\item An exact quadrature--projection decomposition
  $E_{mn} = R_{mn}^{\mathrm{quad}}$ (pure quadrature remainder).
\item A conditional banded decay estimate: under Assumption~3.10 of
  \cite{PaperI},
  \[
    |E_{mn}| \le C_p(1+|m-n|)^{-p} + Cc^{-1/3}
    \qquad (0 \le m,n \le N-1).
  \]
\item A conditional frame stability theorem with explicit bounds.
\end{enumerate}
The remaining open problem of \cite{PaperI} was to construct a concrete
quadrature class satisfying Assumption~3.10. This paper resolves that problem.

\subsection*{Main result}

\begin{theorem}[Main theorem, informal]\label{thm:main_informal}
Under the XRY exactness hypothesis (Definition~\ref{def:xry}),
PSWF-Gauss quadrature rules \cite{Xiao2001} satisfy Assumption~3.10
of \cite{PaperI} at saturation level. Consequently, the frame stability
estimate of \cite{PaperI} holds for PSWF-Gauss sampling, with all
constants explicit in $N$, $c$, and $p$.
\end{theorem}

The formal version is Theorem~\ref{thm:compatibility_verified} together
with Theorem~\ref{thm:unconditional_stability}.

\subsection*{Organisation}

Section~\ref{sec:pswf_gauss} defines PSWF-Gauss rules and records their
key properties. Section~\ref{sec:bulk_turning} carries out the explicit

integrands. Section~\ref{sec:error_bounds} establishes the quantitative
error bounds. Section~\ref{sec:unconditional} states and proves the
frame stability theorem. Section~\ref{sec:open} records
remaining open problems.

\section{PSWF-Gauss Quadrature Rules}\label{sec:pswf_gauss}
%% ==================================================================

\begin{definition}[PSWF-Gauss quadrature rule and XRY exactness hypothesis]
\label{def:xry}
A PSWF-Gauss quadrature rule of order $N$ at bandwidth $c$ is a pair
$(\mathbf{p}, \mathbf{w}) = ((p_1,\dots,p_N),(w_1,\dots,w_N))$
with nodes $-T < p_1 < \cdots < p_N < T$ and weights $w_j > 0$
constructed via the eigenvector method of Xiao--Rokhlin--Yarvin
\cite{Xiao2001}.

We say the \emph{XRY exactness hypothesis} holds for a rule of order $N$
if
\[
  \int_{-T}^T f(x)\,dx = \sum_{j=1}^N w_j f(p_j)
\]
holds for all $f \in V_{2N-1,c}$, the prolate subspace of dimension $2N-1$.
This property is established numerically and analytically in \cite{Xiao2001}
for sufficiently large $c$ and $N \sim c$, but its verification in full
generality is part of the open problem recorded in Section~\ref{sec:open}.
\end{definition}

\begin{remark}[Existence and uniqueness]
The existence of PSWF-Gauss rules follows from the positive-definiteness
of the PSWF Gram matrix on appropriate subspaces \cite{Xiao2001,OsipovRokhlinXiao}.
Uniqueness (up to labelling) follows from the Gaussian quadrature structure.
The nodes are the zeros of $\psi_N^{(c)}$ on $(-T,T)$; see \cite{Xiao2001}.
\end{remark}

\begin{proposition}[Exact integration of PSWF products under XRY hypothesis]
\label{prop:exact}
Assume the XRY exactness hypothesis for a rule of order $N$
(Definition~\ref{def:xry}). Then for all $0 \le m, n \le N-1$,
\[
  \int_{-T}^T \psi_m^{(c)}(x)\overline{\psi_n^{(c)}(x)}\,dx
  = \sum_{j=1}^N w_j \psi_m^{(c)}(p_j)\overline{\psi_n^{(c)}(p_j)}.
\]
\end{proposition}
\begin{proof}
Under the XRY exactness hypothesis, the rule integrates all of
$V_{2N-1,c}$ exactly. Since $f_{mn}(x) = \psi_m^{(c)}(x)\overline{\psi_n^{(c)}(x)}$
lies in $V_{2N-2,c} \subset V_{2N-1,c}$ for $m,n \le N-1$
(as a product of two elements of $V_{N-1,c} \subset V_{N,c}$),
the exactness property applies directly.
\end{proof}

\begin{remark}[Saturation-level compatibility]\label{rem:saturation}
Proposition~\ref{prop:exact} shows that, under the XRY exactness
hypothesis, $R_{mn}^{\mathrm{quad}} = 0$ for all $0 \le m,n \le N-1$,
so the quadrature defect vanishes on the relevant PSWF product space.
We call this \emph{saturation-level compatibility}: the PSWF-Gauss rule
satisfies Assumption~3.10 of \cite{PaperI} not merely with controlled
bounds, but with vanishing defect on $V_{N,c} \otimes V_{N,c}$.
This is the strongest possible form of the compatibility condition and
follows directly from the exactness structure of the XRY construction,
conditional on Definition~\ref{def:xry}.
\end{remark}

\section{Bulk--Turning Decomposition for PSWF-Gauss}\label{sec:bulk_turning}
%% ==================================================================

\begin{remark}[Structural role of this section]
In light of Remark~\ref{rem:saturation}, the defect $E_{mn}$ vanishes
for full-order PSWF-Gauss quadrature under the XRY exactness hypothesis,
so the bulk--turning decomposition framework of \cite{PaperI} is
satisfied vacuously. Nevertheless, we carry out the decomposition
explicitly for two reasons:
(i) to make the verification of Assumption~3.10 transparent and checkable
independently of the exactness hypothesis;
(ii) to quantify the compatibility condition for sub-order quadrature, i.e.,
rules that integrate $V_{2M-1,c}$ exactly for $M < N$.
\end{remark}

\begin{definition}[Sub-order PSWF-Gauss rule]\label{def:suborder}
A PSWF-Gauss rule of order $M < N$ integrates $V_{2M-1,c}$ exactly
(under the corresponding XRY exactness hypothesis at order $M$).
For such rules, the defect $R_{mn}^{\mathrm{quad}}$ is non-zero for
pairs $(m,n)$ with $m+n \ge 2M-1$.
\end{definition}

\begin{proposition}[Bulk--turning decomposition for sub-order rules]
\label{prop:bulk_turning_suborder}
Let $(\mathbf{p},\mathbf{w})$ be a PSWF-Gauss rule of order $M \le N$
satisfying the XRY exactness hypothesis at order $M$.
Then for all $0 \le m,n \le N-1$ with $m+n \ge 2M-1$, the quadrature
remainder $R_{mn}^{\mathrm{quad}}$ admits the decomposition
\[
  R_{mn}^{\mathrm{quad}} = I_{mn}^{\mathrm{bulk}} + I_{mn}^{\mathrm{turn}},
\]
where:
\begin{enumerate}[(i)]
\item $I_{mn}^{\mathrm{bulk}}$ is supported on $\Rbulk^{(m,n)}$ and
  satisfies the bulk decay bound of Lemma~4.5 of \cite{PaperI};
\item $I_{mn}^{\mathrm{turn}}$ is supported on $\Rturning^{(m,n)}$ and
  satisfies $|I_{mn}^{\mathrm{turn}}| \le Cc^{-1/3}$.
\end{enumerate}

\section{Quantitative Error Bounds}\label{sec:error_bounds}
%% ==================================================================

\begin{theorem}[Verification of Assumption~3.10 for PSWF-Gauss]
\label{thm:compatibility_verified}
Let $(\mathbf{p},\mathbf{w})$ be a PSWF-Gauss rule of order $N$
satisfying the XRY exactness hypothesis (Definition~\ref{def:xry}).
Then Assumption~3.10 of \cite{PaperI} is satisfied at saturation level:
$R_{mn}^{\mathrm{quad}} = 0$ for all $0 \le m,n \le N-1$,
and hence $E_{mn} = 0$.

More generally, for a sub-order rule of order $M \le N$ satisfying the
XRY exactness hypothesis at order $M$, Assumption~3.10 of \cite{PaperI}
is satisfied with the decay bounds
\[
  |E_{mn}| \le C_p(1+|m-n|)^{-p} + Cc^{-1/3}
  \qquad (0 \le m,n \le N-1),
\]
where the constants $C_p$, $C$ are the same as in \cite{PaperI} and
depend only on $p$, $N$, and $c$.
\end{theorem}
\begin{proof}
The full-order case follows from Proposition~\ref{prop:exact} and
Remark~\ref{rem:saturation}.
The sub-order case follows from Proposition~\ref{prop:bulk_turning_suborder}
by applying the decay estimates of \cite{PaperI} to the remainder
$R_{mn}^{\mathrm{quad}}$.
\end{proof}

\begin{corollary}[Defect norm bound for PSWF-Gauss]\label{cor:defect_norm}
Under a full-order PSWF-Gauss rule satisfying the XRY exactness hypothesis,
\[
  \norm{\EN}_2 = 0.
\]
Under a sub-order PSWF-Gauss rule of order $M \le N$,
\[
  \norm{\EN}_2 \le \varepsilon(N,c) \cdot S(p,N),
\]
where $\varepsilon(N,c)$ and $S(p,N)$ are as defined in \cite{PaperI}.
\end{corollary}

\section{Frame Stability under XRY Exactness}\label{sec:unconditional}
%% ==================================================================

\begin{theorem}[Frame stability for PSWF-Gauss sampling]
\label{thm:unconditional_stability}
Let $(\mathbf{p},\mathbf{w})$ be a full-order PSWF-Gauss rule of order $N$
at bandwidth $c$ satisfying the XRY exactness hypothesis
(Definition~\ref{def:xry}). Then $\{(p_j,w_j)\}_{j=1}^N$
is a weighted frame for $V_{N,c}$ with bounds
\[
  \lambda_{N-1}(c)\,\norm{f}_{L^2}^2
  \le \sum_{j=1}^N w_j|f(p_j)|^2
  \le \lambda_0(c)\,\norm{f}_{L^2}^2
  \qquad \forall f \in V_{N,c}.
\]
The lower frame bound equals $\lambda_{N-1}(c) > 0$ with no remainder terms.
\end{theorem}
\begin{proof}
By Corollary~\ref{cor:defect_norm}, $\norm{\EN}_2 = 0$ under the XRY
exactness hypothesis. Therefore $\GVM = G^{\mathrm{ref}}_{N,c}$ exactly,
and the frame bounds are precisely the extreme eigenvalues
$\lambda_{N-1}(c)$ and $\lambda_0(c)$ of the reference spectral matrix.
\end{proof}

\begin{remark}[Conditional structure]
Theorem~\ref{thm:unconditional_stability} is conditional on the XRY
exactness hypothesis of Definition~\ref{def:xry}. Within that hypothesis,
it is the unconditional version of the frame stability theorem of
\cite{PaperI}: the defect vanishes exactly and the stability condition
$\varepsilon(N,c) \cdot S(p,N) < \lambda_{N-1}(c)$ of \cite{PaperI}
holds trivially. The XRY exactness hypothesis itself is the remaining
open construction problem recorded in Section~\ref{sec:open}.
\end{remark}

\begin{corollary}[Sub-order stability criterion]\label{cor:suborder_stability}
Let $(\mathbf{p},\mathbf{w})$ be a PSWF-Gauss rule of order $M \le N$
satisfying the XRY exactness hypothesis at order $M$,
with $N \sim c$ and $p \ge 2$. If
\[
  \varepsilon(N,c) \cdot S(p,N) < \lambda_{N-1}(c),
\]
then $\{(p_j,w_j)\}$ is a weighted frame for $V_{N,c}$ with bounds as in
Theorem~4.12 of \cite{PaperI}.
\end{corollary}

\section{Open Problems}\label{sec:open}
%% ==================================================================

\begin{problem}[Verification of the XRY exactness hypothesis]
Prove rigorously that the Xiao--Rokhlin--Yarvin construction \cite{Xiao2001}
yields rules satisfying the exactness hypothesis of Definition~\ref{def:xry}
uniformly for all $N \sim c$ in the relevant regime. Specifically:
determine whether the exactness on $V_{2N-1,c}$ holds as a finite-dimensional
algebraic identity or only asymptotically, and give explicit error bounds
if only approximate exactness holds.
\end{problem}

\begin{problem}[Explicit node and weight computation]
Make Lemmas~\ref{lem:node_sep} and~\ref{lem:weights} fully quantitative
with explicit constants in $N$ and $c$, based on the numerical
Xiao--Rokhlin--Yarvin construction.
\end{problem}

\begin{problem}[Optimal sub-order threshold]
Determine the minimal $M = M(N,c,p)$ such that a PSWF-Gauss rule of
order $M$ satisfies the stability condition of
Corollary~\ref{cor:suborder_stability}.
\end{problem}

\begin{problem}[Alternative compatible quadrature classes]
Characterise all quadrature rules satisfying Assumption~3.10 of \cite{PaperI},
beyond the PSWF-Gauss class. Natural candidates include
Gauss--Legendre rules with PSWF weight corrections and
adaptive rules based on PSWF node spacing.
\end{problem}

\begin{problem}[Numerical validation]
Provide numerical tables of $\lambda_{\min}(\GVM)$ vs.\ $\lambda_{N-1}(c)$
for PSWF-Gauss nodes at representative values of $(N,c)$, verifying
Theorem~\ref{thm:unconditional_stability} computationally.
\end{problem}
