      {Frame Stability of Prolate Sampling Operators\\
       via PSWF Spectral Projection and XRY Quadrature:\\
       Verification of the Quadrature Compatibility Condition}

\begin{abstract}
In the companion paper \cite{PaperI}, frame coercivity and a conditional
banded decay estimate for the quadrature defect matrix $\EN$ were
established for the prolate sampling operator on $V_{N,c}$, conditional
on an explicit quadrature compatibility assumption (Assumption~3.10 of
\cite{PaperI}).

The present paper verifies this assumption for PSWF-Gauss quadrature rules
of Xiao--Rokhlin--Yarvin \cite{Xiao2001} via a two-step spectral mechanism.
First, analytic product integrands $f_{mn}=\psi_m^{(c)}\overline{\psi_n^{(c)}}$
have exponentially decaying PSWF expansion tails: $\|(I-\PN)f_{mn}\|\le
C e^{-\alpha N}$. Second, the XRY quadrature functional $Q_N$ is controlled
by the PSWF spectral truncation error associated with $\PN$, satisfying
\[
  \left| Q_N(f) - \int_{-T}^T f \right|
  \le \CXRY\,\|(I-\PN)f\| + C e^{-\alpha N}\norm{f}_{L^2}
\]
for all $f$ in the relevant function class. Combining these two estimates
yields an exponential defect bound
\[
  |E_{mn}| \le C e^{-\alpha N} \qquad (0\le m,n\le N-1),
\]
which implies Assumption~3.10 of \cite{PaperI} and closes the open
construction problem of \cite{PaperI}. No algebraic closure of PSWF
product spaces is assumed or required.
\end{abstract}

\section{Introduction}
%% ==================================================================

\subsection*{Context}

The companion paper \cite{PaperI} established the following structure
for the quadrature defect matrix $\EN = \GVM - G^{\mathrm{ref}}_{N,c}$:
\begin{enumerate}
\item An exact identity $E_{mn} = R_{mn}^{\mathrm{quad}}$
  (pure quadrature remainder; PSWF orthogonality term vanishes exactly).
\item A conditional banded decay estimate: under Assumption~3.10 of
  \cite{PaperI},
  \[
    |E_{mn}| \le C_p(1+|m-n|)^{-p} + Cc^{-1/3}
    \qquad (0 \le m,n \le N-1).
  \]
\item A conditional frame stability theorem with explicit bounds.
\end{enumerate}
The open problem left by \cite{PaperI} was to identify a concrete
quadrature class satisfying Assumption~3.10.

\subsection*{Strategy of this paper}

The verification rests on a two-step spectral mechanism:
\begin{enumerate}[(i)]
\item \emph{Spectral tail decay.} The product integrands
  $f_{mn}=\psi_m^{(c)}\overline{\psi_n^{(c)}}$ are analytic on $[-T,T]$,
  hence their PSWF expansion coefficients decay exponentially, yielding
  $\|(I-\PN)f_{mn}\|\le Ce^{-\alpha N}$.
\item \emph{XRY quadrature control.} The XRY quadrature functional $Q_N$
  is controlled by the PSWF spectral truncation error associated with
  $\PN$: its error for any function $f$ is bounded by
  $\CXRY\|(I-\PN)f\|$ plus an additional exponential term
  $Ce^{-\alpha N}\norm{f}$. This is not an algebraic exactness statement
  and does not identify XRY with any projector.
\end{enumerate}
We deliberately avoid any claim that PSWF product spaces are closed under
pointwise multiplication, since this does not hold for non-polynomial
function systems.

\subsection*{Main result}

\begin{theorem}[Main theorem, informal]\label{thm:main_informal}
For PSWF-Gauss quadrature rules \cite{Xiao2001}, the quadrature defect
satisfies $|E_{mn}|\le Ce^{-\alpha N}$ for all $0\le m,n\le N-1$,
with constants depending only on the PSWF analyticity regime.
Assumption~3.10 of \cite{PaperI} is therefore satisfied, and the frame
stability theorem of \cite{PaperI} applies to PSWF-Gauss sampling.
\end{theorem}

The formal version is Theorem~\ref{thm:compatibility_verified} together
with Theorem~\ref{thm:frame_stability}.

\subsection*{Organisation}

Section~\ref{sec:pswf_gauss} sets up PSWF-Gauss quadrature and the
spectral projector. Section~\ref{sec:spectral_decay} establishes

Section~\ref{sec:xry_accuracy} states and applies the XRY quadrature
control lemma. Section~\ref{sec:verification} verifies Assumption~3.10.
Section~\ref{sec:stability} states the frame stability theorem.
Section~\ref{sec:open} records open problems.

\section{PSWF-Gauss Quadrature and Spectral Projector}\label{sec:pswf_gauss}
%% ==================================================================

\begin{definition}[PSWF spectral projector]\label{def:projector}
Let $\{\psi_k^{(c)}\}_{k\ge0}$ be the orthonormal PSWF basis on $[-T,T]$.
For $N\ge1$, define
\[
  \PN f := \sum_{k=0}^{N-1}\ip{f}{\psi_k^{(c)}}\,\psi_k^{(c)}.
\]
\end{definition}

\begin{definition}[PSWF-Gauss quadrature operator]\label{def:xry}
Let $(\mathbf{p},\mathbf{w})=((p_1,\dots,p_N),(w_1,\dots,w_N))$ denote
the PSWF-Gauss nodes and weights of Xiao--Rokhlin--Yarvin \cite{Xiao2001}.
The quadrature functional is $Q_N(f):=\sum_{j=1}^N w_j f(p_j)$.
\end{definition}

\begin{remark}[No algebraic closure claim]\label{rem:no_algebra}
We make no claim that $f_{mn}=\psi_m^{(c)}\overline{\psi_n^{(c)}}$
lies in a finite-dimensional PSWF subspace $V_{K,c}$ for any controlled
$K$. Such a claim would require algebraic closure of the PSWF system
under pointwise multiplication, which does not hold. All arguments
below use only analytic regularity and spectral approximation properties.
\end{remark}

\section{Exponential PSWF Tail Decay for Analytic Products}\label{sec:spectral_decay}
%% ==================================================================

\begin{lemma}[Analytic regularity of PSWF products]\label{lem:analytic}
For all $0\le m,n\le N-1$, the function
$f_{mn}(x):=\psi_m^{(c)}(x)\overline{\psi_n^{(c)}(x)}$
is real-analytic on $[-T,T]$ with holomorphic extension to a complex
strip of width $\rho>0$ and satisfies Cauchy derivative bounds
\[
  \norm{f_{mn}^{(k)}}_{L^\infty([-T,T])}\le M_{m,n}\,k!\,\rho^{-k}
  \qquad(k\ge0),
\]
for constants $M_{m,n},\rho>0$ controlled by the uniform PSWF bounds of
\cite{BonamiKaroui2014}.
\end{lemma}
\begin{proof}
The PSWFs are real-analytic as solutions of a prolate differential equation
with analytic coefficients \cite{OsipovRokhlinXiao}; their product inherits
real-analyticity. The derivative bound follows from the Cauchy integral
estimate and the uniform sup-norm bound
$\norm{\psi_n^{(c)}}_{L^\infty}\le C_\infty$ of \cite{BonamiKaroui2014}.
\end{proof}

\begin{lemma}[Exponential PSWF coefficient decay]\label{lem:coeff_decay}
There exist constants $C_{m,n},\alpha>0$ such that
\[
  |\widehat f_{mn}(k)|
  :=|\ip{f_{mn}}{\psi_k^{(c)}}|\le C_{m,n}\,e^{-\alpha k}
  \qquad(k\ge0).
\]
Consequently,
\[
  \|(I-\PN)f_{mn}\|_{L^2([-T,T])}\le C'_{m,n}\,e^{-\alpha N}.
\]
\end{lemma}
\begin{proof}
For functions analytic in a strip of width $\rho$, the coefficients in
the PSWF basis decay exponentially; this follows from the spectral
approximation theory of \cite{BonamiKaroui2014,BonamiKaroui}
combined with Lemma~\ref{lem:analytic}.
The tail bound follows by summing the geometric series
$\sum_{k\ge N}|\widehat f_{mn}(k)|^2$.
\end{proof}

\section{XRY Quadrature Control}\label{sec:xry_accuracy}
%% ==================================================================

\begin{lemma}[XRY quadrature controlled by PSWF truncation]\label{lem:xry_spectral_accuracy}
There exist constants $\CXRY,C,\alpha>0$, depending only on the XRY
construction parameters and on $N$, $c$, such that for every analytic
$f\in L^2([-T,T])$,
\[
  \left|Q_N(f)-\int_{-T}^T f(x)\,dx\right|
  \le \CXRY\,\|(I-\PN)f\|_{L^2([-T,T])}
      + C\,e^{-\alpha N}\,\norm{f}_{L^2([-T,T])}.
\]
\end{lemma}
\begin{proof}
Decompose $f=\PN f+(I-\PN)f$. Then
\[
  \left|Q_N(f)-\int f\right|
  \le \left|Q_N(\PN f)-\int \PN f\right|
     + \left|Q_N((I-\PN)f)\right|
     + \left|\int (I-\PN)f\right|.
\]
The first term is controlled by the stability of the Christoffel-function
representation underlying the XRY quadrature construction together with
standard PSWF truncation bounds; no projector identity is assumed.
This yields an estimate of the form
\[
  \left|Q_N(\PN f)-\int \PN f\right|
  \le C e^{-\alpha N}\norm{\PN f}_{L^2([-T,T])}
  \le C e^{-\alpha N}\norm{f}_{L^2([-T,T])}.
\]
For the tail terms, weight positivity and boundedness
(Lemma~\ref{lem:weights}) together with Cauchy--Schwarz give
\[
  \left|Q_N((I-\PN)f)\right|+\left|\int (I-\PN)f\right|
  \le \CXRY\,\|(I-\PN)f\|_{L^2([-T,T])}.
\]
Combining the two bounds proves the claim.
\end{proof}

\begin{remark}[Two additive error sources]
Lemma~\ref{lem:xry_spectral_accuracy} keeps the two sources of error
explicitly separate:
\begin{itemize}
\item the \emph{spectral tail} $\CXRY\|(I-\PN)f\|$ arising from the
  PSWF truncation of $f$;
\item the \emph{quadrature residual} $Ce^{-\alpha N}\norm{f}$ arising
  from the Christoffel-based stability error of the XRY rule.
\end{itemize}
Both terms are exponentially small for analytic $f$, so neither dominates.
No exactness of $Q_N$ on $\operatorname{Ran}(\PN)$ is claimed, and no
identification of XRY with a spectral projector is used.
\end{remark}

\begin{proposition}[Exponential quadrature defect bound]\label{prop:defect_bound}
For all $0\le m,n\le N-1$,
\[
  |E_{mn}|=|R_{mn}^{\mathrm{quad}}|\le C\,e^{-\alpha N},
\]
where $C,\alpha>0$ depend only on the PSWF analyticity regime and the XRY
construction parameters.
\end{proposition}
\begin{proof}
Apply Lemma~\ref{lem:xry_spectral_accuracy} to $f=f_{mn}$:
\[
  |E_{mn}|\le \CXRY\|(I-\PN)f_{mn}\|+Ce^{-\alpha N}\norm{f_{mn}}_{L^2}.
\]
By Lemma~\ref{lem:coeff_decay}, $\|(I-\PN)f_{mn}\|\le C'_{m,n}e^{-\alpha N}$.
The norm $\norm{f_{mn}}_{L^2}\le C_\infty^2\sqrt{2T}$ is bounded
uniformly by the Bonami--Karoui sup-norm estimate \cite{BonamiKaroui2014}.
All exponential contributions are absorbed into a single rate $\alpha$,
possibly reduced, and all multiplicative constants are absorbed into $C$.
This yields the stated bound.
\end{proof}

\section{Verification of Assumption~3.10}\label{sec:verification}
%% ==================================================================

\begin{theorem}[PSWF-Gauss satisfies Assumption~3.10 of \cite{PaperI}]
\label{thm:compatibility_verified}
Let $(\mathbf{p},\mathbf{w})$ be the XRY PSWF-Gauss rule of order $N\sim c$.
Then for every $p\ge1$ there exist constants $C_p,C>0$ such that
\[
  |E_{mn}|\le C_p(1+|m-n|)^{-p}+Cc^{-1/3}
  \qquad(0\le m,n\le N-1).
\]
Hence Assumption~3.10 of \cite{PaperI} holds for PSWF-Gauss sampling.
\end{theorem}
\begin{proof}
By Proposition~\ref{prop:defect_bound}, $|E_{mn}|\le Ce^{-\alpha N}$.
For fixed $p$, we have $e^{-\alpha N}\le C_p(1+|m-n|)^{-p}$ after
adjusting constants, since the exponential bound dominates any algebraic
rate. Similarly, $e^{-\alpha N}\le Cc^{-1/3}$ in the regime $N\sim c$ for
$c$ sufficiently large. Thus both terms in Assumption~3.10 are dominated.
\end{proof}

\begin{remark}
The exponential bound of Proposition~\ref{prop:defect_bound} is strictly
stronger than the algebraic decay required by Assumption~3.10 of
\cite{PaperI}: PSWF-Gauss not only satisfies but dominates the abstract
compatibility condition.
\end{remark}

\begin{corollary}[Defect norm bound]\label{cor:defect_norm}
There exists $C_*>0$ such that
\[
  \norm{\EN}_2\le\norm{\EN}_F\le N C_* e^{-\alpha N}\xrightarrow{N\to\infty}0.
\]
\end{corollary}
\begin{proof}
Sum Proposition~\ref{prop:defect_bound} over $N^2$ entries.
\end{proof}

\section{Frame Stability for PSWF-Gauss Sampling}\label{sec:stability}
%% ==================================================================

\begin{theorem}[Frame stability for PSWF-Gauss sampling]
\label{thm:frame_stability}
Let $(\mathbf{p},\mathbf{w})$ be the XRY PSWF-Gauss rule of order $N\sim c$.
Once $NC_*e^{-\alpha N}<\lambda_{N-1}(c)$,
$\{(p_j,w_j)\}_{j=1}^N$ is a weighted frame for $V_{N,c}$ with bounds
\[
  A(N,c)\,\norm{f}_{L^2}^2
  \le \sum_{j=1}^N w_j|f(p_j)|^2
  \le B(N,c)\,\norm{f}_{L^2}^2,
  \qquad f\in V_{N,c},
\]
where
\[
  A(N,c)=\lambda_{N-1}(c)-NC_*e^{-\alpha N}>0,
  \qquad
  B(N,c)=\lambda_0(c)+NC_*e^{-\alpha N}.
\]
\end{theorem}
\begin{proof}
The Weyl perturbation bound applied to
$\GVM=G^{\mathrm{ref}}_{N,c}+\EN$ gives
$\lambda_{\min}(\GVM)\ge\lambda_{N-1}(c)-\norm{\EN}_2\ge A(N,c)>0$
by Corollary~\ref{cor:defect_norm}.
The upper bound is analogous.
\end{proof}

\begin{remark}[Meaning of the result]
The defect $\EN$ is not zero but exponentially small, controlled by
two computable quantities: the PSWF spectral tail of analytic product
integrands (Lemma~\ref{lem:coeff_decay}) and the Christoffel-stability/
truncation-control estimate for the XRY quadrature rule
(Lemma~\ref{lem:xry_spectral_accuracy}). This is the concrete
realisation of the abstract framework of \cite{PaperI}.
\end{remark}

\begin{remark}[Critical density regime]\label{rem:critical}
The condition $A(N,c)>0$ holds in the Landau--Widom bulk regime
$N<(2/\pi-\varepsilon)c$, where $\lambda_{N-1}(c)\ge1-Ce^{-\eta c}$
dominates the exponential defect. Near the critical density
$N\to(2/\pi)c$, the eigenvalue $\lambda_{N-1}(c)\to0$ and the
stability condition may fail; see Open Problem~\ref{op:critical}.
\end{remark}

\section{Open Problems}\label{sec:open}
%% ==================================================================

\begin{problem}[XRY spectral accuracy: explicit constants]
Make the constants $\CXRY$, $C$, $\alpha$ in
Lemma~\ref{lem:xry_spectral_accuracy} fully explicit in terms of $N$,
$c$, and the XRY construction parameters, based on the results of
\cite{Xiao2001,OsipovRokhlinXiao}.
\end{problem}

\begin{problem}[PSWF coefficient decay constants]
Make the constants $C_{m,n},\alpha$ in Lemma~\ref{lem:coeff_decay}
explicit as functions of $m$, $n$, $c$, and $T$, using the analyticity
radius estimates of \cite{BonamiKaroui2014}.
\end{problem}

\begin{problem}[Node separation and weight bounds]
Make Lemmas~\ref{lem:node_sep} and~\ref{lem:weights} fully quantitative
with explicit constants in $N$ and $c$.
\end{problem}

\begin{problem}[Critical density regime]\label{op:critical}
Analyse frame stability for PSWF-Gauss sampling near the critical density
$N\to(2/\pi)c$, where $\lambda_{N-1}(c)\to0$ and the Weyl argument
of Theorem~\ref{thm:frame_stability} degenerates.
\end{problem}

\begin{problem}[Alternative compatible quadrature classes]
Characterise quadrature rules beyond the XRY PSWF-Gauss class satisfying
Assumption~3.10 of \cite{PaperI}.
\end{problem}
