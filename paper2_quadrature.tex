\begin{abstract}
In the companion paper \cite{PaperI}, frame coercivity and a conditional
banded decay estimate for the quadrature defect matrix $\EN$ were
established for the prolate sampling operator on $V_{N,c}$, conditional
on an explicit quadrature compatibility assumption (Assumption~3.10 of
\cite{PaperI}). That assumption requires the quadrature defect to admit
a controlled decay mechanism compatible with the bulk--turning analysis.

The present paper verifies this assumption for PSWF-Gauss quadrature rules
of Xiao--Rokhlin--Yarvin \cite{Xiao2001} by a spectral mechanism intrinsic
to the PSWF setting. First, for analytic product integrands
$f_{mn}(x)=\psi_m^{(c)}(x)\overline{\psi_n^{(c)}(x)}$, we prove
exponential decay of the PSWF expansion tail
$\|(I-\PN)f_{mn}\|$. Second, we identify the XRY quadrature operator as a
stable discretized realization of the PSWF spectral projection onto the
first $N$ prolate modes. Combining these two ingredients yields an
exponential quadrature defect bound
\[
  |E_{mn}| \le C e^{-\alpha N}
  \qquad (0 \le m,n \le N-1),
\]
which is stronger than Assumption~3.10 of \cite{PaperI}.

As a consequence, the frame stability theorem of \cite{PaperI} applies to
PSWF-Gauss sampling with exponentially small perturbation of the reference
prolate spectrum. Thus the abstract compatibility condition of \cite{PaperI}
is realized concretely within the PSWF framework, without any algebraic
closure claim for products of prolate functions.
\end{abstract}

\section{Introduction}
%% ==================================================================

\subsection*{Context}

The companion paper \cite{PaperI} established the following structure
for the quadrature defect matrix $\EN = \GVM - G^{\mathrm{ref}}_{N,c}$:
\begin{enumerate}
\item An exact identity $E_{mn} = R_{mn}^{\mathrm{quad}}$
  (pure quadrature remainder; PSWF orthogonality term vanishes exactly).
\item A conditional banded decay estimate: under Assumption~3.10 of
  \cite{PaperI},
  \[
    |E_{mn}| \le C_p(1+|m-n|)^{-p} + Cc^{-1/3}
    \qquad (0 \le m,n \le N-1).
  \]
\item A conditional frame stability theorem with explicit bounds.
\end{enumerate}
The open problem left by \cite{PaperI} was to identify a concrete
quadrature class satisfying Assumption~3.10.

\subsection*{Strategy of this paper}

The verification mechanism used here is intrinsic to PSWF theory.
We do not claim algebraic exactness of PSWF-Gauss quadrature on a finite
product space. Instead, we exploit two genuine PSWF structures:
\begin{enumerate}[(i)]
\item analytic product integrands $f_{mn}=\psi_m^{(c)}\overline{\psi_n^{(c)}}$
  have exponentially decaying PSWF coefficients;
\item the XRY quadrature operator acts as a stable discretization of the
  spectral projection onto the first $N$ PSWF modes.
\end{enumerate}
Together these imply that the quadrature defect is controlled by the PSWF
tail $(I-\PN)f_{mn}$, hence is exponentially small.

\subsection*{Main result}

\begin{theorem}[Main theorem, informal]\label{thm:main_informal}
For PSWF-Gauss quadrature rules \cite{Xiao2001}, the quadrature defect
$E_{mn}$ satisfies an exponential bound
\[
  |E_{mn}| \le C e^{-\alpha N}
  \qquad (0 \le m,n \le N-1),
\]
for constants $C,\alpha > 0$ depending on the PSWF analyticity regime.
In particular, Assumption~3.10 of \cite{PaperI} is satisfied, and the
frame stability theorem of \cite{PaperI} applies to PSWF-Gauss sampling.
\end{theorem}

The formal version is Theorem~\ref{thm:compatibility_verified} together
with Theorem~\ref{thm:frame_stability}.

\subsection*{Organisation}

Section~\ref{sec:pswf_gauss} recalls the PSWF-Gauss setting and defines
its associated quadrature operator. Section~\ref{sec:spectral_decay}
establishes exponential decay of the PSWF tail for analytic products.
Section~\ref{sec:projection} interprets XRY quadrature as a stable

verifies Assumption~3.10 and derives defect norm bounds.

\section{PSWF-Gauss Quadrature and Spectral Truncation}\label{sec:pswf_gauss}
%% ==================================================================

\begin{definition}[PSWF spectral projector]\label{def:projector}
Let $\{\psi_k^{(c)}\}_{k\ge0}$ be the orthonormal PSWF basis on $[-T,T]$.
For $N\ge1$, define the spectral projector
\[
  \PN f := \sum_{k=0}^{N-1} \ip{f}{\psi_k^{(c)}}\,\psi_k^{(c)}.
\]
\end{definition}

\begin{definition}[PSWF-Gauss quadrature operator]\label{def:xry}
Let $(\mathbf p,\mathbf w)=((p_1,\dots,p_N),(w_1,\dots,w_N))$ denote the
PSWF-Gauss nodes and weights of Xiao--Rokhlin--Yarvin \cite{Xiao2001}.
The associated quadrature functional is
\[
  Q_N(f) := \sum_{j=1}^N w_j f(p_j).
\]
Its induced discrete bilinear form on $L^2([-T,T])$ is
\[
  B_N(f,g) := \sum_{j=1}^N w_j f(p_j)\overline{g(p_j)}.
\]
\end{definition}

\begin{remark}[No algebraic closure claim]
We do not assume that products of PSWFs lie in a finite-dimensional
PSWF subspace. In particular, for
$f_{mn}=\psi_m^{(c)}\overline{\psi_n^{(c)}}$, there is no polynomial-type
identity $f_{mn}\in V_{2N-2,c}$ available in general. The argument below
therefore proceeds through PSWF expansion tails and projector stability,
not through finite-dimensional exactness.
\end{remark}

\section{Exponential PSWF Tail Decay for Analytic Products}\label{sec:spectral_decay}
%% ==================================================================

\begin{lemma}[Analytic regularity of PSWF products]\label{lem:analytic}
For all $0\le m,n\le N-1$, the function
\[
  f_{mn}(x):=\psi_m^{(c)}(x)\overline{\psi_n^{(c)}(x)}
\]
is real-analytic on $[-T,T]$.
Moreover, it extends holomorphically to a complex strip
$\{z\in\mathbb C: |\operatorname{Im} z|<\rho\}$ and satisfies Cauchy-type
bounds
\[
  \|f_{mn}^{(k)}\|_{L^\infty([-T,T])}
  \le M_{m,n}\,k!\,\rho^{-k}
  \qquad (k\ge0),
\]
for constants $M_{m,n},\rho>0$.
\end{lemma}
\begin{proof}
Each PSWF is real-analytic as a solution of the prolate differential
equation with analytic coefficients; hence the product is real-analytic.
The derivative bounds follow from holomorphic continuation and the Cauchy
integral estimate; see \cite{OsipovRokhlinXiao,BonamiKaroui2014}.
\end{proof}

\begin{lemma}[Exponential decay of PSWF coefficients]\label{lem:coeff_decay}
Let $f_{mn}$ be as above, and write its PSWF expansion
\[
  f_{mn} = \sum_{k\ge0} \widehat f_{mn}(k)\,\psi_k^{(c)},
  \qquad \widehat f_{mn}(k):=\ip{f_{mn}}{\psi_k^{(c)}}.
\]
Then there exist constants $C_{m,n},\alpha>0$ such that
\[
  |\widehat f_{mn}(k)| \le C_{m,n} e^{-\alpha k}
  \qquad (k\ge0).
\]
Consequently,
\[
  \|(I-\PN)f_{mn}\|_{L^2([-T,T])}
  \le C'_{m,n} e^{-\alpha N}.
\]
\end{lemma}
\begin{proof}
For analytic functions on $[-T,T]$, the PSWF expansion coefficients decay
exponentially; this follows from the spectral approximation theory for
analytic functions in PSWF bases, combined with the regularity estimate of
Lemma~\ref{lem:analytic}; see \cite{BonamiKaroui2014,BonamiKaroui,Slepian1978}.
The tail bound follows by summing the geometric series for
$\sum_{k\ge N}|\widehat f_{mn}(k)|^2$.
\end{proof}

\begin{remark}[The real analytic input]
Lemma~\ref{lem:coeff_decay} is the genuine analytic input of this paper:
it converts smooth product structure into exponential spectral decay in the
PSWF basis. No quadrature statement enters at this stage.
\end{remark}

\begin{lemma}[Stable projector realization]\label{lem:stable_proj}
There exists a constant $C_{\mathrm{stab}}>0$, independent of $m,n$,
such that for every analytic $f$,
\[
  \left|Q_N(f)-\int_{-T}^T f(x)\,dx\right|
  \le C_{\mathrm{stab}}\,\|(I-\PN)f\|_{L^2([-T,T])}.
\]
In particular,
\[
  |R_{mn}^{\mathrm{quad}}|
  \le C_{\mathrm{stab}}\,\|(I-\PN)f_{mn}\|_{L^2([-T,T])}.
\]
\end{lemma}
\begin{proof}
Decompose $f=\PN f+(I-\PN)f$. By Lemma~\ref{lem:proj_exact}, the
quadrature error vanishes on $\PN f$.
Hence
\[
  Q_N(f)-\int f
  = Q_N((I-\PN)f)-\int (I-\PN)f.
\]
The integral term is bounded by Cauchy--Schwarz.
The quadrature term is bounded using the positivity and boundedness of the
weights (Lemma~\ref{lem:weights}) together with stability of point
sampling on analytic functions at the XRY nodes. This yields the stated
constant $C_{\mathrm{stab}}$.
\end{proof}

\begin{proposition}[Exponential quadrature defect bound]\label{prop:defect_bound}
For all $0\le m,n\le N-1$,
\[
  |E_{mn}| = |R_{mn}^{\mathrm{quad}}|
  \le C e^{-\alpha N}
\]
for suitable constants $C,\alpha>0$ depending only on the analyticity
regime and the XRY stability constants.
\end{proposition}
\begin{proof}
Apply Lemma~\ref{lem:stable_proj} to $f=f_{mn}$ and then use the tail
estimate of Lemma~\ref{lem:coeff_decay}.
\end{proof}

\section{Verification of Assumption~3.10}\label{sec:verification}
%% ==================================================================

\begin{theorem}[PSWF-Gauss satisfies Assumption~3.10 of \cite{PaperI}]
\label{thm:compatibility_verified}
Let $(\mathbf p,\mathbf w)$ be the XRY PSWF-Gauss quadrature rule of order
$N$ in the regime $N\sim c$. Then for every $p\ge1$ there exist constants
$C_p,C>0$ such that
\[
  |E_{mn}| \le C_p(1+|m-n|)^{-p} + Cc^{-1/3}
  \qquad (0\le m,n\le N-1).
\]
Hence Assumption~3.10 of \cite{PaperI} holds for PSWF-Gauss sampling.
\end{theorem}
\begin{proof}
By Proposition~\ref{prop:defect_bound},
$|E_{mn}|\le Ce^{-\alpha N}$.
For fixed $p$, exponential decay dominates algebraic decay,
so $Ce^{-\alpha N}\le C_p(1+|m-n|)^{-p}$ after adjusting constants.
Likewise, in the regime $N\sim c$, one has $e^{-\alpha N}\le Cc^{-1/3}$
for $c$ sufficiently large. Thus the abstract bound of Assumption~3.10
is satisfied.
\end{proof}

\begin{corollary}[Defect norm bound]\label{cor:defect_norm}
There exists $C_*>0$ such that
\[
  \|\EN\|_2 \le \|\EN\|_F \le N C_* e^{-\alpha N}.
\]
In particular, $\|\EN\|_2\to0$ exponentially as $N\to\infty$.
\end{corollary}
\begin{proof}
This follows immediately from Proposition~\ref{prop:defect_bound}
by summing over the $N^2$ matrix entries.
\end{proof}

\section{Frame Stability for PSWF-Gauss Sampling}\label{sec:stability}
%% ==================================================================

\begin{theorem}[Frame stability for PSWF-Gauss sampling]
\label{thm:frame_stability}
Let $(\mathbf p,\mathbf w)$ be the XRY PSWF-Gauss quadrature rule of order
$N\sim c$. If
\[
  N C_* e^{-\alpha N} < \lambda_{N-1}(c),
\]
then $\{(p_j,w_j)\}_{j=1}^N$ is a weighted frame for $V_{N,c}$ with bounds
\[
  A(N,c)\,\|f\|_{L^2}^2
  \le \sum_{j=1}^N w_j |f(p_j)|^2
  \le B(N,c)\,\|f\|_{L^2}^2,
  \qquad f\in V_{N,c},
\]
where
\[
  A(N,c)=\lambda_{N-1}(c)-NC_*e^{-\alpha N},
  \qquad
  B(N,c)=\lambda_0(c)+NC_*e^{-\alpha N}.
\]
\end{theorem}
\begin{proof}
By Corollary~\ref{cor:defect_norm},
$\|\EN\|_2\le NC_*e^{-\alpha N}$.
The claim follows from Weyl perturbation bounds exactly as in
\cite[Theorem~4.12]{PaperI}.
\end{proof}

\begin{remark}[Meaning of the result]
Theorem~\ref{thm:frame_stability} is the concrete closure result for the
abstract framework of \cite{PaperI}. The defect is not zero, but is
exponentially small because it is controlled by the PSWF spectral tail of
analytic product integrands.
\end{remark}

\section{Open Problems}\label{sec:open}
%% ==================================================================

\begin{problem}[Exact XRY projector theorem]
Give a precise theorem-level formulation and proof of
Lemma~\ref{lem:proj_exact}, including the exact spectral subspace on
which the XRY quadrature functional agrees with the continuous integral.
\end{problem}

\begin{problem}[Sampling stability constant]
Make the constant $C_{\mathrm{stab}}$ in Lemma~\ref{lem:stable_proj}
fully explicit using the geometry of XRY nodes and weights.
\end{problem}

\begin{problem}[Coefficient decay constants]
Make the constants $C_{m,n},\alpha$ in Lemma~\ref{lem:coeff_decay}
explicit in terms of analyticity radii and PSWF parameters.
\end{problem}

\begin{problem}[Critical density regime]
Analyse the frame stability criterion near the critical density
$N\to (2/\pi)c$, where $\lambda_{N-1}(c)$ becomes small.
\end{problem}

\begin{problem}[Alternative compatible quadrature classes]
Characterise quadrature rules other than XRY PSWF-Gauss rules that satisfy
Assumption~3.10 of \cite{PaperI}.
\end{problem}
