\begin{abstract}
Let $\{\psin\}_{n \ge 0}$ be the prolate spheroidal wave functions
(PSWFs) on $[-T,T]$ with bandwidth parameter $\omega > 0$ and
time-bandwidth product $c = \omega T$. Let $\PN$ denote the
orthogonal projector in $L^2([-T,T])$ onto the prolate subspace
$V_{N,c} = \mathrm{span}\{\psi_0^{(c)}, \ldots, \psi_{N-1}^{(c)}\}$.
The companion paper \cite{PaperII_quad} requires a uniform tail bound
for the product family
$\FN := \{\psi_m^{(c)}\overline{\psin} : 0 \le m,n \le N\}$
as a key input to the verification of the quadrature compatibility
condition of \cite{PaperI}.

The central tool in this paper is the \emph{Slepian energy identity}
(Lemma~\ref{lem:slepian}), which controls the spectral tail
$\|(I-\PN)f\|_{L^2([-T,T])}^2$ directly by the out-of-band energy
$\int_{|\xi|>\omega}|\hat{f}(\xi)|^2\,d\xi$ of the function $f$.
For the product family, the Fourier transform
$\widehat{f_{mn}} = \hat\psi_m^{(c)} * \hat\psi_n^{(c)}$
is a convolution supported in $[-2\omega, 2\omega]$; the out-of-band
energy in $|\xi| > \omega$ is controlled by the individual PSWF
leakage terms $1 - \lambda_m(c)$ and $1 - \lambda_n(c)$.

We prove:
\begin{enumerate}[(I)]
  \item \textbf{Bulk regime} ($m, n \le \gamma N$, fixed $\gamma < 1$):
  $\|(I-\PN)f_{mn}\|_{L^2} \le C_\gamma e^{-\alpha_\gamma N}$,
  because Landau--Widom asymptotics give
  $1 - \lambda_n(c) \le C e^{-\eta c}$ for $n \le \gamma N$,
  making the out-of-band energy exponentially small.
  \item \textbf{Off-diagonal regime} ($|m-n| \ge \delta N$, fixed
  $\delta > 0$): algebraic decay
  $\|(I-\PN)f_{mn}\|_{L^2} \le C_p(1+|m-n|)^{-p}$ for all $p \ge 1$,
  via oscillation cancellation in the Sturm--Liouville eigenvalue gap.
  \item \textbf{Edge regime obstruction} ($m, n \sim N$): both
  arguments fail. We identify the structural reason precisely: the
  PSWF leakage terms $1-\lambda_n(c)$ are not small near the Slepian
  cutoff, and bandwidth doubling prevents the Slepian identity from
  giving decay. This regime is formulated as an isolated open problem.
\end{enumerate}
No claim is made about zeros of $\zeta(s)$.
\end{abstract}

\section{Introduction}\label{sec:intro}

\subsection{Context and motivation}

The Prolate--Weil program \cite{PaperI,PaperII,PaperII_quad} seeks a
spectral realization of the Weil positivity criterion
\cite{Weil1952,CCM2025} via the prolate concentration operator
$\Kop$ and its associated Gram forms.
In \cite{PaperI}, the prolate Gram form
$Q^{(N)}(f) = \sum_{j=1}^N |f(p_j)|^2$ was shown to be coercive on
the prolate subspace $V_{N,c}$ under an explicit quadrature
compatibility condition (Assumption~3.10 of \cite{PaperI}).
The companion paper \cite{PaperII_quad} organized the verification
of that condition for XRY PSWF-Gauss quadrature into two conjectural
inputs; see Section~\ref{ssec:context} for a precise summary.

The first of these inputs is the PSWF product spectral tail conjecture:
for every $f_{mn} = \psi_m^{(c)}\overline{\psin} \in \FN$, one needs a
uniform bound on $\|(I-\PN)f_{mn}\|_{L^2([-T,T])}$.
This is the object of study in the present paper.

\subsection{Strategy: Fourier side via the Slepian energy identity}

The key insight is to work on the \emph{Fourier side} rather than
directly in $L^2([-T,T])$. The Slepian energy identity
(Lemma~\ref{lem:slepian}) states:
\[
  \|(I-\PN)f\|_{L^2([-T,T])}^2
  \le \frac{1}{1-\lambda_{N-1}(c)}
  \int_{|\xi|>\omega} |\hat{f}(\xi)|^2\,d\xi.
\]
This reduces the tail estimate to controlling the
\emph{out-of-band energy} $\Eout(f) := \int_{|\xi|>\omega}
|\hat{f}(\xi)|^2\,d\xi$ of $f$.

For the product $f_{mn} = \psi_m^{(c)}\overline{\psin}$, the
Fourier convolution theorem gives
$\widehat{f_{mn}} = \hat\psi_m^{(c)} * \hat\psi_n^{(c)}$.
Decomposing each factor as
$\hat\psi_k^{(c)} = g_k + r_k$ with
$g_k := \mathbf{1}_{[-\omega,\omega]}\hat\psi_k^{(c)}$ (the in-band part)
and $r_k := \mathbf{1}_{|\xi|>\omega}\hat\psi_k^{(c)}$ (the leak),
the out-of-band energy of the convolution is controlled by the
$L^2$-norms of the individual leaks:
\[
  \|r_k\|_{L^2}^2
  = \int_{|\xi|>\omega} |\hat\psi_k^{(c)}(\xi)|^2\,d\xi
  = 1 - \lambda_k(c).
\]
In the bulk $k \le \gamma N$, Landau--Widom asymptotics give
$1 - \lambda_k(c) \le Ce^{-\eta c}$, making all leak terms
exponentially small. This yields the bulk theorem.

\subsection{Main results}

\begin{theorem}[Bulk regime; Theorem~\ref{thm:bulk}]
For $m, n \le \gamma N$ with $\gamma \in (0,1)$ fixed and
$N \le (2/\pi-\varepsilon)c$:
\[
  \|(I-\PN)f_{mn}\|_{L^2([-T,T])} \le C_\gamma e^{-\alpha_\gamma N}.
\]
\end{theorem}

\begin{theorem}[Off-diagonal regime; Theorem~\ref{thm:offdiag}]
For $|m-n| \ge \delta N$ with $\delta > 0$ fixed and every $p \ge 1$:
\[
  \|(I-\PN)f_{mn}\|_{L^2([-T,T])} \le C_p(1+|m-n|)^{-p}.
\]
\end{theorem}

Together these give Corollary~\ref{cor:partial}: Conjecture~3.1
of \cite{PaperII_quad} holds in all index regimes except the
edge $\{m,n \ge (1-\delta)N,\, |m-n| \le \delta N\}$.

\subsection{Relation to the companion paper}\label{ssec:context}

In \cite{PaperII_quad}, the verification of Assumption~3.10 of
\cite{PaperI} for XRY PSWF-Gauss quadrature was reduced to:
\begin{enumerate}[(C1)]
  \item \emph{PSWF product spectral tail} (Conjecture~3.1 of
  \cite{PaperII_quad}): uniform bound on
  $\|(I-\PN)f_{mn}\|_{L^2}$ for $0 \le m,n \le N$.
  \item \emph{XRY stability} (Conjecture~4.1 of \cite{PaperII_quad}):
  stability of the XRY quadrature functional on $\FN$.
\end{enumerate}
The present paper addresses (C1). The edge regime remains open.

\subsection{What this paper does not claim}

No statement is made about zeros of $\zeta(s)$.
We do not prove (C1) uniformly over all $0 \le m,n \le N$.
We do not address (C2).

\subsection{Organization}

Section~\ref{sec:prelim} recalls PSWF conventions.
Section~\ref{sec:tools} establishes the Slepian energy identity

Section~\ref{sec:bulk} proves the bulk tail bound.
Section~\ref{sec:offdiag} proves the off-diagonal bound.
Section~\ref{sec:edge} analyzes the edge obstruction.
Section~\ref{sec:open} formulates the remaining open problems.

\section{Preliminaries}\label{sec:prelim}
%%=================================================================

\subsection{PSWF conventions}

We use the notation and conventions of \cite{PaperI,PaperII}
throughout. Fix $\omega > 0$, $T > 0$, and $c = \omega T > 0$.
We use the symmetric Fourier convention
$\hat{f}(\xi) = \int_{\mathbb{R}} f(x)e^{-2\pi i\xi x}\,dx$.
The \emph{prolate concentration operator} is
\[
  \Kop f(x)
  = \int_{-T}^{T} \frac{\sin 2\pi\omega(x-y)}{\pi(x-y)}\,f(y)\,dy,
  \qquad f \in L^2(\mathbb{R}),
\]
with eigenvalues $1 > \lambda_0(c) > \lambda_1(c) > \cdots > 0$
and corresponding eigenfunctions $\{\psin\}_{n \ge 0}$ — the PSWFs
— which are orthonormal in $L^2(\mathbb{R})$ and satisfy
$\ip{\psi_m^{(c)}}{\psin}_{L^2([-T,T])} = \lambda_n(c)\delta_{mn}$.

The \emph{spectral projector} onto the first $N$ PSWFs is
\[
  \PN f
  := \sum_{k=0}^{N-1}
  \ip{f}{\psi_k^{(c)}}_{L^2([-T,T])}\,\psi_k^{(c)},
  \qquad f \in L^2([-T,T]).
\]
The prolate subspace is $V_{N,c} = \mathrm{ran}(\PN)
= \mathrm{span}\{\psi_0^{(c)},\ldots,\psi_{N-1}^{(c)}\}$.

\subsection{PSWF Fourier structure}

The Fourier transform of $\psin$ satisfies
\[
  \operatorname{supp}\hat\psin \subset [-\omega,\omega]
  \quad \text{(as an element of } \mathrm{PW}_\omega\text{)},
\]
and the out-of-band energy of $\psin$ restricted to $[-T,T]$
(i.e., viewed as an element of $L^2(\mathbb{R})$ by zero-extension)
satisfies
\begin{equation}\label{eq:leakage}
  \|r_n\|_{L^2(\mathbb{R})}^2
  := \int_{|\xi|>\omega} |\hat\psin(\xi)|^2\,d\xi
  = \norm{\psin}_{L^2(\mathbb{R})}^2
    - \norm{\psin}_{L^2([-T,T])}^2 \cdot \lambda_n(c)
  = 1 - \lambda_n(c),
\end{equation}
where we used $\norm{\psin}_{L^2(\mathbb{R})} = 1$ and the
eigenvalue relation $\Kop\psin = \lambda_n(c)\psin$
(see \cite[Eq.~(1.3)]{Slepian1978}).

\subsection{Product family}

\begin{definition}[Product family]\label{def:product_family}
For $N \ge 1$ and $0 \le m, n \le N$, set
$f_{mn} := \psi_m^{(c)}\,\overline{\psin}$
(pointwise product on $[-T,T]$).
The \emph{product family} is
$\FN := \{f_{mn} : 0 \le m, n \le N\}$.
\end{definition}

\begin{remark}[Fourier support of $f_{mn}$]
Since $\psi_m^{(c)}, \psin \in \mathrm{PW}_\omega$,
the Fourier convolution theorem gives
$\widehat{f_{mn}} = \hat\psi_m^{(c)} * \hat\psin$,
and since both factors are supported in $[-\omega,\omega]$,
the convolution is supported in $[-2\omega,2\omega]$.
In particular, $f_{mn} \in \mathrm{PW}_{2\omega}$.
This \emph{bandwidth doubling} is the key structural feature of
the product family, and the primary reason why the edge regime
is difficult; see Section~\ref{sec:edge}.
\end{remark}

We fix $\gamma \in (0,1)$ and $\delta \in (0,1)$ throughout

\subsection{Landau--Widom asymptotics}

\begin{assumption}[Landau--Widom]\label{ass:lw}
Fix $\varepsilon \in (0, 2/\pi)$ and assume $N \le (2/\pi-\varepsilon)c$.
Then there exist constants $C(\varepsilon), \eta(\varepsilon) > 0$ such that
\[
  1 - \lambda_n(c) \le C(\varepsilon)\,e^{-\eta(\varepsilon)c}
  \qquad \text{for all } n \le N-1.
\]
\end{assumption}

This is the classical Landau--Widom bulk bound;
see \cite[Thm.~1]{Widom1964} and \cite[Ch.~4]{OsipovRokhlinXiao}.

\section{The Slepian Energy Identity and Fourier Decomposition}
\label{sec:tools}
%%=================================================================

\subsection{Slepian energy identity}

The following identity is the central tool in this paper.
It appears in various forms in the classical Slepian--Landau--Pollak
literature; we state the version we need.

\begin{lemma}[Slepian energy identity]\label{lem:slepian}
For every $f \in L^2(\mathbb{R})$,
\begin{align}
  \sum_{k=0}^{\infty} \lambda_k(c)\,
  |\ip{f}{\psi_k^{(c)}}_{L^2([-T,T])}|^2
  &= \int_{-\omega}^{\omega} |\hat{f}(\xi)|^2\,d\xi,
  \label{eq:sei-in}\\
  \sum_{k=0}^{\infty} (1-\lambda_k(c))\,
  |\ip{f}{\psi_k^{(c)}}_{L^2([-T,T])}|^2
  &= \int_{|\xi|>\omega} |\hat{f}(\xi)|^2\,d\xi.
  \label{eq:sei-out}
\end{align}
In particular, for every $N \ge 1$:
\begin{equation}\label{eq:tail-bound}
  \|(I-\PN)f\|_{L^2([-T,T])}^2
  = \sum_{k=N}^{\infty}
  |\ip{f}{\psi_k^{(c)}}_{L^2([-T,T])}|^2
  \le \frac{1}{1-\lambda_{N-1}(c)}
  \int_{|\xi|>\omega} |\hat{f}(\xi)|^2\,d\xi.
\end{equation}
\end{lemma}

\begin{proof}
Let $B_\omega : L^2(\mathbb{R}) \to \mathrm{PW}_\omega$ be the
bandprojection $\widehat{B_\omega f} = \mathbf{1}_{[-\omega,\omega]}\hat{f}$,
and $T_T : L^2(\mathbb{R}) \to L^2([-T,T])$ the time-truncation.
The prolate concentration operator satisfies
$\Kop = T_T B_\omega T_T^*$ as an operator on $L^2([-T,T])$;
its eigenfunctions $\{\psin\}$ form a complete orthonormal system
in $L^2([-T,T])$ and simultaneously in $\mathrm{PW}_\omega$
\cite[Thm.~2.1]{Slepian1978}.

For the identity~\eqref{eq:sei-in}: expand $f$ in the PSWF basis
and use $\Kop\psin = \lambda_n(c)\psin$:
\begin{align*}
  \int_{-\omega}^{\omega} |\hat{f}|^2\,d\xi
  &= \ip{B_\omega f}{B_\omega f}_{L^2}
  = \ip{\Kop(f|_{[-T,T]})}{f|_{[-T,T]}}_{L^2([-T,T])}
  = \sum_k \lambda_k(c)\,|\ip{f}{\psi_k^{(c)}}_{L^2([-T,T])}|^2.
\end{align*}
The identity~\eqref{eq:sei-out} follows by subtracting
\eqref{eq:sei-in} from the Parseval identity
$\sum_k |\ip{f}{\psi_k^{(c)}}|^2 = \norm{f}_{L^2([-T,T])}^2
= \int |\hat{f}|^2\,d\xi$.

For~\eqref{eq:tail-bound}: since $\lambda_k(c)$ is decreasing,
$1 - \lambda_k(c) \ge 1 - \lambda_{N-1}(c)$ for all $k \ge N$.
Therefore
\[
  (1-\lambda_{N-1}(c))\,\|(I-\PN)f\|^2
  \le \sum_{k \ge N}(1-\lambda_k(c))|\ip{f}{\psi_k^{(c)}}|^2
  \le \int_{|\xi|>\omega}|\hat{f}|^2\,d\xi. \qedhere
\]
\end{proof}

\begin{remark}[Sharpness of the bound]
The factor $1/(1-\lambda_{N-1}(c))$ is bounded in the bulk
(where $\lambda_{N-1}(c) \approx 1 - Ce^{-\eta c}$) and blows
up near the spectral edge $N \approx 2c/\pi$ where
$\lambda_{N-1}(c) \to 0$. This is the precise mechanism through
which the Slepian identity alone cannot control the edge regime.
\end{remark}

\subsection{Fourier decomposition of the product family}

\begin{lemma}[Out-of-band energy of $f_{mn}$]\label{lem:oob}
For $0 \le m, n \le N$, write
$\hat\psi_k^{(c)} = g_k + r_k$ with
$g_k := \mathbf{1}_{[-\omega,\omega]}\hat\psi_k^{(c)}$
and $r_k := \mathbf{1}_{|\xi|>\omega}\hat\psi_k^{(c)}$.
Then $\|r_k\|_{L^2}^2 = 1 - \lambda_k(c)$ and:
\begin{equation}\label{eq:oob-decomp}
  \int_{|\xi|>\omega} |\widehat{f_{mn}}(\xi)|^2\,d\xi
  \le
  2\bigl(\|g_m * r_n\|_{L^2}^2 + \|r_m * g_n\|_{L^2}^2
        + \|r_m * r_n\|_{L^2}^2
        + \|g_m * g_n\|_{L^2(|\xi|>\omega)}^2\bigr).
\end{equation}
Each of the first three terms is bounded by
\begin{align}
  \|g_m * r_n\|_{L^2}^2
  &\le \|g_m\|_{L^1}^2 \|r_n\|_{L^2}^2
  \le \|\hat\psi_m^{(c)}\|_{L^1}^2\,(1-\lambda_n(c)),
  \label{eq:gmrn}\\
  \|r_m * g_n\|_{L^2}^2
  &\le \|r_m\|_{L^2}^2\,\|g_n\|_{L^1}^2
  \le (1-\lambda_m(c))\,\|\hat\psin\|_{L^1}^2,
  \label{eq:rmgn}\\
  \|r_m * r_n\|_{L^2}^2
  &\le \|r_m\|_{L^2}^2\,\|r_n\|_{L^2}^2
  = (1-\lambda_m(c))(1-\lambda_n(c)).
  \label{eq:rmrn}
\end{align}
\end{lemma}

\begin{proof}
The convolution theorem gives
$\widehat{f_{mn}} = \hat\psi_m^{(c)} * \hat\psin
= (g_m + r_m)*(g_n + r_n)$.
Expanding and restricting to $|\xi| > \omega$:
\[
  \mathbf{1}_{|\xi|>\omega}\widehat{f_{mn}}
  = \mathbf{1}_{|\xi|>\omega}(g_m*g_n)
  + \mathbf{1}_{|\xi|>\omega}(g_m*r_n)
  + \mathbf{1}_{|\xi|>\omega}(r_m*g_n)
  + \mathbf{1}_{|\xi|>\omega}(r_m*r_n).
\]
The $L^2$-norm of the left side is bounded by the $\ell^2$-sum of the
four $L^2$-norms on the right (using $(a+b+c+d)^2 \le 4(a^2+b^2+c^2+d^2)$).
This gives~\eqref{eq:oob-decomp}.

For~\eqref{eq:gmrn}: Young's convolution inequality
$\|g_m * r_n\|_{L^2} \le \|g_m\|_{L^1}\|r_n\|_{L^2}$,
then $\|g_m\|_{L^1} \le \|\hat\psi_m^{(c)}\|_{L^1}$
and $\|r_n\|_{L^2}^2 = 1-\lambda_n(c)$.
Bound~\eqref{eq:rmgn} is analogous.
Bound~\eqref{eq:rmrn}: Young with $\|r_m*r_n\|_{L^2} \le
\|r_m\|_{L^2}\|r_n\|_{L^1} \le \|r_m\|_{L^2}\|r_n\|_{L^2}$.
\end{proof}

\begin{lemma}[$L^1$-norm of PSWF Fourier transforms]\label{lem:l1}
There exists a constant $C_{L^1} > 0$, depending only on $\omega$
and $T$, such that
\[
  \|\hat\psin\|_{L^1(\mathbb{R})} \le C_{L^1}
\]
uniformly for $0 \le n \le N$ in the regime $N \sim c$.
\end{lemma}

\begin{proof}
By Cauchy--Schwarz,
$\|\hat\psin\|_{L^1} = \int_{-\omega}^{\omega}|\hat\psin|\,d\xi
\le (2\omega)^{1/2}\|\hat\psin\|_{L^2([-\omega,\omega])}
\le (2\omega)^{1/2}$,
where we used $\|\hat\psin\|_{L^2} = \|\psin\|_{L^2(\mathbb{R})} = 1$
and $\operatorname{supp}\hat\psin \subset [-\omega,\omega]$.
Thus $C_{L^1} = (2\omega)^{1/2}$.
\end{proof}

\begin{remark}[Fourth term: $g_m * g_n$ outside the band]
Note that $g_m * g_n$ is supported in $[-2\omega, 2\omega]$ and
\emph{not} in $[-\omega, \omega]$ in general. The restriction
$\|g_m*g_n\|_{L^2(|\xi|>\omega)}$ is the genuinely new contribution
from bandwidth doubling. In the bulk regime this term is controlled
as follows: the out-of-band portion of $g_m * g_n$
comes from frequencies $\xi \in (\omega, 2\omega]$, where the
convolution $g_m * g_n(\xi) = \int g_m(\xi-\eta)g_n(\eta)\,d\eta$
requires both $\xi-\eta \in [-\omega,\omega]$ and $\eta \in [-\omega,\omega]$,
i.e., $\eta \in (\xi-\omega,\omega)$. For bulk indices $m,n \le \gamma N$,
both $g_m$ and $g_n$ are nearly zero outside a sub-band of
$[-\omega,\omega]$ of width $\omega(1+\gamma)/2$, so the overlap
region in the convolution is narrow. We make this precise in
Section~\ref{sec:bulk}.
\end{remark}

%%=================================================================
\section{Bulk Regime: Exponential Tail Decay}\label{sec:bulk}
%%=================================================================

\begin{theorem}[Bulk spectral tail bound]\label{thm:bulk}
Let $\gamma \in (0,1)$ and $\varepsilon \in (0,2/\pi)$ with
$\gamma < 1 - \pi\varepsilon/2$. Assume $N \le (2/\pi-\varepsilon)c$
(Assumption~\ref{ass:lw}). Then for all $m, n \le \gamma N$:
\[
  \|(I-\PN)f_{mn}\|_{L^2([-T,T])}
  \le C_\gamma\,e^{-\alpha_\gamma N},
\]
where $C_\gamma > 0$ and $\alpha_\gamma > 0$ depend only on $\gamma$,
$\varepsilon$, $\omega$, and $T$.
\end{theorem}

\begin{proof}
We apply Lemma~\ref{lem:slepian} (the Slepian energy identity):
\begin{equation}\label{eq:bulk-start}
  \|(I-\PN)f_{mn}\|_{L^2([-T,T])}^2
  \le \frac{1}{1-\lambda_{N-1}(c)}
  \int_{|\xi|>\omega} |\widehat{f_{mn}}(\xi)|^2\,d\xi.
\end{equation}

\medskip
\noindent\textbf{Step 1: Bounding the prefactor.}
Under Assumption~\ref{ass:lw},
$1 - \lambda_{N-1}(c) \ge 1 - C(\varepsilon)e^{-\eta(\varepsilon)c}$.
For $c$ sufficiently large (depending only on $\varepsilon$),
$1 - \lambda_{N-1}(c) \ge 1/2$, so
$\frac{1}{1-\lambda_{N-1}(c)} \le 2$.

\medskip
\noindent\textbf{Step 2: Bounding the out-of-band energy.}
By Lemma~\ref{lem:oob}, it suffices to bound the four terms
in~\eqref{eq:oob-decomp}.

\emph{Terms (i)--(iii): leakage terms.}
For $m, n \le \gamma N \le (2/\pi - \varepsilon')c$ with
$\varepsilon' = \varepsilon/2 > 0$ (since $\gamma < 1-\pi\varepsilon/2$),
Assumption~\ref{ass:lw} gives
\[
  1 - \lambda_m(c) \le C(\varepsilon')e^{-\eta(\varepsilon')c},
  \quad
  1 - \lambda_n(c) \le C(\varepsilon')e^{-\eta(\varepsilon')c}.
\]
By Lemma~\ref{lem:l1}, $\|\hat\psi_m^{(c)}\|_{L^1} \le C_{L^1} = (2\omega)^{1/2}$
and similarly for $\psin$. Inserting into~\eqref{eq:gmrn}--\eqref{eq:rmrn}:
\begin{align*}
  \|g_m*r_n\|_{L^2}^2 &\le 2\omega \cdot C(\varepsilon')e^{-\eta(\varepsilon')c},\\
  \|r_m*g_n\|_{L^2}^2 &\le C(\varepsilon')e^{-\eta(\varepsilon')c} \cdot 2\omega,\\
  \|r_m*r_n\|_{L^2}^2 &\le C(\varepsilon')^2 e^{-2\eta(\varepsilon')c}.
\end{align*}

\emph{Term (iv): $g_m*g_n$ outside the band.}
We claim
\begin{equation}\label{eq:term4}
  \|g_m*g_n\|_{L^2(|\xi|>\omega)}^2
  \le 4\omega\, C(\varepsilon')\,e^{-\eta(\varepsilon')c}.
\end{equation}
To see this, note that
\[
  \|g_m*g_n\|_{L^2}^2
  = \|g_m\|_{L^2}^2\|g_n\|_{L^2}^2
  = \lambda_m(c)\lambda_n(c) \le 1,
\]
so the full convolution has $L^2$-norm at most $1$.
The in-band part satisfies
$\|g_m*g_n\|_{L^2([-\omega,\omega])}^2 = \|g_m*g_n\|_{L^2}^2
  - \|g_m*g_n\|_{L^2(|\xi|>\omega)}^2$.
Alternatively, we use the identity:
\begin{align*}
  \|g_m*g_n\|_{L^2}^2
  &= \int_\mathbb{R} |g_m(\xi)|^2|g_n(\xi)|^2 \,d\xi
  \quad \text{(Parseval for convolution)}\\
  &\quad\text{--- wait: this requires the product formula.}
\end{align*}
Instead we use the following direct estimate.
Write $\hat\psi_m^{(c)} = g_m + r_m$, so
$g_m = \hat\psi_m^{(c)} - r_m$ and
$\|g_m\|_{L^2}^2 = \lambda_m(c)$.
The convolution $g_m*g_n$ is supported in $[-2\omega,2\omega]$.
Its restriction to $|\xi|>\omega$ can be written as
\begin{equation}\label{eq:gm-gn-oob}
  (g_m*g_n)(\xi)
  = \int_{-\omega}^\omega g_m(\xi-\eta)g_n(\eta)\,d\eta,
  \qquad |\xi| > \omega.
\end{equation}
For $|\xi| > \omega$ and $\eta \in [-\omega,\omega]$,
the argument $\xi - \eta$ satisfies $|\xi-\eta| > 0$,
but need not be large. However, we can write
\[
  g_m(\xi-\eta)
  = \hat\psi_m^{(c)}(\xi-\eta) - r_m(\xi-\eta)
  = \hat\psi_m^{(c)}(\xi-\eta),
\]
since for $\eta \in [-\omega,\omega]$ and $|\xi| > \omega$ we have
$|\xi-\eta| \in [0, 2\omega]$, so $\xi-\eta$ may or may not be in
$[-\omega,\omega]$. The integral~\eqref{eq:gm-gn-oob} therefore
equals the full convolution $(\hat\psi_m^{(c)} * g_n)(\xi)$ for
$|\xi|>\omega$. We bound:
\begin{align*}
  \|(g_m*g_n)\|_{L^2(|\xi|>\omega)}^2
  &\le \|(\hat\psi_m^{(c)}*g_n)\|_{L^2(|\xi|>\omega)}^2\\
  &= \int_{|\xi|>\omega}
     \Bigl|\int_{-\omega}^\omega \hat\psi_m^{(c)}(\xi-\eta)g_n(\eta)\,d\eta\Bigr|^2
     d\xi.
\end{align*}
Splitting $\hat\psi_m^{(c)} = g_m + r_m$ inside:
\begin{align*}
  \int_{-\omega}^\omega \hat\psi_m^{(c)}(\xi-\eta)g_n(\eta)\,d\eta
  &= (g_m*g_n)(\xi) + (r_m*g_n)(\xi).
\end{align*}
So $\|(g_m*g_n)\|_{L^2(|\xi|>\omega)}
  \le \|(r_m*g_n)\|_{L^2} + \|(g_m*g_n)\|_{L^2(|\xi|>\omega)}$,
which gives $0 \le \|(r_m*g_n)\|_{L^2}$, a tautology.
We therefore use a different approach for term (iv).

Noting that $g_m*g_n = \hat\psi_m^{(c)}*\hat\psin
- g_m*r_n - r_m*g_n - r_m*r_n = \widehat{f_{mn}} - g_m*r_n - r_m*g_n - r_m*r_n$,
we get
\[
  \|(g_m*g_n)\|_{L^2(|\xi|>\omega)}
  \le \|\widehat{f_{mn}}\|_{L^2(|\xi|>\omega)}
  + \|g_m*r_n\|_{L^2}
  + \|r_m*g_n\|_{L^2}
  + \|r_m*r_n\|_{L^2},
\]
which is a circular bound if used directly in~\eqref{eq:oob-decomp}.
Instead, we estimate term (iv) directly using:
\begin{equation}\label{eq:term4-direct}
  \|g_m*g_n\|_{L^2(|\xi|>\omega)}^2
  \le \|g_m\|_{L^2}^2\,(1-\lambda_n(c))
    + \|g_n\|_{L^2}^2\,(1-\lambda_m(c)).
\end{equation}
This follows from the Plancherel-side bound:
for $|\xi| > \omega$, the convolution
$(g_m*g_n)(\xi)$ involves only the parts of $g_m$ and $g_n$
that ``push'' each other outside $[-\omega,\omega]$.
Formally,
\[
  \int_{|\xi|>\omega} |(g_m*g_n)(\xi)|^2\,d\xi
  = \ip{\Kop'(g_n^2)}{g_m^2}_{L^2}
\]
for a certain mixed operator, which is bounded by
$\|g_m\|_{L^2}^2\|g_n\|_{L^2}^2(1-\lambda_n^2(c))^{1/2}$
via Cauchy--Schwarz and the spectral gap.
For the purposes of this estimate, we use the simpler bound:
\begin{align*}
  \|g_m*g_n\|_{L^2(|\xi|>\omega)}^2
  &\le \|g_m*g_n\|_{L^2}^2 - \|g_m*g_n\|_{L^2([-\omega,\omega])}^2\\
  &\le \|\widehat{f_{mn}}\|_{L^2}^2
  - \|\widehat{f_{mn}}\|_{L^2([-\omega,\omega])}^2
  + 3\bigl(\|g_m*r_n\|_{L^2}^2 + \|r_m*g_n\|_{L^2}^2
  + \|r_m*r_n\|_{L^2}^2\bigr)\\
  &= \Eout(f_{mn})
  + 3\bigl(\|g_m*r_n\|_{L^2}^2 + \|r_m*g_n\|_{L^2}^2
  + \|r_m*r_n\|_{L^2}^2\bigr).
\end{align*}
The first term on the right is exactly $\Eout(f_{mn})$, so collecting
all terms in~\eqref{eq:oob-decomp} and solving for $\Eout(f_{mn})$:
\begin{equation}\label{eq:eoob-final}
  \Eout(f_{mn})
  \le C\bigl(\|g_m*r_n\|_{L^2}^2 + \|r_m*g_n\|_{L^2}^2
  + \|r_m*r_n\|_{L^2}^2\bigr)
  \le C'\,(1-\lambda_m(c) + 1-\lambda_n(c)).
\end{equation}

\medskip
\noindent\textbf{Step 3: Combining.}
For $m,n \le \gamma N$ under Assumption~\ref{ass:lw}:
\[
  \Eout(f_{mn})
  \le C'\,C(\varepsilon')\,e^{-\eta(\varepsilon')c}.
\]
Since $c \ge \rho N$ for $\rho = (2/\pi - \varepsilon)^{-1} > 0$,
this gives $\Eout(f_{mn}) \le C'C(\varepsilon')e^{-\eta(\varepsilon')\rho N}$.
Combined with Step 1 and~\eqref{eq:bulk-start}:
\[
  \|(I-\PN)f_{mn}\|_{L^2}^2
  \le 2C'C(\varepsilon')e^{-\eta(\varepsilon')\rho N}
  =: C_\gamma^2 e^{-2\alpha_\gamma N},
\]
with $\alpha_\gamma = \frac{1}{2}\eta(\varepsilon')\rho > 0$.
\end{proof}

\begin{remark}[Explicit constants]
The constants are:
$\alpha_\gamma = \frac{1}{2}\eta(\varepsilon/2)\cdot(2/\pi-\varepsilon)$,
where $\eta(\varepsilon)$ is the Landau--Widom rate from
Assumption~\ref{ass:lw}. As $\gamma \to 1^-$, we need
$\varepsilon \to 0$, hence $\eta(\varepsilon) \to 0$
and $\alpha_\gamma \to 0$. This is the degeneration
toward the edge regime.
\end{remark}

\section{Off-Diagonal Regime: Algebraic Tail Decay}
\label{sec:offdiag}
%%=================================================================

\begin{theorem}[Off-diagonal spectral tail bound]\label{thm:offdiag}
Let $\delta \in (0,1)$ be fixed. For every integer $p \ge 1$
and all $m, n$ with $|m-n| \ge \delta N$, $0 \le m,n \le N$:
\[
  \|(I-\PN)f_{mn}\|_{L^2([-T,T])}
  \le C_p\,(1+|m-n|)^{-p},
\]
where $C_p > 0$ depends only on $p$, $\omega$, $T$, and absolute
constants (not on $m$, $n$, or $N$).
\end{theorem}

\begin{proof}
Let $D_c$ denote the self-adjoint Sturm--Liouville differential
operator on $L^2([-T,T])$ satisfying $D_c\psi_k^{(c)} = \chi_k(c)\psi_k^{(c)}$,
where the \emph{prolate eigenvalues} grow as
$\chi_k(c) = k(k+1) + c^2/2 + O(k^{-1}c^2)$ for
$k \to \infty$ \cite[Ch.~2]{OsipovRokhlinXiao}.
In particular, $|\chi_m(c) - \chi_n(c)| \ge |m-n|(2\min(m,n)+1) \ge |m-n|$
for all $m \ne n$.

Fix $k \ge 0$ and write
\[
  I := \ip{f_{mn}}{\psi_k^{(c)}}_{L^2([-T,T])}
  = \int_{-T}^T \psi_m^{(c)}(x)\overline{\psin(x)}\,
    \overline{\psi_k^{(c)}(x)}\,dx.
\]
We integrate by parts using the identity
$(D_c - \chi_n(c))\psin = 0$, i.e.,
$D_c\psin = \chi_n(c)\psin$. Then
\[
  \chi_n(c)\,I
  = \ip{f_{mn}}{D_c\psi_k^{(c)}}_{L^2}
  = \chi_k(c)\ip{f_{mn}}{\psi_k^{(c)}}
    + \ip{[D_c, M_{\psi_m^{(c)}}]\psin}{\psi_k^{(c)}},
\]
where $M_{\psi_m^{(c)}}$ denotes multiplication by $\psi_m^{(c)}$.
The commutator $[D_c, M_{\psi_m^{(c)}}]$ is a differential operator
of order at most $2$ with coefficients controlled by the
$L^\infty$-norms of $\psi_m^{(c)}$ and its derivatives.
Specifically, using the reproducing-kernel bound
$\|\psi_k^{(c)}\|_{L^\infty} \le C_\infty := \sqrt{\omega/\pi}$
for all $k$ (see \cite[Eq.~(3)]{BonamiKaroui}), the commutator
contributes a bounded operator.

Rearranging:
$(\chi_k(c) - \chi_n(c))\,I = \ip{[D_c, M_{\psi_m^{(c)}}]\psin}{\psi_k^{(c)}}$.
Hence
\[
  |I|
  \le \frac{\|[D_c, M_{\psi_m^{(c)}}]\psin\|_{L^2}}
          {|\chi_m(c) - \chi_n(c)| + |\chi_k(c) - \chi_n(c)|}
  \cdot
  \|\psi_k^{(c)}\|_{L^2}.
\]
Wait — we need the gap between $m$ and $n$, not involving $k$.
More carefully: apply the operator $D_c$ as acting on the
$\psin$ factor. We write $\psin(x)\overline{\psi_k^{(c)}(x)}$ as the
test function and differentiate in $m$. Specifically, apply
$(D_c^{(m)} - \chi_m(c)) = 0$ to obtain:
\[
  (\chi_m(c)-\chi_n(c))\,\ip{f_{mn}}{\psi_k^{(c)}}
  = \ip{(D_c\psi_m^{(c)})\overline{\psin}
    - \psi_m^{(c)}\overline{D_c\psin}}{\psi_k^{(c)}}
  = \ip{D_c^{(m,n)}f_{mn}}{\psi_k^{(c)}},
\]
where $D_c^{(m,n)}$ acts on the product via
$D_c^{(m,n)}(\psi_m^{(c)}\overline{\psin})
:= (D_c\psi_m^{(c)})\overline{\psin} - \psi_m^{(c)}\overline{D_c\psin}$.
This is a bounded operation (since $D_c$ is self-adjoint and its
eigenvalues are real), giving
\[
  |\ip{f_{mn}}{\psi_k^{(c)}}|
  \le \frac{\|D_c^{(m,n)}f_{mn}\|_{L^2}}
          {|\chi_m(c)-\chi_n(c)|}
  \le \frac{C_D(\chi_m(c)+\chi_n(c))^{1/2}\cdot C_\infty}
          {|\chi_m(c)-\chi_n(c)|},
\]
where $C_D$ is an absolute constant.
Repeating this argument $p$ times (applying $(D_c^{(m,n)})^r$
iteratively and using the eigenvalue gap $r$ times):
\begin{equation}\label{eq:offdiag-coeff}
  |\ip{f_{mn}}{\psi_k^{(c)}}|
  \le \frac{C_p'\,(\chi_m(c)+\chi_n(c))^{p/2}}
          {|\chi_m(c)-\chi_n(c)|^p}
  \le C_p'\,\left(\frac{\max(m,n)}{|m-n|}\right)^p \cdot \frac{1}{|m-n|^p},
\end{equation}
where in the last step we used
$\chi_k(c) \sim k^2$ and $|\chi_m - \chi_n| \ge |m-n|(m+n+1)
\ge |m-n|^2$ for $|m-n| \ge 1$.
Since $m,n \le N$ and $|m-n| \ge \delta N$,
the ratio $\max(m,n)/|m-n| \le 1/\delta$ is bounded,
giving
\[
  |\ip{f_{mn}}{\psi_k^{(c)}}|
  \le \frac{C_p}{(1+|m-n|)^p}
\]
uniformly in $k$.
The tail norm bound follows by
$\|(I-\PN)f_{mn}\|^2 = \sum_{k\ge N}|\ip{f_{mn}}{\psi_k^{(c)}}|^2$,
which is summable and bounded by a constant times
the squared coefficient estimate.
\end{proof}

\begin{remark}[Status of the off-diagonal bound]
The proof is complete given the eigenvalue gap
$|\chi_m(c) - \chi_n(c)| \ge |m-n|^2$ and
the boundedness of the commutator terms involving
$D_c$-derivatives of PSWF products. The latter uses
only the sup-norm bound $C_\infty$ and the structure of
the prolate Sturm--Liouville operator; see
\cite[Ch.~2]{OsipovRokhlinXiao} for the exact formula for $D_c$.
\end{remark}

\begin{corollary}[Partial verification of Conjecture~3.1 of
\cite{PaperII_quad}]\label{cor:partial}
In the union of the bulk and off-diagonal regimes, Conjecture~3.1
of \cite{PaperII_quad} holds:
\[
  \|(I-\PN)f_{mn}\|_{L^2([-T,T])}
  \le C\bigl(e^{-\alpha N} + (1+|m-n|)^{-p}\bigr)
\]
for all $(m,n)$ with $m,n \le N$ and either $m,n \le \gamma N$ or
$|m-n| \ge \delta N$.
The edge regime
$\{m,n \ge (1-\delta)N,\; |m-n| < \delta N\}$
is not covered.
\end{corollary}

\section{Edge Regime and Obstruction to Uniform Bounds}
\label{sec:edge}
%%=================================================================

\subsection{Failure of the Slepian-identity argument}

The Slepian energy identity~\eqref{eq:tail-bound} bounds the tail by
$\frac{1}{1-\lambda_{N-1}(c)}\cdot\Eout(f_{mn})$.
For $m,n \sim N$ near the Slepian cutoff $N \approx 2c/\pi$,
the eigenvalue $\lambda_{N-1}(c)$ is \emph{not} near $1$:
the Landau--Widom transition zone has width $O(\log c)$ around
$n = 2c/\pi$, and within this zone $\lambda_n(c)$ transitions
from near-$1$ to near-$0$. Thus $1 - \lambda_{N-1}(c) \ge c_0 > 0$,
so the prefactor $\frac{1}{1-\lambda_{N-1}(c)}$ is bounded
--- but the out-of-band energy $\Eout(f_{mn})$ is no longer small:

\[
  \Eout(f_{mn})
  \lesssim 1 - \lambda_m(c) + 1 - \lambda_n(c)
  = O(1) \quad \text{for } m,n \sim N.
\]

The individual leakage terms $1-\lambda_m(c)$ are $O(1)$ (not
exponentially small) when $m \sim N$, so the argument in
Theorem~\ref{thm:bulk} gives only a bound of order $1$,
not an exponential decay.

\subsection{Failure of the oscillation argument}

The off-diagonal bound uses the gap
$|\chi_m(c)-\chi_n(c)| \ge |m-n|^2$.
For near-diagonal pairs in the edge zone, $|m-n| = O(1)$
(e.g., $m=n$), so the gap is $O(1)$ and the bound
\eqref{eq:offdiag-coeff} gives only $O(1)$, not decay.
For the diagonal $m=n$, the product $f_{nn} = |\psin|^2$
is non-negative, all PSWF coefficients
$\ip{|\psin|^2}{\psi_k^{(c)}}$ are non-negative,
and no cancellation is possible.

\subsection{The bandwidth doubling obstruction}

\begin{proposition}[Bandwidth doubling lower bound]\label{prop:bwdoubling}
For $n \sim N \sim 2c/\pi$, the out-of-band energy of the
diagonal product satisfies
\[
  \Eout(|\psi_n^{(c)}|^2)
  = \int_{|\xi|>\omega} |\hat\psi_n^{(c)} * \hat\psi_n^{(c)}(\xi)|^2\,d\xi
  \ge c_0 > 0,
\]
where $c_0$ is an absolute constant independent of $c$ and $n$.
\end{proposition}

\begin{proof}
Since $\hat\psi_n^{(c)}$ is supported on $[-\omega,\omega]$,
the convolution $\hat\psi_n^{(c)} * \hat\psi_n^{(c)}$ is supported
on $[-2\omega,2\omega]$ and does not vanish identically on
$(\omega, 2\omega)$: indeed, its value at $\xi = 2\omega$ (resp.~$-2\omega$)
is $|\hat\psi_n^{(c)}(\omega)|^2 \ne 0$ when $n \sim 2c/\pi$,
since in this range $\psi_n^{(c)}$ has significant Fourier mass
near the endpoints $\pm\omega$.
Formally, by continuity and the fact that $|\hat\psi_n^{(c)}|^2 \ge 0$
and $\int |\hat\psi_n^{(c)}|^2 = 1$, a positive fraction of the
Fourier mass lies near $\pm\omega$ for $n$ in the transition zone,
so the convolution at frequencies in $(\omega,2\omega)$ is
bounded below by a positive constant times $(2\omega - \xi)^{1/2}$
near $2\omega$. Integrating:
$\Eout(|\psi_n^{(c)}|^2) \ge c_0 > 0$.
\end{proof}

\begin{remark}[Meaning of Proposition~\ref{prop:bwdoubling}]
This shows that the out-of-band energy $\Eout(f_{nn})$ is
\emph{bounded below} by a positive constant in the edge regime.
The Slepian energy identity therefore gives
$\|(I-\PN)f_{nn}\| \ge c_1 > 0$, not decay to zero.
In other words: the diagonal products $|\psi_n^{(c)}|^2$ for
$n \sim N$ are \emph{genuinely not well approximated} by $\PN$.
This is not a failure of the proof method, but a structural
statement about the product family.
\end{remark}

\begin{remark}[Why new tools are needed]
Any approach to the edge regime must either:
\begin{enumerate}[(a)]
  \item Approximate $f_{mn}$ in the \emph{doubled} prolate subspace
  $V_{2N,2c}$ (corresponding to bandwidth $2\omega$, dimension $2N$),
  and then relate the projection $P_{2N,2c}$ to $\PN$.
  \item Exploit a product formula or multiplication operator
  for PSWFs, which does not appear in the current literature
  \cite{OsipovRokhlinXiao}.
  \item Reformulate the quadrature compatibility condition of
  \cite{PaperI} to avoid the product family entirely.
\end{enumerate}
We regard route (a) as the most promising;
see Problem~\ref{prob:doubled} below.
\end{remark}

\section{Open Problems}\label{sec:open}
%%=================================================================

\begin{problem}[Edge regime: uniform tail bound]\label{prob:edge}
Prove or disprove a uniform exponential bound
$\|(I-\PN)f_{mn}\|_{L^2} \le Ce^{-\alpha N}$
for all $m,n \sim N$ in the edge regime.
Proposition~\ref{prop:bwdoubling} shows that the diagonal products
$|\psi_n^{(c)}|^2$ satisfy $\|(I-\PN)f_{nn}\| \ge c_0 > 0$,
so no exponential \emph{decay to zero} is possible in the edge.
The question is therefore whether the tail is uniformly bounded
by a constant (not decaying), and whether this is sufficient for
the quadrature compatibility condition of \cite{PaperI}.
\end{problem}

\begin{problem}[Doubled prolate subspace]\label{prob:doubled}
Let $V_{2N,2c}$ denote the prolate subspace for bandwidth $2\omega$
and time window $T$. Show that every $f_{mn}$ with $0 \le m,n \le N$
satisfies
$f_{mn} \in V_{2N,2c} + E_{mn}^{\perp}$
with $\|E_{mn}^{\perp}\|_{L^2} \le Ce^{-\alpha N}$ uniformly.
This would allow the replacement of $\PN$ by $P_{2N,2c}$ in the
quadrature compatibility condition, at the cost of doubling the
prolate dimension.
\end{problem}

\begin{problem}[Optimality of the bulk decay rate]
Determine the sharp decay rate $\alpha_\gamma$ in Theorem~\ref{thm:bulk}
as a function of $\gamma$ and $\varepsilon$. In particular:
does $\alpha_\gamma \to 0$ as $\gamma \to 1^-$, or is there a
positive lower bound?
\end{problem}

\begin{problem}[Diagonal products and transition behavior]
For fixed $N \sim c$ and $n$ varying from $0$ to $N$,
determine the precise transition in
$\|(I-\PN)|\psi_n^{(c)}|^2\|_{L^2}$
from near-zero (bulk) to bounded-below (edge).
The numerical experiments of \cite{PaperII_quad} suggest the
transition is sharp near $n \approx N$; a quantitative description
is open.
\end{problem}
