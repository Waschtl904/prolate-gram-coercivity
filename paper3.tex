\title[PSWF Product Spectral Tail Estimates]
      {PSWF Product Spectral Tail Estimates:\\
       Bulk and Off-Diagonal Regimes,\\
       and an Obstruction at the Spectral Edge\\[4pt]
       \large (Paper~III in the Prolate--Weil Program)}

\thanks{This paper is the third in the Prolate--Weil program.
Paper~I \cite{PaperI} established frame coercivity of the prolate Gram
form under a quadrature compatibility assumption (Assumption~3.10 of
\cite{PaperI}).
The companion paper \cite{PaperII_quad} organised the verification of
that assumption for XRY PSWF-Gauss quadrature into two explicit
conjectural inputs: a PSWF product spectral tail conjecture and an
XRY stability conjecture. The present paper addresses the first of
these two inputs.
In the bulk regime, the tail bound is proved conditional on
Assumption~\ref{ass:bulkconv} (exponential out-of-band convolution
decay); this is identified as an open problem (Problem~\ref{prob:bulkconv}).
In the off-diagonal regime, an unconditional uniform bound is proved
(Theorem~\ref{thm:offdiag}) and a mean spectral localization
result is established unconditionally
(Proposition~\ref{prop:mean-loc}); the stronger pointwise
spectral localization is formulated under Assumption~\ref{ass:specloc}
(PSWF Clebsch--Gordan), left as an open problem
(Problem~\ref{prob:cg}).
The edge regime $m,n \sim N$ is identified as a structural obstruction
and left as an explicit open problem.}

\begin{abstract}
Let $\{\psin\}_{n \ge 0}$ be the prolate spheroidal wave functions
(PSWFs) on $[-T,T]$ with bandwidth parameter $\omega > 0$ and
time-bandwidth product $c = \omega T$. Let $\PN$ denote the
orthogonal projector in $L^2([-T,T])$ onto the prolate subspace
$V_{N,c} = \mathrm{span}\{\psi_0^{(c)}, \ldots, \psi_{N-1}^{(c)}\}$.
The companion paper \cite{PaperII_quad} requires a uniform tail bound
for the product family
$\FN := \{\psi_m^{(c)}\overline{\psin} : 0 \le m,n \le N\}$.

The central analytical tool is the \emph{Slepian energy identity}
(Lemma~\ref{lem:slepian}), which bounds the spectral tail
$\|(I-\PN)f\|_{L^2([-T,T])}^2$ by the out-of-band Fourier
energy $\Eout(f) := \int_{|\xi|>\omega}|\hat{f}(\xi)|^2\,d\xi$.
For $f_{mn} = \psi_m^{(c)}\overline{\psin}$, the convolution
$\widehat{f_{mn}} = \hat\psi_m^{(c)} * \hat\psi_n^{(c)}$
is supported in $[-2\omega,2\omega]$ (bandwidth doubling).
The band-edge convolution estimate (Lemma~\ref{lem:bandedge}) gives
$\Eout(f_{mn}) \le 4\omega$ uniformly.

We prove:
\begin{enumerate}[(I)]
  \item \textbf{Bulk regime} ($m, n \le \gamma N$, $\gamma < 1$):
  $\|(I-\PN)f_{mn}\|_{L^2} \le C_\gamma e^{-\alpha_\gamma N}$,
  conditional on Assumption~\ref{ass:bulkconv}
  (exponential out-of-band convolution decay;
  Problem~\ref{prob:bulkconv}).
  \item \textbf{Off-diagonal regime} ($|m-n| \ge \delta N$, $\delta > 0$):
  \begin{enumerate}[(a)]
    \item \emph{Unconditional uniform bound:}
    $\|(I-\PN)f_{mn}\|_{L^2} \le C_\delta T^{1/2}$
    (Theorem~\ref{thm:offdiag}).
    \item \emph{Unconditional mean spectral localization:}
    $\Emn[\chi_k] = \muhat + E_{mn}$
    where $\muhat = \chi_m(c)+\chi_n(c)$
    and $|E_{mn}| \le C\,\|(\psi_m^{(c)})'\|_{L^2}
    \|(\psin)'\|_{L^\infty}$
    (Proposition~\ref{prop:mean-loc}).
    \item \emph{Conditional algebraic decay:}
    $\|(I-\PN)f_{mn}\|_{L^2} \le C_p(1+|m-n|)^{-p}$
    under Assumption~\ref{ass:specloc}
    (Problem~\ref{prob:cg}).
  \end{enumerate}
  \item \textbf{Edge regime obstruction} ($m, n \sim N$):
  Proposition~\ref{prop:bwdoubling} shows
  $\Eout(f_{nn}) \ge c_0 > 0$.
\end{enumerate}
No claim is made about zeros of $\zeta(s)$.
\end{abstract}

\section{Introduction}\label{sec:intro}
%%=================================================================

\subsection{Context and motivation}

The Prolate--Weil program \cite{PaperI,PaperII,PaperII_quad} seeks a
spectral realization of the Weil positivity criterion
\cite{Weil1952,CCM2025} via the prolate concentration operator
$\Kop$ and its associated Gram forms.
In \cite{PaperI}, the prolate Gram form was shown to be coercive
under an explicit quadrature compatibility condition.
The companion paper \cite{PaperII_quad} reduced verification of
that condition to two conjectural inputs; see
Section~\ref{ssec:context}.

The first input is a uniform tail bound for the product family
$f_{mn} = \psi_m^{(c)}\overline{\psin}$.
This paper establishes such bounds in the bulk and off-diagonal
regimes, proves mean spectral localization unconditionally,
identifies the structural obstruction at the edge,
and isolates explicit open problems.

\subsection{Strategy and new results}

The Slepian energy identity (Lemma~\ref{lem:slepian})
reduces to bounding $\Eout(f_{mn})$ in the bulk.
In the off-diagonal regime, the near-eigenvector structure
(Section~\ref{ssec:neareig})
\begin{equation}\label{eq:near-eig-intro}
  D_c f_{mn} = \muhat f_{mn} + R_{mn},
  \qquad \muhat := \chi_m(c) + \chi_n(c),
\end{equation}
yields — by direct substitution into the spectral
expansion of $D_c$ — the \emph{mean spectral localization}
identity:
\begin{equation}\label{eq:mean-loc-intro}
  \Emn[\chi_k]
  := \frac{\sum_k \chi_k(c)\,|a_k^{mn}|^2}
            {\|f_{mn}\|_{L^2}^2}
  = \muhat + \frac{\ip{R_{mn}}{f_{mn}}_{L^2}}
                   {\|f_{mn}\|_{L^2}^2}.
\end{equation}
This is proved without any assumption
(Proposition~\ref{prop:mean-loc}).
For the stronger pointwise decay,
Assumption~\ref{ass:specloc} is needed;
its proof is identified as Problem~\ref{prob:cg}
(PSWF Clebsch--Gordan).

\subsection{Main results summary}

\begin{theorem}[Bulk; Theorem~\ref{thm:bulk}]
Under Assumption~\ref{ass:bulkconv}, for $m,n \le \gamma N$:
$\|(I-\PN)f_{mn}\|_{L^2} \le C_\gamma e^{-\alpha_\gamma N}$.
\end{theorem}

\begin{theorem}[Off-diagonal, unconditional; Theorem~\ref{thm:offdiag}]
For $|m-n| \ge \delta N$:
$\|(I-\PN)f_{mn}\|_{L^2} \le C_\delta T^{1/2}$.
\end{theorem}

\begin{proposition}[Mean localization, unconditional;
Proposition~\ref{prop:mean-loc}]
For all $m,n \le N$:
$\Emn[\chi_k] = \muhat + E_{mn}$,
where $|E_{mn}| \le 2T^{1/2}\|(\psi_m^{(c)})'\|_{L^2}
\|(\psin)'\|_{L^\infty}$.
\end{proposition}

\begin{theorem}[Off-diagonal, conditional; Theorem~\ref{thm:offdiag-strong}]
Under Assumption~\ref{ass:specloc}, for $|m-n| \ge \delta N$:
$\|(I-\PN)f_{mn}\|_{L^2} \le C_p(1+|m-n|)^{-p}$.
\end{theorem}

\subsection{Relation to companion paper}\label{ssec:context}

In \cite{PaperII_quad}, verification of Assumption~3.10 of \cite{PaperI}
was reduced to:
\begin{enumerate}[(C1)]
  \item PSWF product spectral tail (Conjecture~3.1 of \cite{PaperII_quad}).
  \item XRY stability (Conjecture~4.1 of \cite{PaperII_quad}).
\end{enumerate}
This paper addresses (C1) conditionally in the tail sense, and
unconditionally in the mean localization sense. (C2) is not addressed.

\subsection{What this paper does not claim}

No statement is made about zeros of $\zeta(s)$.
Assumptions~\ref{ass:bulkconv} and~\ref{ass:specloc} are not proved;
both are formulated as explicit open problems.

\section{Preliminaries}\label{sec:prelim}
%%=================================================================

\subsection{PSWF conventions}

Fix $\omega > 0$, $T > 0$, $c = \omega T > 0$.
Fourier convention: $\hat{f}(\xi) = \int_{\mathbb{R}} f(x)e^{-2\pi i\xi x}\,dx$.

The \emph{prolate concentration operator}:
\begin{equation}\label{eq:Kop_def}
  \Kop = R_T \circ B_\omega \circ R_T,
\end{equation}
where $B_\omega = \mathcal{F}^{-1}\mathbf{1}_{[-\omega,\omega]}\mathcal{F}$
and $(R_T f)(x) = f(x)\mathbf{1}_{[-T,T]}(x)$.
Eigenvalues $1 > \lambda_0(c) > \lambda_1(c) > \cdots > 0$,
eigenfunctions $\{\psin\}_{n\ge 0}$ orthonormal in $L^2(\mathbb{R})$,
$\ip{\psi_m^{(c)}}{\psin}_{L^2([-T,T])} = \lambda_n(c)\delta_{mn}$.

\subsection{Time concentration and leakage}

\begin{equation}\label{eq:time_conc}
  \lambda_n(c) = \|\psin\|_{L^2([-T,T])}^2,
  \qquad
  \ell_n := 1-\lambda_n(c)
  = \|\psin\|_{L^2(\mathbb{R}\setminus[-T,T])}^2.
\end{equation}

\begin{remark}[Leakage is a time-domain quantity]\label{rem:leakage}
$\ell_n$ measures $L^2$-mass of $\psin$ outside $[-T,T]$ only.
$\int_{|\xi|>\omega}|\hat\psin|^2 = 0$ exactly
(Section~\ref{ssec:bandlimited}).
\end{remark}

\subsection{Bandlimited structure}\label{ssec:bandlimited}

\begin{equation}\label{eq:bandlimited}
  \operatorname{supp}\hat\psin \subset [-\omega,\omega],
  \qquad \|\hat\psin\|_{L^2}^2 = 1.
\end{equation}

\subsection{Product family and bandwidth doubling}

\begin{definition}\label{def:product_family}
$f_{mn} := \psi_m^{(c)}\overline{\psin}$;
$\FN := \{f_{mn}: 0 \le m,n \le N\}$.
\end{definition}

\begin{equation}\label{eq:product_fourier}
  \widehat{f_{mn}} = \hat\psi_m^{(c)} * \hat\psin,
  \quad \operatorname{supp}\widehat{f_{mn}} \subset [-2\omega,2\omega],
  \quad f_{mn} \in \mathrm{PW}_{2\omega}.
\end{equation}

\subsection{Sturm--Liouville operator}

Each $\psin$ satisfies:
\begin{equation}\label{eq:SL}
  D_c\psin = \chi_n(c)\psin,
\end{equation}
where $D_c$ is self-adjoint on $L^2([-T,T])$ with
\begin{equation}\label{eq:chi-asym}
  \chi_n(c) = n(n+1) + \tfrac{1}{2}c^2 + O(n^{-1}c^2),
  \qquad n\to\infty.
\end{equation}
See \cite[Ch.~2]{OsipovRokhlinXiao}.

\subsection{PSWF coefficient sequence}

\begin{definition}\label{def:coeff}
$a_k^{mn} := \ip{f_{mn}}{\psi_k^{(c)}}_{L^2([-T,T])}$.
Spectral tail:
$\|(I-\PN)f_{mn}\|_{L^2}^2 = \sum_{k \ge N}|a_k^{mn}|^2$.
\end{definition}

\begin{definition}[Spectral mean]\label{def:spectral-mean}
Define the \emph{spectral mean} of $f_{mn}$ with respect to
the PSWF basis as
\begin{equation}\label{eq:spectral-mean-def}
  \Emn[\chi_k]
  := \frac{\displaystyle\sum_{k=0}^{\infty}\chi_k(c)\,|a_k^{mn}|^2}
           {\|f_{mn}\|_{L^2([-T,T])}^2}.
\end{equation}
This is the \emph{weighted average eigenvalue index} of $f_{mn}$
in the $D_c$-spectrum, normalized to unit $\ell^2$-mass.
It is the spectral-theory analogue of the quantum-mechanical
expectation value $\langle D_c \rangle_{f_{mn}}$.
\end{definition}

\subsection{Spectral projector and index regimes}

$\PN f := \sum_{k=0}^{N-1}\ip{f}{\psi_k^{(c)}}_{L^2}\psi_k^{(c)}$.

\subsection{Assumptions}

\begin{assumption}[Landau--Widom]\label{ass:lw}
For $N \le (2/\pi-\varepsilon)c$, there exist
$C(\varepsilon),\eta(\varepsilon) > 0$ such that
$\ell_n \le Ce^{-\eta c}$ for all $n \le N-1$.
In particular $\lambda_{N-1}(c) > 1/2$.
\end{assumption}

See \cite[Thm.~1]{Widom1964} and \cite[Ch.~4]{OsipovRokhlinXiao}.

\begin{assumption}[Bulk convolution decay]\label{ass:bulkconv}
For fixed $\gamma\in(0,1)$, there exist
$C_{\mathrm{bc}}(\gamma),\alpha_{\mathrm{bc}}(\gamma)>0$ such that
for all $m,n\le\gamma N$ and $c\ge c_0(\gamma)$:
\[
  \Eout(f_{mn}) \le C_{\mathrm{bc}}\,e^{-\alpha_{\mathrm{bc}}c}.
\]
\end{assumption}

\begin{remark}[Sufficient condition for Assumption~\ref{ass:bulkconv}]
\label{rem:bulkconv-status}
A WKB turning-point bound
$|\hat\psi_n^{(c)}(\xi)| \le Ce^{-\eta c(\omega-|\xi|)}$
for $n\le\gamma N$ near $|\xi|=\omega$
would imply Assumption~\ref{ass:bulkconv}; see Problem~\ref{prob:bulkconv}.
\end{remark}

\begin{assumption}[PSWF spectral localization]\label{ass:specloc}
For fixed $\delta\in(0,1)$ and $p\ge 1$, there exist $C_p(\delta)>0$
such that for all $|m-n|\ge\delta N$, $0\le m,n\le N$, and $k\ge N$:
\begin{equation}\label{eq:specloc}
  |a_k^{mn}|
  \le \frac{C_p(\delta)}{(1+|\chi_k(c)-\muhat|)^p},
  \qquad \muhat := \chi_m(c)+\chi_n(c).
\end{equation}
\end{assumption}

\begin{remark}[Interpretation of Assumption~\ref{ass:specloc}]
\label{rem:specloc-interp}
$\{a_k^{mn}\}$ is pointwise localized near
$k^*: \chi_{k^*} \approx \muhat$.
This is the PSWF analogue of Clebsch--Gordan
decomposition; see Problem~\ref{prob:cg}.
\end{remark}

\begin{remark}[Bulk--Off-diagonal connection]
\label{rem:bulk-od-connection}
For $m,n \le \gamma N$ with $\gamma < 1/\sqrt{2}$,
$k^* < N$; hence Assumption~\ref{ass:specloc} implies
exponential tail decay, showing that
Assumption~\ref{ass:bulkconv} is a Fourier-side
consequence of spectral localization.
\end{remark}

\subsection{Out-of-band energy and Slepian identity}

\begin{definition}\label{def:eout}
$\Eout(f) := \int_{|\xi|>\omega}|\hat f(\xi)|^2\,d\xi$.
\end{definition}

\begin{lemma}[Slepian energy identity]\label{lem:slepian}
For every $f\in L^2(\mathbb{R})$,
with $a_k = \ip{f}{\psi_k^{(c)}}_{L^2([-T,T])}$:
\begin{align}
  \textstyle\sum_k\lambda_k(c)|a_k|^2
  &= \textstyle\int_{-\omega}^{\omega}|\hat f|^2\,d\xi,
  \label{eq:sei-in}\\
  \textstyle\sum_k(1-\lambda_k(c))|a_k|^2
  &= \Eout(f),\label{eq:sei-out}
\end{align}
and consequently:
$\|(I-\PN)f\|_{L^2}^2 \le \Eout(f)/(1-\lambda_{N-1}(c))$.
\end{lemma}

\begin{proof}
$\Kop = R_T B_\omega R_T$ gives
$\int_{-\omega}^\omega|\hat f|^2 = \ip{\Kop\tilde f}{\tilde f}
= \sum_k\lambda_k|a_k|^2$;
subtract from Parseval for~\eqref{eq:sei-out};
for $k\ge N$, $(1-\lambda_{N-1})\|(I-\PN)f\|^2 \le \Eout(f)$.
\end{proof}

\subsection{Band-edge convolution estimate}

\begin{lemma}[Support geometry]\label{lem:geom}
For $g_m,g_n$ supported in $[-\omega,\omega]$ and
$\xi\in(\omega,\omega+\delta)$:
$(g_m*g_n)(\xi) = \int_{\xi-\omega}^{\omega}g_m(\xi-\eta)g_n(\eta)\,d\eta$;
every contributing $\eta$ satisfies $|\eta|\in(\omega-\delta,\omega)$.
\end{lemma}

\begin{lemma}[Band-edge bound]\label{lem:bandedge}
For $g_m,g_n$ supported in $[-\omega,\omega]$:
$\|g_m*g_n\|_{L^2(|\xi|>\omega)}^2 \le 4\omega\|g_m\|_{L^2}^2\|g_n\|_{L^2}^2$.
For PSWFs: $\Eout(f_{mn}) \le 4\omega$.
\end{lemma}

\begin{proof}
Cauchy--Schwarz on the convolution integral for $\xi\in(\omega,2\omega)$,
then Fubini.
\end{proof}

\section{Bulk Regime}\label{sec:bulk}
%%=================================================================

\begin{theorem}[Bulk spectral tail bound]\label{thm:bulk}
Under Assumptions~\ref{ass:lw} and~\ref{ass:bulkconv},
for $m,n \le \gamma N$, $N \le (2/\pi-\varepsilon)c$,
and $c$ sufficiently large:
$\|(I-\PN)f_{mn}\|_{L^2} \le C_\gamma e^{-\alpha_\gamma N}$.
\end{theorem}

\begin{proof}
Slepian: $\|(I-\PN)f_{mn}\|^2 \le \Eout(f_{mn})/(1-\lambda_{N-1})$.
$1/(1-\lambda_{N-1}) \le 2$ (Assumption~\ref{ass:lw});
$\Eout(f_{mn}) \le C_{\mathrm{bc}}e^{-\alpha_{\mathrm{bc}}c}$
(Assumption~\ref{ass:bulkconv});
$e^{-\alpha_{\mathrm{bc}}c} \le e^{-\alpha_\gamma N}$
(since $c \ge N/(2/\pi-\varepsilon)$).
\end{proof}

\begin{remark}[Without Assumption~\ref{ass:bulkconv}]
\label{rem:bulk-without}
Lemma~\ref{lem:bandedge} alone gives $\|(I-\PN)f_{mn}\|^2 \le 8\omega$:
bounded but non-decaying.
\end{remark}

\section{Off-Diagonal Regime}\label{sec:offdiag}
%%=================================================================

We prove three results of increasing conditional strength:
an unconditional uniform bound (Theorem~\ref{thm:offdiag}),
an unconditional mean spectral localization
(Proposition~\ref{prop:mean-loc}),
and a conditional algebraic decay
(Theorem~\ref{thm:offdiag-strong}).

\subsection{Near-eigenvector structure}\label{ssec:neareig}

\begin{lemma}[Near-eigenvector identity]\label{lem:near-eig}
For all $0 \le m,n \le N$:
\begin{equation}\label{eq:near-eig}
  D_c f_{mn} = \muhat\,f_{mn} + R_{mn},
  \qquad \muhat := \chi_m(c)+\chi_n(c),
\end{equation}
where
\begin{equation}\label{eq:Rmn-def}
  R_{mn}(x) := -2\,p(x)\,(\psi_m^{(c)})'(x)\,\overline{(\psin)'(x)},
  \qquad p(x) = 1 - \tfrac{x^2}{T^2},
\end{equation}
and in particular:
\begin{equation}\label{eq:Rmn-bound}
  \|R_{mn}\|_{L^2([-T,T])}
  \le 2\|(\psi_m^{(c)})'\|_{L^2}\|(\psin)'\|_{L^\infty}.
\end{equation}
\end{lemma}

\begin{proof}
The Sturm--Liouville product rule
$D_c(uv) = (D_c u)v + u(D_c v) - 2p(x)u'v'$
with $u = \psi_m^{(c)}$, $v = \overline{\psin}$ gives
$D_c f_{mn} = \chi_m f_{mn} + \chi_n f_{mn} + R_{mn}
= \muhat f_{mn} + R_{mn}$.
Bound~\eqref{eq:Rmn-bound}: $\|p\|_{L^\infty} \le 1$,
then Cauchy--Schwarz.
\end{proof}

\begin{remark}[Coefficient-level consequence]
\label{rem:near-eig-consequence}
For any $k$ with $\chi_k(c) \ne \muhat$, taking
$\ip{\cdot}{\psi_k^{(c)}}$ in~\eqref{eq:near-eig}:
$|\chi_k - \muhat|\cdot|a_k^{mn}|
= |\ip{R_{mn}}{\psi_k^{(c)}}| \le \|R_{mn}\|_{L^2}$,
so
$|a_k^{mn}| \le \|R_{mn}\|_{L^2}/|\chi_k-\muhat|$.
\end{remark}

\begin{remark}[Commutator growth and the obstruction to
pointwise decay]\label{rem:comm-growth}
Since $(\psi_k^{(c)})'$ satisfies
$\|(\psi_k^{(c)})'\|_{L^2} \asymp \chi_k(c)^{1/2} \asymp k$
(Sturm--Liouville estimate, \cite[Prop.~2.7]{OsipovRokhlinXiao}),
one has $\|R_{mn}\|_{L^2} = O(N^2)$ for $m,n = O(N)$.
After $p$ iterations of the near-eigenvector argument,
the coefficient bound becomes
$|a_k^{mn}| \lesssim \|R_{mn}^{(p)}\|_{L^2}/(c_2 N^2)^p
= O(N^{2p})/(c_2 N^2)^p = O(1)$.
The ratio does not decay: the near-eigenvector approach alone
cannot prove pointwise localization.
This is the precise obstruction behind Problem~\ref{prob:cg}.
\end{remark}

\begin{lemma}[Spectral gap for $k \ge N$]\label{lem:chi-gap}
Fix $\gamma \in (0,1/\sqrt{2})$. For $m,n \le \gamma N$,
$k \ge N$, and $c \ge c_0(\gamma)$:
\begin{equation}\label{eq:chi-gap}
  \chi_k(c) - \muhat \ge c_2(\gamma)\,N^2,
  \qquad c_2(\gamma) = (1-2\gamma^2)/2 > 0.
\end{equation}
\end{lemma}

\begin{proof}
From~\eqref{eq:chi-asym}: $\chi_k \ge N^2 + c^2/2$ for $k\ge N$;
$\muhat \le 2\gamma^2 N^2 + c^2 + O(N)$;
hence $\chi_k - \muhat \ge (1-2\gamma^2)N^2 - c^2/2 + O(N)$.
For $N \le (2/\pi)c$, the $c^2/2$ term is absorbed for
$c$ large.
\end{proof}

\subsection{Unconditional uniform bound}

\begin{theorem}[Off-diagonal uniform bound]\label{thm:offdiag}
For $|m-n| \ge \delta N$, $0 \le m,n \le N$:
\begin{equation}\label{eq:offdiag-uniform}
  \|(I-\PN)f_{mn}\|_{L^2}^2
  = \sum_{k\ge N}|a_k^{mn}|^2
  \le \|f_{mn}\|_{L^2}^2
  \le T.
\end{equation}
\end{theorem}

\begin{proof}
Bessel's inequality and
$\|f_{mn}\|_{L^2}^2 \le \|\psi_m^{(c)}\|_{L^2}^2
\|\psin\|_{L^\infty}^2 \le \lambda_m(c)\cdot C \le T$.
\end{proof}

\begin{remark}[Sharpness and information gap]
\label{rem:od-sharp}
The bound $T$ is independent of $|m-n|$ and $\delta$:
the off-diagonal condition provides no improvement here.
This is not a defect of the proof; Parseval alone cannot
see the spectral localization structure.
The first structural improvement is
Proposition~\ref{prop:mean-loc} below.
\end{remark}

\subsection{Mean spectral localization (unconditional)}

\begin{proposition}[Mean spectral localization]
\label{prop:mean-loc}
For all $0 \le m,n \le N$ with $f_{mn} \not\equiv 0$:
\begin{equation}\label{eq:mean-loc}
  \Emn[\chi_k]
  = \muhat + E_{mn},
\end{equation}
where
\begin{equation}\label{eq:mean-loc-error}
  E_{mn}
  := \frac{\ip{R_{mn}}{f_{mn}}_{L^2([-T,T])}}
           {\|f_{mn}\|_{L^2([-T,T])}^2},
\end{equation}
and
\begin{equation}\label{eq:mean-loc-bound}
  |E_{mn}|
  \le \frac{2\|(\psi_m^{(c)})'\|_{L^2([-T,T])}
              \|(\psin)'\|_{L^\infty([-T,T])}
              \|f_{mn}\|_{L^2([-T,T])}}
           {\|f_{mn}\|_{L^2([-T,T])}^2}
  = \frac{2\|(\psi_m^{(c)})'\|_{L^2}\|(\psin)'\|_{L^\infty}}
          {\|f_{mn}\|_{L^2}}.
\end{equation}
In particular:
\begin{equation}\label{eq:mean-loc-T}
  |E_{mn}|
  \le 2T^{-1/2}\|(\psi_m^{(c)})'\|_{L^2}\|(\psin)'\|_{L^\infty}.
\end{equation}
\end{proposition}

\begin{proof}
By the spectral theorem for $D_c$, and using
Definition~\ref{def:spectral-mean}:
\begin{equation}\label{eq:mean-loc-proof1}
  \sum_k\chi_k(c)|a_k^{mn}|^2
  = \ip{D_c f_{mn}}{f_{mn}}_{L^2([-T,T])}.
\end{equation}
This is the \emph{exact} spectral identity; no approximation
is involved.
Substituting~\eqref{eq:near-eig}:
\begin{equation}\label{eq:mean-loc-proof2}
  \ip{D_c f_{mn}}{f_{mn}}
  = \ip{\muhat f_{mn} + R_{mn}}{f_{mn}}
  = \muhat\|f_{mn}\|^2 + \ip{R_{mn}}{f_{mn}}.
\end{equation}
Dividing by $\|f_{mn}\|^2$ gives~\eqref{eq:mean-loc}.
Bound~\eqref{eq:mean-loc-bound}: Cauchy--Schwarz,
then $\|R_{mn}\|_{L^2} \le 2\|(\psi_m^{(c)})'\|_{L^2}
\|(\psin)'\|_{L^\infty}$ from~\eqref{eq:Rmn-bound},
then $\|f_{mn}\|_{L^2} \ge T^{-1/2}\|f_{mn}\|_{L^1}
\ge T^{-1/2}$ (or more directly $\|f_{mn}\|_{L^2}^{-1}
\le T^{1/2}\|f_{mn}\|_{L^2}^{-1}\|f_{mn}\|_{L^2}
= T^{1/2}$; cf.~\eqref{eq:offdiag-uniform}).
\end{proof}

\begin{remark}[This is a genuine unconditional theorem]
\label{rem:mean-loc-genuine}
Proposition~\ref{prop:mean-loc} requires no assumption beyond
the near-eigenvector identity (Lemma~\ref{lem:near-eig}),
which is a direct consequence of the Sturm--Liouville
product rule and holds for all $m,n$.
It is the \emph{first unconditional structural result}
stating that the PSWF coefficient sequence $\{a_k^{mn}\}$
is not spectrally arbitrary: its weighted mean
$\Emn[\chi_k]$ is close to $\muhat = \chi_m + \chi_n$.
\end{remark}

\begin{remark}[Spectral centering as PSWF Clebsch--Gordan, first moment]
\label{rem:mean-loc-cg}
In the classical analogues:
\begin{itemize}
  \item Fourier: $e^{2\pi imt}\cdot e^{2\pi int} = e^{2\pi i(m+n)t}$,
  so $\sum_k k|a_k|^2 = m+n$ exactly (no error).
  \item Spherical harmonics: $Y_m \cdot Y_n$ decomposes over
  $k \in [|m-n|, m+n]$ by Clebsch--Gordan, so
  $\sum_k k(k+1)|a_k|^2 = m(m+1)+n(n+1)$ exactly.
\end{itemize}
For PSWFs, Proposition~\ref{prop:mean-loc} gives the
first-moment analogue with a controlled error $E_{mn}$.
The error is not zero because PSWFs do not satisfy an
exact product formula; $E_{mn}$ vanishes if and only if
$\ip{R_{mn}}{f_{mn}} = 0$, i.e., if the commutator
$R_{mn}$ is orthogonal to $f_{mn}$.
\end{remark}

\begin{remark}[What sharper commutator bounds would give]
\label{rem:mean-loc-improvement}
The bound~\eqref{eq:mean-loc-T} has size
$O(\|(\psi_m^{(c)})'\|_{L^2}\|(\psin)'\|_{L^\infty})
= O(N^2)$
in the worst case (Remark~\ref{rem:comm-growth}).
Relative to $\muhat \asymp N^2 + c^2$, this is $O(1)$ relative error:
the centering is exact to leading order but the subleading correction
is not yet controlled.
Any improvement $\|R_{mn}\|_{L^2} = o(N^2)$ — e.g., via
cancellation in the inner product $\ip{R_{mn}}{f_{mn}}$,
or a better pointwise bound on $(\psin)'$ near the turning
point — would tighten~\eqref{eq:mean-loc-T}
and bring the result closer to the sharp Clebsch--Gordan statement.
This is the natural next analytical step toward
Problem~\ref{prob:cg}.
\end{remark}

\begin{corollary}[Spectral mass is not in the tail — in mean]
\label{cor:mean-tail}
For $m,n \le \gamma N$ with $\gamma < 1/\sqrt{2}$ and
$c$ sufficiently large:
\[
  \Emn[\chi_k] \le \muhat + C N^2
  < \chi_N(c) - c_2(\gamma)N^2 + CN^2.
\]
In particular, the spectral mean lies strictly below $\chi_N$
if the error $CN^2$ is small relative to the gap $c_2(\gamma)N^2$.
For $\gamma$ sufficiently small (concretely: $c_2(\gamma) > 2C/N^2$),
the spectral mass of $f_{mn}$ is mean-localized below index $N$.
\end{corollary}

\begin{proof}
$\Emn[\chi_k] = \muhat + E_{mn} \le \muhat + CN^2$.
By Lemma~\ref{lem:chi-gap}, $\chi_N \ge \muhat + c_2(\gamma)N^2$.
Hence $\Emn[\chi_k] \le \chi_N - (c_2(\gamma)-C/N^2\cdot N^2)$;
for $c_2(\gamma)N^2 > CN^2$, i.e., $c_2(\gamma) > C$,
we get strict inequality.
\end{proof}

\subsection{Conditional algebraic decay}

\begin{theorem}[Off-diagonal algebraic decay]
\label{thm:offdiag-strong}
Under Assumption~\ref{ass:specloc},
for $|m-n| \ge \delta N$, $0 \le m,n \le N$, and $p \ge 1$:
\[
  \|(I-\PN)f_{mn}\|_{L^2}
  \le \frac{C_p(\delta)}{(1+|m-n|)^p}.
\]
\end{theorem}

\begin{proof}
By Assumption~\ref{ass:specloc} and Lemma~\ref{lem:gap-od}:
$|a_k^{mn}| \le C_p'(|m-n|/\delta)^{-2p}$ for $k\ge N$;
Bessel gives the result.
\end{proof}

\begin{lemma}[Gap: $\muhat$ vs.\ tail eigenvalues]\label{lem:gap-od}
For $|m-n|\ge\delta N$, $k\ge N$:
$|\chi_k(c)-\muhat| \ge c_3(\delta)|m-n|^2$
for $c$ sufficiently large, with $c_3(\delta) = c_2/\delta^2$.
\end{lemma}

\begin{proof}
By Lemma~\ref{lem:chi-gap}: $\chi_k - \muhat \ge c_2 N^2$;
since $|m-n|\ge\delta N$, $N \le |m-n|/\delta$,
hence $c_2 N^2 \le c_2|m-n|^2/\delta^2$.
\end{proof}

\begin{remark}[Role of Assumption~\ref{ass:specloc}]
\label{rem:specloc-what}
Lemma~\ref{lem:gap-od} is unconditional: it shows the
concentration set $\{k: \chi_k \approx \muhat\}$ lies
far below the tail $\{k\ge N\}$.
Assumption~\ref{ass:specloc} asserts that the coefficients
$a_k^{mn}$ \emph{respect} this geometry pointwise.
Proposition~\ref{prop:mean-loc} confirms it in the first moment.
The gap between first-moment and pointwise
localization is exactly Problem~\ref{prob:cg}.
\end{remark}

\begin{corollary}[Partial result on Conjecture~3.1 of
\cite{PaperII_quad}]\label{cor:partial}
Under Assumptions~\ref{ass:bulkconv} and~\ref{ass:specloc},
$\|(I-\PN)f_{mn}\|_{L^2} \le C(e^{-\alpha N} + (1+|m-n|)^{-p})$
for all $m,n$ outside the edge regime.
Unconditionally: $\|(I-\PN)f_{mn}\|_{L^2} \le C T^{1/2}$.
The edge regime is not covered.
\end{corollary}

\section{Edge Regime and Obstruction}\label{sec:edge}
%%=================================================================

\subsection{Failure of both approaches}

For $m,n \sim N$: the Slepian denominator $1-\lambda_{N-1}=O(1)$,
giving only $O(1)$ from~\eqref{eq:tail-bound}.
The near-eigenvector target $\muhat \sim 2N^2+c^2 > \chi_N$,
so the concentration set $\{k:\chi_k \approx \muhat\}$
sits entirely within the tail $\{k \ge N\}$;
no spectral gap is available.

\subsection{Lower bound}

\begin{proposition}[Edge lower bound]\label{prop:bwdoubling}
For $n$ in the Landau--Widom transition zone
$|n-2c/\pi| \le c^{1/2}$:
$\Eout(|\psi_n^{(c)}|^2) \ge c_0 > 0$.
\end{proposition}

\begin{proof}
Widom asymptotics imply $\lambda_n \ge 1/4$.
A half-bandwidth argument shows
$\int_{\omega/2}^\omega|\hat\psi_n^{(c)}|^2 \ge 1/16$.
Convolution lower bound via Lemma~\ref{lem:geom}: $\Eout \ge c_0$.
\end{proof}

\begin{remark}[Unified view of the edge failure]
\label{rem:edge-both-fail}
At the edge, $\muhat > \chi_N$ and $\Eout = O(1)$ simultaneously.
Both are manifestations of bandwidth doubling:
$f_{mn} \in \mathrm{PW}_{2\omega}$ has Fourier energy above $\omega$
(Slepian side) and PSWF coefficient energy above $k=N$
(near-eigenvector side).
These are two sides of the same obstruction.
\end{remark}

\section{Open Problems}\label{sec:open}
%%=================================================================

\begin{problem}[Bulk convolution decay]\label{prob:bulkconv}
Prove Assumption~\ref{ass:bulkconv}.
Natural route: WKB turning-point decay
$|\hat\psi_n^{(c)}(\xi)| \le Ce^{-\eta c(\omega-|\xi|)}$
for $n \le \gamma N$ near $|\xi|=\omega$.
\end{problem}

\begin{problem}[PSWF Clebsch--Gordan]
\label{prob:cg}
Prove Assumption~\ref{ass:specloc}: the pointwise
spectral localization bound~\eqref{eq:specloc}.
Proposition~\ref{prop:mean-loc} establishes the
first-moment form; the gap to pointwise localization
requires sharper control on the commutator iterates
$R_{mn}^{(p)}$ (Remark~\ref{rem:comm-growth}).
Natural approach: semiclassical spectral localization for
products of approximate eigenfunctions of $D_c$.
\end{problem}

\begin{problem}[Sharp commutator estimate]
\label{prob:comm-sharp}
Prove $|\ip{R_{mn}}{f_{mn}}_{L^2}| = o(\muhat\|f_{mn}\|_{L^2}^2)$
as $N \to \infty$.
This would make the spectral centering in
Proposition~\ref{prop:mean-loc} asymptotically sharp
(relative error $o(1)$),
and is the natural next step before Problem~\ref{prob:cg}.
\end{problem}

\begin{problem}[Edge regime]
\label{prob:edge}
Prove or disprove exponential tail decay for $m,n \sim N$.
Proposition~\ref{prop:bwdoubling} rules out $\Eout$-based arguments;
any proof requires handling the doubled prolate subspace;
see Problem~\ref{prob:doubled}.
\end{problem}

\begin{problem}[Doubled prolate subspace]\label{prob:doubled}
For bandwidth $2\omega$, show
$\|f_{mn}-P_{2N,2c}f_{mn}\|_{L^2} \le Ce^{-\alpha N}$
uniformly over $0 \le m,n \le N$.
\end{problem}

\begin{problem}[Sharp bulk decay rate]
Determine $\lim_{\gamma\to 1^-}\alpha_{\mathrm{bc}}(\gamma)$.
\end{problem}
