\begin{abstract}
Let $\{\psin\}_{n \ge 0}$ be the prolate spheroidal wave functions
(PSWFs) on $[-T,T]$ with bandwidth parameter $\omega > 0$ and
time-bandwidth product $c = \omega T$. Let $\PN$ denote the
orthogonal projector in $L^2([-T,T])$ onto the prolate subspace
$V_{N,c} = \mathrm{span}\{\psi_0^{(c)}, \ldots, \psi_{N-1}^{(c)}\}$.
The companion paper \cite{PaperII_quad} requires a uniform tail bound
for the product family
$\FN := \{\psi_m^{(c)}\overline{\psin} : 0 \le m,n \le N\}$
as a key input to the verification of the quadrature compatibility
condition of \cite{PaperI}.

The central analytical tool is the \emph{Slepian energy identity}
(Lemma~\ref{lem:slepian}), which bounds the spectral tail
$\|(I-\PN)f\|_{L^2([-T,T])}^2$ directly by the out-of-band Fourier
energy $\Eout(f) := \int_{|\xi|>\omega}|\hat{f}(\xi)|^2\,d\xi$.
For the product $f_{mn} = \psi_m^{(c)}\overline{\psin}$, the Fourier
transform $\widehat{f_{mn}} = \hat\psi_m^{(c)} * \hat\psi_n^{(c)}$
is a convolution of two functions supported in $[-\omega,\omega]$,
so that $f_{mn} \in \mathrm{PW}_{2\omega}$ (bandwidth doubling).
The out-of-band energy $\Eout(f_{mn})$ is the sole object requiring
estimation; it is controlled by the \emph{band-edge convolution
estimate} (Lemma~\ref{lem:bandedge}), which gives $\Eout(f_{mn})
\le 4\omega$ uniformly -- an absolute constant that does not decay
with $c$. Exponential decay of $\Eout(f_{mn})$ in the bulk requires
Assumption~\ref{ass:bulkconv}.

We prove:
\begin{enumerate}[(I)]
  \item \textbf{Bulk regime} ($m, n \le \gamma N$, $\gamma < 1$ fixed):
  $\|(I-\PN)f_{mn}\|_{L^2} \le C_\gamma e^{-\alpha_\gamma N}$,
  \emph{conditional on Assumption~\ref{ass:bulkconv}},
  which asserts exponential decay of
  $\|\hat\psi_m^{(c)} * \hat\psi_n^{(c)}\|_{L^2(|\xi|>\omega)}$
  for bulk-index PSWFs.
  This assumption is identified as an explicit open problem
  (Problem~\ref{prob:bulkconv}) and is expected to follow from
  WKB-type analysis of the prolate Fourier transforms near the
  turning point $|\xi| = \omega$.
  \item \textbf{Off-diagonal regime} ($|m-n| \ge \delta N$,
  $\delta > 0$ fixed): algebraic decay
  $\|(I-\PN)f_{mn}\|_{L^2} \le C_p(1+|m-n|)^{-p}$ for all $p \ge 1$,
  via the Sturm--Liouville eigenvalue gap. This result is unconditional.
  \item \textbf{Edge regime obstruction} ($m, n \sim N$):
  Proposition~\ref{prop:bwdoubling} shows that
  $\Eout(f_{nn}) \ge c_0 > 0$ on the diagonal,
  ruling out exponential decay via any argument based on $\Eout$.
  This regime is an isolated open problem.
\end{enumerate}
No claim is made about zeros of $\zeta(s)$.
\end{abstract}

\section{Introduction}\label{sec:intro}

\subsection{Context and motivation}

The Prolate--Weil program \cite{PaperI,PaperII,PaperII_quad} seeks a
spectral realization of the Weil positivity criterion
\cite{Weil1952,CCM2025} via the prolate concentration operator
$\Kop$ and its associated Gram forms.
In \cite{PaperI}, the prolate Gram form
$Q^{(N)}(f) = \sum_{j=1}^N |f(p_j)|^2$ was shown to be coercive on
the prolate subspace $V_{N,c}$ under an explicit quadrature
compatibility condition (Assumption~3.10 of \cite{PaperI}).
The companion paper \cite{PaperII_quad} organized the verification
of that condition for XRY PSWF-Gauss quadrature into two conjectural
inputs; see Section~\ref{ssec:context} for a precise summary.

The first of these inputs is the PSWF product spectral tail conjecture:
for every $f_{mn} = \psi_m^{(c)}\overline{\psin} \in \FN$, one needs a
uniform bound on $\|(I-\PN)f_{mn}\|_{L^2([-T,T])}$.
This is the object of study in the present paper.

\subsection{Strategy: Slepian identity and bandwidth doubling}

The key insight is to reduce to bounding the out-of-band Fourier energy
$\Eout(f_{mn})$ via the Slepian energy identity (Lemma~\ref{lem:slepian}):
\[
  \|(I-\PN)f\|_{L^2([-T,T])}^2
  \le \frac{\Eout(f)}{1-\lambda_{N-1}(c)}.
\]
For $f_{mn}$, the convolution theorem gives
$\widehat{f_{mn}} = \hat\psi_m^{(c)} * \hat\psi_n^{(c)}$,
supported in $[-2\omega,2\omega]$ (bandwidth doubling).
The sole term to estimate is
$\Eout(f_{mn}) = \|\hat\psi_m^{(c)} * \hat\psi_n^{(c)}\|_{L^2(|\xi|>\omega)}^2$.
The band-edge convolution estimate (Lemma~\ref{lem:bandedge}) gives
$\Eout(f_{mn}) \le 4\omega$, an absolute constant.
Exponential control of this term in the bulk regime requires
Assumption~\ref{ass:bulkconv}.

\subsection{Main results}

\begin{theorem}[Bulk regime; Theorem~\ref{thm:bulk}]
Under Assumption~\ref{ass:bulkconv}, for $m, n \le \gamma N$
with $\gamma \in (0,1)$ fixed,
$N \le (2/\pi-\varepsilon)c$, and $c$ sufficiently large:
\[
  \|(I-\PN)f_{mn}\|_{L^2([-T,T])} \le C_\gamma e^{-\alpha_\gamma N}.
\]
\end{theorem}

\begin{theorem}[Off-diagonal regime; Theorem~\ref{thm:offdiag}]
For $|m-n| \ge \delta N$ with $\delta > 0$ fixed, $0 \le m,n \le N$,
and every integer $p \ge 1$:
\[
  \|(I-\PN)f_{mn}\|_{L^2([-T,T])} \le C_p(1+|m-n|)^{-p}.
\]
\end{theorem}

Together these give Corollary~\ref{cor:partial}: Conjecture~3.1
of \cite{PaperII_quad} holds in all index regimes except the
edge $\{m,n \ge (1-\delta)N,\, |m-n| \le \delta N\}$,
conditionally on Assumption~\ref{ass:bulkconv} in the bulk.

\subsection{Relation to the companion paper}\label{ssec:context}

In \cite{PaperII_quad}, the verification of Assumption~3.10 of
\cite{PaperI} for XRY PSWF-Gauss quadrature was reduced to:
\begin{enumerate}[(C1)]
  \item \emph{PSWF product spectral tail} (Conjecture~3.1 of
  \cite{PaperII_quad}): uniform bound on
  $\|(I-\PN)f_{mn}\|_{L^2}$ for $0 \le m,n \le N$.
  \item \emph{XRY stability} (Conjecture~4.1 of \cite{PaperII_quad}):
  stability of the XRY quadrature functional on $\FN$.
\end{enumerate}
The present paper addresses (C1). The bulk result is conditional on
Assumption~\ref{ass:bulkconv}. The edge regime remains open.

\subsection{What this paper does not claim}

No statement is made about zeros of $\zeta(s)$.
We do not prove (C1) uniformly over all $0 \le m,n \le N$.
We do not prove Assumption~\ref{ass:bulkconv}; it is formulated
as an explicit open problem (Problem~\ref{prob:bulkconv}).
We do not address (C2).

\subsection{Organization}

Section~\ref{sec:prelim} recalls PSWF conventions, establishes the

Section~\ref{sec:tools} establishes the Slepian energy identity

\section{Preliminaries}\label{sec:prelim}

\subsection{PSWF conventions}

We follow the standard setup for prolate spheroidal wave functions;
see \cite{Slepian1978,OsipovRokhlinXiao}.
Fix $\omega > 0$, $T > 0$, and $c = \omega T > 0$.
We use the Fourier convention
$\hat{f}(\xi) = \int_{\mathbb{R}} f(x)e^{-2\pi i\xi x}\,dx$.

The \emph{prolate concentration operator} is the composition
\begin{equation}\label{eq:Kop_def}
  \Kop = R_T \circ B_\omega \circ R_T,
\end{equation}
where $B_\omega := \mathcal{F}^{-1}\mathbf{1}_{[-\omega,\omega]}\mathcal{F}$
is the ideal bandlimiting projection on $L^2(\mathbb{R})$,
and $R_T$ denotes multiplication by $\mathbf{1}_{[-T,T]}$:
$(R_T f)(x) := f(x)\mathbf{1}_{[-T,T]}(x)$.
Explicitly:
\[
  (\Kop f)(x)
  = \int_{-T}^{T}\frac{\sin 2\pi\omega(x-y)}{\pi(x-y)}\,f(y)\,dy,
  \qquad f \in L^2(\mathbb{R}).
\]
As a self-adjoint compact operator on $L^2([-T,T])$, $\Kop$ has
eigenvalues $1 > \lambda_0(c) > \lambda_1(c) > \cdots > 0$
with orthonormal eigenfunctions $\{\psin\}_{n\ge 0}$ satisfying
\[
  \ip{\psi_m^{(c)}}{\psin}_{L^2([-T,T])} = \lambda_n(c)\,\delta_{mn}.
\]

\subsection{Time concentration and leakage}

The eigenvalues $\lambda_n(c)$ measure \emph{time concentration}:
\begin{equation}\label{eq:time_conc}
  \lambda_n(c) = \|\psin\|_{L^2([-T,T])}^2.
\end{equation}
Define the \emph{time-domain leakage}:
\begin{equation}\label{eq:leakage}
  \ell_n := 1 - \lambda_n(c)
  = \|\psin\|_{L^2(\mathbb{R}\setminus[-T,T])}^2 \ge 0.
\end{equation}

\begin{remark}[Leakage is a time-domain quantity]\label{rem:leakage}
The quantity $\ell_n$ is exclusively a \emph{time-domain} object:
it measures the $L^2$-mass of $\psin$ outside $[-T,T]$.
No Fourier-domain interpretation of $\ell_n$ is used anywhere
in this paper. In particular, $\ell_n$ is \emph{not} identified with
any integral of $|\hat\psin|^2$ outside $[-\omega,\omega]$;
such an integral vanishes identically by
Section~\ref{ssec:bandlimited} below.
\end{remark}

\subsection{Bandlimited structure}\label{ssec:bandlimited}

Each $\psin$ belongs to the Paley--Wiener space $\mathrm{PW}_\omega$:
\begin{equation}\label{eq:bandlimited}
  \operatorname{supp}\hat\psin \subset [-\omega,\omega],
  \qquad
  \|\hat\psin\|_{L^2(\mathbb{R})}^2 = 1.
\end{equation}

\begin{remark}[No Fourier leakage for individual PSWFs]
\label{rem:no_fourier_leak}
By~\eqref{eq:bandlimited},
$\int_{|\xi|>\omega}|\hat\psin(\xi)|^2\,d\xi = 0$ exactly.
Individual PSWFs are exactly bandlimited: all their $L^2$-energy
lies in $[-\omega,\omega]$. This is structurally independent of
the time-domain leakage $\ell_n > 0$, and is the reason
the $g_k + r_k$ Fourier decomposition used in earlier drafts
is ill-founded: $r_k \equiv 0$ identically.
\end{remark}

\subsection{Product family and bandwidth doubling}

\begin{definition}[Product family]\label{def:product_family}
For $N \ge 1$ and $0 \le m,n \le N$, set
$f_{mn} := \psi_m^{(c)}\overline{\psin}$.
The \emph{product family} is $\FN := \{f_{mn}: 0 \le m,n \le N\}$.
\end{definition}

By the convolution theorem and~\eqref{eq:bandlimited}:
\begin{equation}\label{eq:product_fourier}
  \widehat{f_{mn}} = \hat\psi_m^{(c)} * \hat\psin,
  \qquad
  \operatorname{supp}\widehat{f_{mn}} \subset [-2\omega,2\omega],
  \qquad
  f_{mn} \in \mathrm{PW}_{2\omega}.
\end{equation}

\begin{remark}[Bandwidth doubling as the structural source]
\label{rem:bwdoubling}
While individual PSWFs lie in $\mathrm{PW}_\omega$, their products
lie in $\mathrm{PW}_{2\omega}$ but not in $\mathrm{PW}_\omega$ in general.
The out-of-band energy
\[
  \Eout(f_{mn})
  := \int_{|\xi|>\omega}|\widehat{f_{mn}}(\xi)|^2\,d\xi
  = \|\hat\psi_m^{(c)} * \hat\psin\|_{L^2(|\xi|>\omega)}^2
\]
is strictly positive in general, and is the \emph{sole} term
requiring estimation throughout this paper.
This is not an approximation artefact but an exact structural
property of the convolution.
\end{remark}

\subsection{Landau--Widom asymptotics}

\begin{assumption}[Landau--Widom]\label{ass:lw}
Fix $\varepsilon \in (0,2/\pi)$ and $N \le (2/\pi-\varepsilon)c$.
There exist $C(\varepsilon),\eta(\varepsilon) > 0$ and
$c_0(\varepsilon) > 0$ such that for all $c \ge c_0(\varepsilon)$:
\[
  \ell_n \le C(\varepsilon)\,e^{-\eta(\varepsilon)c}
  \qquad\text{for all } n \le N-1.
\]
In particular $\lambda_{N-1}(c) > 1/2$ and
$1/(1-\lambda_{N-1}(c)) \le 2$.
\end{assumption}

See \cite[Thm.~1]{Widom1964} and \cite[Ch.~4]{OsipovRokhlinXiao}.

\subsection{Bulk convolution decay assumption}

\begin{assumption}[Bulk convolution decay]\label{ass:bulkconv}
Fix $\gamma \in (0,1)$.
There exist $C_{\mathrm{bc}}(\gamma), \alpha_{\mathrm{bc}}(\gamma) > 0$
and $c_0(\gamma) > 0$ such that for all $m,n \le \gamma N$
and $c \ge c_0(\gamma)$:
\[
  \Eout(f_{mn})
  = \|\hat\psi_m^{(c)} * \hat\psin\|_{L^2(|\xi|>\omega)}^2
  \le C_{\mathrm{bc}}\,e^{-\alpha_{\mathrm{bc}}\,c}.
\]
\end{assumption}

\begin{remark}[Status and sufficient condition for
Assumption~\ref{ass:bulkconv}]\label{rem:bulkconv-status}
By the support geometry of Lemma~\ref{lem:geom}, the integrand
$|(\hat\psi_m^{(c)} * \hat\psin)(\xi)|^2$ for $\xi$ near $\omega^+$
draws exclusively from the near-edge values
$\hat\psi_m^{(c)}(\eta)$, $\hat\psin(\eta)$ with $|\eta|$
near $\omega^-$.
A sufficient condition for Assumption~\ref{ass:bulkconv} is therefore
a uniform \emph{turning-point decay estimate}: the existence of
$C, \eta > 0$ such that for all $n \le \gamma N$
and $|\xi|$ sufficiently close to $\omega$,
\begin{equation}\label{eq:wkb_suff}
  |\hat\psi_n^{(c)}(\xi)| \le C\,e^{-\eta\,c\,(\omega-|\xi|)}.
\end{equation}
The point $|\xi| = \omega$ is a turning point of the prolate
Sturm--Liouville equation in the Fourier representation.
Standard WKB analysis is \emph{consistent with} amplitude suppression
of this form near $\pm\omega$ when $n$ is well below the
Slepian cutoff, but to our knowledge no quantitative form
of~\eqref{eq:wkb_suff} sufficient to imply
Assumption~\ref{ass:bulkconv} through the convolution has been
established in the literature.
Assumption~\ref{ass:bulkconv} is therefore an explicit open
problem; see Problem~\ref{prob:bulkconv}.
\end{remark}

\section{Analytical Tools}\label{sec:tools}
%%=================================================================

\subsection{Out-of-band energy}

\begin{definition}[Out-of-band energy]\label{def:eout}
For $f \in L^2(\mathbb{R})$, define
\[
  \Eout(f) := \int_{|\xi|>\omega}|\hat{f}(\xi)|^2\,d\xi.
\]
\end{definition}

This quantity is zero for any $f \in \mathrm{PW}_\omega$ (in
particular for individual PSWFs by~\eqref{eq:bandlimited}),
and strictly positive for generic elements of $\mathrm{PW}_{2\omega}$
(in particular for the product family $\FN$).

\subsection{Slepian energy identity}

\begin{lemma}[Slepian energy identity]\label{lem:slepian}
For every $f \in L^2(\mathbb{R})$ and $N \ge 1$:
\begin{align}
  \sum_{k=0}^{\infty}\lambda_k(c)\,
  |\ip{f}{\psi_k^{(c)}}_{L^2([-T,T])}|^2
  &= \int_{-\omega}^{\omega}|\hat{f}(\xi)|^2\,d\xi,
  \label{eq:sei-in}\\
  \sum_{k=0}^{\infty}(1-\lambda_k(c))\,
  |\ip{f}{\psi_k^{(c)}}_{L^2([-T,T])}|^2
  &= \Eout(f),
  \label{eq:sei-out}
\end{align}
and consequently:
\begin{equation}\label{eq:tail-bound}
  \|(I-\PN)f\|_{L^2([-T,T])}^2
  \le \frac{\Eout(f)}{1-\lambda_{N-1}(c)}.
\end{equation}
\end{lemma}

\begin{proof}
Recall from~\eqref{eq:Kop_def} that
$\Kop = R_T \circ B_\omega \circ R_T$,
where $B_\omega = \mathcal{F}^{-1}\mathbf{1}_{[-\omega,\omega]}\mathcal{F}$.
For $f \in L^2(\mathbb{R})$, let $\tilde f := R_T f \in L^2([-T,T])$.
Then:
\[
  \int_{-\omega}^{\omega}|\hat{f}(\xi)|^2\,d\xi
  = \ip{B_\omega f}{f}_{L^2(\mathbb{R})}
  = \ip{\Kop \tilde f}{\tilde f}_{L^2([-T,T])}
  = \sum_k \lambda_k(c)\,
  |\ip{\tilde f}{\psi_k^{(c)}}_{L^2([-T,T])}|^2,
\]
where the second equality uses $R_T B_\omega = \Kop R_T$ and
self-adjointness of $R_T$, and the third expands $\tilde f$
in the eigenbasis of $\Kop$.
This gives~\eqref{eq:sei-in}.
Identity~\eqref{eq:sei-out} follows by subtracting~\eqref{eq:sei-in}
from Parseval:
$\sum_k|\ip{\tilde f}{\psi_k^{(c)}}|^2 = \|\tilde f\|_{L^2([-T,T])}^2
= \|f\|_{L^2([-T,T])}^2$.
Bound~\eqref{eq:tail-bound}: for $k \ge N$,
$1-\lambda_k(c) \ge 1-\lambda_{N-1}(c) > 0$, so
\[
  (1-\lambda_{N-1}(c))\|(I-\PN)f\|_{L^2([-T,T])}^2
  \le \sum_{k \ge N}(1-\lambda_k(c))
  |\ip{f}{\psi_k^{(c)}}|^2 \le \Eout(f). \qedhere
\]
\end{proof}

\begin{remark}[Scope of application]\label{rem:slepian-scope}
Bound~\eqref{eq:tail-bound} is applied throughout this paper to
$f = f_{mn} \in \mathrm{PW}_{2\omega}$, for which $\Eout(f_{mn}) > 0$.
For $f = \psin$ with $n < N$: since $\psin \in V_{N,c}$ by definition,
$(I-\PN)\psin = 0$ trivially; consistently, $\Eout(\psin) = 0$
by~\eqref{eq:bandlimited}.
\end{remark}

\subsection{Geometric support restriction}

\begin{lemma}[Support geometry]\label{lem:geom}
Let $g_m, g_n \in L^2(\mathbb{R})$ with supports in
$[-\omega,\omega]$. For $\xi \in (\omega, 2\omega)$:
\[
  (g_m * g_n)(\xi)
  = \int_{\xi-\omega}^{\omega} g_m(\xi-\eta)\,g_n(\eta)\,d\eta.
\]
In particular, for $\xi \in (\omega, \omega+\delta)$, every
contributing $\eta$ satisfies
$\eta \in (\xi-\omega, \omega) \subset (\omega-\delta,\omega)$.
The symmetric statement holds for $\xi \in (-2\omega,-\omega)$.
\end{lemma}

\begin{proof}
$g_m(\xi-\eta) = 0$ unless $\xi-\eta \in [-\omega,\omega]$,
i.e.\ $\eta \ge \xi-\omega$; combined with
$\operatorname{supp}g_n \subset [-\omega,\omega]$, the domain
is $[\xi-\omega,\omega]$.
For $\xi > \omega$ this interval has positive length iff
$\xi < 2\omega$, and its lower endpoint $\xi-\omega > 0$.
The near-edge localization follows immediately.
\end{proof}

\subsection{Band-edge convolution estimate}

\begin{lemma}[Band-edge bound]\label{lem:bandedge}
For $g_m, g_n \in L^2(\mathbb{R})$ supported in $[-\omega,\omega]$:
\begin{equation}\label{eq:bandedge}
  \|g_m * g_n\|_{L^2(|\xi|>\omega)}^2
  \le 4\omega\,\|g_m\|_{L^2}^2\,\|g_n\|_{L^2}^2.
\end{equation}
For PSWFs, $\|\hat\psi_k^{(c)}\|_{L^2}^2 = 1$, yielding:
\begin{equation}\label{eq:eout-uniform}
  \Eout(f_{mn}) \le 4\omega.
\end{equation}
\end{lemma}

\begin{proof}
By Lemma~\ref{lem:geom} and Cauchy--Schwarz, for
$\xi \in (\omega, 2\omega)$:
\[
  |(g_m*g_n)(\xi)|^2
  \le (2\omega-\xi)
  \int_{\xi-\omega}^{\omega}|g_m(\xi-\eta)|^2|g_n(\eta)|^2\,d\eta.
\]
Integrate over $\xi \in (\omega,2\omega)$ and apply Fubini
using the domain $\{0 < \eta < \omega,\; \omega < \xi < \omega+\eta\}$;
substitute $u = \xi-\eta$:
\[
  \|g_m*g_n\|_{L^2(\omega,2\omega)}^2
  \le 2\omega\,\|g_m\|_{L^2}^2\,\|g_n\|_{L^2}^2.
\]
The $(-2\omega,-\omega)$ half is symmetric, giving~\eqref{eq:bandedge}.
\end{proof}

\begin{remark}[$4\omega$ is not exponentially small]
\label{rem:4omega}
The bound~\eqref{eq:eout-uniform} is uniform in $c$, $m$, $n$.
It does not decay as $c \to \infty$. Exponential control of
$\Eout(f_{mn})$ in the bulk therefore requires additional structure
beyond the support geometry, captured in Assumption~\ref{ass:bulkconv}.
\end{remark}

%%=================================================================
\section{Bulk Regime: Conditional Exponential Tail Decay}
\label{sec:bulk}
%%=================================================================

The bulk regime concerns pairs $(m,n)$ with $m,n \le \gamma N$
for a fixed $\gamma \in (0,1)$.
The proof uses exactly three inputs:
the Slepian energy identity (Lemma~\ref{lem:slepian}),
the Landau--Widom bound (Assumption~\ref{ass:lw}),
and the bulk convolution decay assumption
(Assumption~\ref{ass:bulkconv}).
No Fourier-domain property of individual PSWFs beyond
exact bandlimitedness~\eqref{eq:bandlimited} is used.

\begin{theorem}[Bulk spectral tail bound]\label{thm:bulk}
Let $\gamma \in (0,1)$ and $\varepsilon \in (0,2/\pi)$.
Assume Assumptions~\ref{ass:lw} and~\ref{ass:bulkconv},
and suppose $N \le (2/\pi - \varepsilon)c$.
Then there exist $C_\gamma, \alpha_\gamma > 0$ depending only on
$\gamma$, $\varepsilon$, $C_{\mathrm{bc}}$, $\alpha_{\mathrm{bc}}$,
such that for all $c \ge c_0(\gamma, \varepsilon)$
and all $m,n \le \gamma N$:
\[
  \|(I - \PN)f_{mn}\|_{L^2([-T,T])}
  \le C_\gamma\,e^{-\alpha_\gamma N}.
\]
\end{theorem}

\begin{proof}
\textbf{Step 1: Apply the Slepian bound.}
By Lemma~\ref{lem:slepian} (bound~\eqref{eq:tail-bound}):
\begin{equation}\label{eq:bulk-slepian}
  \|(I-\PN)f_{mn}\|_{L^2([-T,T])}^2
  \le \frac{\Eout(f_{mn})}{1 - \lambda_{N-1}(c)}.
\end{equation}

\textbf{Step 2: Bound the denominator.}
By Assumption~\ref{ass:lw}, $\lambda_{N-1}(c) \to 1$
as $c \to \infty$ with $N \le (2/\pi-\varepsilon)c$;
for $c \ge c_0(\varepsilon)$, $\lambda_{N-1}(c) > 1/2$ and:
\begin{equation}\label{eq:bulk-denom}
  \frac{1}{1-\lambda_{N-1}(c)} \le 2.
\end{equation}

\textbf{Step 3: Bound the numerator.}
By Assumption~\ref{ass:bulkconv} with $m,n \le \gamma N$
and $c \ge c_0(\gamma)$:
\begin{equation}\label{eq:bulk-num}
  \Eout(f_{mn})
  = \|\hat\psi_m^{(c)} * \hat\psin\|_{L^2(|\xi|>\omega)}^2
  \le C_{\mathrm{bc}}(\gamma)\,e^{-\alpha_{\mathrm{bc}}(\gamma)\,c}.
\end{equation}

\textbf{Step 4: Combine.}
Inserting~\eqref{eq:bulk-denom} and~\eqref{eq:bulk-num}
into~\eqref{eq:bulk-slepian}:
\[
  \|(I-\PN)f_{mn}\|_{L^2([-T,T])}^2
  \le 2\,C_{\mathrm{bc}}(\gamma)\,e^{-\alpha_{\mathrm{bc}}(\gamma)\,c}.
\]

\textbf{Step 5: Convert $c$ to $N$.}
Since $N \le (2/\pi - \varepsilon)c$, we have
$c \ge N/(2/\pi - \varepsilon)$, so:
\[
  \|(I-\PN)f_{mn}\|_{L^2([-T,T])}^2
  \le 2\,C_{\mathrm{bc}}\,
  e^{-\alpha_{\mathrm{bc}}\cdot N/(2/\pi - \varepsilon)}.
\]
Taking square roots and setting
$C_\gamma := (2C_{\mathrm{bc}})^{1/2}$ and
$\alpha_\gamma := \tfrac{1}{2}\,\alpha_{\mathrm{bc}}/(2/\pi-\varepsilon)$
gives the result.
\end{proof}

\begin{remark}[Proof uses exactly three inputs]\label{rem:bulk-inputs}
The proof of Theorem~\ref{thm:bulk} uses:
\begin{enumerate}[(i)]
  \item Slepian identity: Lemma~\ref{lem:slepian}
        (operator-theoretic; exact)
  \item Denominator bound: Assumption~\ref{ass:lw}
        (Landau--Widom; classical)
  \item Numerator bound: Assumption~\ref{ass:bulkconv}
        (the open input; see Problem~\ref{prob:bulkconv})
\end{enumerate}
No other property of PSWFs is invoked.
In particular, no Fourier-domain structure of individual
$\psin$ beyond exact bandlimitedness~\eqref{eq:bandlimited}
is used anywhere in the proof.
\end{remark}

\begin{remark}[What fails without Assumption~\ref{ass:bulkconv}]
\label{rem:bulk-without}
Without Assumption~\ref{ass:bulkconv}, Lemma~\ref{lem:bandedge}
gives only $\Eout(f_{mn}) \le 4\omega$, and~\eqref{eq:bulk-slepian}
yields $\|(I-\PN)f_{mn}\|_{L^2}^2 \le 8\omega$:
bounded but non-decaying.
Assumption~\ref{ass:bulkconv} is precisely the additional input
needed to convert this into exponential decay.
\end{remark}

\begin{remark}[Degeneration as $\gamma \to 1^-$]
\label{rem:bulk-degeneration}
The rate $\alpha_\gamma = \tfrac{1}{2}\alpha_{\mathrm{bc}}(\gamma)
/(2/\pi-\varepsilon)$ depends on $\gamma$ through
$\alpha_{\mathrm{bc}}(\gamma)$.
If $\alpha_{\mathrm{bc}}(\gamma) \to 0$ as $\gamma \to 1^-$
(expected, since bulk PSWFs approach the Slepian transition
zone), then $\alpha_\gamma \to 0$ and the exponential bound
degenerates, consistent with the edge obstruction of
Section~\ref{sec:edge}.
\end{remark}

\section{Off-Diagonal Regime: Algebraic Tail Decay}
\label{sec:offdiag}
%%=================================================================

\begin{lemma}[Eigenvalue gap estimate]\label{lem:gap}
For $0 \le m,n \le N$ with $m \ne n$, $|m-n| \ge \delta N$
($\delta > 0$ fixed), and $N \le (2/\pi)c$:
\begin{equation}\label{eq:gap}
  |\chi_m(c) - \chi_n(c)| \ge c_1\,|m-n|\cdot N,
\end{equation}
where $c_1 > 0$ is an absolute constant.
Since $|m-n| \ge \delta N$ implies $N \ge |m-n|/\delta$:
\begin{equation}\label{eq:gap2}
  |\chi_m(c) - \chi_n(c)| \ge \frac{c_1}{\delta}\,|m-n|^2.
\end{equation}
\end{lemma}

\begin{proof}
The Sturm--Liouville eigenvalues satisfy
$\chi_k(c) = k(k+1) + c^2/2 + O(k^{-1}c^2)$
\cite[Ch.~2]{OsipovRokhlinXiao}, so:
\[
  \chi_m(c) - \chi_n(c)
  = (m-n)(m+n+1) + O(N^{-1}c^2).
\]
For $m,n \le N$ with $N \le (2/\pi)c$, the error is
$O(N^{-1}c^2) = O(N)$.
The leading term satisfies $(m-n)(m+n+1) \ge |m-n|\cdot N/C_0$
for some absolute $C_0$, since $m+n+1 \ge \max(m,n) \ge N/2$
when $\max(m,n) \ge N/2$. For $c$ large enough the error is
absorbed, giving~\eqref{eq:gap}.
Bound~\eqref{eq:gap2} follows from $N \ge |m-n|/\delta$.
\end{proof}

\begin{theorem}[Off-diagonal spectral tail bound]\label{thm:offdiag}
Let $\delta \in (0,1)$ be fixed. For every integer $p \ge 1$
and all $m, n$ with $|m-n| \ge \delta N$, $0 \le m,n \le N$:
\[
  \|(I-\PN)f_{mn}\|_{L^2([-T,T])}
  \le C_p\,(1+|m-n|)^{-p},
\]
where $C_p > 0$ depends only on $p$, $\delta$, $\omega$, $T$,
and absolute constants.
\end{theorem}

\begin{proof}
Let $D_c$ denote the self-adjoint Sturm--Liouville operator on
$L^2([-T,T])$ with $D_c\psi_k^{(c)} = \chi_k(c)\psi_k^{(c)}$.

\textbf{Gap.} By Lemma~\ref{lem:gap}, since $|m-n| \ge \delta N$
implies $N \ge |m-n|/\delta$:
\[
  |\chi_m(c)-\chi_n(c)|
  \ge c_1\,|m-n|\cdot N
  \ge \frac{c_1}{\delta}\,|m-n|^2.
\]

\textbf{Decay of coefficients.}
Integrating by parts $p$ times using self-adjointness of $D_c$:
\[
  \ip{f_{mn}}{\psi_k^{(c)}}
  = \frac{1}{(\chi_m(c)-\chi_n(c))^p}
  \ip{D_c^{[p]}f_{mn}}{\psi_k^{(c)}}.
\]
The $L^2$-norm of $D_c^{[p]}f_{mn}$ is bounded by
$B_p \le C_p(CN^2)^p$.
By Cauchy--Schwarz and~\eqref{eq:gap}:
\[
  |\ip{f_{mn}}{\psi_k^{(c)}}|
  \le \frac{C_p(CN^2)^p}{\bigl(c_1|m-n|N\bigr)^p}
  = \frac{\hat{C}_p}{|m-n|^p}.
\]

\textbf{Conclusion.}
By Bessel's inequality:
\[
  \|(I-\PN)f_{mn}\|_{L^2}
  \le \frac{\hat{C}_p T}{(1+|m-n|)^p}.
\qedhere
\]
\end{proof}

\begin{corollary}[Partial verification of Conjecture~3.1 of
\cite{PaperII_quad}]\label{cor:partial}
Under Assumption~\ref{ass:bulkconv}, in the bulk and off-diagonal
regimes combined:
\[
  \|(I-\PN)f_{mn}\|_{L^2([-T,T])}
  \le C\bigl(e^{-\alpha N} + (1+|m-n|)^{-p}\bigr)
\]
for all $(m,n)$ with either $m,n \le \gamma N$ or $|m-n| \ge \delta N$.
The edge regime $\{m,n \ge (1-\delta)N,\;|m-n|<\delta N\}$
is not covered.
\end{corollary}

\section{Edge Regime and Obstruction to Uniform Bounds}
\label{sec:edge}
%%=================================================================

\subsection{Failure of the Slepian-identity argument}

For $m,n \sim N$ near the Slepian cutoff $N \approx 2c/\pi$,
the eigenvalue $\lambda_{N-1}(c)$ lies in the Landau--Widom
transition zone, where $\ell_n = 1-\lambda_n(c) = O(1)$.
Thus $\Eout(f_{mn}) = O(1)$, and the Slepian identity
\eqref{eq:tail-bound} yields only $\|(I-\PN)f_{mn}\|^2 = O(1)$.

\subsection{Failure of the oscillation argument}

For near-diagonal pairs in the edge zone, $|m-n| = O(1)$,
the gap $|\chi_m(c)-\chi_n(c)| = O(N)$ by Lemma~\ref{lem:gap},
giving only $O(N^{-p})$ coefficient bounds.
For the diagonal $m=n$, the product $f_{nn} = |\psin|^2$
is non-negative and no oscillation cancellation is available.

\subsection{The bandwidth doubling obstruction}

\begin{proposition}[Bandwidth doubling lower bound]\label{prop:bwdoubling}
For $n$ in the Landau--Widom transition zone, i.e.,
$|n - 2c/\pi| \le c^{1/2}$:
\[
  \Eout(|\psi_n^{(c)}|^2) \ge c_0 > 0,
\]
where $c_0 > 0$ is an absolute constant independent of $c$ and $n$.
\end{proposition}

\begin{proof}
By Widom transition-zone asymptotics \cite[Ch.~4]{OsipovRokhlinXiao},
$\lambda_n(c) \ge 1/4$ for $c$ sufficiently large.

\textbf{Step 1: Positive Fourier mass near the band edge.}
We claim:
\begin{equation}\label{eq:edge-mass-lower}
  \int_{\omega/2}^{\omega}|\hat\psi_n^{(c)}(\xi)|^2\,d\xi \ge \delta_0
\end{equation}
for an absolute constant $\delta_0 > 0$.

Suppose for contradiction that
$\int_{\omega/2}^\omega|\hat\psi_n^{(c)}|^2 < \delta_0$.
Since $\int_{-\omega}^\omega|\hat\psi_n^{(c)}|^2 = \lambda_n(c) \ge 1/4$,
the Fourier mass is concentrated on $(-\omega/2, \omega/2)$:
\[
  \int_{-\omega/2}^{\omega/2}|\hat\psi_n^{(c)}(\xi)|^2\,d\xi
  \ge \tfrac{1}{4} - 2\delta_0.
\]
Now the half-bandwidth concentration operator with bandwidth $\omega/2$
and time-bandwidth product $c_{1/2} = c/2$ has Slepian cutoff
$N_{1/2} \approx c/\pi$.
Since $n \approx 2c/\pi \gg c/\pi$, the index $n$ lies deep in
the superexponential decay regime for bandwidth $\omega/2$.
By Assumption~\ref{ass:lw} applied with $c_{1/2} = c/2$:
\[
  \int_{-\omega/2}^{\omega/2}|\hat\psi_n^{(c)}(\xi)|^2\,d\xi
  \le Ce^{-\eta c/2} \to 0,
\]
contradicting $\ge \tfrac{1}{4} - 2\delta_0 > 0$ for $\delta_0 < 1/8$.
Hence~\eqref{eq:edge-mass-lower} holds with $\delta_0 = 1/16$.

\textbf{Step 2: Lower bound for out-of-band convolution energy.}
Let $h := \hat\psi_n^{(c)} * \hat\psi_n^{(c)}$.
By Lemma~\ref{lem:geom}, for $\xi \in (\omega, 3\omega/2)$,
a direct estimate using~\eqref{eq:edge-mass-lower} gives:
\[
  \int_\omega^{2\omega}|h(\xi)|^2\,d\xi
  \ge \frac{\delta_0^2}{\omega} > 0.
\]

\textbf{Step 3: Conclusion.}
$\Eout(|\psi_n^{(c)}|^2)
\ge \delta_0^2/\omega =: c_0 > 0$.
\end{proof}

\begin{remark}[Structural consequence]
Proposition~\ref{prop:bwdoubling} shows that the diagonal products
$|\psi_n^{(c)}|^2$ in the edge regime are genuinely not approximable
by the prolate subspace $V_{N,c}$ with exponentially small error
via any argument based solely on $\Eout$.
\end{remark}

\begin{remark}[Why new tools are needed]
Any proof for the edge regime must address at least one of:
\begin{enumerate}[(a)]
  \item Controlling $f_{mn}$ in the doubled prolate subspace
  $V_{2N,2c}$ and relating $P_{2N,2c}$ back to $\PN$.
  \item Exploiting a product formula for PSWFs (currently unavailable
  \cite{OsipovRokhlinXiao}).
  \item Reformulating the quadrature condition to avoid the full
  product family.
\end{enumerate}
Route~(a) is addressed in Problem~\ref{prob:doubled}.
\end{remark}

\section{Open Problems}\label{sec:open}
%%=================================================================

\begin{problem}[Bulk convolution decay]\label{prob:bulkconv}
Prove Assumption~\ref{ass:bulkconv}: for fixed $\gamma \in (0,1)$
and all $m,n \le \gamma N$, $c \ge c_0(\gamma)$:
\[
  \|\hat\psi_m^{(c)} * \hat\psin\|_{L^2(|\xi|>\omega)}^2
  \le C_{\mathrm{bc}}(\gamma)\,e^{-\alpha_{\mathrm{bc}}(\gamma)c}.
\]
A natural route is via a uniform turning-point decay estimate
$|\hat\psi_n^{(c)}(\xi)| \le Ce^{-\eta c(\omega-|\xi|)}$
for $|\xi|$ near $\omega$ and $n \le \gamma N$; see
Remark~\ref{rem:bulkconv-status}.
\end{problem}

\begin{problem}[Edge regime: uniform tail bound]\label{prob:edge}
Prove or disprove: there exist $C, \alpha > 0$ such that
$\|(I-\PN)f_{mn}\|_{L^2} \le Ce^{-\alpha N}$
for all $m,n \sim N$.
Proposition~\ref{prop:bwdoubling} rules out exponential decay
on the diagonal via $\Eout$-based arguments.
\end{problem}

\begin{problem}[Doubled prolate subspace]\label{prob:doubled}
Let $V_{2N,2c}$ be the prolate subspace for bandwidth $2\omega$.
Show that every $f_{mn}$ with $0 \le m,n \le N$ satisfies
$\|f_{mn} - P_{2N,2c}f_{mn}\|_{L^2} \le Ce^{-\alpha N}$
uniformly.
\end{problem}

\begin{problem}[Sharp bulk decay rate]
Determine the sharp rate $\alpha_\gamma$ as a function of
$\gamma$ and $\varepsilon$; in particular, whether
$\alpha_\gamma \to 0$ as $\gamma \to 1^-$.
\end{problem}

\begin{problem}[Transition in the diagonal tail]
For fixed $N \sim c$ and $n$ varying from $0$ to $N$,
determine the precise asymptotic behavior of
$\|(I-\PN)|\psi_n^{(c)}|^2\|_{L^2}$.
\end{problem}
