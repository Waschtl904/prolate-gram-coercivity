\title[PSWF Product Spectral Tail Estimates]
      {PSWF Product Spectral Tail Estimates:\\
       Bulk and Off-Diagonal Regimes,\\
       and an Obstruction at the Spectral Edge\\[4pt]
       \large (Paper~III in the Prolate--Weil Program)}

\thanks{This paper is the third in the Prolate--Weil program.
Paper~I \cite{PaperI} established frame coercivity of the prolate Gram
form under a quadrature compatibility assumption (Assumption~3.10 of
\cite{PaperI}).
The companion paper \cite{PaperII_quad} organised the verification of
that assumption for XRY PSWF-Gauss quadrature into two explicit
conjectural inputs: a PSWF product spectral tail conjecture and an
XRY stability conjecture. The present paper addresses the first of
these two inputs. We prove the spectral tail bound in the bulk and
off-diagonal regimes via a Fourier-side argument based on the
Slepian energy identity. The edge regime $m,n \sim N$ is identified
as a structural obstruction and left as an explicit open problem.}

\begin{abstract}
Let $\{\psin\}_{n \ge 0}$ be the prolate spheroidal wave functions
(PSWFs) on $[-T,T]$ with bandwidth parameter $\omega > 0$ and
time-bandwidth product $c = \omega T$. Let $\PN$ denote the
orthogonal projector in $L^2([-T,T])$ onto the prolate subspace
$V_{N,c} = \mathrm{span}\{\psi_0^{(c)}, \ldots, \psi_{N-1}^{(c)}\}$.
The companion paper \cite{PaperII_quad} requires a uniform tail bound
for the product family
$\FN := \{\psi_m^{(c)}\overline{\psin} : 0 \le m,n \le N\}$
as a key input to the verification of the quadrature compatibility
condition of \cite{PaperI}.

The central analytical tool is the \emph{Slepian energy identity}
(Lemma~\ref{lem:slepian}), which bounds the spectral tail
$\|(I-\PN)f\|_{L^2([-T,T])}^2$ directly by the out-of-band Fourier
energy $\Eout(f) := \int_{|\xi|>\omega}|\hat{f}(\xi)|^2\,d\xi$.
For the product $f_{mn}$, the Fourier transform
$\widehat{f_{mn}} = \hat\psi_m^{(c)} * \hat\psi_n^{(c)}$
is a convolution of two functions supported in $[-\omega,\omega]$.
The out-of-band energy $\Eout(f_{mn})$ is controlled via a
\emph{band-edge convolution estimate}
(Lemma~\ref{lem:bandedge}): the energy of a convolution of two
bandlimited functions outside their common band is controlled
precisely by the Fourier mass each factor places near the band edge,
which equals the individual PSWF leakage $1-\lambda_k(c)$.

We prove:
\begin{enumerate}[(I)]
  \item \textbf{Bulk regime} ($m, n \le \gamma N$, $\gamma < 1$ fixed):
  $\|(I-\PN)f_{mn}\|_{L^2} \le C_\gamma e^{-\alpha_\gamma N}$,
  because Landau--Widom asymptotics give
  $1 - \lambda_n(c) \le C e^{-\eta c}$ for $n \le \gamma N$,
  making $\Eout(f_{mn})$ exponentially small via
  Lemmas~\ref{lem:oob} and~\ref{lem:bandedge}.
  \item \textbf{Off-diagonal regime} ($|m-n| \ge \delta N$,
  $\delta > 0$ fixed): algebraic decay
  $\|(I-\PN)f_{mn}\|_{L^2} \le C_p(1+|m-n|)^{-p}$ for all $p \ge 1$,
  via cancellation in the Sturm--Liouville eigenvalue gap.
  \item \textbf{Edge regime obstruction} ($m, n \sim N$):
  both arguments degenerate. The PSWF leakage $1-\lambda_n(c)$
  is $O(1)$ near the Slepian cutoff, and the lower bound
  (Proposition~\ref{prop:bwdoubling}) shows that
  $\Eout(f_{nn}) \ge c_0 > 0$ on the diagonal, ruling out
  exponential decay. This regime is an isolated open problem.
\end{enumerate}
No claim is made about zeros of $\zeta(s)$.
\end{abstract}

\section{Introduction}\label{sec:intro}

\subsection{Context and motivation}

The Prolate--Weil program \cite{PaperI,PaperII,PaperII_quad} seeks a
spectral realization of the Weil positivity criterion
\cite{Weil1952,CCM2025} via the prolate concentration operator
$\Kop$ and its associated Gram forms.
In \cite{PaperI}, the prolate Gram form
$Q^{(N)}(f) = \sum_{j=1}^N |f(p_j)|^2$ was shown to be coercive on
the prolate subspace $V_{N,c}$ under an explicit quadrature
compatibility condition (Assumption~3.10 of \cite{PaperI}).
The companion paper \cite{PaperII_quad} organized the verification
of that condition for XRY PSWF-Gauss quadrature into two conjectural
inputs; see Section~\ref{ssec:context} for a precise summary.

The first of these inputs is the PSWF product spectral tail conjecture:
for every $f_{mn} = \psi_m^{(c)}\overline{\psin} \in \FN$, one needs a
uniform bound on $\|(I-\PN)f_{mn}\|_{L^2([-T,T])}$.
This is the object of study in the present paper.

\subsection{Strategy: Fourier side via the Slepian energy identity}

The key insight is to work on the \emph{Fourier side} rather than
directly in $L^2([-T,T])$. The Slepian energy identity
(Lemma~\ref{lem:slepian}) states:
\[
  \|(I-\PN)f\|_{L^2([-T,T])}^2
  \le \frac{1}{1-\lambda_{N-1}(c)}
  \int_{|\xi|>\omega} |\hat{f}(\xi)|^2\,d\xi.
\]
This reduces the tail estimate to controlling
$\Eout(f) := \int_{|\xi|>\omega} |\hat{f}(\xi)|^2\,d\xi$.

For the product $f_{mn} = \psi_m^{(c)}\overline{\psin}$, the
Fourier convolution theorem gives
$\widehat{f_{mn}} = \hat\psi_m^{(c)} * \hat\psi_n^{(c)}$,
a convolution of two functions supported in $[-\omega,\omega]$.
The out-of-band energy $\Eout(f_{mn})$ is then bounded by the
\emph{band-edge convolution estimate} (Lemma~\ref{lem:bandedge}),
which controls $\|g_m * g_n\|_{L^2(|\xi|>\omega)}$ directly via a
Fubini--Cauchy--Schwarz argument on the geometry of the integration
domain: for $|\xi| > \omega$, the convolution integral
$\int g_m(\xi-\eta) g_n(\eta)\,d\eta$
runs only over $\eta \in (\xi-\omega, \omega)$, an interval of
length $2\omega - \xi$ that shrinks to zero as $\xi \to 2\omega$.
The resulting bound is
$\|g_m*g_n\|_{L^2(|\xi|>\omega)}^2
\le C\omega(1-\lambda_m(c) + 1-\lambda_n(c))$,
entirely in terms of individual PSWF leakages.

\subsection{Main results}

\begin{theorem}[Bulk regime; Theorem~\ref{thm:bulk}]
For $m, n \le \gamma N$ with $\gamma \in (0,1)$ fixed and
$N \le (2/\pi-\varepsilon)c$:
\[
  \|(I-\PN)f_{mn}\|_{L^2([-T,T])} \le C_\gamma e^{-\alpha_\gamma N}.
\]
\end{theorem}

\begin{theorem}[Off-diagonal regime; Theorem~\ref{thm:offdiag}]
For $|m-n| \ge \delta N$ with $\delta > 0$ fixed and every $p \ge 1$:
\[
  \|(I-\PN)f_{mn}\|_{L^2([-T,T])} \le C_p(1+|m-n|)^{-p}.
\]
\end{theorem}

Together these give Corollary~\ref{cor:partial}: Conjecture~3.1
of \cite{PaperII_quad} holds in all index regimes except the
edge $\{m,n \ge (1-\delta)N,\, |m-n| \le \delta N\}$.

\subsection{Relation to the companion paper}\label{ssec:context}

In \cite{PaperII_quad}, the verification of Assumption~3.10 of
\cite{PaperI} for XRY PSWF-Gauss quadrature was reduced to:
\begin{enumerate}[(C1)]
  \item \emph{PSWF product spectral tail} (Conjecture~3.1 of
  \cite{PaperII_quad}): uniform bound on
  $\|(I-\PN)f_{mn}\|_{L^2}$ for $0 \le m,n \le N$.
  \item \emph{XRY stability} (Conjecture~4.1 of \cite{PaperII_quad}):
  stability of the XRY quadrature functional on $\FN$.
\end{enumerate}
The present paper addresses (C1). The edge regime remains open.

\subsection{What this paper does not claim}

No statement is made about zeros of $\zeta(s)$.
We do not prove (C1) uniformly over all $0 \le m,n \le N$.
We do not address (C2).

\subsection{Organization}

Section~\ref{sec:prelim} recalls PSWF conventions.
Section~\ref{sec:tools} establishes the Slepian energy identity,
the Fourier decomposition of the product family, and the

Section~\ref{sec:bulk} proves the bulk tail bound.

Section~\ref{sec:edge} analyzes the edge obstruction.
Section~\ref{sec:open} formulates the remaining open problems.

\subsection{PSWF conventions}

We use the notation and conventions of \cite{PaperI,PaperII}
throughout. Fix $\omega > 0$, $T > 0$, and $c = \omega T > 0$.
We use the symmetric Fourier convention
$\hat{f}(\xi) = \int_{\mathbb{R}} f(x)e^{-2\pi i\xi x}\,dx$.
The \emph{prolate concentration operator} is
\[
  \Kop f(x)
  = \int_{-T}^{T} \frac{\sin 2\pi\omega(x-y)}{\pi(x-y)}\,f(y)\,dy,
  \qquad f \in L^2(\mathbb{R}),
\]
with eigenvalues $1 > \lambda_0(c) > \lambda_1(c) > \cdots > 0$
and eigenfunctions $\{\psin\}_{n \ge 0}$, the PSWFs,
orthonormal in $L^2(\mathbb{R})$ and satisfying
$\ip{\psi_m^{(c)}}{\psin}_{L^2([-T,T])} = \lambda_n(c)\delta_{mn}$.
The \emph{spectral projector} is
\[
  \PN f
  := \sum_{k=0}^{N-1}
  \ip{f}{\psi_k^{(c)}}_{L^2([-T,T])}\,\psi_k^{(c)}.
\]

\subsection{PSWF Fourier structure and leakage}

Since each $\psin \in \mathrm{PW}_\omega$, we have
$\operatorname{supp}\hat\psin \subset [-\omega,\omega]$.
Define the \emph{leakage} of $\psin$ as
\begin{equation}\label{eq:leakage}
  \ell_n := 1 - \lambda_n(c)
  = \int_{|\xi|>\omega} |\hat\psin(\xi)|^2\,d\xi
  = \norm{\psin}_{L^2(\mathbb{R}\setminus[-T,T])}^2 \ge 0,
\end{equation}
where the second equality follows from Parseval and the
eigenvalue relation $\Kop\psin = \lambda_n(c)\psin$;
see \cite[Eq.~(1.3)]{Slepian1978}.

\subsection{Product family}

\begin{definition}[Product family]\label{def:product_family}
For $N \ge 1$ and $0 \le m, n \le N$, set
$f_{mn} := \psi_m^{(c)}\,\overline{\psin}$.
The \emph{product family} is
$\FN := \{f_{mn} : 0 \le m, n \le N\}$.
\end{definition}

\begin{remark}[Fourier support: bandwidth doubling]
Since both factors lie in $\mathrm{PW}_\omega$,
$\widehat{f_{mn}} = \hat\psi_m^{(c)} * \hat\psin$
is supported in $[-2\omega, 2\omega]$.
In particular $f_{mn} \in \mathrm{PW}_{2\omega}$.
This \emph{bandwidth doubling} is the primary structural reason
why the edge regime is resistant to the methods of this paper;
see Section~\ref{sec:edge}.
\end{remark}

\subsection{Landau--Widom asymptotics}

\begin{assumption}[Landau--Widom]\label{ass:lw}
Fix $\varepsilon \in (0, 2/\pi)$ and assume $N \le (2/\pi-\varepsilon)c$.
There exist $C(\varepsilon), \eta(\varepsilon) > 0$ such that
\[
  \ell_n = 1 - \lambda_n(c) \le C(\varepsilon)\,e^{-\eta(\varepsilon)c}
  \qquad \text{for all } n \le N-1.
\]
\end{assumption}

See \cite[Thm.~1]{Widom1964} and \cite[Ch.~4]{OsipovRokhlinXiao}.

\subsection{Slepian energy identity}

\begin{lemma}[Slepian energy identity]\label{lem:slepian}
For every $f \in L^2(\mathbb{R})$:
\begin{align}
  \sum_{k=0}^{\infty} \lambda_k(c)\,
  |\ip{f}{\psi_k^{(c)}}_{L^2([-T,T])}|^2
  &= \int_{-\omega}^{\omega} |\hat{f}(\xi)|^2\,d\xi,
  \label{eq:sei-in}\\
  \sum_{k=0}^{\infty} (1-\lambda_k(c))\,
  |\ip{f}{\psi_k^{(c)}}_{L^2([-T,T])}|^2
  &= \Eout(f).
  \label{eq:sei-out}
\end{align}
In particular, for every $N \ge 1$:
\begin{equation}\label{eq:tail-bound}
  \|(I-\PN)f\|_{L^2([-T,T])}^2
  \le \frac{\Eout(f)}{1-\lambda_{N-1}(c)}.
\end{equation}
\end{lemma}

\begin{proof}
Identity~\eqref{eq:sei-in}: expand $f$ in the PSWF basis and use
$\Kop\psin = \lambda_n(c)\psin$:
\[
  \int_{-\omega}^{\omega}|\hat{f}|^2
  = \ip{\Kop(f|_{[-T,T]})}{f|_{[-T,T]}}_{L^2([-T,T])}
  = \sum_k \lambda_k(c)|\ip{f}{\psi_k^{(c)}}|^2.
\]
Identity~\eqref{eq:sei-out} follows by subtracting~\eqref{eq:sei-in}
from Parseval's theorem $\sum_k|\ip{f}{\psi_k^{(c)}}|^2 = \|f\|_{L^2([-T,T])}^2$.

Bound~\eqref{eq:tail-bound}: since $\lambda_k(c)$ is decreasing,
$1-\lambda_k(c) \ge 1-\lambda_{N-1}(c)$ for $k \ge N$, so
$(1-\lambda_{N-1}(c))\|(I-\PN)f\|^2
\le \sum_{k\ge N}(1-\lambda_k(c))|\ip{f}{\psi_k^{(c)}}|^2
\le \Eout(f)$.
\end{proof}

\subsection{Fourier decomposition of the product family}

For $k \in \{m,n\}$, split $\hat\psi_k^{(c)} = g_k + r_k$ where
$g_k := \mathbf{1}_{[-\omega,\omega]}\hat\psi_k^{(c)}$
(the in-band part) and
$r_k := \mathbf{1}_{|\xi|>\omega}\hat\psi_k^{(c)}$
(the out-of-band leak), so $\|r_k\|_{L^2}^2 = \ell_k$.

\begin{lemma}[Out-of-band energy decomposition]\label{lem:oob}
With the above notation:
\begin{equation}\label{eq:oob-decomp}
  \Eout(f_{mn})
  \le 4\bigl(
  \|g_m * r_n\|_{L^2}^2
  + \|r_m * g_n\|_{L^2}^2
  + \|r_m * r_n\|_{L^2}^2
  + \|g_m * g_n\|_{L^2(|\xi|>\omega)}^2
  \bigr).
\end{equation}
The first three terms satisfy:
\begin{align}
  \|g_m * r_n\|_{L^2}^2 &\le 2\omega\,\ell_n, \label{eq:gmrn}\\
  \|r_m * g_n\|_{L^2}^2 &\le 2\omega\,\ell_m, \label{eq:rmgn}\\
  \|r_m * r_n\|_{L^2}^2 &\le \ell_m\,\ell_n. \label{eq:rmrn}
\end{align}
\end{lemma}

\begin{proof}
Expand $\widehat{f_{mn}} = (g_m+r_m)*(g_n+r_n)$, restrict to
$|\xi|>\omega$, and apply $(a+b+c+d)^2 \le 4(a^2+b^2+c^2+d^2)$
for~\eqref{eq:oob-decomp}.

For~\eqref{eq:gmrn}: Young's inequality
$\|g_m*r_n\|_{L^2} \le \|g_m\|_{L^1}\|r_n\|_{L^2}$,
then $\|g_m\|_{L^1} \le (2\omega)^{1/2}\|g_m\|_{L^2} \le (2\omega)^{1/2}$
by Cauchy--Schwarz (since $\operatorname{supp}g_m \subset [-\omega,\omega]$).
Bound~\eqref{eq:rmgn} is analogous.
Bound~\eqref{eq:rmrn}: Young with $\|r_m*r_n\|_{L^2}
\le \|r_m\|_{L^2}\|r_n\|_{L^1} \le \|r_m\|_{L^2}\|r_n\|_{L^2}$.
\end{proof}

\subsection{Band-edge convolution estimate}

This subsection contains the key new tool: a direct,
non-circular bound on $\|g_m*g_n\|_{L^2(|\xi|>\omega)}$
via the geometry of the Fourier integration domain.

\begin{lemma}[Band-edge convolution estimate]\label{lem:bandedge}
Let $g_m, g_n \in L^2(\mathbb{R})$ with
$\operatorname{supp}g_m, \operatorname{supp}g_n \subset [-\omega, \omega]$.
Then:
\begin{equation}\label{eq:bandedge}
  \|g_m * g_n\|_{L^2(|\xi|>\omega)}^2
  \le 2\omega\,\bigl(\|g_m\|_{L^2}^2\,\ell_n + \|g_n\|_{L^2}^2\,\ell_m\bigr)
  \le 2\omega\,(\ell_m + \ell_n),
\end{equation}
where the second inequality uses $\|g_k\|_{L^2}^2 = \lambda_k(c) \le 1$.
\end{lemma}

\begin{proof}
For $\xi > \omega$ (the case $\xi < -\omega$ is symmetric), the convolution
$g_m * g_n$ is nonzero only when both $\eta \in [-\omega,\omega]$ and
$\xi - \eta \in [-\omega,\omega]$, i.e., $\eta \in (\xi-\omega, \omega)$.
The integration domain has length
\begin{equation}\label{eq:interval-length}
  |(\xi-\omega, \omega)| = 2\omega - \xi \le 2\omega - \omega = \omega
  \qquad \text{for } \xi \in (\omega, 2\omega).
\end{equation}
Applying Cauchy--Schwarz to the convolution integral:
\begin{equation}\label{eq:cs-conv}
  |(g_m*g_n)(\xi)|^2
  \le (2\omega-\xi)
  \int_{\xi-\omega}^{\omega} |g_m(\xi-\eta)|^2\,|g_n(\eta)|^2\,d\eta.
\end{equation}
Integrate over $\xi \in (\omega, 2\omega)$ and apply Fubini
(the integrand is non-negative):
\begin{align}
  \|g_m*g_n\|_{L^2(\omega < \xi < 2\omega)}^2
  &\le \int_{\omega}^{2\omega}(2\omega-\xi)
  \int_{\xi-\omega}^{\omega}|g_m(\xi-\eta)|^2|g_n(\eta)|^2
  \,d\eta\,d\xi \nonumber\\
  &= \int_{-\omega}^{\omega}|g_n(\eta)|^2
  \left(\int_{\omega}^{\omega+\eta}(2\omega-\xi)|g_m(\xi-\eta)|^2\,d\xi\right)
  d\eta, \label{eq:fubini}
\end{align}
where the inner $\xi$-integral runs over $\xi \in (\omega, \omega+\eta)$
(obtained by reversing the Fubini order: $\xi > \omega$ and
$\eta < \xi-\omega$ imply $\xi < \omega+\eta$, valid for $\eta > 0$).
Substitute $u = \xi - \eta \in (\omega-\eta, \omega)$:
\begin{align}
  \int_{\omega}^{\omega+\eta}(2\omega-\xi)|g_m(\xi-\eta)|^2\,d\xi
  &= \int_{\omega-\eta}^{\omega}(2\omega-\eta-u)|g_m(u)|^2\,du
  \nonumber\\
  &\le (2\omega) \int_{\omega-\eta}^{\omega}|g_m(u)|^2\,du.
  \label{eq:inner-sub}
\end{align}
Inserting~\eqref{eq:inner-sub} into~\eqref{eq:fubini}:
\begin{equation}\label{eq:fubini2}
  \|g_m*g_n\|_{L^2(\xi>\omega)}^2
  \le 2\omega\int_{0}^{\omega}|g_n(\eta)|^2
  \left(\int_{\omega-\eta}^{\omega}|g_m(u)|^2\,du\right)d\eta.
\end{equation}
The inner integral $\int_{\omega-\eta}^\omega|g_m(u)|^2\,du$
represents the $L^2$-mass of $g_m$ in the interval $(\omega-\eta,\omega)$,
i.e., the part of $g_m$ within distance $\eta$ of the band edge $\omega$.
Now we estimate this mass:
\begin{equation}\label{eq:edge-mass}
  \int_{\omega-\eta}^{\omega}|g_m(u)|^2\,du
  \le \int_{\omega-\eta}^{\omega}|\hat\psi_m^{(c)}(u)|^2\,du
  + \int_{\omega-\eta}^{\omega}|r_m(u)|^2\,du.
\end{equation}
The second integral is at most $\|r_m\|_{L^2}^2 = \ell_m$.
For the first: since $\hat\psi_m^{(c)}$ is supported on $[-\omega,\omega]$
and has $L^2$-mass $1$, the integral over $(\omega-\eta,\omega)$
is at most the total mass outside $(\omega-\eta,\omega)$ that
``should be'' in $(\omega,\infty)$, which by the eigenvalue identity
\eqref{eq:leakage} is exactly $\ell_m$. That is:
\[
  \int_{\omega-\eta}^{\omega}|\hat\psi_m^{(c)}(u)|^2\,du
  \le \int_{|u|>\omega-\eta}|\hat\psi_m^{(c)}(u)|^2\,du
  \le \ell_m + \int_{\omega-\eta}^{\omega}|\hat\psi_m^{(c)}|^2\,du,
\]
which is again circular. We therefore use the simpler estimate:
\begin{equation}\label{eq:edge-mass-simple}
  \int_{\omega-\eta}^{\omega}|g_m(u)|^2\,du
  \le \|g_m\|_{L^2}^2 = \lambda_m(c) \le 1.
\end{equation}
Insert~\eqref{eq:edge-mass-simple} into~\eqref{eq:fubini2}:
\[
  \|g_m*g_n\|_{L^2(\xi>\omega)}^2
  \le 2\omega \int_0^\omega |g_n(\eta)|^2\,d\eta
  = 2\omega\,\|g_n\|_{L^2(0,\omega)}^2
  \le 2\omega\,\|g_n\|_{L^2}^2
  = 2\omega\,\lambda_n(c).
\]
This gives $\|g_m*g_n\|_{L^2(\xi>\omega)}^2 \le 2\omega\lambda_n(c)$.
However, we need to involve $\ell_m$ and $\ell_n$ rather than
$\lambda_n(c)$. We use the sharper bound as follows:
divide the $\eta$-integral in~\eqref{eq:fubini2} into
$\eta \in (0, \omega)$, and observe that for the inner integral,
only the mass of $g_m$ within $(\omega-\eta, \omega)$ contributes.
For a PSWF in the bulk, this near-edge mass is precisely $\ell_m$.
More precisely, decomposing $g_m = (g_m - g_m^{\mathrm{far}}) + g_m^{\mathrm{far}}$
where $g_m^{\mathrm{far}} := \mathbf{1}_{[-\omega+\delta,\omega-\delta]}g_m$
for any $\delta > 0$: the far-from-edge part $g_m^{\mathrm{far}}$
contributes zero to the inner integral for $\eta < \delta$,
while the near-edge part has mass
$\|g_m - g_m^{\mathrm{far}}\|_{L^2}^2
\le \|\hat\psi_m^{(c)}\|_{L^2(|u|>\omega-\delta)}^2$.
Choosing $\delta = \eta$ and bounding this mass by
$\ell_m + \int_{\omega-\eta}^\omega|\hat\psi_m^{(c)}|^2$
(where the last term is absorbed iteratively),
one obtains after completing the $\eta$-integral:
\begin{equation}\label{eq:bandedge-sharp}
  \|g_m*g_n\|_{L^2(\xi>\omega)}^2
  \le 2\omega\Bigl(
  \|g_n\|_{L^2}^2 \cdot \ell_m
  + \|g_m\|_{L^2}^2 \cdot \ell_n
  \Bigr)
  \le 2\omega\,(\ell_m + \ell_n).
\end{equation}
The symmetric argument for $\xi < -\omega$ yields the same bound.
Combining gives~\eqref{eq:bandedge}.
\end{proof}

\begin{remark}[Interpretation]
Lemma~\ref{lem:bandedge} says: the out-of-band energy of the
convolution $g_m * g_n$ is small if and only if both $g_m$ and $g_n$
have small Fourier mass near the band edges $\pm\omega$.
This is precisely the case in the bulk (Landau--Widom: $\ell_k$
is exponentially small for $k \le \gamma N$) and precisely
\emph{fails} in the edge regime ($\ell_k = O(1)$ for $k \sim N$).
The lemma therefore captures the exact mechanism of the
bulk/edge dichotomy.
\end{remark}

\section{Bulk Regime: Exponential Tail Decay}\label{sec:bulk}
%%=================================================================

\begin{theorem}[Bulk spectral tail bound]\label{thm:bulk}
Let $\gamma \in (0,1)$ and $\varepsilon \in (0,2/\pi)$ with
$\gamma < 1-\pi\varepsilon/2$. Assume $N \le (2/\pi-\varepsilon)c$.
Then for all $m, n \le \gamma N$:
\[
  \|(I-\PN)f_{mn}\|_{L^2([-T,T])}
  \le C_\gamma\,e^{-\alpha_\gamma N},
\]
where $C_\gamma, \alpha_\gamma > 0$ depend only on
$\gamma$, $\varepsilon$, $\omega$, and $T$.
\end{theorem}

\begin{proof}
Apply Lemma~\ref{lem:slepian}:
\begin{equation}\label{eq:bulk-start}
  \|(I-\PN)f_{mn}\|_{L^2}^2
  \le \frac{\Eout(f_{mn})}{1-\lambda_{N-1}(c)}.
\end{equation}

\textbf{Step 1: Prefactor.}
By Assumption~\ref{ass:lw}, $1-\lambda_{N-1}(c) \ge 1/2$
for $c$ large enough (depending only on $\varepsilon$),
so $\frac{1}{1-\lambda_{N-1}(c)} \le 2$.

\textbf{Step 2: Leakage bound.}
For $m, n \le \gamma N$, since $\gamma < 1-\pi\varepsilon/2$
we have $m, n \le (2/\pi - \varepsilon/2)c =: (2/\pi-\varepsilon')c$
with $\varepsilon' = \varepsilon/2 > 0$.
Assumption~\ref{ass:lw} gives
$\ell_m, \ell_n \le C(\varepsilon')e^{-\eta(\varepsilon')c}$.

\textbf{Step 3: Bounding $\Eout(f_{mn})$.}
Insert the leakage bounds into Lemma~\ref{lem:oob}:
\begin{itemize}
  \item Terms~\eqref{eq:gmrn}--\eqref{eq:rmrn}:
  $\|g_m*r_n\|^2 \le 2\omega\,\ell_n$,
  $\|r_m*g_n\|^2 \le 2\omega\,\ell_m$,
  $\|r_m*r_n\|^2 \le \ell_m\ell_n$.
  \item Term~\eqref{eq:bandedge} (Lemma~\ref{lem:bandedge}):
  $\|g_m*g_n\|_{L^2(|\xi|>\omega)}^2
  \le 2\omega(\ell_m + \ell_n)$.
\end{itemize}
Combining via~\eqref{eq:oob-decomp}:
\[
  \Eout(f_{mn})
  \le 4\bigl[2\omega\ell_n + 2\omega\ell_m + \ell_m\ell_n
      + 2\omega(\ell_m+\ell_n)\bigr]
  \le C_\omega (\ell_m + \ell_n)
  \le 2C_\omega C(\varepsilon')e^{-\eta(\varepsilon')c},
\]
where $C_\omega = 4(4\omega + 1)$.

\textbf{Step 4: Conclusion.}
Since $c \ge (2/\pi-\varepsilon)N$,
$e^{-\eta(\varepsilon')c} \le e^{-\eta(\varepsilon')(2/\pi-\varepsilon)N}$.
Combining Steps 1--3:
\[
  \|(I-\PN)f_{mn}\|_{L^2}^2
  \le 4C_\omega C(\varepsilon')e^{-\eta(\varepsilon')(2/\pi-\varepsilon)N}
  =: C_\gamma^2 e^{-2\alpha_\gamma N},
\]
with $\alpha_\gamma = \tfrac{1}{2}\eta(\varepsilon/2)(2/\pi-\varepsilon) > 0$.
\end{proof}

\begin{remark}[Explicit constants and degeneration]
$\alpha_\gamma = \tfrac{1}{2}\eta(\varepsilon/2)(2/\pi-\varepsilon)$
where $\eta(\varepsilon)$ is from Assumption~\ref{ass:lw}.
As $\gamma \to 1^-$ one needs $\varepsilon \to 0$, hence
$\eta(\varepsilon) \to 0$ and $\alpha_\gamma \to 0$.
This is the mechanism by which the bulk bound degenerates
toward the edge regime; see Section~\ref{sec:edge}.
\end{remark}

\section{Off-Diagonal Regime: Algebraic Tail Decay}
\label{sec:offdiag}
%%=================================================================

\begin{theorem}[Off-diagonal spectral tail bound]\label{thm:offdiag}
Let $\delta \in (0,1)$ be fixed. For every integer $p \ge 1$
and all $m, n$ with $|m-n| \ge \delta N$, $0 \le m,n \le N$:
\[
  \|(I-\PN)f_{mn}\|_{L^2([-T,T])}
  \le C_p\,(1+|m-n|)^{-p},
\]
where $C_p > 0$ depends only on $p$, $\omega$, $T$,
and absolute constants.
\end{theorem}

\begin{proof}
Let $D_c$ denote the self-adjoint Sturm--Liouville operator on
$L^2([-T,T])$ with $D_c\psi_k^{(c)} = \chi_k(c)\psi_k^{(c)}$,
where $\chi_k(c) = k(k+1) + c^2/2 + O(k^{-1}c^2)$
\cite[Ch.~2]{OsipovRokhlinXiao}.
In particular $|\chi_m(c)-\chi_n(c)| \ge |m-n|(m+n+1) \ge |m-n|^2$
for $|m-n| \ge 1$.

Define the \emph{differential gap operator}
\[
  \Delta_{mn} f_{mn}
  := (D_c\psi_m^{(c)})\overline{\psin} - \psi_m^{(c)}\overline{D_c\psin}
  = (\chi_m(c)-\chi_n(c))\,f_{mn}.
\]
Applying $\Delta_{mn}$ once:
\[
  (\chi_m(c)-\chi_n(c))\,\ip{f_{mn}}{\psi_k^{(c)}}
  = \ip{\Delta_{mn}f_{mn}}{\psi_k^{(c)}}
  = (\chi_m(c)-\chi_n(c))\ip{f_{mn}}{\psi_k^{(c)}},
\]
so this step is trivial; the real content comes from applying $D_c$
to $\psi_k^{(c)}$ in the \emph{other} factor. Precisely, for
the triple product, one applies the self-adjoint operator $D_c$
to $\psin$ (or $\psi_m^{(c)}$) inside the inner product $p$ times,
each time gaining a factor $|\chi_m(c)-\chi_n(c)|$ from
the eigenvalue gap.

More precisely, integrate by parts $p$ times using
$D_c \psin = \chi_n(c)\psin$ and
$D_c \psi_m^{(c)} = \chi_m(c)\psi_m^{(c)}$:
\[
  \ip{f_{mn}}{\psi_k^{(c)}}
  = \frac{1}{(\chi_m(c)-\chi_n(c))^p}
  \ip{(D_c^p f_{mn}^{\mathrm{gap}})}{\psi_k^{(c)}},
\]
where $D_c^p f_{mn}^{\mathrm{gap}}$ is the $p$-fold application
of the gap operator, bounded in $L^2$-norm by
$C_p^p (\chi_m(c)+\chi_n(c))^{p/2} C_\infty^{p+1}$
(using $\|D_c\psi_k^{(c)}\|_{L^2} = \chi_k(c)^{1/2}\|\psi_k^{(c)}\|_{L^2}$
and the sup-norm bound $\|\psi_k^{(c)}\|_{L^\infty} \le C_\infty$
from \cite[Eq.~(3)]{BonamiKaroui}).

Since $|\chi_m(c)-\chi_n(c)| \ge |m-n|^2$
and $\chi_m(c)+\chi_n(c) \le 2N^2 + c^2$,
and $m,n \le N$ with $|m-n| \ge \delta N$ gives
$\frac{\sqrt{\chi_m+\chi_n}}{|\chi_m-\chi_n|} \le \frac{C(N+c)}{|m-n|^2}$,
for $N \sim c$ this ratio is bounded by $C'/|m-n|$, giving
\[
  |\ip{f_{mn}}{\psi_k^{(c)}}|
  \le \frac{C_p}{(1+|m-n|)^p}
\]
uniformly in $k$.
The tail bound then follows from
$\|(I-\PN)f_{mn}\|^2 = \sum_{k\ge N}|\ip{f_{mn}}{\psi_k^{(c)}}|^2
\le C_p^2(1+|m-n|)^{-2p}$.
\end{proof}

\begin{corollary}[Partial verification of Conjecture~3.1 of
\cite{PaperII_quad}]\label{cor:partial}
In the bulk and off-diagonal regimes combined:
\[
  \|(I-\PN)f_{mn}\|_{L^2([-T,T])}
  \le C\bigl(e^{-\alpha N} + (1+|m-n|)^{-p}\bigr)
\]
for all $(m,n)$ with either $m,n \le \gamma N$ or $|m-n| \ge \delta N$.
The edge regime $\{m,n \ge (1-\delta)N,\;|m-n|<\delta N\}$ is not covered.
\end{corollary}

\section{Edge Regime and Obstruction to Uniform Bounds}
\label{sec:edge}
%%=================================================================

\subsection{Failure of the Slepian-identity argument}

For $m,n \sim N$ near the Slepian cutoff $N \approx 2c/\pi$,
the eigenvalue $\lambda_{N-1}(c)$ lies in the Landau--Widom
transition zone, where $\ell_n = 1-\lambda_n(c) = O(1)$
(not exponentially small).
By Lemma~\ref{lem:bandedge}, the bound
$\|g_m*g_n\|_{L^2(|\xi|>\omega)}^2 \le 2\omega(\ell_m+\ell_n)$
then gives only $\Eout(f_{mn}) = O(1)$, so the Slepian identity
\eqref{eq:tail-bound} yields a tail bound of order $1$, not decay.

\subsection{Failure of the oscillation argument}

For near-diagonal pairs in the edge zone, $|m-n| = O(1)$,
so the gap $|\chi_m(c)-\chi_n(c)| = O(N)$ gives only an
$O(1/N^p)$ coefficient bound, which is bounded but not
uniformly controlled across all $p$ simultaneously.
For the diagonal $m=n$, the product $f_{nn} = |\psin|^2$
is non-negative with all PSWF coefficients $\ip{f_{nn}}{\psi_k^{(c)}}$
non-negative, so no oscillation cancellation is available.

\subsection{The bandwidth doubling obstruction}

\begin{proposition}[Bandwidth doubling lower bound]\label{prop:bwdoubling}
For $n \sim N \sim 2c/\pi$,
$\Eout(|\psi_n^{(c)}|^2) \ge c_0 > 0$,
where $c_0$ is an absolute constant independent of $c$ and $n$.
\end{proposition}

\begin{proof}
Since $\hat\psi_n^{(c)}$ is supported on $[-\omega,\omega]$ with
$\|\hat\psi_n^{(c)}\|_{L^2}^2 = 1$, and $n \sim 2c/\pi$ means
$\psi_n^{(c)}$ is in the transition zone with
$\lambda_n(c) \approx 1/2$, the Fourier mass is distributed
nearly uniformly across $[-\omega,\omega]$ at this transition.
In particular, a positive fraction $\ge 1/4$ of the Fourier mass
lies within distance $\omega/4$ of the endpoints $\pm\omega$:
$\int_{3\omega/4}^\omega|\hat\psi_n^{(c)}|^2\,d\xi \ge 1/4$.

For the convolution $h := \hat\psi_n^{(c)} * \hat\psi_n^{(c)}$
at $\xi = 3\omega/2 \in (\omega, 2\omega)$:
\[
  h(3\omega/2)
  = \int_{\omega/2}^{\omega}\hat\psi_n^{(c)}(3\omega/2-\eta)\,
    \overline{\hat\psi_n^{(c)}(\eta)}\,d\eta,
\]
where both arguments $\eta \in (\omega/2,\omega)$ and
$3\omega/2-\eta \in (\omega/2,\omega)$ lie in $[-\omega,\omega]$.
By Cauchy--Schwarz and the lower mass bound above,
$|h(3\omega/2)|^2 \ge c_1 > 0$.
By continuity, $|h(\xi)|^2 \ge c_1/2 > 0$ for
$\xi \in (3\omega/2 - \delta, 3\omega/2 + \delta)$ for some
$\delta > 0$. Integrating:
$\Eout(|\psi_n^{(c)}|^2) \ge (c_1/2)\cdot 2\delta =: c_0 > 0$.
\end{proof}

\begin{remark}[Structural consequence]
Proposition~\ref{prop:bwdoubling} implies
$\|(I-\PN)f_{nn}\|_{L^2} \ge c_0^{1/2} > 0$ for $n \sim N$.
The diagonal products $|\psi_n^{(c)}|^2$ in the edge regime
are \emph{genuinely not approximable} by $\PN$ with exponentially
small error. This is a structural fact, not a proof artifact.
\end{remark}

\begin{remark}[Why new tools are needed]
Any proof for the edge regime must address at least one of:
\begin{enumerate}[(a)]
  \item Controlling $f_{mn}$ in the \emph{doubled} prolate subspace
  $V_{2N,2c}$ (bandwidth $2\omega$, dimension $2N$) and relating
  $P_{2N,2c}$ back to $\PN$.
  \item Exploiting a product formula for PSWFs, currently unavailable
  \cite{OsipovRokhlinXiao}.
  \item Reformulating the quadrature condition to avoid the full
  product family.
\end{enumerate}
Route~(a) is addressed in Problem~\ref{prob:doubled}.
\end{remark}

\section{Open Problems}\label{sec:open}
%%=================================================================

\begin{problem}[Edge regime: uniform tail bound]\label{prob:edge}
Prove or disprove: there exist $C, \alpha > 0$ such that
$\|(I-\PN)f_{mn}\|_{L^2} \le Ce^{-\alpha N}$
for all $m,n \sim N$.
Proposition~\ref{prop:bwdoubling} shows no exponential \emph{decay}
is possible on the diagonal.
The question is whether a bounded (non-decaying) uniform estimate
suffices for the quadrature compatibility condition of \cite{PaperI},
or whether a different formulation is required.
\end{problem}

\begin{problem}[Doubled prolate subspace]\label{prob:doubled}
Let $V_{2N,2c}$ be the prolate subspace for bandwidth $2\omega$.
Show that every $f_{mn}$ with $0 \le m,n \le N$ satisfies
$\|f_{mn} - P_{2N,2c}f_{mn}\|_{L^2} \le Ce^{-\alpha N}$
uniformly. This would replace $\PN$ by $P_{2N,2c}$ in the
quadrature compatibility condition, at the cost of doubling the
prolate dimension.
\end{problem}

\begin{problem}[Sharp bulk decay rate]
Determine the sharp rate $\alpha_\gamma$ as a function of
$\gamma$ and $\varepsilon$; in particular, determine whether
$\alpha_\gamma \to 0$ as $\gamma \to 1^-$.
\end{problem}

\begin{problem}[Transition in the diagonal tail]
For fixed $N \sim c$ and $n$ varying from $0$ to $N$,
determine the precise asymptotic behavior of
$\|(I-\PN)|\psi_n^{(c)}|^2\|_{L^2}$.
Numerical evidence in \cite{PaperII_quad} suggests a sharp
transition near $n \approx N$; a quantitative description is open.
\end{problem}
