\begin{abstract}
Let $\{\psin\}_{n \ge 0}$ be the prolate spheroidal wave functions
(PSWFs) on $[-T,T]$ with bandwidth parameter $\omega > 0$ and
time-bandwidth product $c = \omega T$. Let $\PN$ denote the
orthogonal projector in $L^2([-T,T])$ onto the prolate subspace
$V_{N,c} = \mathrm{span}\{\psi_0^{(c)}, \ldots, \psi_{N-1}^{(c)}\}$.
The companion paper \cite{PaperII_quad} requires a uniform tail bound
for the product family
$\FN := \{\psi_m^{(c)}\overline{\psin} : 0 \le m,n \le N\}$
as a key input to the verification of the quadrature compatibility
condition of \cite{PaperI}.

The central analytical tool is the \emph{Slepian energy identity}
(Lemma~\ref{lem:slepian}), which bounds the spectral tail
$\|(I-\PN)f\|_{L^2([-T,T])}^2$ directly by the out-of-band Fourier
energy $\Eout(f) := \int_{|\xi|>\omega}|\hat{f}(\xi)|^2\,d\xi$.
For the product $f_{mn}$, the Fourier transform
$\widehat{f_{mn}} = \hat\psi_m^{(c)} * \hat\psi_n^{(c)}$
is a convolution of two functions supported in $[-\omega,\omega]$.
The out-of-band energy $\Eout(f_{mn})$ is bounded via the
\emph{band-edge convolution estimate} (Lemma~\ref{lem:bandedge}),
which gives a rigorous, non-circular upper bound on
$\|g_m*g_n\|_{L^2(|\xi|>\omega)}^2$ entirely in terms of the
$L^2$-norms $\|g_m\|_{L^2}^2 = \lambda_m(c)$ and
$\|g_n\|_{L^2}^2 = \lambda_n(c)$.
The cross-terms (Lemma~\ref{lem:oob}) are exponentially small in the
bulk by Landau--Widom asymptotics. The in-band$\times$in-band
convolution term is $O(\omega)$ (an absolute constant independent of
$c$), and cannot be made exponentially small by the Fourier
decomposition alone. Exponential decay of this term requires
Assumption~\ref{ass:nearedge} (near-edge exponential control).

We prove:
\begin{enumerate}[(I)]
  \item \textbf{Bulk regime} ($m, n \le \gamma N$, $\gamma < 1$ fixed):
  $\|(I-\PN)f_{mn}\|_{L^2} \le C_\gamma e^{-\alpha_\gamma N}$,
  \emph{conditional on Assumption~\ref{ass:nearedge}},
  which asserts that for $n \le \gamma N$ the Fourier mass of
  $\hat\psi_n^{(c)}$ near $|\xi| = \omega$ is exponentially small.
  This assumption is identified as an explicit open problem
  (Problem~\ref{prob:nearedge}) and is expected to follow from
  WKB analysis of the associated Sturm--Liouville problem.
  \item \textbf{Off-diagonal regime} ($|m-n| \ge \delta N$,
  $\delta > 0$ fixed): algebraic decay
  $\|(I-\PN)f_{mn}\|_{L^2} \le C_p(1+|m-n|)^{-p}$ for all $p \ge 1$,
  via the Sturm--Liouville eigenvalue gap. The gap estimate uses
  $N \ge |m-n|/\delta$ explicitly; see Lemma~\ref{lem:gap}.
  \item \textbf{Edge regime obstruction} ($m, n \sim N$):
  both arguments degenerate. The PSWF leakage $1-\lambda_n(c)$
  is $O(1)$ near the Slepian cutoff, and the lower bound
  (Proposition~\ref{prop:bwdoubling}) shows that
  $\Eout(f_{nn}) \ge c_0 > 0$ on the diagonal,
  ruling out exponential decay. This regime is an isolated open problem.
\end{enumerate}
No claim is made about zeros of $\zeta(s)$.
\end{abstract}

\subsection{Context and motivation}

The Prolate--Weil program \cite{PaperI,PaperII,PaperII_quad} seeks a
spectral realization of the Weil positivity criterion
\cite{Weil1952,CCM2025} via the prolate concentration operator
$\Kop$ and its associated Gram forms.
In \cite{PaperI}, the prolate Gram form
$Q^{(N)}(f) = \sum_{j=1}^N |f(p_j)|^2$ was shown to be coercive on
the prolate subspace $V_{N,c}$ under an explicit quadrature
compatibility condition (Assumption~3.10 of \cite{PaperI}).
The companion paper \cite{PaperII_quad} organized the verification
of that condition for XRY PSWF-Gauss quadrature into two conjectural
inputs; see Section~\ref{ssec:context} for a precise summary.

The first of these inputs is the PSWF product spectral tail conjecture:
for every $f_{mn} = \psi_m^{(c)}\overline{\psin} \in \FN$, one needs a
uniform bound on $\|(I-\PN)f_{mn}\|_{L^2([-T,T])}$.
This is the object of study in the present paper.

\subsection{Strategy: Fourier side via the Slepian energy identity}

The key insight is to work on the \emph{Fourier side} rather than
directly in $L^2([-T,T])$. The Slepian energy identity
(Lemma~\ref{lem:slepian}) states:
\[
  \|(I-\PN)f\|_{L^2([-T,T])}^2
  \le \frac{\Eout(f)}{1-\lambda_{N-1}(c)},
\]
reducing the problem to bounding
$\Eout(f) = \int_{|\xi|>\omega}|\hat f(\xi)|^2\,d\xi$.

For $f_{mn}$, the Fourier convolution theorem gives
$\widehat{f_{mn}} = \hat\psi_m^{(c)} * \hat\psi_n^{(c)}$,
a convolution of two functions in $[-\omega,\omega]$.
We decompose $\hat\psi_k^{(c)} = g_k + r_k$ (in-band and leak)
and bound $\Eout(f_{mn})$ by four terms (Lemma~\ref{lem:oob}).
Three involve at least one leak factor $r_k$ with
$\|r_k\|_{L^2}^2 = \ell_k = 1-\lambda_k(c)$
--- these are exponentially small in the bulk by Landau--Widom.
The fourth term, $\|g_m*g_n\|_{L^2(|\xi|>\omega)}^2$,
is bounded by a Fubini--Cauchy--Schwarz argument
(Lemma~\ref{lem:bandedge}) giving
$\|g_m*g_n\|_{L^2(|\xi|>\omega)}^2 \le 4\omega\lambda_m(c)\lambda_n(c) \le 4\omega$.
This is an \emph{absolute constant} in $\omega$, independent of $c$ or $n$,
and does not itself decay. Exponential control of this term requires the
separate near-edge input of Assumption~\ref{ass:nearedge}.

\subsection{Main results}

\begin{theorem}[Bulk regime; Theorem~\ref{thm:bulk}]
Under Assumption~\ref{ass:nearedge}, for $m, n \le \gamma N$
with $\gamma \in (0,1)$ fixed,
$N \le (2/\pi-\varepsilon)c$, and $c$ sufficiently large
(depending only on $\varepsilon$ and $\gamma$):
\[
  \|(I-\PN)f_{mn}\|_{L^2([-T,T])} \le C_\gamma e^{-\alpha_\gamma N}.
\]
\end{theorem}

\begin{theorem}[Off-diagonal regime; Theorem~\ref{thm:offdiag}]
For $|m-n| \ge \delta N$ with $\delta > 0$ fixed, $0 \le m,n \le N$,
and every integer $p \ge 1$:
\[
  \|(I-\PN)f_{mn}\|_{L^2([-T,T])} \le C_p(1+|m-n|)^{-p}.
\]
\end{theorem}

Together these give Corollary~\ref{cor:partial}: Conjecture~3.1
of \cite{PaperII_quad} holds in all index regimes except the
edge $\{m,n \ge (1-\delta)N,\, |m-n| \le \delta N\}$,
conditionally on Assumption~\ref{ass:nearedge} in the bulk.

\subsection{Relation to the companion paper}\label{ssec:context}

In \cite{PaperII_quad}, the verification of Assumption~3.10 of
\cite{PaperI} for XRY PSWF-Gauss quadrature was reduced to:
\begin{enumerate}[(C1)]
  \item \emph{PSWF product spectral tail} (Conjecture~3.1 of
  \cite{PaperII_quad}): uniform bound on
  $\|(I-\PN)f_{mn}\|_{L^2}$ for $0 \le m,n \le N$.
  \item \emph{XRY stability} (Conjecture~4.1 of \cite{PaperII_quad}):
  stability of the XRY quadrature functional on $\FN$.
\end{enumerate}
The present paper addresses (C1). The bulk result is conditional on
Assumption~\ref{ass:nearedge}. The edge regime remains open.

We do not prove (C1) uniformly over all $0 \le m,n \le N$.
We do not prove Assumption~\ref{ass:nearedge}; it is formulated
as an explicit open problem (Problem~\ref{prob:nearedge}).
We do not address (C2).

\subsection{PSWF conventions}

We use the notation and conventions of \cite{PaperI,PaperII}
throughout. Fix $\omega > 0$, $T > 0$, and $c = \omega T > 0$.
We use the symmetric Fourier convention
$\hat{f}(\xi) = \int_{\mathbb{R}} f(x)e^{-2\pi i\xi x}\,dx$.
The \emph{prolate concentration operator} is
\[
  \Kop f(x)
  = \int_{-T}^{T} \frac{\sin 2\pi\omega(x-y)}{\pi(x-y)}\,f(y)\,dy,
  \qquad f \in L^2(\mathbb{R}),
\]
with eigenvalues $1 > \lambda_0(c) > \lambda_1(c) > \cdots > 0$
and eigenfunctions $\{\psin\}_{n \ge 0}$, the PSWFs,
orthonormal in $L^2(\mathbb{R})$ and satisfying
$\ip{\psi_m^{(c)}}{\psin}_{L^2([-T,T])} = \lambda_n(c)\delta_{mn}$.
The \emph{spectral projector} is
\[
  \PN f
  := \sum_{k=0}^{N-1}
  \ip{f}{\psi_k^{(c)}}_{L^2([-T,T])}\,\psi_k^{(c)}.
\]

\subsection{PSWF Fourier structure and leakage}

Since each $\psin \in \mathrm{PW}_\omega$, we have
$\operatorname{supp}\hat\psin \subset [-\omega,\omega]$
and $\|\hat\psin\|_{L^2}^2 = \|\psin\|_{L^2(\mathbb{R})}^2 = 1$.
Define the \emph{leakage} and \emph{in-band energy} of $\psin$ as
\begin{equation}\label{eq:leakage}
  \ell_n := 1 - \lambda_n(c)
  = \int_{|\xi|>\omega} |\hat\psin(\xi)|^2\,d\xi
  = \norm{\psin}_{L^2(\mathbb{R}\setminus[-T,T])}^2 \ge 0,
  \qquad
  \lambda_n(c) = \int_{-\omega}^{\omega}|\hat\psin(\xi)|^2\,d\xi,
\end{equation}
where the identities follow from Parseval and
$\Kop\psin = \lambda_n(c)\psin$ \cite[Eq.~(1.3)]{Slepian1978}.

\begin{remark}\label{rem:leakage-meaning}
Note that $\ell_n$ measures the Fourier energy of $\psin$
\emph{outside} the band $[-\omega,\omega]$, while $\lambda_n(c)$
is the Fourier energy \emph{inside} the band.
These are complementary quantities summing to $1$.
In particular, $\int_{\omega-\delta}^\omega|\hat\psin|^2\,d\xi$
(near-edge in-band mass) and $\ell_n$ are in general
\emph{different} objects and cannot be identified without
additional assumptions. The near-edge in-band mass is precisely
the object controlled by Assumption~\ref{ass:nearedge} below.
\end{remark}

\subsection{Product family}

\begin{definition}[Product family]\label{def:product_family}
For $N \ge 1$ and $0 \le m, n \le N$, set
$f_{mn} := \psi_m^{(c)}\,\overline{\psin}$.
The \emph{product family} is
$\FN := \{f_{mn} : 0 \le m, n \le N\}$.
\end{definition}

\begin{remark}[Fourier support: bandwidth doubling]
Since both factors lie in $\mathrm{PW}_\omega$,
$\widehat{f_{mn}} = \hat\psi_m^{(c)} * \hat\psin$
is supported in $[-2\omega, 2\omega]$.
In particular $f_{mn} \in \mathrm{PW}_{2\omega}$.
This \emph{bandwidth doubling} is the primary structural reason
why the edge regime is resistant to the methods of this paper;
see Section~\ref{sec:edge}.
\end{remark}

\subsection{Landau--Widom asymptotics}

\begin{assumption}[Landau--Widom]\label{ass:lw}
Fix $\varepsilon \in (0, 2/\pi)$ and assume $N \le (2/\pi-\varepsilon)c$.
There exist $C(\varepsilon), \eta(\varepsilon) > 0$, and $c_0(\varepsilon) > 0$
such that for all $c \ge c_0(\varepsilon)$:
\[
  \ell_n = 1 - \lambda_n(c) \le C(\varepsilon)\,e^{-\eta(\varepsilon)c}
  \qquad \text{for all } n \le N-1.
\]
In particular, $1 - \lambda_{N-1}(c) \le C(\varepsilon)e^{-\eta(\varepsilon)c} < 1/2$
for $c \ge c_0(\varepsilon)$, so $\lambda_{N-1}(c) > 1/2$ and
$1/(1-\lambda_{N-1}(c)) \le 2$.
\end{assumption}

See \cite[Thm.~1]{Widom1964} and \cite[Ch.~4]{OsipovRokhlinXiao}.

\subsection{Near-edge exponential control}

\begin{assumption}[Near-edge exponential control]\label{ass:nearedge}
Fix $\gamma \in (0,1)$.
There exist constants $C_{\mathrm{ne}}(\gamma) > 0$,
$\eta_{\mathrm{ne}}(\gamma) > 0$, $\delta_{\mathrm{ne}}(\gamma) > 0$,
and $c_0(\gamma) > 0$ such that for all $n \le \gamma N$
and all $c \ge c_0(\gamma)$:
\[
  \int_{|\xi| \in (\omega - \delta_{\mathrm{ne}}, \omega)}
  |\hat\psi_n^{(c)}(\xi)|^2\,d\xi
  \le C_{\mathrm{ne}}\,e^{-\eta_{\mathrm{ne}} c}.
\]
\end{assumption}

\begin{remark}[Interpretation and status of Assumption~\ref{ass:nearedge}]
\label{rem:nearedge-interp}
Assumption~\ref{ass:nearedge} asserts that bulk PSWFs ($n \le \gamma N$
with $\gamma < 1$) are not only concentrated inside the band
$[-\omega,\omega]$, but exhibit \emph{exponential decay of Fourier mass
near the turning points} $\xi = \pm\omega$.
Heuristically, $\xi = \pm\omega$ is a turning point of the
Sturm--Liouville differential equation for PSWFs: in the WKB approximation,
solutions transition from oscillatory behavior ($|\xi| < \omega$)
to exponential decay ($|\xi| > \omega$), and one expects the amplitude
of $|\hat\psi_n^{(c)}|^2$ to be exponentially small in a neighborhood of
$\pm\omega$ when $n$ is far below the Slepian cutoff.
This is consistent with known pointwise WKB estimates for prolate
functions (see \cite{OsipovRokhlinXiao}), but we are not aware of
an explicit quantitative statement in the literature in the
$L^2$-integrated form required here.
Assumption~\ref{ass:nearedge} is therefore formulated as an explicit
open problem (Problem~\ref{prob:nearedge}).
It is structurally analogous to Assumption~3.10 of \cite{PaperI},
which likewise isolates a precise analytical input required by the
broader Prolate--Weil program.
\end{remark}

\subsection{Slepian energy identity}

\begin{lemma}[Slepian energy identity]\label{lem:slepian}
For every $f \in L^2(\mathbb{R})$:
\begin{align}
  \sum_{k=0}^{\infty} \lambda_k(c)\,
  |\ip{f}{\psi_k^{(c)}}_{L^2([-T,T])}|^2
  &= \int_{-\omega}^{\omega} |\hat{f}(\xi)|^2\,d\xi,
  \label{eq:sei-in}\\
  \sum_{k=0}^{\infty} (1-\lambda_k(c))\,
  |\ip{f}{\psi_k^{(c)}}_{L^2([-T,T])}|^2
  &= \Eout(f).
  \label{eq:sei-out}
\end{align}
In particular, for every $N \ge 1$:
\begin{equation}\label{eq:tail-bound}
  \|(I-\PN)f\|_{L^2([-T,T])}^2
  \le \frac{\Eout(f)}{1-\lambda_{N-1}(c)}.
\end{equation}
\end{lemma}

\begin{proof}
Identity~\eqref{eq:sei-in}: expand $f$ in the PSWF basis and use
$\Kop\psin = \lambda_n(c)\psin$:
\[
  \int_{-\omega}^{\omega}|\hat{f}|^2
  = \ip{\Kop(f|_{[-T,T]})}{f|_{[-T,T]}}_{L^2([-T,T])}
  = \sum_k \lambda_k(c)|\ip{f}{\psi_k^{(c)}}|^2.
\]
Identity~\eqref{eq:sei-out} follows by subtracting~\eqref{eq:sei-in}
from Parseval: $\sum_k|\ip{f}{\psi_k^{(c)}}|^2 = \|f\|_{L^2([-T,T])}^2$.

Bound~\eqref{eq:tail-bound}: since $\lambda_k(c)$ is strictly
decreasing, $1-\lambda_k(c) \ge 1-\lambda_{N-1}(c) > 0$ for $k \ge N$,
so $(1-\lambda_{N-1}(c))\|(I-\PN)f\|^2
\le \sum_{k\ge N}(1-\lambda_k(c))|\ip{f}{\psi_k^{(c)}}|^2 \le \Eout(f)$.
\end{proof}

\subsection{Fourier decomposition of the product family}

For $k \in \{m,n\}$, split $\hat\psi_k^{(c)} = g_k + r_k$ where
$g_k := \mathbf{1}_{[-\omega,\omega]}\hat\psi_k^{(c)}$
(the in-band part) and
$r_k := \mathbf{1}_{|\xi|>\omega}\hat\psi_k^{(c)}$
(the out-of-band leak). By~\eqref{eq:leakage}:
\begin{equation}\label{eq:norms}
  \|g_k\|_{L^2}^2 = \lambda_k(c), \qquad \|r_k\|_{L^2}^2 = \ell_k.
\end{equation}

\begin{lemma}[Out-of-band energy decomposition]\label{lem:oob}
With the above notation:
\begin{equation}\label{eq:oob-decomp}
  \Eout(f_{mn})
  \le 4\bigl(
  \|g_m * r_n\|_{L^2}^2
  + \|r_m * g_n\|_{L^2}^2
  + \|r_m * r_n\|_{L^2}^2
  + \|g_m * g_n\|_{L^2(|\xi|>\omega)}^2
  \bigr).
\end{equation}
The first three terms satisfy:
\begin{align}
  \|g_m * r_n\|_{L^2}^2 &\le 2\omega\,\ell_n, \label{eq:gmrn}\\
  \|r_m * g_n\|_{L^2}^2 &\le 2\omega\,\ell_m, \label{eq:rmgn}\\
  \|r_m * r_n\|_{L^2}^2 &\le \ell_m\,\ell_n. \label{eq:rmrn}
\end{align}
\end{lemma}

\begin{proof}
Expand $\widehat{f_{mn}} = (g_m+r_m)*(g_n+r_n)$, restrict to
$|\xi|>\omega$, apply the triangle inequality, and use
$(a+b+c+d)^2 \le 4(a^2+b^2+c^2+d^2)$ for~\eqref{eq:oob-decomp}.

For~\eqref{eq:gmrn}: Young's convolution inequality
$\|g_m*r_n\|_{L^2} \le \|g_m\|_{L^1}\|r_n\|_{L^2}$,
then $\|g_m\|_{L^1} \le (2\omega)^{1/2}\|g_m\|_{L^2} \le (2\omega)^{1/2}$
by Cauchy--Schwarz and $\operatorname{supp}g_m \subset [-\omega,\omega]$.
Bound~\eqref{eq:rmgn} is symmetric.
Bound~\eqref{eq:rmrn}: Young gives $\|r_m*r_n\|_{L^2}
\le \|r_m\|_{L^2}\|r_n\|_{L^2}$.
\end{proof}

\subsection{Band-edge convolution estimate}

\begin{remark}[Why a naive approach fails]\label{rem:naivefail}
One might try to bound $\int_{\omega-\eta}^\omega|g_m(u)|^2\,du$
by the leakage $\ell_m$.
However, as noted in Remark~\ref{rem:leakage-meaning},
$\ell_m$ measures out-of-band energy while the near-edge integral
is in-band energy. These objects are in general unrelated.
The correct approach, given in Lemma~\ref{lem:bandedge} below,
bounds the near-edge in-band mass by $\lambda_m(c) \le 1$.
To obtain exponential decay of the in-band$\times$in-band term,
one needs the finer near-edge control of Assumption~\ref{ass:nearedge}.
\end{remark}

\begin{lemma}[Geometric support restriction]\label{lem:geom}
Let $g_m, g_n$ be supported in $[-\omega,\omega]$, and let
$\xi > \omega$. Then
\[
  (g_m * g_n)(\xi)
  = \int_{\xi-\omega}^{\omega} g_m(\xi-\eta)\,g_n(\eta)\,d\eta.
\]
In particular, every $\eta$ contributing to this integral satisfies
$\eta \in (\xi-\omega,\,\omega)$, and hence:
\begin{equation}\label{eq:geom}
  \eta \in (-\omega, \omega),\quad \xi-\eta \in (-\omega,\omega),
  \quad \xi > \omega
  \quad\Longrightarrow\quad
  \eta > \xi - \omega.
\end{equation}
For $\xi \in (\omega, \omega + \delta_{\mathrm{ne}})$, this forces
$\eta \in (\xi - \omega,\, \omega) \subset
(\omega - \delta_{\mathrm{ne}},\, \omega)$.
\end{lemma}

\begin{proof}
Since $g_m$ is supported on $[-\omega,\omega]$, the integrand
$g_m(\xi - \eta)$ is zero unless $\xi - \eta \in [-\omega,\omega]$,
i.e.\ $\eta \in [\xi-\omega, \xi+\omega]$.
Combined with $g_n$ supported on $[-\omega,\omega]$ (so $\eta \in [-\omega,\omega]$),
the effective integration domain is $\eta \in [\xi-\omega, \omega]$.
For $\xi > \omega$ this interval is non-empty iff $\xi < 2\omega$,
and its lower endpoint satisfies $\xi - \omega > 0$.
Implication~\eqref{eq:geom} is immediate.
For $\xi \in (\omega, \omega+\delta_{\mathrm{ne}})$, the lower endpoint
$\xi - \omega \in (0, \delta_{\mathrm{ne}})$, so
$\eta > \xi - \omega > \omega - \delta_{\mathrm{ne}}$,
giving $\eta \in (\omega-\delta_{\mathrm{ne}}, \omega)$.
\end{proof}

\begin{remark}[Role of Lemma~\ref{lem:geom} in the bulk proof]
Lemma~\ref{lem:geom} makes the support restriction in Step~3 of
Theorem~\ref{thm:bulk} completely explicit: the restriction of
the convolution integral to $\eta \in (\omega-\delta_{\mathrm{ne}},\omega)$
is an exact consequence of the support geometry,
not an approximation. This is what makes the application of
Assumption~\ref{ass:nearedge} in the bulk proof watertight.
\end{remark}

\begin{lemma}[Band-edge convolution estimate]\label{lem:bandedge}
Let $g_m, g_n \in L^2(\mathbb{R})$ with
$\operatorname{supp}g_m, \operatorname{supp}g_n \subset [-\omega, \omega]$.
Then:
\begin{equation}\label{eq:bandedge}
  \|g_m * g_n\|_{L^2(|\xi|>\omega)}^2
  \le 4\omega\,\|g_m\|_{L^2}^2\,\|g_n\|_{L^2}^2
  \le 4\omega.
\end{equation}
When $g_k = \mathbf{1}_{[-\omega,\omega]}\hat\psi_k^{(c)}$,
we have $\|g_k\|_{L^2}^2 = \lambda_k(c)$, giving:
\begin{equation}\label{eq:bandedge-pswf}
  \|g_m * g_n\|_{L^2(|\xi|>\omega)}^2
  \le 4\omega\,\lambda_m(c)\,\lambda_n(c)
  \le 4\omega.
\end{equation}
This bound is an absolute constant in $\omega$,
independent of $c$, $m$, and $n$.
\end{lemma}

\begin{proof}
By Lemma~\ref{lem:geom}, for $\xi \in (\omega, 2\omega)$:
$(g_m*g_n)(\xi) = \int_{\xi-\omega}^{\omega} g_m(\xi-\eta)g_n(\eta)\,d\eta$
with interval length $2\omega - \xi \le \omega$.
Apply Cauchy--Schwarz:
\[
  |(g_m*g_n)(\xi)|^2
  \le (2\omega-\xi)\int_{\xi-\omega}^{\omega}|g_m(\xi-\eta)|^2|g_n(\eta)|^2\,d\eta.
\]
Integrate over $\xi \in (\omega, 2\omega)$ and apply Fubini.
The domain is
$\{(\xi,\eta): \omega < \xi < 2\omega,\; \xi-\omega < \eta < \omega\}
= \{0 < \eta < \omega,\; \omega < \xi < \omega+\eta\}$.
Thus:
\[
  \|g_m*g_n\|_{L^2(\omega,2\omega)}^2
  \le \int_{0}^{\omega}|g_n(\eta)|^2
  \int_{\omega}^{\omega+\eta}(2\omega-\xi)|g_m(\xi-\eta)|^2\,d\xi\,d\eta.
\]
Substituting $u = \xi - \eta$:
\[
  \int_{\omega}^{\omega+\eta}(2\omega-\xi)|g_m(\xi-\eta)|^2\,d\xi
  \le 2\omega\int_{\omega-\eta}^{\omega}|g_m(u)|^2\,du
  \le 2\omega\,\|g_m\|_{L^2}^2.
\]
Therefore $\|g_m*g_n\|_{L^2(\omega,2\omega)}^2
\le 2\omega\,\|g_m\|_{L^2}^2\,\|g_n\|_{L^2}^2$.
The symmetric bound for $\xi \in (-2\omega,-\omega)$ is identical,
giving the full bound $4\omega\,\|g_m\|_{L^2}^2\,\|g_n\|_{L^2}^2$.
\end{proof}

\begin{remark}[What the estimate actually says]\label{rem:bandedge-interpretation}
Lemma~\ref{lem:bandedge} gives the bound
$\|g_m*g_n\|_{L^2(|\xi|>\omega)}^2 \le 4\omega\lambda_m(c)\lambda_n(c) \le 4\omega$.
This is an \emph{absolute constant} in $\omega$, independent of $c$,
$m$, and $n$. It does \emph{not} provide exponential decay.
The three cross-terms (i)--(iii) in Lemma~\ref{lem:oob} carry factors
of $\ell_m$ or $\ell_n$, which are exponentially small by Landau--Widom
(Assumption~\ref{ass:lw}).
The in-band$\times$in-band term (iv) requires
the separate input of Assumption~\ref{ass:nearedge},
made precise in Theorem~\ref{thm:bulk}.
\end{remark}

\section{Bulk Regime: Exponential Tail Decay}\label{sec:bulk}
%%=================================================================

\begin{theorem}[Bulk spectral tail bound]\label{thm:bulk}
Let $\gamma \in (0,1)$ and $\varepsilon \in (0,2/\pi)$ with
$\gamma < 1-\pi\varepsilon/2$.
Assume Assumptions~\ref{ass:lw} and~\ref{ass:nearedge},
$N \le (2/\pi-\varepsilon)c$, and $c \ge c_0(\varepsilon,\gamma)$.
Then for all $m, n \le \gamma N$:
\[
  \|(I-\PN)f_{mn}\|_{L^2([-T,T])}
  \le C_\gamma\,e^{-\alpha_\gamma N},
\]
where $C_\gamma, \alpha_\gamma > 0$ depend only on
$\gamma$, $\varepsilon$, $\omega$, and $T$.
\end{theorem}

\begin{proof}
Apply Lemma~\ref{lem:slepian}:
\begin{equation}\label{eq:bulk-start}
  \|(I-\PN)f_{mn}\|_{L^2}^2
  \le \frac{\Eout(f_{mn})}{1-\lambda_{N-1}(c)}.
\end{equation}

\textbf{Step 1: Prefactor.}
By Assumption~\ref{ass:lw} with $c \ge c_0(\varepsilon)$,
$1-\lambda_{N-1}(c) \le Ce^{-\eta c} < 1/2$, so
$1/(1-\lambda_{N-1}(c)) \le 2$.

\textbf{Step 2: Leakage bound.}
For $m, n \le \gamma N$ with $\gamma < 1-\pi\varepsilon/2$,
Assumption~\ref{ass:lw} gives
$\ell_m, \ell_n \le C(\varepsilon')e^{-\eta(\varepsilon')c}$
with $\varepsilon' = \varepsilon/2$.

\textbf{Step 3: Bounding $\Eout(f_{mn})$.}
For the cross-terms via~\eqref{eq:gmrn}--\eqref{eq:rmrn}:
\[
  \|g_m*r_n\|^2 + \|r_m*g_n\|^2 + \|r_m*r_n\|^2
  \le 2(4\omega+1)C(\varepsilon')e^{-\eta(\varepsilon')c}.
\]

For the in-band$\times$in-band term: by Lemma~\ref{lem:geom},
for every $\xi \in (\omega, \omega + \delta_{\mathrm{ne}})$,
every $\eta$ contributing to $(g_m*g_n)(\xi)$ satisfies
$\eta \in (\omega - \delta_{\mathrm{ne}}, \omega)$.
The Fubini--Cauchy--Schwarz estimate from Lemma~\ref{lem:bandedge}
therefore gives:
\begin{align*}
  \|g_m*g_n\|_{L^2(|\xi|>\omega)}^2
  &\le 4\omega \cdot \|g_m\|_{L^2}^2 \cdot
  \int_{\omega-\delta_{\mathrm{ne}}}^{\omega}|g_n(\eta)|^2\,d\eta \\
  &\le 4\omega \cdot C_{\mathrm{ne}}(\gamma)\,e^{-\eta_{\mathrm{ne}}(\gamma)c},
\end{align*}
where we used $\|g_m\|_{L^2}^2 \le 1$ and
Assumption~\ref{ass:nearedge}.
By symmetry the same bound holds with $m$ and $n$ interchanged.

Combining via~\eqref{eq:oob-decomp}:
\[
  \Eout(f_{mn})
  \le C_1 e^{-\eta_{\min}c},
  \quad
  \eta_{\min} = \min(\eta(\varepsilon'), \eta_{\mathrm{ne}}(\gamma)) > 0.
\]

\textbf{Step 4: Conclusion.}
\[
  \|(I-\PN)f_{mn}\|_{L^2}^2
  \le 2C_1 e^{-\eta_{\min}c}
  \le 2C_1 e^{-\eta_{\min}(2/\pi-\varepsilon)N}.
\]
Setting $\alpha_\gamma = \tfrac{1}{2}\eta_{\min}(2/\pi-\varepsilon)$
and $C_\gamma = (2C_1)^{1/2}$ gives the result.
\end{proof}

\begin{remark}[Role of Assumption~\ref{ass:nearedge} in the proof]
\label{rem:termiv-role}
Without Assumption~\ref{ass:nearedge}, the argument yields only
$\|(I-\PN)f_{mn}\|_{L^2}^2 \le C$ (bounded but not decaying),
since Lemma~\ref{lem:bandedge} alone gives $O(\omega)$.
Assumption~\ref{ass:nearedge} replaces
$\int_{\omega-\delta}^\omega|g_n|^2 \le 1$
by $\le C_{\mathrm{ne}} e^{-\eta_{\mathrm{ne}} c}$;
combined with the support geometry of Lemma~\ref{lem:geom},
this drives the in-band$\times$in-band term to zero exponentially.
This is the precise and only use of Assumption~\ref{ass:nearedge}.
\end{remark}

\begin{remark}[Explicit constants and degeneration]
$\alpha_\gamma = \tfrac{1}{2}\min(\eta(\varepsilon/2),\eta_{\mathrm{ne}}(\gamma))
\cdot (2/\pi-\varepsilon)$.
As $\gamma \to 1^-$, $\varepsilon \to 0$ and $\alpha_\gamma \to 0$;
see Section~\ref{sec:edge}.
\end{remark}

\section{Off-Diagonal Regime: Algebraic Tail Decay}
\label{sec:offdiag}
%%=================================================================

\begin{lemma}[Eigenvalue gap estimate]\label{lem:gap}
For $0 \le m,n \le N$ with $m \ne n$, $|m-n| \ge \delta N$
($\delta > 0$ fixed), and $N \le (2/\pi)c$:
\begin{equation}\label{eq:gap}
  |\chi_m(c) - \chi_n(c)| \ge c_1\,|m-n|\cdot N,
\end{equation}
where $c_1 > 0$ is an absolute constant.
Since $|m-n| \ge \delta N$ implies $N \ge |m-n|/\delta$:
\begin{equation}\label{eq:gap2}
  |\chi_m(c) - \chi_n(c)| \ge \frac{c_1}{\delta}\,|m-n|^2.
\end{equation}
\end{lemma}

\begin{proof}
The Sturm--Liouville eigenvalues satisfy
$\chi_k(c) = k(k+1) + c^2/2 + O(k^{-1}c^2)$
\cite[Ch.~2]{OsipovRokhlinXiao}, so:
\[
  \chi_m(c) - \chi_n(c)
  = (m-n)(m+n+1) + O(N^{-1}c^2).
\]
For $m,n \le N$ with $N \le (2/\pi)c$, the error is
$O(N^{-1}c^2) = O(N)$ (using $c \le \pi N/2$).
The leading term satisfies $(m-n)(m+n+1) \ge |m-n|\cdot N/C_0$
for some absolute $C_0$, since $m+n+1 \ge \max(m,n) \ge N/2$
when $\max(m,n) \ge N/2$. For $c$ large enough the error is
absorbed, giving~\eqref{eq:gap}.
Bound~\eqref{eq:gap2} follows from $N \ge |m-n|/\delta$.
\end{proof}

\begin{theorem}[Off-diagonal spectral tail bound]\label{thm:offdiag}
Let $\delta \in (0,1)$ be fixed. For every integer $p \ge 1$
and all $m, n$ with $|m-n| \ge \delta N$, $0 \le m,n \le N$:
\[
  \|(I-\PN)f_{mn}\|_{L^2([-T,T])}
  \le C_p\,(1+|m-n|)^{-p},
\]
where $C_p > 0$ depends only on $p$, $\delta$, $\omega$, $T$,
and absolute constants.
\end{theorem}

\begin{proof}
Let $D_c$ denote the self-adjoint Sturm--Liouville operator on
$L^2([-T,T])$ with $D_c\psi_k^{(c)} = \chi_k(c)\psi_k^{(c)}$.

\textbf{Gap.} By Lemma~\ref{lem:gap}, since $|m-n| \ge \delta N$
implies $N \ge |m-n|/\delta$:
\[
  |\chi_m(c)-\chi_n(c)|
  \ge c_1\,|m-n|\cdot N
  \ge \frac{c_1}{\delta}\,|m-n|^2.
\]

\textbf{Decay of coefficients.}
Integrating by parts $p$ times using self-adjointness of $D_c$:
\[
  \ip{f_{mn}}{\psi_k^{(c)}}
  = \frac{1}{(\chi_m(c)-\chi_n(c))^p}
  \ip{D_c^{[p]}f_{mn}}{\psi_k^{(c)}}.
\]
The $L^2$-norm of $D_c^{[p]}f_{mn}$ is bounded by
$B_p \le C_p(CN^2)^p$ (Leibniz rule, $\chi_k \le CN^2$).
By Cauchy--Schwarz and~\eqref{eq:gap}:
\[
  |\ip{f_{mn}}{\psi_k^{(c)}}|
  \le \frac{C_p(CN^2)^p}{\bigl(c_1|m-n|N\bigr)^p}
  = \frac{\hat{C}_p}{|m-n|^p}.
\]

\textbf{Conclusion.}
By Bessel's inequality:
\[
  \|(I-\PN)f_{mn}\|_{L^2}
  \le \frac{\hat{C}_p T}{(1+|m-n|)^p}.
\qedhere
\]
\end{proof}

\begin{corollary}[Partial verification of Conjecture~3.1 of
\cite{PaperII_quad}]\label{cor:partial}
Under Assumption~\ref{ass:nearedge}, in the bulk and off-diagonal
regimes combined:
\[
  \|(I-\PN)f_{mn}\|_{L^2([-T,T])}
  \le C\bigl(e^{-\alpha N} + (1+|m-n|)^{-p}\bigr)
\]
for all $(m,n)$ with either $m,n \le \gamma N$ or $|m-n| \ge \delta N$.
The edge regime $\{m,n \ge (1-\delta)N,\;|m-n|<\delta N\}$ is not covered.
\end{corollary}

\section{Edge Regime and Obstruction to Uniform Bounds}
\label{sec:edge}
%%=================================================================

\subsection{Failure of the Slepian-identity argument}

For $m,n \sim N$ near the Slepian cutoff $N \approx 2c/\pi$,
the eigenvalue $\lambda_{N-1}(c)$ lies in the Landau--Widom
transition zone, where $\ell_n = 1-\lambda_n(c) = O(1)$.
Thus $\Eout(f_{mn}) = O(1)$, and the Slepian identity
\eqref{eq:tail-bound} yields only $\|(I-\PN)f_{mn}\|^2 = O(1)$.

\subsection{Failure of the oscillation argument}

For near-diagonal pairs in the edge zone, $|m-n| = O(1)$,
the gap $|\chi_m(c)-\chi_n(c)| = O(N)$ by Lemma~\ref{lem:gap},
giving only $O(N^{-p})$ coefficient bounds.
For the diagonal $m=n$, the product $f_{nn} = |\psin|^2$
is non-negative and no oscillation cancellation is available.

\subsection{The bandwidth doubling obstruction}

\begin{proposition}[Bandwidth doubling lower bound]\label{prop:bwdoubling}
For $n$ in the Landau--Widom transition zone, i.e.,
$|n - 2c/\pi| \le c^{1/2}$:
\[
  \Eout(|\psi_n^{(c)}|^2) \ge c_0 > 0,
\]
where $c_0 > 0$ is an absolute constant independent of $c$ and $n$.
\end{proposition}

\begin{proof}
By Widom transition-zone asymptotics \cite[Ch.~4]{OsipovRokhlinXiao},
$\lambda_n(c) \ge 1/4$ and $1-\lambda_n(c) \ge 1/4$
for $c$ sufficiently large.

%% ----------------------------------------------------------------
%% FIX 3: Explicit half-bandwidth / Slepian-cutoff argument
%% ----------------------------------------------------------------
\textbf{Step 1: Positive Fourier mass near the band edge.}
We claim:
\begin{equation}\label{eq:edge-mass-lower}
  \int_{\omega/2}^{\omega}|\hat\psi_n^{(c)}(\xi)|^2\,d\xi \ge \delta_0
\end{equation}
for an absolute constant $\delta_0 > 0$.

Suppose for contradiction that
$\int_{\omega/2}^\omega|\hat\psi_n^{(c)}|^2 < \delta_0$.
Since $\int_{-\omega}^\omega|\hat\psi_n^{(c)}|^2 = \lambda_n(c) \ge 1/4$,
the Fourier mass is then concentrated on $(-\omega/2, \omega/2)$:
\[
  \int_{-\omega/2}^{\omega/2}|\hat\psi_n^{(c)}(\xi)|^2\,d\xi
  \ge \tfrac{1}{4} - 2\delta_0.
\]
Now consider the concentration operator for the \emph{half-bandwidth}
$\omega/2$, with time-bandwidth product $c_{1/2} = (\omega/2)\cdot T = c/2$.
Its Slepian cutoff is $N_{1/2} \approx 2c_{1/2}/\pi = c/\pi$.
Since $n \approx 2c/\pi \gg c/\pi = N_{1/2}$, the index $n$ lies
deep in the superexponentially decaying regime for bandwidth $\omega/2$.
By Assumption~\ref{ass:lw} applied with $c_{1/2} = c/2$ and
$n \ge (2/\pi + \varepsilon_0)c_{1/2}$ (for some fixed $\varepsilon_0 > 0$
depending only on the transition zone condition $|n-2c/\pi| \le c^{1/2}$):
\[
  \lambda_n^{(\omega/2)}(c)
  := \int_{-\omega/2}^{\omega/2}|\hat\psi_n^{(c)}(\xi)|^2\,d\xi
  \le Ce^{-\eta c/2} \to 0.
\]
But this contradicts $\int_{-\omega/2}^{\omega/2}|\hat\psi_n^{(c)}|^2
\ge \tfrac{1}{4} - 2\delta_0 > 0$ for $\delta_0 < 1/8$ and $c$ large.
Hence~\eqref{eq:edge-mass-lower} holds with $\delta_0 = 1/16$.

\textbf{Step 2: Lower bound for out-of-band convolution energy.}
Let $h := \hat\psi_n^{(c)} * \hat\psi_n^{(c)}$.
By Lemma~\ref{lem:geom}, for $\xi \in (\omega, 3\omega/2)$,
every $\eta$ contributing to $h(\xi)$ satisfies
$\eta \in (\xi-\omega, \omega)$, and since $\xi - \omega < \omega/2$,
we have $\eta \in (\xi-\omega, \omega) \supset (\omega/2, \omega)$
partially. A direct computation using~\eqref{eq:edge-mass-lower}:
\[
  \int_\omega^{2\omega}|h(\xi)|^2\,d\xi
  \ge \frac{\delta_0^2}{\omega} > 0.
\]

\textbf{Step 3: Conclusion.}
$\Eout(|\psi_n^{(c)}|^2)
\ge \delta_0^2/\omega =: c_0 > 0$.
\end{proof}

\begin{remark}[Structural consequence]
Proposition~\ref{prop:bwdoubling} shows that the diagonal products
$|\psi_n^{(c)}|^2$ in the edge regime are \emph{genuinely not
approximable} by any finite prolate subspace with exponentially
small error. The out-of-band Fourier energy is bounded away from
zero by an absolute constant.
\end{remark}

\begin{remark}[Why new tools are needed]
Any proof for the edge regime must address at least one of:
\begin{enumerate}[(a)]
  \item Controlling $f_{mn}$ in the \emph{doubled} prolate subspace
  $V_{2N,2c}$ and relating $P_{2N,2c}$ back to $\PN$.
  \item Exploiting a product formula for PSWFs, currently unavailable
  \cite{OsipovRokhlinXiao}.
  \item Reformulating the quadrature condition to avoid the full
  product family.
\end{enumerate}
Route~(a) is addressed in Problem~\ref{prob:doubled}.
\end{remark}

\section{Open Problems}\label{sec:open}
%%=================================================================

\begin{problem}[Near-edge exponential control for bulk PSWFs]
\label{prob:nearedge}
Prove Assumption~\ref{ass:nearedge}: for fixed $\gamma \in (0,1)$
and all $n \le \gamma N$, $c \ge c_0(\gamma)$:
\[
  \int_{|\xi| \in (\omega - \delta_{\mathrm{ne}}, \omega)}
  |\hat\psi_n^{(c)}(\xi)|^2\,d\xi
  \le C_{\mathrm{ne}}(\gamma)\,e^{-\eta_{\mathrm{ne}}(\gamma)c}.
\]
A proof would follow from pointwise WKB estimates for
$|\hat\psi_n^{(c)}(\xi)|$ near $\xi = \pm\omega$, integrated over
a width-$\delta_{\mathrm{ne}}$ interval.
\end{problem}

\begin{problem}[Edge regime: uniform tail bound]\label{prob:edge}
Prove or disprove: there exist $C, \alpha > 0$ such that
$\|(I-\PN)f_{mn}\|_{L^2} \le Ce^{-\alpha N}$
for all $m,n \sim N$.
Proposition~\ref{prop:bwdoubling} rules out exponential decay
on the diagonal.
\end{problem}

\begin{problem}[Doubled prolate subspace]\label{prob:doubled}
Let $V_{2N,2c}$ be the prolate subspace for bandwidth $2\omega$.
Show that every $f_{mn}$ with $0 \le m,n \le N$ satisfies
$\|f_{mn} - P_{2N,2c}f_{mn}\|_{L^2} \le Ce^{-\alpha N}$
uniformly.
\end{problem}

\begin{problem}[Sharp bulk decay rate]
Determine the sharp rate $\alpha_\gamma$ as a function of
$\gamma$ and $\varepsilon$; in particular, whether
$\alpha_\gamma \to 0$ as $\gamma \to 1^-$.
\end{problem}

\begin{problem}[Transition in the diagonal tail]
For fixed $N \sim c$ and $n$ varying from $0$ to $N$,
determine the precise asymptotic behavior of
$\|(I-\PN)|\psi_n^{(c)}|^2\|_{L^2}$.
\end{problem}
