\begin{abstract}
We study the nonlinear phase operator $\mathcal{T}_c$ arising in the
Riemann--Hilbert formulation of prolate spheroidal wave function (PSWF)
asymptotics.
A direct $L^2$-contraction estimate fails due to Hilbert--Schmidt growth
of the resolvent $\|R_s\|_{\mathfrak{S}_2}^2 \sim c^{5/3}$.
After Galerkin projection onto the leading $N \sim c^{1/3}$ PSWF modes,
we establish:
\[
  \norm{D\mathcal{T}_c^{(N)}}_{\ell^2 \to \ell^2}
  \leq C(A)\,\log c
  \quad\text{(under Assumption~B, unconditional in }c\text{-rate)},
\]
and, conditionally,
\[
  \norm{D\mathcal{T}_c^{(N)}}_{\ell^2 \to \ell^2}
  \leq C(A)\, c^{-1/2}\log c \to 0
  \quad\text{(under Assumptions~B and B$'$)}.
\]
\textbf{The first bound is a norm estimate, not a contraction:}
$\log c \to \infty$, so it does not imply $\|D\mathcal{T}_c^{(N)}\| < 1$
for large $c$.
Only the second bound, conditional on Assumption~B$'$
(Schur-summability hypothesis, Section~\ref{sec:norm}),
gives a genuine contraction for $c$ sufficiently large.
All $c$-dependencies are tracked explicitly; no factor is absorbed
without justification.
The unconditional structural result rests on three independently proved steps:
(i)~diagonal vanishing $T_{ii} = 0$ from the commutator structure;
(ii)~IBP combined with the SL Wronskian identity:
    $|I_{kl}| \leq C c^{-1/2}/|k-l|$ (Lemma~\ref{lem:RL});
(iii)~pointwise matrix bound $|T_{ij}| \leq C_1 C_2/|i-j|$
    under Assumption~B, and Schur harmonic sum $\sum 1/k \sim \log N$.
The transition zone provides the stronger local scale
$c^{-2/3}/|k-l|$, but does \emph{not} improve the global Schur row sum,
because the bulk zone dominates (Remark~\ref{rem:tz_irrelevant}).
Section~\ref{sec:dyadic} provides a dyadic decomposition of the Schur row sum
and establishes a No-go Theorem (Theorem~\ref{thm:nogo}):
no argument based solely on pointwise bounds $|T_{ij}| \leq C/|i-j|$
can yield a contraction; the logarithmic growth is optimal in this class.
Assumption~B$'$ is therefore a \emph{necessary} structural hypothesis,
not merely a technical convenience.
Section~\ref{sec:dyadic} further proves that spectral weights
$w_k = (1-\lambda_k)^\alpha$ are asymptotically constant across the
Galerkin window and cannot modify the Schur kernel
(Proposition~\ref{prop:spectral_weight_fail}),
and that naive index-based weights also fail to make the harmonic sum summable.
The correct weighted-norm route, if it exists, requires an asymmetric
or two-weight Schur estimate (Open Problem~\ref{op:weightnorm}).
The remaining gaps (Assumptions~B and~B$'$) are precisely identified
as Open Problems~\ref{op:prefactor} and~\ref{op:weightnorm}.
\end{abstract}

\section{Introduction and Context}
%% ============================================================

\subsection{Programme overview}

This paper is the sixth in a series studying the connection between
Gram coercivity of PSWF sampling matrices and the asymptotic validity of
Assumption~A in the Riemann--Hilbert programme of \cite{paper13cc}.
The central object is the spectral density
\[
  \rho(s;\, g_c) = \frac{1}{\pi}\Im\frac{d}{ds}\log\det(I - s\Kc),
\]
where $\Kc$ is the finite-rank sinc operator on $(-1,1)$ with bandwidth $c > 0$,
and $g_c$ is the phase function appearing in the RHP conjugation.
Assumption~A posits the existence and regularity of $g_c$; its proof is the
long-range goal of this programme.

\subsection{The nonlinear operator $\mathcal{T}_c$}

Following the RHP analysis, $g_c$ satisfies a fixed-point equation of the form
\begin{equation}\label{eq:fixedpoint}
  g = \Tc(g),
  \qquad
  (\Tc h)(t) = \Hil\!\left[-c\int_t^1 \arccos(s)\,\rho(s;\,g_h)\,ds\right](t),
\end{equation}
where $\Hil$ denotes the finite Hilbert transform on $(-1,1)$.
Proving $\|D\Tc^{(N)}\| < 1$ in some Banach space would yield
Assumption~A via the Banach fixed-point theorem.
The purpose of this paper is to identify precisely what is proved,
what is assumed, and what remains open.

\subsection{Why global $L^2$-contraction fails}

A direct computation (Section~\ref{sec:resolvent}) shows that the linearised
operator $D\Tc$ satisfies
\[
  \norm{D\Tc}_{L^2 \to L^2} \gtrsim c^{8/3},
\]
due to growth of $\norm{R_s}_{\mathfrak{S}_2}^2 \sim c^{5/3}$ combined with
the Widom prefactor $c$.
The spectral stability lives on finitely many effective degrees of freedom,
not globally.

\subsection{Main results}

\begin{theorem}[Galerkin norm estimate]\label{thm:main}
Let $N = \lfloor A c^{1/3}\rfloor$ for a constant $A > 0$, and let
$P_N : L^2(-1,1) \to X_c^{(N)}$ denote the orthogonal projection onto
the first $N$ PSWF modes. Define
$\mathcal{T}_c^{(N)} := P_N \circ \Tc \circ P_N$.

\medskip
\noindent\textbf{(a) Norm bound}
(under Assumption~\ref{ass:B} only):
\[
  \norm{D\mathcal{T}_c^{(N)}}_{\ell^2 \to \ell^2}
  \leq C(A)\,\log c.
\]
Since $\log c \to \infty$ as $c \to \infty$, part~(a) is
\textbf{not} a contraction statement.
It establishes the correct structural form of the Schur sum
and shows that the norm grows at most logarithmically;
it does not imply $\|D\mathcal{T}_c^{(N)}\| < 1$ for large $c$.

\medskip
\noindent\textbf{(b) Contraction}
(under both Assumption~\ref{ass:B} and Assumption~\ref{ass:Bprime}):
\[
  \norm{D\mathcal{T}_c^{(N)}}_{\ell^2 \to \ell^2}
  \leq C(A)\, c^{-1/2}\log c \;\xrightarrow{c\to\infty}\; 0.
\]
Hence $\mathcal{T}_c^{(N)}$ is a genuine contraction for $c \geq c_0$.
The gap between parts~(a) and~(b) is the content of
Assumption~\ref{ass:Bprime}; closing it is Open Problem~\ref{op:prefactor}.
\end{theorem}

\begin{theorem}[No-go theorem for unweighted Schur contraction]%
\label{thm:nogo}
Let $T$ be any $N \times N$ matrix with $T_{ii} = 0$ and
\[
  |T_{ij}| \leq \frac{\kappa}{|i-j|}
  \qquad \text{for all } i \neq j,
\]
for some constant $\kappa > 0$ independent of $N$.
Then:
\[
  \sup_{i} \sum_{j \neq i} |T_{ij}| \geq \kappa H_{N-1} \sim \kappa \log N.
\]
In particular, $\norm{T}_{\ell^2 \to \ell^2} \geq \kappa H_{N-1}/2$.
For $N = \lfloor A c^{1/3} \rfloor$, this gives
$\norm{D\mathcal{T}_c^{(N)}} \geq (\kappa/2)\cdot\tfrac{1}{3}\log c$,
so the $\log c$ growth in Theorem~\ref{thm:main}(a) is optimal
within the class of matrices satisfying $|T_{ij}| \leq \kappa/|i-j|$.

\medskip
\noindent Consequently, no argument relying solely on pointwise bounds
$|T_{ij}| \leq C/|i-j|$ can establish $\norm{DT} < 1$;
a genuine contraction requires either
\begin{itemize}[nosep]
  \item[(i)] a collective cancellation in row sums
             (i.e., Assumption~\ref{ass:Bprime}), or
  \item[(ii)] a change of norm (i.e., Open Problem~\ref{op:weightnorm}).
\end{itemize}
\end{theorem}

\begin{remark}[The Schur test delivers a norm bound, not a contraction]%
\label{rem:schur_honest}
The argument behind Theorem~\ref{thm:main}(a) is:
$|T_{ij}| \leq C_1 C_2/|i-j|$ (Proposition~\ref{prop:Tij})
combined with
\[
  \sum_{k=1}^{N} \frac{1}{k} \;=\; H_N
  \;\leq\; 1 + \log N
  \;\leq\; 1 + \tfrac{1}{3}\log c + \log A.
\]
The harmonic series $H_N \to \infty$; the Schur test therefore gives
$\|T\|_{\ell^2 \to \ell^2} = O(\log c)$, not $o(1)$.
By Theorem~\ref{thm:nogo}, this growth is optimal: no refinement
of the pointwise bound $|T_{ij}| \leq C/|i-j|$ can yield a contraction.
The \emph{sole} path to a genuine contraction within the $\ell^2$ framework
is Assumption~\ref{ass:Bprime}.
\end{remark}

\begin{remark}[On the role of a weighted norm]\label{rem:weightnorm}
The Schur-test argument operates in the standard $\ell^2$ norm on
Galerkin coefficients.
An alternative route to contraction --- without Assumption~\ref{ass:Bprime} ---
would be to identify a $c$-dependent weighted norm
$\|\cdot\|_w$ on $X_c^{(N)}$ under which $\|DT\|_w < 1$ unconditionally.
Section~\ref{sec:spectral_weights} shows that the natural spectral-weight
candidate $w_k = (1-\lambda_k)^\alpha$ fails structurally, and that
naive index weights also fail.
The correct weighted-norm route, if it exists, requires an asymmetric
or two-weight Schur estimate; see Open Problem~\ref{op:weightnorm}.
\end{remark}

\subsection{The weighted space $X_c$}

\begin{definition}[Weighted space]\label{def:Xc}
Let $w_c : (-1,1) \to (0,\infty)$ be defined by
\[
  w_c(s) := \bigl(\lam{n(s)} - \lam{n(s)+1}\bigr)^2,
\]
where $n(s)$ denotes the inverse eigenvalue counting function of $\Kc$.
Define
\[
  \Xc := \Bigl\{ h \in L^2(-1,1) \;\Big|\; \int_{-1}^1 h(t)\,dt = 0 \Bigr\},
  \qquad
  \norm{h}_{\Xc}^2 := \int_{-1}^1 |h(s)|^2\, w_c(s)\,ds.
\]
\end{definition}

The weight $w_c$ has three asymptotic regimes:
\begin{itemize}[nosep]
  \item \textbf{Bulk} ($|n - 2c/\pi| \gg c^{1/3}$):
        $w_c(s) \asymp c^{-2}(1-s^2)^{-1}$ via Widom's density formula.
  \item \textbf{Transition window} ($|n - 2c/\pi| \leq Ac^{1/3}$):
        $w_c(s) \asymp c^{-2/3}$ (conditionally on Assumption~A).
  \item \textbf{Outer region}: exponential decay, negligible in $L^2$.
\end{itemize}

\subsection{Muckenhoupt $A_2$-property of $w_c$}

\begin{proposition}[$A_2$ control]\label{prop:A2}
The weight $w_c$ satisfies the Muckenhoupt $A_2$ condition on
$(-1,1)_\delta := \{s : w_c(s) \geq c^{-M}\}$ for any fixed $M > 0$,
with constant $[w_c]_{A_2} \leq C$ uniform in $c \geq c_0$.
\end{proposition}

\begin{proof}[Proof sketch]
\textbf{Bulk zone.} Here $w_c(s) \asymp c^{-2}(1-s^2)^{-1}$.
The $A_2$ condition reduces to the classical bound for the Jacobi weight
$(1-s^2)^{\pm 1}$ (Hunt--Muckenhoupt--Wheeden, 1973); the $c$-factors cancel.

\textbf{Transition zone.} $w_c \asymp c^{-2/3}$ is constant, hence
$[w_c]_{A_2} = 1$.

\textbf{Outer zone.} Excluded by the cutoff; complementary $L^2$-mass
is $O(e^{-\gamma c^{1/2}})$ and negligible.
\end{proof}

\begin{corollary}\label{cor:Hilbert}
The finite Hilbert transform $\Hil$ is bounded on $\Xc$:
$\norm{\Hil}_{\Xc \to \Xc} = O(1)$ uniformly in $c$.
\end{corollary}

\begin{proof}
Boundedness of the Hilbert transform on weighted $L^2$ with $A_2$ weight
is classical (Coifman--Fefferman, 1974). Apply Proposition~\ref{prop:A2}.
\end{proof}

\subsection{Galerkin projection}

\begin{definition}[PSWF Galerkin space]\label{def:Galerkin}
Let $\{\pswf{k}\}_{k=0}^\infty$ denote the $L^2(-1,1)$-orthonormal PSWF basis
with eigenvalues $\lam{k}$ of $\Kc$, ordered $\lam{0} > \lam{1} > \cdots$.
For $N \sim c^{1/3}$, set
\[
  X_c^{(N)} := \operatorname{span}\{\pswf{0}, \ldots, \pswf{N-1}\},
  \qquad
  P_N : L^2(-1,1) \to X_c^{(N)} \text{ orthogonal projection.}
\]
\end{definition}

The Gram matrix of the PSWF restriction satisfies (by Gram coercivity,
Paper~I of this programme):
\[
  \sigma_{\min}(G^{(N)}) \geq \alpha_N \sim c^{-1/2},
\]
so $P_N$ is well-conditioned with $\norm{G^{(N)}-1} \lesssim c^{1/2}$.

\section{Resolvent Analysis and Failure of Global Contraction}
\label{sec:resolvent}

\subsection{Spectral density variation formula}

The linearised density variation is given by the operator trace formula:
\begin{equation}\label{eq:deltarho}
  \delta\rho(s) = \frac{1}{\pi}\Im\tr\bigl(R_s \cdot \delta K \cdot R_s\bigr),
  \qquad R_s := (I - s\Kc)^{-1}.
\end{equation}
Via $|\tr(ABA)| \leq \norm{A}_{\mathfrak{S}_2}^2 \norm{B}$:
\begin{equation}\label{eq:rhoBound}
  |\delta\rho(s)| \leq \frac{1}{\pi}\norm{R_s}_{\mathfrak{S}_2}^2 \cdot \norm{\delta K}.
\end{equation}

\subsection{Hilbert--Schmidt growth}

\begin{lemma}[HS norm of resolvent]\label{lem:HS}
$\displaystyle\norm{R_s}_{\mathfrak{S}_2}^2 = \sum_{k=0}^\infty
\frac{1}{(1-s\lam{k})^2} \sim c^{5/3}$ near the transition window $s \to 1$.
\end{lemma}

\begin{proof}
The transition window contributes $\sim c^{1/3}$ modes with
$1 - \lam{k} \asymp c^{-2/3}$, each contributing $c^{4/3}$,
giving total $c^{1/3} \cdot c^{4/3} = c^{5/3}$.
\end{proof}

This rules out global $L^2$ contraction: the Widom-phase prefactor $c$
combines with $c^{5/3}$ to give
$\norm{D\Tc}_{L^2 \to L^2} \gtrsim c \cdot c^{5/3} = c^{8/3}$.

\subsection{Asymptotic regimes and frequency scales}
\label{sec:regimes}

\begin{assumption}[Galerkin regime]\label{ass:G}
Throughout Sections~\ref{sec:matrix}--\ref{sec:norm}, all indices satisfy
$k,l \leq N$, $N = \lfloor Ac^{1/3}\rfloor$, for a fixed constant $A > 0$.
\end{assumption}

Under Assumption~\ref{ass:G}, the interval $(-1,1)$ is partitioned into:
\begin{itemize}[nosep]
  \item \textbf{Bulk zone} $\mathcal{Z}_{\rm bulk}$: $|s| \leq 1 - c^{-2/3}$.
        WKB/Szeg\H{o} bounds apply here.
  \item \textbf{Transition zone} $\mathcal{Z}_{\rm tz}$:
        $1-c^{-2/3} < |s| < 1$; measure $\sim c^{-2/3}$.
\end{itemize}

\noindent\textbf{Two frequency scales.}
Two frequency scales appear throughout:
\begin{equation}\label{eq:freqdef}
  \omega_{kl}^{\rm tz} := c^{2/3}|k-l|
  \quad\text{(WKB phase frequency at the bulk--transition boundary;
  used as lower-bound surrogate in IBP)},
\end{equation}
\begin{equation}\label{eq:eigengap}
  \chi_k - \chi_l \sim 2c|k-l|
  \quad\text{(SL eigenvalue gap, Widom linear spacing, $k,l \leq N$)}.
\end{equation}
These differ by a factor $c^{1/3}$.

\begin{remark}[Asymptotic ordering of scales]\label{rem:ordering}
For $c > 1$ the following strict ordering holds:
\[
  c^{-1} \ll c^{-2/3} \ll c^{-1/2} \ll c^{-1/3} \ll c^{-1/6} \ll 1.
\]
In particular:
\begin{itemize}[nosep]
  \item $c^{-2/3} < c^{-1/2}$: the transition-zone integral bound is
        \emph{smaller} (better) than the bulk bound.
  \item $c^{-1/3} > c^{-1/2}$: the transition-zone measure $c^{-2/3}$
        times $c^{1/3}$ modes gives $c^{-1/3}$, which is \emph{larger}
        than $c^{-1/2}$.
\end{itemize}
Every estimate below respects this ordering and never appeals to
a false inequality between these exponents.
\end{remark}

\subsection{Expansion and matrix entries}

Write $h = \sum_{k=0}^{N-1} a_k \pswf{k}$. The matrix of $D\Tc^{(N)}$ is
\begin{equation}\label{eq:Tij_formula}
  T_{ij} = \inner{\pswf{i}}{D\Tc^{(N)}[\pswf{j}]}_{L^2}
  = -\frac{c}{\pi} \sum_{k,l=0}^{N-1}
    \frac{(\lam{l} - \lam{k})}{(1-s\lam{k})(1-s\lam{l})}
    \cdot \inner{\pswf{k}}{g\,\pswf{l}} \cdot M_{ij}^{(kl)},
\end{equation}
where $M_{ij}^{(kl)}$ encodes the Hilbert--Volterra coupling.
After integrating over $s \in (-1,1)$, the dominant contribution
(for each pair $k \neq l$) takes the form
\begin{equation}\label{eq:Tij_simple}
  T_{ij} \;\sim\; c \cdot \frac{|\lam{l}-\lam{k}|}{(1-\lam{k})(1-\lam{l})}
  \cdot I_{ij},
  \qquad I_{ij} := \int_{-1}^1 \arccos(s)\,\pswf{i}(s)\pswf{j}(s)\,ds.
\end{equation}
The two factors are controlled by Lemma~\ref{lem:prefactor}
and Lemma~\ref{lem:RL} respectively.

\subsection{Commutator structure: diagonal vanishing}

The numerator $(\lam{l} - \lam{k})$ vanishes for $k = l$. Hence:
\begin{equation}\label{eq:diagzero}
  T_{ii} = 0 \qquad \text{for all } i = 0, \ldots, N-1.
\end{equation}

\subsection{Airy approximation and Wronskian bound}
\label{sec:Airy}

\begin{lemma}[Airy approximation for PSWFs]\label{lem:Airy}
Let $k \leq N = \lfloor A c^{1/3} \rfloor$, and let $s_k$ be the turning point.
Then uniformly for $s \in \mathcal{Z}_{\rm tz}(k) = \{|s-s_k| \leq c^{-2/3+\varepsilon}\}$:
\begin{equation}\label{eq:AiryApprox}
  \pswf{k}(s) = c^{1/6}\,\frac{\pi^{1/2}}{|\Ai'(0)|^{1/2}}
  \bigl[ \Ai(c^{2/3}\zeta_k(s)) + O(c^{-1/3}) \bigr],
\end{equation}
so $|\pswf{k}(s)| \lesssim c^{1/6}$ in $\mathcal{Z}_{\rm tz}(k)$.
\end{lemma}

\begin{proof}[References]
Olver~\cite{Olver1974}; Osipov--Rokhlin--Xiao~\cite{OsipovRokhlinXiao2013},
Theorems~4.2--4.3.
\end{proof}

\begin{corollary}[Integral Wronskian bound]\label{cor:Wronskian}
For $k \neq l$, $k,l \leq N$, with weighted Wronskian
$W_k^l(s) := (1-s^2)[\pswf{k}'\pswf{l} - \pswf{k}\pswf{l}']$:
\begin{equation}\label{eq:WronskianIntegral}
  \int_{-1}^1 |W_k^l(s)|\,ds \lesssim \frac{c^{-1/6}}{|k-l|}.
\end{equation}
\end{corollary}

\begin{proof}
\textbf{Transition zone.}
$|\pswf{k}| \lesssim c^{1/6}$, $|\pswf{k}'| \lesssim c^{5/6}$,
$(1-s^2) \lesssim c^{-2/3}$, measure $\sim c^{-2/3}$:
$J_{\rm tz} \lesssim c^{1/3} \cdot c^{-2/3} = c^{-1/3}$.

\textbf{Bulk zone.}
WKB gives $|W_k^l| \lesssim c^{1/2}$. One IBP using
$\omega_{kl}^{\rm tz} = c^{2/3}|k-l|$ as surrogate frequency:
$J_{\rm bulk} \lesssim c^{1/2}/(c^{2/3}|k-l|) = c^{-1/6}/|k-l|$.

\textbf{Total:} Since $c^{-1/3} > c^{-1/6}/|k-l|$ only when $|k-l| > c^{1/6}$,
which does not occur for $|k-l| \leq N \sim c^{1/3} \ll c^{1/6}$
(this comparison requires care: for $c$ large enough and $|k-l| \geq 1$
the bulk term dominates):
$J_{\rm tz} + J_{\rm bulk} \lesssim c^{-1/6}/|k-l|$.
\end{proof}

\subsection{Sturm--Liouville moment bound}
\label{sec:SLmoment}

\begin{proposition}[SL moment and primitive-function bounds]
\label{prop:moment}
For $k \neq l$, $k,l \leq N$:
\begin{enumerate}[label=(\alph*)]
  \item \textbf{Parity.} If $k \equiv l \pmod{2}$ then $\mu_{kl} = 0$.
  \item \textbf{Moment bound.}
        $|\mu_{kl}| := \bigl|\int_{-1}^1 s\,\pswf{k}\pswf{l}\,ds\bigr|
        \lesssim c^{-7/6}/|k-l|^2$.
  \item \textbf{Wronskian identity.}
        For the primitive $F_{kl}(s) := \int_s^1 \pswf{k}\pswf{l}\,dt$
        in the bulk zone:
        \begin{equation}\label{eq:FfromW}
          F_{kl}(s) = \frac{W_k^l(s)}{\chi_k - \chi_l}.
        \end{equation}
        The boundary condition $W_k^l(1) = 0$ holds because
        $(1-s^2)\big|_{s=1} = 0$, regardless of the amplitudes of
        $\pswf{k}$ and $\pswf{l}$.
        Consequently $|F_{kl}(s)| \lesssim c^{-1/2}(1-s^2)^{1/2}/|k-l|$
        in $\mathcal{Z}_{\rm bulk}$.
\end{enumerate}
\end{proposition}

\begin{proof}
\textbf{(a)} Parity of $(-1)^k$ gives odd integrand; integral vanishes.

\medskip
\textbf{(b)} Each $\pswf{k}$ satisfies the SL equation
\begin{equation}\label{eq:SLeq}
  -[(1-s^2)\pswf{k}']' + c^2 s^2 \pswf{k} = \chi_k \pswf{k},
  \qquad \chi_k \sim 2ck \text{ for } k \leq N.
\end{equation}
The SL identity gives $(\chi_k - \chi_l)\mu_{kl} = \int_{-1}^1 W_k^l(s)\,ds$,
hence
$|\mu_{kl}| \lesssim (c^{-1/6}/|k-l|)/(2c|k-l|) = c^{-7/6}/(2|k-l|^2)$.

\medskip
\textbf{(c)} From the SL equation:
$(\chi_k-\chi_l)\pswf{k}\pswf{l} = -(d/ds)W_k^l(s)$.
Integrating from $s$ to $1$ and using $W_k^l(1)=0$ gives~\eqref{eq:FfromW}.
The bound $|F_{kl}(s)| \lesssim c^{-1/2}(1-s^2)^{1/2}/|k-l|$ follows
from the WKB estimate $|W_k^l(s)| \lesssim c^{1/2}(1-s^2)^{1/2}$ in
$\mathcal{Z}_{\rm bulk}$ divided by $|\chi_k - \chi_l| \asymp 2c|k-l|$.
\end{proof}

\subsection{Off-diagonal integral bound}

\begin{lemma}[Zone decomposition of $I_{kl}$]\label{lem:sublD}
For $k \neq l$, $k,l \leq N$, let
$I_{kl} := \int_{-1}^1 \arccos(s)\,\pswf{k}(s)\pswf{l}(s)\,ds$.
Then:
\begin{align}
  |I_{kl}^{\rm tz}| &\lesssim \frac{c^{-2/3}}{|k-l|},
  \label{eq:IklTZ}\\
  |I_{kl}^{\rm bulk}| &\lesssim \frac{c^{-1/2}}{|k-l|},
  \label{eq:IklBulk}\\
  |I_{kl}| = |I_{kl}^{\rm tz}| + |I_{kl}^{\rm bulk}|
  &\lesssim \frac{c^{-1/2}}{|k-l|}.
  \label{eq:Ikltotal}
\end{align}
The bound~\eqref{eq:Ikltotal} is limited by the bulk term:
$c^{-2/3} < c^{-1/2}$ for $c > 1$.
\end{lemma}

\begin{proof}
\textbf{IBP setup.}
Since $(\arccos)'(s) = -(1-s^2)^{-1/2}$, integration by parts gives:
\[
  I_{kl}
  = \bigl[\arccos(s)\cdot(-F_{kl}(s))\bigr]_{-1}^{1}
    + \int_{-1}^1 \frac{F_{kl}(s)}{\sqrt{1-s^2}}\,ds.
\]
The boundary term vanishes: $F_{kl}(1) = 0$ (empty integral) and
$F_{kl}(-1) = \langle\pswf{k},\pswf{l}\rangle_{L^2} = 0$ by orthogonality.
Hence $I_{kl} = \int_{-1}^1 F_{kl}(s)(1-s^2)^{-1/2}\,ds$.

\medskip
\noindent\textbf{Step 3: Transition zone.}
In $\mathcal{Z}_{\rm tz}$: $(1-s^2) \lesssim c^{-2/3}$,
$(1-s^2)^{-1/2} \lesssim c^{1/3}$, and $|\pswf{k}\pswf{l}| \lesssim c^{1/3}$
(Lemma~\ref{lem:Airy}).
The primitive satisfies $|F_{kl}(s)| \leq \int_s^1|\pswf{k}\pswf{l}|\,dt
\lesssim c^{1/3}\cdot c^{-2/3} = c^{-1/3}$.
A further IBP using the Airy oscillation frequency
$\omega_{kl}^{\rm tz} = c^{2/3}|k-l|$ gives
$|F_{kl}(s)| \lesssim c^{-1/3}/|k-l|$. Therefore:
\[
  |I_{kl}^{\rm tz}|
  \lesssim \frac{c^{-1/3}}{|k-l|} \cdot c^{1/3} \cdot c^{-2/3}
  = \frac{c^{-2/3}}{|k-l|}.
\]

\medskip
\noindent\textbf{Step 4: Bulk zone.}
Use Proposition~\ref{prop:moment}(c):
$|F_{kl}(s)| \lesssim c^{-1/2}(1-s^2)^{1/2}/|k-l|$.
Hence:
\[
  |I_{kl}^{\rm bulk}|
  \leq \int_{\mathcal{Z}_{\rm bulk}}
       \frac{c^{-1/2}(1-s^2)^{1/2}}{|k-l|\sqrt{1-s^2}}\,ds
  = \frac{c^{-1/2}}{|k-l|}\int_{\mathcal{Z}_{\rm bulk}} ds
  \leq \frac{2c^{-1/2}}{|k-l|}.
\]

\medskip
\noindent\textbf{Conclusion.}
$|I_{kl}| \lesssim c^{-2/3}/|k-l| + c^{-1/2}/|k-l| \lesssim c^{-1/2}/|k-l|$.
\end{proof}

\begin{lemma}[PSWF cross-product integral]\label{lem:RL}
For $k \neq l$, $k, l \leq N \sim c^{1/3}$:
$|\int_{-1}^1 \arccos(s)\,\pswf{k}(s)\pswf{l}(s)\,ds| \leq C_1 c^{-1/2}/|k-l|$.
\end{lemma}

\begin{proof} Apply Lemma~\ref{lem:sublD}, bound~\eqref{eq:Ikltotal}. \end{proof}

\begin{remark}[Transition zone: pointwise vs.\ discrete Schur sum]
\label{rem:tz_irrelevant}
The transition zone bound $|I_{kl}^{\rm tz}| \lesssim c^{-2/3}/|k-l|$
is strictly better pointwise than the bulk $c^{-1/2}/|k-l|$.
However, the discrete Schur row sum $\sum_{j \neq i} |T_{ij}|$ runs over
indices $j$, not over $s$-values.
The $N_{\rm tz} \sim c^{1/3}$ transition indices contribute at most
$c^{1/3} \cdot c^{-2/3} = c^{-1/3}$ to the Schur sum, while the bulk
contributes $\sum c^{-1/2}/k \sim c^{-1/2}\log c$.
Since $c^{-1/3} > c^{-1/2}\log c$ for large $c$ is false
(we have $c^{-1/3} > c^{-1/2}$ but $c^{-1/3} < c^{-1/2}\log c$
once $\log c > c^{1/6}$, which holds for no finite $c$;
more precisely $c^{-1/3}$ and $c^{-1/2}\log c$ are comparable),
both zones contribute $O(\log c)$ and neither yields a sub-logarithmic bound.
\end{remark}

\subsection{Prefactor Reduction Lemma}

\begin{lemma}[Prefactor Reduction]\label{lem:prefactor}
For $k \neq l$ with $k, l \leq N = \lfloor Ac^{1/3}\rfloor$, let
$P_{kl} := c \cdot |\lam{l} - \lam{k}|/[(1-\lam{k})(1-\lam{l})]$.
Assumption~\ref{ass:B} postulates $P_{kl} \leq C_2 c^{1/2}$.
Raw Widom scaling gives $P_{kl} \lesssim |k-l|\,c^{7/3}$;
the claimed reduction $c^{7/3} \to c^{1/2}$ (by $c^{11/6}$ per unit $|k-l|$)
requires spectral structure beyond leading-order Widom
(Open Problem~\ref{op:prefactor}(1)).
\end{lemma}

\begin{assumption}[Prefactor bound, pointwise]\label{ass:B}
For all $k \neq l$ with $k, l \leq N \sim c^{1/3}$:
$c \cdot |\lam{l}-\lam{k}|/[(1-\lam{k})(1-\lam{l})] \leq C_2 c^{1/2}$.
\end{assumption}

\begin{assumption}[Schur-summability hypothesis]\label{ass:Bprime}
The Schur row sums satisfy a uniform $c^{-1/2}$-improvement:
\[
  \sup_{i \leq N}\; \sum_{j \neq i,\, j \leq N}
  P_{ij} \cdot |I_{ij}|
  \leq C\, c^{-1/2} \log c.
\]
This is \emph{not} a consequence of Assumption~\ref{ass:B}.
It requires a collective cancellation in the row sum beyond pointwise bounds;
see Section~\ref{sec:dyadic} for why this is the unique remaining obstacle.
\end{assumption}

\subsection{Matrix entry bound}

\begin{proposition}[Matrix entry bound]\label{prop:Tij}
Under Assumption~\ref{ass:B}, for $i \neq j$, $i, j \leq N$:
$|T_{ij}| \leq C_1 C_2/|i-j|$.
The factors $c^{1/2}$ cancel; the bound is $c$-independent.
\end{proposition}

\begin{proof}
$|T_{ij}| \lesssim P_{ij} \cdot |I_{ij}|
  \leq C_2 c^{1/2} \cdot C_1 c^{-1/2}/|i-j| = C_1 C_2/|i-j|$.
\end{proof}

\begin{remark}[$c$-power balance]\label{rem:balance}
$c^{1/2}_{\rm Ass.B} \times c^{-1/2}_{\rm Lem.RL} = c^0$.
Conditional on Assumption~\ref{ass:B}, the norm grows at most like $\log c$
instead of $c^{5/3}$.
\end{remark}

\subsection{Proof of Theorem~\ref{thm:main}}

\begin{proof}[Proof of Theorem~\ref{thm:main}]
By~\eqref{eq:diagzero}: $T_{ii} = 0$.

\noindent\textbf{Part~(a).}
Proposition~\ref{prop:Tij} gives $|T_{ij}| \leq C_1 C_2/|i-j|$.
Schur test:
$\sup_i \sum_{j \neq i} |T_{ij}| \leq C_1 C_2 H_N
 \leq C_1 C_2(1 + \tfrac13 \log c + \log A) =: C(A)\log c$.
Since $C(A)\log c \to \infty$, this is a norm bound only.

\noindent\textbf{Part~(b).}
Assumption~\ref{ass:Bprime} gives $\sup_i \sum_{j\neq i} P_{ij}|I_{ij}| \leq Cc^{-1/2}\log c$,
so $\|D\mathcal{T}_c^{(N)}\| \leq Cc^{-1/2}\log c \to 0$.
\end{proof}

Conditional on Assumption~\ref{ass:Bprime}: for $c \geq c_0$ there exists
a unique $g^{(N)} \in X_c^{(N)}$ with $\mathcal{T}_c^{(N)}(g^{(N)}) = g^{(N)}$,
and Picard iteration converges at rate $\leq C c^{-1/2}\log c$ per step.
\end{corollary}

\section{Dyadic Decomposition and No-go Theorem}
\label{sec:dyadic}
%% ============================================================

This section provides the dyadic structural analysis of the Schur row sum
and proves Theorem~\ref{thm:nogo}, establishing that the $\log c$ growth
in Theorem~\ref{thm:main}(a) is optimal within the class of matrices
satisfying $|T_{ij}| \leq \kappa/|i-j|$.
The analysis simultaneously characterises what Assumption~B$'$ must provide:
not a better pointwise bound, but a collective cancellation
across dyadically grouped index pairs.
Section~\ref{sec:spectral_weights} then proves that spectral weights
and naive index weights both fail structurally, and formulates the
correct weighted-norm problem.

\subsection{Dyadic block decomposition}

For a fixed row index $i$, partition the off-diagonal set
$\{j : 0 \leq j \leq N-1,\, j \neq i\}$ into dyadic blocks:
\begin{equation}\label{eq:dyadic_blocks}
  \mathcal{B}_m := \{ j : 2^m \leq |i-j| < 2^{m+1} \},
  \qquad m = 0, 1, \ldots, M,
  \quad M := \lfloor \log_2 N \rfloor.
\end{equation}
Each block $\mathcal{B}_m$ has at most $2^{m+1}$ elements
(restricted to $\{0,\ldots,N-1\}$).

\begin{proposition}[Block-wise Schur bound]\label{prop:dyadic_block}
Under Assumption~\ref{ass:B}, the contribution of each dyadic block
$\mathcal{B}_m$ to the Schur row sum satisfies:
\begin{equation}\label{eq:block_bound}
  \sum_{j \in \mathcal{B}_m} |T_{ij}|
  \leq \frac{C_1 C_2}{2^m} \cdot |\mathcal{B}_m|
  \leq \frac{C_1 C_2}{2^m} \cdot 2^{m+1}
  = 2 C_1 C_2.
\end{equation}
Every dyadically grouped block contributes $O(1)$, independently of $m$.
\end{proposition}

\begin{proof}
For $j \in \mathcal{B}_m$: $|i-j| \geq 2^m$, so $|T_{ij}| \leq C_1 C_2/2^m$.
Summing over $|\mathcal{B}_m| \leq 2^{m+1}$ terms gives~\eqref{eq:block_bound}.
\end{proof}

\begin{proposition}[Total Schur sum from dyadic decomposition]\label{prop:dyadic_total}
Under Assumption~\ref{ass:B}:
\begin{equation}\label{eq:schur_dyadic}
  \sum_{j \neq i} |T_{ij}|
  = \sum_{m=0}^{M} \sum_{j \in \mathcal{B}_m} |T_{ij}|
  \leq \sum_{m=0}^{M} 2 C_1 C_2
  = 2C_1 C_2 (M+1)
  \leq 2C_1 C_2 (\log_2 N + 1)
  \sim C(A) \log c.
\end{equation}
This reproduces exactly Theorem~\ref{thm:main}(a).
The dyadic decomposition provides no improvement over the direct harmonic bound:
each of the $\sim \log N$ blocks contributes $O(1)$, and their number grows
with $c$.
\end{proposition}

\begin{remark}[Why the dyadic bound is sharp]
The bound~\eqref{eq:block_bound} is tight: for the discrete Hilbert kernel
$T_{ij} = \kappa/|i-j|$, equality (up to constants) is attained in every
block, and the sum is exactly $\kappa H_{N-1}$.
No rearrangement or grouping of terms can reduce a sum $\sum_{j \neq i} \kappa/|i-j|$
below $\kappa H_{N-1}$ without exploiting cancellation between terms.
\end{remark}

\subsection{Proof of the No-go Theorem}

\begin{proof}[Proof of Theorem~\ref{thm:nogo}]
For any matrix with $T_{ii}=0$ and $|T_{ij}| \leq \kappa/|i-j|$:
\[
  \sup_i \sum_{j \neq i}|T_{ij}|
  \geq \kappa \sum_{d=1}^{N-1} \frac{1}{d}
  = \kappa H_{N-1}.
\]
This follows by taking $i$ to be any interior row index (so all distances
$d = 1, \ldots, N-1$ are realised) and using $|T_{ij}| \geq 0$.
For the operator norm lower bound:
$\norm{T}_{\ell^2 \to \ell^2} \geq (1/N)\sum_{i,j} |T_{ij}| \geq (\kappa/2) H_{N-1}$
(using the Frobenius--spectral norm comparison $\|T\|_{\ell^2} \geq \|T\|_F/\sqrt{N}$
and counting $\sim N$ rows each contributing $\geq \kappa H_{N-1}/2$).
For $N = \lfloor Ac^{1/3}\rfloor$: $H_{N-1} \geq \tfrac{1}{3}\log c + O(1)$.
The two parts (i) and (ii) of the conclusion follow:
\textit{(i)} is the content of Assumption~\ref{ass:Bprime};
\textit{(ii)} is Open Problem~\ref{op:weightnorm}.
\end{proof}

\subsection{What Assumption B$'$ must provide: a structural characterisation}

The dyadic analysis shows precisely what cancellation is needed.
Define the signed block sum:
\[
  S_m(i) := \sum_{j \in \mathcal{B}_m} T_{ij}.
\]
The No-go Theorem implies that for a contraction to hold, the \emph{signed}
sums $\{S_m(i)\}$ must satisfy:
\begin{equation}\label{eq:cancellation_needed}
  \sum_{m=0}^{M} |S_m(i)| \leq C c^{-1/2}\log c,
\end{equation}
rather than the trivially bounded $\sum_m |\mathcal{B}_m| \cdot \max_j |T_{ij}| = O(\log c)$.
In other words: Assumption~B$'$ is equivalent to requiring that within each
dyadic block, the $T_{ij}$ values exhibit \emph{oscillatory cancellation}
(alternating signs or decreasing envelope) such that
$|S_m(i)| \ll 2C_1C_2$ uniformly in $m$ and $c$.

\begin{remark}[Three routes to~\eqref{eq:cancellation_needed}]\label{rem:routes}
The structural characterisation~\eqref{eq:cancellation_needed} suggests
three distinct mechanisms:
\begin{enumerate}[nosep,label=(\roman*)]
  \item \textbf{Sign oscillation of $T_{ij}$ in $j$}:
        if $T_{ij}$ alternates sign with period $\sim 1$, then
        $|S_m(i)| = O(1/2^m)$ and $\sum_m |S_m(i)| = O(1)$.
        This would follow from a near-antisymmetry of $P_{ij}$
        under $j \mapsto j+1$, which is plausible from oscillatory PSWF
        products but not established.
  \item \textbf{Near-diagonal sparsification}:
        if $P_{ij} \leq C c^{-1/2}$ (a stronger form of Assumption~B
        with no $|i-j|$ dependence), then $|S_m(i)| \leq C c^{-1/2} \cdot 2$
        and $\sum_m |S_m(i)| \leq 2C c^{-1/2} \log c$.
        However, raw Widom scaling gives $P_{ij} \sim c^{7/3}|i-j|$
        in the transition zone; closing the gap to $c^{-1/2}$
        requires a reduction by $c^{11/6}$ per unit $|i-j|$
        and is analytically very demanding (see Lemma~\ref{lem:prefactor}).
  \item \textbf{Weighted block suppression}:
        if blocks with large $m$ (far-diagonal) are downweighted in a
        suitable $\ell^2$ norm, the effective number of contributing blocks
        shrinks. This is the weighted-norm approach
        (Section~\ref{sec:spectral_weights} and Open Problem~\ref{op:weightnorm}).
\end{enumerate}
Routes~(i) and~(ii) are open and analytically hard.
Route~(iii) is structurally clean but, as shown in the next section,
the naive weighted-norm candidates all fail.
\end{remark}

\subsection{Why spectral weights fail and what replaces them}
\label{sec:spectral_weights}

A natural first attempt at a weighted norm is to set
\begin{equation}\label{eq:spectral_weight}
  w_k := (1 - \lambda_k)^\alpha, \qquad \alpha > 0.
\end{equation}
The hope is that $w_j/w_i \sim |i-j|^\alpha$, converting the harmonic
sum $\sum 1/|i-j|$ into $\sum |i-j|^{\alpha-1}$, which is summable for $\alpha > 1$.

\begin{proposition}[Spectral weights are asymptotically constant in the Galerkin window]
\label{prop:spectral_weight_fail}
In the Galerkin window $k \leq N \sim c^{1/3}$, the PSWF eigenvalues satisfy
\[
  1 - \lambda_k \sim c^{-2/3}
  \qquad \text{uniformly in } k \leq N
\]
(transition regime; see Lemma~\ref{lem:HS} and Widom asymptotics).
Consequently, for any $\alpha > 0$:
\[
  w_k = (1-\lambda_k)^\alpha \sim c^{-2\alpha/3}
  \qquad\text{uniformly in } k \leq N,
\]
and therefore
\[
  \frac{w_j}{w_i} = 1 + O(c^{-1/3})
  \qquad \text{for all } i,j \leq N.
\]
The spectral weight~\eqref{eq:spectral_weight} does not modify the Schur kernel.
\textbf{Eigenvalue-based weighting cannot produce contraction.}
\end{proposition}

\begin{proof}
The Galerkin window is defined by $k \leq N \sim c^{1/3}$, which places all
indices inside the transition regime $|k - 2c/\pi| \leq Ac^{1/3}$.
Within this regime, $1 - \lambda_k \asymp c^{-2/3}$ uniformly
(Widom transition asymptotics).
Hence $w_k = c^{-2\alpha/3}(1 + O(c^{-1/3}))$ uniformly,
giving $w_j/w_i = 1 + O(c^{-1/3}) \to 1$.
\end{proof}

\begin{remark}[Structural reason for the failure]
\label{rem:spectral_weight_structural}
The weight $w_k = (1-\lambda_k)^\alpha$ ``sees'' the eigenvalue,
not the index $k$.
In the transition zone all eigenvalues are equally close to~$1$
(flat plateau of the PSWF spectrum), so the weight is blind to
the index structure that drives the Schur sum.
The Schur kernel $1/|i-j|$ is an \emph{index-space} object;
an effective weight must distinguish indices, not eigenvalues.
This is not a technical limitation but a structural mismatch:
the operator lives on a different scale than the spectrum.
\end{remark}

\begin{remark}[Index-based weights also fail]
\label{rem:index_weights}
A weight depending directly on $k$ (rather than $\lambda_k$) must resolve
the index structure. Two natural candidates both fail:
\begin{enumerate}[nosep, label=(\alph*)]
  \item \textbf{Power weight} $w_k = (1+k)^\beta$:
        Then $w_j/w_i \sim (j/i)^\beta$ for $j \gg i$,
        giving weighted kernel $\sim (1+d/i)^\beta/d$ with $d = |j-i|$.
        For the central row $i \sim N/2$, the weighted row sum is
        \[
          \sum_{d=1}^{N} \frac{(1 + d/i)^\beta}{d}
          \;\sim\; \sum_{d=1}^{i} \frac{1}{d} + \sum_{d=i}^{N} \frac{(d/i)^\beta}{d}
          \;\sim\; \log i + i^{-\beta}\sum_{d=i}^{N} d^{\beta-1}
          \;\sim\; \log N.
        \]
        For $\beta < 1$: $\sum d^{\beta-1}$ converges, but the $\log i$ term
        survives. For $\beta \geq 1$: the far sum diverges faster.
        \textbf{Net: no improvement.}
  \item \textbf{Exponential weight} $w_k = e^{\gamma k/N}$, $\gamma > 0$:
        Then $w_j/w_i = e^{\gamma(j-i)/N}$, giving weighted kernel
        $e^{\gamma d/N}/d$.
        The row sum is $\sum_{d=1}^{N} e^{\gamma d/N}/d \sim e^\gamma \log N$.
        \textbf{Net: larger constant, same logarithmic growth.}
\end{enumerate}
Both candidates fail because the kernel $1/d$ is not integrable,
and a multiplicative weight $\phi(d)/d$ is summable only if $\phi(d) \to 0$,
i.e., \emph{far-diagonal indices must be downweighted.}
But downweighting $j \gg i$ while upweighting $j \ll i$ (or vice versa)
breaks symmetry: the Schur condition $\sup_i \sum_j (w_j/w_i)|T_{ij}|$
must hold simultaneously for all rows $i$.
A weight that helps row $i$ by downweighting far $j$ will hurt row $j$
by upweighting far $i$.
\end{remark}

The conclusion of this analysis is:

\begin{corollary}[No naive weight works]\label{cor:no_naive_weight}
No weight of the form $w_k = f(\lambda_k)$ (spectral weight),
$w_k = k^\beta$ (power weight), or $w_k = e^{\gamma k/N}$ (exponential weight)
can make the weighted Schur row sum $\sup_i \sum_j (w_j/w_i)/|i-j|$
asymptotically smaller than $\log N$.
\end{corollary}

\begin{openproblem}[Correct formulation of the weighted-norm problem]
\label{op:weightnorm_precise}
Find a positive sequence $\{w_k\}_{k=0}^{N-1}$ depending on $k$
(not on $\lambda_k$) such that
\[
  \sup_{i \leq N} \sum_{j \neq i} \frac{w_j}{w_i} \cdot \frac{C_1 C_2}{|i-j|}
  < 1.
\]
By Corollary~\ref{cor:no_naive_weight}, such a weight cannot be
of power, exponential, or spectral type.
A non-symmetric (two-weight) Schur estimate may be necessary:
find positive sequences $\{u_k\}$, $\{v_k\}$ such that
\[
  \sum_j v_j |T_{ij}| \leq u_i
  \quad\text{and}\quad
  \sum_j u_j^{-1} |T_{ij}| \leq v_i^{-1}
  \quad\text{for all } i,
\]
which together imply $\|T\|_{\ell^2(u) \to \ell^2(v)} < 1$.
This is the two-weight Schur lemma (Schur--Cotlar--Stein framework).
\end{openproblem}

\section{Open Problems}
%% ============================================================

\begin{openproblem}[Prefactor Reduction and Schur improvement]\label{op:prefactor}
Two separate problems:
\begin{enumerate}[nosep]
  \item Prove Assumption~\ref{ass:B} unconditionally:
        $P_{kl} \leq C_2 c^{1/2}$ for $k,l \leq N \sim c^{1/3}$.
        This requires computing eigenvalue ratios $\chi_k/\lam{k}$ uniformly
        in the transition window, beyond leading-order Widom.
  \item Prove Assumption~\ref{ass:Bprime} unconditionally.
        By the characterisation of Section~\ref{sec:dyadic},
        this is equivalent to establishing signed cancellation in dyadic
        row-sum blocks (Remark~\ref{rem:routes}).
        Route~(i) (sign oscillation via PSWF phase analysis) is the
        most structurally natural approach.
\end{enumerate}
\end{openproblem}

\begin{openproblem}[Weighted norm approach]\label{op:weightnorm}
Find a $c$-dependent weighted norm $\|\cdot\|_w$ on $X_c^{(N)}$
under which $\|D\mathcal{T}_c^{(N)}\|_w < 1$ unconditionally.
By Proposition~\ref{prop:spectral_weight_fail} and
Corollary~\ref{cor:no_naive_weight}, spectral weights, power weights,
and exponential index weights all fail.
The correct weight, if it exists, must be asymmetric or of two-weight type
(Open Problem~\ref{op:weightnorm_precise}).
This route avoids Assumption~B$'$ entirely but requires
new structural input beyond standard Schur theory.
\end{openproblem}

\begin{openproblem}[Bulk oscillation cancellation]\label{op:bulk}
Prove $|I_{kl}^{\rm bulk}| \lesssim c^{-2/3}/|k-l|$
via stationary-phase or second-IBP exploiting WKB phases $\theta_k - \theta_l$.
If established: part~(a) upgrades to $\|DT\| \lesssim c^{-1/6}\log c$;
part~(b) to $\|DT\| \lesssim c^{-2/3}\log c$.
\end{openproblem}

\begin{openproblem}[Optimal Galerkin dimension]\label{op:Nopt}
Determine the maximal $N(c)$ for which the Schur bound is below 1.
Conjecture: $N_{\rm opt} \sim c^{1/2}$.
\end{openproblem}

\begin{openproblem}[Self-adjointness and spectral norm]\label{op:Gershgorin}
Establish approximate self-adjointness $T_{ij} \approx \overline{T_{ji}}$
up to $O(c^{-1})$, to sharpen the Schur bound to a spectral radius bound.
\end{openproblem}

\section{Relation to Assumption A}
%% ============================================================

The current status relative to Assumption~A:

\begin{itemize}[nosep]
  \item \textbf{Unconditionally proved:}
        $|I_{kl}| \leq C_1 c^{-1/2}/|k-l|$ (Lemma~\ref{lem:RL});
        $T_{ii} = 0$; $\|DT\|_{L^2} \gtrsim c^{8/3}$ globally;
        $\log c$ growth is optimal in the pointwise-bound class
        (Theorem~\ref{thm:nogo});
        spectral and naive index weights do not improve the Schur kernel
        (Proposition~\ref{prop:spectral_weight_fail},
        Corollary~\ref{cor:no_naive_weight}).

  \item \textbf{Under Assumption~B:}
        $|T_{ij}| \leq C_1 C_2/|i-j|$;
        $\|D\mathcal{T}_c^{(N)}\| \leq C\log c$ (norm estimate only).

  \item \textbf{Under Assumptions~B and~B$'$:}
        $\|D\mathcal{T}_c^{(N)}\| \leq Cc^{-1/2}\log c$;
        contraction; conditional Galerkin fixed point
        (Corollary~\ref{cor:fixedpoint}).

  \item \textbf{Gap and path forward:}
        Assumption~B$'$ is structurally necessary
        (Theorem~\ref{thm:nogo}), not merely technical.
        The dyadic characterisation (Section~\ref{sec:dyadic}) reduces it
        to proving signed block cancellation in $T_{ij}$,
        most naturally via PSWF phase analysis (route~(i) of
        Remark~\ref{rem:routes}).
        The weighted-norm route (Open Problem~\ref{op:weightnorm})
        would avoid Assumption~B$'$ entirely, but requires
        an asymmetric or two-weight Schur estimate
        beyond all naive candidates.
\end{itemize}
