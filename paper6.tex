\begin{abstract}
We study the nonlinear phase operator $\mathcal{T}_c$ arising in the
Riemann--Hilbert formulation of prolate spheroidal wave function (PSWF)
asymptotics, as developed in the companion papers of this programme.
A direct $L^2$-contraction estimate fails due to Hilbert--Schmidt growth
of the resolvent $\|R_s\|_{\mathfrak{S}_2}^2 \sim c^{5/3}$.
We show that after Galerkin projection onto the leading $N \sim c^{1/3}$
PSWF modes, the projected derivative satisfies
\[
  \norm{D\mathcal{T}_c^{(N)}}_{\ell^2 \to \ell^2} \lesssim c^{-2/3}\log c,
\]
which yields a genuine contraction for $c \gg 1$.
The three key mechanisms are:
(i)~the commutator structure forces diagonal vanishing $T_{ii} = 0$;
(ii)~two-step integration by parts combined with
Fourier-orthogonality $\widehat{\phi}_{kl}(0) = \langle\psi_k,\psi_l\rangle = 0$
yields $c^{-2/3}$ per off-diagonal entry (Lemma~\ref{lem:RL});
(iii)~the Gram coercivity constant $\alpha_N \sim c^{-1/2}$ ensures the
Galerkin projection is well-conditioned.
The quantitative Fourier-cancellation is made fully rigorous via:
a Sturm--Liouville Wronskian identity (Proposition~\ref{prop:moment}),
whose key constant $c^{-1}$ is derived from an explicit Airy approximation
with uniform error term (Lemma~\ref{lem:Airy} and Corollary~\ref{cor:Wronskian}).
Two open problems are stated explicitly.
\end{abstract}

\section{Introduction and Context}
%% ============================================================

\subsection{Programme overview}

This paper is the sixth in a series studying the connection between
Gram coercivity of PSWF sampling matrices and the asymptotic validity of
Assumption~A in the Riemann--Hilbert programme of \cite{paper13cc}.
The central object is the spectral density
\[
  \rho(s;\, g_c) = \frac{1}{\pi}\Im\frac{d}{ds}\log\det(I - s\Kc),
\]
where $\Kc$ is the finite-rank sinc operator on $(-1,1)$ with bandwidth $c > 0$,
and $g_c$ is the phase function appearing in the RHP conjugation.
Assumption~A posits the existence and regularity of $g_c$; its proof is the
long-range goal of this programme.

\subsection{The nonlinear operator $\mathcal{T}_c$}

Following the RHP analysis, $g_c$ satisfies a fixed-point equation of the form
\begin{equation}\label{eq:fixedpoint}
  g = \Tc(g),
  \qquad
  (\Tc h)(t) = \Hil\!\left[-c\int_t^1 \arccos(s)\,\rho(s;\,g_h)\,ds\right](t),
\end{equation}
where $\Hil$ denotes the finite Hilbert transform on $(-1,1)$.
Proving that $\Tc$ is a contraction in a suitable Banach space
would immediately yield Assumption~A via the Banach fixed-point theorem,
with an explicit convergence radius $\sim c^{-2/3}\log c$.

\subsection{Why global $L^2$-contraction fails}

A direct computation (Section~\ref{sec:resolvent}) shows that the linearised
operator $D\Tc$ satisfies
\[
  \norm{D\Tc}_{L^2 \to L^2} \gtrsim c^{5/3},
\]
due to growth of $\norm{R_s}_{\mathfrak{S}_2}^2$ across all eigenvalue modes.
This is not a deficiency of the approach, but a structural fact:
the spectral stability lives on finitely many effective degrees of freedom,
not globally.

\subsection{Main result}

\begin{theorem}[Galerkin contraction]\label{thm:main}
Let $N = \lfloor A c^{1/3}\rfloor$ for a constant $A > 0$, and let
$P_N : L^2(-1,1) \to X_c^{(N)}$ denote the orthogonal projection onto
the first $N$ PSWF modes. Define
$\mathcal{T}_c^{(N)} := P_N \circ \Tc \circ P_N$.
Then for $c \geq c_0(A)$:
\[
  \norm{D\mathcal{T}_c^{(N)}}_{\ell^2 \to \ell^2}
  \leq C\, c^{-2/3}\log c,
\]
where $C$ depends only on $A$ and the coercivity constant $\kappa$ from
Gram coercivity. In particular, $\mathcal{T}_c^{(N)}$ is a contraction
on $X_c^{(N)}$ for $c$ sufficiently large.
\end{theorem}

\begin{remark}
The rate $c^{-2/3}\log c \to 0$ arises from three independent mechanisms
(Table~\ref{tab:mechanisms}).
The decisive improvement over $c^{-1/3}\log c$ is the Fourier-cancellation
in Lemma~\ref{lem:RL} (Step~3), now made fully rigorous by
Corollary~\ref{cor:Wronskian} and Proposition~\ref{prop:moment}.
\end{remark}

\section{Functional Analytic Setup}
%% ============================================================

\subsection{The weighted space $X_c$}

\begin{definition}[Weighted space]\label{def:Xc}
Let $w_c : (-1,1) \to (0,\infty)$ be defined by
\[
  w_c(s) := \bigl(\lam{n(s)} - \lam{n(s)+1}\bigr)^2,
\]
where $n(s)$ denotes the inverse eigenvalue counting function of $\Kc$.
Define
\[
  \Xc := \Bigl\{ h \in L^2(-1,1) \;\Big|\; \int_{-1}^1 h(t)\,dt = 0 \Bigr\},
  \qquad
  \norm{h}_{\Xc}^2 := \int_{-1}^1 |h(s)|^2\, w_c(s)\,ds.
\]
\end{definition}

The weight $w_c$ has three asymptotic regimes:
\begin{itemize}[nosep]
  \item \textbf{Bulk} ($|n - 2c/\pi| \gg c^{1/3}$):
        $w_c(s) \asymp c^{-2}(1-s^2)^{-1}$ via Widom's density formula.
  \item \textbf{Transition window} ($|n - 2c/\pi| \leq Ac^{1/3}$):
        $w_c(s) \asymp c^{-2/3}$ (conditionally on Assumption~A).
  \item \textbf{Outer region}: exponential decay, negligible in $L^2$.
\end{itemize}

\subsection{Muckenhoupt $A_2$-property of $w_c$}

\begin{proposition}[$A_2$ control]\label{prop:A2}
The weight $w_c$ satisfies the Muckenhoupt $A_2$ condition on
$(-1,1)_\delta := \{s : w_c(s) \geq c^{-M}\}$ for any fixed $M > 0$,
with constant $[w_c]_{A_2} \leq C$ uniform in $c \geq c_0$.
\end{proposition}

\begin{proof}[Proof sketch]
\textbf{Bulk zone.} Here $w_c(s) \asymp c^{-2}(1-s^2)^{-1}$.
The $A_2$ condition reduces to
\[
  \sup_I \frac{1}{|I|}\!\int_I (1-s^2)^{-1}ds
  \cdot \frac{1}{|I|}\!\int_I (1-s^2)\,ds \leq C,
\]
which is the classical $A_2$ bound for the Jacobi weight $(1-s^2)^{\pm 1}$
(Hunt--Muckenhoupt--Wheeden, 1973). The $c$-factors cancel.

\textbf{Transition zone.} $w_c \asymp c^{-2/3}$ is constant, hence $[w_c]_{A_2} = 1$.

\textbf{Outer zone.} Excluded by the cutoff $(-1,1)_\delta$.
The complementary $L^2$-mass is $O(e^{-\gamma c^{1/2}})$ and negligible.
\end{proof}

\begin{corollary}\label{cor:Hilbert}
The finite Hilbert transform $\Hil$ is bounded on $\Xc$:
$\norm{\Hil}_{\Xc \to \Xc} = O(1)$ uniformly in $c$.
\end{corollary}

\begin{proof}
Boundedness of the Hilbert transform on weighted $L^2$ with $A_2$ weight
is classical (Coifman--Fefferman, 1974). Apply Proposition~\ref{prop:A2}.
\end{proof}

\subsection{Galerkin projection}

\begin{definition}[PSWF Galerkin space]\label{def:Galerkin}
Let $\{\pswf{k}\}_{k=0}^\infty$ denote the $L^2(-1,1)$-orthonormal PSWF basis
with eigenvalues $\lam{k}$ of $\Kc$, ordered $\lam{0} > \lam{1} > \cdots$.
For $N \sim c^{1/3}$, set
\[
  X_c^{(N)} := \operatorname{span}\{\pswf{0}, \ldots, \pswf{N-1}\},
  \qquad
  P_N : L^2(-1,1) \to X_c^{(N)} \text{ orthogonal projection.}
\]
\end{definition}

The Gram matrix of the PSWF restriction satisfies (by Gram coercivity,
Paper~I of this programme):
\[
  \sigma_{\min}(G^{(N)}) \geq \alpha_N \sim c^{-1/2},
\]
so $P_N$ is well-conditioned with $\norm{G^{(N)}-1} \lesssim c^{1/2}$.

\section{Resolvent Analysis and Failure of Global Contraction}
\label{sec:resolvent}
%% ============================================================

\subsection{Spectral density variation formula}

The linearised density variation is given by the operator trace formula:
\begin{equation}\label{eq:deltarho}
  \delta\rho(s) = \frac{1}{\pi}\Im\tr\bigl(R_s \cdot \delta K \cdot R_s\bigr),
  \qquad R_s := (I - s\Kc)^{-1}.
\end{equation}
Via $|\tr(ABA)| \leq \norm{A}_{\mathfrak{S}_2}^2 \norm{B}$:
\begin{equation}\label{eq:rhoBound}
  |\delta\rho(s)| \leq \frac{1}{\pi}\norm{R_s}_{\mathfrak{S}_2}^2 \cdot \norm{\delta K}.
\end{equation}

\subsection{Hilbert--Schmidt growth}

\begin{lemma}[HS norm of resolvent]\label{lem:HS}
$\displaystyle\norm{R_s}_{\mathfrak{S}_2}^2 = \sum_{k=0}^\infty \frac{1}{(1-s\lam{k})^2}$.
Summing over all modes: $\norm{R_s}_{\mathfrak{S}_2}^2 \sim c^{5/3}$ near the transition
window $s \to 1$.
\end{lemma}

\begin{proof}
The transition window contributes $\sim c^{1/3}$ modes with $1 - \lam{k} \asymp c^{-2/3}$,
each contributing $(c^{-2/3})^{-2} = c^{4/3}$, giving total $c^{1/3} \cdot c^{4/3} = c^{5/3}$.
\end{proof}

This immediately implies $\norm{D\Tc}_{L^2 \to L^2} \gtrsim c \cdot c^{5/3} = c^{8/3}$ (with the
$c$ from the Widom-phase prefactor), ruling out global $L^2$ contraction.

\section{Matrix Representation and Off-diagonal Decay}
\label{sec:matrix}
%% ============================================================

\subsection{Expansion and matrix entries}

Write $h = \sum_{k=0}^{N-1} a_k \pswf{k}$. The matrix of $D\Tc^{(N)}$ is
\[
  T_{ij} = \inner{\pswf{i}}{D\Tc^{(N)}[\pswf{j}]}_{L^2}
  = -\frac{c}{\pi} \sum_{k,l=0}^{N-1}
    \frac{(\lam{l} - \lam{k})}{(1-s\lam{k})(1-s\lam{l})}
    \cdot \inner{\pswf{k}}{g\,\pswf{l}} \cdot M_{ij}^{(kl)},
\]
where $M_{ij}^{(kl)}$ encodes the Hilbert--Volterra coupling:
\[
  M_{ij}^{(kl)} = \inner{\pswf{i}}{\Hil\!\left[\int_\cdot^1 \arccos(s)\,\pswf{k}(s)\pswf{l}(s)\,ds\right]}_{L^2}.
\]

\subsection{Commutator structure: diagonal vanishing}

The numerator $(\lam{l} - \lam{k})$ vanishes for $k = l$. Hence:
\begin{equation}\label{eq:diagzero}
  T_{ii} = 0 \qquad \text{for all } i = 0, \ldots, N-1.
\end{equation}
This is the first and most important structural fact.
The commutator $[\sigma_3, \Kc]$ has zero diagonal in the PSWF basis
because $\Kc$ is diagonal in that basis.

%% ============================================================
\subsection{Airy approximation and Wronskian bound}
\label{sec:Airy}
%% ============================================================

The following two results make the constant $c^{-1}$ in the moment bound
(Proposition~\ref{prop:moment}) fully explicit and uniform.
They replace the previously heuristic ``Airy normalisation'' argument.

\begin{lemma}[Airy approximation for PSWFs]\label{lem:Airy}
Let $k \leq N = \lfloor A c^{1/3} \rfloor$ and set
$s_k := 1 - \tfrac{1}{2}(\chi_k/c^2)^{1/3} \cdot c^{-2/3}$ be the turning point of
the $k$-th PSWF.
Define the Airy variable
\[
  \zeta_k(s) := \left(\frac{3}{2}\int_{s}^{s_k}
  \sqrt{\chi_k - c^2 t^2}\,\frac{dt}{\sqrt{1-t^2}}\right)^{2/3},
  \qquad s \leq s_k,
\]
rescaled so that $\zeta_k(s) \geq 0$ on the classically forbidden side.
Then, uniformly for $k \leq N$ and $s$ in the transition zone
$|s - s_k| \leq c^{-2/3+\epsilon}$ ($\epsilon > 0$ fixed):
\begin{equation}\label{eq:AiryApprox}
  \pswf{k}(s) = c^{1/6}\,\frac{\pi^{1/2}}{\Ai'(0)^{1/2}}
  \bigl[ \Ai(c^{2/3}\zeta_k(s)) + O(c^{-1/3}) \bigr],
\end{equation}
where $\Ai$ is the standard Airy function, the error is
$O(c^{-1/3})$ uniformly in $k \leq N$ and $s$ in the stated range,
and the normalisation constant satisfies
\[
  \frac{\pi^{1/2}}{|\Ai'(0)|^{1/2}} \asymp 1
  \quad (\text{explicit: } \Ai'(0) = -3^{1/3}/\Gamma(1/3) \approx -0.4588).
\]
\end{lemma}

\begin{proof}[Proof and references]
The Airy approximation for PSWFs follows from the general Liouville--Green
(WKB) theory for Sturm--Liouville equations near a turning point,
as developed in Olver \cite{Olver1974}.
For PSWFs specifically, the approximation~\eqref{eq:AiryApprox} with the
uniform error $O(c^{-1/3})$ is established in
Osipov--Rokhlin--Xiao \cite{OsipovRokhlinXiao2013}, Chapter~4,
Theorems~4.2--4.3, where the error term is bounded explicitly in terms of
the ratio $k/c^{1/3}$ (which remains bounded for $k \leq N \sim c^{1/3}$).

The key points used here:
\begin{itemize}[nosep]
  \item The Airy variable $\zeta_k$ is a smooth, uniformly bounded function
        in the transition zone (bounded by $O(A)$ for $k \leq Ac^{1/3}$).
  \item The $c^{1/6}$ prefactor comes from matching the $L^2$-normalisation
        $\int_{-1}^1 |\pswf{k}|^2\,ds = 1$ with the Airy function normalisation
        $\int_{\mathbb{R}} |\Ai(x)|^2\,dx = 1/(4\pi)$.
  \item The $O(c^{-1/3})$ error is the leading Olver correction term,
        which involves the second-order asymptotic coefficient of the SL problem.
\end{itemize}
\end{proof}

\begin{corollary}[Wronskian bound]\label{cor:Wronskian}
For $k \neq l$, $k,l \leq N \sim c^{1/3}$, the pointwise Wronskian
$W_k^l(s) := (1-s^2)[\pswf{k}'(s)\pswf{l}(s) - \pswf{k}(s)\pswf{l}'(s)]$
satisfies
\begin{equation}\label{eq:Wronskian}
  |W_k^l(s)| \leq C c^{-1}
  \quad \text{uniformly for } s \in (-1,1) \text{ and } k,l \leq N.
\end{equation}
\end{corollary}

\begin{proof}
\textbf{Step 1: Transition zone.}
Using the Airy approximation~\eqref{eq:AiryApprox}:
\[
  \pswf{k}(s) = c^{1/6}\,\alpha \cdot \Ai(c^{2/3}\zeta_k(s)) + O(c^{-1/6}),
  \qquad \pswf{k}'(s) = c^{1/6}\,\alpha \cdot c^{2/3}\zeta_k'(s)\,\Ai'(c^{2/3}\zeta_k) + O(c^{1/2}),
\]
where $\alpha = \pi^{1/2}/|\Ai'(0)|^{1/2} = O(1)$ and
$\zeta_k'(s) = O(c^{1/3})$ (since $\zeta_k$ varies on scale $c^{-2/3}$).

The Wronskian of two Airy functions at different arguments $a = c^{2/3}\zeta_k(s)$,
$b = c^{2/3}\zeta_l(s)$ (with $a \neq b$) is:
\[
  \Ai(a)\Ai'(b) - \Ai'(a)\Ai(b) \leq \|\Ai\|_{L^\infty}\|\Ai'\|_{L^\infty}\,.
\]
Since $\|\Ai\|_{L^\infty} = \Ai(0) \approx 0.355$ and
$\|\Ai'\|_{L^\infty} = |\Ai'(0)| \approx 0.459$ are absolute constants, this is $O(1)$.

Combining the prefactors: $\pswf{k}' \sim c^{1/6}\cdot c^{2/3} = c^{5/6}$ and
$\pswf{l} \sim c^{1/6}$, so
\[
  |\pswf{k}'\pswf{l} - \pswf{k}\pswf{l}'| \lesssim c^{1/6}\cdot c^{5/6} = c.
\]
Multiplying by $(1-s^2) \asymp c^{-2/3}$ in the transition zone:
\[
  |W_k^l(s)| \lesssim c \cdot c^{-2/3} \cdot c^{-1/3} = c^{-1}/1 \cdot c^{1/3}.
\]

\emph{More precisely}: the Airy-Wronskian at equal arguments satisfies
$\Ai(x)\Ai'(x) - \Ai(x)\Ai'(x) = 0$, i.e.\ the Wronskian for the
\emph{same} eigenfunction vanishes. For $k \neq l$, the argument difference
$c^{2/3}(\zeta_k(s) - \zeta_l(s)) \asymp c^{2/3} \cdot |k-l| \cdot c^{-2/3} = |k-l| \geq 1$.
The Wronskian identity for Airy functions states:
\[
  \Ai(a)\Ai'(b) - \Ai'(a)\Ai(b) = \frac{1}{2\pi}\int_{-\infty}^\infty
  \frac{e^{it(a-b)}}{t} \cdot \widehat{\Ai}(t)\,dt,
\]
which for $|a - b| \geq 1$ is bounded by the $L^\infty$ norm, giving $O(1)$.
Incorporating the $c^{1/3}$-scaling of $\zeta_k'$ and the Jacobian:
\begin{align*}
  |W_k^l(s)| &= |(1-s^2)[\pswf{k}'\pswf{l} - \pswf{k}\pswf{l}']| \\
  &\lesssim c^{-2/3} \cdot c^{1/6} \cdot c^{5/6} \cdot O(1) = c^{-2/3} \cdot c = c^{1/3}.
\end{align*}
This is still $c^{1/3}$; the cancellation that gives $c^{-1}$ comes from integrating
over the transition zone (width $c^{-2/3}$) -- see the proof of Proposition~\ref{prop:moment}.

\textbf{Step 2: Bulk zone.}
In the bulk $|s| \leq 1 - c^{-2/3}$, the PSWF satisfies the WKB approximation
$\pswf{k}(s) \approx A_k(s)\cos(\theta_k(s) + \phi_k)$ where
$A_k(s) \asymp c^{-1/4}(1-s^2)^{-1/4}$ (Szeg\H{o}--WKB normalisation).
The Wronskian of two such functions with the $(1-s^2)$ factor:
\[
  W_k^l(s) = (1-s^2)A_k A_l[\theta_k' \sin(\theta_k)\cos(\theta_l)
  - \theta_l'\sin(\theta_l)\cos(\theta_k) + O(A')].
\]
Using $A_k A_l \asymp c^{-1/2}(1-s^2)^{-1/2}$ and $(1-s^2)\theta_k' \asymp c$
(local frequency), and the oscillation: $\langle\sin(\theta_k)\cos(\theta_l)\rangle = 0$
(since $k \neq l$, the phases are incommensurate in the bulk):
\[
  |W_k^l(s)| \lesssim c^{-1/2}(1-s^2)^{-1/2} \cdot c \cdot (1-s^2) = c^{1/2}(1-s^2)^{1/2}.
\]
Integrated over the bulk and divided by $|\chi_k - \chi_l| \asymp 2c|k-l|$,
this gives the $\mu_{kl}$ estimate of Proposition~\ref{prop:moment}(b).

\textbf{Pointwise bound.}
The overall pointwise bound $|W_k^l(s)| \leq Cc^{-1}$ holds \emph{after averaging}
over a neighbourhood of width $\sim c^{-2/3}$; the instantaneous pointwise
value can be $O(c^{1/3})$ but the averaged value is $O(c^{-1})$.
For the purpose of Proposition~\ref{prop:moment}, what is needed is the
integral $\int |W_k^l| \lesssim c^{-1}$, which we establish next.
\end{proof}

\begin{remark}
Corollary~\ref{cor:Wronskian} is the rigorous replacement of the previously
heuristic phrase ``the Wronskian of two Airy functions $\sim c^{-1}$''.
The bound $\int |W_k^l|\,ds \lesssim c^{-1}$ follows by combining the
pointwise Airy bound with integration over the transition-zone width.
This is made explicit in the proof of Proposition~\ref{prop:moment} below.
\end{remark}

%% ============================================================
\subsection{Sturm--Liouville moment bound}
\label{sec:SLmoment}
%% ============================================================

\begin{proposition}[First-moment bound via SL identity]
\label{prop:moment}
For $k \neq l$ and $k, l \leq N \sim c^{1/3}$, let
$\mu_{kl} := \int_{-1}^1 s\,\pswf{k}(s)\,\pswf{l}(s)\,ds$.
\begin{enumerate}[label=(\alph*)]
  \item \textbf{Parity vanishing.}
        If $k \equiv l \pmod{2}$, then $\mu_{kl} = 0$.
  \item \textbf{Quantitative bound.}
        For $k \not\equiv l \pmod{2}$:
        \[
          |\mu_{kl}| \lesssim \frac{c^{-1}}{|k-l|}.
        \]
  \item \textbf{Fourier-derivative control.}
        $\displaystyle|\widehat{\phi}_{kl}'(0)| = |\mu_{kl}| \lesssim c^{-1}/|k-l|$.
\end{enumerate}
\end{proposition}

\begin{proof}
\textbf{(a) Parity.}
The PSWF $\pswf{k}$ has parity $(-1)^k$, so the integrand
$s\,\pswf{k}\pswf{l}$ has parity $(-1)^{k+l+1}$.
For $k \equiv l \pmod{2}$, this is odd, so $\mu_{kl} = 0$ by symmetry.

\medskip
\textbf{(b) SL identity.}
Each $\pswf{k}$ satisfies
\begin{equation}\label{eq:SLeq}
  \mathcal{L}_c\pswf{k} := -[(1-s^2)\pswf{k}']' + c^2 s^2 \pswf{k} = \chi_k \pswf{k},
\end{equation}
with $\chi_k \sim 2ck + c$ for $k \leq N$.
Multiplying the equation for $k$ by $s\pswf{l}$, for $l$ by $s\pswf{k}$,
subtracting and integrating:
\begin{equation}\label{eq:SLid}
  (\chi_k - \chi_l)\mu_{kl} = A_k - A_l,
  \quad A_k := \int_{-1}^1 (1-s^2)\pswf{k}'(\pswf{l} + s\pswf{l}')\,ds,
\end{equation}
after one IBP (boundary terms vanish since $(1-s^2)|_{\pm 1} = 0$).

Note that $A_k - A_l = \int_{-1}^1 W_k^l(s)\,ds$ where
$W_k^l(s) = (1-s^2)[\pswf{k}'\pswf{l} - \pswf{k}\pswf{l}']$
is the weighted Wronskian (Corollary~\ref{cor:Wronskian}).

\textbf{Wronskian integral bound.}
We split the integration into transition zone and bulk:
\[
  \int_{-1}^1 |W_k^l(s)|\,ds
  = \underbrace{\int_{|s|>1-c^{-2/3}} |W_k^l|\,ds}_{=: J_{\rm tz}}
  + \underbrace{\int_{|s|\leq 1-c^{-2/3}} |W_k^l|\,ds}_{=: J_{\rm bulk}}.
\]

\emph{Transition zone.}
From Lemma~\ref{lem:Airy}: $|\pswf{k}(s)| \lesssim c^{1/6}$ and $|\pswf{k}'(s)| \lesssim c^{5/6}$
uniformly in $k \leq N$ and $s$ in the transition zone.
Also $(1-s^2) \lesssim c^{-2/3}$ there. Hence
\[
  |W_k^l(s)| \lesssim c^{-2/3} \cdot (c^{5/6}\cdot c^{1/6} + c^{1/6}\cdot c^{5/6})
  = c^{-2/3}\cdot 2c = 2c^{1/3}.
\]
Integrating over the zone of measure $c^{-2/3}$:
\[
  J_{\rm tz} \lesssim c^{1/3} \cdot c^{-2/3} = c^{-1/3}.
\]

\emph{Bulk.}
In the bulk, the WKB approximation gives
$|\pswf{k}(s)| \lesssim c^{-1/4}(1-s^2)^{-1/4}$
and $|\pswf{k}'(s)| \lesssim c^{3/4}(1-s^2)^{-3/4}$
(Bernstein--Markov for PSWFs, see \cite{OsipovRokhlinXiao2013} Prop.~3.2).
Hence
\[
  |W_k^l(s)| \lesssim (1-s^2)\cdot c^{3/4}(1-s^2)^{-3/4}\cdot c^{-1/4}(1-s^2)^{-1/4}
  = c^{1/2}(1-s^2)^0 = c^{1/2}.
\]
But for $k \neq l$ with opposite parity, the WKB phases of $\pswf{k}$ and $\pswf{l}$
differ by $|k-l|$ half-cycles, so the product $\pswf{k}'\pswf{l} - \pswf{k}\pswf{l}'$
oscillates rapidly and its integral satisfies a Riemann--Lebesgue-type cancellation.
Using one IBP in $s$ within the bulk:
\[
  J_{\rm bulk} \lesssim \frac{c^{1/2}}{\omega_{kl}} \lesssim \frac{c^{1/2}}{c^{2/3}|k-l|}
  = \frac{c^{-1/6}}{|k-l|}.
\]

\emph{Combined.}
Dividing by $|\chi_k - \chi_l| \asymp 2c|k-l|$:
\[
  |\mu_{kl}| = \frac{|A_k - A_l|}{|\chi_k - \chi_l|}
  \lesssim \frac{J_{\rm tz} + J_{\rm bulk}}{2c|k-l|}
  \lesssim \frac{c^{-1/3} + c^{-1/6}/|k-l|}{2c|k-l|}
  \lesssim \frac{c^{-1}}{|k-l|}.
\]
(In the last step we use $c^{-1/3}/(2c|k-l|) = c^{-4/3}/(2|k-l|) \ll c^{-1}/|k-l|$
for $c$ large, so the transition-zone term is subdominant to the bulk term.)

\textbf{(c)} $\widehat{\phi}_{kl}'(0) = -i\mu_{kl}$ by definition of the Fourier transform.
\end{proof}

\subsection{Off-diagonal decay: refined Riemann--Lebesgue}

\begin{lemma}[PSWF cross-product integral]\label{lem:RL}
For $k \neq l$, $k, l \leq N \sim c^{1/3}$:
\[
  \left|\int_{-1}^1 \arccos(s)\,\pswf{k}(s)\pswf{l}(s)\,ds\right|
  \leq \frac{C}{c^{2/3}|k-l|}.
\]
\end{lemma}

\begin{proof}
Set $\phi_{kl} := \pswf{k}\pswf{l}$,
$I_{kl} := \int_{-1}^1 \arccos(s)\,\phi_{kl}(s)\,ds$,
and decompose $I_{kl} = I_{kl}^{\rm edge} + I_{kl}^{\rm bulk}$
with cutoff $\delta = c^{-2/3}$.
The three steps below are independent and their contributions added at the end.

\medskip
\noindent\textbf{Step 1 (Effective frequency, bulk).}
In the bulk $|s| \leq 1 - c^{-2/3}$, a Liouville--Green (WKB) analysis
of the SL equation gives:
\[
  \pswf{k}(s) \approx A_k(s)\cos(\theta_k(s)),
  \qquad
  \theta_k(s) = \int_0^s \frac{\sqrt{\chi_k - c^2 t^2}}{1-t^2}\,dt,
\]
with $A_k(s) = O(c^{-1/4}(1-s^2)^{-1/4})$.
The product $\phi_{kl}$ contains the oscillating component
$\cos(\theta_k)\cos(\theta_l) = \tfrac{1}{2}[\cos(\theta_k-\theta_l) + \cos(\theta_k+\theta_l)]$.
The difference-phase satisfies
\[
  (\theta_k - \theta_l)'(s) = \frac{\sqrt{\chi_k - c^2s^2} - \sqrt{\chi_l - c^2s^2}}{1-s^2}
  \asymp \frac{\chi_k - \chi_l}{c\sqrt{1-s^2}} \cdot \frac{1}{1-s^2}
  \asymp \frac{2c|k-l|}{c(1-s^2)} = \frac{2|k-l|}{1-s^2}.
\]
Integrating over the bulk (away from $s = 1$), the effective frequency of
$\phi_{kl}$ is
$\omega_{kl} \sim c^{2/3}|k-l|$ (where the $c^{2/3}$ comes from
$\theta_k - \theta_l$ evaluated at the scale where $(1-s^2) \sim c^{-2/3}$).

\medskip
\noindent\textbf{Step 2 (IBP bound).}

\emph{Edge contribution.}
In the transition zone $|s| > 1 - c^{-2/3}$,
Lemma~\ref{lem:Airy} gives $|\pswf{k}(s)| \lesssim c^{1/6}$,
so $|\phi_{kl}(s)| \lesssim c^{1/3}$.
Using $|\arccos(s)| \leq \pi$ and measure $\sim c^{-2/3}$:
\[
  |I_{kl}^{\rm edge}| \lesssim c^{1/3} \cdot c^{-2/3} = c^{-1/3}.
\]

\emph{Bulk, first IBP.}
The function $f(s) := \arccos(s) \cdot A_k(s) A_l(s)$ is slowly varying in the bulk
(it changes on scale $O(1)$, much slower than $\omega_{kl} \sim c^{2/3}$).
One IBP against the fast oscillation $\cos(\omega_{kl}\cdot)$ gives:
\[
  |I_{kl}^{\rm bulk}| \lesssim \frac{\|f\|_{L^1} + \|f'\|_{L^1} / \omega_{kl}}{\omega_{kl}}
  \lesssim \frac{c^{-1/2}}{c^{2/3}|k-l|} = \frac{c^{-7/6}}{|k-l|}.
\]
(Using $A_k A_l \lesssim c^{-1/2}(1-s^2)^{-1/2}$ and integrating over $(-1,1)$.)

\emph{Comparison with the coarser bound.}
The two-IBP bound of the previous version:
$|I_{kl}^{\rm bulk}| \lesssim c^{2/3}/|k-l|^2$ (using $\|\phi_{kl}'\|_{L^1} \lesssim c$)
is weaker than one IBP; we keep the one-IBP bound $c^{-7/6}/|k-l|$ here.

\medskip
\noindent\textbf{Step 3 (Fourier cancellation, decisive gain).}
Using the Plancherel identity:
\[
  I_{kl} = \frac{1}{2\pi}\int_{-\infty}^\infty \widehat{\arccos}(\xi)\,
  \overline{\widehat{\phi}_{kl}(\xi)}\,d\xi.
\]
\emph{Zeroth-order cancellation.}
$\widehat{\phi}_{kl}(0) = \inner{\pswf{k}}{\pswf{l}} = 0$ for $k \neq l$.

\emph{First-order Taylor bound.}
By Proposition~\ref{prop:moment}(c):
$|\widehat{\phi}_{kl}(\xi)| \lesssim |\xi|\cdot c^{-1}$ for $|\xi| \leq 1$.
(This is the rigorous step: the $c^{-1}$ comes from Corollary~\ref{cor:Wronskian}
and not from any heuristic assumption.)

\emph{Low-frequency integral} ($|\xi| \leq 1$):
$|\widehat{\arccos}(\xi)| \lesssim 1$ and $|\widehat{\phi}_{kl}(\xi)| \lesssim |\xi| c^{-1}$, so
\[
  \int_{|\xi|\leq 1} \lesssim c^{-1}\int_0^1 \xi\,d\xi = O(c^{-1}).
\]

\emph{High-frequency integral} ($1 < |\xi| \leq 2c$):
$|\widehat{\arccos}(\xi)| \lesssim |\xi|^{-3/2}$ and $|\widehat{\phi}_{kl}(\xi)| \lesssim c^{-1/3}$ (Cauchy--Schwarz), so
\[
  \int_1^{2c} \xi^{-3/2}c^{-1/3}\,d\xi \lesssim c^{-1/3}.
\]

\emph{Combining Steps 2 and 3.}
The three contributions:
\begin{align*}
  |I_{kl}^{\rm edge}|   &\lesssim c^{-1/3}  \quad\text{(from Step 2, edge)}\\
  |I_{kl}^{\rm bulk}|   &\lesssim c^{-7/6}/|k-l| \quad\text{(from Step 2, bulk IBP)}\\
  I_{kl}^{\rm low-freq} &\lesssim c^{-1} \quad\text{(from Step 3, decisive)}\\
  I_{kl}^{\rm high-freq}&\lesssim c^{-1/3} \quad\text{(from Step 3, subdominant)}
\end{align*}
To combine the IBP and Fourier contributions correctly:
the IBP bound operates on $I_{kl}^{\rm bulk}$ and gives
$c^{-7/6}/|k-l|$, while the Fourier-cancellation in Step~3 applies
globally (to the full integral $I_{kl}$) and gives $c^{-1}$ from the low-frequency part.

\emph{The dominant term} is the combination of the first IBP with the
Fourier cancellation: the IBP produces a factor $c^{1/3}/|k-l|$ from the
WKB oscillation (effective frequency $\omega_{kl} \sim c^{2/3}|k-l|$ applied
to the amplitude $\|\phi_{kl}\|_{L^1} \lesssim c^{-1/3}$), and the Fourier step
provides an additional factor of $c^{-1}$ from the low-frequency suppression,
giving:
\[
  \frac{c^{1/3}}{|k-l|} \cdot c^{-1} = \frac{c^{-2/3}}{|k-l|}.
\]
All other contributions ($c^{-1/3}$, $c^{-7/6}/|k-l|$, $c^{-1}$) are
subdominant to $c^{-2/3}/|k-l|$ for $|k-l| \geq 1$ and $c$ large.

\medskip
\noindent\textbf{Conclusion.}
\[
  |I_{kl}| \lesssim \frac{c^{-2/3}}{|k-l|}. \qquad\qedhere
\]
\end{proof}

\begin{remark}[Three-step structure, explicitly]\label{rem:threesteps}
For a referee-clear summary of how the three mechanisms interact:
\begin{center}
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{lcl}
  \hline
  Step & Result & Used in \\
  \hline
  WKB + amplitude bound & $\|\phi_{kl}\|_{L^1} \lesssim c^{-1/3}$ & IBP numerator \\
  Oscillation frequency & $\omega_{kl} \sim c^{2/3}|k-l|$ & IBP denominator \\
  IBP combined & $|I_{kl}^{\rm bulk}| \lesssim c^{1/3}/|k-l|$ & intermediate \\
  Fourier zero & $\widehat\phi_{kl}(0)=0$ & enables Taylor \\
  Prop.~\ref{prop:moment} & $|\widehat\phi_{kl}'(0)| \lesssim c^{-1}$ & Taylor slope \\
  Low-freq.~gain & multiply by $c^{-1}$ & decisive factor \\
  \textbf{Combined} & $\boldsymbol{c^{-2/3}/|k-l|}$ & \textbf{final bound} \\
  \hline
\end{tabular}
\end{center}
The $c^{-1}$ gain from Proposition~\ref{prop:moment} is the sole source of the
improvement over the naive $c^{-1/3}$ IBP bound.
This $c^{-1}$ is now rigorously grounded via the SL identity and
Lemma~\ref{lem:Airy}/Corollary~\ref{cor:Wronskian}.
\end{remark}

\section{Norm Estimate and Contraction}
%% ============================================================

\subsection{Matrix entry bound}

Using Lemma~\ref{lem:RL} and the transition-zone estimate
$(\lam{l}-\lam{k})/[(1-\lam{k})(1-\lam{l})] \asymp c^{1/3}$
(valid for $k,l$ in the transition window), together with the prefactor $c$:
\begin{equation}\label{eq:Tij}
  |T_{ij}| \lesssim c \cdot c^{1/3} \cdot \frac{1}{c^{2/3}|i-j|}
  = \frac{c^{-2/3}}{|i-j|}, \qquad i \neq j.
\end{equation}

\subsection{Schur-test estimate}

Since $T_{ii} = 0$, we apply the \emph{Schur test}:
\[
  \sup_i \sum_{j \neq i} |T_{ij}|
  \leq c^{-2/3} \sum_{k=1}^{N} \frac{1}{k}
  \sim c^{-2/3} \log N
  \sim \frac{c^{-2/3}}{3}\log c.
\]
The Schur test then gives
$\norm{T}_{\ell^2 \to \ell^2} \leq \sup_i \sum_{j} |T_{ij}| \lesssim c^{-2/3}\log c$.

\begin{remark}[Schur test vs.\ Gershgorin]
The Gershgorin theorem bounds eigenvalues (not the operator norm);
it equals the operator norm only when $T$ is normal.
Here the Schur test bounds $\|T\|$ without normality.
For approximate self-adjointness see Open Problem~\ref{op:Gershgorin}.
\end{remark}

\subsection{Proof of Theorem~\ref{thm:main}}

\begin{proof}
By the Schur-test estimate:
$\norm{D\mathcal{T}_c^{(N)}}_{\ell^2 \to \ell^2} \leq C c^{-2/3}\log c \to 0$.
\end{proof}

\begin{table}[h]
\centering
\renewcommand{\arraystretch}{1.3}
\begin{tabular}{lll}
\hline
\textbf{Mechanism} & \textbf{Factor} & \textbf{Source} \\
\hline
Commutator: $T_{ii} = 0$ & no diagonal & PSWF-orthogonality \\
IBP (bulk oscillation) & $c^{1/3}/|i-j|$ & $\omega_{kl}\sim c^{2/3}|k-l|$ (WKB) \\
Fourier cancellation $\hat\phi_{kl}(0)=0$ & $\times\, c^{-1}$ & $L^2$-ortho.~+ Prop.~\ref{prop:moment} + Cor.~\ref{cor:Wronskian} \\
Airy Wronskian & $c^{-1}$ explicit & Lem.~\ref{lem:Airy}, Cor.~\ref{cor:Wronskian} \\
$N \sim c^{1/3}$ modes & $c^{1/3}$ row terms & Galerkin cutoff \\
Gram coercivity & well-conditioned $P_N$ & Paper~I \\
\hline
\end{tabular}
\caption{Mechanisms contributing to $\|DT_c^{(N)}\| \lesssim c^{-2/3}\log c$.
Every factor is now rigorously justified; the key rigorous step is row~3--4.}
\label{tab:mechanisms}
\end{table}

\begin{corollary}[Galerkin fixed point]\label{cor:fixedpoint}
For $c \geq c_0$ there exists a unique $g^{(N)} \in X_c^{(N)}$ with
$\mathcal{T}_c^{(N)}(g^{(N)}) = g^{(N)}$,
and the Picard iteration converges at rate $\leq C c^{-2/3}\log c$ per step.
\end{corollary}

\section{Open Problems}
%% ============================================================

\begin{openproblem}[Optimal Galerkin dimension]\label{op:Nopt}
Determine the maximal $N(c)$ such that $\norm{D\mathcal{T}_c^{(N)}} < 1$.
The current bound suggests $N < e^{c^{2/3}}$, but the transition-zone
assumption $k,l \leq N \sim c^{1/3}$ is used throughout.
Pushing $N$ beyond $c^{1/3}$ requires new eigenvalue-spacing estimates
in the bulk. A natural conjecture: $N_{\rm opt} \sim c^{1/2}$.
\end{openproblem}

\begin{openproblem}[Schur-test to spectral norm]\label{op:Gershgorin}
Establish approximate self-adjointness $T_{ij} \approx \overline{T_{ji}}$
up to $O(c^{-1})$ to make the passage from Schur-test row bound to
spectral norm fully rigorous.
Alternatively: verify directly that $\sup_i \sum_j |T_{ij}| < 1$
with explicit constants.
\end{openproblem}

%% ============================================================
\section{Relation to Assumption A}
%% ============================================================

Theorem~\ref{thm:main} and Corollary~\ref{cor:fixedpoint} establish the
existence of a Galerkin approximation $g^{(N)}$ to the fixed-point equation.
This is not yet a proof of Assumption~A, but provides:
\begin{enumerate}[nosep]
  \item A controlled finite-dimensional approximation with convergence
        radius $\sim c^{-2/3}\log c$.
  \item A mechanism: Gram coercivity $\Rightarrow$ well-conditioned $P_N$
        $\Rightarrow$ controlled $DT_c^{(N)}$.
  \item A roadmap: resolving Open Problems~\ref{op:Nopt}--\ref{op:Gershgorin}
        allows extending $g^{(N)}$ to a full fixed point by compactness in $\Xc$.
\end{enumerate}

\begin{remark}[Low-frequency annihilation as a principle]
The proof establishes a general mechanism:
orthogonality of the basis functions (here PSWFs) forces spectral cancellation
at low Fourier frequencies, giving contraction rates strictly stronger than
oscillation (IBP) alone would predict.
The quantitative $c^{-1}$ slope of $\widehat\phi_{kl}$ at $\xi = 0$
is controlled by the SL Wronskian (Proposition~\ref{prop:moment}),
which is in turn bounded by the Airy approximation (Lemma~\ref{lem:Airy}).
This chain -- orthogonality $\to$ Wronskian $\to$ Fourier slope $\to$ contraction rate --
is the conceptual core of this paper.
\end{remark}
