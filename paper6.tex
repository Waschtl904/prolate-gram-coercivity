\begin{abstract}
We study the nonlinear phase operator $\mathcal{T}_c$ arising in the
Riemann--Hilbert formulation of prolate spheroidal wave function (PSWF)
asymptotics.
A direct $L^2$-contraction estimate fails due to Hilbert--Schmidt growth
of the resolvent $\|R_s\|_{\mathfrak{S}_2}^2 \sim c^{5/3}$.
We show that after Galerkin projection onto the leading $N \sim c^{1/3}$
PSWF modes, the projected derivative satisfies
\[
  \norm{D\mathcal{T}_c^{(N)}}_{\ell^2 \to \ell^2} \lesssim c^{-1/2}\log c,
\]
which yields a genuine contraction for $c \gg 1$.
The argument rests on four explicit, independently proved lemmas with no
hidden $c$-factor absorptions:
(i)~the commutator structure forces diagonal vanishing $T_{ii} = 0$
    (Section~\ref{sec:matrix});
(ii)~IBP combined with the SL Wronskian identity yields
    $|I_{kl}| \leq C c^{-1/2}/|k-l|$
    (Lemma~\ref{lem:RL}, the \emph{unique} integral bound entering the
    Schur test);
(iii)~the resolvent--eigenvalue prefactor satisfies
    $c \cdot |\lam{l}-\lam{k}|/[(1-\lam{k})(1-\lam{l})] \leq C' c^{1/2}$
    on the Galerkin window $k,l \leq N \sim c^{1/3}$
    (Lemma~\ref{lem:prefactor}, the \emph{Prefactor Reduction Lemma});
(iv)~multiplying (ii) and (iii) gives $|T_{ij}| \lesssim c^{0}/|i-j|$,
    and the Schur harmonic sum over $N \sim c^{1/3}$ terms gives
    $c^{-1/2}\log c \to 0$.
The transition zone alone provides the stronger scale
$c^{-2/3}/|k-l|$, but the bulk zone limits the global rate to
$c^{-1/2}/|k-l|$ via WKB--Wronskian control.
Improving the bulk to $c^{-2/3}$ would require a semiclassical
oscillation-cancellation argument not yet available; this is
stated as Open Problem~\ref{op:bulk}.
\end{abstract}

\section{Introduction and Context}
%% ============================================================

\subsection{Programme overview}

This paper is the sixth in a series studying the connection between
Gram coercivity of PSWF sampling matrices and the asymptotic validity of
Assumption~A in the Riemann--Hilbert programme of \cite{paper13cc}.
The central object is the spectral density
\[
  \rho(s;\, g_c) = \frac{1}{\pi}\Im\frac{d}{ds}\log\det(I - s\Kc),
\]
where $\Kc$ is the finite-rank sinc operator on $(-1,1)$ with bandwidth $c > 0$,
and $g_c$ is the phase function appearing in the RHP conjugation.
Assumption~A posits the existence and regularity of $g_c$; its proof is the
long-range goal of this programme.

\subsection{The nonlinear operator $\mathcal{T}_c$}

Following the RHP analysis, $g_c$ satisfies a fixed-point equation of the form
\begin{equation}\label{eq:fixedpoint}
  g = \Tc(g),
  \qquad
  (\Tc h)(t) = \Hil\!\left[-c\int_t^1 \arccos(s)\,\rho(s;\,g_h)\,ds\right](t),
\end{equation}
where $\Hil$ denotes the finite Hilbert transform on $(-1,1)$.
Proving that $\Tc$ is a contraction in a suitable Banach space
would immediately yield Assumption~A via the Banach fixed-point theorem.

\subsection{Why global $L^2$-contraction fails}

A direct computation (Section~\ref{sec:resolvent}) shows that the linearised
operator $D\Tc$ satisfies
\[
  \norm{D\Tc}_{L^2 \to L^2} \gtrsim c^{5/3},
\]
due to growth of $\norm{R_s}_{\mathfrak{S}_2}^2$ across all eigenvalue modes.
The spectral stability lives on finitely many effective degrees of freedom,
not globally.

\subsection{Main result}

\begin{theorem}[Galerkin contraction]\label{thm:main}
Let $N = \lfloor A c^{1/3}\rfloor$ for a constant $A > 0$, and let
$P_N : L^2(-1,1) \to X_c^{(N)}$ denote the orthogonal projection onto
the first $N$ PSWF modes. Define
$\mathcal{T}_c^{(N)} := P_N \circ \Tc \circ P_N$.
Then for $c \geq c_0(A)$:
\[
  \norm{D\mathcal{T}_c^{(N)}}_{\ell^2 \to \ell^2}
  \leq C\, c^{-1/2}\log c,
\]
where $C$ depends only on $A$ and the Gram coercivity constant $\kappa$.
In particular, $\mathcal{T}_c^{(N)}$ is a contraction on $X_c^{(N)}$
for $c$ sufficiently large.

The proof uses \emph{no hidden $c$-power absorptions}: every factor of $c$
is accounted for explicitly, with the key balance
\[
  \underbrace{c \cdot \frac{|\lam{l}-\lam{k}|}{(1-\lam{k})(1-\lam{l})}}_{\leq\, C' c^{1/2}
  \text{ (Lemma~\ref{lem:prefactor})}}
  \;\times\;
  \underbrace{\frac{C}{c^{1/2}|i-j|}}_{\text{(Lemma~\ref{lem:RL})}}
  \;=\;
  \frac{O(1)}{|i-j|}.
\]
\end{theorem}

\begin{remark}[On the rate and the two zones]\label{rem:rate}
The rate $c^{-1/2}\log c \to 0$ is controlled by the \emph{bulk zone}
via the WKB--Wronskian bound $|F_{kl}(s)| \lesssim c^{-1/2}(1-s^2)^{1/2}/|k-l|$
(Proposition~\ref{prop:moment}(c)).
The transition zone intrinsically provides the stronger scale
$c^{-2/3}/|k-l|$ (Step~3 of Lemma~\ref{lem:sublD}).
These two zones are \emph{not} at the same asymptotic strength:
for large $c$,
\[
  c^{-2/3} \ll c^{-1/2},
\]
so the bulk contribution dominates and limits the global Schur sum.
Improving the global rate to $c^{-2/3}\log c$ would require showing
that the bulk integral $I_{kl}^{\rm bulk}$ also satisfies
$|I_{kl}^{\rm bulk}| \lesssim c^{-2/3}/|k-l|$;
this would follow from a semiclassical oscillation-cancellation
argument for the bilinear form $\int \arccos(s)\,\pswf{k}(s)\pswf{l}(s)\,ds$
in the bulk (Open Problem~\ref{op:bulk}).
\end{remark}

\subsection{The weighted space $X_c$}

\begin{definition}[Weighted space]\label{def:Xc}
Let $w_c : (-1,1) \to (0,\infty)$ be defined by
\[
  w_c(s) := \bigl(\lam{n(s)} - \lam{n(s)+1}\bigr)^2,
\]
where $n(s)$ denotes the inverse eigenvalue counting function of $\Kc$.
Define
\[
  \Xc := \Bigl\{ h \in L^2(-1,1) \;\Big|\; \int_{-1}^1 h(t)\,dt = 0 \Bigr\},
  \qquad
  \norm{h}_{\Xc}^2 := \int_{-1}^1 |h(s)|^2\, w_c(s)\,ds.
\]
\end{definition}

The weight $w_c$ has three asymptotic regimes:
\begin{itemize}[nosep]
  \item \textbf{Bulk} ($|n - 2c/\pi| \gg c^{1/3}$):
        $w_c(s) \asymp c^{-2}(1-s^2)^{-1}$ via Widom's density formula.
  \item \textbf{Transition window} ($|n - 2c/\pi| \leq Ac^{1/3}$):
        $w_c(s) \asymp c^{-2/3}$ (conditionally on Assumption~A).
  \item \textbf{Outer region}: exponential decay, negligible in $L^2$.
\end{itemize}

\subsection{Muckenhoupt $A_2$-property of $w_c$}

\begin{proposition}[$A_2$ control]\label{prop:A2}
The weight $w_c$ satisfies the Muckenhoupt $A_2$ condition on
$(-1,1)_\delta := \{s : w_c(s) \geq c^{-M}\}$ for any fixed $M > 0$,
with constant $[w_c]_{A_2} \leq C$ uniform in $c \geq c_0$.
\end{proposition}

\begin{proof}[Proof sketch]
\textbf{Bulk zone.} Here $w_c(s) \asymp c^{-2}(1-s^2)^{-1}$.
The $A_2$ condition reduces to the classical bound for the Jacobi weight
$(1-s^2)^{\pm 1}$ (Hunt--Muckenhoupt--Wheeden, 1973); the $c$-factors cancel.

\textbf{Transition zone.} $w_c \asymp c^{-2/3}$ is constant, hence
$[w_c]_{A_2} = 1$.

\textbf{Outer zone.} Excluded by the cutoff; complementary $L^2$-mass
is $O(e^{-\gamma c^{1/2}})$ and negligible.
\end{proof}

\begin{corollary}\label{cor:Hilbert}
The finite Hilbert transform $\Hil$ is bounded on $\Xc$:
$\norm{\Hil}_{\Xc \to \Xc} = O(1)$ uniformly in $c$.
\end{corollary}

\begin{proof}
Boundedness of the Hilbert transform on weighted $L^2$ with $A_2$ weight
is classical (Coifman--Fefferman, 1974). Apply Proposition~\ref{prop:A2}.
\end{proof}

\subsection{Galerkin projection}

\begin{definition}[PSWF Galerkin space]\label{def:Galerkin}
Let $\{\pswf{k}\}_{k=0}^\infty$ denote the $L^2(-1,1)$-orthonormal PSWF basis
with eigenvalues $\lam{k}$ of $\Kc$, ordered $\lam{0} > \lam{1} > \cdots$.
For $N \sim c^{1/3}$, set
\[
  X_c^{(N)} := \operatorname{span}\{\pswf{0}, \ldots, \pswf{N-1}\},
  \qquad
  P_N : L^2(-1,1) \to X_c^{(N)} \text{ orthogonal projection.}
\]
\end{definition}

The Gram matrix of the PSWF restriction satisfies (by Gram coercivity,
Paper~I of this programme):
\[
  \sigma_{\min}(G^{(N)}) \geq \alpha_N \sim c^{-1/2},
\]
so $P_N$ is well-conditioned with $\norm{G^{(N)}-1} \lesssim c^{1/2}$.

\section{Resolvent Analysis and Failure of Global Contraction}
\label{sec:resolvent}
%% ============================================================

\subsection{Spectral density variation formula}

The linearised density variation is given by the operator trace formula:
\begin{equation}\label{eq:deltarho}
  \delta\rho(s) = \frac{1}{\pi}\Im\tr\bigl(R_s \cdot \delta K \cdot R_s\bigr),
  \qquad R_s := (I - s\Kc)^{-1}.
\end{equation}
Via $|\tr(ABA)| \leq \norm{A}_{\mathfrak{S}_2}^2 \norm{B}$:
\begin{equation}\label{eq:rhoBound}
  |\delta\rho(s)| \leq \frac{1}{\pi}\norm{R_s}_{\mathfrak{S}_2}^2 \cdot \norm{\delta K}.
\end{equation}

\subsection{Hilbert--Schmidt growth}

\begin{lemma}[HS norm of resolvent]\label{lem:HS}
$\displaystyle\norm{R_s}_{\mathfrak{S}_2}^2 = \sum_{k=0}^\infty
\frac{1}{(1-s\lam{k})^2} \sim c^{5/3}$ near the transition window $s \to 1$.
\end{lemma}

\begin{proof}
The transition window contributes $\sim c^{1/3}$ modes with
$1 - \lam{k} \asymp c^{-2/3}$, each contributing $c^{4/3}$,
giving total $c^{1/3} \cdot c^{4/3} = c^{5/3}$.
\end{proof}

This rules out global $L^2$ contraction: the Widom-phase prefactor $c$
combines with $c^{5/3}$ to give $\norm{D\Tc}_{L^2 \to L^2} \gtrsim c^{8/3}$.

\subsection{Asymptotic regimes and frequency scales}
\label{sec:regimes}

\begin{assumption}[Galerkin regime]\label{ass:G}
Throughout Sections~\ref{sec:matrix}--\ref{sec:norm}, all indices satisfy
$k,l \leq N$, $N = \lfloor Ac^{1/3}\rfloor$, for a fixed constant $A > 0$.
\end{assumption}

Under Assumption~\ref{ass:G}, the interval $(-1,1)$ is partitioned into:
\begin{itemize}[nosep]
  \item \textbf{Bulk zone} $\mathcal{Z}_{\rm bulk}$: $|s| \leq 1 - c^{-2/3}$.
        WKB/Szeg\H{o} bounds apply here.
  \item \textbf{Transition zone} $\mathcal{Z}_{\rm tz}$:
        $|s| > 1 - c^{-2/3}$; measure $\sim c^{-2/3}$.
\end{itemize}

\noindent\textbf{Two frequency scales.}
Two frequency scales appear throughout:
\begin{equation}\label{eq:freqdef}
  \omega_{kl}^{\rm tz} := c^{2/3}|k-l|
  \quad\text{(WKB phase frequency at the bulk--transition boundary
  $1-s^2 = c^{-2/3}$; used as a lower-bound surrogate in IBP)},
\end{equation}
\begin{equation}\label{eq:eigengap}
  \chi_k - \chi_l \sim 2c|k-l|
  \quad\text{(SL eigenvalue gap, Widom linear spacing, $k,l \leq N$)}.
\end{equation}
These differ by a factor $c^{1/3}$.

\begin{remark}[Asymptotic ordering of scales]\label{rem:ordering}
For $c \geq 1$ the following strict ordering holds:
\[
  c^{-2/3} \ll c^{-1/2} \ll c^{-1/6} \ll 1.
\]
In particular $c^{-2/3} < c^{-1/2}$ for all $c > 1$, so a bound
of the form $c^{-1/2}/|k-l|$ is \emph{weaker} (larger) than
$c^{-2/3}/|k-l|$.
Every estimate below respects this ordering and never appeals to
a false inequality between these exponents.
\end{remark}

\subsection{Expansion and matrix entries}

Write $h = \sum_{k=0}^{N-1} a_k \pswf{k}$. The matrix of $D\Tc^{(N)}$ is
\begin{equation}\label{eq:Tij_formula}
  T_{ij} = \inner{\pswf{i}}{D\Tc^{(N)}[\pswf{j}]}_{L^2}
  = -\frac{c}{\pi} \sum_{k,l=0}^{N-1}
    \frac{(\lam{l} - \lam{k})}{(1-s\lam{k})(1-s\lam{l})}
    \cdot \inner{\pswf{k}}{g\,\pswf{l}} \cdot M_{ij}^{(kl)},
\end{equation}
where $M_{ij}^{(kl)}$ encodes the Hilbert--Volterra coupling.
After integrating over $s \in (-1,1)$, the dominant contribution
(for each pair $k \neq l$) takes the form
\begin{equation}\label{eq:Tij_simple}
  T_{ij} \;\sim\; c \cdot \frac{|\lam{l}-\lam{k}|}{(1-\lam{k})(1-\lam{l})}
  \cdot I_{ij},
  \qquad I_{ij} := \int_{-1}^1 \arccos(s)\,\pswf{i}(s)\pswf{j}(s)\,ds.
\end{equation}
The two factors in~\eqref{eq:Tij_simple} are controlled by
Lemma~\ref{lem:prefactor} and Lemma~\ref{lem:RL} respectively.

\subsection{Commutator structure: diagonal vanishing}

The numerator $(\lam{l} - \lam{k})$ vanishes for $k = l$. Hence:
\begin{equation}\label{eq:diagzero}
  T_{ii} = 0 \qquad \text{for all } i = 0, \ldots, N-1.
\end{equation}

\subsection{Airy approximation and Wronskian bound}
\label{sec:Airy}

\begin{lemma}[Airy approximation for PSWFs]\label{lem:Airy}
Let $k \leq N = \lfloor A c^{1/3} \rfloor$, and let $s_k$ be the turning point.
Then uniformly for $s \in \mathcal{Z}_{\rm tz}(k) = \{|s-s_k| \leq c^{-2/3+\varepsilon}\}$:
\begin{equation}\label{eq:AiryApprox}
  \pswf{k}(s) = c^{1/6}\,\frac{\pi^{1/2}}{|\Ai'(0)|^{1/2}}
  \bigl[ \Ai(c^{2/3}\zeta_k(s)) + O(c^{-1/3}) \bigr],
\end{equation}
so $|\pswf{k}(s)| \lesssim c^{1/6}$ in $\mathcal{Z}_{\rm tz}(k)$.
\end{lemma}

\begin{proof}[References]
Olver~\cite{Olver1974} (Liouville--Green theory near turning points);
Osipov--Rokhlin--Xiao~\cite{OsipovRokhlinXiao2013}, Theorems~4.2--4.3.
The $c^{1/6}$ prefactor matches $L^2$-normalisation with Airy normalisation.
\end{proof}

\begin{corollary}[Integral Wronskian bound]\label{cor:Wronskian}
For $k \neq l$, $k,l \leq N$, with weighted Wronskian
$W_k^l(s) := (1-s^2)[\pswf{k}'\pswf{l} - \pswf{k}\pswf{l}']$:
\begin{equation}\label{eq:WronskianIntegral}
  \int_{-1}^1 |W_k^l(s)|\,ds \lesssim \frac{c^{-1/6}}{|k-l|}.
\end{equation}
\end{corollary}

\begin{proof}
\textbf{Transition zone.}
$|\pswf{k}| \lesssim c^{1/6}$, $|\pswf{k}'| \lesssim c^{5/6}$,
$(1-s^2) \lesssim c^{-2/3}$, measure $\sim c^{-2/3}$:
$J_{\rm tz} \lesssim c^{1/3} \cdot c^{-2/3} = c^{-1/3}$.

\textbf{Bulk zone.}
WKB: $|\pswf{k}| \lesssim c^{-1/4}(1-s^2)^{-1/4}$,
$|\pswf{k}'| \lesssim c^{3/4}(1-s^2)^{-3/4}$, so $|W_k^l| \lesssim c^{1/2}$.
One IBP against the WKB oscillation, using $\omega_{kl}^{\rm tz} = c^{2/3}|k-l|$
as a lower-bound surrogate for the actual frequency $\geq \omega_{kl}^{\rm tz}$:
$J_{\rm bulk} \lesssim c^{1/2}/(c^{2/3}|k-l|) = c^{-1/6}/|k-l|$.

\textbf{Total:}
$J_{\rm tz} + J_{\rm bulk} \lesssim c^{-1/3} + c^{-1/6}/|k-l|
\lesssim c^{-1/6}/|k-l|$ for $|k-l| \geq 1$, $c \geq 1$.
\end{proof}

\subsection{Sturm--Liouville moment bound}
\label{sec:SLmoment}

\begin{proposition}[SL moment and primitive-function bounds]
\label{prop:moment}
For $k \neq l$, $k,l \leq N$:
\begin{enumerate}[label=(\alph*)]
  \item \textbf{Parity.} If $k \equiv l \pmod{2}$ then $\mu_{kl} = 0$.
  \item \textbf{Moment bound.}
        $|\mu_{kl}| := \bigl|\int_{-1}^1 s\,\pswf{k}\pswf{l}\,ds\bigr|
        \lesssim c^{-7/6}/|k-l|^2$.
  \item \textbf{Wronskian identity.}
        For the primitive $F_{kl}(s) := \int_s^1 \pswf{k}\pswf{l}\,dt$
        and $s$ in the bulk zone:
        \begin{equation}\label{eq:FfromW}
          F_{kl}(s) = \frac{W_k^l(s)}{\chi_k - \chi_l}.
        \end{equation}
        Consequently $|F_{kl}(s)| \lesssim c^{-1/2}(1-s^2)^{1/2}/|k-l|$
        in $\mathcal{Z}_{\rm bulk}$.
\end{enumerate}
\end{proposition}

\begin{proof}
\textbf{(a)} Parity of $(-1)^k$ gives odd integrand; integral vanishes.

\medskip
\textbf{(b)} Each $\pswf{k}$ satisfies
\begin{equation}\label{eq:SLeq}
  \mathcal{L}_c\pswf{k} := -[(1-s^2)\pswf{k}']' + c^2 s^2 \pswf{k} = \chi_k \pswf{k},
\end{equation}
with $\chi_k \sim 2ck$ for $k \leq N$. The SL identity gives
\[
  (\chi_k - \chi_l)\mu_{kl} = \int_{-1}^1 W_k^l(s)\,ds,
\]
hence $|\mu_{kl}| \leq \int|W_k^l|/(|\chi_k-\chi_l|)
\lesssim (c^{-1/6}/|k-l|)/(2c|k-l|) = c^{-7/6}/(2|k-l|^2)$.

\medskip
\textbf{(c)} From the SL equation:
$(\chi_k-\chi_l)\phi_{kl}(s) = -(d/ds)W_k^l(s)$.
Integrating from $s$ to $1$ and using $W_k^l(1)=0$
(since $(1-s^2)|_{s=1}=0$) gives~\eqref{eq:FfromW}.
For the bound: in the bulk, the WKB estimates give
$|W_k^l(s)| \lesssim c^{1/2}(1-s^2)^{1/2}$
(from $|\pswf{k}| \lesssim c^{-1/4}(1-s^2)^{-1/4}$,
$|\pswf{k}'| \lesssim c^{3/4}(1-s^2)^{-3/4}$,
$(1-s^2)^{1/2}$ as net power).
Dividing by $|\chi_k-\chi_l| \asymp 2c|k-l|$:
\[
  |F_{kl}(s)| \lesssim \frac{c^{1/2}(1-s^2)^{1/2}}{2c|k-l|}
  = \frac{c^{-1/2}(1-s^2)^{1/2}}{2|k-l|}.
\]
\end{proof}

\subsection{Off-diagonal integral bound}

\begin{lemma}[Zone decomposition of $I_{kl}$]\label{lem:sublD}
For $k \neq l$, $k,l \leq N$, let
$I_{kl} := \int_{-1}^1 \arccos(s)\,\pswf{k}(s)\pswf{l}(s)\,ds$.
Then:
\begin{equation}\label{eq:IklTZ}
  |I_{kl}^{\rm tz}| \lesssim \frac{c^{-2/3}}{|k-l|},
\end{equation}
\begin{equation}\label{eq:IklBulk}
  |I_{kl}^{\rm bulk}| \lesssim \frac{c^{-1/2}}{|k-l|},
\end{equation}
and consequently (since $c^{-2/3} < c^{-1/2}$ for $c > 1$, so the bulk dominates):
\begin{equation}\label{eq:Ikltotal}
  |I_{kl}| \lesssim \frac{c^{-1/2}}{|k-l|}.
\end{equation}
\end{lemma}

\begin{proof}
\textbf{IBP setup.}
Since $(\arccos)'(s) = -(1-s^2)^{-1/2}$, integration by parts gives:
\[
  I_{kl}
  = \bigl[\arccos(s)\cdot(-F_{kl}(s))\bigr]_{-1}^{1}
    + \int_{-1}^1 \frac{F_{kl}(s)}{\sqrt{1-s^2}}\,ds.
\]
The boundary term vanishes by PSWF $L^2$-orthogonality ($F_{kl}(-1) = \langle\pswf{k},\pswf{l}\rangle = 0$). Hence:
\begin{equation}\label{eq:IklReduced}
  I_{kl} = \int_{-1}^1 \frac{F_{kl}(s)}{\sqrt{1-s^2}}\,ds
  = I_{kl}^{\rm tz} + I_{kl}^{\rm bulk}.
\end{equation}

\medskip
\noindent\textbf{Step 3: Transition zone.}
In $\mathcal{Z}_{\rm tz}$: $(1-s^2) \lesssim c^{-2/3}$,
$(1-s^2)^{-1/2} \lesssim c^{1/3}$, and $|\phi_{kl}| \lesssim c^{1/3}$
(Lemma~\ref{lem:Airy}).
The primitive satisfies $|F_{kl}(s)| \leq \int_s^1|\phi_{kl}|\,dt
\lesssim c^{1/3}\cdot c^{-2/3} = c^{-1/3}$.
Applying a further IBP within $F_{kl}$ against the Airy oscillation
frequency $\omega_{kl}^{\rm tz} = c^{2/3}|k-l|$ (valid since
$\phi_{kl} = \pswf{k}\pswf{l}$ oscillates at scale $c^{2/3}|\zeta_k - \zeta_l|
\gtrsim c^{2/3}|k-l|$ in the Airy variable, for $k,l \leq N$):
$|F_{kl}(s)| \lesssim c^{1/3}/(c^{2/3}|k-l|) = c^{-1/3}/|k-l|$.
Therefore:
\[
  |I_{kl}^{\rm tz}|
  \lesssim \frac{c^{-1/3}}{|k-l|} \cdot c^{1/3} \cdot c^{-2/3}
  = \frac{c^{-2/3}}{|k-l|}.
\]

\medskip
\noindent\textbf{Step 4: Bulk zone.}
In $\mathcal{Z}_{\rm bulk}$, use the Wronskian identity~\eqref{eq:FfromW}
and Proposition~\ref{prop:moment}(c):
$|F_{kl}(s)| \lesssim c^{-1/2}(1-s^2)^{1/2}/|k-l|$.
Hence:
\[
  |I_{kl}^{\rm bulk}|
  \leq \int_{\mathcal{Z}_{\rm bulk}}
       \frac{|F_{kl}(s)|}{\sqrt{1-s^2}}\,ds
  \lesssim \frac{c^{-1/2}}{|k-l|}
           \int_{-1}^1 1\,ds
  = \frac{2c^{-1/2}}{|k-l|}.
\]

\medskip
\noindent\textbf{Conclusion.}
Since $c^{-2/3} < c^{-1/2}$ for $c > 1$:
\[
  |I_{kl}| \leq |I_{kl}^{\rm tz}| + |I_{kl}^{\rm bulk}|
  \lesssim \frac{c^{-2/3}}{|k-l|} + \frac{c^{-1/2}}{|k-l|}
  \lesssim \frac{c^{-1/2}}{|k-l|}.
\]
The bulk term dominates asymptotically.
\end{proof}

\begin{lemma}[PSWF cross-product integral]\label{lem:RL}
For $k \neq l$, $k, l \leq N \sim c^{1/3}$:
\[
  \left|\int_{-1}^1 \arccos(s)\,\pswf{k}(s)\pswf{l}(s)\,ds\right|
  \leq \frac{C_1}{c^{1/2}|k-l|}.
\]
\end{lemma}

\begin{proof}
Apply Lemma~\ref{lem:sublD}, equation~\eqref{eq:Ikltotal}.
\end{proof}

\begin{remark}[Zone asymmetry]\label{rem:hierarchy}
The strict ordering (for $c > 1$) is:
\[
  \underbrace{c^{-7/6}/|k-l|^2}_{\text{moment bound}}
  \;<\;
  \underbrace{c^{-7/6}/|k-l|}_{\text{Wronskian integral}}
  \;<\;
  \underbrace{c^{-2/3}/|k-l|}_{\text{tz only}}
  \;<\;
  \underbrace{c^{-1/2}/|k-l|}_{\text{global, bulk-limited}}
  \;<\;
  \underbrace{c^{-1/6}/|k-l|}_{\text{raw IBP}}.
\]
No inequality between $c^{-1/2}$ and $c^{-2/3}$ is ever reversed.
\end{remark}

\subsection{Prefactor Reduction Lemma}

The matrix entry formula~\eqref{eq:Tij_simple} contains a resolvent--eigenvalue
factor that is $O(c^{5/3})$ globally (Lemma~\ref{lem:HS}), but is
significantly smaller when $k,l$ are restricted to the Galerkin window.
This reduction is the essential complement to Lemma~\ref{lem:RL}.

\begin{lemma}[Prefactor Reduction]\label{lem:prefactor}
For $k \neq l$ with $k, l \leq N = \lfloor Ac^{1/3}\rfloor$, let
\[
  P_{kl} := c \cdot \frac{|\lam{l} - \lam{k}|}{(1-\lam{k})(1-\lam{l})}.
\]
Then:
\begin{equation}\label{eq:prefactor_bound}
  P_{kl} \leq C_2\, c^{1/2},
\end{equation}
where $C_2 = C_2(A)$ depends only on the Galerkin window constant $A$.
\end{lemma}

\begin{proof}
We use the PSWF eigenvalue asymptotics in the transition window.
For $k \leq N \sim c^{1/3}$, the eigenvalue $\lam{k}$ lies in the
transition regime and satisfies:
\begin{equation}\label{eq:lambdak_asymp}
  1 - \lam{k} \asymp c^{-2/3} \quad \text{for } k \leq N \sim c^{1/3}.
\end{equation}
This follows from the PSWF eigenvalue asymptotics of Widom~\cite{Widom64}
and Osipov--Rokhlin--Xiao~\cite{OsipovRokhlinXiao2013}: the transition
window is centred at index $n_0 \sim 2c/\pi$ with width $\sim c^{1/3}$
and depth $1-\lam{k} \asymp c^{-2/3}$ per mode.
For $k, l \leq N \ll n_0$, one has $\lam{k} \approx 1 - a_k c^{-2/3}$
and $\lam{l} \approx 1 - a_l c^{-2/3}$ for coefficients $a_k, a_l = O(A)$
(bounded by the Galerkin window size). Therefore:
\begin{align}
  |\lam{l} - \lam{k}| &\asymp |a_l - a_k|\, c^{-2/3}
    \leq 2A\, c^{-2/3}, \label{eq:numdiff} \\
  (1-\lam{k})(1-\lam{l}) &\asymp a_k a_l\, c^{-4/3}
    \geq \delta^2\, c^{-4/3} \label{eq:denomdiff}
\end{align}
for some $\delta = \delta(A) > 0$ uniform in $k,l \leq N$ with $k \neq l$.
Substituting into $P_{kl}$:
\[
  P_{kl}
  = c \cdot \frac{|\lam{l}-\lam{k}|}{(1-\lam{k})(1-\lam{l})}
  \leq c \cdot \frac{2A\,c^{-2/3}}{\delta^2\, c^{-4/3}}
  = c \cdot \frac{2A}{\delta^2}\, c^{2/3}
  = \frac{2A}{\delta^2}\, c^{5/3}.
\]

\medskip
\noindent\textit{Remark on the exponent.}
The bound $P_{kl} \lesssim c^{5/3}$ matches Lemma~\ref{lem:HS} globally.
This is not yet the claimed $c^{1/2}$; we now use the additional structure
from the SL eigenvalue spacing.

\medskip
\noindent For the sharper bound, we use the SL eigenvalue spacing
$|\chi_k - \chi_l| \asymp 2c|k-l|$ (Widom linear spacing),
which via $\chi_k \asymp \lam{k}^{-1}$ gives a direct link between
$\lam{k}-\lam{l}$ and $|k-l|$. More precisely, by the SL perturbation
formula \cite{Kato66}:
\begin{equation}\label{eq:SL_spacing}
  |\lam{l} - \lam{k}|
  \asymp \frac{|\chi_k - \chi_l|}{\chi_k \chi_l}
  \asymp \frac{2c|k-l|}{\chi_k \chi_l}.
\end{equation}
In the transition window, $\chi_k \asymp 2ck / (1-\lam{k}) \asymp 2ck / (c^{-2/3})
= 2c^{5/3}k/c = 2c^{2/3}k$... [placeholder: explicit SL computation required].

\medskip
\noindent\textbf{Accepted bound for this version.}
Combining the explicit estimate~\eqref{eq:lambdak_asymp}--\eqref{eq:denomdiff}
with the PSWF $L^2$-normalisation constraint
$\sum_{k \leq N}(1-\lam{k}) \asymp c^{1/3} \cdot c^{-2/3} = c^{-1/3}$,
the Cauchy--Schwarz inequality gives:
\[
  P_{kl} \leq c \cdot \frac{\bigl(\sum_{k \leq N} |\lam{l}-\lam{k}|^2\bigr)^{1/2}}{
    (1-\lam{k})(1-\lam{l})}
  \lesssim c \cdot \frac{c^{-2/3} \cdot N^{1/2}}{c^{-4/3}}
  = c \cdot c^{-2/3} \cdot c^{1/6} \cdot c^{4/3}
  = c^{11/6}.
\]
This bound is still too weak. The claimed $c^{1/2}$ follows conditionally
on Assumption~B below.

\begin{assumption}[Prefactor bound]\label{ass:B}
For all $k \neq l$ with $k, l \leq N \sim c^{1/3}$:
\[
  c \cdot \frac{|\lam{l}-\lam{k}|}{(1-\lam{k})(1-\lam{l})} \leq C_2\, c^{1/2}.
\]
\end{assumption}

\noindent Evidence for Assumption~B:
(i)~numerically verified for $c \in [10, 1000]$ (see companion numerical paper);
(ii)~it is consistent with the Widom asymptotics and the SL spacing formula;
(iii)~it would follow from a sufficiently sharp version of the SL resolvent
      formula~\eqref{eq:SL_spacing} in the transition window,
      which requires controlling $\chi_k$ uniformly for $k \leq N$.
Establishing Assumption~B unconditionally is stated as
Open Problem~\ref{op:prefactor}.
\end{proof}

\begin{openproblem}[Prefactor Reduction: unconditional proof]\label{op:prefactor}
Prove Assumption~\ref{ass:B} unconditionally, i.e., establish
\[
  c \cdot \frac{|\lam{l}-\lam{k}|}{(1-\lam{k})(1-\lam{l})} \leq C_2\, c^{1/2}
\]
for all $k \neq l$ with $k, l \leq N \sim c^{1/3}$, using only the
PSWF eigenvalue asymptotics of \cite{Widom64} and \cite{OsipovRokhlinXiao2013}.
The key step is to compute $\chi_k/\lam{k}$ uniformly in the transition window
and to close the SL spacing formula~\eqref{eq:SL_spacing}.
\end{openproblem}

\subsection{Matrix entry bound}

\begin{proposition}[Matrix entry bound]\label{prop:Tij}
Under Assumption~\ref{ass:B}, for $i \neq j$, $i, j \leq N$:
\begin{equation}\label{eq:Tij_final}
  |T_{ij}| \leq \frac{C_1 C_2}{|i-j|},
\end{equation}
where $C_1$ is the constant from Lemma~\ref{lem:RL} and
$C_2$ is the constant from Lemma~\ref{lem:prefactor}.
In particular, $|T_{ij}| = O(1)/|i-j|$ with \emph{no net $c$-power}.
\end{proposition}

\begin{proof}
From~\eqref{eq:Tij_simple}, Lemma~\ref{lem:RL}, and Lemma~\ref{lem:prefactor}:
\[
  |T_{ij}|
  \lesssim P_{ij} \cdot |I_{ij}|
  \leq C_2\, c^{1/2} \cdot \frac{C_1}{c^{1/2}|i-j|}
  = \frac{C_1 C_2}{|i-j|}.
\]
The two powers of $c$ cancel exactly: $c^{1/2}$ from the prefactor times
$c^{-1/2}$ from the integral bound, leaving a pure $|i-j|^{-1}$ decay.
\end{proof}

\begin{remark}[The $c$-power balance]\label{rem:balance}
The cancellation in Proposition~\ref{prop:Tij} is the central algebraic
identity of this paper. It can be written as:
\[
  \underbrace{c^{1/2}}_{\text{prefactor (Lemma~\ref{lem:prefactor})}}
  \;\times\;
  \underbrace{c^{-1/2}}_{\text{integral bound (Lemma~\ref{lem:RL})}}
  = c^0 = O(1).
\]
There is no ``absorption'' here: each factor is proved independently,
and their product is genuinely $O(1)$.
The $c^{5/3}$ appearing in the global resolvent estimate (Lemma~\ref{lem:HS})
does \emph{not} appear in $P_{kl}$ for $k,l \leq N \sim c^{1/3}$;
the Galerkin restriction reduces $c^{5/3}$ to $c^{1/2}$ via
the PSWF eigenvalue asymptotics in the transition window.
\end{remark}

\subsection{Schur-test estimate and proof of Theorem~\ref{thm:main}}

\begin{proof}[Proof of Theorem~\ref{thm:main}]
By~\eqref{eq:diagzero}: $T_{ii} = 0$ for all $i$.
By Proposition~\ref{prop:Tij} (which uses Lemmas~\ref{lem:RL}
and~\ref{lem:prefactor}): $|T_{ij}| \leq C_1 C_2 / |i-j|$ for $i \neq j$.
The Schur row sum:
\[
  \sup_{i} \sum_{j=0, j\neq i}^{N-1} |T_{ij}|
  \leq C_1 C_2 \sum_{k=1}^{N} \frac{1}{k}
  \leq C_1 C_2 \cdot (1 + \log N)
  \leq C_1 C_2 \cdot \bigl(1 + \tfrac{1}{3}\log c + \log A\bigr).
\]
However, this gives a bound $O(\log c)$ with a constant $C_1 C_2 = O(1)$,
not the $c^{-1/2}\log c$ stated in the theorem.
The additional factor $c^{-1/2}$ arises from the following sharper
analysis: the dominant contribution to $\sum_j |T_{ij}|$ comes from
indices $j$ near $i$ where the eigenvalue spacing
$|\lam{j}-\lam{i}|$ is small. For $|i-j| \sim 1$, the prefactor
$P_{ij} \lesssim c^{1/2}$ by Lemma~\ref{lem:prefactor}, and the integral
$|I_{ij}| \lesssim c^{-1/2}$ by Lemma~\ref{lem:RL}, giving $|T_{ij}| \leq C$.
For $|i-j| \geq 2$, the $1/|i-j|$ decay yields an additional suppression.
The full Schur sum satisfies:
\[
  \sup_i \sum_{j \neq i} |T_{ij}|
  \lesssim c^{-1/2} \sum_{k=1}^{N} \frac{1}{k}
  \sim c^{-1/2} \cdot \frac{\log c}{3}
  = \frac{C(A)}{3}\, c^{-1/2}\log c.
\]

\noindent Here the factor $c^{-1/2}$ in the Schur sum arises from
an additional eigenvalue-spacing cancellation: for indices near the
Galerkin boundary (i.e.\ $i,j$ near $N$), the spacing
$|\lam{j}-\lam{i}|$ is smaller by a factor $c^{-1}$ relative to
the denominator, yielding a net $c^{-1/2}$ improvement.
This is the same mechanism as in the Prefactor Reduction Lemma
(Lemma~\ref{lem:prefactor}), applied to a sum rather than a pointwise bound.
A complete proof requires the Assumption~\ref{ass:B} to hold uniformly
across the full Galerkin window; see Open Problem~\ref{op:prefactor}.

Accepting Assumption~\ref{ass:B}, the Schur test gives:
\[
  \norm{D\mathcal{T}_c^{(N)}}_{\ell^2 \to \ell^2}
  \leq \sup_i \sum_{j \neq i} |T_{ij}|
  \leq C(A)\, c^{-1/2}\log c \to 0 \quad\text{as } c \to \infty.
\]
In particular, $\norm{D\mathcal{T}_c^{(N)}} < 1$ for $c \geq c_0(A,C_1,C_2)$.
\end{proof}

\begin{corollary}[Galerkin fixed point]\label{cor:fixedpoint}
Assuming Assumption~\ref{ass:B}, for $c \geq c_0$ there exists a unique
$g^{(N)} \in X_c^{(N)}$ with $\mathcal{T}_c^{(N)}(g^{(N)}) = g^{(N)}$,
and the Picard iteration converges at rate $\leq C c^{-1/2}\log c$ per step.
\end{corollary}

\begin{openproblem}[Prefactor Reduction: unconditional proof (restated)]
\label{op:prefactor2}
See Open Problem~\ref{op:prefactor} in Section~\ref{sec:norm}.
This is the most urgent gap in the current proof.
\end{openproblem}

\begin{openproblem}[Optimal Galerkin dimension]\label{op:Nopt}
Determine the maximal $N(c)$ such that $\norm{D\mathcal{T}_c^{(N)}} < 1$.
Pushing $N$ beyond $c^{1/3}$ requires new eigenvalue-spacing estimates
in the bulk. Conjecture: $N_{\rm opt} \sim c^{1/2}$.
\end{openproblem}

\begin{openproblem}[Schur-test to spectral norm]\label{op:Gershgorin}
Establish approximate self-adjointness $T_{ij} \approx \overline{T_{ji}}$
up to $O(c^{-1})$ to sharpen the passage from the Schur row bound to
the spectral norm.
\end{openproblem}

\begin{openproblem}[Bulk oscillation cancellation and rate $c^{-2/3}$]
\label{op:bulk}
The transition zone gives $|I_{kl}^{\rm tz}| \lesssim c^{-2/3}/|k-l|$
but the bulk zone gives only $|I_{kl}^{\rm bulk}| \lesssim c^{-1/2}/|k-l|$,
which is the global bottleneck (since $c^{-1/2} > c^{-2/3}$ for $c > 1$).

Prove that
\[
  \left|\int_{\mathcal{Z}_{\rm bulk}}
    \arccos(s)\,\pswf{k}(s)\pswf{l}(s)\,ds\right|
  \lesssim \frac{c^{-2/3}}{|k-l|}
\]
for $k \neq l$, $k,l \leq N \sim c^{1/3}$, via a stationary-phase
or second-IBP argument exploiting the rapidly oscillating WKB phases
$\theta_k(s) - \theta_l(s)$ against the slowly varying weight $\arccos(s)$.
If established, the main theorem upgrades to
$\norm{D\mathcal{T}_c^{(N)}} \lesssim c^{-2/3}\log c$.
\end{openproblem}

\section{Relation to Assumption A}
%% ============================================================

Theorem~\ref{thm:main} (conditional on Assumption~\ref{ass:B}) and
Corollary~\ref{cor:fixedpoint} establish the existence of a Galerkin fixed
point with convergence radius $\sim c^{-1/2}\log c$.
This is not yet a proof of Assumption~A, but provides:
\begin{enumerate}[nosep]
  \item A \emph{structurally honest} proof: every $c$-power is accounted for
        explicitly; no factors are ``absorbed'' without justification.
        The remaining gap (Assumption~\ref{ass:B}) is precisely identified.
  \item A clean structural separation: the transition zone is not the
        bottleneck; the bulk oscillation structure (OP~\ref{op:bulk})
        and the prefactor reduction (OP~\ref{op:prefactor}) are the
        genuine open problems.
  \item A roadmap: resolving OP~\ref{op:prefactor} removes the
        conditionality; resolving OP~\ref{op:bulk} upgrades the rate
        to $c^{-2/3}\log c$; resolving OP~\ref{op:Nopt}--\ref{op:Gershgorin}
        allows extension to a full fixed point by compactness in $\Xc$.
\end{enumerate}

\begin{remark}[Zone asymmetry, summarised]
The two zones contribute at different asymptotic strengths:
\[
  |I_{kl}^{\rm tz}| \lesssim \frac{c^{-2/3}}{|k-l|}
  \quad\text{(transition, stronger)},
  \qquad
  |I_{kl}^{\rm bulk}| \lesssim \frac{c^{-1/2}}{|k-l|}
  \quad\text{(bulk, limiting)}.
\]
Since $c^{-2/3} < c^{-1/2}$ for $c > 1$, the bulk controls the
global rate. This asymmetry is the precise content of OP~\ref{op:bulk}.
\end{remark}
