\begin{abstract}
We study the nonlinear phase operator $\mathcal{T}_c$ arising in the
Riemann--Hilbert formulation of prolate spheroidal wave function (PSWF)
asymptotics.
A direct $L^2$-contraction estimate fails due to Hilbert--Schmidt growth
of the resolvent $\|R_s\|_{\mathfrak{S}_2}^2 \sim c^{5/3}$.
After Galerkin projection onto the leading $N \sim c^{1/3}$ PSWF modes,
the analysis has two conditional layers above an unconditional base:

\smallskip
\noindent\textbf{Layer~0 (unconditional):}
(i)~$T_{ii}=0$;
(ii)~$|I_{kl}|\leq C_1c^{-1/2}/|k-l|$ (IBP $+$ Wronskian $+$ SL);
(iii)~the bulk oscillation barrier is localised to
$|k-l|\lesssim c^{1/6}$;
(iv)~$\|T\|\geq\tfrac{\kappa}{2}\log c$ in the pointwise-bound class
(No-go Theorem).
Items~(i)--(iii) are the primary analytic content.

\smallskip
\noindent\textbf{Layer~1 (Assumption~B):}
$\norm{D\mathcal{T}_c^{(N)}}\leq C(A)\log c$.
Since $\log c\to\infty$, this is not a contraction.

\smallskip
\noindent\textbf{Layer~2 (Assumptions~B and B$'$):}
$\norm{D\mathcal{T}_c^{(N)}}\leq C(A)c^{-1/2}\log c\to 0$.

\smallskip
\noindent
Assumption~B$'$ is equivalent to
Conjecture~\ref{conj:phase} (linear WKB phase equidistribution)
modulo a Dirichlet-kernel argument, and is strictly weaker than
the contraction it implies.
The No-go Theorem shows the Layer~1 bound is optimal within the
class $|T_{ij}|\leq C/|i-j|$; no pointwise improvement can reach
Layer~2.
The near/far splitting $T=T^{\rm near}+T^{\rm far}$
(width $c^{1/6}$, relative bandwidth $c^{-1/6}\to 0$)
reduces the global Schur problem to a band-matrix of shrinking
relative bandwidth; under Conjecture~\ref{conj:phase},
$T^{\rm near}$ formally approaches a Toeplitz-type operator in
the scaling limit.
All remaining gaps are precisely isolated as Open Problems.
\end{abstract}

\section{Introduction and Context}
%% ============================================================

\subsection{Programme overview}

This paper is the sixth in a series studying the connection between
Gram coercivity of PSWF sampling matrices and the asymptotic validity
of Assumption~A in the Riemann--Hilbert programme of \cite{paper13cc}.
The central object is the spectral density
\[
  \rho(s;\,g_c)
  = \frac{1}{\pi}\Im\frac{d}{ds}\log\det(I-s\Kc),
\]
where $\Kc$ is the finite-rank sinc operator on $(-1,1)$ with
bandwidth $c>0$, and $g_c$ is the phase function in the RHP
conjugation. Assumption~A posits the existence and regularity of~$g_c$.

\subsection{The nonlinear operator $\mathcal{T}_c$}

The phase $g_c$ satisfies a fixed-point equation
\begin{equation}\label{eq:fixedpoint}
  g=\Tc(g),\qquad
  (\Tc h)(t)
  =\Hil\!\left[-c\int_t^1\arccos(s)\,\rho(s;\,g_h)\,ds\right](t),
\end{equation}
where $\Hil$ is the finite Hilbert transform on $(-1,1)$.
Proving $\|D\Tc^{(N)}\|<1$ yields Assumption~A via the Banach
fixed-point theorem.

\subsection{Why global $L^2$-contraction fails}

A direct computation gives
$\norm{D\Tc}_{L^2\to L^2}\gtrsim c^{8/3}$
(HS growth $\norm{R_s}_{\mathfrak{S}_2}^2\sim c^{5/3}$ times
Widom prefactor $c$).
Spectral stability lives on finitely many modes.

\subsection{Main results}

\begin{theorem}[Galerkin norm estimate]\label{thm:main}
Let $N=\lfloor Ac^{1/3}\rfloor$, $P_N$ the orthogonal projection
onto the first $N$ PSWF modes,
$\mathcal{T}_c^{(N)}:=P_N\circ\Tc\circ P_N$.

\medskip\noindent\textbf{(a) Norm bound}
(Assumption~\ref{ass:B} only):
\[
  \norm{D\mathcal{T}_c^{(N)}}_{\ell^2\to\ell^2}\leq C(A)\log c.
\]
Since $\log c\to\infty$, this is \textbf{not} a contraction.

\medskip\noindent\textbf{(b) Contraction}
(Assumptions~\ref{ass:B} and~\ref{ass:Bprime}):
\[
  \norm{D\mathcal{T}_c^{(N)}}_{\ell^2\to\ell^2}
  \leq C(A)\,c^{-1/2}\log c\xrightarrow{c\to\infty}0.
\]
\end{theorem}

\begin{theorem}[No-go theorem]\label{thm:nogo}
Let $T$ be $N\times N$ with $T_{ii}=0$ and
$|T_{ij}|\leq\kappa/|i-j|$ for $i\neq j$.
Then:
\[
  \norm{T}_{\ell^2\to\ell^2}
  \geq\frac{1}{2}\sup_i\sum_{j\neq i}|T_{ij}|
  \geq\frac{\kappa}{2}H_{N-1}
  \sim\frac{\kappa}{6}\log c.
\]
The $\log c$ growth in Theorem~\ref{thm:main}(a) is optimal in
the class $|T_{ij}|\leq\kappa/|i-j|$.
No argument based solely on pointwise bounds can establish
$\norm{DT}<1$.
\end{theorem}

\begin{proof}
Classical Schur inequality for quasi-symmetric operators
(Young's bilinear inequality):
$\norm{T}_{\ell^2}\geq\tfrac{1}{2}\sup_i\sum_{j\neq i}|T_{ij}|$.
Taking any interior row $i$: $\sum_{j\neq i}|T_{ij}|\geq\kappa H_{N-1}$.
For $N=\lfloor Ac^{1/3}\rfloor$: $H_{N-1}\geq\tfrac{1}{3}\log c+O(1)$.
\end{proof}

\begin{remark}[Schur test: norm bound, not contraction]
\label{rem:schur_honest}
$H_N\to\infty$; Schur gives $O(\log c)$, optimal by
Theorem~\ref{thm:nogo}.
The sole paths to contraction are oscillatory cancellation
(Assumption~\ref{ass:Bprime}) or a change of norm
(Open Problem~\ref{op:weightnorm}).
\end{remark}

\begin{remark}[Structural decomposition]\label{rem:decomposition}
The contraction problem splits into two independent subproblems:
\begin{enumerate}[nosep,label=(\arabic*)]
  \item \textbf{Pointwise decay} (largely resolved):
        $|T_{ij}|\leq C/|i-j|$;
        mechanism fully identified (IBP $+$ Airy $+$ SL);
        Assumption~B quantifies the remaining gap.
  \item \textbf{Summation cancellation} (open):
        control $\sum_j T_{ij}$ \emph{with signs};
        a Calder\'{o}n--Zygmund / Fourier problem.
\end{enumerate}
Theorem~\ref{thm:nogo}: no improvement to~(1) alone resolves~(2).
\end{remark}

\subsection{Proof architecture}
\label{sec:architecture}

The logical structure has three layers.

\medskip\noindent\textbf{Layer~0 (unconditional).}
Four results require no assumptions beyond the Galerkin setup:
\begin{enumerate}[nosep,label=(\arabic*)]
  \item $T_{ii}=0$
        (commutator structure, \S\ref{sec:matrix});
  \item $|I_{kl}|\leq C_1c^{-1/2}/|k-l|$
        (IBP $+$ Wronskian $+$ SL, Lemma~\ref{lem:RL});
  \item bulk oscillation barrier localised to
        $|k-l|\lesssim c^{1/6}$
        (Proposition~\ref{prop:bulk_barrier});
  \item $\|T\|\geq\tfrac{\kappa}{2}\log c$
        in the pointwise-bound class
        (Theorem~\ref{thm:nogo}).
\end{enumerate}
Items~(1)--(3) are the primary analytic content of this paper.
Item~(4) is a sharp optimality statement that eliminates
an entire family of approaches.

\medskip\noindent\textbf{Layer~1 (conditional on Assumption~B).}
Under the prefactor bound $P_{kl}\leq C_2c^{1/2}$:
\[
  |T_{ij}|\leq C_1C_2/|i-j|
  \quad\Longrightarrow\quad
  \norm{T}_{\ell^2}\leq C(A)\log c.
\]
Assumption~B quantifies a phase-sensitive cancellation at the
turning point $s_*=1-c^{-2/3}$
(Remark~\ref{rem:assB_heuristic});
it is open (Open Problem~\ref{op:prefactor}).

\medskip\noindent\textbf{Layer~2 (conditional on Assumption~B$'$).}
Under oscillatory cancellation in Schur row sums:
$\norm{T}\leq Cc^{-1/2}\log c\to 0$.
By Remark~\ref{rem:Bprime_chain},
Assumption~B$'$ is implied by Conjecture~\ref{conj:phase}
(linear WKB phase equidistribution) via a Dirichlet-kernel argument,
and is \emph{strictly weaker} than the contraction it implies.

\medskip\noindent
The paper's primary contribution is Layer~0.
Layers~1--2 identify precisely what remains to be proved
and why it cannot be obtained by further pointwise analysis.

\subsection{Comparison with standard PSWF theory}
\label{sec:new}

Standard PSWF asymptotic theory
(Osipov--Rokhlin--Xiao \cite{OsipovRokhlinXiao2013},
Widom \cite{Widom64}, Slepian--Pollak \cite{SP61})
provides pointwise bounds on $\pswf{k}$, eigenvalue asymptotics,
and WKB approximations in bulk and transition zones.
The present paper contributes three results not contained in the
standard theory:

\begin{enumerate}[nosep,label=(\arabic*)]
  \item \textbf{Barrier localisation}
        (Proposition~\ref{prop:bulk_barrier}):
        the second-IBP obstruction for $I_{kl}$ is confined to
        $|k-l|\lesssim c^{1/6}$, a band of relative width
        $c^{-1/6}\to 0$.
        This does not follow from standard pointwise bounds.

  \item \textbf{Schur optimality}
        (Theorem~\ref{thm:nogo}):
        the $\log c$ growth is sharp within
        $\{|T_{ij}|\leq C/|i-j|\}$, eliminating an entire
        family of potential improvements.

  \item \textbf{Near/far decomposition}
        (Remark~\ref{rem:routes}(iii),
        Open Problem~\ref{op:nearfar}):
        the contraction problem reduces to a band matrix of
        shrinking relative bandwidth $c^{-1/6}\to 0$,
        a regime inaccessible to global Schur estimates and
        suggesting a Toeplitz-type spectral limit under
        Conjecture~\ref{conj:phase}.
\end{enumerate}

Items~(1)--(3) depend only on the Galerkin structure and are
independent of Assumptions~B and~B$'$.

\subsection{The weighted space $X_c$}

\begin{definition}\label{def:Xc}
$w_c(s):=(\lam{n(s)}-\lam{n(s)+1})^2$;
$\Xc:=\{h\in L^2(-1,1):\int h=0\}$,
$\norm{h}_{\Xc}^2:=\int|h|^2w_c\,ds$.
\end{definition}

Regimes: bulk $w_c\asymp c^{-2}(1-s^2)^{-1}$;
transition $w_c\asymp c^{-2/3}$; outer negligible.

\begin{proposition}[$A_2$ control]\label{prop:A2}
$[w_c]_{A_2}\leq C$ uniformly in $c\geq c_0$.
Hence $\norm{\Hil}_{\Xc\to\Xc}=O(1)$
(Coifman--Fefferman 1974).
\end{proposition}

\subsection{Galerkin projection}

\begin{definition}\label{def:Galerkin}
$X_c^{(N)}:=\operatorname{span}\{\pswf{0},\ldots,\pswf{N-1}\}$,
$P_N$ orthogonal projection.
By Gram coercivity (Paper~I):
$\sigma_{\min}(G^{(N)})\geq\alpha_N\sim c^{-1/2}$.
\end{definition}

\subsection{Matrix entries and commutator structure}

Dominant contribution:
\begin{equation}\label{eq:Tij_simple}
  T_{ij}\sim c\cdot
  \frac{|\lam{j}-\lam{i}|}{(1-\lam{i})(1-\lam{j})}
  \cdot I_{ij},\qquad
  I_{ij}:=\int_{-1}^1\arccos(s)\,\pswf{i}(s)\pswf{j}(s)\,ds.
\end{equation}
Numerator $(\lam{j}-\lam{i})$ vanishes for $i=j$:
\begin{equation}\label{eq:diagzero}
  T_{ii}=0\quad\text{for all }i.
\end{equation}

\subsection{Airy approximation and Wronskian bound}

\begin{lemma}[Airy approximation]\label{lem:Airy}
$|\pswf{k}(s)|\lesssim c^{1/6}$ uniformly in
$\mathcal{Z}_{\rm tz}(k)$.
\emph{(Olver~\cite{Olver1974}; ORX~\cite{OsipovRokhlinXiao2013})}
\end{lemma}

\begin{corollary}[Wronskian bound]\label{cor:Wronskian}
For $k\neq l$, $k,l\leq N$,
$W_k^l(s):=(1-s^2)[\pswf{k}'\pswf{l}-\pswf{k}\pswf{l}']$:
\[
  \int_{-1}^1|W_k^l(s)|\,ds\lesssim\frac{c^{-1/6}}{|k-l|}.
\]
\end{corollary}

\begin{proof}
TZ: $J_{\rm tz}\lesssim c^{1/3}\cdot c^{-2/3}=c^{-1/3}$.
Bulk: $J_{\rm bulk}\lesssim c^{1/2}/(c^{2/3}|k-l|)=c^{-1/6}/|k-l|$.
For $|k-l|\geq 1$, $c>1$: $c^{-1/6}>c^{-1/3}$, bulk dominates.
\end{proof}

\subsection{SL moment and primitive bounds}

\begin{proposition}\label{prop:moment}
For $k\neq l$, $k,l\leq N$:
\begin{enumerate}[label=(\alph*)]
  \item Parity: $k\equiv l\pmod{2}$
        $\Rightarrow\mu_{kl}:=\int s\pswf{k}\pswf{l}\,ds=0$.
  \item Moment: $|\mu_{kl}|\lesssim c^{-7/6}/|k-l|^2$.
  \item Wronskian identity in $\mathcal{Z}_{\rm bulk}$:
        \[
          F_{kl}(s):=\int_s^1\pswf{k}\pswf{l}\,dt
          =\frac{W_k^l(s)}{\chi_k-\chi_l},
          \quad
          |F_{kl}(s)|\lesssim
          \frac{c^{-1/2}(1-s^2)^{1/2}}{|k-l|}.
        \]
        $W_k^l(1)=0$ since $(1-s^2)|_{s=1}=0$.
\end{enumerate}
\end{proposition}

\begin{proof}
(b) SL: $(\chi_k-\chi_l)\mu_{kl}=\int W_k^l\,ds
\lesssim c^{-1/6}/|k-l|$;
divide by $\chi_k-\chi_l\asymp 2c|k-l|$.
(c) Integrate $(\chi_k-\chi_l)\pswf{k}\pswf{l}=-(d/ds)W_k^l$
from $s$ to $1$;
WKB: $|W_k^l|\lesssim c^{1/2}(1-s^2)^{1/2}$ in bulk.
\end{proof}

\subsection{Off-diagonal integral bounds}

\begin{lemma}[Zone decomposition]\label{lem:sublD}
For $k\neq l$:
$|I_{kl}^{\rm tz}|\lesssim c^{-2/3}/|k-l|$,
$|I_{kl}^{\rm bulk}|\lesssim c^{-1/2}/|k-l|$,
$|I_{kl}|\lesssim c^{-1/2}/|k-l|$.
\end{lemma}

\begin{proof}
IBP via $(\arccos)'=-(1-s^2)^{-1/2}$; boundary terms vanish
($F_{kl}(1)=0$, $F_{kl}(-1)=0$).
TZ: $|F_{kl}|\lesssim c^{-1/3}/|k-l|$, $(1-s^2)^{-1/2}\lesssim c^{1/3}$,
measure $\sim c^{-2/3}$: total $c^{-2/3}/|k-l|$.
Bulk: $|F_{kl}|\lesssim c^{-1/2}(1-s^2)^{1/2}/|k-l|$,
$(1-s^2)^{-1/2}$ cancels: total $2c^{-1/2}/|k-l|$.
\end{proof}

\begin{lemma}[IBP integral bound]\label{lem:RL}
$|I_{kl}|\leq C_1c^{-1/2}/|k-l|$
for $k\neq l$, $k,l\leq N$.
\end{lemma}

\begin{lemma}[Zone Schur contributions]\label{lem:zone_schur}
Under Assumption~\ref{ass:B}, for any fixed row $i$:
\begin{align}
  \sum_{j\in\mathrm{TZ},j\neq i}|T_{ij}|
  &\leq\tfrac{1}{3}C_2\,c^{-1/6}\log c\to 0,
  \label{eq:zone_tz}\\
  \sum_{j\in\mathrm{bulk},j\neq i}|T_{ij}|
  &\leq C_1C_2(1+\log N)\sim C_1C_2\log c.
  \label{eq:zone_bulk}
\end{align}
\end{lemma}

\begin{proof}
\eqref{eq:zone_tz}: $|T_{ij}|\leq C_2c^{1/2}\cdot c^{-2/3}/|i-j|$
for TZ pairs; $N_{\rm tz}\sim c^{1/3}$ terms:
sum $\leq C_2c^{-1/6}\cdot\tfrac{1}{3}\log c\to 0$.
\eqref{eq:zone_bulk}: Proposition~\ref{prop:Tij} and harmonic sum.
\end{proof}

\begin{remark}[TZ: asymptotically irrelevant]
\label{rem:tz_irrelevant}
The TZ contribution $O(c^{-1/6}\log c)\to 0$.
For large $c$, the Schur row sum is determined entirely by the bulk.
\end{remark}

\subsection{Bulk oscillation barrier and near-diagonal localisation}
\label{sec:bulk_barrier}

\begin{proposition}[Bulk oscillation barrier]\label{prop:bulk_barrier}
Let $k\neq l$, $k,l\leq N$, WKB in $\mathcal{Z}_{\rm bulk}$:
$\pswf{k}(s)\sim c^{-1/4}(1-s^2)^{-1/4}\cos(\theta_k(s))$.

\medskip\noindent\textbf{(i)} First IBP: $|I_{kl}^{\rm bulk}|
\lesssim c^{-1/2}/|k-l|$.

\medskip\noindent\textbf{(ii)} WKB product:
difference-phase frequency
$\partial_s(\theta_k-\theta_l)\asymp|k-l|(1-s^2)^{-1/2}$.

\medskip\noindent\textbf{(iii)} Second IBP:
$I_{kl}^{\rm bulk}
=\underbrace{\rm interior}_{O(c^{-2/3}/|k-l|)}
+\underbrace{\rm boundary\ at\ }s_*=1-c^{-2/3}_{\rm dominant}$.

\medskip\noindent\textbf{(iv)} At $s_*$:
$|F_{kl}(s_*)|\lesssim c^{-5/6}/|k-l|$,
$|\partial_s(\theta_k-\theta_l)|_{s_*}|\asymp|k-l|c^{1/3}$.
\[
  |\text{boundary term}|
  \asymp\frac{c^{-1/2}}{|k-l|^2}.
  \qquad
  \text{Ratio to target }c^{-2/3}/|k-l|:
  \quad
  \frac{c^{1/6}}{|k-l|}.
\]
\begin{itemize}[nosep]
  \item $|k-l|\lesssim c^{1/6}$:
        ratio $\gtrsim 1$ — obstruction active.
  \item $|k-l|\gg c^{1/6}$:
        ratio $\to 0$ — second IBP succeeds.
\end{itemize}

\medskip\noindent
\textbf{The bulk oscillation obstruction is localised to
$|k-l|\lesssim c^{1/6}$.}

\medskip\noindent\textbf{Equivalence.}
$|I_{kl}^{\rm bulk}|\lesssim c^{-2/3}/|k-l|$ for all
$|k-l|\leq N$
$\;\Longleftrightarrow\;$
$|\text{boundary term at }s_*|
\lesssim c^{-2/3}/|k-l|$
for $|k-l|\lesssim c^{1/6}$.
\end{proposition}

\begin{remark}[Near-diagonal localisation]\label{rem:barrier_geometry}
Width $c^{1/6}$, dimension $N\sim c^{1/3}$,
relative bandwidth $c^{1/6}/N=c^{-1/6}\to 0$:
the obstruction region shrinks relative to the full matrix.
This is a Layer~0 result, unconditional.
\end{remark}

\subsection{Prefactor bounds}

\begin{lemma}[Prefactor]\label{lem:prefactor}
For $k\neq l$, $k,l\leq N$:
$P_{kl}:=c\cdot|\lam{l}-\lam{k}|/[(1-\lam{k})(1-\lam{l})]$.
Raw Widom: $P_{kl}\lesssim|k-l|\,c^{7/3}$.
Assumption~\ref{ass:B}: $P_{kl}\leq C_2c^{1/2}$.
\end{lemma}

\begin{remark}[Assumption~B: phase-sensitive gap]
\label{rem:assB_heuristic}
In the transition regime:
$|\lam{k}-\lam{l}|\asymp c^{-2/3}|k-l|$,
$(1-\lam{k})(1-\lam{l})\asymp c^{-4/3}$, giving
$P_{kl}\asymp c^{5/3}|k-l|$
(reduction from $c^{7/3}$ by $c^{2/3}$).
The further reduction to $c^{1/2}$ (factor $c^{7/6}$) requires
\emph{phase-sensitive} Airy--WKB matching — cancellation between
oscillatory terms at the turning point, not merely amplitude bounds.
This is the precise analytic content of Assumption~B.
\end{remark}

\begin{assumption}[Prefactor bound]\label{ass:B}
For all $k\neq l$, $k,l\leq N\sim c^{1/3}$:
\[
  c\cdot\frac{|\lam{l}-\lam{k}|}{(1-\lam{k})(1-\lam{l})}
  \leq C_2c^{1/2}.
\]
\end{assumption}

\begin{assumption}[Schur-summability]\label{ass:Bprime}
\[
  \sup_{i\leq N}\sum_{j\neq i}P_{ij}\cdot|I_{ij}|
  \leq C\,c^{-1/2}\log c.
\]
Not a consequence of Assumption~\ref{ass:B};
requires oscillatory cancellation.
\end{assumption}

\begin{remark}[Implication chain for B$'$]\label{rem:Bprime_chain}
The following implications hold and none reverses in general:
\[
  \text{Conjecture~\ref{conj:phase}}
  \;\Rightarrow\;
  \text{Assumption~B$'$}
  \;\Rightarrow\;
  \norm{T}\leq Cc^{-1/2}\log c
  \;\Rightarrow\;
  \text{Assumption~A}.
\]
Conjecture~\ref{conj:phase} is a statement about WKB phases at the
single point $s_*=1-c^{-2/3}$, computable in principle from
Airy--WKB connection formulae independently of the contraction
problem.
It is therefore \emph{strictly weaker} than the contraction
conclusion.
The apparent circularity (cancellation $\to$ phases $\to$
cancellation) dissolves when one observes that the phase condition
is a one-point asymptotic statement, while contraction is a global
operator-norm statement.
\end{remark}

\subsection{Matrix entry bound and proof of Theorem~\ref{thm:main}}

\begin{proposition}[Matrix entry bound]\label{prop:Tij}
Under Assumption~\ref{ass:B}, for $i\neq j$:
$|T_{ij}|\leq C_1C_2/|i-j|$.
\end{proposition}

\begin{proof}
$P_{ij}\cdot|I_{ij}|\leq C_2c^{1/2}\cdot C_1c^{-1/2}/|i-j|
=C_1C_2/|i-j|$.
\end{proof}

\begin{remark}[$c$-power balance]\label{rem:balance}
$c^{+1/2}_{\rm Ass.\,B}\times c^{-1/2}_{\rm Lem.\,\ref{lem:RL}}
=c^0$.
Under Assumption~B the norm grows like $\log c$, not $c^{8/3}$.
\end{remark}

\begin{proof}[Proof of Theorem~\ref{thm:main}]
$T_{ii}=0$ by~\eqref{eq:diagzero}.

\noindent\textbf{(a).}
Schur test $+$ Proposition~\ref{prop:Tij}:
$\norm{T}_{\ell^2}\leq\sup_i\sum_{j\neq i}|T_{ij}|
\leq C_1C_2H_N=:C(A)\log c$.

\noindent\textbf{(b).}
Assumption~\ref{ass:Bprime}:
$\sup_i\sum_{j\neq i}P_{ij}|I_{ij}|\leq Cc^{-1/2}\log c\to 0$.
\end{proof}

\section{Dyadic Decomposition and Routes to Contraction}
\label{sec:dyadic}
%% ============================================================

\subsection{Dyadic block decomposition}

For fixed row $i$:
$\mathcal{B}_m:=\{j:2^m\leq|i-j|<2^{m+1}\}$,
$m=0,\ldots,M=\lfloor\log_2 N\rfloor$.

\begin{proposition}[Block-wise and total bounds]
\label{prop:dyadic_block}
Under Assumption~\ref{ass:B}:
$\sum_{j\in\mathcal{B}_m}|T_{ij}|\leq 2C_1C_2$ (each $m$);
$\sum_{j\neq i}|T_{ij}|\leq 2C_1C_2(M+1)\sim C(A)\log c$.
\end{proposition}

\subsection{Structural characterisation of Assumption~B$'$}

Define $S_m(i):=\sum_{j\in\mathcal{B}_m}T_{ij}$ (signed).
Contraction requires:
\begin{equation}\label{eq:cancellation_needed}
  \sum_{m=0}^{M}|S_m(i)|\leq Cc^{-1/2}\log c,
\end{equation}
i.e., $|S_m(i)|\ll 2C_1C_2$ uniformly — a
Calder\'{o}n--Zygmund-type condition, not a decay condition.

\begin{remark}[B$'$ $\Leftrightarrow$ signed dyadic cancellation]
\label{rem:Bprime_equiv}
Assumption~B$'$ is equivalent to~\eqref{eq:cancellation_needed}.
No pointwise improvement establishes it (Theorem~\ref{thm:nogo}).
By Remark~\ref{rem:Bprime_chain}, it is implied by
Conjecture~\ref{conj:phase} via Dirichlet-kernel cancellation.
\end{remark}

\subsection{Routes to contraction}

\begin{remark}[Four routes to~\eqref{eq:cancellation_needed}]
\label{rem:routes}
\begin{enumerate}[nosep,label=(\roman*)]
  \item \textbf{Airy--WKB matching at $s_*$}
        (OP~\ref{op:bulk_refined}):
        cancellation of the boundary term for
        $|k-l|\lesssim c^{1/6}$;
        rate $c^{-1/6}\log c$ under Assumption~B alone.
        Most immediate analytic target;
        feeds directly into route~(iii).

  \item \textbf{Sign oscillation in $j$}:
        if $T_{ij}$ alternates sign at period $\sim 1$,
        $|S_m(i)|=O(1/2^m)$, $\sum|S_m|=O(1)$.
        Equivalent to route~(iv) under
        Conjecture~\ref{conj:phase}.

  \item \textbf{Near/far splitting}
        (OP~\ref{op:nearfar}):
        $T=T^{\rm near}+T^{\rm far}$,
        $T^{\rm near}_{ij}:=T_{ij}\cdot\mathbf{1}_{|i-j|\leq c^{1/6}}$.
        Far part: free of boundary-term obstruction.
        Near part: band matrix, width $c^{1/6}$,
        dimension $N\sim c^{1/3}$,
        relative bandwidth $c^{-1/6}\to 0$.
        Reduces global Schur to finite-band spectral problem.
        Route~(i) controls $T^{\rm near}$ directly.

  \item \textbf{Weighted norm}
        (OP~\ref{op:weightnorm}):
        Section~\ref{sec:spectral_weights} shows all natural
        candidates fail.
\end{enumerate}
Routes~(i) and~(iii) are \emph{complementary}:
Airy--WKB matching controls precisely the near-diagonal block.
Routes~(ii) and~(iii) converge under Conjecture~\ref{conj:phase}:
linear phase equidistribution implies both sign oscillation
(route~(ii)) and the Toeplitz spectral limit (route~(iii)).
\end{remark}

\subsection{Phase structure and Fourier-multiplier strategy}
\label{sec:fourier}

\begin{conjecture}[Linear phase equidistribution]\label{conj:phase}
The WKB phases $\Theta_k^{(c)}$ at $s_*=1-c^{-2/3}$ satisfy,
for $k\leq N$:
\[
  \Theta_k^{(c)}=\alpha^{(c)}k+\beta^{(c)}+O(k^2/c),
\]
$\alpha^{(c)}\in(0,2\pi)$ bounded away from $0$ and $\pi$.
Implies Assumption~B$'$ via Dirichlet-kernel cancellation
(Remark~\ref{rem:Bprime_chain}).
\end{conjecture}

\begin{remark}[Fourier-multiplier route]\label{rem:fourier_route}
If $T_{ij}=f(i-j)/(i-j)$ for bounded oscillatory $f$,
then $\norm{T}_{\ell^2}=\|\widehat{f/|\cdot|}\|_{\ell^\infty}=O(1)$
by $\ell^2$-boundedness of the discrete Hilbert transform.
Approximate translation-invariance $T_{ij}\approx g(i-j)$
follows from Conjecture~\ref{conj:phase}.
\end{remark}

\begin{remark}[Toeplitz effective model for $T^{\rm near}$]
\label{rem:toeplitz}
Under Conjecture~\ref{conj:phase}, $T_{ij}\approx g(i-j)$
holds in $|i-j|\leq c^{1/6}$.
In this regime $T^{\rm near}$ formally approaches a Toeplitz
band operator of symbol $g$ in the WKB scaling limit.
By the Szeg\H{o}--Toeplitz limit theorem:
$\norm{T^{\rm near}}_{\ell^2}\to\norm{g}_{L^\infty(\mathbb{T})}$.
Whether $\norm{g}_{L^\infty}<1$ is then a concrete spectral question
about a fixed function — not an asymptotic one.
This reduction is \emph{conditional} on Conjecture~\ref{conj:phase}
and is a Toeplitz-type effective model in the scaling limit,
not a claim of exact Toeplitz structure.
Conjecture~\ref{conj:phase} thus unifies routes~(ii)--(iii):
linear phase equidistribution implies both the Fourier-multiplier
bound and the Toeplitz spectral picture simultaneously.
\end{remark}

\subsection{Why spectral and index weights fail}
\label{sec:spectral_weights}

\begin{proposition}\label{prop:spectral_weight_fail}
For $k\leq N$: $1-\lambda_k\asymp c^{-2/3}$ uniformly (transition
plateau); spectral weights $w_k=(1-\lambda_k)^\alpha$ satisfy
$w_j/w_i=1+O(c^{-1/3})$ and do not modify the Schur kernel.
\end{proposition}

\begin{remark}
Power weights $(1+k)^\beta$ and exponential weights $e^{\gamma k/N}$
both give row sum $\sim\log N$ — $\sum 1/d$ diverges regardless.
No weight of spectral, power, or exponential type makes
$\sup_i\sum_j(w_j/w_i)/|i-j|=o(\log N)$.
\end{remark}

\section{Open Problems}
\label{sec:open}
%% ============================================================

\begin{openproblem}[Prefactor and cancellation]\label{op:prefactor}
\begin{enumerate}[nosep]
  \item Prove Assumption~\ref{ass:B}: $P_{kl}\leq C_2c^{1/2}$.
        Remaining gap: $c^{7/6}$ beyond transition asymptotics;
        requires phase-sensitive Airy--WKB matching
        (Remark~\ref{rem:assB_heuristic}).
  \item Prove Assumption~\ref{ass:Bprime}
        $\Leftrightarrow$ signed dyadic cancellation
        (Remark~\ref{rem:Bprime_equiv});
        implied by Conjecture~\ref{conj:phase}
        (Remark~\ref{rem:Bprime_chain}).
\end{enumerate}
\end{openproblem}

\begin{openproblem}[Airy--WKB matching]\label{op:bulk_refined}
Establish cancellation of the boundary term at $s_*$ for
$|k-l|\lesssim c^{1/6}$ via Airy--WKB connection formulae.
Set $s=1-c^{-2/3}x$: difference phase linearises,
boundary term reduces to explicit $\Ai\cdot\Ai$ integral.
If cancellation holds: unconditional rate $c^{-1/6}\log c$
under Assumption~B; simultaneously resolves OP~\ref{op:nearfar}(2).
\end{openproblem}

\begin{openproblem}[Near/far splitting]\label{op:nearfar}
\begin{enumerate}[nosep]
  \item Unconditional estimates for $\norm{T^{\rm far}}_{\ell^2}$
        (no boundary-term obstruction for $|i-j|\gg c^{1/6}$).
  \item Control $\norm{T^{\rm near}}_{\ell^2}$:
        band matrix, width $c^{1/6}$, dimension $c^{1/3}$,
        relative bandwidth $c^{-1/6}\to 0$.
        Under Conjecture~\ref{conj:phase}: Toeplitz effective model
        reduces to $\norm{g}_{L^\infty(\mathbb{T})}<1$?
        (Remark~\ref{rem:toeplitz}).
\end{enumerate}
\end{openproblem}

\begin{openproblem}[Weighted norm]\label{op:weightnorm}
Find $c$-dependent $\norm{\cdot}_w$ with
$\norm{D\mathcal{T}_c^{(N)}}_w<1$ unconditionally.
All natural candidates fail (\S\ref{sec:spectral_weights}).
A successful weight must satisfy an asymmetric two-weight condition:
$\{u_k\},\{v_k\}>0$ with
$\sum_j v_j|T_{ij}|\leq u_i$ and
$\sum_j u_j^{-1}|T_{ij}|\leq v_i^{-1}$ for all $i$.
\end{openproblem}

\begin{openproblem}[Optimal Galerkin dimension]\label{op:Nopt}
Determine maximal $N(c)$ with Schur bound $<1$.
Conjecture: $N_{\rm opt}\sim c^{1/2}$.
\end{openproblem}

\section{Summary: Layer Structure and What Is New}
\label{sec:summary}
%% ============================================================

\subsection*{Layer~0 (unconditional, primary content)}
\begin{itemize}[nosep]
  \item $T_{ii}=0$; $|I_{kl}|\leq C_1c^{-1/2}/|k-l|$
  \item Barrier localised to $|k-l|\lesssim c^{1/6}$
        (Prop.~\ref{prop:bulk_barrier})
  \item $\norm{T}\geq\tfrac{\kappa}{2}\log c$ optimal
        (Thm~\ref{thm:nogo})
  \item TZ contribution $o(1)$, bulk dominates
  \item Weights (spectral, power, exponential) all fail
\end{itemize}

\subsection*{Layer~1 (Assumption~B)}
\begin{itemize}[nosep]
  \item $|T_{ij}|\leq C_1C_2/|i-j|$;
        $\norm{T}\leq C\log c$
\end{itemize}

\subsection*{Layer~2 (Assumption~B$'$
$\Leftarrow$ Conjecture~\ref{conj:phase})}
\begin{itemize}[nosep]
  \item $\norm{T}\leq Cc^{-1/2}\log c\to 0$;
        conditional fixed point (Cor.~\ref{cor:fixedpoint})
\end{itemize}

\subsection*{Four routes to unconditional contraction}
\begin{enumerate}[nosep,label=(\roman*)]
  \item Airy--WKB at $s_*$ (most immediate; feeds (iii))
  \item Sign oscillation (equivalent to (iv) under
        Conj.~\ref{conj:phase})
  \item Near/far splitting (band spectral problem,
        Toeplitz limit under Conj.~\ref{conj:phase};
        complementary to (i))
  \item Two-weight norm (constructive path unknown)
\end{enumerate}
Routes~(i)+(iii) are complementary.
Routes~(ii)+(iii) converge under Conjecture~\ref{conj:phase}.

\subsection*{What is genuinely new vs.\ standard PSWF theory}
Items independent of Assumptions~B and~B$'$:
\textbf{(1)} barrier localisation to relative width $c^{-1/6}\to 0$;
\textbf{(2)} Schur optimality ($\log c$ sharp);
\textbf{(3)} near/far decomposition as band spectral problem
with shrinking relative bandwidth.
