\begin{abstract}
We study the nonlinear phase operator $\mathcal{T}_c$ arising in the
Riemann--Hilbert formulation of prolate spheroidal wave function (PSWF)
asymptotics.
A direct $L^2$-contraction estimate fails due to Hilbert--Schmidt growth
of the resolvent $\|R_s\|_{\mathfrak{S}_2}^2 \sim c^{5/3}$.
After Galerkin projection onto the leading $N \sim c^{1/3}$ PSWF modes,
we establish:
\[
  \norm{D\mathcal{T}_c^{(N)}}_{\ell^2 \to \ell^2}
  \leq C(A)\,\log c
  \quad\text{(under Assumption~B, unconditional in }c\text{-rate)},
\]
and, conditionally,
\[
  \norm{D\mathcal{T}_c^{(N)}}_{\ell^2 \to \ell^2}
  \leq C(A)\, c^{-1/2}\log c \to 0
  \quad\text{(under Assumptions~B and B$'$)}.
\]
\textbf{The first bound is a norm estimate, not a contraction:}
$\log c \to \infty$, so it does not imply $\|D\mathcal{T}_c^{(N)}\| < 1$
for large $c$.
Only the second bound, conditional on Assumption~B$'$
(Schur-summability hypothesis, Section~\ref{sec:norm}),
gives a genuine contraction for $c$ sufficiently large.
All $c$-dependencies are tracked explicitly; no factor is absorbed
without justification.
The unconditional structural result rests on three independently proved steps:
(i)~diagonal vanishing $T_{ii} = 0$ from the commutator structure;
(ii)~IBP combined with the SL Wronskian identity:
    $|I_{kl}| \leq C c^{-1/2}/|k-l|$ (Lemma~\ref{lem:RL});
(iii)~pointwise matrix bound $|T_{ij}| \leq C_1 C_2/|i-j|$
    under Assumption~B, and Schur harmonic sum $\sum 1/k \sim \log N$.
The transition zone contributes $o(1)$ to the Schur row sum and is
asymptotically irrelevant; the bulk zone dominates (Lemma~\ref{lem:zone_schur}).
The contraction problem decomposes into two logically independent subproblems:
pointwise decay (analytic, largely resolved) and summation cancellation
(oscillatory, open).
No progress on the former alone can resolve the latter.
Section~\ref{sec:dyadic} establishes a No-go Theorem (Theorem~\ref{thm:nogo}),
proves that spectral and index weights fail structurally, and identifies
three routes to contraction: Airy--WKB matching, phase equidistribution
(Conjecture~\ref{conj:phase}), and a near/far matrix splitting
exploiting the localisation of the bulk oscillation barrier to the
near-diagonal regime $|i-j| \lesssim c^{1/6}$
(Proposition~\ref{prop:bulk_barrier}).
The remaining gaps are precisely identified as Open Problems~\ref{op:prefactor}
and~\ref{op:weightnorm}.
\end{abstract}

\section{Introduction and Context}
%% ============================================================

\subsection{Programme overview}

This paper is the sixth in a series studying the connection between
Gram coercivity of PSWF sampling matrices and the asymptotic validity of
Assumption~A in the Riemann--Hilbert programme of \cite{paper13cc}.
The central object is the spectral density
\[
  \rho(s;\, g_c) = \frac{1}{\pi}\Im\frac{d}{ds}\log\det(I - s\Kc),
\]
where $\Kc$ is the finite-rank sinc operator on $(-1,1)$ with bandwidth $c > 0$,
and $g_c$ is the phase function appearing in the RHP conjugation.
Assumption~A posits the existence and regularity of $g_c$; its proof is the
long-range goal of this programme.

\subsection{The nonlinear operator $\mathcal{T}_c$}

Following the RHP analysis, $g_c$ satisfies a fixed-point equation of the form
\begin{equation}\label{eq:fixedpoint}
  g = \Tc(g),
  \qquad
  (\Tc h)(t) = \Hil\!\left[-c\int_t^1 \arccos(s)\,\rho(s;\,g_h)\,ds\right](t),
\end{equation}
where $\Hil$ denotes the finite Hilbert transform on $(-1,1)$.
Proving $\|D\Tc^{(N)}\| < 1$ in some Banach space would yield
Assumption~A via the Banach fixed-point theorem.
The purpose of this paper is to identify precisely what is proved,
what is assumed, and what remains open.

\subsection{Why global $L^2$-contraction fails}

A direct computation (Section~\ref{sec:resolvent}) shows that the linearised
operator $D\Tc$ satisfies
\[
  \norm{D\Tc}_{L^2 \to L^2} \gtrsim c^{8/3},
\]
due to growth of $\norm{R_s}_{\mathfrak{S}_2}^2 \sim c^{5/3}$ combined with
the Widom prefactor $c$.
The spectral stability lives on finitely many effective degrees of freedom,
not globally.

\subsection{Main results}

\begin{theorem}[Galerkin norm estimate]\label{thm:main}
Let $N = \lfloor A c^{1/3}\rfloor$ for a constant $A > 0$, and let
$P_N : L^2(-1,1) \to X_c^{(N)}$ denote the orthogonal projection onto
the first $N$ PSWF modes. Define
$\mathcal{T}_c^{(N)} := P_N \circ \Tc \circ P_N$.

\medskip
\noindent\textbf{(a) Norm bound}
(under Assumption~\ref{ass:B} only):
\[
  \norm{D\mathcal{T}_c^{(N)}}_{\ell^2 \to \ell^2}
  \leq C(A)\,\log c.
\]
Since $\log c \to \infty$, part~(a) is \textbf{not} a contraction statement.

\medskip
\noindent\textbf{(b) Contraction}
(under both Assumption~\ref{ass:B} and Assumption~\ref{ass:Bprime}):
\[
  \norm{D\mathcal{T}_c^{(N)}}_{\ell^2 \to \ell^2}
  \leq C(A)\, c^{-1/2}\log c \;\xrightarrow{c\to\infty}\; 0.
\]
Hence $\mathcal{T}_c^{(N)}$ is a genuine contraction for $c \geq c_0$.
\end{theorem}

\begin{theorem}[No-go theorem for unweighted Schur contraction]%
\label{thm:nogo}
Let $T$ be any $N \times N$ matrix with $T_{ii} = 0$ and
$|T_{ij}| \leq \kappa/|i-j|$ for all $i \neq j$.
Then:
\[
  \norm{T}_{\ell^2 \to \ell^2}
  \;\geq\;
  \frac{1}{2}\sup_{i}\sum_{j \neq i}|T_{ij}|
  \;\geq\;
  \frac{\kappa}{2} H_{N-1}
  \;\sim\; \frac{\kappa}{2}\log N.
\]
For $N = \lfloor A c^{1/3} \rfloor$:
$\norm{D\mathcal{T}_c^{(N)}} \geq (\kappa/6)\log c + O(1)$.
The $\log c$ growth in Theorem~\ref{thm:main}(a) is optimal within the
class $|T_{ij}| \leq \kappa/|i-j|$; no argument based solely on pointwise
bounds can establish $\norm{DT} < 1$.
\end{theorem}

\begin{proof}
The lower bound $\norm{T}_{\ell^2} \geq \frac{1}{2}\sup_i \sum_{j\neq i}|T_{ij}|$
is the classical Schur inequality (see e.g.\ Young's inequality for
bilinear forms): for any unit vector $e_i$,
$\norm{Te_i}^2 \leq \norm{T}_{\ell^2}^2$, and summing the Schur condition
gives $\norm{T}_{\ell^2} \geq \frac{1}{2}\sup_i \sum_j |T_{ij}|$
when $T$ is symmetric (or quasi-symmetric, as here up to $O(c^{-1})$).
The harmonic lower bound follows by taking any interior row $i$,
for which all distances $d = 1,\ldots,N-1$ are realised:
$\sum_{j\neq i}|T_{ij}| \geq \kappa H_{N-1}$.
\end{proof}

\begin{remark}[The Schur test delivers a norm bound, not a contraction]%
\label{rem:schur_honest}
The harmonic series $H_N \to \infty$; the Schur test gives
$\|T\|_{\ell^2} = O(\log c)$, not $o(1)$.
By Theorem~\ref{thm:nogo}, this growth is optimal.
The \emph{sole} path to contraction within $\ell^2$ is
Assumption~\ref{ass:Bprime} (oscillatory cancellation)
or a change of norm (Open Problem~\ref{op:weightnorm}).
\end{remark}

\begin{remark}[Structural decomposition of the contraction problem]
\label{rem:decomposition}
The problem decomposes into two logically independent subproblems:
\begin{enumerate}[nosep,label=(\arabic*)]
  \item \textbf{Pointwise decay} (analytic, largely resolved):
        establish $|T_{ij}| \leq C/|i-j|$.
        This requires Assumption~B (open), but the mechanism
        (Wronskian IBP $+$ Airy $+$ SL identity) is fully identified.
  \item \textbf{Summation mechanism} (oscillatory, open):
        control $\sum_j T_{ij}$ \emph{with signs}.
        This is a Calder\'{o}n--Zygmund / Fourier problem,
        not an asymptotic one.
\end{enumerate}
Theorem~\ref{thm:nogo} shows that no progress on subproblem~(1) alone
can resolve subproblem~(2): the obstruction is structural, not quantitative.
\end{remark}

\subsection{The weighted space $X_c$}

\begin{definition}[Weighted space]\label{def:Xc}
Let $w_c : (-1,1) \to (0,\infty)$ be defined by
$w_c(s) := \bigl(\lam{n(s)} - \lam{n(s)+1}\bigr)^2$,
where $n(s)$ is the inverse eigenvalue counting function of $\Kc$. Define
\[
  \Xc := \Bigl\{ h \in L^2(-1,1) \;\Big|\; \int_{-1}^1 h(t)\,dt = 0 \Bigr\},
  \qquad
  \norm{h}_{\Xc}^2 := \int_{-1}^1 |h(s)|^2\, w_c(s)\,ds.
\]
\end{definition}

The weight $w_c$ has three asymptotic regimes:
\begin{itemize}[nosep]
  \item \textbf{Bulk} ($|n - 2c/\pi| \gg c^{1/3}$):
        $w_c(s) \asymp c^{-2}(1-s^2)^{-1}$.
  \item \textbf{Transition window} ($|n - 2c/\pi| \leq Ac^{1/3}$):
        $w_c(s) \asymp c^{-2/3}$.
  \item \textbf{Outer region}: exponential decay, negligible.
\end{itemize}

\subsection{Muckenhoupt $A_2$-property and Hilbert transform}

\begin{proposition}[$A_2$ control]\label{prop:A2}
The weight $w_c$ satisfies $[w_c]_{A_2} \leq C$ uniformly in $c \geq c_0$.
\end{proposition}

\begin{proof}[Proof sketch]
Bulk: reduces to classical Jacobi weight bound (Hunt--Muckenhoupt--Wheeden,
1973). Transition: $w_c \asymp c^{-2/3}$ constant, $[w_c]_{A_2}=1$.
Outer: excluded by cutoff.
\end{proof}

\begin{corollary}\label{cor:Hilbert}
$\norm{\Hil}_{\Xc \to \Xc} = O(1)$ uniformly in $c$
(Coifman--Fefferman, 1974).
\end{corollary}

\subsection{Galerkin projection}

\begin{definition}[PSWF Galerkin space]\label{def:Galerkin}
For $N \sim c^{1/3}$:
$X_c^{(N)} := \operatorname{span}\{\pswf{0}, \ldots, \pswf{N-1}\}$,
$P_N : L^2(-1,1) \to X_c^{(N)}$ orthogonal projection.
\end{definition}

By Gram coercivity (Paper~I):
$\sigma_{\min}(G^{(N)}) \geq \alpha_N \sim c^{-1/2}$,
so $P_N$ is well-conditioned.

\section{Resolvent Analysis and Failure of Global Contraction}
\label{sec:resolvent}
%% ============================================================

\begin{lemma}[HS norm of resolvent]\label{lem:HS}
$\norm{R_s}_{\mathfrak{S}_2}^2 \sim c^{5/3}$ near $s \to 1$.
\end{lemma}

\begin{proof}
$\sim c^{1/3}$ transition modes with $1-\lam{k} \asymp c^{-2/3}$,
each contributing $c^{4/3}$: total $c^{5/3}$.
\end{proof}

Combined with the Widom prefactor:
$\norm{D\Tc}_{L^2 \to L^2} \gtrsim c \cdot c^{5/3} = c^{8/3}$.
Global $L^2$ contraction fails.

\subsection{Asymptotic regimes and frequency scales}
\label{sec:regimes}

\begin{assumption}[Galerkin regime]\label{ass:G}
All indices satisfy $k,l \leq N = \lfloor Ac^{1/3}\rfloor$.
\end{assumption}

Partition $(-1,1)$ into:
\begin{itemize}[nosep]
  \item \textbf{Bulk zone} $\mathcal{Z}_{\rm bulk}$: $|s| \leq 1 - c^{-2/3}$.
  \item \textbf{Transition zone} $\mathcal{Z}_{\rm tz}$:
        $1-c^{-2/3} < |s| < 1$, measure $\sim c^{-2/3}$.
\end{itemize}

Two frequency scales:
\begin{align}
  \omega_{kl}^{\rm tz} &:= c^{2/3}|k-l|
  \quad\text{(WKB phase frequency at bulk--transition boundary)},
  \label{eq:freqdef}\\
  \chi_k - \chi_l &\sim 2c|k-l|
  \quad\text{(SL eigenvalue gap)}.
  \label{eq:eigengap}
\end{align}

\begin{remark}[Asymptotic ordering]\label{rem:ordering}
$c^{-1} \ll c^{-2/3} \ll c^{-1/2} \ll c^{-1/3} \ll c^{-1/6} \ll 1$
for $c > 1$. Every estimate below respects this ordering.
\end{remark}

\subsection{Matrix entries and commutator structure}

Write $h = \sum_{k=0}^{N-1} a_k \pswf{k}$. The dominant contribution is
\begin{equation}\label{eq:Tij_simple}
  T_{ij} \;\sim\; c \cdot \frac{|\lam{l}-\lam{k}|}{(1-\lam{k})(1-\lam{l})}
  \cdot I_{ij},
  \qquad I_{ij} := \int_{-1}^1 \arccos(s)\,\pswf{i}(s)\pswf{j}(s)\,ds.
\end{equation}
The numerator $(\lam{l}-\lam{k})$ vanishes for $k=l$, giving:
\begin{equation}\label{eq:diagzero}
  T_{ii} = 0 \qquad \text{for all } i.
\end{equation}

\subsection{Airy approximation and Wronskian bound}
\label{sec:Airy}

\begin{lemma}[Airy approximation]\label{lem:Airy}
Uniformly in $\mathcal{Z}_{\rm tz}(k)$:
$|\pswf{k}(s)| \lesssim c^{1/6}$.
\end{lemma}

\begin{proof}[References]
Olver~\cite{Olver1974};
Osipov--Rokhlin--Xiao~\cite{OsipovRokhlinXiao2013}, Thms~4.2--4.3.
\end{proof}

\begin{corollary}[Wronskian bound]\label{cor:Wronskian}
For $k \neq l$, $k,l \leq N$, with
$W_k^l(s) := (1-s^2)[\pswf{k}'\pswf{l} - \pswf{k}\pswf{l}']$:
\[
  \int_{-1}^1 |W_k^l(s)|\,ds \lesssim \frac{c^{-1/6}}{|k-l|}.
\]
\end{corollary}

\begin{proof}
\textbf{TZ:} $J_{\rm tz} \lesssim c^{1/3} \cdot c^{-2/3} = c^{-1/3}$.
\textbf{Bulk:} $J_{\rm bulk} \lesssim c^{1/2}/(c^{2/3}|k-l|) = c^{-1/6}/|k-l|$.
\textbf{Total:} For $|k-l| \geq 1$ and $c > 1$:
$c^{-1/6} > c^{-1/3}$ (since $c^{1/6} > 1$),
so the bulk term dominates and
$J_{\rm tz} + J_{\rm bulk} \lesssim c^{-1/6}/|k-l|$.
\end{proof}

\subsection{Sturm--Liouville moment and primitive bounds}
\label{sec:SLmoment}

\begin{proposition}[SL bounds]\label{prop:moment}
For $k \neq l$, $k,l \leq N$:
\begin{enumerate}[label=(\alph*)]
  \item \textbf{Parity.} $k \equiv l \pmod{2}$ $\Rightarrow$ $\mu_{kl} = 0$.
  \item \textbf{Moment.} $|\mu_{kl}| \lesssim c^{-7/6}/|k-l|^2$.
  \item \textbf{Wronskian identity.}
        $F_{kl}(s) := \int_s^1\pswf{k}\pswf{l}\,dt
        = W_k^l(s)/(\chi_k-\chi_l)$
        in $\mathcal{Z}_{\rm bulk}$,
        with $|F_{kl}(s)| \lesssim c^{-1/2}(1-s^2)^{1/2}/|k-l|$.
\end{enumerate}
\end{proposition}

\begin{proof}
\textbf{(b)} SL identity: $(\chi_k-\chi_l)\mu_{kl} = \int W_k^l\,ds
\lesssim c^{-1/6}/|k-l|$; divide by $\chi_k-\chi_l \asymp 2c|k-l|$.

\textbf{(c)} SL equation gives $(\chi_k-\chi_l)\pswf{k}\pswf{l}
= -(d/ds)W_k^l$; integrate from $s$ to $1$ using $W_k^l(1)=0$.
Bound from WKB: $|W_k^l| \lesssim c^{1/2}(1-s^2)^{1/2}$ in bulk.
\end{proof}

\subsection{Off-diagonal integral bounds}

\begin{lemma}[Zone decomposition]\label{lem:sublD}
For $k \neq l$, $k,l \leq N$:
\begin{align}
  |I_{kl}^{\rm tz}| &\lesssim c^{-2/3}/|k-l|, \label{eq:IklTZ}\\
  |I_{kl}^{\rm bulk}| &\lesssim c^{-1/2}/|k-l|, \label{eq:IklBulk}\\
  |I_{kl}| &\lesssim c^{-1/2}/|k-l|. \label{eq:Ikltotal}
\end{align}
\end{lemma}

\begin{proof}
IBP via $(\arccos)'(s) = -(1-s^2)^{-1/2}$: boundary terms vanish
by $F_{kl}(1)=0$ and $F_{kl}(-1)=\langle\pswf{k},\pswf{l}\rangle=0$.
Hence $I_{kl} = \int_{-1}^1 F_{kl}(s)(1-s^2)^{-1/2}\,ds$.

\textbf{TZ:} $|F_{kl}| \lesssim c^{-1/3}/|k-l|$ (IBP with $\omega_{kl}^{\rm tz}$);
$(1-s^2)^{-1/2} \lesssim c^{1/3}$; measure $\sim c^{-2/3}$:
$|I_{kl}^{\rm tz}| \lesssim c^{-1/3}/|k-l| \cdot c^{1/3} \cdot c^{-2/3}
= c^{-2/3}/|k-l|$.

\textbf{Bulk:} $|F_{kl}| \lesssim c^{-1/2}(1-s^2)^{1/2}/|k-l|$
(Prop.~\ref{prop:moment}(c));
$(1-s^2)^{-1/2}$ cancels: $|I_{kl}^{\rm bulk}| \leq 2c^{-1/2}/|k-l|$.
\end{proof}

\begin{lemma}[PSWF integral bound]\label{lem:RL}
$\bigl|\int_{-1}^1\arccos(s)\,\pswf{k}\pswf{l}\,ds\bigr|
\leq C_1 c^{-1/2}/|k-l|$.
\end{lemma}

\begin{lemma}[Zone contributions to Schur row sum]\label{lem:zone_schur}
Under Assumption~\ref{ass:B}, for any fixed row $i$:
\begin{align}
  \sum_{j \in \mathrm{TZ},\,j\neq i} |T_{ij}|
  &\leq \tfrac{1}{3}C_2\,c^{-1/6}\log c \;\to\; 0,
  \label{eq:zone_tz}\\
  \sum_{j \in \mathrm{bulk},\,j\neq i} |T_{ij}|
  &\leq C_1 C_2(1+\log N) \;\sim\; C_1 C_2\log c.
  \label{eq:zone_bulk}
\end{align}
The Schur row sum is asymptotically determined by the bulk zone alone.
\end{lemma}

\begin{proof}
\eqref{eq:zone_tz}: Using $|i-j| \geq 1$ and
$|T_{ij}| \leq C_2 c^{1/2} \cdot c^{-2/3}/|i-j|$ for TZ pairs:
$\sum_{j\in\rm TZ} |T_{ij}|
\leq C_2 c^{-1/6}\sum_{d=1}^{N_{\rm tz}} 1/d
\leq C_2 c^{-1/6}\cdot\tfrac{1}{3}\log c \to 0$.
\eqref{eq:zone_bulk}: direct harmonic bound via Prop.~\ref{prop:Tij}.
\end{proof}

\begin{remark}[Transition zone: asymptotic irrelevance]\label{rem:tz_irrelevant}
The transition-zone contribution to the Schur row sum is
$O(c^{-1/6}\log c) \to 0$.
The transition zone is not merely subdominant but asymptotically invisible:
for large $c$, the Schur row sum is determined entirely by the bulk.
\end{remark}

\subsection{Bulk oscillation barrier}
\label{sec:bulk_barrier}

\begin{proposition}[Bulk oscillation barrier]\label{prop:bulk_barrier}
Let $k \neq l$, $k,l \leq N$.
Under the WKB representation
$\pswf{k}(s) \sim c^{-1/4}(1-s^2)^{-1/4}\cos(\theta_k(s))$
in $\mathcal{Z}_{\rm bulk}$:

\medskip
\noindent\textbf{(i)} First IBP (Wronskian):
$|I_{kl}^{\rm bulk}| \lesssim c^{-1/2}/|k-l|$.

\medskip
\noindent\textbf{(ii)} WKB decomposition:
$\pswf{k}\pswf{l} = \tfrac{1}{2}A_kA_l
[\cos(\theta_k-\theta_l)+\cos(\theta_k+\theta_l)]$.
The sum-phase term is $O(c^{-1}/|k+l|)$ after one IBP.
The difference-phase frequency is
$\partial_s(\theta_k-\theta_l) \asymp |k-l|(1-s^2)^{-1/2}$.

\medskip
\noindent\textbf{(iii)} Second IBP applied to the difference term yields:
\[
  I_{kl}^{\rm bulk}
  = \underbrace{\text{interior integral}}_{O(c^{-2/3}/|k-l|)}
  + \underbrace{\text{boundary term at }s_* = 1-c^{-2/3}}_{\text{dominant}}.
\]

\medskip
\noindent\textbf{(iv)} At $s_* = 1-c^{-2/3}$:
$(1-s_*^2)^{1/2} \asymp c^{-1/3}$,
$\partial_s(\theta_k-\theta_l)|_{s_*} \asymp |k-l|c^{1/3}$,
$|F_{kl}(s_*)| \lesssim c^{-5/6}/|k-l|$.
The boundary term satisfies:
\[
  \left|\text{boundary term}\right|
  \asymp \frac{c^{-5/6}/|k-l|}{|k-l|\cdot c^{1/3}}
  = \frac{c^{-7/6}}{|k-l|^2}
  \cdot c^{?}
\]

Recomputing carefully:
boundary term $\asymp F_{kl}(s_*) / \partial_s(\theta_k-\theta_l)|_{s_*}$,
so
\[
  \left|\text{boundary term}\right|
  \asymp \frac{c^{-1/2}\cdot c^{-1/3}/|k-l|}{|k-l|\cdot c^{1/3}}
  = \frac{c^{-1/2}}{|k-l|^2}.
\]
Compared to target $c^{-2/3}/|k-l|$:
\[
  \frac{|\text{boundary term}|}{c^{-2/3}/|k-l|}
  = \frac{c^{-1/2}/|k-l|^2}{c^{-2/3}/|k-l|}
  = \frac{c^{1/6}}{|k-l|}.
\]
The obstruction satisfies:
\begin{itemize}[nosep]
  \item For $|k-l| \lesssim c^{1/6}$: ratio $\gtrsim 1$
        — boundary term dominates, second IBP fails.
  \item For $|k-l| \gg c^{1/6}$: ratio $\to 0$
        — boundary term is negligible, second IBP succeeds.
\end{itemize}
\textbf{The bulk oscillation obstruction is localised to the
near-diagonal regime $|k-l| \lesssim c^{1/6}$.}

\medskip
\noindent\textbf{Equivalence.}
$|I_{kl}^{\rm bulk}| \lesssim c^{-2/3}/|k-l|$
for all $|k-l| \leq N$
$\Longleftrightarrow$
$|\text{boundary term at }s_*| \lesssim c^{-2/3}/|k-l|$
for $|k-l| \lesssim c^{1/6}$.
\end{proposition}

\begin{proof}
Parts~(i)--(iii): standard IBP using
$F_{kl}(s) \lesssim c^{-1/2}(1-s^2)^{1/2}/|k-l|$
(Prop.~\ref{prop:moment}(c)).
Part~(iv): substitute $|F_{kl}(s_*)| \leq c^{-1/2}\cdot c^{-1/3}/|k-l|$
and $|\partial_s(\theta_k-\theta_l)|_{s_*}| \asymp |k-l|c^{1/3}$;
the ratio $c^{1/6}/|k-l|$ determines where the obstruction is active.
\end{proof}

\begin{remark}[Geometric interpretation]\label{rem:barrier_geometry}
The boundary term at $s_*$ marks the transition from oscillatory (WKB)
to monotone (Airy) behaviour.
The obstruction is not global but concentrated in the near-diagonal
band $|k-l| \lesssim c^{1/6}$, suggesting that a near/far decomposition
of $T$ may isolate the difficulty (see Route~(iv), Section~\ref{sec:dyadic}).
\end{remark}

\section{Norm Estimate and Contraction}
\label{sec:norm}

\subsection{Prefactor bounds}

\begin{lemma}[Prefactor]\label{lem:prefactor}
For $k \neq l$, $k,l \leq N$, let
$P_{kl} := c\cdot|\lam{l}-\lam{k}|/[(1-\lam{k})(1-\lam{l})]$.
Raw Widom scaling gives $P_{kl} \lesssim |k-l|\,c^{7/3}$;
Assumption~\ref{ass:B} postulates the reduction $P_{kl} \leq C_2 c^{1/2}$.
\end{lemma}

\begin{remark}[Heuristic for Assumption~B]\label{rem:assB_heuristic}
In the transition regime:
$|\lam{k}-\lam{l}| \asymp c^{-2/3}|k-l|$ and
$(1-\lam{k})(1-\lam{l}) \asymp c^{-4/3}$, giving
$P_{kl} \asymp c^{5/3}|k-l|$.
This is already a reduction from $c^{7/3}|k-l|$ by $c^{2/3}$,
and shows that the worst-case growth originates in the transition regime.
The further reduction from $c^{5/3}|k-l|$ to $c^{1/2}$
(by an additional $c^{7/6}$ per unit $|k-l|$) requires
second-order transition asymptotics and is currently open.
\end{remark}

\begin{assumption}[Prefactor bound]\label{ass:B}
For all $k \neq l$, $k,l \leq N \sim c^{1/3}$:
\[
  c \cdot \frac{|\lam{l}-\lam{k}|}{(1-\lam{k})(1-\lam{l})} \leq C_2 c^{1/2}.
\]
\end{assumption}

\begin{assumption}[Schur-summability]\label{ass:Bprime}
The Schur row sums satisfy:
\[
  \sup_{i \leq N}\sum_{j \neq i,\,j \leq N}
  P_{ij}\cdot|I_{ij}|
  \leq C\,c^{-1/2}\log c.
\]
This is \emph{not} a consequence of Assumption~\ref{ass:B}.
It requires collective oscillatory cancellation in row sums;
see Section~\ref{sec:dyadic}.
\end{assumption}

\subsection{Matrix entry bound and proof of Theorem~\ref{thm:main}}

\begin{proposition}[Matrix entry bound]\label{prop:Tij}
Under Assumption~\ref{ass:B}, for $i \neq j$:
$|T_{ij}| \leq C_1 C_2/|i-j|$.
\end{proposition}

\begin{proof}
$|T_{ij}| \lesssim P_{ij}\cdot|I_{ij}|
\leq C_2 c^{1/2}\cdot C_1 c^{-1/2}/|i-j| = C_1C_2/|i-j|$.
\end{proof}

\begin{remark}[$c$-power balance]\label{rem:balance}
$c^{+1/2}_{\rm Ass.\,B} \times c^{-1/2}_{\rm Lem.\,\ref{lem:RL}} = c^0$.
Conditional on Assumption~B, the matrix norm grows at most like $\log c$
rather than $c^{8/3}$.
\end{remark}

\begin{proof}[Proof of Theorem~\ref{thm:main}]
$T_{ii}=0$ by~\eqref{eq:diagzero}.

\noindent\textbf{Part~(a).}
Proposition~\ref{prop:Tij} gives $|T_{ij}|\leq C_1C_2/|i-j|$.
By the Schur test:
\[
  \norm{T}_{\ell^2\to\ell^2}
  \leq \sup_i\sum_{j\neq i}|T_{ij}|
  \leq C_1C_2 H_N
  \leq C_1C_2(1+\tfrac{1}{3}\log c+\log A)
  =: C(A)\log c.
\]

\noindent\textbf{Part~(b).}
Assumption~\ref{ass:Bprime} directly gives
$\sup_i\sum_{j\neq i}P_{ij}|I_{ij}|\leq Cc^{-1/2}\log c$,
hence $\norm{D\mathcal{T}_c^{(N)}}\leq Cc^{-1/2}\log c\to 0$.
\end{proof}

Under Assumptions~B and~B$'$: for $c\geq c_0$ there exists a unique

and Picard iteration converges at rate $\leq Cc^{-1/2}\log c$.
\end{corollary}

\section{Dyadic Decomposition and Routes to Contraction}
\label{sec:dyadic}
%% ============================================================

\subsection{Dyadic block decomposition}

For fixed row $i$, partition off-diagonal indices:
\begin{equation}\label{eq:dyadic_blocks}
  \mathcal{B}_m := \{j : 2^m \leq |i-j| < 2^{m+1}\},
  \quad m=0,\ldots,M,\quad M:=\lfloor\log_2 N\rfloor.
\end{equation}

\begin{proposition}[Block-wise bound]\label{prop:dyadic_block}
Under Assumption~\ref{ass:B}:
$\sum_{j\in\mathcal{B}_m}|T_{ij}| \leq 2C_1C_2$
for every $m$.
\end{proposition}

\begin{proof}
$|T_{ij}|\leq C_1C_2/2^m$ for $j\in\mathcal{B}_m$;
$|\mathcal{B}_m|\leq 2^{m+1}$.
\end{proof}

\begin{proposition}[Total Schur sum]\label{prop:dyadic_total}
$\sum_{j\neq i}|T_{ij}|
\leq 2C_1C_2(M+1)\sim C(A)\log c$.
Each of the $\sim\log N$ dyadic scales contributes $O(1)$;
the total grows logarithmically.
\end{proposition}

\begin{remark}[Sharpness]
For $T_{ij}=\kappa/|i-j|$, equality holds in every block.
No rearrangement can reduce the sum without signed cancellation.
\end{remark}

\subsection{Proof of the No-go Theorem}

\begin{proof}[Proof of Theorem~\ref{thm:nogo}]
For any interior row $i$, all distances $d=1,\ldots,N-1$ are realised:
\[
  \sup_i\sum_{j\neq i}|T_{ij}|
  \geq \kappa H_{N-1}.
\]
The operator norm lower bound
$\norm{T}_{\ell^2}\geq\tfrac{1}{2}\sup_i\sum_{j\neq i}|T_{ij}|$
follows from the classical Schur inequality for quasi-symmetric operators
(see Young's bilinear inequality).
For $N=\lfloor Ac^{1/3}\rfloor$: $H_{N-1}\geq\tfrac{1}{3}\log c+O(1)$.
\end{proof}

\subsection{Structural characterisation of Assumption~B$'$}

Define the signed dyadic block sum $S_m(i):=\sum_{j\in\mathcal{B}_m}T_{ij}$.
For contraction, the signed sums must satisfy:
\begin{equation}\label{eq:cancellation_needed}
  \sum_{m=0}^{M}|S_m(i)| \leq Cc^{-1/2}\log c,
\end{equation}
rather than the absolute bound $\sum_m 2C_1C_2=O(\log c)$.

\begin{remark}[Explicit equivalence for B$'$]\label{rem:Bprime_equiv}
Assumption~B$'$ is equivalent to oscillatory cancellation in the
signed dyadic block sums $S_m(i)$, i.e., $|S_m(i)|\ll 2C_1C_2$
uniformly in $m$ and $c$.
This is a \emph{Calder\'{o}n--Zygmund-type condition}, not a decay condition.
Pointwise improvements to $|T_{ij}|$ alone cannot establish it
(Theorem~\ref{thm:nogo}).
\end{remark}

\subsection{Routes to contraction}

\begin{remark}[Four routes to~\eqref{eq:cancellation_needed}]
\label{rem:routes}
\begin{enumerate}[nosep,label=(\roman*)]
  \item \textbf{Airy--WKB matching at $s_*$}
        (Open Problem~\ref{op:bulk_refined}):
        establish cancellation of the boundary term at $s_*=1-c^{-2/3}$
        via Airy--WKB connection formulae.
        If successful: unconditional contraction under Assumption~B alone,
        rate $c^{-1/6}\log c\to 0$.
        This is the most immediate analytic target.

  \item \textbf{Sign oscillation of $T_{ij}$ in $j$}:
        if $T_{ij}$ alternates sign with period $\sim 1$,
        then $|S_m(i)|=O(1/2^m)$ and $\sum_m|S_m(i)|=O(1)$.
        This would follow from near-antisymmetry of $P_{ij}$ under
        $j\mapsto j+1$, plausible from oscillatory PSWF products
        but not yet established.

  \item \textbf{Near/far splitting}
        (motivated by Proposition~\ref{prop:bulk_barrier}):
        write $T = T^{\rm near} + T^{\rm far}$ with
        \[
          T^{\rm near}_{ij}
          = T_{ij}\cdot\mathbf{1}_{|i-j|\leq c^{1/6}}.
        \]
        By Proposition~\ref{prop:bulk_barrier}, the bulk oscillation
        obstruction is localised to $|i-j|\lesssim c^{1/6}$.
        The far part $T^{\rm far}$ is therefore free of boundary-term
        obstructions and may admit improved summation estimates.
        The near part is a band matrix of width $c^{1/6}$ acting on
        a space of dimension $N\sim c^{1/3}$.
        Since $c^{1/6}/N = c^{-1/6}\to 0$, its operator norm may be
        accessible via spectral or local analysis, bypassing global
        Schur estimates.
        This reduces the global contraction problem to a
        \emph{finite-band spectral problem}, a structurally different regime.
        Whether $\norm{T^{\rm near}}_{\ell^2}<1$ can be established
        unconditionally remains open.

  \item \textbf{Weighted norm}
        (Open Problem~\ref{op:weightnorm}):
        find a $c$-dependent weighted $\ell^2$ norm under which
        $\norm{DT}_w<1$ unconditionally.
        Section~\ref{sec:spectral_weights} shows all natural candidates fail.
\end{enumerate}
\end{remark}

\subsection{Phase structure and the Fourier-multiplier strategy}
\label{sec:fourier}

\begin{remark}[Heuristic phase structure of $T_{ij}$]\label{rem:phase_Tij}
Under a WKB approximation, products of PSWFs admit:
\[
  \pswf{i}(s)\pswf{j}(s)
  \approx A_{ij}(s)\cos(\theta_i(s)-\theta_j(s)),
\]
with slowly varying amplitude and rapidly varying phases.
Formally, this suggests:
\[
  T_{ij}
  \approx \frac{\widetilde{A}_{ij}}{|i-j|}\cos(\Theta_i-\Theta_j)
\]
for effective phases $\Theta_k$ arising from the $s$-integration.
If $\Theta_i-\Theta_j$ exhibits non-degeneracy or equidistribution
in $j$ (for fixed $i$), the signed block sums $S_m(i)$ would exhibit
oscillatory cancellation analogous to Calder\'{o}n--Zygmund singular
integrals, yielding~\eqref{eq:cancellation_needed}.
Making this rigorous requires quantitative control of $\theta_k$
in the transition and bulk regimes, which is currently open.
\end{remark}

\begin{conjecture}[Linear phase equidistribution]\label{conj:phase}
The effective WKB phases $\Theta_k^{(c)}$ at $s_*=1-c^{-2/3}$ satisfy,
for $k\leq N\sim c^{1/3}$:
\[
  \Theta_k^{(c)} = \alpha^{(c)}k + \beta^{(c)} + O(k^2/c),
\]
where $\alpha^{(c)}\in(0,2\pi)$ is bounded away from $0$ and $\pi$
uniformly in $c$.
Under this conjecture, Assumption~B$'$ follows from a standard
Dirichlet-kernel cancellation argument applied to $S_m(i)$.
\end{conjecture}

\begin{remark}[Fourier-multiplier route to B$'$]\label{rem:fourier_route}
If $T_{ij}=f(i-j)/(i-j)$ for some bounded oscillatory
$f:\mathbb{Z}\to[-1,1]$, then $T$ is a discrete convolution operator and
\[
  \norm{T}_{\ell^2\to\ell^2}
  = \bigl\|\widehat{(f/|\cdot|)}\bigr\|_{\ell^\infty}
  = O(1),
\]
by $\ell^2$-boundedness of the discrete Hilbert transform with
oscillatory weight.
This would establish B$'$ unconditionally, bypassing the Schur
argument entirely.
The key hypothesis is \emph{approximate translation-invariance}:
$T_{ij}\approx g(i-j)$ for some $g$.
Whether the PSWF matrix $T$ has this structure is an open question
that would follow from Conjecture~\ref{conj:phase}.
\end{remark}

\subsection{Why spectral and index weights fail}
\label{sec:spectral_weights}

\begin{proposition}[Spectral weights are constant in the Galerkin window]
\label{prop:spectral_weight_fail}
For $k\leq N\sim c^{1/3}$: $1-\lambda_k\asymp c^{-2/3}$ uniformly,
so $w_k=(1-\lambda_k)^\alpha\asymp c^{-2\alpha/3}$ and
$w_j/w_i=1+O(c^{-1/3})$.
Spectral weighting does not modify the Schur kernel.
\end{proposition}

\begin{proof}
All indices lie in the transition regime (Widom transition asymptotics).
\end{proof}

\begin{remark}[Why spectral weights fail structurally]
\label{rem:spectral_weight_structural}
The weight $w_k=(1-\lambda_k)^\alpha$ is blind to index structure:
on the transition plateau all eigenvalues are equivalent.
An effective weight for the Schur kernel $1/|i-j|$ must
distinguish indices, not eigenvalues.
\end{remark}

\begin{remark}[Index-based weights also fail]\label{rem:index_weights}
\begin{enumerate}[nosep,label=(\alph*)]
  \item Power weight $w_k=(1+k)^\beta$:
        weighted row sum $\sum_{d=1}^N(1+d/i)^\beta/d\sim\log N$
        for all $\beta$.
  \item Exponential weight $w_k=e^{\gamma k/N}$:
        row sum $\sum_d e^{\gamma d/N}/d\sim e^\gamma\log N$.
\end{enumerate}
Both fail because $1/d$ is not summable and no symmetric polynomial
or exponential weight can make it so.
\end{remark}

\begin{corollary}[No naive weight works]\label{cor:no_naive_weight}
No weight of spectral, power, or exponential type makes
$\sup_i\sum_j(w_j/w_i)/|i-j|=o(\log N)$.
\end{corollary}

\section{Open Problems}
\label{sec:open}
%% ============================================================

\begin{openproblem}[Prefactor and cancellation]\label{op:prefactor}
\begin{enumerate}[nosep]
  \item Prove Assumption~\ref{ass:B}: $P_{kl}\leq C_2c^{1/2}$.
        Remark~\ref{rem:assB_heuristic} reduces the gap from $c^{7/3}$
        to $c^{5/3}$; closing the remaining $c^{7/6}$ requires
        second-order Airy estimates.
  \item Prove Assumption~\ref{ass:Bprime}.
        By Remark~\ref{rem:Bprime_equiv}, this is equivalent to
        signed cancellation in $S_m(i)$.
        Conjecture~\ref{conj:phase} would imply B$'$
        via Dirichlet-kernel / Fourier-multiplier argument
        (Remark~\ref{rem:fourier_route}).
\end{enumerate}
\end{openproblem}

\begin{openproblem}[Weighted norm]\label{op:weightnorm}
Find a $c$-dependent weighted norm $\norm{\cdot}_w$ on $X_c^{(N)}$
under which $\norm{D\mathcal{T}_c^{(N)}}_w<1$ unconditionally.
By Proposition~\ref{prop:spectral_weight_fail} and
Corollary~\ref{cor:no_naive_weight}, all natural candidates fail.
A successful weight must be asymmetric or of two-weight type:
find $\{u_k\},\{v_k\}>0$ such that
\[
  \sum_j v_j|T_{ij}|\leq u_i
  \quad\text{and}\quad
  \sum_j u_j^{-1}|T_{ij}|\leq v_i^{-1}
  \quad\text{for all }i.
\]
\end{openproblem}

\begin{openproblem}[Airy--WKB matching]\label{op:bulk_refined}
By Proposition~\ref{prop:bulk_barrier}, the improved bound
$|I_{kl}^{\rm bulk}|\lesssim c^{-2/3}/|k-l|$ is equivalent to
cancellation of the boundary term at $s_*=1-c^{-2/3}$,
which is active only for $|k-l|\lesssim c^{1/6}$.

\medskip
\noindent\textbf{Approach.}
Set $s=1-c^{-2/3}x$, $x=O(1)$.
In these coordinates:
\begin{itemize}[nosep]
  \item $\pswf{k}$ transitions from monotone Airy to oscillatory WKB.
  \item The difference phase linearises:
        $\theta_k(s)-\theta_l(s)\approx(k-l)\phi(x)$.
  \item The boundary term becomes an explicit $\Ai\cdot\Ai$ integral.
\end{itemize}
If this integral exhibits stationary-phase cancellation, the boundary
term reduces to $O(c^{-2/3}/|k-l|)$, and
Theorem~\ref{thm:main}(a) upgrades unconditionally to
$\norm{D\mathcal{T}_c^{(N)}}\lesssim c^{-1/6}\log c\to 0$
under Assumption~B alone.
\end{openproblem}

\begin{openproblem}[Near/far splitting]\label{op:nearfar}
Formalise the near/far decomposition $T=T^{\rm near}+T^{\rm far}$
with $T^{\rm near}_{ij}=T_{ij}\cdot\mathbf{1}_{|i-j|\leq c^{1/6}}$:
\begin{enumerate}[nosep]
  \item Establish unconditional estimates for $\norm{T^{\rm far}}_{\ell^2}$,
        exploiting the absence of the bulk boundary obstruction
        for $|i-j|\gg c^{1/6}$.
  \item Control $\norm{T^{\rm near}}_{\ell^2}$ via spectral or local methods
        for the band matrix of width $c^{1/6}$ and dimension $N\sim c^{1/3}$.
        The bandwidth-to-dimension ratio $c^{1/6}/c^{1/3}=c^{-1/6}\to 0$
        suggests that local structure dominates.
\end{enumerate}
\end{openproblem}

\begin{openproblem}[Optimal Galerkin dimension]\label{op:Nopt}
Determine the maximal $N(c)$ for which the Schur bound is below~$1$.
Conjecture: $N_{\rm opt}\sim c^{1/2}$.
\end{openproblem}

\section{Relation to Assumption~A and Summary of Results}
%% ============================================================

\begin{itemize}[nosep]
  \item \textbf{Unconditionally proved:}
        $|I_{kl}|\leq C_1c^{-1/2}/|k-l|$;
        $T_{ii}=0$;
        $\norm{DT}_{L^2}\gtrsim c^{8/3}$;
        $\log c$ growth is optimal in pointwise-bound class
        (Thm~\ref{thm:nogo});
        spectral and index weights cannot improve Schur kernel
        (Prop.~\ref{prop:spectral_weight_fail},
        Cor.~\ref{cor:no_naive_weight});
        TZ Schur contribution is $o(1)$
        (Lem.~\ref{lem:zone_schur}, Rem.~\ref{rem:tz_irrelevant});
        bulk oscillation barrier localised to $|k-l|\lesssim c^{1/6}$
        (Prop.~\ref{prop:bulk_barrier}).

  \item \textbf{Under Assumption~B:}
        $|T_{ij}|\leq C_1C_2/|i-j|$;
        $\norm{D\mathcal{T}_c^{(N)}}\leq C\log c$.

  \item \textbf{Under Assumptions~B and~B$'$:}
        $\norm{D\mathcal{T}_c^{(N)}}\leq Cc^{-1/2}\log c\to 0$;
        conditional Galerkin fixed point (Cor.~\ref{cor:fixedpoint}).

  \item \textbf{Four routes to unconditional contraction,
        in increasing depth:}
        \begin{enumerate}[nosep,label=(\roman*)]
          \item Airy--WKB matching at $s_*$
                (OP~\ref{op:bulk_refined}):
                unconditional contraction rate $c^{-1/6}\log c$
                under Assumption~B.
          \item Near/far splitting
                (OP~\ref{op:nearfar}):
                reduces global Schur to a finite-band spectral problem
                of shrinking relative bandwidth $c^{-1/6}$.
          \item Linear phase equidistribution
                (Conj.~\ref{conj:phase} $\Rightarrow$ B$'$
                via Fourier-multiplier, Rem.~\ref{rem:fourier_route}):
                contraction under Assumption~B.
          \item Two-weight Schur estimate
                (OP~\ref{op:weightnorm}):
                contraction without B$'$, no Schur required.
        \end{enumerate}
        Route~(i) is the most immediate analytic target.
        Route~(ii) is structurally new: it recasts the problem
        as a local band-matrix problem rather than a global estimate.
\end{itemize}
