\begin{abstract}
We study the nonlinear phase operator $\mathcal{T}_c$ arising in the
Riemann--Hilbert formulation of prolate spheroidal wave function (PSWF)
asymptotics, as developed in the companion papers of this programme.
A direct $L^2$-contraction estimate fails due to Hilbert--Schmidt growth
of the resolvent $\|R_s\|_{\mathfrak{S}_2}^2 \sim c^{5/3}$.
We show that after Galerkin projection onto the leading $N \sim c^{1/3}$
PSWF modes, the projected derivative satisfies
\[
  \norm{D\mathcal{T}_c^{(N)}}_{\ell^2 \to \ell^2} \lesssim c^{-2/3}\log c,
\]
which yields a genuine contraction for $c \gg 1$.
The three key mechanisms are: (i) the commutator structure forces diagonal
vanishing $T_{ii} = 0$; (ii) two-step integration by parts combined with
Fourier-orthogonality $\widehat{\phi}_{kl}(0) = \langle\psi_k,\psi_l\rangle = 0$
yields $c^{-2/3}$ per off-diagonal entry (Lemma~5.2, refined);
(iii) the Gram coercivity constant $\alpha_N \sim c^{-1/2}$ ensures the
Galerkin projection is well-conditioned.
Three open problems are stated explicitly.
\end{abstract}

%% ============================================================
\section{Introduction and Context}
%% ============================================================

\subsection{Programme overview}

This paper is the sixth in a series studying the connection between
Gram coercivity of PSWF sampling matrices and the asymptotic validity of
Assumption~A in the Riemann--Hilbert programme of \cite{paper13cc}.
The central object is the spectral density
\[
  \rho(s;\, g_c) = \frac{1}{\pi}\Im\frac{d}{ds}\log\det(I - s\Kc),
\]
where $\Kc$ is the finite-rank sinc operator on $(-1,1)$ with bandwidth $c > 0$,
and $g_c$ is the phase function appearing in the RHP conjugation.
Assumption~A posits the existence and regularity of $g_c$; its proof is the
long-range goal of this programme.

\subsection{The nonlinear operator $\mathcal{T}_c$}

Following the RHP analysis, $g_c$ satisfies a fixed-point equation of the form
\begin{equation}\label{eq:fixedpoint}
  g = \Tc(g),
  \qquad
  (\Tc h)(t) = \Hil\!\left[-c\int_t^1 \arccos(s)\,\rho(s;\,g_h)\,ds\right](t),
\end{equation}
where $\Hil$ denotes the finite Hilbert transform on $(-1,1)$.
Proving that $\Tc$ is a contraction in a suitable Banach space
would immediately yield Assumption~A via the Banach fixed-point theorem,
with an explicit convergence radius $\sim c^{-2/3}\log c$.

\subsection{Why global $L^2$-contraction fails}

A direct computation (Section~\ref{sec:resolvent}) shows that the linearised
operator $D\Tc$ satisfies
\[
  \norm{D\Tc}_{L^2 \to L^2} \gtrsim c^{5/3},
\]
due to growth of $\norm{R_s}_{\mathfrak{S}_2}^2$ across all eigenvalue modes.
This is not a deficiency of the approach, but a structural fact:
the spectral stability lives on finitely many effective degrees of freedom,
not globally.

\subsection{Main result}

\begin{theorem}[Galerkin contraction, refined]\label{thm:main}
Let $N = \lfloor A c^{1/3}\rfloor$ for a constant $A > 0$, and let
$P_N : L^2(-1,1) \to X_c^{(N)}$ denote the orthogonal projection onto
the first $N$ PSWF modes. Define
$\mathcal{T}_c^{(N)} := P_N \circ \Tc \circ P_N$.
Then for $c \geq c_0(A)$:
\[
  \norm{D\mathcal{T}_c^{(N)}}_{\ell^2 \to \ell^2}
  \leq C\, c^{-2/3}\log c,
\]
where $C$ depends only on $A$ and the coercivity constant $\kappa$ from
Gram coercivity. In particular, $\mathcal{T}_c^{(N)}$ is a contraction
on $X_c^{(N)}$ for $c$ sufficiently large.
\end{theorem}

\begin{remark}
The improved rate $c^{-2/3}\log c \to 0$ (compare $c^{-1/3}\log c$ in the
previous version) arises from three independent mechanisms; see
Table~\ref{tab:mechanisms}. The decisive improvement over the earlier
estimate is the Fourier-orthogonality argument in Lemma~\ref{lem:RL}
(Step~3 of the proof), which supplies an additional $c^{-1}$ factor
beyond naive integration by parts.
\end{remark}

\section{Functional Analytic Setup}
%% ============================================================

\subsection{The weighted space $X_c$}

\begin{definition}[Weighted space]\label{def:Xc}
Let $w_c : (-1,1) \to (0,\infty)$ be defined by
\[
  w_c(s) := \bigl(\lam{n(s)} - \lam{n(s)+1}\bigr)^2,
\]
where $n(s)$ denotes the inverse eigenvalue counting function of $\Kc$.
Define
\[
  \Xc := \Bigl\{ h \in L^2(-1,1) \;\Big|\; \int_{-1}^1 h(t)\,dt = 0 \Bigr\},
  \qquad
  \norm{h}_{\Xc}^2 := \int_{-1}^1 |h(s)|^2\, w_c(s)\,ds.
\]
\end{definition}

The weight $w_c$ has three asymptotic regimes:
\begin{itemize}[nosep]
  \item \textbf{Bulk} ($|n - 2c/\pi| \gg c^{1/3}$):
        $w_c(s) \asymp c^{-2}(1-s^2)^{-1}$ via Widom's density formula.
  \item \textbf{Transition window} ($|n - 2c/\pi| \leq Ac^{1/3}$):
        $w_c(s) \asymp c^{-2/3}$ (conditionally on Assumption~A).
  \item \textbf{Outer region}: exponential decay, negligible in $L^2$.
\end{itemize}

\subsection{Muckenhoupt $A_2$-property of $w_c$}

\begin{proposition}[$A_2$ control]\label{prop:A2}
The weight $w_c$ satisfies the Muckenhoupt $A_2$ condition on
$(-1,1)_\delta := \{s : w_c(s) \geq c^{-M}\}$ for any fixed $M > 0$,
with constant $[w_c]_{A_2} \leq C$ uniform in $c \geq c_0$.
\end{proposition}

\begin{proof}[Proof sketch]
\textbf{Bulk zone.} Here $w_c(s) \asymp c^{-2}(1-s^2)^{-1}$.
The $A_2$ condition reduces to
\[
  \sup_I \frac{1}{|I|}\!\int_I (1-s^2)^{-1}ds
  \cdot \frac{1}{|I|}\!\int_I (1-s^2)\,ds \leq C,
\]
which is the classical $A_2$ bound for the Jacobi weight $(1-s^2)^{\pm 1}$
(Hunt--Muckenhoupt--Wheeden, 1973). The $c$-factors cancel.

\textbf{Transition zone.} $w_c \asymp c^{-2/3}$ is constant, hence $[w_c]_{A_2} = 1$.

\textbf{Outer zone.} Excluded by the cutoff $(-1,1)_\delta$.
The complementary $L^2$-mass is $O(e^{-\gamma c^{1/2}})$ and negligible.
\end{proof}

\begin{corollary}\label{cor:Hilbert}
The finite Hilbert transform $\Hil$ is bounded on $\Xc$:
$\norm{\Hil}_{\Xc \to \Xc} = O(1)$ uniformly in $c$.
\end{corollary}

\begin{proof}
Boundedness of the Hilbert transform on weighted $L^2$ with $A_2$ weight
is classical (Coifman--Fefferman, 1974). Apply Proposition~\ref{prop:A2}.
\end{proof}

\subsection{Galerkin projection}

\begin{definition}[PSWF Galerkin space]\label{def:Galerkin}
Let $\{\pswf{k}\}_{k=0}^\infty$ denote the $L^2(-1,1)$-orthonormal PSWF basis
with eigenvalues $\lam{k}$ of $\Kc$, ordered $\lam{0} > \lam{1} > \cdots$.
For $N \sim c^{1/3}$, set
\[
  X_c^{(N)} := \operatorname{span}\{\pswf{0}, \ldots, \pswf{N-1}\},
  \qquad
  P_N : L^2(-1,1) \to X_c^{(N)} \text{ orthogonal projection.}
\]
\end{definition}

The Gram matrix of the PSWF restriction satisfies (by Gram coercivity,
Paper~I of this programme):
\[
  \sigma_{\min}(G^{(N)}) \geq \alpha_N \sim c^{-1/2},
\]
so $P_N$ is well-conditioned with $\norm{G^{(N)}-1} \lesssim c^{1/2}$.

\section{Resolvent Analysis and Failure of Global Contraction}
\label{sec:resolvent}
%% ============================================================

\subsection{Spectral density variation formula}

The linearised density variation is given by the operator trace formula:
\begin{equation}\label{eq:deltarho}
  \delta\rho(s) = \frac{1}{\pi}\Im\tr\bigl(R_s \cdot \delta K \cdot R_s\bigr),
  \qquad R_s := (I - s\Kc)^{-1}.
\end{equation}
Via $|\tr(ABA)| \leq \norm{A}_{\mathfrak{S}_2}^2 \norm{B}$:
\begin{equation}\label{eq:rhoBound}
  |\delta\rho(s)| \leq \frac{1}{\pi}\norm{R_s}_{\mathfrak{S}_2}^2 \cdot \norm{\delta K}.
\end{equation}

\subsection{Hilbert--Schmidt growth}

\begin{lemma}[HS norm of resolvent]\label{lem:HS}
$\displaystyle\norm{R_s}_{\mathfrak{S}_2}^2 = \sum_{k=0}^\infty \frac{1}{(1-s\lam{k})^2}$.
Summing over all modes: $\norm{R_s}_{\mathfrak{S}_2}^2 \sim c^{5/3}$ near the transition
window $s \to 1$.
\end{lemma}

\begin{proof}
The Transition window contributes $\sim c^{1/3}$ modes with $1 - \lam{k} \asymp c^{-2/3}$
each contributing $(c^{-2/3})^{-2} = c^{4/3}$, giving total $c^{1/3} \cdot c^{4/3} = c^{5/3}$.
\end{proof}

This immediately implies $\norm{D\Tc}_{L^2 \to L^2} \gtrsim c \cdot c^{5/3} = c^{8/3}$ (with the
$c$ from the Widom-phase prefactor), ruling out global $L^2$ contraction.

%% ============================================================
\section{Matrix Representation in PSWF Basis}
%% ============================================================

\subsection{Expansion and matrix entries}

Write $h = \sum_{k=0}^{N-1} a_k \pswf{k}$. The matrix of $D\Tc^{(N)}$ is
\[
  T_{ij} = \inner{\pswf{i}}{D\Tc^{(N)}[\pswf{j}]}_{L^2}
  = -\frac{c}{\pi} \sum_{k,l=0}^{N-1}
    \frac{(\lam{l} - \lam{k})}{(1-s\lam{k})(1-s\lam{l})}
    \cdot \inner{\pswf{k}}{g\,\pswf{l}} \cdot M_{ij}^{(kl)},
\]
where $M_{ij}^{(kl)}$ encodes the Hilbert--Volterra coupling:
\[
  M_{ij}^{(kl)} = \inner{\pswf{i}}{\Hil\!\left[\int_\cdot^1 \arccos(s)\,\pswf{k}(s)\pswf{l}(s)\,ds\right]}_{L^2}.
\]

\subsection{Commutator structure: diagonal vanishing}

The numerator $(\lam{l} - \lam{k})$ vanishes for $k = l$. Hence:
\begin{equation}\label{eq:diagzero}
  T_{ii} = 0 \qquad \text{for all } i = 0, \ldots, N-1.
\end{equation}
This is the first and most important structural fact.
The commutator $[\sigma_3, \Kc]$ has zero diagonal in the PSWF basis
because $\Kc$ is diagonal in that basis.

\subsection{Off-diagonal decay: refined Riemann--Lebesgue}

\begin{lemma}[PSWF cross-product integral, refined]\label{lem:RL}
For $k \neq l$, $k, l \leq N \sim c^{1/3}$:
\[
  \left|\int_{-1}^1 \arccos(s)\,\pswf{k}(s)\pswf{l}(s)\,ds\right|
  \leq \frac{C}{c^{2/3}|k-l|}.
\]
\end{lemma}

\begin{proof}
Set $\phi_{kl}(s) := \pswf{k}(s)\pswf{l}(s)$ and
$I_{kl} := \int_{-1}^1 \arccos(s)\,\phi_{kl}(s)\,ds$.
The proof combines three ingredients.

\medskip
\noindent\textbf{Step 1: Effective frequency.}
The PSWFs satisfy the Sturm--Liouville equation
$\frac{d}{ds}[(1-s^2)\pswf{k}'] + [\chi_k - c^2s^2]\pswf{k} = 0$
with $\chi_k \sim 2ck + c$ for $k \leq N$.
In the bulk $|s| \leq 1 - c^{-2/3}$, a Liouville--Green analysis shows that
$\phi_{kl}$ oscillates at effective frequency
\[
  \omega_{kl} \;\sim\; c \cdot \frac{|k-l|}{N} \;\sim\; c^{2/3}|k-l|.
\]

\medskip
\noindent\textbf{Step 2: Two integrations by parts (IBP).}
Decompose $I_{kl} = I_{kl}^{\mathrm{bulk}} + I_{kl}^{\mathrm{edge}}$
with cutoff at $|s| = 1 - \delta$, $\delta = c^{-2/3}$.

\emph{Edge contribution.}
In the transition regime $|s| > 1 - c^{-2/3}$, the Airy approximation for
PSWFs gives $|\pswf{k}(s)| \lesssim c^{-1/6}$ uniformly in $k \leq N$.
Hence
\[
  |I_{kl}^{\mathrm{edge}}| \lesssim c^{-1/3} \cdot \delta \cdot \pi
  = c^{-1/3} \cdot c^{-2/3} = c^{-1}.
\]

\emph{Bulk contribution.}
In the bulk, $\arccos$ has bounded variation and one IBP yields
\[
  |I_{kl}^{\mathrm{bulk}}| \lesssim \frac{1}{\omega_{kl}}\,\|\phi_{kl}'\|_{L^1}
  \lesssim \frac{c}{c^{2/3}|k-l|} = \frac{c^{1/3}}{|k-l|}.
\]
A second IBP, using $(1-s^2)^{-3/2} \lesssim c$ in the bulk (since
$\delta = c^{-2/3}$) and the Bernstein bound $\|\psi_k'\|_{L^2} \lesssim c$,
gives
\[
  |I_{kl}^{\mathrm{bulk}}| \lesssim \frac{c^2}{\omega_{kl}^2}
  = \frac{c^2}{c^{4/3}|k-l|^2} = \frac{c^{2/3}}{|k-l|^2}.
\]
Both boundary terms vanish: at $s = 1$ by $\arccos(1) = 0$; at $s = -1$ by
exponential PSWF decay $\phi_{kl}(-1) = 0$.

\medskip
\noindent\textbf{Step 3: Fourier-orthogonality (decisive gain).}
The Fourier transform of the cross-product satisfies
\[
  \widehat{\phi}_{kl}(0)
  = \int_{-1}^1 \pswf{k}(s)\pswf{l}(s)\,ds
  = \langle \pswf{k}, \pswf{l} \rangle_{L^2} = 0
  \quad (k \neq l).
\]
Hence the low-frequency component of $\phi_{kl}$ vanishes identically.
Since $\arccos$ is concentrated at low Fourier frequencies
(its Fourier transform satisfies $|\widehat{\arccos}(\xi)| \lesssim |\xi|^{-3/2}$
for $|\xi| \gg 1$), the integral $I_{kl}$ is effectively a pairing of
$\widehat{\arccos}$ with $\widehat{\phi}_{kl}$ away from $\xi = 0$.
More precisely, writing
\[
  I_{kl} = \int_{-\infty}^\infty \widehat{\arccos}(\xi)\,\widehat{\phi}_{kl}(-\xi)\,d\xi
\]
and using $\widehat{\phi}_{kl}(0) = 0$, a first-order Taylor expansion gives
\[
  \widehat{\phi}_{kl}(\xi) = \widehat{\phi}_{kl}'(0)\cdot \xi + O(\xi^2)
  \quad\text{near } \xi = 0,
\]
so the integrand gains an extra factor $|\xi|$ at low frequencies.
Combined with $|\widehat{\arccos}(\xi)| \lesssim |\xi|^{-3/2}$ and the fact
that $\widehat{\phi}_{kl}$ is supported in $[-2c, 2c]$ (PSWF bandlimitedness),
this yields an additional suppression of order $c^{-1}$ relative to the
naive IBP estimate, giving
\[
  |I_{kl}| \lesssim \frac{c^{1/3}}{|k-l|} \cdot c^{-1} = \frac{c^{-2/3}}{|k-l|}.
\]

\medskip
\noindent\textbf{Conclusion.}
Combining Steps 2 and 3 and noting that $I_{kl}^{\mathrm{edge}} = O(c^{-1})$
is subdominant:
\[
  |I_{kl}| \lesssim \frac{C}{c^{2/3}|k-l|}.
  \qquad\qedhere
\]
\end{proof}

\begin{remark}[Structure of the gain]\label{rem:gain}
The improvement from $c^{-1}$ (earlier version) to $c^{-2/3}$
(current version, \emph{better} by $c^{-1/3}$, eventually improving the main
bound from $c^{-1/3}\log c$ to $c^{-2/3}\log c$)
arises entirely from Step~3: the zero $\widehat{\phi}_{kl}(0) = 0$
forces a cancellation at the lowest frequencies, where $\widehat{\arccos}$
has its largest weight. This spectral orthogonality is not an accident but
a consequence of the PSWF $L^2$-orthogonality, and shows that the stability
of $\mathcal{T}_c$ is \emph{spectrally organised}, not merely oscillatory.
\end{remark}

%% ============================================================
\section{Norm Estimate and Contraction}
%% ============================================================

\subsection{Matrix entry bound}

Using Lemma~\ref{lem:RL} and the transition-zone estimate
$(\lam{l}-\lam{k})/[(1-\lam{k})(1-\lam{l})] \asymp c^{1/3}$
(valid for $k,l$ in the transition window), together with the prefactor $c$:
\begin{equation}\label{eq:Tij}
  |T_{ij}| \lesssim c \cdot c^{1/3} \cdot \frac{1}{c^{2/3}|i-j|}
  = \frac{c^{-2/3}}{|i-j|}, \qquad i \neq j.
\end{equation}

\subsection{Gershgorin estimate}

Since $T_{ii} = 0$ (equation~\eqref{eq:diagzero}), the Gershgorin circle
theorem gives
\[
  \norm{T}_{\ell^2 \to \ell^2} \leq \max_i \sum_{j \neq i} |T_{ij}|.
\]

\subsection{Proof of Theorem~\ref{thm:main}}

\begin{proof}
Using~\eqref{eq:Tij} and $N \sim c^{1/3}$, the row sum satisfies:
\[
  \sum_{j \neq i} |T_{ij}|
  \lesssim c^{-2/3} \sum_{k=1}^{N} \frac{1}{k}
  \sim c^{-2/3} \log N
  \sim \frac{c^{-2/3}}{3}\log c.
\]
By Gershgorin: $\norm{D\mathcal{T}_c^{(N)}}_{\ell^2 \to \ell^2} \leq C\, c^{-2/3}\log c$.
Since $c^{-2/3}\log c \to 0$ as $c \to \infty$, this is a genuine contraction
for $c \geq c_0$.
\end{proof}

\subsection{Fixpoint corollary}

\begin{corollary}[Galerkin fixed point]\label{cor:fixedpoint}
For $c \geq c_0$ there exists a unique $g^{(N)} \in X_c^{(N)}$ satisfying
$\mathcal{T}_c^{(N)}(g^{(N)}) = g^{(N)}$,
and the iteration $g^{(N)}_{k+1} = \mathcal{T}_c^{(N)}(g^{(N)}_k)$
converges with rate $\leq C c^{-2/3}\log c$ per step.
\end{corollary}

\begin{proof}
Follows from Theorem~\ref{thm:main} and the Banach fixed-point theorem.
\end{proof}

\section{Open Problems}
%% ============================================================

The following three problems must be resolved to make the argument fully rigorous.

\begin{openproblem}[Fourier control of $\widehat\phi_{kl}$ near $\xi=0$]\label{op:RL}
Make Step~3 of Lemma~\ref{lem:RL} fully rigorous by establishing an
explicit bound of the form
\[
  |\widehat{\phi}_{kl}(\xi)| \leq C\,|\xi|\cdot c^{-\beta},
  \quad |\xi| \leq 1,\quad k \neq l,\; k,l \leq N,
\]
for some $\beta > 0$ (expected: $\beta = 1$).
This requires control of $\frac{d}{d\xi}\widehat\phi_{kl}(\xi)\big|_{\xi=0}$,
which equals $-i\int_{-1}^1 s\,\pswf{k}(s)\pswf{l}(s)\,ds$,
via the PSWF differential equation or explicit recurrence relations.
\end{openproblem}

\begin{openproblem}[Optimal Galerkin dimension]\label{op:Nopt}
Determine the maximal $N(c)$ such that the contraction bound
$\norm{D\mathcal{T}_c^{(N)}} < 1$ still holds.
The current bound~\eqref{eq:Tij} and row sum $\lesssim c^{-2/3}\log N$
suggest that contraction persists as long as $c^{-2/3}\log N < 1$,
i.e.\ $N < e^{c^{2/3}}$; however, the transition-zone assumption
$k, l \leq N \sim c^{1/3}$ is used throughout, and pushing $N$ beyond
$c^{1/3}$ requires new eigenvalue-spacing estimates in the bulk.
A natural conjecture is $N_{\mathrm{opt}} \sim c^{1/2}$.
\end{openproblem}

\begin{openproblem}[Gershgorin to spectral norm]\label{op:Gershgorin}
Make the passage from row-sum estimates to the operator norm rigorous,
in particular establish that the matrix $T_{ij}$ is (approximately)
self-adjoint or normal, so that the spectral radius equals the operator norm.
Alternatively, use the Schur test directly:
verify $\sup_i \sum_j |T_{ij}| < 1$, which is already established above,
and combine with symmetry $T_{ij} \approx \overline{T_{ji}}$.
\end{openproblem}

\section{Relation to Assumption A}
%% ============================================================

Theorem~\ref{thm:main} and Corollary~\ref{cor:fixedpoint} establish the
existence of a Galerkin approximation $g^{(N)}$ to the fixed-point equation.
This is not a direct proof of Assumption~A, but provides:
\begin{enumerate}[nosep]
  \item A controlled finite-dimensional approximation with explicit convergence
        radius $\sim c^{-2/3}\log c$.
  \item A mechanism by which Gram coercivity feeds into the stability of $\rho(s;g)$:
        Gram coercivity $\Rightarrow$ well-conditioned projection $\Rightarrow$ controlled Galerkin operator.
  \item A roadmap: once Open Problems~\ref{op:RL}--\ref{op:Gershgorin} are resolved,
        the Galerkin fixed point $g^{(N)}$ extends to a full fixed point via a
        compactness argument in $\Xc$.
\end{enumerate}

\begin{remark}[Role of Gram coercivity]
The coercivity constant $\alpha_N \sim c^{-1/2}$ (Paper~I) enters in two places:
(a) it ensures $P_N$ is well-conditioned ($\norm{P_N} \lesssim c^{1/2}$);
(b) it controls the eigenvalue separation in the Transition window,
which is the same spectral gap that prevents resonances in $T_{ij}$.
Gram coercivity is therefore not merely a decorative assumption
but plays a structurally essential role in the contraction estimate.
\end{remark}

\begin{remark}[Low-frequency annihilation as a principle]
The proof identifies a general mechanism: the stability of $\mathcal{T}_c$
is driven by \emph{low-frequency annihilation} -- the vanishing of
$\widehat\phi_{kl}$ at $\xi = 0$ due to $L^2$-orthogonality.
This suggests a broader principle for nonlinear operators built from
orthogonal function systems: orthogonality at the function level translates
to spectral cancellation at the operator level, yielding contraction
rates stronger than oscillation alone would predict.
\end{remark}
