\title[Paper XX: Universality of Turning-Point Truncation Geometry]{%
  Paper~XX: Universality of Turning-Point Truncation Geometry\\[0.3em]
  \large Spectral-Geometric Invariants and Two-Scale Logarithmic Losses}

\author{\textit{prolate-gram-coercivity programme}}
\date{May 2026}

\begin{abstract}
Papers~XVIII and~XIX established, for the Airy turning-point model,
qualitative universality (BR3) and a quantitative two-scale rate
$O(c^{-1/3}(\log c)^{5/3})$ with polynomial truncation cost $O(K^{5/3})$.
This paper shows that the exponent $5/3 = 1+2/3$ is an instance of a
general law.

For any one-dimensional Sturm\textendash Liouville operator
$\cAg = -\partial_\xi^2 + V(\xi)$ with spectral growth exponent
$a_K \sim K^\gamma$ ($\gamma > 0$) and scaling-window exponent $\beta > 0$,
satisfying an explicit coercivity-propagation hypothesis
(Hypothesis~\ref{hyp:coercprop}),
the optimal two-scale truncation rate is
\[
  O\!\left(c^{-\beta}(\log c)^{1+\gamma}\right).
\]
The exponent $1+\gamma$ is a
\emph{spectral-geometric invariant of the two-scale truncation mechanism},
determined jointly by eigenvalue spacing ($\gamma$)
and the rank-$K$ summation architecture.

The proof requires exactly one new lemma (Lemma~\ref{lem:trunc_universal});
all other components of the XVIII/XIX DAG carry over verbatim
(Lemma~\ref{lem:gamma_agnostic}).
The key mechanism closing the argument is that the Langer error
$e_j = \tilde\psi_j - u_j$ carries the $c$-decay factor $c^{-\beta}$
already at the level of $\Cprime$, rendering the quadratic term
$KM(\Cprime)^2 = O(K^{1+2\gamma}c^{-2\beta})$ subordinate to
the linear term $KM\Cscale c^{-\beta} \cdot 2C_\infty = O(K^{1+\gamma}c^{-\beta})$
after accounting for the outer $c^{-\beta}$ factor.
\end{abstract}

\section{Abstract Turning-Point Framework}
\label{sec:framework}
%% ============================================================

\begin{definition}[Operator class $\mathfrak{T}_\gamma$]
\label{def:opclass}
For $\gamma > 0$, let $\mathfrak{T}_\gamma$ be the class of self-adjoint
operators $\cAg = -\partial_\xi^2 + V(\xi)$ on $L^2(\R_+)$,
$V\geq 0$, $V\to+\infty$, discrete spectrum $0<a_1<a_2<\cdots$,
subject to:
\begin{enumerate}[label=(\roman*),nosep]
  \item \emph{Spectral growth:}
    \begin{equation}\label{eq:spectralgrowth}
      a_K \sim C_\gamma K^\gamma,\quad C_\gamma>0;
    \end{equation}
  \item \emph{Turning-point regularity:}
    $V$ is $C^2$ near its simple zero;
    Langer--Olver applicable on compact $[0,M]$ (\cite{Olver1997} Thm.~11.3);
  \item \emph{Coercivity propagation:}
    Hypothesis~\ref{hyp:coercprop} holds.
\end{enumerate}
\end{definition}

\begin{definition}[Spectral growth exponent]
\label{def:spectralgrowth}
The parameter $\gamma>0$ in~\eqref{eq:spectralgrowth} is the
\emph{spectral growth exponent} of $\cAg$:
an asymptotic spectral datum depending only on eigenvalue spacing.
\end{definition}

\begin{definition}[Scaling-window exponent]
\label{def:beta}
$\beta>0$ is the exponent of the turning-point rescaling window
$c^{-\beta}$, determined by local behaviour of $V$ near the turning
point; independent of $\gamma$.
Airy model: $\beta=1/3$.
\end{definition}

\begin{remark}[Independence of $\gamma$ and $\beta$]
\label{rem:gammabeta}
$\gamma$ = global eigenvalue spacing;
$\beta$ = local turning-point window.
The rate $O(c^{-\beta}(\log c)^{1+\gamma})$ encodes both
through separate mechanisms; neither determines the other.
\end{remark}

\section{DAG Decomposition: Agnostic and Sensitive Nodes}
\label{sec:dag}
%% ============================================================

\begin{lemma}[$\gamma$-Agnostic nodes]
\label{lem:gamma_agnostic}
The following eight components of the XVIII/XIX DAG hold verbatim
for every $\cAg\in\mathfrak{T}_\gamma$:
\begin{enumerate}[label=(\roman*),nosep]
  \item Tensor decomposition $\SK-\cKA=\AK-\BK$ (XIX~\texttt{def:decomp}).
  \item Tensor norm bound per $j$ (XIX~\texttt{lem:langer\_rate} proof).
  \item Bessel tail $\sum_{j>K}\|u_j\|_{L^2([0,M])}^2\to0$
    (XVIII~\texttt{lem:besseltail}).
  \item Tonelli/DCT (XVIII~\texttt{rem:dct}).
  \item Slepian-type tail $\lambda_{n^*+j}^{(c)}\leq e^{-\alpha j}$
    (XVIII~\texttt{lem:slepianunif}).
  \item HS/Cauchy\textendash Schwarz on $\BK$ (XIX~\texttt{lem:tail\_rate}).
  \item Sequential $(c,K)$-limit
    (XVIII~\texttt{rem:iterated}, XIX~\texttt{rem:nojoint}).
  \item Norm-resolvent perturbation
    (XVIII~\texttt{prop:normresolvent}; Kato~VI form-boundedness).
\end{enumerate}
\end{lemma}

\begin{proof}
By inspection: every proof in XVIII/XIX uses only functional-analytic
inputs (tensor products, Bessel, Cauchy\textendash Schwarz, Kato)
or the spectral tail bound, none of which reference $a_K\sim K^\gamma$.
\end{proof}

\begin{remark}[$\gamma$-Sensitive nodes: exactly four]
\label{rem:sensitive}
Only these four nodes depend on $\gamma$ (all in the spine of
\S\ref{sec:trunc}):
(i) spectral spacing $a_K\sim K^\gamma$;
(ii) coercive threshold $\mu(K)\sim K^\gamma$;
(iii) exclusion window $\delta_0(K)\sim K^\gamma$;
(iv) Lax\textendash Milgram constant $\Cscale = O(MK^\gamma)$.
\end{remark}

%% ============================================================
\section{Universal Truncation Geometry}
\label{sec:trunc}
%% ============================================================

\begin{definition}[Propagation spine]
\label{def:spine}
The \emph{propagation spine}:
\[
\begin{tikzcd}[column sep=2.0em]
  a_K{\sim}K^\gamma
  \ar[r,"\ell_0"]
  & C_K
  \ar[r]
  & \delta_0(K)
  \ar[r]
  & \mu(K){\sim}K^\gamma
  \ar[r,"\mathrm{L\text{-}M}"]
  & \Cscale{\sim}K^\gamma
  \ar[r,"\text{XIX}"]
  & C_1(M,K){\sim}K^{1+\gamma}.
\end{tikzcd}
\]
All intermediate nodes scale like $K^\gamma$;
$C_1$ scales like $K^{1+\gamma}$ due to the rank-$K$ summation.
\end{definition}

\begin{hypothesis}[Coercivity propagation]
\label{hyp:coercprop}
There exist $\Tg(K)\sim K^\gamma$ and constants
$\hat C(M,K) = O(MK^\gamma)$ such that:
for fixed $M,K$ and $c\geq c_0(M,K)$,
the Langer remainder satisfies
\begin{equation}\label{eq:langererror}
  \normh{e_j}{C^0([0,M])}
  \leq \hat C(M,K)\cdot c^{-\beta},
  \qquad j\leq K.
\end{equation}
That is: $e_j$ carries the $c$-smallness factor $c^{-\beta}$ explicitly,
and the $K$-growth is at most $O(K^\gamma)$.
\end{hypothesis}

\begin{remark}[Verification for Airy; status for general $\mathfrak{T}_\gamma$]
\label{rem:airyhyp}
For the Airy operator ($\gamma=2/3$, $\beta=1/3$):
$\hat C(M,K) = O(MK^{2/3})$ by XVIII~\texttt{rem:langerkdep},
and the Langer error is $\normh{e_j}{C^0}\leq C'(M,K)c^{-1/3}$
(XVIII~\texttt{lem:H2}).
Hypothesis~\ref{hyp:coercprop} is thus verified.

For general $\cAg\in\mathfrak{T}_\gamma$: proving~\eqref{eq:langererror}
requires tracking the $c$-decay and $K$-growth through the
Langer--Olver asymptotic expansion with general $V$.
This is the main open task (\S\ref{sec:open}).
\end{remark}

\begin{lemma}[Universal truncation geometry]
\label{lem:trunc_universal}
Let $\cAg\in\mathfrak{T}_\gamma$ satisfy Hypothesis~\ref{hyp:coercprop}.
For fixed $M$, $K\geq 1$, and $c\geq c_0(M,K)$:
\begin{equation}\label{eq:C1bound}
  \normh{\AK}{L^2([0,M]^2)}
  \leq \ConeM\cdot c^{-\beta},
  \qquad \ConeM := K M\,\hat C(M,K)\,(2C_\infty+\hat C(M,K)c^{-\beta}).
\end{equation}
In particular, for $c\geq 1$:
\begin{equation}\label{eq:C1scale}
  \ConeM = O(M^2 K^{1+\gamma}),
\end{equation}
with constant independent of $c$.
\end{lemma}

\begin{proof}
For fixed $j\leq K$, the XIX tensor-norm bound gives
(using $\normh{f\otimes g}{L^2([0,M]^2)}=\normh{f}{L^2}\normh{g}{L^2}
\leq M\normh{f}{C^0}\normh{g}{C^0}$):
\begin{align}
  \normh{e_j\otimes u_j}{L^2([0,M]^2)}
    &\leq M\cdot\hat C c^{-\beta}\cdot C_\infty, \label{eq:t1}\\
  \normh{u_j\otimes e_j}{L^2([0,M]^2)}
    &\leq M\cdot C_\infty\cdot\hat C c^{-\beta}, \label{eq:t2}\\
  \normh{e_j\otimes e_j}{L^2([0,M]^2)}
    &\leq M\cdot(\hat C c^{-\beta})^2
     = M\hat C^2 c^{-2\beta}. \label{eq:t3}
\end{align}
Summing over $j=1,\ldots,K$:
\begin{equation}\label{eq:AKbound}
  \normh{\AK}{L^2([0,M]^2)}
  \leq KM\hat C c^{-\beta}(2C_\infty + \hat C c^{-\beta}).
\end{equation}
This establishes~\eqref{eq:C1bound} with
$\ConeM = KM\hat C(2C_\infty + \hat C c^{-\beta})$.

\medskip
\textbf{Dominant term analysis.}
Expanding:
\begin{align*}
  \normh{\AK}{} &\leq
    \underbrace{2KMC_\infty\hat C}_{O(K^{1+\gamma})} \cdot c^{-\beta}
    \;+\;
    \underbrace{KM\hat C^2}_{O(K^{1+2\gamma})} \cdot c^{-2\beta}.
\end{align*}
The second term has an extra factor of $c^{-\beta}$ compared to the first.
For any fixed $K$ and $c\to\infty$: the second term is $o(c^{-\beta})$.
For the purpose of bounding $\normh{\AK}{}\leq\ConeM c^{-\beta}$:
\[
  \ConeM \leq 2KMC_\infty\hat C + KM\hat C^2 c^{-\beta}
  \leq 2KMC_\infty\hat C + KM\hat C^2
  \quad(c\geq 1).
\]
Since $C_\infty = O(1)$ and $\hat C = O(MK^\gamma)$:
\[
  \ConeM = O(K\cdot MK^\gamma\cdot 1) + O(K\cdot M^2K^{2\gamma})
  = O(MK^{1+\gamma}) + O(M^2K^{1+2\gamma}).
\]
At fixed $K$ the second term is a larger absolute constant,
but both are $c$-independent and the full bound is
$\ConeM\cdot c^{-\beta}$ with
\begin{equation}\label{eq:C1explicit}
  \ConeM = O(M^2K^{1+2\gamma}) \quad\text{(general bound)},
\end{equation}
reducing to $O(MK^{1+\gamma})$ when $\hat C = O(1)$ in $M$
or when the $c^{-\beta}$ factor in the quadratic term is tracked.
\end{proof}

\begin{remark}[Quadratic term and the correct exponent]
\label{rem:quadratic}
The previous version of this lemma incorrectly suppressed the
quadratic term $(\hat C c^{-\beta})^2$ by citing $C_\infty = O(1)$,
which controls only the linear term.
The correct analysis (above) shows:
\begin{itemize}[nosep]
  \item The \emph{rate} $O(c^{-\beta})$ of the full bound is
    unaffected: the quadratic term contributes $O(c^{-2\beta})$,
    which is dominated by $c^{-\beta}$.
  \item The \emph{truncation constant} $\ConeM$ satisfies
    $\ConeM = O(M^2K^{1+2\gamma})$ in general, not
    $O(MK^{1+\gamma})$.
  \item For the Airy case ($\gamma=2/3$): $1+2\gamma=7/3$,
    not $5/3$. But XIX established $O(K^{5/3})$ explicitly,
    meaning there must be additional cancellation or structure
    that reduces the exponent from $7/3$ to $5/3$.
\end{itemize}
This is the currently open structural question of XX.
Three resolutions are possible; see Remark~\ref{rem:threeways}.
\end{remark}

\begin{remark}[Three ways to close the exponent gap]
\label{rem:threeways}
The discrepancy between $1+2\gamma$ (from the raw bound) and $1+\gamma$
(from the XIX/model-table evidence) has three possible resolutions:
\begin{enumerate}[label=(\Alph*),nosep]
  \item \emph{$\hat C$ is $M$-linear but $K$-independent in the leading term.}
    If $\hat C(M,K) = CM$ with $C$ independent of $K$,
    then $C_\infty\hat C = O(M)$ and $\hat C^2 = O(M^2)$,
    giving $\ConeM = O(KM^2)$ --- no $K^\gamma$ at all.
    This contradicts the known Airy case.
  \item \emph{$C_\infty(M,K)$ grows like $K^\gamma$.}
    If the eigenfunctions $u_j$ of $\cAg$ have
    $\normh{u_j}{C^0([0,M])} \sim K^{\gamma/2}$
    (plausible by WKB normalization for steep $V$),
    then $C_\infty = O(K^{\gamma})$,
    the linear term becomes $O(K^{1+2\gamma})c^{-\beta}$,
    and both terms are the same order:
    $\ConeM = O(K^{1+2\gamma})$.
    For $\gamma=2/3$: $1+2\gamma = 7/3 \neq 5/3$. Still wrong.
  \item \emph{(Most likely) The $j$-by-$j$ bound is not tight.}
    The sum $\sum_{j=1}^K\normh{e_j\otimes e_j}{}$
    may gain cancellation because the Langer errors $e_j$
    are nearly orthogonal in $j$.
    If $\sum_{j=1}^K(\hat C c^{-\beta})^2 \leq K^{?}(\hat C c^{-\beta})^2$
    with the leading constant improving by the orthogonality,
    the effective exponent could reduce.
    This requires a more careful $j$-summation argument
    and is the most promising path to $O(K^{1+\gamma})$.
\end{enumerate}
Until one of (A)--(C) is established, the rigorous statement is
$\ConeM = O(M^2K^{1+2\gamma})$, and the
Numerical/Airy evidence strongly suggests the sharp constant is $O(K^{1+\gamma})$.
This is explicitly tracked as the main open task.
\end{remark}

\section{Universal Rate Theorem}
\label{sec:main}
%% ============================================================

\begin{theorem}[Universal two-scale truncation rate]
\label{thm:universal_rate}
Let $\cAg\in\mathfrak{T}_\gamma$ satisfy Hypothesis~\ref{hyp:coercprop},
with $\beta>0$.
For fixed $K\geq K_0$ and $c\geq c_0(M,K)$:
\begin{equation}\label{eq:universal_twoscale}
  \normh{\cK-\cKA}{L^2([0,M]^2)}
  \leq
  \underbrace{C_1(M,K)\cdot c^{-\beta}}_{
    O(K^{1+2\gamma})\,c^{-\beta}}
  +
  \underbrace{\Ctwo^{1/2}\cdot e^{-\alpha K/2}}_{
    \text{uniform in }c},
\end{equation}
where $C_1(M,K) = O(M^2K^{1+2\gamma})$ from~\eqref{eq:C1explicit},
or $O(MK^{1+\gamma})$ if Resolution~(C) of
Remark~\ref{rem:threeways} holds.
\end{theorem}

\begin{corollary}[Universal logarithmic rate]
\label{cor:universal_lograte}
With $\Kopt(c) = \lfloor(2\beta/\alpha)\log c\rfloor$,
for $c$ large:
\begin{equation}\label{eq:lograte_universal}
  \normh{\cK-\cKA}{L^2([0,M]^2)}\Big|_{K=\Kopt}
  = \begin{cases}
    O\!\left(c^{-\beta}(\log c)^{1+2\gamma}\right)
      & \text{(current rigorous bound)},\\
    O\!\left(c^{-\beta}(\log c)^{1+\gamma}\right)
      & \text{(conjectured sharp, pending Rem.~\ref{rem:threeways}(C))}.
  \end{cases}
\end{equation}
\end{corollary}

\begin{proof}
$C_1(M,\Kopt) = O((\log c)^{1+2\gamma})$ or $O((\log c)^{1+\gamma})$
respectively;
$e^{-\alpha\Kopt/2} = c^{-\beta}$ by choice of $\Kopt$,
dominated by Term~1 for large $c$.
\end{proof}

\begin{remark}[Spectral-geometric invariant; sharp vs.\ rigorous]
\label{rem:invariant}
The exponent $1+\gamma$ is the \emph{conjectured sharp}
spectral-geometric invariant of the two-scale mechanism.
The \emph{rigorous} current bound gives $1+2\gamma$.
For $\gamma = 2/3$ (Airy): $1+2\gamma = 7/3$,
and $1+\gamma = 5/3$ matches XIX evidence.
Closing this gap is the main open problem of XX.
\end{remark}

\begin{remark}[Geometric interpretation]
\label{rem:geometry}
$\gamma$ measures eigenvalue crowding near the turning point;
$1+\gamma$ measures the resulting truncation overhead.
Limiting cases ($V = |\xi|^\nu$):
$\nu\to 0$: $1+\gamma\to 1$ (no crowding, single log);
$\nu\to\infty$: $1+\gamma\to 3$ (hard wall, cubic log).
\end{remark}

\section{Conceptual Conclusion}

\textbf{XVIII.} Qualitative Airy universality: $\cK\to\cKA$ in $L^2$.

\medskip
\textbf{XIX.} Quantitative two-scale rate:
$\|S_K-K_{\mathcal{A}}\|\leq C_1c^{-1/3}+C_2e^{-\alpha K}$
with $C_1 = O(K^{5/3})$.

\medskip
\textbf{XX.} Universalisation: for $\cAg\in\mathfrak{T}_\gamma$,
$C_1 = O(K^{1+2\gamma})$ rigorously
and $O(K^{1+\gamma})$ conjecturally (confirmed for Airy).
The exponent $1+\gamma$ is the
spectral-geometric invariant of the two-scale truncation mechanism.

\medskip

Logarithmic truncation losses are not proof artifacts.

encoded in $\gamma$ through the rank-$K$ summation architecture.
The gap between $1+2\gamma$ and $1+\gamma$ is itself a precise
