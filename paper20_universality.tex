  Paper~XX: Universality of Turning-Point Truncation Geometry\\[0.3em]
  \large Spectral-Geometric Invariants, Two-Scale Logarithmic Losses,
  and Structural Obstruction to Rate Improvement}

\author{\textit{prolate-gram-coercivity programme}}
\date{May 2026 --- FROZEN}

\begin{abstract}
For operators $\cAg\in\mathfrak{T}_\gamma$
(spectrally regularized WKB class, Definition~\ref{def:opclass}),
the Langer term $\AK = \AKlin + \AKquad$ satisfies,
under the pointwise Langer scaling law
$\|e_j\|_{C^0}\leq C_0 j^\gamma c^{-\beta}$
(Hypothesis~\ref{hyp:langer_pointwise}),
\[
  \|\AKlin\| \leq C_{\rm lin}\,K^{1+\gamma}c^{-\beta},
  \qquad
  \|\AKquad\| \leq C_{\rm quad}\,K^{1+2\gamma}c^{-2\beta}
\]
for all $K\geq K_0$, $c\geq c_0$, with $c$-admissible constants.
The optimal truncation $\Kopt$ is the unique minimizer of
$F(K;c) := L\,K^{1+\gamma}c^{-\beta} + T\,e^{-\alpha K/2}$
over $K\in[c_1\log c,c_2\log c]$, satisfying
$K^*(c) = \frac{2\beta}{\alpha}\log c + \frac{2\gamma}{\alpha}\log\log c + O(1)$.
For each fixed $\gamma>0$, the diagonal rate
$O(c^{-\beta}(\log c)^{1+\gamma})$
holds for every $\cAg\in\mathfrak{T}_\gamma$ satisfying H1--H2
(Corollary~\ref{cor:universal_lograte}).
The central result of this paper is that this rate is not merely achievable
but \emph{structurally rigid}: within the simultaneous validity of
the Langer scaling law~(H1), the Euler--Maclaurin structural property,
and the substitution stability principle, no asymptotically better rate
is compatible with the two-scale truncation geometry
(Theorem~\ref{thm:rate_obstruction}).
All results are conditional on H1--H2.
\end{abstract}

\begin{remark}[Scope of XX and meaning of universality]
\label{rem:logstatus}
All results are theorems within $\mathfrak{T}_\gamma$;
H1--H2 are imposed axioms, not derived for general $V$.
Universality in this paper has two layers:
\begin{itemize}[nosep]
  \item \textbf{Rate universality:} for each fixed $\gamma>0$,
    the rate $O(c^{-\beta}(\log c)^{1+\gamma})$ holds uniformly
    over all operators in $\mathfrak{T}_\gamma$ satisfying H1--H2.
    No uniformity in $\gamma$ across different operator classes is claimed.
  \item \textbf{Structural rigidity:} the exponent $1+\gamma$ and the
    logarithmic scale $\log c$ are fixed by the interplay of
    Langer scaling, EM growth, and tail decay;
    no improvement is possible without breaking this structure
    (Theorem~\ref{thm:rate_obstruction}).
\end{itemize}
\end{remark}

\begin{remark}[$c$-admissible constants]
\label{rem:cadmissible}
A constant is \emph{$c$-admissible} if it is independent of
the approximation parameter $c$.
Every constant in this paper falls under exactly one of
four declared sources:
\begin{itemize}[nosep]
  \item \textbf{Axiomatic:} $C_0$, $C_\infty$ by H1--H2.
  \item \textbf{Analytic:} $C_\sigma = \frac{|B_2|}{12}\sigma(\sigma-1)$,
    function of exponent $\sigma$ only.
  \item \textbf{Derived:} products and finite sums of $c$-admissible
    constants (Lemma~\ref{lem:cadm_closure}).
  \item \textbf{Operator-intrinsic:} $C_2(\alpha)$,
    $c$-admissible by definition.
\end{itemize}
\end{remark}

\begin{lemma}[$c$-admissible closure]
\label{lem:cadm_closure}
The class of $c$-admissible constants is closed under
finite products and finite sums.
\end{lemma}
\begin{proof}
$A,B$ independent of $c$ $\Rightarrow$ $AB$ and $A+B$ independent of $c$.
\end{proof}

\begin{proposition}[Substitution Stability Principle]
\label{prop:substitution_stability}
\emph{Intent.}
This proposition records an inference step, not a new analytical
estimate: substituting a deterministic map $K(c)$ into a
pointwise-valid inequality is admissible as a pointwise composition.

\emph{Setup.}
Let $|E(K,c)| \leq \sum_{n=1}^N C_n\,K^{a_n}c^{-b_n}$
hold for all $K\geq K_0$, $c\geq c_0$, $c$-admissible $C_n$.
Let $K(c)\in[c_1\log c,\,c_2\log c]$ for all large $c$.

\begin{itemize}[nosep]
  \item[(i)] \textbf{Pointwise substitution.}
    $|E(K(c),c)|\leq\sum_n C_n(K(c))^{a_n}c^{-b_n}$;
    no uniform-in-$K$ structure is used.
  \item[(ii)] \textbf{Window control via monotonicity.}
    Each $K\mapsto K^{a_n}$ ($a_n>0$) is monotone increasing;
    for $K(c)\in[c_1\log c, c_2\log c]$:
    \[
      \sum_n C_n (K(c))^{a_n} c^{-b_n}
      \leq \sum_n C_n (c_2\log c)^{a_n} c^{-b_n}.
    \]
    Uniform bound via monotonicity only; no additional uniformity assumed.
  \item[(iii)] \textbf{Asymptotic order.}
    $(K(c))^{a_n}=O(\log^{a_n}c)$;
    precise correction by~\eqref{eq:Kstar_power}.
  \item[(iv)] \textbf{Dominance.}
    Asymptotic comparisons by
    Propositions~\ref{prop:asymp_order} and~\ref{prop:scale_algebra}.
\end{itemize}
\end{proposition}

\begin{proof}
(i) pointwise; (ii) monotonicity + window upper bound;
(iii) sandwich~\eqref{eq:sandwich}; (iv) Props.~\ref{prop:asymp_order}--\ref{prop:scale_algebra}.
\end{proof}

\section{Asymptotic Scale}
\label{sec:asymp}

\begin{definition}[Asymptotic scale]
\label{def:scale}
$\mathcal{S}=\{c^{-s}(\log c)^t : s>0,t\in\R\}$;
$c^{-s_1}(\log c)^{t_1}\succ c^{-s_2}(\log c)^{t_2}$
iff $s_1<s_2$ or ($s_1=s_2$, $t_1>t_2$).
The relation $\prec$ is transitive: it is the lexicographic order
on $(s,t)\in\R^2$; composition of dominance relations closed under $\mathcal{S}$.
\end{definition}

\begin{proposition}[Scale algebra]
\label{prop:scale_algebra}
$f\cdot g\in\mathcal{S}$ with exponents added;
$f/g\to\infty\Rightarrow f+g=O(f)$.
\end{proposition}

\begin{proposition}[Layer dominance]
\label{prop:asymp_order}
With $K=K^*(c)$, $n\geq 2$:
$c^{-n\beta}(\log c)^{1+n\gamma}\prec c^{-\beta}(\log c)^{1+\gamma}$.
\end{proposition}

\section{Truncation Geometry}
\label{sec:trunc}

\begin{lemma}[Universal truncation geometry]
\label{lem:trunc_universal}
Under H1--H2, for all $K\geq 1$, $c\geq c_0$:
\begin{align}
  \normh{\AKlin}{}
    &\leq C_{\rm lin}K^{1+\gamma}c^{-\beta}
    +C_{\sigma,1}K^\gamma c^{-\beta},
    \label{eq:AKlin}\\
  \normh{\AKquad}{}
    &\leq C_{\rm quad}K^{1+2\gamma}c^{-2\beta}
    +C_{\sigma,2}K^{2\gamma}c^{-2\beta},
    \label{eq:AKquad}
\end{align}
all constants $c$-admissible.
\end{lemma}

\begin{proof}
Euler--Maclaurin:
\begin{equation}\label{eq:EM}
  \sum_{j=1}^K j^\sigma
  =\frac{K^{1+\sigma}}{1+\sigma}+\frac{1}{2}K^\sigma+R_\sigma(K),
\end{equation}
\begin{equation}\label{eq:EM_remainder}
  |R_\sigma(K)|\leq C_\sigma K^{\sigma-1},\quad
  C_\sigma=\tfrac{|B_2|}{12}\sigma(\sigma-1),\quad B_2=\tfrac{1}{6}.
\end{equation}
Multiply by $c$-admissible prefactors; close by Lemma~\ref{lem:cadm_closure}.
\end{proof}

\begin{remark}[EM structural property]
\label{rem:EM_growth}
Uniform structural property of all EM remainders in this paper:
\begin{itemize}[nosep]
  \item $|R_\sigma(K)|\leq C_\sigma K^{\sigma-1}$;
    exponent $\sigma-1$ strictly below leading $1+\sigma$.
  \item $C_\sigma$ independent of $K,c$.
  \item $K$ appears only as explicit power, never inside constants.
  \item Each EM expansion applied once per sum, never iterated
    or composed; all occurrences are independent instances.
\end{itemize}
This property is one of the three structural components
(S2) in Theorem~\ref{thm:rate_obstruction}.
\end{remark}

\section{Variational Optimal Truncation}
\label{sec:kopt}

\begin{lemma}[$K^*(c)\to\infty$]
\label{lem:kstar_to_infty}
$K^*(c)\to\infty$ as $c\to\infty$.
\end{lemma}

\begin{lemma}[Sandwich]
\label{lem:loglog}
With $c_1=2\beta/\alpha$, $c_2=2(\beta+1)/\alpha$:
\begin{equation}\label{eq:sandwich}
  c_1\log c\leq K^*(c)\leq c_2\log c,
  \qquad \log K^*(c)=\log\log c+O(1).
\end{equation}
Proof: sign analysis of $F'$; no convexity used.
\end{lemma}

\begin{lemma}[Variational $K^*$]
\label{lem:kopt_variational}
\textup{(i)} On $[c_1\log c,c_2\log c]$, $F''(K;c)>0$;
unique $K^*(c)$ with
\begin{equation}\label{eq:FOC}
  L(1+\gamma){K^*}^\gamma c^{-\beta}=\tfrac{\alpha}{2}Te^{-\alpha K^*/2}.
\end{equation}

\textup{(ii)} Iterated asymptotics:
\begin{equation}\label{eq:kopt_iterated}
  K^*(c)=\frac{2\beta}{\alpha}\log c+\frac{2\gamma}{\alpha}\log\log c+O(1).
\end{equation}

\textup{(ii')} $\lambda(c)=\frac{2\gamma}{\alpha K^*(c)}\to 0$:
self-consistency certificate only.

\textup{(iii)} $F(K^*;c)=O(c^{-\beta}(\log c)^{1+\gamma})$.
\end{lemma}

\begin{lemma}[Stability]
\label{lem:stability_kopt}
$O(1)$ perturbation of~\eqref{eq:FOC} shifts $K^*$ by $O(1)$.
\begin{proof}
$K\mapsto\beta\log c+\gamma\log K+C_1$ is strictly increasing
for $K\geq 2\gamma/\alpha$ (holds for large $c$).
Implicit function: $\frac{dK^*}{dC_1}=\frac{1}{\alpha/2-\gamma/K^*}=O(1)$.
\end{proof}
\end{lemma}

\section{Diagonal Substitution}
\label{sec:diagonal}
%% ============================================================

\begin{lemma}[Diagonal substitution]
\label{lem:diagonal_sub}
Applies Proposition~\ref{prop:substitution_stability} at $K(c)=K^*(c)$.

\emph{$h(c)$-control.}
$h(c):=K^*(c)/(\tfrac{2\beta}{\alpha}\log c)\to 1$;
$|h(c)-1|=O(\log\log c/\log c)$.
For large $c$: $h(c)\in[1/2,3/2]$;
exponentiation $h(c)^{1+\gamma}$ uniformly Lipschitz on $[1/2,3/2]$ (fixed $\gamma$).
By MVT, constant bounded by $(1+\gamma)2^\gamma$,
finite on every compact $[\gamma_0,\gamma_1]\subset(0,\infty)$.
Uniformity fails for large $\gamma$ via $(1+\gamma)2^\gamma\to\infty$;
no claim for $\gamma\to\infty$.
Hence
\begin{equation}\label{eq:Kstar_power}
  (K^*(c))^{1+\gamma}
  =\Bigl(\tfrac{2\beta}{\alpha}\log c\Bigr)^{1+\gamma}
  \Bigl(1+O\Bigl(\tfrac{\log\log c}{\log c}\Bigr)\Bigr).
\end{equation}

\emph{Conclusions:}
$\normh{\cK-\cKA}{}=O(c^{-\beta}(\log c)^{1+\gamma})$; linear term dominant.
\end{lemma}

\begin{theorem}[Two-scale rate]
\label{thm:universal_rate}
For each fixed $\gamma>0$, $\cAg\in\mathfrak{T}_\gamma$,
all $K\geq K_0$, $c\geq c_0$:
\[
  \normh{\cK-\cKA}{}
  \leq C_{\rm lin}K^{1+\gamma}c^{-\beta}
  +C_{\rm quad}K^{1+2\gamma}c^{-2\beta}
  +C_2(\alpha)^{1/2}e^{-\alpha K/2},
\]
all constants $c$-admissible.
No uniformity in $\gamma$ across operator classes is claimed.
\end{theorem}

\begin{corollary}[Universal logarithmic rate]
\label{cor:universal_lograte}
For each fixed $\gamma>0$, at $K=\lfloor K^*(c)\rceil$:
\[
  \normh{\cK-\cKA}{}=O(c^{-\beta}(\log c)^{1+\gamma}).
\]
Universality: rate holds for every $\cAg\in\mathfrak{T}_\gamma$
at fixed $\gamma$; constants depend on $\gamma$, not on $c$.
\end{corollary}

\section{Structural Obstruction to Rate Improvement}
\label{sec:obstruction}
%% ============================================================

\begin{theorem}[Structural Obstruction]
\label{thm:rate_obstruction}
Let $\cAg\in\mathfrak{T}_\gamma$ satisfy H1--H2 and the
EM structural property (Remark~\ref{rem:EM_growth}),
and let $\normh{\cK-\cKA}{}$ admit the fixed-parameter
decomposition~\eqref{eq:AKlin}--\eqref{eq:AKquad} with $c$-admissible constants.
Assume that for some choice of $K(c)$,
\begin{equation}\label{eq:improved_rate_hyp}
  \normh{\cK-\cKA}{} = o\!\big(c^{-\beta}(\log c)^{1+\gamma}\big)
  \quad\text{as }c\to\infty.
\end{equation}
Then at least one of the following structural components must fail:
\begin{itemize}[nosep]
  \item[\textnormal{(S1)}] the Langer scaling law H1;
  \item[\textnormal{(S2)}] the EM structural bound
    $|R_\sigma(K)|\leq C_\sigma K^{\sigma-1}$ with $c$-admissible $C_\sigma$;
  \item[\textnormal{(S3)}] the substitution stability principle
    (Proposition~\ref{prop:substitution_stability}).
\end{itemize}
Equivalently: within the simultaneous validity of (S1)--(S3),
the rate $c^{-\beta}(\log c)^{1+\gamma}$ is the minimal asymptotic
scale compatible with the two-scale truncation geometry
of $\mathfrak{T}_\gamma$.
\end{theorem}

\begin{proof}
Assume (S1)--(S3) all hold. We derive a contradiction with~\eqref{eq:improved_rate_hyp}.

\textbf{Step 1 (Three-term balance).}
The fixed-parameter bound has three contributions:
$c^{-\beta}K^{1+\gamma}$ (linear/EM),
$c^{-2\beta}K^{1+2\gamma}$ (quadratic),
$e^{-\alpha K/2}$ (tail).
Any improvement requires all three to be
$o(c^{-\beta}(\log c)^{1+\gamma})$ simultaneously.

\textbf{Step 2 (Lower rigidity: EM constraint).}
By (S2), the EM remainder satisfies
$|R_\sigma(K)|\leq C_\sigma K^{\sigma-1}$;
under (S1) the leading linear term satisfies
$c^{-\beta}K^{1+\gamma}\sim C_{\rm lin}c^{-\beta}K^{1+\gamma}$.
Reducing $K(c)$ below $C\log c$ for any $C<c_1=2\beta/\alpha$
forces the linear term to scale as
$o(c^{-\beta}(\log c)^{1+\gamma})$,
but then $e^{-\alpha K/2}\sim c^{-\alpha C/2}$,
which dominates $c^{-\beta}(\log c)^{1+\gamma}$
if $\alpha C/2<\beta$, i.e., $C<2\beta/\alpha=c_1$.
Therefore reducing $K$ below logarithmic scale breaks tail dominance.

\textbf{Step 3 (Upper rigidity: tail constraint).}
To keep the exponential tail $e^{-\alpha K/2}\prec c^{-\beta}(\log c)^{1+\gamma}$,
one must have $K(c)\gtrsim c_1\log c$ (from $e^{-\alpha K/2}=O(c^{-\beta})$).
Increasing $K$ above $c_2\log c$ forces the linear term
$c^{-\beta}K^{1+\gamma}\succ c^{-\beta}(\log c)^{1+\gamma}$,
which contradicts improvement.

\textbf{Step 4 (Uniqueness of balance).}
Steps~2--3 together pin
$K(c)\in[c_1\log c, c_2\log c]$,
which is precisely the sandwich of Lemma~\ref{lem:loglog}.
Within this window the leading term satisfies
$c^{-\beta}(K(c))^{1+\gamma}\asymp c^{-\beta}(\log c)^{1+\gamma}$
(by Proposition~\ref{prop:substitution_stability}(ii)--(iii));
no $K(c)$ in the window can produce $o(c^{-\beta}(\log c)^{1+\gamma})$.

\textbf{Step 5 (Contradiction).}
Any $K(c)$ satisfying both tail and EM constraints lies in
$[c_1\log c, c_2\log c]$ (Step~3) and yields
linear-term contribution $\asymp c^{-\beta}(\log c)^{1+\gamma}$ (Step~4).
Hence~\eqref{eq:improved_rate_hyp} is impossible under (S1)--(S3);
at least one structural component must fail.
\end{proof}

\begin{remark}[Scope of obstruction]
\label{rem:obstruction_scope}
Theorem~\ref{thm:rate_obstruction} is an obstruction within the
method class defined by (S1)--(S3); it is not a lower bound
in the sense of O3a (which would require a matching construction).
The theorem asserts: \emph{any method operating within this
structural framework cannot exceed the rate} $c^{-\beta}(\log c)^{1+\gamma}$.
O3a remains open and would provide the complementary lower bound;
its resolution is independent of the upper-bound programme.
\end{remark}
