\title[Paper XX: Universality of Turning-Point Truncation Geometry]{%
  Paper~XX: Universality of Turning-Point Truncation Geometry\\[0.3em]
  \large Spectral-Geometric Invariants, Two-Scale Logarithmic Losses,
  and Structural Obstruction to Rate Improvement}

\author{\textit{prolate-gram-coercivity programme}}
\date{May 2026 --- FROZEN}

\begin{abstract}
For operators $\cAg\in\mathfrak{T}_\gamma$
(spectrally regularized WKB class, Definition~\ref{def:opclass}),
the Langer term $\AK = \AKlin + \AKquad$ satisfies,
under the pointwise Langer scaling law
$\|e_j\|_{C^0}\leq C_0 j^\gamma c^{-\beta}$
(Hypothesis~\ref{hyp:langer_pointwise}),
\[
  \|\AKlin\| \leq C_{\rm lin}\,K^{1+\gamma}c^{-\beta},
  \qquad
  \|\AKquad\| \leq C_{\rm quad}\,K^{1+2\gamma}c^{-2\beta}
\]
for all $K\geq K_0$, $c\geq c_0$, with $c$-admissible constants.
The optimal truncation $\Kopt$ is the unique minimizer of
$F(K;c) := L\,K^{1+\gamma}c^{-\beta} + T\,e^{-\alpha K/2}$
over $K\in[c_1\log c,c_2\log c]$, satisfying
$K^*(c) = \frac{2\beta}{\alpha}\log c + \frac{2\gamma}{\alpha}\log\log c + O(1)$.
For each fixed $\gamma>0$, the diagonal rate
$O(c^{-\beta}(\log c)^{1+\gamma})$
holds for every $\cAg\in\mathfrak{T}_\gamma$ satisfying H1--H2
(Corollary~\ref{cor:universal_lograte}).
The central result of this paper is that this rate is not merely achievable
but \emph{structurally rigid}: within the simultaneous validity of
the Langer scaling law~(H1), the Euler--Maclaurin structural property,
and the substitution stability principle, no asymptotically better rate
is compatible with the two-scale truncation geometry for any admissible
truncation-based representation
(Theorem~\ref{thm:rate_obstruction}).
The rigidity is supported by decomposition stability
(Lemma~\ref{lem:decomp_stability}),
balance fixed-point uniqueness
(Lemma~\ref{lem:balance_fixpoint}),
H1 exponent rigidity
(Lemma~\ref{lem:H1_rigidity}),
and explicit scope remarks
(Remarks~\ref{rem:non_tautology}--\ref{rem:method_vs_operator}).
All results are conditional on H1--H2.
\end{abstract}

\begin{remark}[Scope of XX and meaning of universality]
\label{rem:logstatus}
All results are theorems within $\mathfrak{T}_\gamma$;
H1--H2 are imposed axioms, not derived for general $V$.
Universality in this paper has two layers:
\begin{itemize}[nosep]
  \item \textbf{Rate universality:} for each fixed $\gamma>0$,
    the rate $O(c^{-\beta}(\log c)^{1+\gamma})$ holds uniformly
    over all operators in $\mathfrak{T}_\gamma$ satisfying H1--H2.
    No uniformity in $\gamma$ across different operator classes is claimed.
  \item \textbf{Structural rigidity:} the exponent $1+\gamma$ and the
    logarithmic scale $\log c$ are fixed by the interplay of
    Langer scaling, EM growth, and tail decay;
    no improvement is possible within any admissible truncation-based
    representation without breaking this structure
    (Theorem~\ref{thm:rate_obstruction}).
\end{itemize}
\end{remark}

\begin{remark}[$c$-admissible constants]
\label{rem:cadmissible}
A constant is \emph{$c$-admissible} if it is independent of
the approximation parameter $c$.
Every constant in this paper falls under exactly one of
four declared sources:
\begin{itemize}[nosep]
  \item \textbf{Axiomatic:} $C_0$, $C_\infty$ by H1--H2.
  \item \textbf{Analytic:} $C_\sigma = \frac{|B_2|}{12}\sigma(\sigma-1)$,
    function of exponent $\sigma$ only.
  \item \textbf{Derived:} products and finite sums of $c$-admissible
    constants (Lemma~\ref{lem:cadm_closure}).
  \item \textbf{Operator-intrinsic:} $C_2(\alpha)$,
    $c$-admissible by definition.
\end{itemize}
\end{remark}

\begin{lemma}[$c$-admissible closure]
\label{lem:cadm_closure}
The class of $c$-admissible constants is closed under
finite products and finite sums.
\end{lemma}
\begin{proof}
$A,B$ independent of $c$ $\Rightarrow$ $AB$ and $A+B$ independent of $c$.
\end{proof}

\begin{remark}[Two sources of $c$-dependence, strictly separated]

  \item \textbf{Optimization domain:} $c$ in window
    $[c_1\log c, c_2\log c]$ and $K^*(c)$ via FOC.

The formal bridge is Proposition~\ref{prop:substitution_stability}.
\end{remark}

\begin{proposition}[Substitution Stability Principle]
\label{prop:substitution_stability}
\emph{Intent.}
This proposition records an inference step, not a new analytical
estimate: substituting a deterministic map $K(c)$ into a
pointwise-valid inequality is admissible as a pointwise composition.

\emph{Setup.}
Let $|E(K,c)| \leq \sum_{n=1}^N C_n\,K^{a_n}c^{-b_n}$
hold for all $K\geq K_0$, $c\geq c_0$, $c$-admissible $C_n$.
Let $K(c)\in[c_1\log c,\,c_2\log c]$ for all large $c$.

\begin{itemize}[nosep]
  \item[(i)] \textbf{Pointwise substitution.}
    $|E(K(c),c)|\leq\sum_n C_n(K(c))^{a_n}c^{-b_n}$;
    no uniform-in-$K$ structure is used.
  \item[(ii)] \textbf{Window control via monotonicity.}
    Each $K\mapsto K^{a_n}$ ($a_n>0$) is monotone increasing;
    for $K(c)\in[c_1\log c, c_2\log c]$:
    \[
      \sum_n C_n (K(c))^{a_n} c^{-b_n}
      \leq \sum_n C_n (c_2\log c)^{a_n} c^{-b_n}.
    \]
    Uniform bound via monotonicity only; no additional uniformity assumed.
  \item[(iii)] \textbf{Asymptotic order.}
    $(K(c))^{a_n}=O(\log^{a_n}c)$;
    precise correction by~\eqref{eq:Kstar_power}.
  \item[(iv)] \textbf{Dominance.}
    Asymptotic comparisons by
    Propositions~\ref{prop:asymp_order} and~\ref{prop:scale_algebra}.
\end{itemize}
\end{proposition}

\begin{proof}
(i) pointwise; (ii) monotonicity + window upper bound;
(iii) sandwich~\eqref{eq:sandwich}; (iv) Props.~\ref{prop:asymp_order}--\ref{prop:scale_algebra}.
\end{proof}

\section{Asymptotic Scale}
\label{sec:asymp}

\begin{definition}[Asymptotic scale]
\label{def:scale}
$\mathcal{S}=\{c^{-s}(\log c)^t : s>0,t\in\R\}$;
$c^{-s_1}(\log c)^{t_1}\succ c^{-s_2}(\log c)^{t_2}$
iff $s_1<s_2$ or ($s_1=s_2$, $t_1>t_2$).
The relation $\prec$ is transitive: it is the lexicographic order
on $(s,t)\in\R^2$; composition of dominance relations closed under $\mathcal{S}$.
\end{definition}

\begin{proposition}[Scale algebra]
\label{prop:scale_algebra}
$f\cdot g\in\mathcal{S}$ with exponents added;
$f/g\to\infty\Rightarrow f+g=O(f)$.
\end{proposition}

\begin{proposition}[Layer dominance]
\label{prop:asymp_order}
With $K=K^*(c)$, $n\geq 2$:
$c^{-n\beta}(\log c)^{1+n\gamma}\prec c^{-\beta}(\log c)^{1+\gamma}$.
\end{proposition}

\section{Truncation Geometry}
\label{sec:trunc}
%% ============================================================

\begin{lemma}[Universal truncation geometry]
\label{lem:trunc_universal}
Under H1--H2, for all $K\geq 1$, $c\geq c_0$:
\begin{align}
  \normh{\AKlin}{}
    &\leq C_{\rm lin}K^{1+\gamma}c^{-\beta}
    +C_{\sigma,1}K^\gamma c^{-\beta},
    \label{eq:AKlin}\\
  \normh{\AKquad}{}
    &\leq C_{\rm quad}K^{1+2\gamma}c^{-2\beta}
    +C_{\sigma,2}K^{2\gamma}c^{-2\beta},
    \label{eq:AKquad}
\end{align}
all constants $c$-admissible.
\end{lemma}

\begin{proof}
Euler--Maclaurin:
\begin{equation}\label{eq:EM}
  \sum_{j=1}^K j^\sigma
  =\frac{K^{1+\sigma}}{1+\sigma}+\frac{1}{2}K^\sigma+R_\sigma(K),
\end{equation}
\begin{equation}\label{eq:EM_remainder}
  |R_\sigma(K)|\leq C_\sigma K^{\sigma-1},\quad
  C_\sigma=\tfrac{|B_2|}{12}\sigma(\sigma-1),\quad B_2=\tfrac{1}{6}.
\end{equation}
Multiply by $c$-admissible prefactors; close by Lemma~\ref{lem:cadm_closure}.
\end{proof}

\begin{remark}[EM structural property]
\label{rem:EM_growth}
Uniform structural property of all EM remainders in this paper:
\begin{itemize}[nosep]
  \item $|R_\sigma(K)|\leq C_\sigma K^{\sigma-1}$;
    exponent $\sigma-1$ strictly below leading $1+\sigma$.
  \item $C_\sigma$ independent of $K,c$.
  \item $K$ appears only as explicit power, never inside constants.
  \item Each EM expansion applied once per sum, never iterated
    or composed; all occurrences are independent instances.
\end{itemize}
This property is structural component (S2) in
Theorem~\ref{thm:rate_obstruction}.
\end{remark}

\section{Variational Optimal Truncation}
\label{sec:kopt}
%% ============================================================

\begin{lemma}[$K^*(c)\to\infty$]
\label{lem:kstar_to_infty}
$K^*(c)\to\infty$ as $c\to\infty$.
\end{lemma}

\begin{lemma}[Sandwich]
\label{lem:loglog}
With $c_1=2\beta/\alpha$, $c_2=2(\beta+1)/\alpha$:
\begin{equation}\label{eq:sandwich}
  c_1\log c\leq K^*(c)\leq c_2\log c,
  \qquad \log K^*(c)=\log\log c+O(1).
\end{equation}
Proof: sign analysis of $F'$; no convexity used.
\end{lemma}

\begin{lemma}[Variational $K^*$]
\label{lem:kopt_variational}
\textup{(i)} On $[c_1\log c,c_2\log c]$, $F''(K;c)>0$;
unique $K^*(c)$ with
\begin{equation}\label{eq:FOC}
  L(1+\gamma){K^*}^\gamma c^{-\beta}=\tfrac{\alpha}{2}Te^{-\alpha K^*/2}.
\end{equation}

\textup{(ii)} Iterated asymptotics:
\begin{equation}\label{eq:kopt_iterated}
  K^*(c)=\frac{2\beta}{\alpha}\log c+\frac{2\gamma}{\alpha}\log\log c+O(1).
\end{equation}

\textup{(ii')} $\lambda(c)=\frac{2\gamma}{\alpha K^*(c)}\to 0$:
self-consistency certificate only.

\textup{(iii)} $F(K^*;c)=O(c^{-\beta}(\log c)^{1+\gamma})$.
\end{lemma}

\begin{lemma}[Stability]
\label{lem:stability_kopt}
$O(1)$ perturbation of~\eqref{eq:FOC} shifts $K^*$ by $O(1)$.
\begin{proof}
$K\mapsto\beta\log c+\gamma\log K+C_1$ is strictly increasing
for $K\geq 2\gamma/\alpha$ (holds for large $c$).
Implicit function: $\frac{dK^*}{dC_1}=\frac{1}{\alpha/2-\gamma/K^*}=O(1)$.
\end{proof}
\end{lemma}

\section{Diagonal Substitution}
\label{sec:diagonal}
%% ============================================================

\begin{lemma}[Diagonal substitution]
\label{lem:diagonal_sub}
Applies Proposition~\ref{prop:substitution_stability} at $K(c)=K^*(c)$.

\emph{$h(c)$-control.}
$h(c):=K^*(c)/(\tfrac{2\beta}{\alpha}\log c)\to 1$;
$|h(c)-1|=O(\log\log c/\log c)$.
For large $c$: $h(c)\in[1/2,3/2]$;
exponentiation $h(c)^{1+\gamma}$ uniformly Lipschitz on $[1/2,3/2]$ (fixed $\gamma$).
By MVT, constant bounded by $(1+\gamma)2^\gamma$,
finite on every compact $[\gamma_0,\gamma_1]\subset(0,\infty)$.
Uniformity fails for large $\gamma$ via $(1+\gamma)2^\gamma\to\infty$;
no claim for $\gamma\to\infty$.
Hence
\begin{equation}\label{eq:Kstar_power}
  (K^*(c))^{1+\gamma}
  =\Bigl(\tfrac{2\beta}{\alpha}\log c\Bigr)^{1+\gamma}
  \Bigl(1+O\Bigl(\tfrac{\log\log c}{\log c}\Bigr)\Bigr).
\end{equation}

\emph{Conclusions:}
$\normh{\cK-\cKA}{}=O(c^{-\beta}(\log c)^{1+\gamma})$; linear term dominant.
\end{lemma}

\begin{theorem}[Two-scale rate]
\label{thm:universal_rate}
For each fixed $\gamma>0$, $\cAg\in\mathfrak{T}_\gamma$,
all $K\geq K_0$, $c\geq c_0$:
\[
  \normh{\cK-\cKA}{}
  \leq C_{\rm lin}K^{1+\gamma}c^{-\beta}
  +C_{\rm quad}K^{1+2\gamma}c^{-2\beta}
  +C_2(\alpha)^{1/2}e^{-\alpha K/2},
\]
all constants $c$-admissible.
No uniformity in $\gamma$ across operator classes is claimed.
\end{theorem}

\begin{corollary}[Universal logarithmic rate]
\label{cor:universal_lograte}
For each fixed $\gamma>0$, at $K=\lfloor K^*(c)\rceil$:
\[
  \normh{\cK-\cKA}{}=O(c^{-\beta}(\log c)^{1+\gamma}).
\]
Universality: rate holds for every $\cAg\in\mathfrak{T}_\gamma$
at fixed $\gamma$; constants depend on $\gamma$, not on $c$.
\end{corollary}

\section{H1 Exponent Rigidity}
\label{sec:H1rigidity}
%% ============================================================

\begin{lemma}[H1 exponent rigidity]
\label{lem:H1_rigidity}
Let $\widetilde{\mathrm{H1}}$ be any hypothesis of the form
\begin{equation}\label{eq:H1tilde}
  \|e_j\|_{C^0([0,M])} \leq \tilde C_0\, j^{\tilde\gamma} c^{-\tilde\beta},
  \qquad \tilde C_0 \text{ $c$-admissible},
\end{equation}
and let $\widetilde\Phi_1 := \tilde C_{\rm lin} K^{1+\tilde\gamma} c^{-\tilde\beta}$
be the resulting dominant truncation term.

\begin{itemize}[nosep]
  \item[\textup{(i)}] \textbf{Exponent propagation rigidity.}
  The first-order optimality condition~\eqref{eq:FOC} with exponents
  $(\tilde\gamma,\tilde\beta)$ has a unique solution
  \[
    \widetilde K^*(c)
    = \frac{2\tilde\beta}{\alpha}\log c
    + \frac{2\tilde\gamma}{\alpha}\log\log c + O(1),
  \]
  by the same iterated-asymptotics argument as Lemma~\ref{lem:kopt_variational}(ii).
  The resulting diagonal rate satisfies
  \[
    \widetilde\Phi_1(\widetilde K^*(c),c)
    = O\!\big(c^{-\tilde\beta}(\log c)^{1+\tilde\gamma}\big).
  \]
  The map
  \[
    (\tilde\gamma,\tilde\beta) \longmapsto
    \bigl(\text{log-exponent}\;1+\tilde\gamma,\;
    \text{power-exponent}\;\tilde\beta\bigr)
  \]
  is injective on $(0,\infty)^2$.
  Hence distinct exponent pairs yield distinct asymptotic rates;
  the rate $c^{-\beta}(\log c)^{1+\gamma}$ uniquely identifies $(\gamma,\beta)$
  within the power-law class~\eqref{eq:H1tilde}.

  \item[\textup{(ii)}] \textbf{Logarithmic perturbation instability.}
  Suppose instead
  \begin{equation}\label{eq:H1log}
    \|e_j\|_{C^0} \leq C_0\, j^\gamma (\log j)^\delta c^{-\beta},
    \qquad \delta > 0.
  \end{equation}
  Then $\widetilde\Phi_1 = C\,K^{1+\gamma}(\log K)^\delta c^{-\beta}$.
  Under substitution $K = K^*(c)$ via
  Proposition~\ref{prop:substitution_stability}:
  \[
    (\log K^*(c))^\delta
    = (\log\log c + O(1))^\delta
    \sim (\log\log c)^\delta,
  \]
  so the diagonal rate becomes
  \[
    O\!\big(c^{-\beta}(\log c)^{1+\gamma}(\log\log c)^\delta\big).
  \]
  By Proposition~\ref{prop:scale_algebra},
  $(\log\log c)^\delta \to \infty$ for $\delta > 0$,
  so this rate is \emph{strictly larger} than the baseline
  $c^{-\beta}(\log c)^{1+\gamma}$.
  Thus any logarithmic weakening of H1 inflates the rate
  into a strictly higher iterated-logarithmic scale.
\end{itemize}
\end{lemma}

\begin{proof}
\textup{(i)} The FOC~\eqref{eq:FOC} in the form
$\log(1+\tilde\gamma) + \tilde\gamma\log K - \tilde\beta\log c
= \log(\alpha T/2L) - \alpha K/2$
(taking logarithms) shows that $\widetilde K^*(c)$ satisfies the
same implicit equation structure as $K^*(c)$, with $(\gamma,\beta)$
replaced by $(\tilde\gamma,\tilde\beta)$. The iterated-asymptotics
argument of Lemma~\ref{lem:kopt_variational}(ii) applies verbatim
(it uses only the shape of~\eqref{eq:FOC}, not the specific values).
Injectivity: $(1+\tilde\gamma_1,\tilde\beta_1) = (1+\tilde\gamma_2,\tilde\beta_2)$
iff $\tilde\gamma_1=\tilde\gamma_2$ and $\tilde\beta_1=\tilde\beta_2$.

\textup{(ii)} $K^*(c) \sim \frac{2\beta}{\alpha}\log c$
(Lemma~\ref{lem:loglog}), so $\log K^*(c) = \log\log c + O(1)$.
Substitution via Proposition~\ref{prop:substitution_stability}(i)
is pointwise and yields $(\log K^*(c))^\delta \sim (\log\log c)^\delta$.
Strictly larger scale by Proposition~\ref{prop:scale_algebra}
and $\log\log c \to \infty$.
\end{proof}

\begin{remark}[What H1 rigidity does and does not establish]
\label{rem:H1_rigidity_scope}
Lemma~\ref{lem:H1_rigidity} establishes:
\begin{itemize}[nosep]
  \item[(a)] Within the power-law class~\eqref{eq:H1tilde},
    the exponent pair $(\gamma,\beta)$ is \emph{uniquely identified}
    by the rate $c^{-\beta}(\log c)^{1+\gamma}$.
  \item[(b)] Any logarithmic weakening~\eqref{eq:H1log} strictly
    inflates the rate; H1 is \emph{logarithmically stable}.
\end{itemize}
Lemma~\ref{lem:H1_rigidity} does \emph{not} establish:
\begin{itemize}[nosep]
  \item[(c)] That H1 is the weakest hypothesis in any sense beyond the
    power-law class (the \emph{envelope minimality} question, O1).
  \item[(d)] That no hypothesis outside the power-law class can reproduce
    the same rate with a different analytic form.
\end{itemize}
These are the content of Conjecture~\ref{conj:H1_envelope} below.
\end{remark}

\section{Structural Obstruction to Rate Improvement}
\label{sec:obstruction}

\begin{lemma}[Decomposition stability under asymptotic comparison]
\label{lem:decomp_stability}
Let the total bound be
\begin{equation}\label{eq:total_bound}
  \|\cK-\cKA\| \leq \Phi_1(K,c) + \Phi_2(K,c) + \Phi_3(K,c),
\end{equation}
where
\[
  \Phi_1 := C_{\rm lin}K^{1+\gamma}c^{-\beta},\quad
  \Phi_2 := C_{\rm quad}K^{1+2\gamma}c^{-2\beta},\quad
  \Phi_3 := C_2(\alpha)^{1/2}e^{-\alpha K/2},
\]
with $c$-admissible constants, and let $\Phi_1\succ\Phi_2$
(Proposition~\ref{prop:asymp_order}).

If $\|\cK-\cKA\| = o(\Phi_1(K^*(c),c))$, then no cancellation within
the decomposition~\eqref{eq:total_bound} is possible: for any admissible
truncation-based representation compatible with \textnormal{(S1)--(S3)},
one cannot have
$\Phi_1(K(c),c) + \Phi_3(K(c),c)
= o(c^{-\beta}(\log c)^{1+\gamma})$
via mutual cancellation of $\Phi_1$ and $\Phi_3$.
\end{lemma}

\begin{proof}
$\Phi_1,\Phi_3\geq 0$; hence $\Phi_1+\Phi_3\geq\Phi_1\geq 0$.
Cancellation would require $\Phi_3 < 0$, which contradicts
$\Phi_3$ being a norm bound.
\end{proof}

\begin{remark}[Role of decomposition stability]
\label{rem:decomp_role}
Any improvement of the total rate must come from reducing
each dominant term individually.
Under (S1)--(S3) this forces the analysis of $K(c)$
in Lemma~\ref{lem:balance_fixpoint}.
\end{remark}

\begin{lemma}[Balance fixed-point: strict separation outside logarithmic window]
\label{lem:balance_fixpoint}
Define $\Psi(K,c) := \Phi_1(K,c) + \Phi_3(K,c)$.
Under \textnormal{(S1)--(S3)}:
\begin{itemize}[nosep]
  \item[\textup{(i)}] $K(c)=o(\log c)$:
    $\Phi_3\succ c^{-\beta}(\log c)^{1+\gamma}$, so
    $\Psi\succ c^{-\beta}(\log c)^{1+\gamma}$.
  \item[\textup{(ii)}] $K(c)\geq C\log c$, $C>c_2$:
    $\Phi_1\succ c^{-\beta}(\log c)^{1+\gamma}$, so
    $\Psi\succ c^{-\beta}(\log c)^{1+\gamma}$.
  \item[\textup{(iii)}] $K(c)\in[c_1\log c,c_2\log c]$:
    $\Psi\asymp c^{-\beta}(\log c)^{1+\gamma}$;
    no $K(c)$ in the window achieves $o(c^{-\beta}(\log c)^{1+\gamma})$.
  \item[\textup{(iv)}] For every $K(c)$,
    $\Psi(K(c),c)\asymp c^{-\beta}(\log c)^{1+\gamma}$ or strictly larger.
\end{itemize}
\end{lemma}

\begin{proof}
(i) $K=o(\log c)\Rightarrow e^{-\alpha K/2}\succ c^{-\varepsilon}$ all $\varepsilon>0$.
(ii) $(C\log c)^{1+\gamma}c^{-\beta}\succ(\log c)^{1+\gamma}c^{-\beta}$ for $C>c_2$.
(iii) Equation~\eqref{eq:Kstar_power} gives
$(K(c))^{1+\gamma}\asymp(\log c)^{1+\gamma}$; conclude by
Proposition~\ref{prop:substitution_stability}(ii)--(iii).
(iv) Exhaustion of (i)--(iii).
\end{proof}

\begin{remark}[Nature of the fixed point]
\label{rem:fixpoint_nature}
The logarithmic window is the only regime where $\Psi\asymp$
target scale. Outside, $\Psi$ is strictly larger.
The balance point is an exact invariant of
$(\mathrm{H1},\mathrm{EM},\mathrm{tail})$, not a local minimum.
\end{remark}

\begin{theorem}[Structural Obstruction]
\label{thm:rate_obstruction}
Let $\cAg\in\mathfrak{T}_\gamma$ satisfy H1--H2 and the
EM structural property (Remark~\ref{rem:EM_growth}),
and let $\normh{\cK-\cKA}{}$ admit the decomposition~\eqref{eq:total_bound}
with $c$-admissible constants.
For any admissible truncation-based representation compatible with
\textnormal{(S1)--(S3)}, if for some $K(c)$,
\[
  \normh{\cK-\cKA}{} = o\!\big(c^{-\beta}(\log c)^{1+\gamma}\big),
\]
then at least one of \textnormal{(S1)--(S3)} must fail.
Equivalently: within \textnormal{(S1)--(S3)}, the rate
$c^{-\beta}(\log c)^{1+\gamma}$ is the minimal asymptotic scale
compatible with the two-scale truncation geometry of $\mathfrak{T}_\gamma$.
\end{theorem}

\begin{proof}
Step~1 (no cancellation): Lemma~\ref{lem:decomp_stability} reduces
the problem to suppressing $\Psi(K(c),c)$.
Step~2 (outside window): Lemma~\ref{lem:balance_fixpoint}(i)--(ii)
contradicts $\Psi=o(c^{-\beta}(\log c)^{1+\gamma})$ for $K(c)$
outside $[c_1\log c,c_2\log c]$.
Step~3 (inside window): Lemma~\ref{lem:balance_fixpoint}(iii)
contradicts $\Psi=o(c^{-\beta}(\log c)^{1+\gamma})$ for $K(c)$
inside the window.
Step~4 (exhaustion): all $K(c)$ covered; contradiction.
\end{proof}

\begin{remark}[Scope of obstruction]
\label{rem:obstruction_scope}
Theorem~\ref{thm:rate_obstruction} is an obstruction within
$\mathcal{M}(\mathrm{S1,S2,S3})$; not a lower bound in the sense
of O3a. The following three remarks close the standard referee
objections (R1--R3).
\end{remark}

\begin{remark}[Non-tautology: independent analytic origins of $K^*(c)$]
\label{rem:non_tautology}
The optimal scale $K^*(c)$ arises from a balance between two
\emph{analytically independent} contributions:
\begin{itemize}[nosep]
  \item[(i)] \emph{H1-driven algebraic growth} $\Phi_1$: origin in
    Hypothesis~\ref{hyp:langer_pointwise}.
  \item[(ii)] \emph{Spectral tail decay} $\Phi_3$: origin in $\BK$
    alone (Remark~\ref{rem:BK_definition}), entirely independent of H1.
\end{itemize}
Changing H1 (modifying $\gamma,\beta$) shifts $\Phi_1$ but leaves
$\Phi_3$ invariant; the balance point changes accordingly.
The balance is a genuine coupling of two structurally distinct
mechanisms, not a reparametrization of H1.
\end{remark}

\begin{remark}[Constant dependence: uniformity in $c$, explicit in $\gamma$]
\label{rem:constant_dependence}
All constants in Lemma~\ref{lem:balance_fixpoint} are uniform in $c$;
they may depend on $(\gamma,\beta,\alpha,C_{\rm lin},C_2(\alpha))$,
all $c$-admissible.
No dependence on $K(c)$ enters constants after substitution
(Proposition~\ref{prop:substitution_stability}(i) is pointwise).
Dependence on $\gamma$ is explicit ($(1+\gamma)2^\gamma$,
Lemma~\ref{lem:diagonal_sub}); no uniformity in $\gamma\to\infty$ claimed.
\end{remark}

\begin{remark}[Method class vs.~operator class]
\label{rem:method_vs_operator}
Theorem~\ref{thm:rate_obstruction} classifies the method class
$\mathcal{M}(\mathrm{S1,S2,S3}) :=
\{\text{truncation-based representations satisfying (S1)--(S3)}\}$,
not the operator class $\mathfrak{T}_\gamma$.
Methods not admitting the decomposition~\eqref{eq:total_bound}
or not using EM summation are outside scope.
This is the Tauberian structure: (S1)--(S3) are the Tauberian condition;
the rate $c^{-\beta}(\log c)^{1+\gamma}$ is the forced consequence
within that condition. O3a (lower bound for methods outside
$\mathcal{M}$) remains open.
\end{remark}

\begin{conjecture}[H1 envelope minimality --- O1]
\label{conj:H1_envelope}
H1 is the \emph{minimal power-law growth envelope} for the rate
$c^{-\beta}(\log c)^{1+\gamma}$ in the following sense:
there exists no hypothesis of the form
\[
  \|e_j\|_{C^0} \leq C_0\, j^\gamma f(j)\, c^{-\beta}
\]
with $f(j) = o(1)$ as $j\to\infty$ and $f$ monotone
that still yields $\Phi_1(K^*(c),c) = O(c^{-\beta}(\log c)^{1+\gamma})$
with $c$-admissible constants.

\emph{Status.}
The \emph{sufficiency} direction (H1 $\Rightarrow$ rate) is
Theorem~\ref{thm:universal_rate}.
The \emph{necessity} direction (rate $\Rightarrow$ H1-class sharpness)
requires a pointwise lower bound of the form
$\limsup_{j\to\infty} j^{-\gamma}c^\beta\|e_j\|_{C^0} > 0$,
which is operator-specific and not derivable from (S1)--(S3) alone.
For the Airy operator ($\gamma=2/3$, $\beta=1/3$), this lower bound
follows from the known asymptotics of Airy function maxima
(Abramowitz--Stegun~\S10.4); the conjecture is therefore true
in that case. For general $V\in\mathfrak{T}_\gamma$, it is open.
\end{conjecture}

\begin{remark}[Why O1 is structurally different from O3a]
\label{rem:O1_vs_O3a}
O3a asks: can a method \emph{outside} $\mathcal{M}(\mathrm{S1,S2,S3})$
achieve a better rate?
O1 asks: is H1 \emph{necessary} within $\mathcal{M}(\mathrm{S1,S2,S3})$
for the current rate, or could it be weakened while keeping the same rate?
These are logically independent.
O3a is a method-class question; O1 is an axiom-minimality question.
Neither follows from the other.
\end{remark}

\section{Programme Conclusion}
\label{sec:conclusion}
%% ============================================================

\textbf{XVIII--XIX--XX.}
Universal rate $O(c^{-\beta}(\log c)^{1+\gamma})$ for fixed $\gamma>0$,
structural obstruction (Theorem~\ref{thm:rate_obstruction}),
and H1 exponent rigidity (Lemma~\ref{lem:H1_rigidity}):
no improvement is possible within $\mathcal{M}(\mathrm{S1,S2,S3})$,
and the exponent pair $(\gamma,\beta)$ is uniquely identified by the rate.
Closure inventory:
\begin{itemize}[nosep,leftmargin=1.5em]
  \item $c$-admissible chain, four sources;
  \item EM structural property: independent instances;
  \item $\BK$, $C_2(\alpha)$: $c$-free;
  \item domain vs.\ operator $c$-dependence: disjoint;
  \item anti-circularity: $F'\to$window$\to$convexity;
  \item Substitution Stability with monotonicity window control;
  \item $\gamma$-uniformity on compact sets; $(1+\gamma)2^\gamma\to\infty$ explicit;
  \item log-log stability: Lipschitz implicit function;
  \item $\prec$ transitive via lexicographic order;
  \item Decomposition stability: no cancellation between $\Phi_1,\Phi_3$;
  \item Balance fixed-point: strict separation outside window;
    $\asymp$-unavoidability inside;
  \item \textbf{H1 exponent rigidity:}
    distinct $(\gamma,\beta)$ yield distinct rates (injective map);
    logarithmic weakening inflates rate strictly (Lemma~\ref{lem:H1_rigidity});
  \item Non-tautology: $\Phi_1,\Phi_3$ have disjoint analytic origins;
  \item Constant uniformity: all $\asymp$-constants uniform in $c$;
  \item Method vs.\ operator scope: Tauberian structure made precise;
  \item Structural obstruction: rate rigid within $\mathcal{M}(\mathrm{S1,S2,S3})$;
  \item \textbf{O1 formalized:} H1 envelope conjecture
    (Conjecture~\ref{conj:H1_envelope}); Airy case verified;
    general case open;
  \item DAG: obstruction as terminal node, H1 rigidity as sibling input;
  \item Layer-Survival; invariants $1+\gamma$, $1+2\gamma$.
\end{itemize}

The leading logarithmic exponent is not merely achieved ---

The exponent pair $(\gamma,\beta)$ is uniquely identified by the rate;

No method in $\mathcal{M}(\mathrm{S1,S2,S3})$ can escape this balance;
