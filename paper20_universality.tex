\title[Paper XX: Universality of Turning-Point Truncation Geometry]{%
  Paper~XX: Universality of Turning-Point Truncation Geometry\\[0.3em]
  \large Spectral-Geometric Invariants and Two-Scale Logarithmic Losses}

\author{\textit{prolate-gram-coercivity programme}}
\date{May 2026}

\begin{abstract}
For operators $\cAg\in\mathfrak{T}_\gamma$
(spectrally regularized WKB class, Definition~\ref{def:opclass}),
the Langer term $\AK = \AKlin + \AKquad$ satisfies,
under the pointwise Langer scaling law
$\|e_j\|_{C^0}\leq C_0 j^\gamma c^{-\beta}$
(Hypothesis~\ref{hyp:langer_pointwise}),
$\Clin = O(K^{1+\gamma})$ and $\Cquad = O(K^{1+2\gamma})$
via summation $\sum_{j=1}^K j^\gamma\sim K^{1+\gamma}/(1+\gamma)$.

The optimal truncation $\Kopt$ is characterized as the
\emph{exact minimizer} of the upper-bound functional
$F(K) := L(c)\,K^{1+\gamma}c^{-\beta} + T(c)\,e^{-\alpha K/2}$
(Lemma~\ref{lem:kopt_variational}):
\[
  \Kopt = \operatorname{argmin}_{K\geq 1} F(K),
  \qquad
  \Kopt \sim \frac{2\beta}{\alpha(1+\gamma)}\log c
  \quad (c\to\infty).
\]
The Layer-Survival Selection Principle
(Theorem~\ref{thm:layer_survival}) shows that
for every interaction order $n\geq 2$,
layer $n$ is $\prec$ layer~$1$ in the total
asymptotic scale $\mathcal{S}$ (Definition~\ref{def:scale}).
Hence:
\[
  \normh{\cK-\cKA}{}\big|_{K=\Kopt}
  = O\!\left(c^{-\beta}(\log c)^{1+\gamma}\right).
\]
All results are theorems conditional on
Hypotheses~\ref{hyp:langer_pointwise}--\ref{hyp:Cinfty},
which are verified for the Airy case and for the
power family $V=C_V\xi^p$ (proof sketch).
\end{abstract}

\section{Asymptotic Scale and Order}
\label{sec:asymp}
%% ============================================================

\begin{definition}[Total asymptotic scale $\mathcal{S}$]
\label{def:scale}
Let $\mathcal{S}$ be the set of positive functions of the form
$c^{-s}(\log c)^t$ ($s>0$, $t\in\R$), totally ordered by:
\[
  c^{-s_1}(\log c)^{t_1} \succ c^{-s_2}(\log c)^{t_2}
  \;\Longleftrightarrow\;
  s_1 < s_2,\text{ or }(s_1=s_2\text{ and }t_1 > t_2),
\]
as $c\to\infty$.
All $o$- and $O$-comparisons in this paper use this order.
\end{definition}

\begin{proposition}[Layer dominance]
\label{prop:asymp_order}
For $\gamma,\beta>0$ and $K\sim\kappa\log c$ ($\kappa>0$):
\[
  c^{-n\beta}(\log c)^{1+n\gamma}
  \prec c^{-\beta}(\log c)^{1+\gamma}
  \quad \text{for all } n\geq 2,
\]
because $n\beta > \beta$; the comparison holds for all
$n\geq 2$, uniformly in $\gamma, \kappa$.
\end{proposition}

\begin{proof}
$c^{-n\beta}(\log c)^{1+n\gamma} / c^{-\beta}(\log c)^{1+\gamma}
= c^{-(n-1)\beta}(\log c)^{(n-1)\gamma}\to 0$
since $c^{-\delta}\cdot(\log c)^m\to 0$ for all $\delta>0$, $m>0$.
\end{proof}

\begin{remark}[Explicit uses of $c^{-\beta}\succ (\log c)^m$]
\label{rem:scales}
The comparison $c^{-\beta}\succ(\log c)^{-m}$ (for every fixed $m$)
is used in exactly two places:
\begin{enumerate}[label=(\roman*),nosep]
  \item Proposition~\ref{prop:asymp_order}:
    layer $n\geq 2$ dominated by layer $n=1$.
  \item Corollary~\ref{cor:universal_lograte}:
    spectral tail $e^{-\alpha\Kopt/2}\sim c^{-\beta/(1+\gamma)}
    \prec c^{-\beta}(\log c)^{1+\gamma}$.
\end{enumerate}
No other comparison between polynomial $c$-factors and
logarithms is needed.
\end{remark}

\section{Framework: Spectrally Regularized WKB Class}
\label{sec:framework}
%% ============================================================

\begin{definition}[Spectrally regularized WKB class $\mathfrak{T}_\gamma$]
\label{def:opclass}
For $\gamma>0$, $\beta>0$: $\mathfrak{T}_\gamma$ is the class
of self-adjoint operators $\cAg=-\partial_\xi^2+V(\xi)$
on $L^2(\R_+)$, $V\geq 0$, $V\to+\infty$,
discrete spectrum $0<a_1<a_2<\cdots$, satisfying:
\begin{enumerate}[label=(\roman*),nosep]
  \item $a_j\sim C_\gamma j^\gamma$ (spectral growth);
  \item $V$ is $C^2$, simple turning points, Langer--Olver applicable;
  \item Hypothesis~\ref{hyp:langer_pointwise} (pointwise Langer);
  \item Hypothesis~\ref{hyp:Cinfty} ($C_\infty<\infty$).
\end{enumerate}
\end{definition}

\begin{hypothesis}[Pointwise Langer scaling law]
\label{hyp:langer_pointwise}
$\exists\, C_0=C_0(M,V)<\infty$, independent of $j$, $K$, $c$,
such that for $c\geq c_0(j)$:
\begin{equation}\label{eq:langer_pointwise}
  \|e_j\|_{C^0([0,M])} \leq C_0\,j^\gamma\,c^{-\beta}.
\end{equation}
\emph{Status.}
Airy ($p=1$): verified by XVIII \texttt{rem:langerkdep}.
Power family $V=C_V\xi^p$:
\texttt{olver\_growth\_lemma.tex} gives a proof sketch
(Remark~\ref{rem:h1_mechanism} explains the mechanism).
General $V$: open; requires a regular variation condition.
\end{hypothesis}

\begin{remark}[Mechanism behind H1 for the power family]
\label{rem:h1_mechanism}
For $V=C_V\xi^p$, the Olver remainder bound gives
$\|e_j\|\leq 2AQ_j c^{-\beta}$ with
$Q_j = \int_0^M|\Psi(\zeta_j(t))|\,dt$
(\texttt{olver\_growth\_lemma.tex}, eq.~(2.3)).
The turning point $\xi_j\sim j^{\gamma/p}$ and the
Olver error potential $\Psi$ involves $V'(\xi_j)$
and $V''(\xi_j)$, giving
$Q_j\sim D_p j^{\gamma/p}$.
Since $\gamma/p\leq\gamma$ for $p\geq 1$
(with equality at $p=1$, Airy is the worst case):
$Q_j\leq D_1 j^\gamma$ with $D_1$ independent of $j$.
This is the constructive bridge from O4 to H1
in the DAG of \texttt{trilogy\_dag.tex}:
the mechanism is Olver remainder $+$
turning-point WKB scaling.
\end{remark}

\begin{hypothesis}[$C_\infty$ boundedness]
\label{hyp:Cinfty}
$C_\infty:=\sup_{j\geq 1}\|u_j\|_{C^0([0,M])}<\infty$.

\emph{Status.}
Airy: standard $L^\infty$ Airy bound.
Power family: follows from WKB $L^\infty$ normalization.
General: open.
\end{hypothesis}

\section{Truncation Geometry via Summation}
\label{sec:trunc}
%% ============================================================

\begin{lemma}[Universal truncation geometry]
\label{lem:trunc_universal}
Under Hypotheses~\ref{hyp:langer_pointwise} and~\ref{hyp:Cinfty},
for $K\geq 1$ and $c\geq c_0$:
\begin{align}
  \normh{\AKlin}{}
    &\leq 2MC_\infty C_0\,c^{-\beta}\sum_{j=1}^K j^\gamma
    \leq \underbrace{\frac{2MC_\infty C_0}{1+\gamma}K^{1+\gamma}
      +O(K^\gamma)}_{\Clin}\cdot c^{-\beta},
    \label{eq:AKlin}\\
  \normh{\AKquad}{}
    &\leq MC_0^2\,c^{-2\beta}\sum_{j=1}^K j^{2\gamma}
    \leq \underbrace{\frac{MC_0^2}{1+2\gamma}K^{1+2\gamma}
      +O(K^{2\gamma})}_{\Cquad}\cdot c^{-2\beta}.
    \label{eq:AKquad}
\end{align}
Hence $\Clin=O(K^{1+\gamma})$, $\Cquad=O(K^{1+2\gamma})$
with constants depending only on $M$, $C_0$, $C_\infty$, $\gamma$.
\end{lemma}

\begin{proof}
For each $j\leq K$:
$\|u_j\otimes e_j+e_j\otimes u_j\|\leq 2MC_\infty C_0 j^\gamma c^{-\beta}$;
$\|e_j\otimes e_j\|\leq MC_0^2 j^{2\gamma}c^{-2\beta}$
($c^{-2\beta}$ is not absorbed).
Sum over $j=1,\ldots,K$ using Euler--Maclaurin:
$\sum_{j=1}^K j^\sigma = K^{1+\sigma}/(1+\sigma)+O(K^\sigma)$.
\end{proof}

\section{Variational Optimal Truncation}
\label{sec:kopt}
%% ============================================================

\begin{lemma}[Variational characterization of $\Kopt$]
\label{lem:kopt_variational}
Define the upper-bound functional (for $K\geq 1$, $c\geq 2$):
\begin{equation}\label{eq:F}
  F(K) := L\,K^{1+\gamma}\,c^{-\beta} + T\,e^{-\alpha K/2},
\end{equation}
where $L = 2MC_\infty C_0/(1+\gamma)$ and $T=\Ctwo^{1/2}$
are positive constants independent of $K$ and $c$.

\begin{enumerate}[label=(\roman*)]
  \item \textbf{(Existence of minimizer).}
  $F(K)$ is convex in the real variable $K>0$
  (sum of a strictly convex increasing and a strictly convex
  decreasing function), so it has a unique minimizer
  $K^*(c)\in(0,\infty)$ satisfying the first-order condition
  \begin{equation}\label{eq:FOC}
    L(1+\gamma)\,{K^*}^{\gamma}\,c^{-\beta}
    = \tfrac{\alpha}{2}\,T\,e^{-\alpha K^*/2}.
  \end{equation}

  \item \textbf{(Asymptotics of $K^*$).}
  Taking logarithms of~\eqref{eq:FOC}:
  \[
    \gamma\log K^* + \log(L(1+\gamma)c^{-\beta})
    = \log(\tfrac{\alpha T}{2}) - \tfrac{\alpha}{2}K^*.
  \]
  Since $\log K^* = o(K^*)$, the dominant balance as $c\to\infty$ is
  $-\beta\log c \sim -\frac{\alpha}{2}K^*$, giving
  \begin{equation}\label{eq:kopt_leading}
    K^*(c) \sim \frac{2\beta}{\alpha}\log c
    \quad (\text{leading order}).
  \end{equation}
  Substituting back: $\gamma\log K^*\sim\gamma\log\log c = o(\log c)$,
  so the next-order correction gives
  \begin{equation}\label{eq:kopt_refined}
    K^*(c) \sim \frac{2\beta}{\alpha(1+\gamma)}\log c
    \qquad (c\to\infty).
  \end{equation}
  Both~\eqref{eq:kopt_leading} and~\eqref{eq:kopt_refined}
  are asymptotic; the rate $O(c^{-\beta}(\log c)^{1+\gamma})$
  holds for any $\Kopt = \Theta(\log c)$.

  \item \textbf{(Value at $K^*$).}
  At $K=K^*(c)\sim\kappa\log c$
  ($\kappa=2\beta/(\alpha(1+\gamma))$):
  \[
    F(K^*) = O\!\left(c^{-\beta}(\log c)^{1+\gamma}\right).
  \]
\end{enumerate}
\end{lemma}

\begin{proof}
(i)\; $F$ is the sum of $K\mapsto LK^{1+\gamma}c^{-\beta}$
(strictly increasing for $K>0$, $L>0$) and
$K\mapsto Te^{-\alpha K/2}$ (strictly decreasing);
both are convex, hence $F$ is strictly convex with a unique
minimizer. Setting $F'(K^*)=0$ gives~\eqref{eq:FOC}.

(ii)\; Take logarithms of~\eqref{eq:FOC} and note
$\log K^* = o(K^*)$ as $K^*\to\infty$.

(iii)\; $LK^{1+\gamma}c^{-\beta}|_{K=\kappa\log c}
= L\kappa^{1+\gamma}(\log c)^{1+\gamma}c^{-\beta}$.
\end{proof}

\begin{remark}[Variational vs.\ heuristic saddle balance]
\label{rem:variational_vs_heuristic}
Earlier versions (XIX, first XX draft) obtained $\Kopt$ by
a heuristic order-of-magnitude balance
$K^{1+\gamma}c^{-\beta}\asymp e^{-\alpha K/2}$.
Lemma~\ref{lem:kopt_variational} replaces this with an exact
variational statement: $K^*$ is the unique minimizer of
a strictly convex function $F(K)$, and the asymptotics
follow from the first-order condition~\eqref{eq:FOC}.
The two approaches give the same leading-order formula~\eqref{eq:kopt_leading};
the variational formulation provides the
missing ``minimizing principle'' that makes $\Kopt$
canonical rather than heuristic.
\end{remark}

\section{Layer-Survival Selection Principle}
\label{sec:layersurvival}
%% ============================================================

\begin{theorem}[Layer-Survival Selection Principle]
\label{thm:layer_survival}
For $n\geq 1$, the $n$-th interaction layer satisfies,
under Hypothesis~\ref{hyp:langer_pointwise},
\[
  \normh{A_K^{(n)}}{}
  \leq \frac{MC_\infty^{(2-n)_+}C_0^n}{1+n\gamma}
    K^{1+n\gamma}\,c^{-n\beta}
  + O(K^{n\gamma}c^{-n\beta}),
\]
where $(2-n)_+ = \max(2-n,0)$.
At $K=K^*(c)\sim\kappa\log c$:
\[
  \normh{A_K^{(n)}}{} = O\!\left(c^{-n\beta}(\log c)^{1+n\gamma}\right)
  \prec c^{-\beta}(\log c)^{1+\gamma} \quad \forall\, n\geq 2,
\]
by Proposition~\ref{prop:asymp_order}.
Consequently, the rate is governed by the $n=1$ layer alone.

\medskip
\textbf{Structural interpretation.}
For each additional Langer error ($n\to n+1$):
the $K$-exponent increases by $\gamma$ but the
$c$-exponent increases by $\beta$.
Since $c^{-\beta}\succ(\log c)^m$ for all $m$
(Definition~\ref{def:scale}),
every additional $c^{-\beta}$ factor dominates any
logarithmic gain. The selection is determined entirely
by the total asymptotic order $\mathcal{S}$:
the layer with fewest Langer errors (smallest $c$-exponent)
governed by layer $n=1$.
\end{theorem}

\section{Universal Rate Theorem}
\label{sec:main}
%% ============================================================

\begin{theorem}[Universal two-scale rate]
\label{thm:universal_rate}
Let $\cAg\in\mathfrak{T}_\gamma$, $K\geq K_0$, $c\geq c_0$:
\begin{equation}\label{eq:twoscale_universal}
  \normh{\cK-\cKA}{}
  \leq \Clin\cdot c^{-\beta}
  + \Cquad\cdot c^{-2\beta}
  + \Ctwo^{1/2}\cdot e^{-\alpha K/2}.
\end{equation}
\end{theorem}

\begin{corollary}[Universal logarithmic rate]
\label{cor:universal_lograte}
With $\Kopt = K^*(c)$ from Lemma~\ref{lem:kopt_variational}:
\begin{equation}\label{eq:lograte_universal}
  \normh{\cK-\cKA}{}\big|_{K=\Kopt}
  = O\!\left(c^{-\beta}(\log c)^{1+\gamma}\right).
\end{equation}
All three contributions are explicit:
\begin{enumerate}[label=(\roman*),nosep]
  \item Linear: $O(c^{-\beta}(\log c)^{1+\gamma})$ — leading.
  \item Quadratic: $O(c^{-2\beta}(\log c)^{1+2\gamma})
    \prec$ linear (Prop.~\ref{prop:asymp_order}).
  \item Tail: $O(c^{-\beta/(1+\gamma)})
    \prec c^{-\beta}(\log c)^{1+\gamma}$
    (Remark~\ref{rem:scales}(ii)).
\end{enumerate}
\end{corollary}

\begin{remark}[The two spectral-geometric invariants]
\label{rem:invariants}
$1+\gamma$: exponent of the linear layer;
governs the rate (survives $c$-hierarchy);
arises from $\sum_{j=1}^K j^\gamma\sim K^{1+\gamma}/(1+\gamma)$.

$1+2\gamma$: exponent of the quadratic layer;
structurally present; arises from $\sum_{j=1}^K j^{2\gamma}$;
subleading by one factor $c^{-\beta}$ in $\mathcal{S}$.

The gap $\gamma=(1+2\gamma)-(1+\gamma)$ measures the
additional crowding cost of one extra Langer interaction;
both invariants are determined by the turning-point geometry.
\end{remark}
