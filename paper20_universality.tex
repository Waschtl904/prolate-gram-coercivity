\begin{abstract}
For operators $\cAg\in\mathfrak{T}_\gamma$
(spectrally regularized WKB class, Definition~\ref{def:opclass}),
the Langer term $\AK = \AKlin + \AKquad$ satisfies,
under the pointwise Langer scaling law
$\|e_j\|_{C^0}\leq C_0 j^\gamma c^{-\beta}$
(Hypothesis~\ref{hyp:langer_pointwise}),
$\Clin = O(K^{1+\gamma})$ and $\Cquad = O(K^{1+2\gamma})$
via summation $\sum_{j=1}^K j^\gamma\sim K^{1+\gamma}/(1+\gamma)$.

The optimal truncation $\Kopt$ is the unique minimizer of
$F(K;c) := L\,K^{1+\gamma}c^{-\beta} + T\,e^{-\alpha K/2}$,
where $F$ is a \emph{fixed-$c$ variational functional}
(Lemma~\ref{lem:kopt_variational}).
The minimizer satisfies
\[
  K^*(c) = \frac{2\beta}{\alpha}\log c
          + \frac{2\gamma}{\alpha}\log\log c + O(1).
\]
All operator bounds are derived for fixed $K$ and $c$,
then evaluated along the coupled diagonal $K = K^*(c)$,
$c\to\infty$ (Remark~\ref{rem:diagonal_regime}).
The logical dependency structure is displayed in
Section~\ref{sec:dag_logical}.
The scale $\mathcal{S}$ is used as a representative
system of asymptotic equivalence classes under $\sim$
(Proposition~\ref{prop:scale_algebra}).
All results are conditional on H1--H2.
\end{abstract}

\begin{remark}[Diagonal asymptotic regime and quantifier discipline]
\label{rem:diagonal_regime}
Two distinct regimes appear in this paper and are kept strictly separate:
\begin{itemize}[nosep]
  \item \textbf{Fixed-parameter regime.}
    All operator inequalities
    (Lemmas~\ref{lem:gamma_agnostic}--\ref{lem:trunc_universal},
    Theorem~\ref{thm:universal_rate})
    hold for all $(K,c)$ in a stated parameter range,
    with no coupling between $K$ and $c$ assumed.
    Quantifier structure: $\forall K\geq K_0,\; \forall c\geq c_0$.
  \item \textbf{Diagonal evaluation.}
    The variational problem $\min_K F(K;c)$
    is solved \emph{for each fixed $c$} independently
    (Section~\ref{sec:kopt}).
    The solution $K^*(c)$ is a function of $c$ alone;
    its asymptotics $K^*(c)\sim\frac{2\beta}{\alpha}\log c$
    are derived as $c\to\infty$ with $K$ determined by $c$.
    Quantifier structure:
    $\forall c\geq c_0,\;\exists! K^*(c)\in(0,\infty)$ solving $F'(K;c)=0$.
\end{itemize}
The final rate result
(Corollary~\ref{cor:universal_lograte})
is obtained by substituting the diagonal $K = K^*(c)$
into the fixed-parameter bounds.
No two-parameter joint limit $(K,c)\to(\infty,\infty)$
is taken; the coupling is one-directional: $c\mapsto K^*(c)$.
\end{remark}

\section{Asymptotic Scale and Order}
\label{sec:asymp}
%% ============================================================

\begin{definition}[Asymptotic scale representatives $\mathcal{S}$]
\label{def:scale}
Let $\mathcal{S} = \{c^{-s}(\log c)^t : s>0, t\in\R\}$.
$\mathcal{S}$ is used as a \emph{representative system}
of asymptotic equivalence classes under $\sim$
(i.e.\ $f\sim g$ iff $f/g\to 1$ as $c\to\infty$).
The induced total order $\succ$ on $\mathcal{S}/\!\sim$:
$[c^{-s_1}(\log c)^{t_1}] \succ [c^{-s_2}(\log c)^{t_2}]$
iff $s_1<s_2$, or ($s_1=s_2$ and $t_1>t_2$).

All $O(\cdot)$, $o(\cdot)$, $\prec$, $\sim$ notation in this paper
refers to the equivalence classes in $\mathcal{S}/\!\sim$;
statements like ``$f\in O(g)$'' are statements about
representatives, valid up to asymptotic equivalence.
\end{definition}

\begin{proposition}[Algebraic stability of $\mathcal{S}/\!\sim$]
\label{prop:scale_algebra}
The ordered set $(\mathcal{S}/\!\sim,\succ)$ is stable under:
\begin{itemize}[nosep]
\item[\textup{(i)}] \textbf{Multiplication} (exact on representatives):
  $c^{-s_1}(\log c)^{t_1}\cdot c^{-s_2}(\log c)^{t_2}
  = c^{-(s_1+s_2)}(\log c)^{t_1+t_2}\in\mathcal{S}$.
\item[\textup{(ii)}] \textbf{Asymptotic addition
  (dominant-term rule on equivalence classes):}
  if $[f]\succ[g]$ then $[f+g]=[f]$ in $\mathcal{S}/\!\sim$,
  since $g/f\to 0$ implies $(f+g)/f\to 1$.
\end{itemize}
Note: $\mathcal{S}$ is \emph{not} literally closed under addition as a set
(e.g.\ $c^{-1}+c^{-2}\notin\mathcal{S}$),
but its equivalence classes are closed under addition
in the sense of~\textup{(ii)}.
Every finite sum and product of elements of $\mathcal{S}$
has a well-defined leading class in $\mathcal{S}/\!\sim$.
\end{proposition}

\begin{proof}
(i) Direct computation.
(ii) $[f]\succ[g]\Rightarrow g/f\to 0
\Rightarrow (f+g)/f=1+g/f\to 1
\Rightarrow [f+g]=[f]$.
\end{proof}

\begin{proposition}[Layer dominance]
\label{prop:asymp_order}
For $\gamma,\beta>0$, $K\sim\kappa\log c$, and all $n\geq 2$:
$[c^{-n\beta}(\log c)^{1+n\gamma}]
\prec [c^{-\beta}(\log c)^{1+\gamma}]$
in $\mathcal{S}/\!\sim$.
\end{proposition}

\begin{proof}
Ratio $= c^{-(n-1)\beta}(\log c)^{(n-1)\gamma}\to 0$.
\end{proof}

\begin{remark}[Uses of $c^{-\beta}\succ(\log c)^m$]
\label{rem:scales}
Used at: Proposition~\ref{prop:asymp_order};
Corollary~\ref{cor:universal_lograte}(iii).
Both are instances of Proposition~\ref{prop:scale_algebra}(i).
\end{remark}

\begin{remark}[DAG guarantees: no hidden coupling loops]
\label{rem:dag_nocycle}
The DAG is acyclic by construction:
\begin{itemize}[nosep]
  \item $F(K;c)$ is defined independently of $K^*(c)$
    (it is a function of two independent variables $K,c$).
  \item $K^*(c)$ is derived from $F$ via $F'(K;c)=0$
    for each fixed $c$; this is a one-directional dependency.
  \item The sandwich (Lemma~\ref{lem:loglog}) uses
    $K^*\to\infty$ (from $F$ structure)
    and $F'$ sign analysis at $c_1\log c$;
    no property of the asymptotic expansion~\eqref{eq:kopt_iterated}
    is used in its proof.
  \item Corollary~\ref{cor:universal_lograte} uses
    the fixed-parameter bounds of
    Theorem~\ref{thm:universal_rate} and the
    diagonal substitution $K = K^*(c)$; these are
    logically disjoint steps connected only by substitution.
\end{itemize}
In particular, there is no cycle of the form:
``bound on $K^*$ used to derive bound on $K^*$''.
\end{remark}

\begin{definition}[$\mathfrak{T}_\gamma$]
\label{def:opclass}
For $\gamma,\beta>0$: $\mathfrak{T}_\gamma$ is the class
of self-adjoint $\cAg = -\partial_\xi^2+V(\xi)$ on $L^2(\R_+)$,
$V\geq 0$, $V\to+\infty$, discrete spectrum, satisfying:
(i) $a_j\sim C_\gamma j^\gamma$;
(ii) $V\in C^2$, simple turning points;
(iii) H1; (iv) H2.
\end{definition}

\begin{hypothesis}[H1: Pointwise Langer scaling]
\label{hyp:langer_pointwise}
$\exists\, C_0<\infty$ (independent of $j,K,c$) such that
$\|e_j\|_{C^0([0,M])} \leq C_0\,j^\gamma\,c^{-\beta}$
for $c\geq c_0(j)$.

\emph{Status:} Airy: XVIII; power family: sketch
(\texttt{olver\_growth\_lemma.tex});
general: follows from H1-K
(\texttt{olver\_growth\_lemma.tex}, Corollary~\texttt{cor:h1\_derived}).
\end{hypothesis}

\begin{remark}[Mechanism for H1, power family]
\label{rem:h1_mechanism}
Olver: $\|e_j\|\leq 2AQ_jc^{-\beta}$;
$\xi_j\sim j^{\gamma/p}\Rightarrow Q_j\leq D_1j^\gamma$;
$C_0 = 2AD_1$ independent of $j$.
\end{remark}

\begin{hypothesis}[H2: $C_\infty$ boundedness]
\label{hyp:Cinfty}
$C_\infty = \sup_{j\geq 1}\|u_j\|_{C^0([0,M])}<\infty$.
\emph{Status:} Airy: standard; power family: WKB; general: open.
\end{hypothesis}

\begin{lemma}[Universal truncation geometry]
\label{lem:trunc_universal}
Under H1--H2, for all fixed $K\geq 1$, $c\geq c_0$
(fixed-parameter regime; Remark~\ref{rem:diagonal_regime}):
\begin{align}
  \normh{\AKlin}{}
    &\leq \frac{2MC_\infty C_0}{1+\gamma}K^{1+\gamma}c^{-\beta}
    + O(K^\gamma c^{-\beta}),\label{eq:AKlin}\\
  \normh{\AKquad}{}
    &\leq \frac{MC_0^2}{1+2\gamma}K^{1+2\gamma}c^{-2\beta}
    + O(K^{2\gamma}c^{-2\beta}).\label{eq:AKquad}
\end{align}
\end{lemma}

\begin{proof}
Euler--Maclaurin: $\sum_{j=1}^K j^\sigma
= K^{1+\sigma}/(1+\sigma)+O(K^\sigma)$.
\end{proof}

\section{Variational Optimal Truncation}
\label{sec:kopt}
%% ============================================================

\begin{lemma}[$K^*(c)\to\infty$]
\label{lem:kstar_to_infty}
Let $F(K;c) = LK^{1+\gamma}c^{-\beta} + Te^{-\alpha K/2}$
with $L,T,\alpha,\gamma,\beta>0$.
For each fixed $c$, $F(\,\cdot\,;c)$ is a fixed-$c$ variational
functional and has a unique minimizer $K^*(c)$.
As $c\to\infty$: $K^*(c)\to\infty$.
\end{lemma}

\begin{proof}
For any fixed $K_0>0$: $F'(K_0;c)
= L(1+\gamma)K_0^\gamma c^{-\beta} - \frac{\alpha}{2}Te^{-\alpha K_0/2}
< 0$ for all large $c$ (first term $\to 0$, second fixed).
Hence $K^*(c)>K_0$ for large $c$; $K_0$ arbitrary gives $K^*\to\infty$.
\end{proof}

\begin{lemma}[Sandwich on $K^*(c)$]
\label{lem:loglog}
With $c_1:=2\beta/\alpha$, $c_2:=2(\beta+1)/\alpha$,
for all $c$ sufficiently large:
\begin{equation}\label{eq:sandwich}
  c_1\log c \leq K^*(c) \leq c_2\log c,
  \qquad \log K^*(c) = \log\log c + O(1).
\end{equation}
\end{lemma}

\begin{proof}
\textit{Upper.}
From the logarithmic FOC (derived in Lemma~\ref{lem:kopt_variational}
using only strict convexity and existence, not this sandwich):
$\tfrac{\alpha}{2}K^* = \beta\log c + \gamma\log K^* + C_0$.
Set $g(c):=\gamma(\log K^*(c))/K^*(c)$.
Since $K^*\to\infty$ and $t\mapsto(\log t)/t\searrow 0$ for $t>e$,
$g(c)\to 0$ monotonically;
hence $\exists c_0$ (depending only on $\alpha,\beta,\gamma,L,T$)
with $g(c)\leq\alpha/4$ for $c\geq c_0$,
giving $K^*\leq c_2\log c$.
\textit{Lower.}
$F'(c_1\log c;c) = [L(1+\gamma)(2\beta/\alpha)^\gamma(\log c)^\gamma
- \frac{\alpha}{2}T]c^{-\beta} > 0$ for large $c$;
since $F'(0^+;c)<0$, the zero $K^*\geq c_1\log c$.
\textit{Log extraction} is immediate from~\eqref{eq:sandwich}.
\end{proof}

\begin{lemma}[Variational $K^*$]
\label{lem:kopt_variational}
Set $F(K;c) := LK^{1+\gamma}c^{-\beta} + Te^{-\alpha K/2}$,
$L=2MC_\infty C_0/(1+\gamma)$, $T=\Ctwo^{1/2}$.

\textup{(i)} \textbf{Fixed-$c$ existence and uniqueness.}
For each fixed $c$,
$F(\,\cdot\,;c)$ is strictly convex on $(0,\infty)$
($F''>0$), with $F(0^+)=T>0$, $F'(0^+)<0$, $F(\infty)=\infty$:
unique interior minimizer $K^*(c)$ satisfying
\begin{equation}\label{eq:FOC}
  L(1+\gamma){K^*}^{\gamma}c^{-\beta}
  = \tfrac{\alpha}{2}\,T\,e^{-\alpha K^*/2}.
\end{equation}

\textup{(ii)} \textbf{Iterated asymptotics.}
Logarithm of~\eqref{eq:FOC}:
\begin{equation}\label{eq:log_FOC}
  \tfrac{\alpha}{2}K^*
  = \beta\log c + \gamma\log K^* + C_1.
\end{equation}
Deriving~\eqref{eq:kopt_iterated} without circularity:
\begin{itemize}[nosep]
  \item \emph{Step 1.} $K^*\to\infty$
    (Lemma~\ref{lem:kstar_to_infty}, independent of any expansion)
    $\Rightarrow$ $\gamma\log K^*/K^*\to 0$
    $\Rightarrow$ zeroth balance $K_0 = \frac{2\beta}{\alpha}\log c$.
  \item \emph{Step 2.} Sandwich~\eqref{eq:sandwich}
    (proved without~\eqref{eq:kopt_iterated})
    $\Rightarrow$ $\log K^* = \log\log c + O(1)$.
  \item \emph{Step 3.} Substitute Step~2 into~\eqref{eq:log_FOC}:
\end{itemize}
\begin{equation}\label{eq:kopt_iterated}
  K^*(c) = \frac{2\beta}{\alpha}\log c
           + \frac{2\gamma}{\alpha}\log\log c + O(1).
\end{equation}

\textup{(ii')} \textbf{Metric reformulation of the fixed-point equation
(stability check).}
Equation~\eqref{eq:log_FOC} is equivalent to $K^*=\Phi(K^*)$ with
$\Phi(K):=\frac{2\beta}{\alpha}\log c+\frac{2\gamma}{\alpha}\log K+C_1$.
This reformulation does \emph{not} provide a logically independent
derivation of~\eqref{eq:kopt_iterated}
(all inputs---sandwich, $K^*\to\infty$---are shared with~\textup{(ii)});
its purpose is to:
\begin{itemize}[nosep]
  \item certify that the iterative scheme of~\textup{(ii)}
    operates within the correct scaling regime $K\asymp\log c$;
  \item provide a quantitative convergence rate
    $\lambda(c) = \frac{2\gamma}{\alpha c_1\log c}\to 0$;
  \item verify via Banach FPT (pointwise in $c$;
    not a uniform contraction over all $c$)
    that the unique fixed point coincides with $K^*(c)$
    from~\textup{(i)}.
\end{itemize}
See Remark~\ref{rem:dag_nocycle} for the DAG position of this arm.

\textup{(iii)} $F(K^*) = O(c^{-\beta}(\log c)^{1+\gamma})$.
\end{lemma}

\begin{proof}
(i)--(iii) As in previous version.
(ii') is a reformulation, not an independent derivation;
all cited lemmas are already in the acyclic DAG upstream.
\end{proof}

\begin{lemma}[Stability of iterated asymptotics]
\label{lem:stability_kopt}
An $O(1)$ perturbation of the right-hand side of~\eqref{eq:log_FOC}
shifts $K^*$ by $O(1)$.
\end{lemma}

\begin{proof}
Let $K_i$ solve
$\tfrac{\alpha}{2}K_i=\beta\log c+\gamma\log K_i+C_i$,
$|C_1-C_2|\leq D$.
Subtracting and applying MVT with $\min K_i\geq c_1\log c$:
$\tfrac{\alpha}{2}|K_1-K_2|
\leq \frac{\gamma}{c_1\log c}|K_1-K_2|+D
\leq \tfrac{\alpha}{4}|K_1-K_2|+D$
for large $c$, giving $|K_1-K_2|\leq 4D/\alpha$.
\end{proof}

\begin{remark}[Discrete projection]
\label{rem:kopt_discrete}
$K_{\rm int}=\lfloor K^*\rceil$:
$F(K_{\rm int})-F(K^*)\leq\tfrac{1}{8}F''(K^*)
=O(c^{-\beta}(\log c)^{\gamma-1})
\prec c^{-\beta}(\log c)^{1+\gamma}$
(Proposition~\ref{prop:scale_algebra}(ii)).
\end{remark}

\begin{theorem}[Layer-Survival]
\label{thm:layer_survival}
For $n\geq 1$, under H1, and for fixed $K,c$:
$\normh{A_K^{(n)}}{} = O(K^{1+n\gamma}c^{-n\beta})$.
Evaluated at $K=K^*(c)$:
$O(c^{-n\beta}(\log c)^{1+n\gamma})
\prec c^{-\beta}(\log c)^{1+\gamma}$
for $n\geq 2$ (Proposition~\ref{prop:asymp_order});
the rate is governed by $n=1$.
\end{theorem}

\section{Main Theorems}
\label{sec:main}
%% ============================================================

\begin{theorem}[Two-scale rate, fixed-parameter]
\label{thm:universal_rate}
For $\cAg\in\mathfrak{T}_\gamma$,
all fixed $K\geq K_0$, $c\geq c_0$:
\[
  \normh{\cK-\cKA}{}
  \leq \Clin c^{-\beta}
  + \Cquad c^{-2\beta}
  + \Ctwo^{1/2}e^{-\alpha K/2}.
\]
\end{theorem}

\begin{corollary}[Universal logarithmic rate]
\label{cor:universal_lograte}
Substituting the diagonal $K=K_{\rm int}=\lfloor K^*(c)\rceil$
(diagonal evaluation; Remarks~\ref{rem:diagonal_regime},
\ref{rem:kopt_discrete}):
$\normh{\cK-\cKA}{} = O(c^{-\beta}(\log c)^{1+\gamma})$.
Contributions:
\textup{(i)} linear: $O(c^{-\beta}(\log c)^{1+\gamma})$;
\textup{(ii)} quadratic $\prec$ linear
  (Proposition~\ref{prop:asymp_order});
\textup{(iii)} tail $O(c^{-\beta}(\log c)^{-\gamma})\prec$ linear
  (Proposition~\ref{prop:scale_algebra}(i)).
\end{corollary}

\begin{remark}[Spectral-geometric invariants]
\label{rem:invariants}
$1+\gamma$: leading exponent from $\sum j^\gamma$;
governed by turning-point crowding.
$1+2\gamma$: quadratic-layer exponent;
subleading by $c^{-\beta}$;
gap $\gamma$ = crowding cost per Langer interaction.
$\gamma$ is absent from the leading coefficient $2\beta/\alpha$
but governs the log-exponent and log-log correction.
\end{remark}

\begin{remark}[O3: sharpness gap and axiom status]
\label{rem:O3_sharpness}
The rate $O(c^{-\beta}(\log c)^{1+\gamma})$ is an upper bound.
Two distinct questions arise:
\begin{itemize}[nosep]
  \item \textbf{O3a (independent of H1--H2):}
    Does there exist $\cAg\in\mathfrak{T}_\gamma$
    with $\normh{\AKlin}{} = \Omega(K^{1+\gamma}c^{-\beta})$?
    This is a purely existential question about the class
    $\mathfrak{T}_\gamma$ and does not require strengthening
    H1 or H2.
  \item \textbf{O3b (potential hidden axiom):}
    A lower bound valid for \emph{all} $\cAg\in\mathfrak{T}_\gamma$
    would require a positive-mass condition on the spectral measure,
    i.e.\ that eigenfunction $L^2$-masses below the truncation
    threshold do not asymptotically cancel.
    This condition is not implied by H1 or H2,
    and would constitute a genuinely new axiom H3
    if imposed.
\end{itemize}
Currently: O3a is open (true mathematical unknown);
O3b is open and would require H3 (not imposed).
Neither blocks any upper-bound conclusion of this paper.
\end{remark}
