\begin{abstract}
For operators $\cAg\in\mathfrak{T}_\gamma$
(spectrally regularized WKB class, Definition~\ref{def:opclass}),
the Langer term $\AK = \AKlin + \AKquad$ satisfies,
under the pointwise Langer scaling law
$\|e_j\|_{C^0}\leq C_0 j^\gamma c^{-\beta}$
(Hypothesis~\ref{hyp:langer_pointwise}),
\[
  \|\AKlin\| \leq C_{\rm lin}\,K^{1+\gamma}c^{-\beta},
  \qquad
  \|\AKquad\| \leq C_{\rm quad}\,K^{1+2\gamma}c^{-2\beta}
\]
for all $K\geq K_0$, $c\geq c_0$, with $c$-admissible constants
(Lemma~\ref{lem:trunc_universal}; Remark~\ref{rem:EM_growth}).
The optimal truncation $\Kopt$ is the unique minimizer of
$F(K;c) := L\,K^{1+\gamma}c^{-\beta} + T\,e^{-\alpha K/2}$
over $K\in[c_1\log c,c_2\log c]$
(Lemma~\ref{lem:kopt_variational}), satisfying
$K^*(c) = \frac{2\beta}{\alpha}\log c + \frac{2\gamma}{\alpha}\log\log c + O(1)$.
The transition from fixed-parameter bounds to the diagonal regime
is licensed by Proposition~\ref{prop:substitution_stability},
the unique formal bridge between the two quantifier regimes.
All implied constants are $c$-admissible with no hidden dependence
(Remark~\ref{rem:cadmissible}, Lemma~\ref{lem:cadm_closure}).
All results are conditional on H1--H2.
\end{abstract}

\begin{remark}[$c$-admissible constants]
\label{rem:cadmissible}
A constant is \emph{$c$-admissible} if it is independent of
the approximation parameter $c$.
Every constant in this paper falls under exactly one of
four declared sources:
\begin{itemize}[nosep]
  \item \textbf{Axiomatic:} $C_0$, $C_\infty$ by H1--H2.
  \item \textbf{Analytic:} $C_\sigma = \frac{|B_2|}{12}\sigma(\sigma-1)$,
    function of exponent $\sigma$ only
    (eq.~\eqref{eq:EM_remainder}; see Remark~\ref{rem:EM_growth}).
  \item \textbf{Derived:} products and finite sums of $c$-admissible
    constants (Lemma~\ref{lem:cadm_closure}); covers
    $C_{\rm lin}$, $C_{\rm quad}$, $C_{\sigma,1}$, $C_{\sigma,2}$,
    and all $C_n$.
  \item \textbf{Operator-intrinsic:} $C_2(\alpha)$,
    $c$-admissible by definition
    (Remark~\ref{rem:BK_definition}).
\end{itemize}
No intermediate estimate introduces hidden $c$-dependence;
all constants are traceable to one of these four sources.
\end{remark}

\begin{lemma}[$c$-admissible closure]
\label{lem:cadm_closure}
The class of $c$-admissible constants is closed under
finite products and finite sums.
\end{lemma}

\begin{proof}
$A,B$ independent of $c$ $\Rightarrow$ $AB$ and $A+B$ independent of $c$.
\end{proof}

\begin{remark}[Two sources of $c$-dependence, strictly separated]
\label{rem:dag_csplit}
$c$ appears in two logically distinct, disjoint roles:
\begin{itemize}[nosep]
  \item \textbf{Operator estimates:} $c^{-\beta}$ as scaling factor;
    all multiplying constants $c$-admissible; $K,c$ independent.
  \item \textbf{Optimization domain:} $c$ in window
    $[c_1\log c, c_2\log c]$ and $K^*(c)$ via FOC.
    This is domain-dependence, not operator-geometry dependence.
\end{itemize}
The formal bridge is Proposition~\ref{prop:substitution_stability}.
\end{remark}

\begin{remark}[Two-regime quantifier structure]
\label{rem:diagonal_regime}
\begin{itemize}[nosep]
  \item \textbf{Fixed-parameter:}
    $\forall K\geq K_0,\, c\geq c_0$ independently;
    all constants $c$-admissible.
  \item \textbf{Diagonal:}
    $\forall c\geq c_0,\;\exists!\,K^*(c)\in[c_1\log c,c_2\log c]$;
    asymptotics as $c\to\infty$.
\end{itemize}
Transition licensed by Proposition~\ref{prop:substitution_stability}
(Remark~\ref{rem:dag_csplit}).
\end{remark}

\section{Substitution Stability Principle}
\label{sec:substprin}

\begin{proposition}[Substitution Stability Principle]
\label{prop:substitution_stability}
\emph{Intent.}
This proposition records an inference step, not a new analytical
estimate: it makes explicit that substituting a deterministic
map $K(c)$ into a pointwise-valid inequality is admissible as a
pointwise composition and requires no additional uniformity argument.

\emph{Setup.}
Let
\begin{equation}\label{eq:subst_input}
  |E(K,c)| \leq \sum_{n=1}^N C_n\,K^{a_n}c^{-b_n}
\end{equation}
hold for all $K\geq K_0$, $c\geq c_0$,
with $c$-admissible $C_n$ (Remark~\ref{rem:cadmissible}).
Let $K(c)$ satisfy
\begin{equation}\label{eq:Kc_window}
  K(c)\in[c_1\log c,\,c_2\log c]
  \quad\text{for all large }c.
\end{equation}

\emph{Conclusions.}
\begin{itemize}[nosep]
  \item[(i)] \textbf{Pointwise substitution.}
    $|E(K(c),c)|\leq\sum_n C_n(K(c))^{a_n}c^{-b_n}$,
    since~\eqref{eq:subst_input} is valid for each fixed $K$
    and evaluation along a deterministic map $K(c)$ is
    admissible as a pointwise composition;
    no uniform-in-$K$ structure is used.
  \item[(ii)] \textbf{Asymptotic order.}
    $(K(c))^{a_n}=O(\log^{a_n}c)$ from the upper bound
    in~\eqref{eq:Kc_window};
    for $K(c)\sim\tfrac{2\beta}{\alpha}\log c$ the precise
    correction is given by~\eqref{eq:Kstar_power}.
  \item[(iii)] \textbf{Dominance.}
    Asymptotic comparisons among terms follow from
    Propositions~\ref{prop:asymp_order} and~\ref{prop:scale_algebra},
    evaluated termwise.
\end{itemize}
Every diagonal evaluation in this paper is a reduction under
this principle; no separate uniformity argument is needed.
\end{proposition}

\begin{proof}
(i) pointwise substitution into a pointwise-valid inequality.
(ii) upper sandwich bound~\eqref{eq:Kc_window}.
(iii) Propositions~\ref{prop:asymp_order}--\ref{prop:scale_algebra}.
\end{proof}

\section{Asymptotic Scale and Regime-Dependent Order}
\label{sec:asymp}

\begin{definition}[Asymptotic scale]
\label{def:scale}
$\mathcal{S}=\{c^{-s}(\log c)^t : s>0,t\in\R\}$;
$f\sim g\Leftrightarrow f/g\to 1$;
$c^{-s_1}(\log c)^{t_1}\succ c^{-s_2}(\log c)^{t_2}$
iff $s_1<s_2$ or ($s_1=s_2$, $t_1>t_2$).
All comparisons evaluated in the diagonal regime
$K=K^*(c)\sim\tfrac{2\beta}{\alpha}\log c$;
verified per pair, not assumed globally.
\end{definition}

\begin{proposition}[Scale algebra]
\label{prop:scale_algebra}
$f\cdot g\in\mathcal{S}$ with exponents added.
If $f/g\to\infty$: $f+g=O(f)$ with constant $2$.
\end{proposition}

\begin{proposition}[Layer dominance]
\label{prop:asymp_order}
With $K=K^*(c)$, for $n\geq 2$:
$c^{-n\beta}(\log c)^{1+n\gamma}\prec c^{-\beta}(\log c)^{1+\gamma}$.
\end{proposition}

\begin{definition}[$\mathfrak{T}_\gamma$]
\label{def:opclass}
$\mathfrak{T}_\gamma$ ($\gamma,\beta>0$): self-adjoint
$\cAg=-\partial_\xi^2+V(\xi)$ on $L^2(\R_+)$,
$V\geq 0$, $V\to+\infty$, discrete spectrum;
(i) $a_j\sim C_\gamma j^\gamma$;
(ii) $V\in C^2$, simple turning points; (iii) H1; (iv) H2.
$c$ is an approximation scale parameter, not a spectral parameter;

\label{hyp:langer_pointwise}
$\exists\,C_0<\infty$ (independent of $j,K,c$):
$\|e_j\|_{C^0([0,M])}\leq C_0\,j^\gamma c^{-\beta}$.
\end{hypothesis}

\begin{hypothesis}[H2]
\label{hyp:Cinfty}
$C_\infty=\sup_{j}\|u_j\|_{C^0([0,M])}<\infty$
(operator-intrinsic, independent of $c$).
\end{hypothesis}

\begin{remark}[$\BK$: definition-level $c$-independence]
\label{rem:BK_definition}
$\BK$ is the spectral remainder of $\cAg$ after truncation
at index $K$: a functional of $\{P_j:j>K\}$ of $\cAg$ alone;
no $c$ enters its definition.
Consequently:
\begin{itemize}[nosep]
  \item $C_2(\alpha)=\sup_{K\geq 0}\|\BK\|^2 e^{\alpha K}$
    is a constant of $\cAg$ and $\alpha$;
    $c$-admissible by definition.
  \item The sup over all $K$ is compatible with evaluation
    at any $K(c)$: the bound $\|\BK\|\leq C_2^{1/2}e^{-\alpha K/2}$
    holds pointwise in $K$; $K^*(c)$ is one choice.
    No uniformity in $K$ is required
    (Proposition~\ref{prop:substitution_stability}(i)).
\end{itemize}
\end{remark}

\begin{lemma}[$\gamma$-Agnostic nodes]
\label{lem:gamma_agnostic}
Without H1--H2, for all $\cAg\in\mathfrak{T}_\gamma$:
(i) $\SK-\cKA=\AK-\BK$;
(ii) tensor norm;
(iii) Bessel tail;
(iv) Tonelli/DCT;
(v) $\|\BK\|\leq C_2(\alpha)^{1/2}e^{-\alpha K/2}$
  ($C_2(\alpha)$ $c$-admissible, Remark~\ref{rem:BK_definition});
(vi) Kato.
\end{lemma}

\section{Truncation Geometry}
\label{sec:trunc}

\begin{lemma}[Universal truncation geometry]
\label{lem:trunc_universal}
Under H1--H2, with
$C_{\rm lin}=\frac{2MC_\infty C_0}{1+\gamma}$,
$C_{\rm quad}=\frac{MC_0^2}{1+2\gamma}$
($c$-admissible, Lemma~\ref{lem:cadm_closure}),
for all $K\geq 1$, $c\geq c_0$:
\begin{align}
  \normh{\AKlin}{}
    &\leq C_{\rm lin}K^{1+\gamma}c^{-\beta}
    +C_{\sigma,1}K^\gamma c^{-\beta},\label{eq:AKlin}\\
  \normh{\AKquad}{}
    &\leq C_{\rm quad}K^{1+2\gamma}c^{-2\beta}
    +C_{\sigma,2}K^{2\gamma}c^{-2\beta},\label{eq:AKquad}
\end{align}
$C_{\sigma,1}=\tfrac{1}{2}C_0 C_\infty$,
$C_{\sigma,2}=\tfrac{1}{2}C_0^2$ ($c$-admissible).
\end{lemma}

\begin{proof}
Euler--Maclaurin applied to $\sum_{j=1}^K j^\sigma$:
\begin{equation}\label{eq:EM}
  \sum_{j=1}^K j^\sigma
  =\frac{K^{1+\sigma}}{1+\sigma}+\frac{1}{2}K^\sigma+R_\sigma(K),
\end{equation}
\begin{equation}\label{eq:EM_remainder}
  |R_\sigma(K)|\leq C_\sigma K^{\sigma-1},
  \quad C_\sigma=\frac{|B_2|}{12}\sigma(\sigma-1),\quad B_2=\tfrac{1}{6}.
\end{equation}
$C_\sigma$ depends only on $\sigma$; see Remark~\ref{rem:EM_growth}.
Multiply by $c$-admissible prefactors from H1--H2;
closure by Lemma~\ref{lem:cadm_closure}.
\end{proof}

\begin{remark}[EM remainder: global growth invariant]
\label{rem:EM_growth}
All Euler--Maclaurin remainders in this paper have the form
$R_\sigma(K)=O(K^{\sigma-1})$ with $c$-admissible implied constant
$C_\sigma=\frac{|B_2|}{12}\sigma(\sigma-1)$.
In particular:
\begin{itemize}[nosep]
  \item every EM remainder is polynomial in $K$
    with exponent strictly less than the leading term's exponent
    ($\sigma-1 < \sigma < 1+\sigma$);
  \item the implied constant is independent of $K$ and $c$
    (function of $\sigma$ only);
  \item no $K$-dependent growth amplification enters the
    constant hierarchy; the variable $K$ appears only as the
    explicit power in the bound, never inside any constant.
\end{itemize}
This is a global invariant of the EM expansion used throughout;
it prevents hidden $K$-dependence from leaking into
the $c$-admissible constant chain.
\end{remark}

\section{Variational Optimal Truncation}
\label{sec:kopt}
%% ============================================================

\begin{lemma}[$K^*(c)\to\infty$]
\label{lem:kstar_to_infty}
For each fixed $c$, $F(\cdot;c)$ has unique minimizer $K^*(c)$;
$K^*(c)\to\infty$ as $c\to\infty$.
\end{lemma}

\begin{lemma}[Sandwich on $K^*(c)$]
\label{lem:loglog}
With $c_1=2\beta/\alpha$, $c_2=2(\beta+1)/\alpha$:
\begin{equation}\label{eq:sandwich}
  c_1\log c\leq K^*(c)\leq c_2\log c,
  \qquad \log K^*(c)=\log\log c+O(1).
\end{equation}
\emph{Proof method:} sign analysis of $F'$ at endpoints $c_1\log c$
and the limit $F'(0^+)<0$;
no convexity argument is used here.
\end{lemma}

\begin{lemma}[Variational $K^*$]
\label{lem:kopt_variational}

\textup{(i)} \textbf{Fixed-$c$ uniqueness (convexity on window).}
By Lemma~\ref{lem:loglog} (sign analysis only, no circularity;
see Remark~\ref{rem:dag_nocycle}),
$K^*(c)\in[c_1\log c,c_2\log c]$.
On this window $F''(K;c)>0$ for each fixed $c$
(no uniform-in-$c$ claim); unique $K^*(c)$ with
\begin{equation}\label{eq:FOC}
  L(1+\gamma){K^*}^\gamma c^{-\beta}=\tfrac{\alpha}{2}Te^{-\alpha K^*/2}.
\end{equation}

\textup{(ii)} \textbf{Iterated asymptotics.}
\begin{equation}\label{eq:log_FOC}
  \tfrac{\alpha}{2}K^*=\beta\log c+\gamma\log K^*+C_1.
\end{equation}
Three steps (no circularity; Remark~\ref{rem:dag_nocycle}):
Step~1: $K^*\to\infty\Rightarrow$ leading term.
Step~2: sandwich $\Rightarrow$ $\log K^*=\log\log c+O(1)$.
Step~3:
\begin{equation}\label{eq:kopt_iterated}
  K^*(c)=\frac{2\beta}{\alpha}\log c+\frac{2\gamma}{\alpha}\log\log c+O(1).
\end{equation}

\textup{(ii')} \textbf{Role of $\lambda(c)$: self-consistency only.}
Lipschitz constant of iteration map:
$\lambda(c)=\frac{2\gamma}{\alpha K^*(c)}\to 0$,
hence $\lambda(c)<\tfrac{1}{2}$ for large $c$.
\emph{This certifies self-consistency of~\eqref{eq:kopt_iterated}
only.}
Existence and uniqueness of $K^*(c)$ comes from (i),
not from the Banach fixed-point theorem;
no conclusion depends on $\lambda(c)$ being small uniformly in $c$.

\textup{(iii)} $F(K^*;c)=O(c^{-\beta}(\log c)^{1+\gamma})$.
\end{lemma}

\begin{lemma}[Stability of asymptotics]
\label{lem:stability_kopt}
An $O(1)$ perturbation of~\eqref{eq:log_FOC}
shifts $K^*$ by $O(1)$.
\end{lemma}

\section{Diagonal Substitution}
\label{sec:diagonal}
%% ============================================================

\begin{lemma}[Diagonal substitution: explicit evaluation]
\label{lem:diagonal_sub}
This lemma applies Proposition~\ref{prop:substitution_stability}
to the specific choice $K(c)=K^*(c)$.

\emph{Input} (Theorem~\ref{thm:universal_rate},
all constants $c$-admissible):
\begin{equation}\label{eq:fixed_param_bound}
  \normh{\cK-\cKA}{}
  \leq C_{\rm lin}K^{1+\gamma}c^{-\beta}
  +C_{\rm quad}K^{1+2\gamma}c^{-2\beta}
  +C_{\rm tail}e^{-\alpha K/2}.
\end{equation}

\emph{Slowly-varying stability
($\gamma$-uniformity on compact sets).}
Let $h(c):=K^*(c)/(\tfrac{2\beta}{\alpha}\log c)\to 1$
with $|h(c)-1|=O(\log\log c/\log c)$.
By MVT, the $O$-constant in
\begin{equation}\label{eq:hgamma}
  h(c)^{1+\gamma}=1+O\!\left(\frac{\log\log c}{\log c}\right)
\end{equation}
is bounded by $(1+\gamma)2^\gamma$,
finite on every compact $[\gamma_0,\gamma_1]\subset(0,\infty)$.
(This is the authoritative $\gamma$-uniformity statement;
all references to compact-$\gamma$ uniformity in this paper
refer to this bound.)
No uniformity as $\gamma\to\infty$ is claimed anywhere.
Hence
\begin{equation}\label{eq:Kstar_power}
  (K^*(c))^{1+\gamma}
  =\Bigl(\tfrac{2\beta}{\alpha}\log c\Bigr)^{1+\gamma}
  \Bigl(1+O\Bigl(\tfrac{\log\log c}{\log c}\Bigr)\Bigr).
\end{equation}

\emph{Conclusion}
(Proposition~\ref{prop:substitution_stability}(i)--(iii)):
\begin{itemize}[nosep]
  \item[(i)] $K^*(c)^{1+\gamma}c^{-\beta}=O(c^{-\beta}(\log c)^{1+\gamma})$.
  \item[(ii)] $K^*(c)^{1+2\gamma}c^{-2\beta}\prec c^{-\beta}(\log c)^{1+\gamma}$.
  \item[(iii)] $C_{\rm tail}e^{-\alpha K^*(c)/2}\prec c^{-\beta}(\log c)^{1+\gamma}$.
  \item[(iv)] $\normh{\cK-\cKA}{}=O(c^{-\beta}(\log c)^{1+\gamma})$;
    linear term dominant.
\end{itemize}
\end{lemma}

\begin{remark}[Discrete projection]
\label{rem:kopt_discrete}
$K_{\rm int}=\lfloor K^*\rceil$:
$F(K_{\rm int})-F(K^*)\prec c^{-\beta}(\log c)^{1+\gamma}$.
\end{remark}

\begin{theorem}[Layer-Survival]
\label{thm:layer_survival}
For $n\geq 1$, H1, all $K\geq K_0$, $c\geq c_0$:
$\normh{A_K^{(n)}}{}\leq C_n K^{1+n\gamma}c^{-n\beta}$
($C_n$ $c$-admissible, Lemma~\ref{lem:cadm_closure}).
Diagonal by Proposition~\ref{prop:substitution_stability}:
\[
  \normh{A_K^{(n)}}{}_{K=K^*(c)}
  =O(c^{-n\beta}(\log c)^{1+n\gamma})
  \prec c^{-\beta}(\log c)^{1+\gamma}\quad(n\geq 2).
\]
\end{theorem}

\section{Main Theorems}
\label{sec:main}
%% ============================================================

\begin{theorem}[Two-scale rate, fixed-parameter]
\label{thm:universal_rate}
For $\cAg\in\mathfrak{T}_\gamma$, all $K\geq K_0$, $c\geq c_0$:
\[
  \normh{\cK-\cKA}{}
  \leq C_{\rm lin}K^{1+\gamma}c^{-\beta}
  +C_{\rm quad}K^{1+2\gamma}c^{-2\beta}
  +C_2(\alpha)^{1/2}e^{-\alpha K/2},
\]
all constants $c$-admissible
(Remark~\ref{rem:cadmissible}, Lemma~\ref{lem:cadm_closure}).
\end{theorem}

\begin{corollary}[Universal logarithmic rate]
\label{cor:universal_lograte}
By Proposition~\ref{prop:substitution_stability}
and Lemma~\ref{lem:diagonal_sub} at $K=\lfloor K^*(c)\rceil$:
\[
  \normh{\cK-\cKA}{}=O(c^{-\beta}(\log c)^{1+\gamma}).
\]
Leading constant $C_{\rm lin}(2\beta/\alpha)^{1+\gamma}$
($c$-admissible).
\end{corollary}

\begin{remark}[Spectral-geometric invariants]
\label{rem:invariants}
$1+\gamma$: leading; $1+2\gamma$: subleading.
$\gamma$ governs the log-exponent; absent from $2\beta/\alpha$.
\end{remark}
