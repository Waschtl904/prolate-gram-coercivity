\begin{abstract}
For operators $\cAg\in\mathfrak{T}_\gamma$
(spectrally regularized WKB class, Definition~\ref{def:opclass}),
the Langer term $\AK = \AKlin + \AKquad$ satisfies,
under the pointwise Langer scaling law
$\|e_j\|_{C^0}\leq C_0 j^\gamma c^{-\beta}$
(Hypothesis~\ref{hyp:langer_pointwise}),
$\Clin = O(K^{1+\gamma})$ and $\Cquad = O(K^{1+2\gamma})$
via summation $\sum_{j=1}^K j^\gamma\sim K^{1+\gamma}/(1+\gamma)$.

The optimal truncation $\Kopt$ is characterized as the
\emph{exact minimizer} of the upper-bound functional
$F(K) := L\,K^{1+\gamma}c^{-\beta} + T\,e^{-\alpha K/2}$
(Lemma~\ref{lem:kopt_variational}).
The unique minimizer satisfies the iterated asymptotics
\[
  K^*(c) = \frac{2\beta}{\alpha}\log c
          + \frac{2\gamma}{\alpha}\log\log c + O(1),
  \qquad\text{so in particular}\quad
  K^*(c)\sim\frac{2\beta}{\alpha}\log c.
\]
This expansion is derived as the unique fixed point of
$\Phi(K)=\frac{2\beta}{\alpha}\log c+\frac{2\gamma}{\alpha}\log K+O(1)$,
whose Lipschitz constant $\Phi'(K)=\frac{2\gamma}{\alpha K}$
decays to zero along the scaling regime $K\asymp\log c$
(parameter-dependent asymptotic contraction,
applied pointwise in $c$ for each fixed large $c$;
Lemma~\ref{lem:kopt_variational}(ii')),
and is stable under $O(1)$ perturbations of the logarithmic FOC
(Lemma~\ref{lem:stability_kopt}).
The integer projection $K_{\rm int}=\lfloor K^*\rceil$
satisfies $F(K_{\rm int}) = F(K^*)\cdot(1+o(1))$
(Remark~\ref{rem:kopt_discrete}).

The Layer-Survival Selection Principle
(Theorem~\ref{thm:layer_survival}) shows that
for every $n\geq 2$,
layer~$n$ is $\prec$ layer~$1$ in $\mathcal{S}$.
Hence: $\normh{\cK-\cKA}{}|_{K=\Kopt}
= O(c^{-\beta}(\log c)^{1+\gamma})$.

All results are theorems conditional on H1--H2;
the structural stability of the theory is certified
by \texttt{structural\_stability\_metatheorem.tex}.
\end{abstract}

\section{Asymptotic Scale and Order}
\label{sec:asymp}
%% ============================================================

\begin{definition}[Total asymptotic scale $\mathcal{S}$]
\label{def:scale}
Let $\mathcal{S} = \{c^{-s}(\log c)^t : s>0, t\in\R\}$,
totally ordered by:
$c^{-s_1}(\log c)^{t_1} \succ c^{-s_2}(\log c)^{t_2}$
iff $s_1<s_2$, or ($s_1=s_2$ and $t_1>t_2$), as $c\to\infty$.
\end{definition}

\begin{proposition}[Layer dominance]
\label{prop:asymp_order}
For $\gamma,\beta>0$, $K\sim\kappa\log c$, and all $n\geq 2$:
$c^{-n\beta}(\log c)^{1+n\gamma}
\prec c^{-\beta}(\log c)^{1+\gamma}$
in $\mathcal{S}$.
\end{proposition}

\begin{proof}
$c^{-n\beta}(\log c)^{1+n\gamma} / c^{-\beta}(\log c)^{1+\gamma}
= c^{-(n-1)\beta}(\log c)^{(n-1)\gamma}\to 0$.
\end{proof}

\begin{remark}[Uses of $c^{-\beta}\succ(\log c)^m$]
\label{rem:scales}
This comparison is used in exactly two places:
(i) Proposition~\ref{prop:asymp_order};
(ii) Corollary~\ref{cor:universal_lograte}(iii)
(tail $\prec$ linear).
\end{remark}

\begin{definition}[$\mathfrak{T}_\gamma$]
\label{def:opclass}
For $\gamma,\beta>0$: $\mathfrak{T}_\gamma$ is the class
of self-adjoint $\cAg = -\partial_\xi^2+V(\xi)$ on $L^2(\R_+)$,
$V\geq 0$, $V\to+\infty$, discrete spectrum, satisfying:
(i) $a_j\sim C_\gamma j^\gamma$;
(ii) $V\in C^2$, simple turning points;
(iii) H1; (iv) H2.
\end{definition}

\begin{hypothesis}[H1: Pointwise Langer scaling]
\label{hyp:langer_pointwise}
$\exists\, C_0<\infty$ (independent of $j,K,c$) such that
$\|e_j\|_{C^0([0,M])} \leq C_0\,j^\gamma\,c^{-\beta}$
for $c\geq c_0(j)$.

\emph{Status:} Airy: XVIII; power family: sketch
(\texttt{olver\_growth\_lemma.tex}, Remark~\ref{rem:h1_mechanism});
general: follows from H1-K
(\texttt{olver\_growth\_lemma.tex}, Corollary~\texttt{cor:h1\_derived}).
\end{hypothesis}

\begin{remark}[Mechanism for H1, power family]
\label{rem:h1_mechanism}
Olver gives $\|e_j\|\leq 2AQ_jc^{-\beta}$;
$\xi_j\sim j^{\gamma/p}$ implies $Q_j\sim D_pj^{\gamma/p}$;
$p\geq 1$ implies $Q_j\leq D_1j^\gamma$;
$C_0 = 2AD_1$ independent of $j$.
\end{remark}

\begin{hypothesis}[H2: $C_\infty$ boundedness]
\label{hyp:Cinfty}
$C_\infty = \sup_{j\geq 1}\|u_j\|_{C^0([0,M])}<\infty$.

\emph{Status:} Airy: standard;
power family: WKB $L^\infty$ normalization;
general: open.
\end{hypothesis}

\begin{lemma}[Universal truncation geometry]
\label{lem:trunc_universal}
Under H1--H2, for $K\geq 1$, $c\geq c_0$:
\begin{align}
  \normh{\AKlin}{}
    &\leq \frac{2MC_\infty C_0}{1+\gamma}K^{1+\gamma}c^{-\beta}
    + O(K^\gamma c^{-\beta}),\label{eq:AKlin}\\
  \normh{\AKquad}{}
    &\leq \frac{MC_0^2}{1+2\gamma}K^{1+2\gamma}c^{-2\beta}
    + O(K^{2\gamma}c^{-2\beta}).\label{eq:AKquad}
\end{align}
\end{lemma}

\begin{proof}
Sum over $j=1,\ldots,K$:
$\|u_j\otimes e_j+e_j\otimes u_j\|
\leq 2MC_\infty C_0 j^\gamma c^{-\beta}$;
$\|e_j\otimes e_j\|\leq MC_0^2 j^{2\gamma}c^{-2\beta}$.
Apply Euler--Maclaurin $\sum_{j=1}^K j^\sigma
= K^{1+\sigma}/(1+\sigma)+O(K^\sigma)$.
\end{proof}

%% Step 0: K* -> infinity (logically prior to all log-arguments)

\begin{lemma}[$K^*(c)\to\infty$]
\label{lem:kstar_to_infty}
Let $F(K) = LK^{1+\gamma}c^{-\beta} + Te^{-\alpha K/2}$
with $L,T,\alpha>0$, $\gamma>0$, $\beta>0$.
The unique minimizer $K^*(c)$ satisfies $K^*(c)\to\infty$
as $c\to\infty$.
\end{lemma}

\begin{proof}
$F'(K) = L(1+\gamma)K^\gamma c^{-\beta} - \frac{\alpha}{2}Te^{-\alpha K/2}$.
For any fixed $K_0 > 0$:
$F'(K_0) = L(1+\gamma)K_0^\gamma c^{-\beta} - \frac{\alpha}{2}Te^{-\alpha K_0/2}$.
As $c\to\infty$, the first term $\to 0$ while the second is fixed,
so $F'(K_0) < 0$ for all sufficiently large $c$.
Hence $K^*(c) > K_0$ for large $c$.
Since $K_0$ is arbitrary, $K^*(c)\to\infty$.
\end{proof}

%%------------------------------------------------------------
%% Auxiliary: log-of-log isolation (with sandwich)
%%------------------------------------------------------------
\begin{lemma}[Logarithm of the minimizer]
\label{lem:loglog}
Let $K^*(c)$ be the unique minimizer from
Lemma~\ref{lem:kopt_variational}.
Set
\[
  c_1 := \frac{2\beta}{\alpha},
  \qquad
  c_2 := \frac{2(\beta+1)}{\alpha},
\]
both depending only on $\alpha,\beta$.
For all $c$ sufficiently large:
\begin{equation}\label{eq:sandwich}
  c_1 \log c \leq K^*(c) \leq c_2 \log c.
\end{equation}
In particular $K^*(c)\to\infty$ (Lemma~\ref{lem:kstar_to_infty}),
and taking logarithms of~\eqref{eq:sandwich} yields
\begin{equation}\label{eq:loglog}
  \log K^*(c) = \log\log c + O(1).
\end{equation}
\end{lemma}

\begin{proof}
\textit{Upper bound.}
From the logarithmic FOC~\eqref{eq:log_FOC}
(proved in Lemma~\ref{lem:kopt_variational}
using only $F''>0$ and existence of $K^*$, not the sandwich):
\[
  \tfrac{\alpha}{2}K^*
  = \beta\log c + \gamma\log K^* + C_0,
\]
where $C_0 = \log(L(1+\gamma)/(\frac{\alpha}{2}T))$ is a fixed constant.
By Lemma~\ref{lem:kstar_to_infty}, $K^*\to\infty$.
Define $g(c) := \gamma\log K^*(c)/K^*(c)$.
Since $K^*\to\infty$ and $t\mapsto (\log t)/t$ is strictly
decreasing for $t > e$, $g(c)\to 0$ monotonically for
$K^*(c) > e$, which holds for all large $c$.
Hence there exists $c_0(\alpha,\beta,\gamma,L,T)$,
depending only on the parameters of $F$,
such that $g(c) \leq \alpha/4$ for all $c \geq c_0$.
(The existence of $c_0$ follows from the monotone decrease of $g$
toward $0$, not from any circular use of the bound~\eqref{eq:sandwich}.)
For $c \geq c_0$:
$\frac{\alpha}{4}K^* \leq \beta\log c + C_0$,
so $K^* \leq \frac{4(\beta\log c+C_0)}{\alpha} \leq c_2\log c$
for $c \geq c_0(\alpha,\beta,C_0)$.

\textit{Lower bound.}
$F'(K) = L(1+\gamma)K^\gamma c^{-\beta}
- \frac{\alpha}{2}Te^{-\alpha K/2}$.
Note $F'(0^+) = -\frac{\alpha}{2}T < 0$.
At $K = c_1\log c = \frac{2\beta}{\alpha}\log c$,
using $\frac{\alpha}{2}\cdot c_1 = \beta$ so
$e^{-\alpha c_1(\log c)/2} = c^{-\beta}$:
\[
  F'(c_1\log c)
  = \Bigl[L(1+\gamma)\Bigl(\tfrac{2\beta}{\alpha}\Bigr)^\gamma (\log c)^\gamma
    - \tfrac{\alpha}{2}T\Bigr]c^{-\beta}.
\]
The bracket is positive for all $c \geq c_{**}$,
where $c_{**}$ depends only on $L, T, \alpha, \beta, \gamma$.
Since $F'(0^+) < 0 < F'(c_1\log c)$ for $c\geq c_{**}$,
the unique zero $K^*$ satisfies $K^* \geq c_1\log c$.

\textit{Log extraction.}
From~\eqref{eq:sandwich}:
$\log c_1 + \log\log c
\leq \log K^*
\leq \log c_2 + \log\log c$,
giving~\eqref{eq:loglog} with $|O(1)| \leq \max(|\log c_1|,|\log c_2|)$.
\end{proof}

%% Main variational lemma

\begin{lemma}[Variational $K^*$]
\label{lem:kopt_variational}
Set $F(K) := L K^{1+\gamma}c^{-\beta} + Te^{-\alpha K/2}$
($L=2MC_\infty C_0/(1+\gamma)$, $T=\Ctwo^{1/2}$).

\textup{(i)} \textbf{Existence and uniqueness.}
$F$ is strictly convex on $(0,\infty)$ since
$F''(K) = L\gamma(1+\gamma)K^{\gamma-1}c^{-\beta}
+ \frac{\alpha^2}{4}Te^{-\alpha K/2} > 0$.
$F(0^+)=T>0$, $F'(0^+)=-\frac{\alpha}{2}T<0$,
$F(K)\to+\infty$ as $K\to+\infty$:
hence $F$ has a unique interior minimum $K^*(c)\in(0,\infty)$
satisfying the FOC
\begin{equation}\label{eq:FOC}
  L(1+\gamma){K^*}^{\gamma}c^{-\beta}
  = \tfrac{\alpha}{2}\,T\,e^{-\alpha K^*/2}.
\end{equation}

\textup{(ii)} \textbf{Iterated asymptotics via fixed-point iteration.}
Taking logarithms of~\eqref{eq:FOC}:
\begin{equation}\label{eq:log_FOC}
  \frac{\alpha}{2}K^*
  = \beta\log c + \gamma\log K^*
  + \log\!\Bigl(\tfrac{L(1+\gamma)}{(\alpha/2)T}\Bigr).
\end{equation}
This is equivalently the fixed-point equation $K^* = \Phi(K^*)$ with
\begin{equation}\label{eq:fixpoint}
  \Phi(K) := \frac{2\beta}{\alpha}\log c
  + \frac{2\gamma}{\alpha}\log K + C_1,
  \quad C_1 = \frac{2}{\alpha}\log\!\Bigl(\tfrac{L(1+\gamma)}{(\alpha/2)T}\Bigr).
\end{equation}
The expansion~\eqref{eq:kopt_iterated} is derived as follows,
without a priori assuming $K^*\asymp\log c$:
\begin{itemize}[nosep]
  \item \emph{Zeroth iterate.}
    By Lemma~\ref{lem:kstar_to_infty}, $K^*\to\infty$,
    so $\gamma\log K^*/K^*\to 0$;
    the dominant balance in~\eqref{eq:log_FOC} gives
    $K_0 := \frac{2\beta}{\alpha}\log c$.
  \item \emph{First iterate.}
    The sandwich~\eqref{eq:sandwich} gives
    $\log K^* = \log\log c + O(1)$ (Lemma~\ref{lem:loglog}).
    Substituting into~\eqref{eq:log_FOC}:
\end{itemize}
\begin{equation}\label{eq:kopt_iterated}
  K^*(c) = \frac{2\beta}{\alpha}\log c
           + \frac{2\gamma}{\alpha}\log\log c
           + O(1)
  \qquad(c\to\infty).
\end{equation}
In particular, $K^*(c)\sim\frac{2\beta}{\alpha}\log c$,
and $\gamma$ does not appear in the leading coefficient.

\textup{(ii')} \textbf{Parameter-dependent asymptotic contraction.}
On the invariant region $\mathcal{I}(c) := [c_1\log c, c_2\log c]$
(guaranteed non-empty for large $c$ by~\eqref{eq:sandwich}),
$\Phi$ maps $\mathcal{I}(c)$ into itself for large $c$, and
\begin{equation}\label{eq:contraction}
  \sup_{K\in\mathcal{I}(c)}|\Phi'(K)|
  = \frac{2\gamma}{\alpha c_1 \log c}
  \xrightarrow{c\to\infty} 0.
\end{equation}
For each fixed $c \geq c_0$, the Lipschitz constant
$\lambda(c) := \frac{2\gamma}{\alpha c_1\log c} \leq \frac{1}{2}$,
so $\Phi(\,\cdot\,;c)$ is a strict contraction on the
complete metric space $(\mathcal{I}(c), |\cdot|)$
(Banach fixed-point theorem applied pointwise in $c$;
the interval $\mathcal{I}(c)$ and Lipschitz constant $\lambda(c)$
both depend on $c$, so this is \emph{not} a uniform contraction
over all $c$ simultaneously).
The unique fixed point is $K^*(c)$,
and the iteration $K_{n+1}=\Phi(K_n)$, started from any
$K_0\in\mathcal{I}(c)$, converges at rate $\lambda(c)^n\to 0$.
This provides an independent derivation of~\eqref{eq:kopt_iterated}.

\textup{(iii)} \textbf{Value at $K^*$.}
At $K = K^*(c)$:
\[
  F(K^*) = O\!\left(c^{-\beta}(\log c)^{1+\gamma}\right).
\]
\end{lemma}

\begin{proof}
(i) $F'' > 0$ as computed;
$F(0^+)=T>0$, $F'(0^+)<0$, $F\to+\infty$:
unique interior minimum.
(ii) \eqref{eq:log_FOC} is exact.
The fixed-point structure~\eqref{eq:fixpoint} is algebraically equivalent.
Zeroth iterate: $K^*\to\infty$ (Lemma~\ref{lem:kstar_to_infty},
proved independently); $\gamma\log K^*/K^*\to 0$ follows.
First iterate: $\log K^*=\log\log c+O(1)$ (Lemma~\ref{lem:loglog}).
(ii') For each fixed large $c$:
$\Phi$ maps $\mathcal{I}(c)$ into itself since
$\Phi(c_i\log c)\in\mathcal{I}(c)$ for $i=1,2$ and large $c$.
$|\Phi'(K)|=\frac{2\gamma}{\alpha K}\leq\frac{2\gamma}{\alpha c_1\log c}
=\frac{\gamma}{\beta\log c}\leq\frac{1}{2}$ for $c\geq c_0$.
Banach fixed-point theorem on $(\mathcal{I}(c),|\cdot|)$
(a closed bounded interval, hence complete)
gives a unique fixed point $K^*(c)\in\mathcal{I}(c)$,
confirming~\eqref{eq:kopt_iterated} for each fixed large $c$.
(The application is pointwise in $c$; uniformity over all $c$
is not claimed or needed.)
(iii) $L(K^*)^{1+\gamma}c^{-\beta}
= L\bigl(\frac{2\beta}{\alpha}\log c\bigr)^{1+\gamma}
c^{-\beta}(1+o(1))
= O(c^{-\beta}(\log c)^{1+\gamma})$.
\end{proof}

\begin{lemma}[Stability of iterated asymptotics]
\label{lem:stability_kopt}
Let $K^*(c)$ satisfy~\eqref{eq:FOC},
and let $\widetilde K^*(c)$ satisfy the perturbed equation
obtained by adding an $O(1)$ term to the right-hand side
of the logarithmic FOC~\eqref{eq:log_FOC}.
Then
\[
  K^*(c) - \widetilde K^*(c) = O(1).
\]
In particular, the two-term expansion~\eqref{eq:kopt_iterated}
is stable under all such perturbations,
and all conclusions of Lemma~\ref{lem:kopt_variational}
and Corollary~\ref{cor:universal_lograte} remain valid.
\end{lemma}

\begin{proof}
Let $K_1=K^*$ and $K_2=\widetilde K^*$ solve
\[
  \tfrac{\alpha}{2}K_i = \beta\log c + \gamma\log K_i + C_i,
  \quad i=1,2,
\]
with $|C_1-C_2|\leq D$ for some fixed $D$.
Subtracting:
$\tfrac{\alpha}{2}(K_1-K_2)
= \gamma(\log K_1 - \log K_2) + (C_1-C_2)$.
By the mean value theorem,
$|\log K_1 - \log K_2| \leq |K_1-K_2|/\min(K_1,K_2)$.
By Lemma~\ref{lem:loglog} (sandwich~\eqref{eq:sandwich}),
$\min(K_1,K_2)\geq c_1\log c = \frac{2\beta}{\alpha}\log c$, so
$\tfrac{\alpha}{2}|K_1-K_2|
\leq \frac{\gamma\alpha}{2\beta\log c}|K_1-K_2| + D$.
For $c$ sufficiently large,
$\frac{\gamma\alpha}{2\beta\log c} < \alpha/4$,
hence $|K_1-K_2|\leq \frac{4D}{\alpha} = O(1)$.
\end{proof}

\begin{remark}[Discrete projection: continuous minimizer controls integer minimizer]
\label{rem:kopt_discrete}
Lemma~\ref{lem:kopt_variational} minimizes $F$ over $\R_+$.
The integer projection $K_{\rm int}=\lfloor K^*\rceil\in\N$
satisfies $|K_{\rm int}-K^*|\leq\frac{1}{2}$, and by Taylor:
\[
  F(K_{\rm int}) - F(K^*)
  \leq \tfrac{1}{8}F''(K^*) = O(c^{-\beta}(\log c)^{\gamma-1})
  \prec c^{-\beta}(\log c)^{1+\gamma}
\]
in $\mathcal{S}$ (Definition~\ref{def:scale}).
Hence the continuous minimizer controls the integer minimizer
up to a term asymptotically negligible relative to $F(K^*)$,
and Corollary~\ref{cor:universal_lograte}
holds for $K=K_{\rm int}\in\N$.
\end{remark}

\begin{remark}[Variational replaces heuristic saddle]
\label{rem:variational_vs_heuristic}
Prior to commit \texttt{679b3f2}, $\Kopt$ was obtained
by heuristic order-of-magnitude balance.
Lemma~\ref{lem:kopt_variational} replaces this
with a strictly convex functional, a FOC,
an iterated asymptotic expansion~\eqref{eq:kopt_iterated}
justified by the fixed-point structure~\eqref{eq:fixpoint}
and parameter-dependent contraction~\eqref{eq:contraction},
and a stability certificate (Lemma~\ref{lem:stability_kopt}).
The sandwich~\eqref{eq:sandwich} provides an
explicit two-sided bound on $K^*$ prior to any
logarithmic manipulation.
\end{remark}

\section{Layer-Survival Selection Principle}
\label{sec:layersurvival}
%% ============================================================

\begin{theorem}[Layer-Survival]
\label{thm:layer_survival}
For $n\geq 1$, under H1,
$\normh{A_K^{(n)}}{}
= O(K^{1+n\gamma}c^{-n\beta})$.
At $K=K^*\sim\frac{2\beta}{\alpha}\log c$:
$\normh{A_K^{(n)}}{}
= O(c^{-n\beta}(\log c)^{1+n\gamma})
\prec c^{-\beta}(\log c)^{1+\gamma}$
for all $n\geq 2$
(Proposition~\ref{prop:asymp_order}).
The rate is governed by $n=1$ alone.

\smallskip
\textit{Mechanism:}
each additional Langer factor adds $+\beta$ to the
$c$-exponent and $+\gamma$ to the log-exponent;
$c^{-\beta}\succ(\log c)^m$ for all $m$
$\Rightarrow$ $c$-hierarchy always selects $n=1$.
\end{theorem}

\begin{theorem}[Two-scale rate]
\label{thm:universal_rate}
For $\cAg\in\mathfrak{T}_\gamma$, $K\geq K_0$, $c\geq c_0$:
\[
  \normh{\cK-\cKA}{}
  \leq \Clin c^{-\beta}
  + \Cquad c^{-2\beta}
  + \Ctwo^{1/2}e^{-\alpha K/2}.
\]
\end{theorem}

\begin{corollary}[Universal logarithmic rate]
\label{cor:universal_lograte}
With $K=K_{\rm int}=\lfloor K^*(c)\rceil$
(Remark~\ref{rem:kopt_discrete}):
$\normh{\cK-\cKA}{}
= O(c^{-\beta}(\log c)^{1+\gamma})$.
The three contributions:
\textup{(i)} linear $O(c^{-\beta}(\log c)^{1+\gamma})$: leading;
\textup{(ii)} quadratic $\prec$ linear
  (Proposition~\ref{prop:asymp_order});
\textup{(iii)} tail $O(e^{-\alpha K^*/2})
  = O(c^{-\beta}(\log c)^{-\gamma})
  \prec$ linear (Remark~\ref{rem:scales}(ii)).
\end{corollary}

\begin{remark}[Spectral-geometric invariants]
\label{rem:invariants}
$1+\gamma$: leading exponent; from $\sum j^\gamma$;
governed by turning-point crowding.
$1+2\gamma$: quadratic-layer exponent;
subleading by one factor $c^{-\beta}$;
gap $\gamma$ = crowding cost per Langer interaction.

\smallskip
Note: $\gamma$ does not appear in the leading saddle
coefficient $2\beta/\alpha$ of $K^*$
(Lemma~\ref{lem:kopt_variational}(ii));
it governs the log-exponent of the rate and the
log-log correction~\eqref{eq:kopt_iterated}
to the optimal truncation.
\end{remark}

\begin{remark}[O3: sharpness and the lower-bound gap]
\label{rem:O3_sharpness}
The rate $O(c^{-\beta}(\log c)^{1+\gamma})$ is an upper bound.
A matching lower bound $\Omega(K^{1+\gamma})$ for the Langer term
(O3 in Section~\ref{sec:open}) would establish sharpness.
The gap is structural: the upper bound uses
$\sum_{j=1}^K j^\gamma \leq \frac{K^{1+\gamma}}{1+\gamma} + O(K^\gamma)$
(Euler--Maclaurin), while a lower bound requires a
positive-mass condition on the spectral measure below threshold,
i.e.\ that eigenfunction concentrations do not cancel.
Within $\mathfrak{T}_\gamma$ this is plausible but not proved;
a concrete obstruction would require constructing
$\cAg\in\mathfrak{T}_\gamma$ with
$\normh{\AKlin}{} = o(K^{1+\gamma}c^{-\beta})$,
which appears impossible under H1 but is not ruled out
by the current axioms.
\textit{Status:} open; non-blocking for all upper-bound conclusions.
\end{remark}
