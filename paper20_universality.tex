\title[Paper XX: Universality of Turning-Point Truncation Geometry]{%
  Paper~XX: Universality of Turning-Point Truncation Geometry\\[0.3em]
  \large Spectral-Geometric Invariants and Two-Scale Logarithmic Losses}

\author{\textit{prolate-gram-coercivity programme}}
\date{May 2026}

\begin{abstract}
Papers~XVIII and~XIX established, for the Airy turning-point model,
qualitative universality (BR3) and a quantitative two-scale rate
$O(c^{-1/3}(\log c)^{5/3})$ with polynomial truncation cost $O(K^{5/3})$.
This paper shows that the exponent $5/3 = 1 + 2/3$ is an instance of a
general law.

For any one-dimensional Sturm\textendash Liouville operator
$\cAg = -\partial_\xi^2 + V(\xi)$ with spectral growth exponent
$a_K \sim K^\gamma$ ($\gamma > 0$) satisfying an explicit
\emph{coercivity-propagation hypothesis}
(Hypothesis~\ref{hyp:coercprop}),
the optimal two-scale truncation rate is
\[
  O\!\left(c^{-\beta}(\log c)^{1+\gamma}\right),
\]
where $\beta > 0$ is the scaling-window exponent of the turning-point
rescaling.
The exponent $1+\gamma$ is a
\emph{spectral-geometric invariant of the two-scale truncation mechanism}:
it is determined by eigenvalue spacing and the rank-$K$ summation
structure, and cannot be improved without a $K$-uniform Langer bound.

The proof requires exactly one new lemma
(Lemma~\ref{lem:trunc_universal}); all other components of the
XVIII/XIX dependency DAG carry over verbatim
(Lemma~\ref{lem:gamma_agnostic}).
Logarithmic truncation losses are therefore not artifacts of
proof strategy, but spectral-geometric properties of turning-point models.
\end{abstract}

XIX showed \emph{that} the truncation cost is $O(K^{5/3})$ for the Airy

turning-point model in the class considered here.

\begin{remark}[Scope]
\label{rem:scope}
Paper~XX is restricted to one-dimensional, self-adjoint,
semibounded Sturm\textendash Liouville operators with simple turning points
and the spectral growth hypothesis~\eqref{eq:spectralgrowth}.
The parameter $\gamma$ ranges over all of $(0,\infty)$;
the constraint $\gamma \in (0,1]$ is \emph{not} imposed
(the power-potential family $V = |\xi|^\nu$ gives $\gamma = 2\nu/(\nu+2)
\nearrow 2$ as $\nu \to \infty$).
\end{remark}

\section{Abstract Turning-Point Framework}
\label{sec:framework}
%% ============================================================

\begin{definition}[Operator class $\mathfrak{T}_\gamma$]
\label{def:opclass}
For $\gamma > 0$, let $\mathfrak{T}_\gamma$ denote the class of
self-adjoint operators
\[
  \cAg = -\partial_\xi^2 + V(\xi)
  \quad\text{on } L^2(\R_+),
  \quad \mathcal{D}(\cAg) = H^2 \cap H^1_0(\R_+),
\]
with $V \geq 0$, $V(\xi) \to +\infty$, discrete spectrum
$0 < a_1 < a_2 < \cdots$, subject to:
\begin{enumerate}[label=(\roman*),nosep]
  \item \emph{Spectral growth:}
    \begin{equation}\label{eq:spectralgrowth}
      a_K \sim C_\gamma K^\gamma \quad (K\to\infty),
      \quad C_\gamma > 0;
    \end{equation}
  \item \emph{Turning-point regularity:} $V$ is $C^2$ near its
    simple zero, Langer--Olver applicable on compact $[0,M]$
    (\cite{Olver1997} Thm.~11.3);
  \item \emph{Coercivity propagation:}
    Hypothesis~\ref{hyp:coercprop} below holds.
\end{enumerate}
\end{definition}

\begin{definition}[Spectral growth exponent]
\label{def:spectralgrowth}
The parameter $\gamma > 0$ in~\eqref{eq:spectralgrowth}
is called the \emph{spectral growth exponent} of $\cAg$.
It is an asymptotic spectral datum, determined solely by
the eigenvalue spacing of $\cAg$.
\end{definition}

\begin{definition}[Scaling-window exponent]
\label{def:beta}
$\beta > 0$ denotes the exponent of the turning-point rescaling
window $c^{-\beta}$.
For the Airy model: $\beta = 1/3$ (XVIII~\texttt{def:airycoord}).
The value of $\beta$ is determined by the local behaviour of $V$
near the turning point; it is treated as a given input,
\emph{independent} of $\gamma$.
\end{definition}

\begin{remark}[Independence of $\gamma$ and $\beta$]
\label{rem:gammabeta}
$\gamma$ governs global eigenvalue spacing;
$\beta$ governs the local turning-point window.
The universal rate $O(c^{-\beta}(\log c)^{1+\gamma})$ depends
on both through separate mechanisms: $\beta$ controls the $c$-decay
and $1+\gamma$ controls the logarithmic overhead.
\end{remark}

\section{DAG Decomposition: Agnostic and Sensitive Nodes}
\label{sec:dag}
%% ============================================================

\begin{lemma}[$\gamma$-Agnostic nodes]
\label{lem:gamma_agnostic}
The following eight components of the XVIII/XIX dependency DAG
hold verbatim for every $\cAg \in \mathfrak{T}_\gamma$,
with no reference to the value of $\gamma$:
\begin{enumerate}[label=(\roman*),nosep]
  \item \emph{Tensor decomposition}
    $\SK - \cKA = \AK - \BK$ (XIX~\texttt{def:decomp}): pure algebra.
  \item \emph{Tensor norm bound per $j$}
    (XIX~\texttt{lem:langer\_rate} proof):
    only $\|f\otimes g\|_{L^2}=\|f\|_{L^2}\|g\|_{L^2}$ used.
  \item \emph{Bessel tail}
    $\sum_{j>K}\|u_j\|_{L^2([0,M])}^2\to 0$
    (XVIII~\texttt{lem:besseltail}):
    Bessel inequality and $L^1$ diagonal; no $V$.
  \item \emph{Tonelli/DCT}
    (XVIII~\texttt{rem:dct}):
    $L^1([0,M]^2)$ domination; no $V$.
  \item \emph{Slepian-type tail}
    $\lambda_{n^*+j}^{(c)}\leq e^{-\alpha j}$
    (XVIII~\texttt{lem:slepianunif}):
    ORX/Slepian; independent of $\gamma$.
  \item \emph{HS/Cauchy\textendash Schwarz on $\BK$}
    (XIX~\texttt{lem:tail\_rate}):
    spectral orthogonality only.
  \item \emph{Sequential $(c,K)$-limit}
    (XVIII~\texttt{rem:iterated}, XIX~\texttt{rem:nojoint}):
    no joint uniformity; no $V$.
  \item \emph{Norm-resolvent perturbation}
    (XVIII~\texttt{prop:normresolvent}):
    Kato~VI form-boundedness; requires only relative bound $0$.
\end{enumerate}
\end{lemma}

\begin{proof}
Each item is verified by inspection: in every case the only
inputs are functional-analytic (tensor products, Bessel,
Cauchy\textendash Schwarz, Kato) or the spectral tail bound (ORX/Slepian),
neither of which references $a_K\sim K^\gamma$.
\end{proof}

\begin{remark}[$\gamma$-Sensitive nodes: exactly four]
\label{rem:sensitive}
The following four nodes depend on $\gamma$,
all within the propagation spine:
\begin{enumerate}[label=(\roman*),nosep]
  \item Spectral spacing: $a_K\sim K^\gamma$ (hypothesis);
  \item Coercive threshold: $\mu(K) = C_K+1\sim K^\gamma$;
  \item Exclusion window: $\delta_0(K) = 2C_K+2\sim K^\gamma$;
  \item Lax\textendash Milgram constant:
    $\Cprime = O(M\Tg(K))$ under Hypothesis~\ref{hyp:coercprop}.
\end{enumerate}
\end{remark}

\section{Universal Truncation Geometry}
\label{sec:trunc}
%% ============================================================

\begin{definition}[Propagation spine]
\label{def:spine}
The \emph{propagation spine} of the truncation geometry is:
\[
\begin{tikzcd}[column sep=2.0em]
  a_K{\sim}K^\gamma
  \ar[r,"\ell_0"]
  & C_K
  \ar[r]
  & \delta_0(K)
  \ar[r,"\text{Step 2}"]
  & \mu(K){\sim}K^\gamma
  \ar[r,"\text{L-M}"]
  & C'(M,K)
  \ar[r,"\text{XIX}"]
  & C_1(M,K).
\end{tikzcd}
\]
The first five nodes scale like $K^\gamma$;
the last scales like $K^{1+\gamma}$ due to the rank-$K$ summation
$C_1 = KM\cdot C'(2C_\infty + C')$ with $C_\infty = O(1)$.
\end{definition}

\begin{hypothesis}[Coercivity propagation]
\label{hyp:coercprop}
There exists a function $\Tg\colon \mathbb{N}\to(0,\infty)$
with $\Tg(K)\sim K^\gamma$ such that the following holds:
for fixed $M$ and $K\geq 1$, there exist $c_0(M,K)$ and
a Langer-type $C^0([0,M])$-error bound
\begin{equation}\label{eq:langererror}
  \normh{e_j}{C^0([0,M])} \leq M\,\Tg(K)\,c^{-\beta},
  \qquad j\leq K,\; c\geq c_0(M,K),
\end{equation}
where $e_j = \tilde\psi_j - u_j$ is the Langer remainder.
\end{hypothesis}

\begin{remark}[Verification for Airy]
\label{rem:airyhyp}
For $\cAg = $ Airy ($\gamma = 2/3$, $\beta = 1/3$):
Hypothesis~\ref{hyp:coercprop} holds with $\Tg(K) = C_\gamma K^{2/3}$
by XVIII~\texttt{rem:langerkdep}.
The propagation chain
$a_K\sim K^{2/3}\Rightarrow\mu(K)\sim K^{2/3}\Rightarrow
\Cprime = O(MK^{2/3})$
is the Airy instance of the general spine.
\end{remark}

\begin{remark}[Status of the hypothesis for general $\mathfrak{T}_\gamma$]
\label{rem:hypstatus}
Hypothesis~\ref{hyp:coercprop} is an explicit, checkable condition:
it requires that the Langer--Lax\textendash Milgram coercivity argument
of XVIII~\texttt{lem:langer} Steps~1--3 propagates the threshold
$\mu(K)\sim K^\gamma$ into the $C^0([0,M])$ error bound at rate $c^{-\beta}$.
For operators in $\mathfrak{T}_\gamma$ with $C^2$ potential
and Olver-applicable turning point, this is
\emph{expected} to follow from the standard Langer theory
(\cite{Olver1997} Ch.~11) combined with
$\mu(K)\sim K^\gamma$, but a complete proof for general $\gamma$
remains to be written and constitutes the main open task of XX.
\end{remark}

\begin{lemma}[Universal truncation geometry]
\label{lem:trunc_universal}
Let $\cAg \in \mathfrak{T}_\gamma$ satisfy
Hypothesis~\ref{hyp:coercprop}.
For fixed $M < \infty$ and $K \geq 1$:
\begin{align}
  \Cprime &= O(M\,\Tg(K)) = O(MK^\gamma), \label{eq:Cprimescale}\\
  \ConeM  &= O(M^2 K^{1+\gamma}), \label{eq:C1scale}
\end{align}
with constants independent of $c$.
\end{lemma}

\begin{proof}
\textbf{Step 1: Coercive threshold.}
By~\eqref{eq:spectralgrowth} and XVIII~\texttt{lem:slbound}:
$C_K = a_K+1\sim C_\gamma K^\gamma$,
$\mu(K) = C_K+1\sim K^\gamma$,
$\delta_0(K) = 2C_K+2\sim K^\gamma$.

\medskip
\textbf{Step 2: Lax\textendash Milgram constant.}
By Hypothesis~\ref{hyp:coercprop}:
$\Cprime = O(M\,\Tg(K)) = O(MK^\gamma)$.
This is~\eqref{eq:Cprimescale}.

\medskip
\textbf{Step 3: Truncation constant.}
With $C_\infty = O(1)$ (uniform $C^0$ bound,
XVIII~\texttt{lem:linf}) and~\eqref{eq:Cprimescale}:
\begin{equation}\label{eq:C1expand}
  \ConeM = KM\cdot\Cprime\cdot(2C_\infty + \Cprime)
  = O(K\cdot MK^\gamma \cdot 1)
  = O(MK^{1+\gamma}),
\end{equation}
since the dominant term is $KM\cdot\Cprime\cdot 2C_\infty$
(the $(\Cprime)^2$ term is $O(K^{1+2\gamma})$ and is
\emph{not} dominant when $C_\infty = O(1)$).
Verification: $\gamma = 2/3$ gives $C_1 = O(MK^{5/3})$,
matching XIX~\texttt{eq:C1scale}. $\hfill\square$
\end{proof}

\begin{remark}[Why $K^{1+\gamma}$ and not $K^{1+2\gamma}$]
\label{rem:exponent}
The bound~\eqref{eq:C1expand} uses $C_\infty = O(1)$
(the eigenfunctions $u_j$ of the limit operator $\cAg$ are
uniformly bounded on $[0,M]$, independent of $K$ and $c$).
This is the key that suppresses the $(\Cprime)^2 = O(K^{2\gamma})$
term in favour of $\Cprime \cdot C_\infty = O(K^\gamma)$,
yielding $K \cdot O(K^\gamma) = O(K^{1+\gamma})$.
If $C_\infty$ were to grow with $K$, the dominant exponent
would be $1+2\gamma$ instead.
\end{remark}

\section{Universal Rate Theorem}
\label{sec:main}
%% ============================================================

\begin{theorem}[Universal two-scale truncation rate]
\label{thm:universal_rate}
Let $\cAg \in \mathfrak{T}_\gamma$ satisfy
Hypothesis~\ref{hyp:coercprop}, with scaling-window exponent $\beta>0$.
For fixed $K\geq K_0$ and $c\geq c_0(M,K)$:
\begin{equation}\label{eq:universal_twoscale}
  \normh{\cK-\cKA}{L^2([0,M]^2)}
  \leq
  \underbrace{C_1(M,K)\cdot c^{-\beta}}_{
    O(K^{1+\gamma})\,c^{-\beta}}
  +
  \underbrace{\Ctwo^{1/2}\cdot e^{-\alpha K/2}}_{
    \text{uniform in }c}.
\end{equation}
The two bounds come from vertex-disjoint parts of the dependency DAG.
No joint $(c,K)$-uniformity is assumed.
\end{theorem}

\begin{proof}
Decomposition + triangle inequality (Lemma~\ref{lem:gamma_agnostic}(i));
bounds from Lemma~\ref{lem:trunc_universal} and XIX~\texttt{lem:tail\_rate}
(Lemma~\ref{lem:gamma_agnostic}(vi)).
\end{proof}

\begin{corollary}[Universal logarithmic rate]
\label{cor:universal_lograte}
Let
\begin{equation}\label{eq:Kopt}
  \Kopt(c) := \left\lfloor\frac{2\beta}{\alpha}\log c\right\rfloor.
\end{equation}
For $c$ large enough that $\Kopt\geq K_0$:
\begin{equation}\label{eq:lograte_universal}
  \normh{\cK-\cKA}{L^2([0,M]^2)}\Big|_{K=\Kopt}
  = O\!\left(c^{-\beta}(\log c)^{1+\gamma}\right).
\end{equation}
\end{corollary}

\begin{proof}
With $\Kopt\sim(2\beta/\alpha)\log c$:
\begin{itemize}[nosep]
  \item \emph{Term 1.}
    $C_1(M,\Kopt) = O(\Kopt^{1+\gamma}) = O((\log c)^{1+\gamma})$,
    so Term~1 is $O(c^{-\beta}(\log c)^{1+\gamma})$.
  \item \emph{Term 2.}
    $e^{-\alpha\Kopt/2} = e^{-\beta\log c} = c^{-\beta}$,
    so Term~2 is $O(c^{-\beta})$.
\end{itemize}
Since $c^{-\beta} = o(c^{-\beta}(\log c)^{1+\gamma})$,
Term~1 dominates and~\eqref{eq:lograte_universal} follows.
\end{proof}

\begin{remark}[Correction vs.\ XIX coupling constant]
\label{rem:Koptcorrection}
Paper~XIX used $\Kopt = \lfloor(\beta/\alpha)\log c\rfloor$,
which gives $e^{-\alpha\Kopt/2} = O(c^{-\beta/2})$;
this is larger than $c^{-\beta}$ and would not balance Term~1.
The correct factor is $2\beta/\alpha$
(as in~\eqref{eq:Kopt}), ensuring $e^{-\alpha\Kopt/2} = O(c^{-\beta})$.
For the Airy case ($\beta=1/3$, $\alpha$ as in XVIII):
$\Kopt = \lfloor(2/(3\alpha))\log c\rfloor$.
This corrects XIX~\texttt{cor:lograte} at the level of the
coupling constant only; the rate $O(c^{-1/3}(\log c)^{5/3})$ is unchanged.
\end{remark}

\begin{remark}[Spectral-geometric invariant]
\label{rem:invariant}
The exponent $1+\gamma$ is a
\emph{spectral-geometric invariant of the two-scale truncation mechanism}:
it is determined by the eigenvalue spacing ($a_K\sim K^\gamma$, spectral)
and the rank-$K$ summation in $C_1 = KM\cdot C'\cdot(2C_\infty+C')$
(geometric/truncation architecture).
Neither factor alone determines $1+\gamma$; both are required.
The exponent cannot be improved to $1$ (i.e., a single logarithm)
without eliminating the $K$-growth of $\Cprime$,
which would require a $K$-uniform Langer bound.
\end{remark}

\begin{remark}[Limiting cases and geometric interpretation]
\label{rem:limits}
\begin{itemize}[nosep]
  \item $\nu\to 0$ (flat potential): $\gamma\to 0$, $1+\gamma\to 1$.
    Eigenvalues crowd slowly; truncation penalty approaches a
    single logarithm.
  \item $\nu\to\infty$ (hard-wall / box): $\gamma\to 2$,
    $1+\gamma\to 3$.
    Eigenvalues crowd rapidly; truncation at rank $K$ discards
    many geometrically relevant modes, producing cubic-logarithmic cost.
  \item $\gamma > 1$ is admissible in $\mathfrak{T}_\gamma$
    (e.g.\ $\nu > 2$ in the power family); the range $\gamma\in(0,\infty)$
    is necessary.
\end{itemize}
Geometrically: $\gamma$ measures the rate of eigenvalue crowding
near the turning point, and $1+\gamma$ measures the
truncation overhead this crowding imposes on the two-scale approximation.
\end{remark}

\section{Conceptual Conclusion}
\label{sec:conclusion}
%% ============================================================

Papers~XVIII, XIX, and XX form a trilogy:

\medskip
\textbf{XVIII.} The rescaled PSWF kernel converges to the Airy kernel:
qualitative universality via two-scale iterated limit.

\medskip
\textbf{XIX.} The convergence rate is quantified by the decomposition
$S_K - K_{\mathcal{A}} = A_K - B_K$, with separated rates
$O_K(c^{-1/3})$ and $O(e^{-\alpha K})$;
the truncation cost is $O(K^{5/3})$,
with $5/3$ determined by Airy zero spacing.

\medskip
\textbf{XX.} The exponent $5/3 = 1+2/3$ is an instance of $1+\gamma$:
for any turning-point model in $\mathfrak{T}_\gamma$
satisfying Hypothesis~\ref{hyp:coercprop},
the optimal two-scale truncation rate is
$O(c^{-\beta}(\log c)^{1+\gamma})$,
and $1+\gamma$ is a spectral-geometric invariant of
the two-scale truncation mechanism.

\medskip
The deepest consequence:
\begin{quote}\itshape
Logarithmic truncation losses are not artifacts of proof strategy.
They are spectral-geometric properties of turning-point models,
encoded in the eigenvalue growth exponent $\gamma$
through the rank-$K$ summation architecture of the
two-scale decomposition.
\end{quote}
