  Paper~XX: Universality of Turning-Point Truncation Geometry\\[0.3em]
  \large Spectral-Geometric Invariants, Two-Scale Logarithmic Losses,
  and Structural Obstruction to Rate Improvement}

\author{\textit{prolate-gram-coercivity programme}}
\date{May 2026 --- FROZEN}

\begin{abstract}
For operators $\cAg\in\mathfrak{T}_\gamma$
(spectrally regularized WKB class, Definition~\ref{def:opclass}),
the Langer term $\AK = \AKlin + \AKquad$ satisfies,
under the pointwise Langer scaling law
$\|e_j\|_{C^0}\leq C_0 j^\gamma c^{-\beta}$
(Hypothesis~\ref{hyp:langer_pointwise}),
\[
  \|\AKlin\| \leq C_{\rm lin}\,K^{1+\gamma}c^{-\beta},
  \qquad
  \|\AKquad\| \leq C_{\rm quad}\,K^{1+2\gamma}c^{-2\beta}
\]
for all $K\geq K_0$, $c\geq c_0$, with $c$-admissible constants.
The optimal truncation $\Kopt$ is the unique minimizer of
$F(K;c) := L\,K^{1+\gamma}c^{-\beta} + T\,e^{-\alpha K/2}$
over $K\in[c_1\log c,c_2\log c]$, satisfying
$K^*(c) = \frac{2\beta}{\alpha}\log c + \frac{2\gamma}{\alpha}\log\log c + O(1)$.
For each fixed $\gamma>0$, the diagonal rate
$O(c^{-\beta}(\log c)^{1+\gamma})$
holds for every $\cAg\in\mathfrak{T}_\gamma$ satisfying H1--H2
(Corollary~\ref{cor:universal_lograte}).
The central result is that this rate is \emph{structurally rigid}:
within the simultaneous validity of (S1)--(S3), no asymptotically
better rate is compatible with the two-scale truncation geometry
(Theorem~\ref{thm:rate_obstruction}).
The three structural components play distinct roles:
(S1) generates the exponent pair $(\gamma,\beta)$;
(S2) --- the Euler--Maclaurin structural property ---
ensures $c$-admissibility of remainder coefficients,
without which the asymptotic scale $\mathcal{S}$ loses its
poset structure and the balance fixed-point argument collapses
(Lemma~\ref{lem:S2_knockdown});
(S3) licenses the diagonal substitution $K\mapsto K^*(c)$.
Rigidity is further supported by decomposition stability
(Lemma~\ref{lem:decomp_stability}),
balance fixed-point uniqueness
(Lemma~\ref{lem:balance_fixpoint}),
H1 exponent rigidity
(Lemma~\ref{lem:H1_rigidity}),
and scope remarks
(Remarks~\ref{rem:non_tautology}--\ref{rem:method_vs_operator}).
All results are conditional on H1--H2.
\end{abstract}

\begin{remark}[Roles of S1, S2, S3]
\label{rem:S123_roles}
The three structural components of Theorem~\ref{thm:rate_obstruction}
play logically distinct and mutually irreducible roles:
\begin{itemize}[nosep]
  \item \textbf{(S1) = H1 (Langer scaling):}
    generates the exponent pair $(\gamma,\beta)$ and produces
    the algebraic growth $\Phi_1 = C_{\rm lin}K^{1+\gamma}c^{-\beta}$.
    Without S1, no growth term exists and the balance equation is vacuous.
  \item \textbf{(S2) = EM structural property:}
    the \emph{scale admissibility axiom}.
    It ensures that remainder coefficients are $c$-admissible,
    which is necessary and sufficient for the asymptotic scale
    $\mathcal{S} = \{c^{-s}(\log c)^t\}$ to carry a well-defined
    poset structure.
    Without S2, the dominance relation $\Phi_1\succ\Phi_2$ is not
    well-defined and the balance fixed-point argument of
    Lemma~\ref{lem:balance_fixpoint}(iii) collapses
    (Lemma~\ref{lem:S2_knockdown}).
  \item \textbf{(S3) = Substitution Stability:}
    licenses the passage from the fixed-parameter regime
    $(\forall K,c$ independently) to the diagonal regime
    ($K = K^*(c)$). Without S3, the optimality condition~\eqref{eq:FOC}
    cannot be substituted into the truncation bound.
\end{itemize}
The three roles are:\quad
exponent generation (S1),\quad
scale comparability (S2),\quad
diagonal projection (S3).
\end{remark}

\begin{remark}[$c$-admissible constants]
\label{rem:cadmissible}
A constant is \emph{$c$-admissible} if it is independent of $c$.
Every constant falls under one of four declared sources:
\begin{itemize}[nosep]
  \item \textbf{Axiomatic:} $C_0$, $C_\infty$ by H1--H2.
  \item \textbf{Analytic:} $C_\sigma = \frac{|B_2|}{12}\sigma(\sigma-1)$.
  \item \textbf{Derived:} products/sums of $c$-admissible constants
    (Lemma~\ref{lem:cadm_closure}).
  \item \textbf{Operator-intrinsic:} $C_2(\alpha)$.
\end{itemize}
\end{remark}

\begin{lemma}[$c$-admissible closure]
\label{lem:cadm_closure}
The class of $c$-admissible constants is closed under
finite products and finite sums.
\end{lemma}
\begin{proof}
$A,B$ independent of $c$ $\Rightarrow$ $AB$ and $A+B$ independent of $c$.
\end{proof}

\begin{proposition}[Substitution Stability Principle]
\label{prop:substitution_stability}
\emph{Intent.}
This proposition records an inference step, not a new analytical
estimate: substituting a deterministic map $K(c)$ into a
pointwise-valid inequality is admissible as a pointwise composition.

\emph{Setup.}
Let $|E(K,c)| \leq \sum_{n=1}^N C_n\,K^{a_n}c^{-b_n}$
hold for all $K\geq K_0$, $c\geq c_0$, $c$-admissible $C_n$.
Let $K(c)\in[c_1\log c,\,c_2\log c]$ for all large $c$.

\begin{itemize}[nosep]
  \item[(i)] \textbf{Pointwise substitution.}
    $|E(K(c),c)|\leq\sum_n C_n(K(c))^{a_n}c^{-b_n}$;
    no uniform-in-$K$ structure is used.
  \item[(ii)] \textbf{Window control via monotonicity.}
    For $K(c)\in[c_1\log c, c_2\log c]$:
    $\sum_n C_n (K(c))^{a_n} c^{-b_n}
      \leq \sum_n C_n (c_2\log c)^{a_n} c^{-b_n}$.
  \item[(iii)] \textbf{Asymptotic order.}
    $(K(c))^{a_n}=O(\log^{a_n}c)$;
    precise correction by~\eqref{eq:Kstar_power}.
  \item[(iv)] \textbf{Dominance.}
    By Propositions~\ref{prop:asymp_order} and~\ref{prop:scale_algebra}.
\end{itemize}
\end{proposition}

\begin{proof}
(i) pointwise; (ii) monotonicity + window upper bound;
(iii) sandwich~\eqref{eq:sandwich}; (iv) Props.~\ref{prop:asymp_order}--\ref{prop:scale_algebra}.
\end{proof}

\section{Asymptotic Scale}

\begin{definition}[Asymptotic scale]
\label{def:scale}
$\mathcal{S}=\{c^{-s}(\log c)^t : s>0,t\in\R\}$;
$c^{-s_1}(\log c)^{t_1}\succ c^{-s_2}(\log c)^{t_2}$
iff $s_1<s_2$ or ($s_1=s_2$, $t_1>t_2$).
The dominance relation is the lexicographic order on $(s,t)\in\R^2$.
\emph{The relation $\prec$ on $\mathcal{S}$ is well-defined only when
all implicit constants in asymptotic comparisons are $c$-admissible.}
\end{definition}

\begin{proposition}[Scale algebra]
\label{prop:scale_algebra}
$f\cdot g\in\mathcal{S}$ with exponents added;
$f/g\to\infty\Rightarrow f+g=O(f)$.
\end{proposition}

\begin{proposition}[Layer dominance]
\label{prop:asymp_order}
With $K=K^*(c)$, $n\geq 2$:
$c^{-n\beta}(\log c)^{1+n\gamma}\prec c^{-\beta}(\log c)^{1+\gamma}$.
\end{proposition}

\section{Truncation Geometry and S2 Knockdown}
\label{sec:trunc}
%% ============================================================

\begin{lemma}[Universal truncation geometry]
\label{lem:trunc_universal}
Under H1--H2, for all $K\geq 1$, $c\geq c_0$:
\begin{align}
  \normh{\AKlin}{}
    &\leq C_{\rm lin}K^{1+\gamma}c^{-\beta}
    +C_{\sigma,1}K^\gamma c^{-\beta},
    \label{eq:AKlin}\\
  \normh{\AKquad}{}
    &\leq C_{\rm quad}K^{1+2\gamma}c^{-2\beta}
    +C_{\sigma,2}K^{2\gamma}c^{-2\beta},
    \label{eq:AKquad}
\end{align}
all constants $c$-admissible.
\end{lemma}

\begin{proof}
Euler--Maclaurin:
\begin{equation}\label{eq:EM}
  \sum_{j=1}^K j^\sigma
  =\frac{K^{1+\sigma}}{1+\sigma}+\frac{1}{2}K^\sigma+R_\sigma(K),
\end{equation}
\begin{equation}\label{eq:EM_remainder}
  |R_\sigma(K)|\leq C_\sigma K^{\sigma-1},\quad
  C_\sigma=\tfrac{|B_2|}{12}\sigma(\sigma-1),\quad B_2=\tfrac{1}{6}.
\end{equation}
Multiply by $c$-admissible prefactors; close by Lemma~\ref{lem:cadm_closure}.
\end{proof}

\begin{remark}[EM structural property --- scale admissibility axiom]
\label{rem:EM_growth}
The Euler--Maclaurin formula serves two distinct functions in this paper:
\begin{enumerate}[nosep,label=(\roman*)]
  \item \textbf{Analytic function:}
    converts the discrete spectral sum $\sum_{j=1}^K j^\sigma$
    into a leading power $K^{1+\sigma}/(1+\sigma)$ plus controlled lower-order terms.
  \item \textbf{Scale admissibility function:}
    the remainder coefficient $C_\sigma = \frac{|B_2|}{12}\sigma(\sigma-1)$
    is a \emph{universal explicit constant}, independent of $K$ and $c$.
    This makes every coefficient in~\eqref{eq:AKlin}--\eqref{eq:AKquad}
    $c$-admissible, which is the necessary precondition for the
    asymptotic dominance relation $\Phi_1 \succ \Phi_2$ to be
    well-defined in $\mathcal{S}$ (Definition~\ref{def:scale}).
\end{enumerate}
Role (ii) is structural component (S2).
Lemma~\ref{lem:S2_knockdown} below shows that (ii) cannot be replaced
by a weaker summation method without destroying the poset structure of $\mathcal{S}$.
\end{remark}

\begin{lemma}[S2 knockdown: loss of scale admissibility]
\label{lem:S2_knockdown}
Assume that the EM structural property~(S2) is replaced by
an arbitrary summation method yielding only
\begin{equation}\label{eq:S2weak}
  \sum_{j=1}^K j^\gamma
  = a_\gamma K^{1+\gamma} + R_\gamma^*(K),
  \qquad a_\gamma \in \R_{>0},
  \qquad R_\gamma^*(K) = O_*(K^\gamma),
\end{equation}
where the implicit constant in $O_*(\cdot)$ is \emph{not assumed
$c$-admissible} (i.e.\ may depend on $K$ or $c$).
Then:
\begin{itemize}[nosep]
  \item[\textup{(i)}] \textbf{Loss of scale admissibility.}
    The truncation bound~\eqref{eq:AKlin} takes the form
    $\normh{\AKlin}{} \leq C_*(K,c)\,K^{1+\gamma}c^{-\beta}$,
    where $C_*(K,c)$ is not $c$-admissible in general.
    Hence $\Phi_1 = C_*(K,c)\,K^{1+\gamma}c^{-\beta}$ is
    not an element of $\mathcal{S}$ in the sense of
    Definition~\ref{def:scale}.

  \item[\textup{(ii)}] \textbf{Collapse of the poset structure.}
    The dominance relation $\Phi_1 \succ \Phi_2$ is defined
    via lexicographic comparison of the exponent pairs
    $(\beta, 1+\gamma)$ vs.~$(2\beta, 1+2\gamma)$
    in Definition~\ref{def:scale}.
    This comparison is valid only when all implicit constants are
    $c$-admissible; see the closing sentence of Definition~\ref{def:scale}.
    Without S2, $\Phi_1$ carries a non-admissible factor,
    so the comparison $\Phi_1 \succ \Phi_2$ is not well-defined
    as a statement in $\mathcal{S}$.

  \item[\textup{(iii)}] \textbf{Failure of the balance fixed-point argument.}
    Lemma~\ref{lem:balance_fixpoint}(iii) asserts
    $\Psi(K(c),c) \asymp c^{-\beta}(\log c)^{1+\gamma}$
    for $K(c)$ in the logarithmic window.
    This $\asymp$-equivalence requires that all constants in
    $(K(c))^{1+\gamma} \asymp (\log c)^{1+\gamma}$
    are $c$-admissible (Proposition~\ref{prop:substitution_stability}(ii)).
    Without S2, the constant $C_*(K,c)$ in $\Phi_1$ is not
    $c$-admissible, so the $\asymp$-relation does not hold in $\mathcal{S}$,
    and the strict separation argument of Lemma~\ref{lem:balance_fixpoint}
    has no valid conclusion.
\end{itemize}

\emph{Upshot.}
S2 is not an analytical convenience but a
\emph{structural admissibility axiom}: it is the unique point in the
argument where the discrete-to-continuous spectral transfer
$j \mapsto j^\sigma$
is endowed with $c$-admissible coefficients.
The logical gap created by its removal is a \emph{quantifier shift}:
with S2, every remainder coefficient satisfies
$\forall K,c\;\exists C_\sigma^{\rm EM}\text{ ($c$-admissible)}: |R_\sigma(K)|\leq C_\sigma^{\rm EM} K^{\sigma-1}$;
without S2, one has only
$\forall K,c\;\exists C_*(K,c): |R^*(K)|\leq C_*(K,c) K^\gamma$,
where the \emph{existence} of $C_*(K,c)$ does not imply
$c$-admissibility.
\end{lemma}

\begin{proof}
\textup{(i)} The leading coefficient $C_{\rm lin}$ in
Lemma~\ref{lem:trunc_universal} is a product of $C_0$ (from H1)
and the EM coefficient $1/(1+\gamma)$ from~\eqref{eq:EM}.
Without~\eqref{eq:EM_remainder}, the remainder $R_\gamma^*(K)$
has an implicit constant that is not controlled independently of $c$;
after absorbing it into the leading factor,
the resulting $C_*(K,c)$ may depend on $c$.
By Definition~\ref{def:scale}, only elements with $c$-admissible
implicit constants belong to $\mathcal{S}$.

\textup{(ii)} The lexicographic comparison in Definition~\ref{def:scale}
compares exponent pairs; it implicitly assumes that all prefactors
are $c$-admissible so that the comparison is independent of $c$.
If $C_*(K,c)$ is not $c$-admissible, then for different values of $c$
the prefactor of $K^{1+\gamma}c^{-\beta}$ changes, and
$\Phi_1 \succ \Phi_2$ may hold for some $c$ but not others.

\textup{(iii)} The $\asymp$-equivalence in
Lemma~\ref{lem:balance_fixpoint}(iii) is established via
Proposition~\ref{prop:substitution_stability}(ii), which requires
all implicit constants to be $c$-admissible to obtain a
$c$-uniform upper and lower bound.
Without S2, $C_*(K,c)$ may grow with $c$,
invalidating the window-based uniform bound.
\end{proof}

\section{Variational Optimal Truncation}
\label{sec:kopt}
%% ============================================================

\begin{lemma}[$K^*(c)\to\infty$]
\label{lem:kstar_to_infty}
$K^*(c)\to\infty$ as $c\to\infty$.
\end{lemma}

\begin{lemma}[Sandwich]
\label{lem:loglog}
With $c_1=2\beta/\alpha$, $c_2=2(\beta+1)/\alpha$:
\begin{equation}\label{eq:sandwich}
  c_1\log c\leq K^*(c)\leq c_2\log c,
  \qquad \log K^*(c)=\log\log c+O(1).
\end{equation}
\end{lemma}

\begin{lemma}[Variational $K^*$]
\label{lem:kopt_variational}
\textup{(i)} On $[c_1\log c,c_2\log c]$, $F''(K;c)>0$;
unique $K^*(c)$ with
\begin{equation}\label{eq:FOC}
  L(1+\gamma){K^*}^\gamma c^{-\beta}=\tfrac{\alpha}{2}Te^{-\alpha K^*/2}.
\end{equation}

\textup{(ii)} Iterated asymptotics:
\begin{equation}\label{eq:kopt_iterated}
  K^*(c)=\frac{2\beta}{\alpha}\log c+\frac{2\gamma}{\alpha}\log\log c+O(1).
\end{equation}

\textup{(ii')} $\lambda(c)=\frac{2\gamma}{\alpha K^*(c)}\to 0$:
self-consistency certificate only.

\textup{(iii)} $F(K^*;c)=O(c^{-\beta}(\log c)^{1+\gamma})$.
\end{lemma}

\begin{lemma}[Stability]
\label{lem:stability_kopt}
$O(1)$ perturbation of~\eqref{eq:FOC} shifts $K^*$ by $O(1)$.
\begin{proof}
Implicit function:
$\frac{dK^*}{dC_1}=\frac{1}{\alpha/2-\gamma/K^*}=O(1)$.
\end{proof}
\end{lemma}

\section{Diagonal Substitution}
\label{sec:diagonal}
%% ============================================================

\begin{lemma}[Diagonal substitution]
\label{lem:diagonal_sub}
Applies Proposition~\ref{prop:substitution_stability} at $K(c)=K^*(c)$.

$h(c):=K^*(c)/(\tfrac{2\beta}{\alpha}\log c)\to 1$;
$|h(c)-1|=O(\log\log c/\log c)$.
For large $c$: $h(c)\in[1/2,3/2]$;
constant by MVT bounded by $(1+\gamma)2^\gamma$,
finite on every compact $[\gamma_0,\gamma_1]$;
$(1+\gamma)2^\gamma\to\infty$ for $\gamma\to\infty$ explicit.
Hence
\begin{equation}\label{eq:Kstar_power}
  (K^*(c))^{1+\gamma}
  =\Bigl(\tfrac{2\beta}{\alpha}\log c\Bigr)^{1+\gamma}
  \Bigl(1+O\Bigl(\tfrac{\log\log c}{\log c}\Bigr)\Bigr).
\end{equation}
Conclusion: $\normh{\cK-\cKA}{}=O(c^{-\beta}(\log c)^{1+\gamma})$.
\end{lemma}

\begin{theorem}[Layer-Survival]

For $n\geq 1$, H1, all $K\geq K_0$:

\end{theorem}

\section{Main Rate Theorem}
\label{sec:main}
%% ============================================================

\begin{theorem}[Two-scale rate]
\label{thm:universal_rate}
For each fixed $\gamma>0$, $\cAg\in\mathfrak{T}_\gamma$,
all $K\geq K_0$, $c\geq c_0$:
\[
  \normh{\cK-\cKA}{}
  \leq C_{\rm lin}K^{1+\gamma}c^{-\beta}
  +C_{\rm quad}K^{1+2\gamma}c^{-2\beta}
  +C_2(\alpha)^{1/2}e^{-\alpha K/2},
\]
all constants $c$-admissible.
\end{theorem}

\begin{corollary}[Universal logarithmic rate]
\label{cor:universal_lograte}
For each fixed $\gamma>0$, at $K=\lfloor K^*(c)\rceil$:
$\normh{\cK-\cKA}{}=O(c^{-\beta}(\log c)^{1+\gamma})$.
\end{corollary}

\section{H1 Exponent Rigidity}
\label{sec:H1rigidity}
%% ============================================================

\begin{lemma}[H1 exponent rigidity]
\label{lem:H1_rigidity}
Let $\widetilde{\mathrm{H1}}$ be any hypothesis of the form
$\|e_j\|_{C^0} \leq \tilde C_0 j^{\tilde\gamma} c^{-\tilde\beta}$,
$\tilde C_0$ $c$-admissible,
and let $\widetilde\Phi_1 := \tilde C_{\rm lin} K^{1+\tilde\gamma} c^{-\tilde\beta}$.

\begin{itemize}[nosep]
  \item[\textup{(i)}] \textbf{Exponent propagation.}
    The FOC~\eqref{eq:FOC} with $(\tilde\gamma,\tilde\beta)$ yields
    $\widetilde K^*(c) = \frac{2\tilde\beta}{\alpha}\log c
    + \frac{2\tilde\gamma}{\alpha}\log\log c + O(1)$
    and rate $O(c^{-\tilde\beta}(\log c)^{1+\tilde\gamma})$.
    The map $(\tilde\gamma,\tilde\beta)\mapsto(1+\tilde\gamma,\tilde\beta)$
    is injective; distinct exponent pairs yield distinct rates.

  \item[\textup{(ii)}] \textbf{Logarithmic perturbation instability.}
    If $\|e_j\|_{C^0} \leq C_0 j^\gamma (\log j)^\delta c^{-\beta}$,
    $\delta>0$, then the rate becomes
    $O(c^{-\beta}(\log c)^{1+\gamma}(\log\log c)^\delta)$,
    strictly larger than the baseline.
\end{itemize}
\end{lemma}

\begin{proof}
(i) The log-form of~\eqref{eq:FOC} is shape-invariant under
replacement $(\gamma,\beta)\to(\tilde\gamma,\tilde\beta)$;
iterated asymptotics of Lemma~\ref{lem:kopt_variational}(ii) applies verbatim.
(ii) $\log K^*(c) = \log\log c + O(1)$ by Lemma~\ref{lem:loglog};
pointwise substitution yields $(\log K^*)^\delta\sim(\log\log c)^\delta$;
Proposition~\ref{prop:scale_algebra} gives strict inflation.
\end{proof}

\begin{remark}[Scope of H1 rigidity]
\label{rem:H1_rigidity_scope}
(a)--(b) within power-law class: established by
Lemma~\ref{lem:H1_rigidity}.
(c) Envelope minimality and (d) hypotheses outside the power-law
class: open; see Conjecture~\ref{conj:H1_envelope}.
\end{remark}

\section{Structural Obstruction to Rate Improvement}
\label{sec:obstruction}
%% ============================================================

\begin{lemma}[Decomposition stability]
\label{lem:decomp_stability}
With $\Phi_1,\Phi_3\geq 0$ and $c$-admissible constants
(guaranteed by S2 via Lemma~\ref{lem:S2_knockdown}(i)):
$\Phi_1+\Phi_3\geq\Phi_1$;
no cancellation between $\Phi_1$ and $\Phi_3$ is possible.
\end{lemma}

\begin{remark}[Role]
\label{rem:decomp_role}
Any rate improvement must reduce each dominant term individually;
under (S1)--(S3) this forces the analysis of $K(c)$
in Lemma~\ref{lem:balance_fixpoint}.
\end{remark}

\begin{lemma}[Balance fixed-point]
\label{lem:balance_fixpoint}
Define $\Psi := \Phi_1+\Phi_3$.
Under \textnormal{(S1)--(S3)} (and in particular with S2 ensuring
that $\Phi_1\in\mathcal{S}$ and constants $c$-admissible):
\begin{itemize}[nosep]
  \item[\textup{(i)}] $K(c)=o(\log c)$:
    $\Psi\succ c^{-\beta}(\log c)^{1+\gamma}$.
  \item[\textup{(ii)}] $K(c)\geq C\log c$, $C>c_2$:
    $\Psi\succ c^{-\beta}(\log c)^{1+\gamma}$.
  \item[\textup{(iii)}] $K(c)\in[c_1\log c,c_2\log c]$:
    $\Psi\asymp c^{-\beta}(\log c)^{1+\gamma}$;
    no $K(c)$ in the window achieves $o(\cdot)$.
    \emph{(Requires S2: the $\asymp$-constants must be $c$-admissible;
    Lemma~\ref{lem:S2_knockdown}(iii) shows this fails without S2.)}
  \item[\textup{(iv)}] For every $K(c)$,
    $\Psi\asymp c^{-\beta}(\log c)^{1+\gamma}$ or strictly larger.
\end{itemize}
\end{lemma}

\begin{proof}
(i)--(ii): regime analysis as before.
(iii): $\asymp$ from~\eqref{eq:Kstar_power} via
Proposition~\ref{prop:substitution_stability}(ii);
$c$-admissibility of constants certified by S2
(Remark~\ref{rem:EM_growth}(ii) and Lemma~\ref{lem:cadm_closure}).
(iv): exhaustion.
\end{proof}

\begin{remark}[Nature of the fixed point]
\label{rem:fixpoint_nature}
The balance point is an exact invariant of (S1,S2,S3),
not a local minimum. The scale $c^{-\beta}(\log c)^{1+\gamma}$
is \emph{forced} by the constraint geometry.
\end{remark}

\begin{theorem}[Structural Obstruction]
\label{thm:rate_obstruction}
Let $\cAg\in\mathfrak{T}_\gamma$ satisfy H1--H2 and (S2),
admitting decomposition~\eqref{eq:AKlin}--\eqref{eq:AKquad}
with $c$-admissible constants.
For any admissible truncation-based representation compatible with
\textnormal{(S1)--(S3)}, if for some $K(c)$,
$\normh{\cK-\cKA}{} = o(c^{-\beta}(\log c)^{1+\gamma})$,
then at least one of \textnormal{(S1)--(S3)} must fail.
\end{theorem}

\begin{proof}
Step~1: Lemma~\ref{lem:decomp_stability} (uses S2 for $c$-admissibility).
Step~2: Lemma~\ref{lem:balance_fixpoint}(i)--(ii) (outside window).
Step~3: Lemma~\ref{lem:balance_fixpoint}(iii) (inside window; uses S2).
Step~4: exhaustion of all $K(c)$.
\end{proof}

\begin{remark}[Scope]
\label{rem:obstruction_scope}
Obstruction within $\mathcal{M}(\mathrm{S1,S2,S3})$;
not a lower bound in the sense of O3a.
Remarks~\ref{rem:non_tautology}--\ref{rem:method_vs_operator} close R1--R3.
\end{remark}

\begin{remark}[Non-tautology]
\label{rem:non_tautology}
$\Phi_1$ (H1-driven) and $\Phi_3$ ($\BK$-driven, H1-independent)
have disjoint analytic origins. Changing H1 shifts $\Phi_1$;
$\Phi_3$ is invariant. The balance is a genuine coupling.
\end{remark}

\begin{remark}[Constant dependence]
\label{rem:constant_dependence}
All constants in Lemma~\ref{lem:balance_fixpoint} are $c$-uniform;
dependence on $(\gamma,\beta,\alpha,C_{\rm lin},C_2)$ is explicit.
No $K(c)$-dependence enters after substitution (S3 pointwise).
No uniformity in $\gamma\to\infty$ claimed.
\end{remark}

\begin{remark}[Method class vs.\ operator class]
\label{rem:method_vs_operator}
$\mathcal{M}(\mathrm{S1,S2,S3})$ classifies the method, not $\mathfrak{T}_\gamma$.
(S1)--(S3) are the Tauberian condition;
$c^{-\beta}(\log c)^{1+\gamma}$ is the forced consequence.
O3a (outside $\mathcal{M}$) remains open.
\end{remark}

\begin{conjecture}[H1 envelope minimality --- O1]
\label{conj:H1_envelope}
There exists no hypothesis
$\|e_j\|_{C^0} \leq C_0 j^\gamma f(j) c^{-\beta}$
with $f(j)=o(1)$ monotone that still yields
$\Phi_1(K^*(c),c) = O(c^{-\beta}(\log c)^{1+\gamma})$
with $c$-admissible constants.

\emph{Status.}
Sufficiency: Theorem~\ref{thm:universal_rate}.
Necessity requires $\limsup_{j\to\infty} j^{-\gamma}c^\beta\|e_j\|_{C^0}>0$;
operator-specific, not derivable from (S1)--(S3).
Airy ($\gamma=2/3$, $\beta=1/3$): true via Abramowitz--Stegun~\S10.4.
General $V$: open.
\end{conjecture}

\end{remark}

\section{Programme Conclusion}
\label{sec:conclusion}
%% ============================================================

\textbf{XVIII--XIX--XX.}
Universal rate, structural obstruction, H1 exponent rigidity,
and S2 knockdown: the scale $c^{-\beta}(\log c)^{1+\gamma}$
is rigid within $\mathcal{M}(\mathrm{S1,S2,S3})$;
$(\gamma,\beta)$ is uniquely identified by the rate;
and S2 is the indispensable scale admissibility axiom.

Closure inventory:
\begin{itemize}[nosep,leftmargin=1.5em]
  \item $c$-admissible chain, four sources;
  \item \textbf{S1 (H1): exponent generation};
  \item \textbf{S2 (EM): scale admissibility axiom} ---
    $c$-admissible remainder coefficients;
    poset structure of $\mathcal{S}$ depends on S2;
    removal causes quantifier shift and poset collapse
    (Lemma~\ref{lem:S2_knockdown});
  \item \textbf{S3 (subst): diagonal projection};
  \item $\BK$, $C_2(\alpha)$: $c$-free;
  \item domain vs.\ operator $c$-dependence: disjoint;
  \item anti-circularity: $F'\to$window$\to$convexity;
  \item $\gamma$-uniformity on compact sets;
    $(1+\gamma)2^\gamma\to\infty$ explicit;
  \item Decomposition stability: no cancellation;
  \item Balance fixed-point: strict separation + $\asymp$-unavoidability
    (both depend on S2 for $c$-admissibility of window constants);
  \item H1 exponent rigidity: injective exponent map;
    logarithmic inflation strictly positive;
  \item Non-tautology, constant uniformity, method/operator scope;
  \item O1 formalized: Airy case closed; general case open;
  \item DAG: S2 enters as dual-role node
    (analytic + admissibility).
\end{itemize}
