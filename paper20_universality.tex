\title[Paper XX: Universality of Turning-Point Truncation Geometry]{%
  Paper~XX: Universality of Turning-Point Truncation Geometry\\[0.3em]
  \large Spectral-Geometric Invariants and Two-Scale Logarithmic Losses}

\author{\textit{prolate-gram-coercivity programme}}
\date{May 2026}

\begin{abstract}
Papers~XVIII and~XIX established, for the Airy turning-point model,
qualitative universality (BR3) and a quantitative two-scale rate
$O(c^{-1/3}(\log c)^{5/3})$.
Paper~XIX's Langer term $\normh{A_K}{}$ splits into two internal scales:
\[
  \normh{A_K}{} \leq C_{\mathrm{lin}}\cdot c^{-1/3} + C_{\mathrm{quad}}\cdot c^{-2/3},
\]
where the linear interaction $C_{\mathrm{lin}} = O(K^{5/3})$ dominates
because the quadratic interaction $C_{\mathrm{quad}} = O(K^{7/3})$ carries
an extra $c^{-\beta}$ factor (two Langer errors vs.\ one).

This paper shows that this two-scale structure within $A_K$ is universal:
for any $\cAg \in \mathfrak{T}_\gamma$ with spectral growth exponent $\gamma > 0$,
\[
  C_{\mathrm{lin}}(M,K) = O(K^{1+\gamma}), \qquad
  C_{\mathrm{quad}}(M,K) = O(K^{1+2\gamma}),
\]
and at $K = K_{\mathrm{opt}} \sim (2\beta/\alpha)\log c$ the quadratic term
is $O(c^{-2\beta}(\log c)^{1+2\gamma}) = o(c^{-\beta}(\log c)^{1+\gamma})$,
yielding the rigorous rate
\[
  O\!\left(c^{-\beta}(\log c)^{1+\gamma}\right).
\]
The exponent $1+\gamma$ is the spectral-geometric invariant of the
\emph{linear interaction layer} of the two-scale truncation mechanism;
the quadratic layer contributes a strictly subleading $1+2\gamma$.

The proof uses exactly one new lemma (Lemma~\ref{lem:trunc_universal});
all other components carry over from XVIII/XIX
(Lemma~\ref{lem:gamma_agnostic}).
\end{abstract}

\medskip
The key insight is that the Langer term $A_K$ has two internal
interaction layers:
\begin{itemize}[nosep]
  \item \emph{Linear layer} ($u_j\otimes e_j + e_j\otimes u_j$):
    one exact eigenfunction $\times$ one Langer error;
    rate $O(K^{1+\gamma})c^{-\beta}$.
  \item \emph{Quadratic layer} ($e_j\otimes e_j$):
    two Langer errors;
    rate $O(K^{1+2\gamma})c^{-2\beta}$.
\end{itemize}
The quadratic layer is subleading by one extra factor $c^{-\beta}$,
so the asymptotic rate is governed by the linear layer alone.
This is not a bookkeeping accident but the structural content of
the two-scale architecture.

\begin{remark}[Scope]
\label{rem:scope}
$\gamma > 0$ arbitrary. The power family $V = |\xi|^\nu$ gives
$\gamma = 2\nu/(\nu+2)\nearrow 2$, so $\gamma\in(0,1]$ is insufficient.
\end{remark}

\section{Abstract Turning-Point Framework}
\label{sec:framework}
%% ============================================================

\begin{definition}[Operator class $\mathfrak{T}_\gamma$]
\label{def:opclass}
For $\gamma > 0$: $\mathfrak{T}_\gamma$ is the class of self-adjoint operators
$\cAg = -\partial_\xi^2 + V(\xi)$ on $L^2(\R_+)$,
$V\geq 0$, $V\to+\infty$, discrete spectrum $0<a_1<a_2<\cdots$,
with:
\begin{enumerate}[label=(\roman*),nosep]
  \item \emph{Spectral growth:} $a_K \sim C_\gamma K^\gamma$,
    $C_\gamma > 0$. \hfill\eqref{eq:spectralgrowth}
  \item \emph{Turning-point regularity:} $V$ is $C^2$ near its simple zero;
    Langer--Olver applicable on $[0,M]$.
  \item \emph{Coercivity propagation:}
    Hypothesis~\ref{hyp:coercprop}.
\end{enumerate}
\begin{equation}\label{eq:spectralgrowth}
  a_K \sim C_\gamma K^\gamma \quad(K\to\infty),\quad C_\gamma>0.
\end{equation}
\end{definition}

\begin{definition}[Spectral growth exponent]
\label{def:spectralgrowth}
$\gamma > 0$ in~\eqref{eq:spectralgrowth} is the
\emph{spectral growth exponent} of $\cAg$.
\end{definition}

\begin{definition}[Scaling-window and tail exponents]
\label{def:beta}
$\beta > 0$: turning-point rescaling window $c^{-\beta}$ (local geometry);
independent of $\gamma$. Airy: $\beta = 1/3$.\\
$\alpha > 0$: Slepian tail decay rate; from XVIII~\texttt{lem:slepianunif}.
\end{definition}

\begin{remark}[Independence of $\gamma$, $\beta$, $\alpha$]
\label{rem:threeparams}
$\gamma$ (global eigenvalue spacing), $\beta$ (local turning-point window),
and $\alpha$ (spectral tail decay) are logically independent.
The universal rate $O(c^{-\beta}(\log c)^{1+\gamma})$ encodes $\gamma$ and $\beta$
through separate mechanisms.
\end{remark}

\section{DAG Decomposition: Agnostic and Sensitive Nodes}
\label{sec:dag}

\begin{lemma}[$\gamma$-Agnostic nodes]
\label{lem:gamma_agnostic}
The following eight components of the XVIII/XIX DAG hold verbatim
for every $\cAg\in\mathfrak{T}_\gamma$:
\begin{enumerate}[label=(\roman*),nosep]
  \item Tensor decomposition $\SK-\cKA=\AK-\BK$.
  \item Tensor norm bound per $j$:
    $\|f\otimes g\|_{L^2}\leq M\|f\|_{C^0}\|g\|_{C^0}$.
  \item Bessel tail (XVIII~\texttt{lem:besseltail}).
  \item Tonelli/DCT (XVIII~\texttt{rem:dct}).
  \item Slepian-type tail (XVIII~\texttt{lem:slepianunif}).
  \item HS/Cauchy--Schwarz on $\BK$ (XIX~\texttt{lem:tail\_rate}).
  \item Sequential $(c,K)$-limit structure.
  \item Norm-resolvent perturbation (Kato~VI).
\end{enumerate}
\end{lemma}

\begin{proof}
By inspection: no proof in XVIII/XIX references $a_K\sim K^\gamma$
except through the four $\gamma$-sensitive nodes of
Remark~\ref{rem:sensitive}.
\end{proof}

\begin{remark}[$\gamma$-Sensitive nodes: exactly four]
\label{rem:sensitive}
Only these four depend on $\gamma$ (all in the propagation spine):
(i) spectral spacing $a_K\sim K^\gamma$;
(ii) coercive threshold $\mu(K)\sim K^\gamma$;
(iii) exclusion window $\delta_0(K)\sim K^\gamma$;
(iv) Langer $C^0$-constant $\Cprime = O(MK^\gamma)$.
\end{remark}

\section{Universal Truncation Geometry}
\label{sec:trunc}
%% ============================================================

\begin{definition}[Propagation spine]
\label{def:spine}
\[
\begin{tikzcd}[column sep=1.9em]
  a_K{\sim}K^\gamma
  \ar[r]
  & \mu(K){\sim}K^\gamma
  \ar[r,"\mathrm{L\text{-}M}"]
  & \Cprime{\sim}K^\gamma
  \ar[r,"\times K\text{ (lin)}"]
  & \Clin{\sim}K^{1+\gamma}
  \ar[r,"\times K\text{ (quad)}"]
  & \Cquad{\sim}K^{1+2\gamma}.
\end{tikzcd}
\]
The exponent increases by $\gamma$ (one Langer error) at each
multiplication step.
\end{definition}

\begin{hypothesis}[Coercivity propagation]
\label{hyp:coercprop}
There exists $\Cprime = O(MK^\gamma)$ (independent of $c$)
such that: for fixed $M,K$ and $c\geq c_0(M,K)$,
\begin{equation}\label{eq:langererror}
  \normh{e_j}{C^0([0,M])} \leq \Cprime\cdot c^{-\beta},
  \qquad j\leq K.
\end{equation}
Airy ($\gamma=2/3$, $\beta=1/3$): $\Cprime = O(MK^{2/3})$ by
XVIII~\texttt{rem:langerkdep}. $\checkmark$
\end{hypothesis}

\begin{lemma}[Universal truncation geometry]
\label{lem:trunc_universal}
Let $\cAg\in\mathfrak{T}_\gamma$ satisfy Hypothesis~\ref{hyp:coercprop},
and let $\Cinfty := \sup_{j\leq K}\normh{u_j}{C^0([0,M])} = O(1)$.
For fixed $M$, $K\geq 1$, and $c\geq c_0(M,K)$:
\begin{align}
  \normh{\AK}{L^2([0,M]^2)}
  &\leq
  \underbrace{2KM\Cprime\Cinfty}_{\Clin}\cdot c^{-\beta}
  \;+\;
  \underbrace{KM(\Cprime)^2}_{\Cquad}\cdot c^{-2\beta}, \label{eq:AKsplit}
\end{align}
where:
\begin{equation}\label{eq:Cscales}
  \Clin = O(MK^{1+\gamma}), \qquad \Cquad = O(M^2K^{1+2\gamma}).
\end{equation}
\end{lemma}

\begin{proof}
For each $j\leq K$, using $\|f\otimes g\|_{L^2([0,M]^2)}
\leq M\|f\|_{C^0}\|g\|_{C^0}$
and Hypothesis~\ref{hyp:coercprop}:
\begin{align*}
  \normh{u_j\otimes e_j + e_j\otimes u_j}{}
    &\leq 2M\Cinfty\Cprime\,c^{-\beta}, \\
  \normh{e_j\otimes e_j}{}
    &\leq M(\Cprime)^2\,c^{-2\beta}.
\end{align*}
Summing over $j=1,\ldots,K$ and using~\eqref{eq:spectralgrowth}
with $\Cprime=O(MK^\gamma)$, $\Cinfty=O(1)$
gives~\eqref{eq:AKsplit}--\eqref{eq:Cscales}.

\medskip
\textbf{Airy verification.}
$\gamma=2/3$: $\Clin=O(MK^{5/3})$ $\checkmark$
(XIX~\texttt{def:langerconstants} Rem.~\ref{rem:C1scale}),
$\Cquad=O(M^2K^{7/3})$. \qed
\end{proof}

\begin{remark}[Why $1+\gamma$ governs the rate, not $1+2\gamma$]
\label{rem:whylinear}
At the optimal coupling $\Kopt \sim (2\beta/\alpha)\log c$:
\[
  \Cquad\cdot c^{-2\beta}\Big|_{K=\Kopt}
  = O\!\left((\log c)^{1+2\gamma}\,c^{-2\beta}\right)
  = o\!\left((\log c)^{1+\gamma}\,c^{-\beta}\right).
\]
The quadratic layer is suppressed by the extra $c^{-\beta}$ factor
arising from the second Langer error.
The asymptotic rate is therefore governed exclusively by
the linear layer, with invariant $1+\gamma$.
This is a structural property of the two-scale architecture,
not a bookkeeping choice.
\end{remark}

\section{Universal Rate Theorem}
\label{sec:main}
%% ============================================================

\begin{theorem}[Universal two-scale rate]
\label{thm:universal_rate}
Let $\cAg\in\mathfrak{T}_\gamma$ satisfy Hypothesis~\ref{hyp:coercprop}.
For fixed $K\geq K_0$ and $c\geq c_0(M,K)$:
\begin{equation}\label{eq:twoscale_universal}
  \normh{\cK-\cKA}{L^2([0,M]^2)}
  \leq
  \Clin\cdot c^{-\beta}
  + \Cquad\cdot c^{-2\beta}
  + \Ctwo^{1/2}\cdot e^{-\alpha K/2}.
\end{equation}
\end{theorem}

\begin{proof}
Triangle inequality on $\AK - \BK$ + Lemma~\ref{lem:trunc_universal}
+ XIX~\texttt{lem:tail\_rate} (Lemma~\ref{lem:gamma_agnostic}(vi)).
\end{proof}

\begin{corollary}[Universal logarithmic rate]
\label{cor:universal_lograte}
With $\Kopt(c) = \lfloor(2\beta/\alpha)\log c\rfloor$,
for $c$ large:
\begin{equation}\label{eq:lograte_universal}
  \normh{\cK-\cKA}{}\Big|_{K=\Kopt}
  = O\!\left(c^{-\beta}(\log c)^{1+\gamma}\right).
\end{equation}
\end{corollary}

\begin{proof}
At $K=\Kopt\sim(2\beta/\alpha)\log c$:
\begin{itemize}[nosep]
  \item \emph{Linear term:}
    $\Clin c^{-\beta} = O((\log c)^{1+\gamma}c^{-\beta})$.
  \item \emph{Quadratic term:}
    $\Cquad c^{-2\beta} = O((\log c)^{1+2\gamma}c^{-2\beta})
    = o(c^{-\beta}(\log c)^{1+\gamma})$.
  \item \emph{Tail:}
    $e^{-\alpha\Kopt/2} = c^{-\beta}$;
    then $\Ctwo^{1/2}c^{-\beta} = O(c^{-\beta})
    = o(c^{-\beta}(\log c)^{1+\gamma})$.
\end{itemize}
Linear term dominates; \eqref{eq:lograte_universal} follows.
\end{proof}

\begin{remark}[Spectral-geometric invariants]
\label{rem:invariants}
The two spectral-geometric invariants of the two-scale mechanism are:
\begin{itemize}[nosep]
  \item $1+\gamma$: exponent of the linear interaction layer
    ($u_j\otimes e_j + e_j\otimes u_j$); governs the asymptotic rate.
  \item $1+2\gamma$: exponent of the quadratic interaction layer
    ($e_j\otimes e_j$); subleading by $c^{-\beta}$.
\end{itemize}
Both are determined by $\gamma$ alone (and the rank-$K$ summation);
neither is a proof artifact.
\end{remark}

\begin{remark}[Interpretation]
\label{rem:geometry}
$\nu\to 0$: $1+\gamma\to 1$, $1+2\gamma\to 1$ (flat potential, both layers trivial).\\
$\nu\to\infty$: $1+\gamma\to 3$, $1+2\gamma\to 5$ (hard wall, steep penalties).\\
The two-layer gap $1+2\gamma - (1+\gamma) = \gamma$
is itself a spectral quantity: it measures how much
``extra crowding'' the quadratic interaction incurs relative to the linear.
\end{remark}
