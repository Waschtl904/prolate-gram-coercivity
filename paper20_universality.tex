\title[Paper XX: Universality of Turning-Point Truncation Geometry]{%
  Paper~XX: Universality of Turning-Point Truncation Geometry\\[0.3em]
  \large Spectral-Geometric Invariants and Two-Scale Logarithmic Losses}

\author{\textit{prolate-gram-coercivity programme}}
\date{May 2026}

\begin{abstract}
Papers~XVIII and~XIX established, for the Airy turning-point model,
qualitative universality (BR3) and a quantitative two-scale rate
$O(c^{-1/3}(\log c)^{5/3})$.
Paper~XIX's Langer term $\normh{A_K}{}$ splits into two
\emph{interaction layers}:
\begin{align*}
  \AKlin  &= \textstyle\sum_{j=1}^K
    (u_j\otimes e_j + e_j\otimes u_j)
    &&\text{(linear: one exact mode, one Langer error)},\\
  \AKquad &= \textstyle\sum_{j=1}^K e_j\otimes e_j
    &&\text{(quadratic: two Langer errors)},
\end{align*}
with bounds
$\normh{\AKlin}{}\leq C_{\mathrm{lin}}c^{-\beta}$
and
$\normh{\AKquad}{}\leq C_{\mathrm{quad}}c^{-2\beta}$.

This paper shows that this two-layer structure is universal:
for every $\cAg\in\mathfrak{T}_\gamma$ with spectral growth exponent $\gamma>0$,
\[
  C_{\mathrm{lin}} = O(K^{1+\gamma}), \qquad
  C_{\mathrm{quad}} = O(K^{1+2\gamma}).
\]
At $K=K_{\mathrm{opt}}\sim(2\beta/\alpha)\log c$,
the quadratic layer is
$O(c^{-2\beta}(\log c)^{1+2\gamma}) = o(c^{-\beta}(\log c)^{1+\gamma})$,
so the rate is governed by the linear layer alone:
\[
  \normh{\cK-\cKA}{} = O\!\left(c^{-\beta}(\log c)^{1+\gamma}\right).
\]

The central mechanism is expressed in the
\emph{Layer-Survival Selection Principle}
(Remark~\ref{rem:selection_principle}):
\begin{quote}\itshape
  The leading logarithmic exponent is selected not by maximal
  $K$-growth, but by the highest interaction layer
  surviving the $c$-scaling hierarchy.
\end{quote}
$1+2\gamma > 1+\gamma$ algebraically, yet $1+\gamma$ governs the rate
because the quadratic layer pays one additional factor $c^{-\beta}$
for its second Langer error, placing it below the $c$-horizon at
$K=K_{\mathrm{opt}}$.
\end{abstract}

The Langer term $\AK = \AKlin + \AKquad$ has three

\medskip
\begin{center}
\renewcommand{\arraystretch}{1.4}
\begin{tabular}{l l l l}
\toprule
\textbf{Layer} & \textbf{Structure}
  & \textbf{$K$-invariant} & \textbf{$c$-scale} \\
\midrule
Linear (leading)
  & $u_j\otimes e_j + e_j\otimes u_j$
  & $1+\gamma$ & $c^{-\beta}$ \\
Quadratic (subleading)
  & $e_j\otimes e_j$
  & $1+2\gamma$ & $c^{-2\beta}$ \\

  & $B_K = \sum_{j>K}u_j\otimes u_j$
  & $e^{-\alpha K/2}$ & uniform in $c$ \\
\bottomrule
\end{tabular}
\end{center}

\medskip
Each layer is governed by a separate geometric mechanism;
the asymptotic rate is selected by the \emph{survival} criterion
of Remark~\ref{rem:selection_principle}.

\begin{remark}[Layer-Survival Selection Principle]
\label{rem:selection_principle}
Let $\lambda_1 > \lambda_2 > \cdots$ be the
$K$-growth exponents of the interaction layers of $\AK$,
with associated $c$-scales $c^{-\sigma_1}, c^{-\sigma_2},\ldots$
($\sigma_1 < \sigma_2 < \cdots$).
At the optimal coupling $K = K_{\mathrm{opt}}(c)$,
an interaction layer $\ell$ survives the $c$-hierarchy if and only if
\[
  (\log c)^{\lambda_\ell}\, c^{-\sigma_\ell}
  \not= o\!\left((\log c)^{\lambda_1}\,c^{-\sigma_1}\right).
\]
The leading logarithmic exponent of the total rate
is determined by the highest-$K$-exponent layer that survives
(i.e.\ the layer with the smallest $\sigma$).

\medskip
\textbf{In our setting:}
\begin{itemize}[nosep]
  \item Linear layer: $\lambda_1 = 1+\gamma$,\; $\sigma_1 = \beta$.
  \item Quadratic layer: $\lambda_2 = 1+2\gamma$,\; $\sigma_2 = 2\beta$.
\end{itemize}
Since $\sigma_2 > \sigma_1$:
$
(\log c)^{1+2\gamma}c^{-2\beta} = c^{-\beta}\cdot(\log c)^{1+2\gamma}c^{-\beta}
= o\bigl((\log c)^{1+\gamma}c^{-\beta}\bigr).
$
The quadratic layer does not survive the $c$-hierarchy;
the rate exponent is $1+\gamma$, determined exclusively by
the linear layer.

\medskip
\textbf{Conceptual content.}
Algebraic $K$-size ($1+2\gamma > 1+\gamma$) is insufficient to
determine the rate.
What matters is the combined $(K, c)$-scaling after
$K$-optimization.
The layer hierarchy is:
\[
  \text{linear geometry} \;\succ\; \text{quadratic geometry}
  \quad\text{(in the asymptotic sense)},
\]
not because $1+\gamma < 1+2\gamma$,
but because the quadratic layer pays an additional $c^{-\beta}$
for its second Langer error, dropping it below the $c$-horizon.
\end{remark}

\begin{remark}[Scope]
\label{rem:scope}
$\gamma > 0$ arbitrary. The power family $V = |\xi|^\nu$ gives
$\gamma = 2\nu/(\nu+2)$, ranging over $(0,2)$.
\end{remark}

\section{Abstract Turning-Point Framework}

\begin{definition}[Operator class $\mathfrak{T}_\gamma$]
\label{def:opclass}
For $\gamma > 0$: $\mathfrak{T}_\gamma$ is the class of self-adjoint operators
$\cAg = -\partial_\xi^2 + V(\xi)$ on $L^2(\R_+)$,
$V\geq 0$, $V\to+\infty$, discrete spectrum $0<a_1<a_2<\cdots$,
with:
\begin{enumerate}[label=(\roman*),nosep]
  \item \emph{Spectral growth:} $a_K \sim C_\gamma K^\gamma$.
  \item \emph{Turning-point regularity:} $V$ is $C^2$ near its simple zero;
    Langer--Olver applicable on $[0,M]$.
  \item \emph{Coercivity propagation:}
    Hypothesis~\ref{hyp:coercprop}.
\end{enumerate}
\end{definition}

\begin{definition}[Parameters]
\label{def:params}
\mbox{}
\begin{itemize}[nosep]
  \item $\gamma > 0$: spectral growth exponent ($a_K\sim K^\gamma$);
    governs $K$-size of interaction layers.
  \item $\beta > 0$: turning-point window ($c^{-\beta}$ per Langer error);
    governs $c$-survival of layers. Independent of $\gamma$.
  \item $\alpha > 0$: Slepian tail decay; independent of $\gamma$, $\beta$.
\end{itemize}
Airy: $\gamma=2/3$, $\beta=1/3$, $\alpha>0$ from XVIII.
\end{definition}

\section{Universal Truncation Geometry}
\label{sec:trunc}
%% ============================================================

\begin{definition}[Propagation spine]
\label{def:spine}
The $\gamma$-sensitive chain is:
\[
\begin{tikzcd}[column sep=2.1em]
  a_K{\sim}K^\gamma
  \ar[r]
  & \mu(K){\sim}K^\gamma
  \ar[r,"\mathrm{L\text{-}M}"]
  & \hat C{\sim}K^\gamma
  \ar[r,"\times K \cdot C_\infty \;\text{(lin)}"]
  & C_{\rm lin}{\sim}K^{1+\gamma}
\end{tikzcd}
\]
\[
\begin{tikzcd}[column sep=2.1em]
  \hat C{\sim}K^\gamma
  \ar[r,"\hat C^2\,{\times}\,K \;\text{(quad)}"]
  & C_{\rm quad}{\sim}K^{1+2\gamma}.
\end{tikzcd}
\]
The $K$-exponent increases by $\gamma$ per Langer error factor:
linear interaction carries one error ($+\gamma$),
quadratic carries two ($+2\gamma$).
Each additional error also adds one factor $c^{-\beta}$,
which is the $c$-counterpart of the $K$-growth.
\end{definition}

\begin{hypothesis}[Coercivity propagation]
\label{hyp:coercprop}
$\exists\, \hat{C}=O(MK^\gamma)$ (independent of $c$) such that:
for fixed $M,K$ and $c\geq c_0(M,K)$,
\begin{equation}\label{eq:langererror}
  \normh{e_j}{C^0([0,M])} \leq \hat{C}\cdot c^{-\beta},
  \qquad j\leq K.
\end{equation}
Airy verification: $\gamma=2/3$, $\beta=1/3$,
$\hat{C} = O(MK^{2/3})$ by XVIII~\texttt{rem:langerkdep}. $\checkmark$
\end{hypothesis}

\begin{lemma}[Universal truncation geometry]
\label{lem:trunc_universal}
Let $\cAg\in\mathfrak{T}_\gamma$ satisfy Hypothesis~\ref{hyp:coercprop},
$\Cinfty := \sup_{j\leq K}\normh{u_j}{C^0([0,M])} = O(1)$.
For fixed $M$, $K\geq 1$, $c\geq c_0(M,K)$:
\begin{align}
  \normh{\AKlin}{L^2([0,M]^2)}
    &\leq \underbrace{2KM\hat{C}\Cinfty}_{\Clin}
      \cdot c^{-\beta}, \label{eq:AKlin} \\
  \normh{\AKquad}{L^2([0,M]^2)}
    &\leq \underbrace{KM\hat{C}^2}_{\Cquad}
      \cdot c^{-2\beta}, \label{eq:AKquad}
\end{align}
where
\begin{equation}\label{eq:Cscales}
  \Clin = O(MK^{1+\gamma}), \qquad
  \Cquad = O(M^2K^{1+2\gamma}).
\end{equation}
\end{lemma}

\begin{proof}
For each $j\leq K$, by Hypothesis~\ref{hyp:coercprop}
and $\|f\otimes g\|_{L^2}\leq M\|f\|_{C^0}\|g\|_{C^0}$:
\[
  \normh{u_j\otimes e_j + e_j\otimes u_j}{}
    \leq 2M\Cinfty\hat{C}\,c^{-\beta},
  \qquad
  \normh{e_j\otimes e_j}{}
    \leq M\hat{C}^2\,c^{-2\beta}.
\]
Note that the quadratic term carries $c^{-2\beta}$ exactly:
no absorption $c^{-2\beta}\leq c^{-\beta}$ is performed here.
Summing over $j=1,\ldots,K$ and using $\hat{C}=O(MK^\gamma)$,
$\Cinfty=O(1)$ gives~\eqref{eq:AKlin}--\eqref{eq:Cscales}.

\medskip
\textbf{Airy check} ($\gamma=2/3$, $\beta=1/3$):
$\Clin = O(MK^{5/3})$ (XIX~\texttt{def:langerconstants})~$\checkmark$,
$\Cquad = O(M^2K^{7/3})$~$\checkmark$.
\end{proof}

\begin{remark}[Layer-survival at $K=\Kopt$: the selection mechanism]
\label{rem:whylinear}
By Remark~\ref{rem:selection_principle} with
$K=\Kopt\sim(2\beta/\alpha)\log c$:
\[
  \Cquad\cdot c^{-2\beta}\Big|_{K=\Kopt}
  = O\!\left((\log c)^{1+2\gamma}\cdot c^{-2\beta}\right)
  = c^{-\beta}\cdot O\!\left((\log c)^{1+2\gamma}\cdot c^{-\beta}\right)
  = o\!\left((\log c)^{1+\gamma}\cdot c^{-\beta}\right),
\]
since $c^{-\beta}\gg(\log c)^m$ for every fixed $m$ as $c\to\infty$.
The quadratic layer does not survive the $c$-hierarchy
at $K=\Kopt$; it falls below the $c$-horizon.
The rate exponent $1+\gamma$ is therefore selected by the linear layer,
which is the highest $K$-exponent layer that survives.
\end{remark}

\section{Universal Rate Theorem}
\label{sec:main}
%% ============================================================

\begin{theorem}[Universal two-scale rate]
\label{thm:universal_rate}
Let $\cAg\in\mathfrak{T}_\gamma$ satisfy Hypothesis~\ref{hyp:coercprop}.
For fixed $K\geq K_0$ and $c\geq c_0(M,K)$:
\begin{equation}\label{eq:twoscale_universal}
  \normh{\cK-\cKA}{}
  \leq
  \Clin\cdot c^{-\beta}
  + \Cquad\cdot c^{-2\beta}
  + \Ctwo^{1/2}\cdot e^{-\alpha K/2}.
\end{equation}
\end{theorem}

\begin{proof}
Triangle inequality on $\AK - \BK$:
$\normh{\AK}{}\leq\normh{\AKlin}{}+\normh{\AKquad}{}$
+ Lemma~\ref{lem:trunc_universal} + tail bound
(Lemma~\ref{lem:gamma_agnostic}(vi)).
\end{proof}

\begin{corollary}[Universal logarithmic rate]
\label{cor:universal_lograte}
With $\Kopt(c) = \lfloor(2\beta/\alpha)\log c\rfloor$,
for $c$ large enough that $\Kopt\geq K_0$:
\begin{equation}\label{eq:lograte_universal}
  \normh{\cK-\cKA}{}\Big|_{K=\Kopt}
  = O\!\left(c^{-\beta}(\log c)^{1+\gamma}\right).
\end{equation}
The quadratic layer and the spectral tail are both
$o(c^{-\beta}(\log c)^{1+\gamma})$.
\end{corollary}

\begin{proof}
At $K=\Kopt\sim(2\beta/\alpha)\log c$:
\begin{enumerate}[label=(\roman*),nosep]
  \item \emph{Linear:}
    $\Clin c^{-\beta} = O((\log c)^{1+\gamma}c^{-\beta})$.
  \item \emph{Quadratic:}
    $\Cquad c^{-2\beta} = O((\log c)^{1+2\gamma}c^{-2\beta})
    = o(c^{-\beta}(\log c)^{1+\gamma})$
    (Remark~\ref{rem:whylinear}).
  \item \emph{Tail:}
    $\Ctwo^{1/2}e^{-\alpha\Kopt/2}\leq\Ctwo^{1/2}c^{-\beta}
    = O(c^{-\beta})
    = o(c^{-\beta}(\log c)^{1+\gamma})$.
\end{enumerate}
Sum dominated by term (i). \qed
\end{proof}

\begin{remark}[Two spectral-geometric invariants]
\label{rem:invariants}
The two spectral-geometric invariants of the two-scale
truncation mechanism are:
\begin{itemize}[nosep]
  \item $1+\gamma$: $K$-exponent of the linear layer;
    \emph{determines the rate} (survives $c$-hierarchy).
  \item $1+2\gamma$: $K$-exponent of the quadratic layer;
    \emph{structurally present but subleading}
    (suppressed by additional $c^{-\beta}$).
\end{itemize}
Both are real geometric quantities;
none is a proof artifact.
The gap $\gamma = (1+2\gamma)-(1+\gamma)$ is itself spectral:
it measures the extra crowding cost of the quadratic interaction.
\end{remark}

\begin{remark}[Geometry of the model family]
\label{rem:geometry}
$\nu\to 0$: $1+\gamma\to 1$, $1+2\gamma\to 1$
(flat potential; both layers cheap, gap $\gamma\to 0$).\\
$\nu\to\infty$: $1+\gamma\to 3$, $1+2\gamma\to 5$
(hard wall; both layers expensive, gap $\gamma\to 2$).\\
The inter-layer gap $\gamma$ interpolates monotonically with
potential steepness.
\end{remark}
