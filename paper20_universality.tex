\begin{abstract}
For operators $\cAg\in\mathfrak{T}_\gamma$
(spectrally regularized WKB class, Definition~\ref{def:opclass}),
the Langer term $\AK = \AKlin + \AKquad$ satisfies,
under the pointwise Langer scaling law
$\|e_j\|_{C^0}\leq C_0 j^\gamma c^{-\beta}$
(Hypothesis~\ref{hyp:langer_pointwise}),
\[
  \|\AKlin\| \leq C_{\rm lin}\,K^{1+\gamma}c^{-\beta},
  \qquad
  \|\AKquad\| \leq C_{\rm quad}\,K^{1+2\gamma}c^{-2\beta}
\]
for all $K\geq K_0$, $c\geq c_0$, with constants $C_{\rm lin}, C_{\rm quad}$
independent of $K$ and $c$
(Lemma~\ref{lem:trunc_universal}).
The optimal truncation $\Kopt$ is the unique minimizer of the
fixed-$c$ functional $F(K;c) := L\,K^{1+\gamma}c^{-\beta} + T\,e^{-\alpha K/2}$
over the window $K\in[c_1\log c,c_2\log c]$
(Lemma~\ref{lem:kopt_variational}), satisfying
$K^*(c) = \frac{2\beta}{\alpha}\log c + \frac{2\gamma}{\alpha}\log\log c + O(1)$.
The transition from fixed-parameter bounds to the diagonal
$K = K^*(c)$ is carried out by a single explicit lemma
(Lemma~\ref{lem:diagonal_sub}), which is the unique point
where coupled scaling is introduced;
logarithmic instability under exponentiation is controlled
by an explicit slowly-varying substitution estimate.
All results are conditional on H1--H2.
\end{abstract}

\section{Asymptotic Scale and Regime-Dependent Order}
\label{sec:asymp}

\begin{definition}[Asymptotic scale representatives]
\label{def:scale}
Let $\mathcal{S} = \{c^{-s}(\log c)^t : s>0,\, t\in\R\}$,
used as a representative system of asymptotic equivalence
classes: $f\sim g \Leftrightarrow f/g\to 1$ as $c\to\infty$.
The total order $\succ$ on equivalence classes:
$c^{-s_1}(\log c)^{t_1}\succ c^{-s_2}(\log c)^{t_2}$
iff $s_1<s_2$, or ($s_1=s_2$ and $t_1>t_2$).
When we write $f = O(g)$ or $f\prec g$
for $f,g\in\mathcal{S}$, this is a statement about the
asymptotic ratio $f/g$ as $c\to\infty$;
no algebraic structure on $\mathcal{S}$ is assumed.
\end{definition}

\begin{remark}[Dominance is regime-dependent, not algebraic]
\label{rem:scale_regime}
All asymptotic comparisons ($\prec$, $O(\cdot)$, $\sim$)
in this paper are evaluated in the diagonal regime
$K = K^*(c) \sim \frac{2\beta}{\alpha}\log c$.
Dominance of one term over another is verified explicitly
for each pair appearing in the proofs
(Propositions~\ref{prop:asymp_order},
Lemma~\ref{lem:diagonal_sub}(ii)--(iii));
it is not assumed as a global property of $\mathcal{S}$.
\end{remark}

\begin{proposition}[Multiplication on $\mathcal{S}$]
\label{prop:scale_algebra}
For $f = c^{-s_1}(\log c)^{t_1}$,
$g = c^{-s_2}(\log c)^{t_2}\in\mathcal{S}$:
\[
  f\cdot g = c^{-(s_1+s_2)}(\log c)^{t_1+t_2}\in\mathcal{S}.
\]
Asymptotic addition: if $f/g\to\infty$ as $c\to\infty$,
then $(f+g)/f\to 1$, so $f+g = O(f)$ with constant $2$.
Dominance ($f/g\to\infty$) is verified for each specific
pair used in this paper; see Remark~\ref{rem:scale_regime}.
\end{proposition}

\begin{proposition}[Layer dominance in diagonal regime]
\label{prop:asymp_order}
With $K = K^*(c)\sim\frac{2\beta}{\alpha}\log c$, for $n\geq 2$:
\[
  \frac{c^{-n\beta}(\log c)^{1+n\gamma}}{c^{-\beta}(\log c)^{1+\gamma}}
  = c^{-(n-1)\beta}(\log c)^{(n-1)\gamma}\to 0,
\]
so $c^{-n\beta}(\log c)^{1+n\gamma} \prec c^{-\beta}(\log c)^{1+\gamma}$.
\end{proposition}

\section{Truncation Geometry}
\label{sec:trunc}
%% ============================================================

\begin{lemma}[Universal truncation geometry, fixed-parameter]
\label{lem:trunc_universal}
Under H1--H2, there exist constants
\[
  C_{\rm lin} = \frac{2MC_\infty C_0}{1+\gamma},
  \qquad
  C_{\rm quad} = \frac{MC_0^2}{1+2\gamma},
\]
both independent of $K$ and $c$, such that
for all $K\geq 1$, $c\geq c_0$:
\begin{align}
  \normh{\AKlin}{}
    &\leq C_{\rm lin}\,K^{1+\gamma}c^{-\beta}
    + C_{\sigma,1}\,K^\gamma c^{-\beta},\label{eq:AKlin}\\
  \normh{\AKquad}{}
    &\leq C_{\rm quad}\,K^{1+2\gamma}c^{-2\beta}
    + C_{\sigma,2}\,K^{2\gamma}c^{-2\beta},\label{eq:AKquad}
\end{align}
where $C_{\sigma,1}, C_{\sigma,2}$ are also independent of $K$ and $c$.
\end{lemma}

\begin{proof}
Apply the Euler--Maclaurin formula to $\sum_{j=1}^K j^\sigma$
with $\sigma = \gamma$ (for the linear term) and $\sigma = 2\gamma$
(for the quadratic term):
\begin{equation}\label{eq:EM}
  \sum_{j=1}^K j^\sigma
  = \frac{K^{1+\sigma}}{1+\sigma}
  + \frac{1}{2}K^\sigma
  + R_\sigma(K),
\end{equation}
where the remainder satisfies
\begin{equation}\label{eq:EM_remainder}
  |R_\sigma(K)| \leq C_\sigma\,K^{\sigma-1},
  \qquad
  C_\sigma = \frac{|B_2|}{12}\,\sigma(\sigma-1),
\end{equation}
with $B_2 = \tfrac{1}{6}$ the second Bernoulli number.
The constant $C_\sigma$ depends only on $\sigma$, not on $K$
or on any external parameter such as $c$.
This makes the leading constant and all subsequent
error terms in~\eqref{eq:AKlin}--\eqref{eq:AKquad}
uniform in both $K$ and $c$.
The constants $C_{\rm lin}, C_{\rm quad}$ are read off from
the leading term of~\eqref{eq:EM} after multiplying by
$C_0 C_\infty c^{-\beta}$ and $C_0^2 c^{-2\beta}$ respectively;
$C_{\sigma,1} = \frac{1}{2}C_0 C_\infty$ and
$C_{\sigma,2} = \frac{1}{2}C_0^2$ come from the middle term.
\end{proof}

\section{Variational Optimal Truncation}
\label{sec:kopt}
%% ============================================================

\begin{lemma}[$K^*(c)\to\infty$]
\label{lem:kstar_to_infty}
For each fixed $c$,
$F(\,\cdot\,;c)=LK^{1+\gamma}c^{-\beta}+Te^{-\alpha K/2}$
has a unique minimizer $K^*(c)$.
As $c\to\infty$: $K^*(c)\to\infty$.
\end{lemma}

\begin{proof}
For fixed $K_0$:
$F'(K_0;c)=L(1+\gamma)K_0^\gamma c^{-\beta}-\tfrac{\alpha}{2}Te^{-\alpha K_0/2}<0$
for large $c$; hence $K^*(c)>K_0$ for large $c$.
\end{proof}

\begin{lemma}[Sandwich on $K^*(c)$]
\label{lem:loglog}
With $c_1=2\beta/\alpha$, $c_2=2(\beta+1)/\alpha$,
for all sufficiently large $c$:
\begin{equation}\label{eq:sandwich}
  c_1\log c\leq K^*(c)\leq c_2\log c,
  \qquad \log K^*(c)=\log\log c+O(1).
\end{equation}
\end{lemma}

\begin{proof}
\emph{Upper.}
Logarithmic FOC:
$\tfrac{\alpha}{2}K^*=\beta\log c+\gamma\log K^*+C_0$.
$g(c):=\gamma(\log K^*(c))/K^*(c)\to 0$ monotonically
(since $K^*\to\infty$ and $(\log t)/t\searrow 0$);
hence $\exists c_0$: $g(c)\leq\alpha/4$,
giving $K^*\leq c_2\log c$.
\emph{Lower.}
$F'(c_1\log c;c)=[L(1+\gamma)(2\beta/\alpha)^\gamma(\log c)^\gamma
-\tfrac{\alpha}{2}T]c^{-\beta}>0$ for large $c$;
$F'(0^+)<0$ gives $K^*\geq c_1\log c$.
\end{proof}

\begin{lemma}[Variational $K^*$]
\label{lem:kopt_variational}

\textup{(i)} \textbf{Fixed-$c$ existence and uniqueness
(on the restricted domain).}
By Lemma~\ref{lem:loglog}, $K^*(c)\in[c_1\log c,c_2\log c]$
for all large $c$.
For each fixed $c$, the functional $F(\,\cdot\,;c)$
is strictly convex on this window:
\[
  F''(K;c)
  = L\gamma(1+\gamma)K^{\gamma-1}c^{-\beta}
  + \tfrac{\alpha^2}{4}Te^{-\alpha K/2}
  > 0
  \quad\forall K\in[c_1\log c,\,c_2\log c].
\]
Note: $F''(K;c)\to 0$ as $c\to\infty$ for fixed $K$;
no uniform-in-$c$ convexity is claimed or needed.
Strict convexity at each fixed $c$, on the window
determined by the sandwich lemma,
is sufficient for uniqueness of the minimizer $K^*(c)$.
The unique minimizer satisfies
\begin{equation}\label{eq:FOC}
  L(1+\gamma){K^*}^\gamma c^{-\beta}=\tfrac{\alpha}{2}T e^{-\alpha K^*/2}.
\end{equation}

\textup{(ii)} \textbf{Iterated asymptotics.}
Logarithm of~\eqref{eq:FOC}:
\begin{equation}\label{eq:log_FOC}
  \tfrac{\alpha}{2}K^*=\beta\log c+\gamma\log K^*+C_1.
\end{equation}
Derivation without circularity (DAG: Remark~\ref{rem:dag_nocycle}):
\begin{itemize}[nosep]
  \item Step~1: $K^*\to\infty$
    (Lemma~\ref{lem:kstar_to_infty}, independent of any expansion)
    $\Rightarrow\gamma\log K^*/K^*\to 0
    \Rightarrow K_0=\frac{2\beta}{\alpha}\log c$.
  \item Step~2: sandwich~\eqref{eq:sandwich}
    $\Rightarrow\log K^*=\log\log c+O(1)$.
  \item Step~3: substitute into~\eqref{eq:log_FOC}:
\end{itemize}
\begin{equation}\label{eq:kopt_iterated}
  K^*(c)=\frac{2\beta}{\alpha}\log c+\frac{2\gamma}{\alpha}\log\log c+O(1).
\end{equation}

\textup{(ii')} \textbf{Local regularity of the iteration
(non-uniform, pointwise in $c$).}
Equation~\eqref{eq:log_FOC} is equivalent to $K^*=\Phi_c(K^*)$ with
$\Phi_c(K)=\frac{2\beta}{\alpha}\log c+\frac{2\gamma}{\alpha}\log K+C_1$.
$\Phi_c$ is not a uniform contraction in $c$:
$|\Phi_c'(K^*(c))|=\frac{2\gamma}{\alpha K^*(c)}\to 0$,
so contraction strength vanishes asymptotically.
For each fixed $c\geq c_0$, $\Phi_c$ has Lipschitz constant
$\lambda(c)<\tfrac{1}{2}$ on $[c_1\log c, c_2\log c]$.
The unique fixed point of $\Phi_c$ coincides with $K^*(c)$
by uniqueness from~\textup{(i)}, not by contraction strength.
Purpose: certify that iteration~\textup{(ii)} operates within
$[c_1\log c,c_2\log c]$; provide convergence rate $\lambda(c)$;
confirm regime self-consistency.
No new asymptotic information is generated.

\textup{(iii)} $F(K^*;c)=O(c^{-\beta}(\log c)^{1+\gamma})$.
\end{lemma}

\begin{lemma}[Stability of asymptotics]
\label{lem:stability_kopt}
An $O(1)$ perturbation of the RHS of~\eqref{eq:log_FOC}
shifts $K^*$ by $O(1)$.
\end{lemma}

\begin{proof}
MVT on the difference of two solutions with $\min K_i\geq c_1\log c$;
for large $c$: $|K_1-K_2|\leq 4D/\alpha=O(1)$.
\end{proof}

\section{Diagonal Substitution}
\label{sec:diagonal}
%% ============================================================

\begin{lemma}[Diagonal substitution principle]
\label{lem:diagonal_sub}
This is the unique lemma where $K$ and $c$ are coupled.

\emph{Hypothesis.}
There exist constants $C_{\rm lin}, C_{\rm quad}, C_{\rm tail}$
independent of $K$ and $c$ such that for all $K\geq K_0$,
$c\geq c_0$:
\begin{equation}\label{eq:fixed_param_bound}
  \normh{\cK-\cKA}{}
  \leq C_{\rm lin}\,K^{1+\gamma}c^{-\beta}
  + C_{\rm quad}\,K^{1+2\gamma}c^{-2\beta}
  + C_{\rm tail}\,e^{-\alpha K/2}.
\end{equation}
These constants are inherited from Lemma~\ref{lem:trunc_universal}
($C_{\rm lin}, C_{\rm quad}$, with explicit EM remainder~\eqref{eq:EM_remainder})
and Lemma~\ref{lem:gamma_agnostic}(v)
($C_{\rm tail} = \Ctwo^{1/2}$).

\emph{Slowly-varying substitution stability.}
Let
\[
  h(c) := \frac{K^*(c)}{\frac{2\beta}{\alpha}\log c}.
\]
By~\eqref{eq:kopt_iterated}, as $c\to\infty$:
\begin{equation}\label{eq:h_rate}
  h(c) \to 1
  \quad\text{and}\quad
  |h(c) - 1|
  = O\!\left(\frac{\log\log c}{\log c}\right).
\end{equation}
For every fixed $\gamma > 0$, by the mean-value inequality
$(1+u)^{1+\gamma} = 1 + O(|u|)$ for $|u|\leq\tfrac{1}{2}$:
\begin{equation}\label{eq:hgamma}
  h(c)^{1+\gamma}
  = 1 + O\!\left(\frac{\log\log c}{\log c}\right),
\end{equation}
and consequently
\begin{equation}\label{eq:Kstar_power}
  (K^*(c))^{1+\gamma}
  = \left(\frac{2\beta}{\alpha}\log c\right)^{1+\gamma}
  \!\left(1 + O\!\left(\frac{\log\log c}{\log c}\right)\right).
\end{equation}
This controls the logarithmic instability under exponentiation:
the relative error in substituting $K^*(c)$ for
$\frac{2\beta}{\alpha}\log c$ in a power $(\cdot)^{1+\gamma}$
is $O(\log\log c/\log c)$, which is $o(1)$ and does not
degrade the asymptotic order of any term in~\eqref{eq:fixed_param_bound}.

\emph{Conclusion.}
Let $K=K^*(c)$ (Lemma~\ref{lem:kopt_variational}(ii)).
Then, as $c\to\infty$:
\begin{itemize}[nosep]
  \item[\textup{(i)}] By~\eqref{eq:Kstar_power}:
    $K^*(c)^{1+\gamma}c^{-\beta}
    = \bigl(\frac{2\beta}{\alpha}\bigr)^{1+\gamma}
    c^{-\beta}(\log c)^{1+\gamma}\bigl(1+O(\tfrac{\log\log c}{\log c})\bigr)
    = O(c^{-\beta}(\log c)^{1+\gamma})$.
  \item[\textup{(ii)}] $K^*(c)^{1+2\gamma}c^{-2\beta}
    \prec c^{-\beta}(\log c)^{1+\gamma}$:
    ratio $= K^*(c)^\gamma c^{-\beta}
    = O((\log c)^\gamma c^{-\beta})\to 0$
    (Proposition~\ref{prop:asymp_order}).
  \item[\textup{(iii)}] $C_{\rm tail}\,e^{-\alpha K^*(c)/2}
    = C'\,c^{-\beta}(\log c)^{-\gamma}
    \prec c^{-\beta}(\log c)^{1+\gamma}$:
    ratio $= O((\log c)^{-(1+2\gamma)})\to 0$.
  \item[\textup{(iv)}] Hence $\normh{\cK-\cKA}{}
    = O(c^{-\beta}(\log c)^{1+\gamma})$,
    with the linear term~\textup{(i)} dominant.
\end{itemize}
\end{lemma}

\begin{proof}
Items (i)--(iii): ratios computed explicitly above;
dominance verified for these specific pairs
(Remark~\ref{rem:scale_regime}).
(iv): since (i)/(ii) and (i)/(iii) both $\to\infty$,
asymptotic addition
(Proposition~\ref{prop:scale_algebra})
gives sum $= O(\text{term (i)})$.
\end{proof}

\begin{remark}[Discrete projection]
\label{rem:kopt_discrete}
$K_{\rm int}=\lfloor K^*\rceil$:
$F(K_{\rm int})-F(K^*)=O(c^{-\beta}(\log c)^{\gamma-1})
\prec c^{-\beta}(\log c)^{1+\gamma}$
(ratio $= O((\log c)^{-2})\to 0$).
\end{remark}

\section{Layer-Survival Selection Principle}
\label{sec:layersurvival}
%% ============================================================

\begin{theorem}[Layer-Survival]
\label{thm:layer_survival}
For $n\geq 1$, under H1, and for all fixed $K\geq K_0$, $c\geq c_0$:
$\normh{A_K^{(n)}}{}\leq C_n\,K^{1+n\gamma}c^{-n\beta}$
with constant $C_n$ independent of $K,c$
(Euler--Maclaurin, as in Lemma~\ref{lem:trunc_universal}).
Diagonal evaluation via Lemma~\ref{lem:diagonal_sub}
(applying the slowly-varying stability~\eqref{eq:Kstar_power}
with exponent $1+n\gamma$):
\[
  \normh{A_K^{(n)}}{}_{K=K^*(c)}
  = O(c^{-n\beta}(\log c)^{1+n\gamma})
  \prec c^{-\beta}(\log c)^{1+\gamma}
  \quad(n\geq 2)
\]
by Proposition~\ref{prop:asymp_order};
the rate is governed by $n=1$.
\end{theorem}

\begin{theorem}[Two-scale rate, fixed-parameter]
\label{thm:universal_rate}
For $\cAg\in\mathfrak{T}_\gamma$,
all $K\geq K_0$, $c\geq c_0$:
\[
  \normh{\cK-\cKA}{}
  \leq C_{\rm lin}\,K^{1+\gamma}c^{-\beta}
  + C_{\rm quad}\,K^{1+2\gamma}c^{-2\beta}
  + \Ctwo^{1/2}e^{-\alpha K/2},
\]
with constants $C_{\rm lin}, C_{\rm quad}$ from
Lemma~\ref{lem:trunc_universal} (explicit, independent of $K$ and $c$).
\end{theorem}

\begin{corollary}[Universal logarithmic rate]
\label{cor:universal_lograte}
Apply Lemma~\ref{lem:diagonal_sub}
(with $C_{\rm lin}, C_{\rm quad}$ from Theorem~\ref{thm:universal_rate})
at $K=K_{\rm int}=\lfloor K^*(c)\rceil$:
\[
  \normh{\cK-\cKA}{}=O(c^{-\beta}(\log c)^{1+\gamma}).
\]
The leading constant is
$C_{\rm lin}\bigl(\frac{2\beta}{\alpha}\bigr)^{1+\gamma}$;
all three contributions and their dominance relations
are explicitly computed in Lemma~\ref{lem:diagonal_sub}(i)--(iii).
\end{corollary}

\begin{remark}[Spectral-geometric invariants]
\label{rem:invariants}
$1+\gamma$: leading exponent from $\sum j^\gamma$.
$1+2\gamma$: quadratic-layer exponent, subleading by $c^{-\beta}$.
$\gamma$ is absent from $K^*$'s leading coefficient $2\beta/\alpha$,
but governs the log-exponent and log-log correction.
\end{remark}

\begin{remark}[O3: sharpness gap and axiom status]
\label{rem:O3_sharpness}
\begin{itemize}[nosep]
  \item \textbf{O3a} (independent of H1--H2, true open problem):
    does $\exists\,\cAg\in\mathfrak{T}_\gamma$ with
    $\normh{\AKlin}{}=\Omega(K^{1+\gamma}c^{-\beta})$?
  \item \textbf{O3b} (would require H3, not imposed):
    a uniform lower bound for all $\cAg\in\mathfrak{T}_\gamma$
    requires a positive-mass / non-cancellation condition
    on the spectral measure; this is a genuinely new axiom
    not implied by H1 or H2.
\end{itemize}
Neither blocks any upper-bound conclusion.
\end{remark}
