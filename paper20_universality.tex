\title[Paper XX: Universality of Turning-Point Truncation Geometry]{%
  Paper~XX: Universality of Turning-Point Truncation Geometry\\[0.3em]
  \large Spectral-Geometric Invariants and Two-Scale Logarithmic Losses}

\author{\textit{prolate-gram-coercivity programme}}
\date{May 2026}

\begin{abstract}
For operators $\cAg\in\mathfrak{T}_\gamma$ (the
\emph{spectrally regularized WKB class}, Definition~\ref{def:opclass}),
the Langer term $A_K = A_K^{\rm lin} + A_K^{\rm quad}$ satisfies,
under the pointwise Langer scaling law
$\|e_j\|_{C^0}\leq C_0 j^\gamma c^{-\beta}$
(Hypothesis~\ref{hyp:langer_pointwise}),
\[
  \Clin = O(K^{1+\gamma}),\qquad \Cquad = O(K^{1+2\gamma}),
\]
via summation $\sum_{j=1}^K j^\gamma\sim K^{1+\gamma}/(1+\gamma)$.
At the first-order saddle balance
$K_{\rm opt}\sim 2\beta/(\alpha(1+\gamma))\log c$,
the quadratic layer is $o$ of the linear layer in the
total asymptotic order on $\mathcal{S} = \{c^{-s}(\log c)^t\}$
(Proposition~\ref{prop:asymp_order}),
and the rate is
\[
  \normh{\cK-\cKA}{} = O\!\left(c^{-\beta}(\log c)^{1+\gamma}\right).
\]
The \emph{Layer-Survival Selection Principle}
(Theorem~\ref{thm:layer_survival}) gives a complete characterisation:
for each interaction order $n\geq 1$, the $n$-th layer has
$K$-exponent $1+n\gamma$ and $c$-exponent $n\beta$;
at $K_{\rm opt}$ every layer with $n\geq 2$ is $o$ of the $n=1$ layer.
\end{abstract}

\section{Asymptotic Order and the $o$-Scale}
\label{sec:asymp}
%% ============================================================

\begin{definition}[Total asymptotic scale $\mathcal{S}$]
\label{def:scale}
Let $\mathcal{S}$ be the set of functions of the form
$c^{-s}(\log c)^t$ ($s>0$, $t\in\R$) ordered by:
\[
  c^{-s_1}(\log c)^{t_1} \succ c^{-s_2}(\log c)^{t_2}
  \quad\Longleftrightarrow\quad
  s_1 < s_2, \text{ or } (s_1=s_2 \text{ and } t_1 > t_2).
\]
This is a total order on $\mathcal{S}$ as $c\to\infty$.
\end{definition}

\begin{proposition}[Asymptotic order of the two layers]
\label{prop:asymp_order}
For any $\gamma>0$, $\beta>0$, and $K\sim\kappa\log c$ ($\kappa>0$):
\begin{equation}\label{eq:order}
  c^{-2\beta}(\log c)^{1+2\gamma}
  \prec
  c^{-\beta}(\log c)^{1+\gamma}
  \quad \text{in }\mathcal{S},
\end{equation}
because $2\beta > \beta$ (the $c$-exponent is strictly larger).
The comparison holds for all $K=O(\log c)$,
uniformly in $\gamma$ and $\kappa$.
In particular:
$c^{-2\beta}(\log c)^{1+2\gamma} = o(c^{-\beta}(\log c)^{1+\gamma})$
as $c\to\infty$.
\end{proposition}

\begin{proof}
$c^{-2\beta}(\log c)^{1+2\gamma} / c^{-\beta}(\log c)^{1+\gamma}
= c^{-\beta}(\log c)^\gamma \to 0$,
since $c^{-\beta}\cdot(\log c)^m\to 0$ for all $m>0$.
\end{proof}

\begin{remark}[Where $c^{-\beta}\gg(\log c)^m$ is used]
\label{rem:scales}
The comparison $c^{-\beta}\succ(\log c)^{-m}$ for every fixed $m$
is used in two places:
\begin{enumerate}[label=(\roman*),nosep]
  \item Proposition~\ref{prop:asymp_order}:
    linear layer dominates quadratic.
  \item Corollary~\ref{cor:universal_lograte}:
    spectral tail $e^{-\alpha K/2}\sim c^{-\alpha\kappa/2}$
    with $\alpha\kappa/2 = \beta/(1+\gamma) < \beta$;
    tail is strictly below the linear layer in $\mathcal{S}$.
\end{enumerate}
Both uses are explicit applications of Definition~\ref{def:scale};
the Layer-Survival Principle is a theorem about this order,
not an independent axiom.
\end{remark}

\section{Framework: Spectrally Regularized WKB Class}
\label{sec:framework}
%% ============================================================

\begin{definition}[Spectrally regularized WKB class $\mathfrak{T}_\gamma$]
\label{def:opclass}
For $\gamma>0$ and $\beta>0$:
$\mathfrak{T}_\gamma$ is the class of self-adjoint operators
$\cAg=-\partial_\xi^2+V(\xi)$ on $L^2(\R_+)$,
$V\geq 0$, $V\to+\infty$, discrete spectrum $0<a_1<a_2<\cdots$,
satisfying:
\begin{enumerate}[label=(\roman*),nosep]
  \item $a_j\sim C_\gamma j^\gamma$ (spectral growth);
  \item $V$ is $C^2$, simple turning points, Langer--Olver applicable;
  \item Hypothesis~\ref{hyp:langer_pointwise}
    (pointwise Langer scaling law);
  \item $C_\infty:=\sup_{j\geq 1}\|u_j\|_{C^0([0,M])}<\infty$
    (Hypothesis~\ref{hyp:Cinfty}).
\end{enumerate}
The term \emph{spectrally regularized} refers to (iii)--(iv):
these are additional regularity conditions on the WKB structure
beyond what classical Olver theory guarantees automatically.
\end{definition}

\begin{hypothesis}[Pointwise Langer scaling law]
\label{hyp:langer_pointwise}
There exists $C_0=C_0(M,V)<\infty$, independent of $j$, $K$, $c$,
such that for $c\geq c_0(j)$:
\begin{equation}\label{eq:langer_pointwise}
  \|e_j\|_{C^0([0,M])} \leq C_0\,j^\gamma\,c^{-\beta}.
\end{equation}
Status: verified for Airy; proof sketch for power family
(\texttt{olver\_growth\_lemma.tex},
Lemma~\texttt{lem:olver\_growth}); open in general.
\end{hypothesis}

\begin{hypothesis}[$C_\infty$ boundedness]
\label{hyp:Cinfty}
$C_\infty:=\sup_{j\geq 1}\|u_j\|_{C^0([0,M])}<\infty$.

Status: standard for Airy ($L^\infty$ Airy bound);
follows from WKB normalization for the power family;
open in general.
\end{hypothesis}

\section{$K_{\rm opt}$: First-Order Saddle Balance}
\label{sec:kopt}
%% ============================================================

\begin{remark}[$K_{\rm opt}$ as asymptotic saddle balance]
\label{rem:kopt}
The optimal truncation $\Kopt$ is determined by the
\emph{first-order saddle balance} between the linear layer and
the spectral tail in Theorem~\ref{thm:universal_rate}:
\begin{equation}\label{eq:balance}
  K^{1+\gamma}\,c^{-\beta} \asymp e^{-\alpha K/2}.
\end{equation}
Logarithmating~\eqref{eq:balance}:
$(1+\gamma)\log K - \beta\log c \sim -\frac{\alpha}{2}K$.
Since $\log K = o(K)$ for large $K$,
the dominant balance is $-\beta\log c \sim -\frac{\alpha}{2}K$,
giving the leading-order relation
\begin{equation}\label{eq:kopt}
  \Kopt \sim \frac{2\beta}{\alpha}\log c.
\end{equation}
Including the $\log K$ correction:
$K_{\rm opt}\log K_{\rm opt} = \frac{\beta}{\alpha/(1+\gamma)/2}$,
which at leading order gives the refined estimate
\begin{equation}\label{eq:kopt_refined}
  \Kopt \sim \frac{2\beta}{\alpha(1+\gamma)}\log c
  \qquad (\text{first correction from }\log K\text{ term}).
\end{equation}

\textbf{Caution.} Both~\eqref{eq:kopt} and~\eqref{eq:kopt_refined}
are asymptotic identities, not exact canonical scales.
The rate $O(c^{-\beta}(\log c)^{1+\gamma})$ holds for any
$\Kopt = \Theta(\log c)$; the precise prefactor
$2\beta/(\alpha(1+\gamma))$ gives the tightest constant
but is not structurally canonical in the sense of a fixed-point.
We use~\eqref{eq:kopt_refined} as the stated coupling.

\textbf{Airy check:} $\gamma=2/3$, $\beta=1/3$:
$\Kopt\sim 2/(5\alpha)\log c$.
\end{remark}

\section{DAG: Agnostic and Sensitive Nodes}

\begin{lemma}[$\gamma$-Agnostic nodes]
\label{lem:gamma_agnostic}
The following hold verbatim for every $\cAg\in\mathfrak{T}_\gamma$:
\begin{enumerate}[label=(\roman*),nosep]
  \item Split $\SK-\cKA = \AK-\BK$;
    $\AK = \AKlin + \AKquad$.
  \item $\|f\otimes g\|_{L^2}\leq M\|f\|_{C^0}\|g\|_{C^0}$.
  \item Bessel tail (XVIII~\texttt{lem:besseltail}).
  \item Tonelli/DCT.
  \item Slepian tail (XVIII~\texttt{lem:slepianunif}):
    $\|\BK\|\leq \Ctwo^{1/2}e^{-\alpha K/2}$.
  \item HS/Cauchy--Schwarz on $\BK$.
  \item Sequential $(c,K)$-limit structure.
  \item Norm-resolvent perturbation (Kato~VI).
\end{enumerate}
\end{lemma}

\begin{remark}[$\gamma$-Sensitive nodes]
\label{rem:sensitive}
Exactly four nodes depend on $\gamma$:
(i) $a_j\sim j^\gamma$;
(ii) turning-point location $\xi_j\sim j^{\gamma/p}$;
(iii) pointwise Langer bound $\|e_j\|\leq C_0 j^\gamma c^{-\beta}$;
(iv) summation exponents $1+\gamma$, $1+2\gamma$.
\end{remark}

\section{Truncation Geometry via Summation}
\label{sec:trunc}
%% ============================================================

\begin{definition}[Propagation spine]
\label{def:spine}
\[
\begin{tikzcd}[column sep=1.8em]
  a_j{\sim}j^\gamma
  \ar[r,"\text{WKB}"]
  & \|e_j\|{\leq} C_0 j^\gamma c^{-\beta}
  \ar[r,"\textstyle\sum_j\cdot C_\infty"]
  & \Clin{\sim} K^{1+\gamma}c^{-\beta}
\end{tikzcd}
\]
\[
\begin{tikzcd}[column sep=1.8em]
  \|e_j\|{\leq} C_0 j^\gamma c^{-\beta}
  \ar[r,"\textstyle\sum_j(\cdot)^2"]
  & \Cquad{\sim} K^{1+2\gamma}c^{-2\beta}.
\end{tikzcd}
\]
\end{definition}

\begin{lemma}[Universal truncation geometry]
\label{lem:trunc_universal}
Under Hypotheses~\ref{hyp:langer_pointwise} and~\ref{hyp:Cinfty},
for $K\geq 1$ and $c\geq c_0$:
\begin{align}
  \normh{\AKlin}{}
    &\leq 2MC_\infty C_0 c^{-\beta}\sum_{j=1}^K j^\gamma
    \leq \underbrace{\frac{2MC_\infty C_0}{1+\gamma}K^{1+\gamma}
      +O(K^\gamma)}_{\Clin}\cdot c^{-\beta}, \label{eq:AKlin}\\
  \normh{\AKquad}{}
    &\leq MC_0^2\,c^{-2\beta}\sum_{j=1}^K j^{2\gamma}
    \leq \underbrace{\frac{MC_0^2}{1+2\gamma}K^{1+2\gamma}
      +O(K^{2\gamma})}_{\Cquad}\cdot c^{-2\beta}. \label{eq:AKquad}
\end{align}
Hence $\Clin=O(K^{1+\gamma})$, $\Cquad=O(K^{1+2\gamma})$.
\end{lemma}

\begin{proof}
For each $j\leq K$:
$\|u_j\otimes e_j+e_j\otimes u_j\|\leq 2MC_\infty C_0 j^\gamma c^{-\beta}$;
$\|e_j\otimes e_j\|\leq MC_0^2 j^{2\gamma}c^{-2\beta}$.
Note $c^{-2\beta}$ is not absorbed into $c^{-\beta}$.
Sum over $j=1,\ldots,K$ using
$\sum_{j=1}^K j^\sigma = K^{1+\sigma}/(1+\sigma)+O(K^\sigma)$.
\end{proof}

\section{Layer-Survival Selection Principle}
\label{sec:layersurvival}
%% ============================================================

\begin{theorem}[Layer-Survival Selection Principle]
\label{thm:layer_survival}
For $n\geq 1$, define the $n$-th interaction layer as the
collection of terms in $A_K$ with exactly $n$ factors of $e_j$
(and $2-n$ factors of $u_j$ in the $n=1$ case):
\begin{equation}\label{eq:nth_layer}
  \text{layer }n:\quad
  \normh{A_K^{(n)}}{} \leq
  \frac{MC_\infty^{2-n}C_0^n}{1+n\gamma}\,K^{1+n\gamma}\,c^{-n\beta}
  + O(K^{n\gamma}\,c^{-n\beta}).
\end{equation}
(For $n=1$: linear layer; $n=2$: quadratic layer.
Higher $n$ arise in expanded versions.)

At the saddle balance $K=\Kopt\sim\kappa\log c$
($\kappa=2\beta/(\alpha(1+\gamma))$):
\begin{equation}\label{eq:survival}
  \normh{A_K^{(n)}}{} = O\!\left(c^{-n\beta}(\log c)^{1+n\gamma}\right)
  \prec c^{-\beta}(\log c)^{1+\gamma}
  \quad\text{for all }n\geq 2,
\end{equation}
because $n\beta > \beta$ and therefore $c^{-n\beta}\prec c^{-\beta}$
in the total order $\mathcal{S}$ of Definition~\ref{def:scale}.

Consequently, the rate is governed by the $n=1$ layer alone:
\[
  \normh{\cK-\cKA}{} = O\!\left(c^{-\beta}(\log c)^{1+\gamma}\right).
\]

\textbf{Conceptual content.}
For each additional Langer error ($n\to n+1$):
$K$-exponent increases by $\gamma$ ($1+n\gamma\to 1+(n+1)\gamma$),
but $c$-exponent also increases by $\beta$ ($n\beta\to(n+1)\beta$).
Since $c^{-\beta}\succ(\log c)^m$ for all $m$,
every additional factor of $c^{-\beta}$ dominates any
logarithmic gain. The $c$-scaling hierarchy always selects
the layer with fewest Langer errors.
\end{theorem}

\begin{proof}
Layer $n$ bound follows from Lemma~\ref{lem:trunc_universal}
applied to $n$-fold products with
$n$ factors $\|e_j\|\leq C_0 j^\gamma c^{-\beta}$
and $\sum_{j=1}^K j^{n\gamma}\sim K^{1+n\gamma}/(1+n\gamma)$.
Dominance~\eqref{eq:survival}: by Proposition~\ref{prop:asymp_order}
generalised to $(c^{-(n-1)\beta}(\log c)^{(n-1)\gamma})\to 0$.
\end{proof}

\section{Universal Rate Theorem}
\label{sec:main}
%% ============================================================

\begin{theorem}[Universal two-scale rate]
\label{thm:universal_rate}
Let $\cAg\in\mathfrak{T}_\gamma$ (Definition~\ref{def:opclass}),
$K\geq K_0$, $c\geq c_0$:
\begin{equation}\label{eq:twoscale_universal}
  \normh{\cK-\cKA}{}
  \leq \Clin\cdot c^{-\beta}
  + \Cquad\cdot c^{-2\beta}
  + \Ctwo^{1/2}\cdot e^{-\alpha K/2}.
\end{equation}
\end{theorem}

\begin{corollary}[Universal logarithmic rate]
\label{cor:universal_lograte}
With $\Kopt$ from~\eqref{eq:kopt_refined}:
\begin{equation}\label{eq:lograte_universal}
  \normh{\cK-\cKA}{}\Big|_{K=\Kopt}
  = O\!\left(c^{-\beta}(\log c)^{1+\gamma}\right).
\end{equation}
All three terms (linear, quadratic, tail) are explicit;
quadratic and tail are $\prec$ linear in $\mathcal{S}$
(Remark~\ref{rem:scales} + Theorem~\ref{thm:layer_survival}).
\end{corollary}

\begin{remark}[Two spectral-geometric invariants]
\label{rem:invariants}
\begin{itemize}[nosep]
  \item $1+\gamma$: exponent of the linear layer;
    governs the rate; arises from $\sum j^\gamma$.
  \item $1+2\gamma$: exponent of the quadratic layer;
    structurally present; arises from $\sum j^{2\gamma}$;
    subleading by one factor $c^{-\beta}$ in $\mathcal{S}$.
\end{itemize}
The gap $\gamma$ measures the additional crowding cost
of one extra Langer interaction. Both invariants are
spectral-geometric quantities of $\cAg$.
\end{remark}
