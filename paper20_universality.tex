\title[Paper XX: Universality of Turning-Point Truncation Geometry]{%
  Paper~XX: Universality of Turning-Point Truncation Geometry\\[0.3em]
  \large Spectral-Geometric Invariants and Two-Scale Logarithmic Losses}

\author{\textit{prolate-gram-coercivity programme}}
\date{May 2026}

\begin{abstract}
Papers~XVIII--XIX established qualitative Airy universality (BR3) and
a quantitative two-scale rate $O(c^{-1/3}(\log c)^{5/3})$.
The Langer term $A_K$ splits into two interaction layers:
\begin{align*}
  A_K^{\mathrm{lin}}
    &= \textstyle\sum_{j=1}^K(u_j\otimes e_j + e_j\otimes u_j)
    &&\text{one exact mode, one Langer error},\\
  A_K^{\mathrm{quad}}
    &= \textstyle\sum_{j=1}^K e_j\otimes e_j
    &&\text{two Langer errors}.
\end{align*}
The key structure of this paper is the \emph{pointwise Langer scaling law}:
\[
  \|e_j\|_{C^0([0,M])} \leq C\,j^\gamma\,c^{-\beta},
  \qquad j\leq K,
\]
(Hypothesis~\ref{hyp:langer_pointwise}),
which is the correct Olver-compatible formulation replacing
any uniform-in-$j$ assumption.
Summation over $j=1,\ldots,K$ then gives, via $\sum_{j=1}^K j^\gamma\sim K^{1+\gamma}$
and $\sum_{j=1}^K j^{2\gamma}\sim K^{1+2\gamma}$:
\[
  C_{\mathrm{lin}} = O(K^{1+\gamma}), \qquad C_{\mathrm{quad}} = O(K^{1+2\gamma}),
\]
and the two-scale rate
\[
  \|\cK-\cKA\| = O\!\left(c^{-\beta}(\log c)^{1+\gamma}\right)
\]
at
$K_{\mathrm{opt}} \sim \dfrac{\beta}{\alpha(1+\gamma)}\log c$.

The central mechanism is the \emph{Layer-Survival Selection Principle}
(Remark~\ref{rem:selection_principle}): the exponent $1+\gamma$ governs
the rate (not $1+2\gamma$, despite $1+2\gamma > 1+\gamma$) because
the quadratic layer pays $c^{-2\beta}$ for two Langer errors,
dropping it below the $c$-horizon at $K_{\mathrm{opt}}$.
\end{abstract}

\begin{remark}[Layer-Survival Selection Principle]
\label{rem:selection_principle}
At the optimal coupling $K=\Kopt$, an interaction layer with
$K$-exponent $\lambda$ and $c$-scale $c^{-\sigma}$ survives the
$c$-hierarchy if and only if
$(\log c)^\lambda c^{-\sigma} \not= o((\log c)^{\lambda_1}c^{-\sigma_1})$,
where layer~$1$ is the layer with smallest $\sigma$.

In our setting:
\begin{itemize}[nosep]
  \item Linear layer: $\lambda_1=1+\gamma$, $\sigma_1=\beta$ — survives.
  \item Quadratic layer: $\lambda_2=1+2\gamma$, $\sigma_2=2\beta$ —
    does not survive: $(\log c)^{1+2\gamma}c^{-2\beta} =
    c^{-\beta}\cdot(\log c)^{1+2\gamma}c^{-\beta} =
    o((\log c)^{1+\gamma}c^{-\beta})$.
\end{itemize}
\emph{The rate exponent $1+\gamma$ is determined not by maximal
$K$-growth, but by the highest interaction layer surviving
the $c$-scaling hierarchy.}
\end{remark}

\section{Framework and Olver Compatibility}
\label{sec:framework}
%% ============================================================

\begin{definition}[Operator class $\mathfrak{T}_\gamma$]
\label{def:opclass}
For $\gamma>0$: $\mathfrak{T}_\gamma$ is the class of self-adjoint operators
$\cAg = -\partial_\xi^2+V(\xi)$ on $L^2(\R_+)$,
$V\geq 0$, $V\to+\infty$, discrete spectrum $0<a_1<a_2<\cdots$, with:
\begin{enumerate}[label=(\roman*),nosep]
  \item $a_j \sim C_\gamma j^\gamma$ (spectral growth);
  \item $V$ is $C^2$ near its turning points; Langer--Olver applicable;
  \item Hypothesis~\ref{hyp:langer_pointwise} holds.
\end{enumerate}
\end{definition}

\begin{definition}[Parameters $\gamma$, $\beta$, $\alpha$]
\label{def:params}
$\gamma>0$: spectral growth exponent ($a_j\sim j^\gamma$);
$\beta>0$: turning-point window ($c^{-\beta}$ per Langer error;
Airy: $\beta=1/3$, independent of $\gamma$);
$\alpha>0$: Slepian tail decay (XVIII).
\end{definition}

\begin{remark}[Why uniform-in-$j$ is the wrong formulation]
\label{rem:olver_uniformity}
The classical Olver theory (\cite{Olver1997}, Ch.~11--12) provides,
for fixed $j$, a local uniform estimate
\[
  \|e_j\|_{C^0([0,M])} \leq C_j\,c^{-\beta}.
\]
However, the theory does \emph{not} guarantee
$\sup_{j\leq K} C_j < \infty$.

The source of the $j$-dependence is the turning-point location:
$V(\xi_j)\sim a_j\sim j^\gamma$, so $\xi_j\sim j^{\gamma/p}$
(for $V\sim\xi^p$).
The Olver remainder involves derivatives of $V$ at $\xi_j$:
\[
  e_j(\xi) = c^{-\beta}\Big(
    A(\xi,\xi_j)\,V''(\xi_j) + B(\xi,\xi_j)\,V'(\xi_j)^2 + \cdots\Big),
\]
where $A, B$ are bounded functions of $(\xi-\xi_j)$.
For $V\sim\xi^p$:
$V'(\xi_j)\sim j^{\gamma(1-1/p)}$ and $V''(\xi_j)\sim j^{\gamma(1-2/p)}$,
so the Olver remainder grows algebraically in $j$.

The \emph{correct} formulation is therefore a \emph{pointwise}
scaling law $\|e_j\|_{C^0}\leq C j^\gamma c^{-\beta}$
(Hypothesis~\ref{hyp:langer_pointwise}).
Uniformity in $j$ is replaced by controlled growth;
the exponents $1+\gamma$ and $1+2\gamma$ then arise by summation,
not by a global norm bound.
\end{remark}

\begin{hypothesis}[Pointwise Langer scaling law]
\label{hyp:langer_pointwise}
There exists $C_0 = C_0(M,V) < \infty$ (independent of $j$, $K$, $c$)
such that, for fixed $M$ and $c\geq c_0(j)$:
\begin{equation}\label{eq:langer_pointwise}
  \|e_j\|_{C^0([0,M])} \leq C_0\,j^\gamma\,c^{-\beta},
  \qquad j = 1,2,\ldots
\end{equation}
The constant $C_0$ absorbs $M$-dependent geometry but is
uniform in $j$; the $j$-dependence is entirely in $j^\gamma$.

\medskip
\textbf{Airy verification} ($V=\xi$, $\gamma=2/3$, $\beta=1/3$):
$a_j = $ $j$-th Airy zero $\sim (3\pi j/2)^{2/3}\sim j^{2/3}$;
XVIII~\texttt{rem:langerkdep} gives $C_j = O(j^{2/3})$,
hence $\|e_j\|_{C^0}\leq C_0 j^{2/3}c^{-1/3}$. \checkmark
\end{hypothesis}

\begin{remark}[Hypothesis~\ref{hyp:langer_pointwise} vs.\ previous formulation]
\label{rem:hyp_comparison}
Previous versions of this paper used
$\|e_j\|_{C^0}\leq \hat C(M,K)c^{-\beta}$ with $\hat C = O(K^\gamma)$
uniform in $j\leq K$.
This is \emph{a priori} stronger (it implies~\eqref{eq:langer_pointwise}
with $C_0 = \hat C / j^\gamma \leq \hat C / 1 = \hat C$, but growing
in $K$ and hence $j$-dependent in the wrong direction).
Hypothesis~\ref{hyp:langer_pointwise} is weaker and more natural:
it requires only that the pointwise growth is $\leq C_0 j^\gamma$
with a genuinely universal constant $C_0$.
The summation then automatically produces the correct exponents.
\end{remark}

(iv) $\|e_j\|_{C^0} \leq C_0 j^\gamma c^{-\beta}$ (Langer pointwise).

\section{Universal Truncation Geometry via Summation}
\label{sec:trunc}
%% ============================================================

\begin{definition}[Propagation spine]
\label{def:spine}
The $\gamma$-sensitive chain, now with explicit summation:
\[
\begin{tikzcd}[column sep=2.0em]
  a_j{\sim}j^\gamma
  \ar[r]
  & \|e_j\|_{C^0}{\leq} C_0 j^\gamma c^{-\beta}
  \ar[r,"\sum_j (\cdot)\cdot C_\infty"]
  & C_{\rm lin}{\sim}K^{1+\gamma}c^{-\beta}
\end{tikzcd}
\]
\[
\begin{tikzcd}[column sep=2.0em]
  \|e_j\|_{C^0}{\leq} C_0 j^\gamma c^{-\beta}
  \ar[r,"\sum_j (\cdot)^2"]
  & C_{\rm quad}{\sim}K^{1+2\gamma}c^{-2\beta}.
\end{tikzcd}
\]
Each additional Langer error adds $\gamma$ to the $K$-exponent
\emph{and} $\beta$ to the $c$-exponent.
The exponents $1+\gamma$ and $1+2\gamma$ arise from
$\sum_{j=1}^K j^\gamma \sim \frac{K^{1+\gamma}}{1+\gamma}$
and $\sum_{j=1}^K j^{2\gamma} \sim \frac{K^{1+2\gamma}}{1+2\gamma}$,
respectively.
\end{definition}

\begin{lemma}[Universal truncation geometry]
\label{lem:trunc_universal}
Let $\cAg\in\mathfrak{T}_\gamma$ satisfy
Hypothesis~\ref{hyp:langer_pointwise},
and let $C_\infty := \sup_{j\geq 1}\|u_j\|_{C^0([0,M])} < \infty$.
For fixed $M$, $K\geq 1$, $c\geq c_0$:
\begin{align}
  \normh{\AKlin}{L^2([0,M]^2)}
    &\leq 2MC_\infty C_0 c^{-\beta}
      \sum_{j=1}^K j^\gamma
    \leq \underbrace{\frac{2MC_\infty C_0}{1+\gamma}\,K^{1+\gamma}+O(K^\gamma)}_
      {=:\Clin}
      \cdot c^{-\beta}, \label{eq:AKlin} \\
  \normh{\AKquad}{L^2([0,M]^2)}
    &\leq MC_0^2\,c^{-2\beta}
      \sum_{j=1}^K j^{2\gamma}
    \leq \underbrace{\frac{MC_0^2}{1+2\gamma}\,K^{1+2\gamma}+O(K^{2\gamma})}_
      {=:\Cquad}
      \cdot c^{-2\beta}. \label{eq:AKquad}
\end{align}
Hence $\Clin = O(K^{1+\gamma})$ and $\Cquad = O(K^{1+2\gamma})$
with constants depending only on $M$, $C_0$, $C_\infty$, $\gamma$.
\end{lemma}

\begin{proof}
For each $j\leq K$, by
$\|f\otimes g\|_{L^2([0,M]^2)}\leq M\|f\|_{C^0}\|g\|_{C^0}$
and Hypothesis~\ref{hyp:langer_pointwise}:
\[
  \normh{u_j\otimes e_j + e_j\otimes u_j}{}
  \leq 2MC_\infty\cdot C_0 j^\gamma\,c^{-\beta},
  \qquad
  \normh{e_j\otimes e_j}{}
  \leq M\cdot C_0^2 j^{2\gamma}\,c^{-2\beta}.
\]
Note: the quadratic term carries $j^{2\gamma}c^{-2\beta}$ exactly,
without any $c$-absorption.
Summing over $j=1,\ldots,K$ and applying
$\sum_{j=1}^K j^\sigma = K^{1+\sigma}/(1+\sigma)+O(K^\sigma)$
($\sigma > 0$) gives~\eqref{eq:AKlin}--\eqref{eq:AKquad}.

\textbf{Airy check:}
$C_0=O(1)$ (Airy constants), $\gamma=2/3$:
$\Clin = O(K^{5/3})$~$\checkmark$ (XIX), $\Cquad = O(K^{7/3})$~$\checkmark$.
\end{proof}

\begin{remark}[Summation is the correct proof mechanism]
\label{rem:summation}
The derivation above shows that $C_{\rm lin}=O(K^{1+\gamma})$
and $C_{\rm quad}=O(K^{1+2\gamma})$ are \emph{not} consequences
of a global norm bound on $\{e_j\}_{j\leq K}$,
but of the summation of the pointwise Olver error
$\|e_j\|_{C^0}\leq C_0 j^\gamma c^{-\beta}$ via
$\sum_{j=1}^K j^\gamma\sim K^{1+\gamma}/(1+\gamma)$.
This is consistent with the classical Olver theory
(Remark~\ref{rem:olver_uniformity}) and
requires no assumption beyond Hypothesis~\ref{hyp:langer_pointwise}.
\end{remark}

\section{Universal Rate Theorem}
\label{sec:main}
%% ============================================================

\begin{theorem}[Universal two-scale rate]
\label{thm:universal_rate}
Let $\cAg\in\mathfrak{T}_\gamma$ satisfy
Hypothesis~\ref{hyp:langer_pointwise}.
For $K\geq K_0$ and $c\geq c_0$:
\begin{equation}\label{eq:twoscale_universal}
  \normh{\cK-\cKA}{}
  \leq
  \Clin\cdot c^{-\beta}
  + \Cquad\cdot c^{-2\beta}
  + \Ctwo^{1/2}\cdot e^{-\alpha K/2}.
\end{equation}
\end{theorem}

\begin{proof}
Triangle inequality: $\normh{\AK}{}\leq\normh{\AKlin}{}+\normh{\AKquad}{}$,
then Lemma~\ref{lem:trunc_universal} + tail bound
(Lemma~\ref{lem:gamma_agnostic}(vi)).
\end{proof}

\begin{remark}[Optimal $K_{{\rm opt}}$ and the tail balance]
\label{rem:kopt}
The three terms in~\eqref{eq:twoscale_universal} are balanced by
choosing $K=\Kopt$ so that the dominant linear term equals the tail:
\[
  K^{1+\gamma}\,c^{-\beta} \sim e^{-\alpha K/2}.
\]
Logarithmating: $(1+\gamma)\log K - \beta\log c \sim -\frac{\alpha}{2}K$,
which gives (for large $c$, where $\log K \ll K$):
\begin{equation}\label{eq:kopt}
  \Kopt \sim \frac{\beta}{\alpha(1+\gamma)/2}\log c
  = \frac{2\beta}{\alpha(1+\gamma)}\log c.
\end{equation}

\textbf{Airy check:} $\gamma=2/3$, $\beta=1/3$:
$\Kopt \sim \frac{2/3}{\alpha\cdot 5/3}\log c = \frac{2}{5\alpha}\log c$.

\medskip
\textbf{Correction to earlier versions.}
Previous versions used $\Kopt\sim(\log c)/(3\alpha)$ (XIX)
or $\Kopt\sim(2\beta/\alpha)\log c$ (XX first version).
Both are incorrect: the factor $(1+\gamma)$ in the denominator
was missing. The rate $O(c^{-\beta}(\log c)^{1+\gamma})$ is
unaffected by this correction (it holds for any
$\Kopt = \Theta(\log c)$), but the precise constant in the coupling
changes.
\end{remark}

\begin{corollary}[Universal logarithmic rate]
\label{cor:universal_lograte}
With $\Kopt$ as in~\eqref{eq:kopt}, for $c$ large:
\begin{equation}\label{eq:lograte_universal}
  \normh{\cK-\cKA}{}\Big|_{K=\Kopt}
  = O\!\left(c^{-\beta}(\log c)^{1+\gamma}\right).
\end{equation}
\end{corollary}

\begin{proof}
At $K=\Kopt\sim \frac{2\beta}{\alpha(1+\gamma)}\log c$:
\begin{enumerate}[label=(\roman*),nosep]
  \item \emph{Linear:}
    $\Clin c^{-\beta} = O(K^{1+\gamma}c^{-\beta})
    = O((\log c)^{1+\gamma}c^{-\beta})$.
  \item \emph{Quadratic:}
    $\Cquad c^{-2\beta} = O(K^{1+2\gamma}c^{-2\beta})
    = O((\log c)^{1+2\gamma}c^{-2\beta})
    = o(c^{-\beta}(\log c)^{1+\gamma})$
    since $c^{-\beta}\gg(\log c)^m$ for every fixed $m$.
  \item \emph{Tail:}
    $e^{-\alpha\Kopt/2}\leq c^{-\beta/(1+\gamma)}
    = o(c^{-\beta}(\log c)^{1+\gamma})$.
\end{enumerate}
Term (i) dominates. \qed
\end{proof}

\begin{remark}[Two spectral-geometric invariants]
\label{rem:invariants}
\begin{itemize}[nosep]
  \item $1+\gamma$: exponent of the linear layer;
    governs the rate (survives $c$-hierarchy).
    Arises from $\sum_{j=1}^K j^\gamma\sim K^{1+\gamma}/(1+\gamma)$.
  \item $1+2\gamma$: exponent of the quadratic layer;
    real geometric quantity, subleading by $c^{-\beta}$ at $\Kopt$.
    Arises from $\sum_{j=1}^K j^{2\gamma}\sim K^{1+2\gamma}/(1+2\gamma)$.
\end{itemize}
The gap $(1+2\gamma)-(1+\gamma)=\gamma$ measures the additional
crowding cost of one extra Langer error in the quadratic interaction.
All three quantities $1+\gamma$, $1+2\gamma$, $\gamma$ are spectral:
they are determined by the turning-point geometry alone.
\end{remark}
