\begin{abstract}
This paper proves Lemma~\ref{lem:main}:
$\cK(\xi,\eta)\to\cKA(\xi,\eta)$ for fixed $\xi,\eta\geq 0$,
closing BR3 with Paper~XVII.

\medskip
\textbf{Key corrections in v5.}
\begin{enumerate}[nosep]
  \item \emph{Turning-point scaling (Lemma~\ref{lem:tpshift}):}
    in Airy coordinates, $\xi_{n^*+j}^* = O(c^{-1/3})$ for
    \emph{all} $|j|\leq K$; the Airy window contains all $K$ turning
    points, with spacing $O(c^{-1/3})\to 0$.
    (Previous versions incorrectly had $\xi_{n^*+j}^*=O(c^{1/3})$.)
  \item \emph{I2b without Airy leakage (Lemma~\ref{lem:I2b}):}
    the Olver bound on $[0,M]$ is derived from the oscillatory
    WKB amplitude, valid uniformly for all $j$ since all turning
    points cluster at $O(1)$ in Airy coords.
    No Airy-form input required.
  \item \emph{Projector convergence closed (Proposition~\ref{prop:PR}
    + Remark~\ref{rem:projconv}):}
    $\PcK\to\PK$ in operator norm (finite-dimensional);
    index identification via eigenvalue ordering.
  \item \emph{Airy kernel identity (Remark~\ref{rem:airykernel}):}
    $\cKA=\sum_k u_k\otimes u_k$ cited as spectral theorem
    for $\cA$ on $L^2(\R_+)$ with Dirichlet BC.
\end{enumerate}
\end{abstract}

\section{Setup and Coordinate Systems}
\label{sec:setup}
%% ============================================================

\begin{definition}[Coordinate systems and turning points]
\label{def:scales}
\mbox{}
\begin{enumerate}[nosep,label=(\Roman*)]
  \item \emph{Physical:} $x\in[-1,1]$,
    native domain of $\psi_n^{(c)}$.
  \item \emph{Airy:} $\xi = c^{2/3}(1-x)\geq 0$,
    so $x = 1-c^{-2/3}\xi$.
    Fixed window $\xi\in[0,M]$ $\leftrightarrow$
    $x\in[1-Mc^{-2/3},1]$.
  \item \emph{Spectral:} $s_n := (n-2c/\pi)/c^{1/3}$.
    Fermi index $n^*=\lfloor 2c/\pi\rfloor$,
    so $s_{n^*}=O(c^{-1/3})$;
    for $n=n^*+j$: $s_{n^*+j} = jc^{-1/3}+O(c^{-2/3})$.
\end{enumerate}
\emph{Turning point of $\psi_{n^*+j}^{(c)}$.}
The PSWF Sturm--Liouville equation (Paper~XVI Def.~1.1) has
turning point at physical coordinate
$x_{n^*+j}^* = (n^*+j)\pi/(2c)$
(where the potential changes sign).
In Airy coordinates:
\begin{equation}\label{eq:tp}
  \xi_{n^*+j}^*
  := c^{2/3}(1-x_{n^*+j}^*)
  = c^{2/3}\Bigl(1-\frac{(n^*+j)\pi}{2c}\Bigr)
  = c^{2/3}\cdot\frac{2c/\pi - n^* - j}{c}
  \cdot\frac{\pi}{2}
  = -s_{n^*+j}\cdot\frac{\pi c^{1/3}}{2}\cdot c^{-1/3}
  = -\frac{\pi}{2}\,s_{n^*+j}
\end{equation}
where we used $c^{2/3}(1-(n^*+j)\pi/(2c))
= c^{2/3}(2c/\pi - n^* - j)\pi/(2c)
= (\pi/2)(2c/\pi-n^*-j)/c^{1/3}
= (\pi/2)(-s_{n^*+j})$.
More precisely: $\xi_{n^*+j}^* = -(\pi/2)s_{n^*+j}+O(c^{-2/3})$,
so
\begin{equation}\label{eq:tp2}
  \xi_{n^*+j}^*
  = -\frac{\pi}{2}\cdot jc^{-1/3} + O(c^{-2/3})
  = O(c^{-1/3})\quad\text{for fixed }|j|\leq K.
\end{equation}
\textbf{Key point:} all turning points $\xi_{n^*+j}^*$, $|j|\leq K$,
converge to $0$ in Airy coordinates as $c\to\infty$,
with spacing $O(c^{-1/3})$ between adjacent turning points.
\end{definition}

\begin{remark}[All $K$ modes are Airy modes]
\label{rem:allairyK}
Equation~\eqref{eq:tp2} corrects v3--v4:
the turning points do \emph{not} diverge to $\pm\infty$ in Airy coords.
Rather, all $K$ turning points cluster near $\xi=0$,
all within $[0,M]$ for $M\geq 2$ and $c$ large.
Every mode $|j|\leq K$ is an ``Airy mode'';
$j=0$ is not distinguished geometrically.
The Airy mode structure is an \emph{ensemble} of $K$ modes,
not a single one.
\end{remark}

\begin{definition}[Kernels, rescaled PSWFs, identification map]
\label{def:kernels}
Rescaled PSWF: $\tilde\psi_j(\xi) := c^{1/3}\psi_{n^*+j}^{(c)}(1-c^{-2/3}\xi)$.\\
Rescaled kernel: $\cK(\xi,\eta) := c^{-2/3}K_c(1-c^{-2/3}\xi,1-c^{-2/3}\\eta)$.\\
Airy eigenfunctions: $u_k\in L^2(\R_+)$, $\cA u_k = -a_k u_k$
($\cA=-\partial_\xi^2+\xi$, Dirichlet at~$0$,
eigenvalues $0<a_1<a_2<\cdots$).\\
Airy kernel: $\cKA(\xi,\eta) := \sum_{k=1}^\infty u_k(\xi)u_k(\eta)$
(Remark~\ref{rem:airykernel}).\\
Identification map: $\Uc : L^2([-1,1])\to L^2(\R_+)$,
$(\Uc f)(\xi) := c^{1/3}f(1-c^{-2/3}\xi)$.
Then $\Uc\psi_{n^*+j}^{(c)} = \tilde\psi_j$ and
$\|\Uc f\|_{L^2(\R_+)} = \|f\|_{L^2([-1,1])}(1+O(e^{-\pi c}))$.\\
Projectors: $\PcK := \sum_{|j|\leq K}|\tilde\psi_j\rangle\langle\tilde\psi_j|$,
$\PK := \sum_{k=1}^K|u_k\rangle\langle u_k|$
(both on $L^2(\R_+)$).
\end{definition}

\begin{remark}[Airy kernel as spectral theorem]
\label{rem:airykernel}
The equality $\cKA(\xi,\eta) = \sum_{k\geq 1}u_k(\xi)u_k(\eta)$
is the spectral resolution of the identity for $\cA=-\partial_\xi^2+\xi$
on $L^2(\R_+)$ with Dirichlet BC (a positive self-adjoint operator
with compact resolvent).
It coincides with the integral form
$\int_0^\infty\Ai(\xi+t)\Ai(\eta+t)\,dt$
by completeness of Airy functions on $\R_+$
(see e.g.\ \cite{Olver1997} Ch.~11, or \cite{Simon2005} Vol.~4
for the spectral theorem version).
In particular, $\cKA$ is well-defined as an $L^2(\R_+\times\R_+)$ kernel
and the eigenfunction expansion converges in $L^2$ and pointwise for
$\xi,\eta>0$.
No Tracy--Widom ``soft edge'' identification is assumed;
only the spectral theorem for $\cA$ is used.
\end{remark}

\begin{lemma}[Main Lemma]
\label{lem:main}
For fixed $\xi,\eta\geq 0$:
$\displaystyle\lim_{c\to\infty}\cK(\xi,\eta) = \cKA(\xi,\eta)$.
\end{lemma}

\noindent Proof in Sections~\ref{sec:I2b}--\ref{sec:proof}.

%% ============================================================
\section{(I2b): Two-Regime Kernel Domination}
\label{sec:I2b}
%% ============================================================

\begin{lemma}[Two-regime domination --- corrected]
\label{lem:I2b}
For fixed $M>0$ and $K\geq 1$, there exist $C_{M,K},C'>0$
uniform in $c\geq c_0(M,K)$:\footnote{The constant $C_{M,K}$ depends on $M$ and $K$
through the number of turning points and the size of the Airy window;
it is independent of $c$.}
\begin{enumerate}[nosep,label=(\roman*)]
  \item \emph{Airy window, $\xi,\eta\in[0,M]$:}
    \begin{equation}\label{eq:airybound}
      |\cK(\xi,\eta)| \leq C_{M,K}(1+\xi)^{-1/4}(1+\eta)^{-1/4}.
    \end{equation}
  \item \emph{Far field, $\xi>M$ or $\eta>M$:}
    \begin{equation}\label{eq:farbound}
      |\cK(\xi,\eta)| \leq C'\,e^{-\frac{4}{3}(\xi\wedge\eta-M)_+^{3/2}}.
    \end{equation}
\end{enumerate}
Together: $h(\xi,\eta) := C_{M,K}(1+\xi)^{-1/4}(1+\eta)^{-1/4}
+ C'e^{-\frac{4}{3}(\xi\wedge\eta-M)_+^{3/2}}$
is in $L^2(w\otimes w)$ for $w=(1+\xi)^{-\beta}$, $\beta>1/4$.
\end{lemma}

\begin{proof}
\emph{CD decomposition.}
ORX Proposition~5.2 (algebraic, no asymptotics):
$K_c(x,y) = \frac{\psi_{n^*}^{(c)}(x)\partial_y\psi_{n^*}^{(c)}(y)
  - \psi_{n^*}^{(c)}(y)\partial_x\psi_{n^*}^{(c)}(x)}{x-y}$.
Rescaled: $\cK(\xi,\eta) = \frac{\tilde\psi_0(\xi)\partial_\eta\tilde\psi_0(\eta)
- \tilde\psi_0(\eta)\partial_\xi\tilde\psi_0(\xi)}{\xi-\eta}$
(plus terms from $|j|\leq K$ in the edge window;
see below).

\emph{Regime (i): $\xi,\eta\in[0,M]$.}
By~\eqref{eq:tp2}, all turning points $\xi_{n^*+j}^* = O(c^{-1/3})$
lie in $[0,1]$ for $c$ large.
On the compact set $[0,M]$,
Olver~\cite{Olver1997} Ch.~11 \S11.3 gives the oscillatory WKB amplitude:
for any smooth potential with simple turning points in $[0,M]$,
the solution is bounded by
$|\tilde\psi_j(\xi)|\leq C_0(1+|\xi-\xi_{n^*+j}^*|)^{-1/4}$
\emph{uniformly in $c$ and in $|j|\leq K$},
where $C_0$ depends only on $M$, $K$,
and the (bounded) potential on $[0,M]$.
The uniformity in $c$ comes from the fact that
the potential $V_j^{(c)}(\xi) := \xi - \xi_{n^*+j}^*$
(after rescaling) is bounded on $[0,M]$ with bound $M$,
independent of $c$ (since $\xi_{n^*+j}^*=O(c^{-1/3})\to 0$,
not varying the potential significantly on $[0,M]$).
Since $(1+|\xi-\xi_{n^*+j}^*|)^{-1/4}\leq (1+\xi)^{-1/4}+C$
and $\xi_{n^*+j}^*\to 0$, we get~\eqref{eq:airybound}
with $C_{M,K} = 2KC_0^2$.

\emph{Why no Airy-form input:}
the Olver oscillatory amplitude $(1+|\xi-\xi^*|)^{-1/4}$
is the WKB amplitude of the oscillatory region;
it does \emph{not} require the Airy-function form of the solution
near the turning point (which is the hard content of Prop.~\ref{prop:PR}).
The $(1+\xi)^{-1/4}$ bound holds from the WKB energy conservation:
$|\tilde\psi_j|^2|\tilde\psi_j'| \sim \mathrm{const}$
(standard WKB).

\emph{Regime (ii): $\xi>M$.}
For $\xi>M$ and all turning points in $[0,1]$,
the solution $\tilde\psi_j$ is in the exponentially decaying
region (beyond all turning points).
Olver subdominant solution:
$|\tilde\psi_j(\xi)|\leq C\exp(-\frac{2}{3}(\xi-\xi_{n^*+j}^*)^{3/2})
\leq Ce^{-\frac{2}{3}(\xi-1)^{3/2}}$
for $\xi>M>1$.
Product in CD: $(ii)$ follows with $C'$ depending on $K$ and $M$.
\end{proof}

\section{Uniformity and Residual Control}
\label{sec:uniformity}
%% ============================================================

\begin{lemma}[Turning-point clustering --- corrected]
\label{lem:tpshift}
For $|j|\leq K$ and $c\geq 1$:
$\xi_{n^*+j}^* = -\frac{\pi}{2}jc^{-1/3} + O(c^{-2/3})$,
so $|\xi_{n^*+j}^*|\leq \frac{\pi K}{2}c^{-1/3} + O(c^{-2/3})\to 0$.
All $K$ turning points lie in $(-1,1)$ for $c\geq c_0(K)$.
The inter-turning-point spacing is $\pi c^{-1/3}/2$.
\end{lemma}

\begin{proof}
Direct from~\eqref{eq:tp2}: $\xi_{n^*+j}^* = -(\pi/2)s_{n^*+j}+O(c^{-2/3})$
and $s_{n^*+j}=jc^{-1/3}+O(c^{-2/3})$.
\end{proof}

\begin{remark}[Correct interpretation of the Airy window]
\label{rem:airywindow}
The fixed window $\xi\in[0,M]$ with $M\geq 2$ contains \emph{all}
$K$ turning points for large $c$: they cluster at $\xi=0^+$ with spacing
$\pi c^{-1/3}/2\to 0$.
In the $c\to\infty$ limit, all turning points coalesce to $0$,
which is consistent with the Airy zeros $a_k$ being $O(1)$:
the $k$-th Airy zero corresponds to the $k$-th turning point
shifting to $0$ at rate $c^{-1/3}$.
\end{remark}

\begin{lemma}[ODE residual]
\label{lem:residunif}
For $|j|\leq K$ and $c\geq c_0(K)$:
$\|r_j^{(c)}\|_{L^2(\R_+)}\leq C_0(1+|j|)\,c^{-1/3}$,
with $C_0$ independent of $j,K,c$.
\end{lemma}

\begin{proof}
Paper~XVI Theorem~2.1: $\|R_c\|_{\mathrm{op}}=O(c^{-1/3})$.
$|a_{j,c}-a_j|=O(|j|c^{-1/3})$ (Paper~XVII Lem.~2.1).
$\|r_j^{(c)}\|_{L^2}\leq \|R_c\|_{\mathrm{op}}+|a_{j,c}-a_j|
= O((1+|j|)c^{-1/3})$.
\end{proof}

%% ============================================================
\section{Proof of the Main Lemma}
\label{sec:proof}
%% ============================================================

\textbf{Step 1: Edge/bulk split.}
Fermi tail (Paper~XVII Lem.~2.1):
$|\cK(\xi,\eta)-\cK^{(K)}(\xi,\eta)|\leq h(\xi,\eta)e^{-\pi R}$,
uniform in $c$.

\medskip
\textbf{Step 2: Spectral projector convergence.}

\begin{proposition}[Finite-window Gram and projector convergence]
\label{prop:PR}
Fix $K\geq 1$. Let $G^{(c)}_{jk}:=\langle\tilde\psi_j,u_k\rangle_{L^2(\R_+)}$.
Then $\|G^{(c)}-I_K\|_{\mathrm{op}}=O_K(c^{-1/3})$
and $\|\PcK-\PK\|_{\mathrm{op}}=O_K(c^{-1/3})$.
\end{proposition}

\begin{proof}
\emph{(A) ODE.}
$(-\partial_\xi^2+\xi)\tilde\psi_j = -a_{j,c}\tilde\psi_j + r_j^{(c)}$
(Paper~XVI Thm.~2.1),
where $a_{j,c}\to a_j$ in increasing order:
$a_{1,c}<a_{2,c}<\cdots<a_{K,c}$ converging to $a_1<\cdots<a_K$
(Airy zeros, increasing, by Paper~XVII).

\emph{(B) Residual.}
$\|r_j^{(c)}\|_{L^2}\leq C_0(1+K)c^{-1/3}$ (Lemma~\ref{lem:residunif}).

\emph{(C) Airy gap.}
$\delta_K:=\min_{k\leq K}(a_{k+1}-a_k)>0$, fixed for fixed $K$.

\emph{(D) Davis--Kahan.}
Since $a_{j,c}\to a_j$ in increasing order,
the Davis--Kahan theorem~\cite{DavisKahan1970}
identifies $\tilde\psi_j$ with $u_j$ (same index $j$,
by matching in eigenvalue order):\footnote{The index identification
is canonical: the $j$-th approximate eigenvalue $a_{j,c}$
is $O(c^{-1/3})$-close to the $j$-th Airy zero $a_j$
and separated from $a_{j\pm 1}$ by gap $\delta_K>0$.
Davis--Kahan gives the subspace closeness;
the eigenvalue ordering gives the index matching.}
$\|\tilde\psi_j-u_j\|_{L^2}\leq\|r_j^{(c)}\|_{L^2}/\delta_K
= O(K^{4/3}c^{-1/3})$.

\emph{(E) Gram.}
$G^{(c)}_{jk} = \delta_{jk}+O_K(c^{-1/3})$,
so $\|G^{(c)}-I_K\|_{\mathrm{op}}=O_K(c^{-1/3})$.

\emph{(F) Projector norm.}
In the finite-dimensional setting ($K$ fixed):
$\|\PcK-\PK\|_{\mathrm{op}}
\leq 2\max_j\|\tilde\psi_j-u_j\|_{L^2}
= O_K(c^{-1/3})$.
(Standard: $\|P-Q\|_{\mathrm{op}}\leq\|\text{basis diff}\|$ for
finite-rank projectors onto $\varepsilon$-close ONBs.)
\end{proof}

\begin{remark}[Projector convergence in a single space]
\label{rem:projconv}
Both $\PcK$ and $\PK$ act on $L^2(\R_+)$ via the identification map $\Uc$
(Definition~\ref{def:kernels}).
Proposition~\ref{prop:PR} gives convergence in \emph{operator norm}
(stronger than strong), which is valid because $K$ is finite.
The strong convergence $\chi_{(-\infty,0]}(\cA_c)\to\chi_{(-\infty,0]}(\cA)$
(all modes) follows by $K\to\infty$ with the Fermi tail from Step~5.
\end{remark}

\begin{lemma}[$H^2\to C^0$]
\label{lem:H2}
For fixed $K,M$:
$\|\tilde\psi_j-u_j\|_{C^0([0,M])} = O_K(c^{-1/3})$.
\end{lemma}

\begin{proof}
$e_j:=\tilde\psi_j-u_j$ satisfies
$-e_j''+\xi e_j = -a_{j,c}e_j+(a_{j,c}+a_j)u_j+r_j^{(c)}$.
$\|e_j''\|_{L^2([0,M+1])}\leq(M+1)\|e_j\|_{L^2}+O_K(c^{-1/3})
= O_K(c^{-1/3})$.
$H^2([0,M])\hookrightarrow C^0([0,M])$:
$\|e_j\|_{C^0([0,M])}=O_K(c^{-1/3})$.
\end{proof}

\textbf{Step 3: Eigenfunction products.}
For $\xi,\eta\in[0,M]$:
\[
  \cK^{(K)}(\xi,\eta)
  = \sum_{|j|\leq K}\lam{c}{n^*+j}
  (u_j(\xi)+O_K(c^{-1/3}))(u_j(\eta)+O_K(c^{-1/3}))
  \to\sum_{k=1}^K u_k(\xi)u_k(\eta)
  = \cKA^{(K)}(\xi,\eta),
\]
using $\lam{c}{n^*+j}\to\mathbf{1}_{j<0}+\tfrac12\mathbf{1}_{j=0}$
and Lemma~\ref{lem:H2}.

\textbf{Step 4: $K\to\infty$.}
$|\cK(\xi,\eta)-\cK^{(K)}|\leq Ce^{-\pi K/c^{1/3}}\to 0$;
$\cKA^{(K)}\to\cKA$ (Remark~\ref{rem:airykernel}).
\hfill$\square$
