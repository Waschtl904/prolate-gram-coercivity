\begin{abstract}
This paper proves Lemma~\ref{lem:main}:
$\cK(\xi,\eta)\to\cKA(\xi,\eta)$ pointwise,
closing BR3 with Paper~XVII.

\medskip
\textbf{Central addition in v6.}
Proposition~\ref{prop:normresolvent} establishes
\begin{equation}\label{eq:nrc-abstract}
  \bigl\|(\Uc\Lc\Uc^{-1}-z)^{-1} - (\cA-z)^{-1}\bigr\|_{L^2(\R_+)\to L^2(\R_+)}
  = O(c^{-1/3}), \quad z\notin\R,
\end{equation}
i.e.\ norm-resolvent convergence of the rescaled PSWF operator to the
Airy operator $\cA = -\partial_\xi^2+\xi$ on $L^2(\R_+)$.
All spectral consequences (eigenvalue convergence, eigenfunction
convergence, projector convergence, index identification)
follow as standard corollaries of~\eqref{eq:nrc-abstract}.
\end{abstract}

\section{Setup}
\label{sec:setup}
%% ============================================================

\begin{definition}[Rescaling and operators]
\label{def:setup}
\emph{Airy coordinate:} $\xi = c^{2/3}(1-x)\geq 0$.\\
\emph{Identification map:}
$(\Uc f)(\xi) := c^{1/3}f(1-c^{-2/3}\xi)$;
$\Uc: L^2([-1,1])\to L^2(\R_+)$ is an approximate isometry with
$\|\Uc\| = 1+O(e^{-\pi c})$.\\
\emph{Rescaled PSWF operator:}
$\Lc := c^{-2/3}(\Uc\mathcal{L}_c\Uc^{-1})$
where $\mathcal{L}_c = -(1-x^2)\partial_x^2+2x\partial_x+c^2x^2$
is the PSWF Sturm--Liouville operator on $L^2([-1,1])$.\\
\emph{Airy operator:} $\cA = -\partial_\xi^2+\xi$ on $L^2(\R_+)$,
Dirichlet BC at $\xi=0$, domain $H^2(\R_+)\cap H^1_0(\R_+)$.\\
\emph{Kernels and eigenfunctions:}
$\tilde\psi_j(\xi):=\Uc\psi_{n^*+j}^{(c)}(\xi)$;
$\cK(\xi,\eta):=c^{-2/3}K_c(1-c^{-2/3}\xi,1-c^{-2/3}\eta)$;
$u_k\in L^2(\R_+)$: $L^2$-normalised eigenfunctions of $\cA$,
$\cA u_k = a_k u_k$, $0<a_1<a_2<\cdots$;
$\cKA(\xi,\eta):=\sum_{k\geq 1}u_k(\xi)u_k(\eta)$
(spectral theorem for $\cA$, see Remark~\ref{rem:airykernel}).
\end{definition}

\begin{remark}[Airy kernel as spectral theorem]
\label{rem:airykernel}
$\cA = -\partial_\xi^2+\xi$ on $L^2(\R_+)$ with Dirichlet BC
is a positive self-adjoint operator with compact resolvent
(Weyl's law: $a_k\sim(\frac{3\pi}{2}(k-\frac14))^{2/3}$).
The spectral theorem gives
$\sum_{k=1}^\infty u_k\otimes u_k = \mathrm{Id}_{L^2(\R_+)}$ (Parseval),
so $\cKA(\xi,\eta)=\sum_k u_k(\xi)u_k(\eta)$
is the kernel of the identity on $L^2(\R_+)$
(up to the spectral projection onto $[0,R]$;
the full Airy kernel in RMT sense is $P_{(-\infty,0]}(\cA)$,
but our $\cKA$ here is the sum over all positive eigenvalues,
consistent with Lemma~\ref{lem:main}).
The equality with the integral form
$\int_0^\infty\Ai(\xi+t)\Ai(\eta+t)\,dt$
holds by completeness of Airy functions~\cite{Olver1997} Ch.~11.
No Tracy--Widom identification is assumed.
\end{remark}

\begin{lemma}[Main Lemma]
\label{lem:main}
For fixed $\xi,\eta\geq 0$:
$\displaystyle\lim_{c\to\infty}\cK(\xi,\eta) = \cKA(\xi,\eta)$.
\end{lemma}

Proof in Sections~\ref{sec:nrc}--\ref{sec:proof}.

\section{(I2b): Two-Regime Domination}
\label{sec:I2b}
%% ============================================================

\begin{lemma}[Kernel domination]
\label{lem:I2b}
For fixed $M>0$, $K\geq 1$, there exist $C_{M,K},C'>0$
uniform in $c\geq c_0(M,K)$:
\begin{enumerate}[nosep,label=(\roman*)]
  \item $\xi,\eta\in[0,M]$:
    $|\cK(\xi,\eta)|\leq C_{M,K}(1+\xi)^{-1/4}(1+\eta)^{-1/4}$.
  \item $\xi>M$ or $\eta>M$:
    $|\cK(\xi,\eta)|\leq C'e^{-\frac{4}{3}(\xi\wedge\eta-M)_+^{3/2}}$.
\end{enumerate}
\end{lemma}

\begin{proof}
\emph{Christoffel--Darboux} (ORX Prop.~5.2, purely algebraic):
$K_c(x,y) = \frac{\psi_{n^*}^{(c)}(x)\partial_y\psi_{n^*}^{(c)}(y)
- \psi_{n^*}^{(c)}(y)\partial_x\psi_{n^*}^{(c)}(x)}{x-y}$.

\emph{WKB oscillatory amplitude on $[0,M]$.}
The rescaled PSWF operator $\Lc$ has potential
$V_c(\xi) := -c^{-2/3}\lam{c}{n^*}+\xi+O(c^{-2/3})$
(from Def.~\ref{def:setup} and Paper~XVI Def.~1.1).
On the compact set $[0,M]$, $V_c\to\xi$ uniformly in $c$,
so the classical WKB amplitude of an oscillatory solution
$(-\partial_\xi^2+V_c(\xi))\tilde\psi = 0$ satisfies
$|\tilde\psi_j(\xi)|\leq C_{M,K}(1+\xi)^{-1/4}$
uniformly in $c$ and $|j|\leq K$.
This uses \emph{only} the oscillatory WKB amplitude
(energy conservation: $|\tilde\psi|^2|\partial_\xi V_c|/V_c^{3/2}\sim\text{const}$;
Olver~\cite{Olver1997} Ch.~11 \S11.3 for the uniform version).
Crucially, no Airy-function form is assumed for $\tilde\psi_j$.
Substituting into CD and rescaling gives~(i).

\emph{Exponential decay for $\xi>M$.}
For $\xi\gg 1$: $V_c(\xi)\approx\xi>0$,
so $\tilde\psi_j$ is in the exponentially decaying (``right of turning point'') regime.
Olver subdominant solution gives $|\tilde\psi_j(\xi)|\leq Ce^{-\frac{2}{3}(\xi-1)^{3/2}}$,
giving~(ii).
\end{proof}

%% ============================================================
\section{Norm-Resolvent Convergence}
\label{sec:nrc}
%% ============================================================

This section provides the central operator-theoretic foundation.
All subsequent spectral results are corollaries.

\begin{proposition}[Norm-resolvent convergence]
\label{prop:normresolvent}
For $z\in\C\setminus\R$ and $c\geq c_0$:
\begin{equation}\label{eq:nrc}
  \bigl\|(\Lc - z)^{-1} - (\cA - z)^{-1}\bigr\|_{L^2(\R_+)\to L^2(\R_+)}
  \leq \frac{C\,c^{-1/3}}{(\mathrm{Im}\,z)^2},
\end{equation}
where $\Lc := \Uc\mathcal{L}_c^{\mathrm{resc}}\Uc^{-1}$ acts on $L^2(\R_+)$
and $C$ is independent of $c$ and $z$.
\end{proposition}

\begin{proof}
\emph{Step 1: Intertwining identity.}
Paper~XVI Theorem~2.1 establishes the intertwining:
\begin{equation}\label{eq:intertwine}
  \Lc = \cA + c^{-1/3}R_c,
  \qquad \|R_c\|_{L^2(\R_+)\to L^2(\R_+)} \leq C_0,
\end{equation}
where $R_c$ is the remainder operator from the conjugation
$\Uc\mathcal{L}_c\Uc^{-1}-\cA$.
Explicitly: in the Airy coordinate, the PSWF potential is
$V_c(\xi) = \xi - c^{-1/3}W_c(\xi)$
where $W_c(\xi) = O(1)$ uniformly on $[0,M]$
(from Taylor expansion of $(1-(1-c^{-2/3}\xi)^2)^{1/2}$
in Paper~XVI Def.~1.1),
so $\Lc = -\partial_\xi^2+\xi - c^{-1/3}W_c = \cA - c^{-1/3}W_c =: \cA + c^{-1/3}R_c$
with $R_c = -W_c$ and $\|R_c\|_{\infty}\leq C_0$ on $[0,M]$,
extended to $L^2(\R_+)$ by the exponential decay of the PSWFs.

\emph{Step 2: Resolvent identity.}
The second resolvent identity:
\begin{equation}\label{eq:resolvent-id}
  (\Lc-z)^{-1} - (\cA-z)^{-1}
  = -(\Lc-z)^{-1}\cdot c^{-1/3}R_c\cdot(\cA-z)^{-1}.
\end{equation}

\emph{Step 3: Norm estimate.}
Taking $L^2\to L^2$ norms:
\begin{align}
  \|(\Lc-z)^{-1}-(\cA-z)^{-1}\|
  &\leq \|(\Lc-z)^{-1}\|\cdot c^{-1/3}\|R_c\|\cdot\|(\cA-z)^{-1}\|\notag\\
  &\leq \frac{1}{|\mathrm{Im}\,z|}\cdot C_0 c^{-1/3}\cdot\frac{1}{|\mathrm{Im}\,z|}
  = \frac{C_0\,c^{-1/3}}{(\mathrm{Im}\,z)^2},\label{eq:nrcbound}
\end{align}
using the self-adjointness bounds
$\|(\Lc-z)^{-1}\|\leq|\mathrm{Im}\,z|^{-1}$
and $\|(\cA-z)^{-1}\|\leq|\mathrm{Im}\,z|^{-1}$
(both operators are self-adjoint on $L^2(\R_+)$).
\end{proof}

\begin{remark}[Self-adjointness of $\Lc$]
\label{rem:sa}
$\Lc = \cA + c^{-1/3}R_c$ is self-adjoint on the same domain as $\cA$
(i.e.\ $H^2(\R_+)\cap H^1_0(\R_+)$) because $R_c = -W_c(\xi)$
is a bounded symmetric multiplication operator.
Dirichlet BC at $\xi=0$ is preserved by $\Uc$:
$\tilde\psi_j(0) = c^{1/3}\psi_{n^*+j}^{(c)}(1)$;
the Dirichlet condition $\psi_{n^*+j}^{(c)}(1)=0$ holds by
PSWF parity and boundary behaviour (Paper~XVI Rem.~1.2).
\end{remark}

\begin{corollary}[Spectral convergence]
\label{cor:spectral}
From Proposition~\ref{prop:normresolvent} and standard spectral theory
(Kato~\cite{Kato1966} Ch.~IV Thm.~3.16;\footnote{%
  Norm-resolvent convergence $\Rightarrow$ eigenvalue convergence
  and eigenprojector convergence for isolated eigenvalues.}
or Reed--Simon~\cite{ReedSimon4} Thm.~VIII.23):
\begin{enumerate}[nosep,label=(\roman*)]
  \item \emph{Eigenvalue convergence:}
    $a_{j,c}\to a_j$ for each fixed $j$,
    in increasing order, at rate $O(c^{-1/3})$.
  \item \emph{Eigenprojector convergence:}
    $\|P_{j,c}-P_j\|_{L^2\to L^2} = O_j(c^{-1/3})$
    (operator norm, for fixed $j$).
  \item \emph{$K$-window convergence:}
    $\|\PcK-\PK\|_{L^2\to L^2} = O_K(c^{-1/3})$
    (Gram matrix $G^{(c)}\to I_K$ follows).
  \item \emph{Index identification:}
    the $j$-th eigenvalue $a_{j,c}$ is $O(c^{-1/3})$-close to
    $a_j$ and separated from $a_{j\pm 1}$ by $\delta_K>0$ (fixed),
    so the identification $\tilde\psi_j\leftrightarrow u_j$ is canonical.
\end{enumerate}
\end{corollary}

\begin{remark}[Davis--Kahan as secondary tool]
\label{rem:dk}
Corollary~\ref{cor:spectral}(ii)--(iii) can also be derived
via Davis--Kahan once~\eqref{eq:intertwine} is established;
see Proposition~\ref{prop:gram} below for the explicit Gram estimate.
However, the primary foundation is now
Proposition~\ref{prop:normresolvent},
which anchors the spectral convergence globally
(not just mode-by-mode).
\end{remark}

\section{Proof of the Main Lemma}
\label{sec:proof}
%% ============================================================

\begin{proposition}[Gram convergence]
\label{prop:gram}
For fixed $K$, $G^{(c)}_{jk}:=\langle\tilde\psi_j,u_k\rangle$:
$\|G^{(c)}-I_K\|_{\mathrm{op}}=O_K(c^{-1/3})$,
$\|\PcK-\PK\|_{\mathrm{op}}=O_K(c^{-1/3})$.
\end{proposition}

\begin{proof}
Corollary~\ref{cor:spectral}(iii), or explicitly:
$\|\tilde\psi_j-u_j\|_{L^2} = \|P_{j,c}\phi - P_j\phi\|/\|\phi\|
\leq\|P_{j,c}-P_j\|_{\mathrm{op}} = O_j(c^{-1/3})$
(where $\phi$ is any normalised vector with $P_j\phi=\phi$).
$G^{(c)}_{jk}=\delta_{jk}+O_K(c^{-1/3})$ follows.
\end{proof}

\begin{lemma}[$H^2\to C^0$]
\label{lem:H2}
For fixed $K,M$:
$\|\tilde\psi_j-u_j\|_{C^0([0,M])}=O_K(c^{-1/3})$.
\end{lemma}

\begin{proof}
Let $e_j:=\tilde\psi_j-u_j$.
From~\eqref{eq:intertwine}:
$\cA e_j = a_j e_j - (a_{j,c}-a_j)\tilde\psi_j - c^{-1/3}R_c\tilde\psi_j$,
so $\|\cA e_j\|_{L^2}\leq |a_{j,c}-a_j|+C_0 c^{-1/3} = O_j(c^{-1/3})$.
Since $\cA = -\partial_\xi^2+\xi$:
$\|e_j''\|_{L^2([0,M])}\leq\|\cA e_j\|_{L^2}+(M+1)\|e_j\|_{L^2}
= O_K(c^{-1/3})$.
$H^2([0,M])\hookrightarrow C^0([0,M])$:
$\|e_j\|_{C^0([0,M])}\leq C_M\|e_j\|_{H^2}=O_K(c^{-1/3})$.
\end{proof}

\textbf{Step 1: Edge/bulk split.}
$|\cK(\xi,\eta)-\cK^{(K)}(\xi,\eta)|\leq h(\xi,\eta)e^{-\pi R}$
(Fermi tail, Paper~XVII Lem.~2.1, domination by Lemma~\ref{lem:I2b}).

\textbf{Step 2: Projector convergence.}
Corollary~\ref{cor:spectral}(iii) and the eigenvalue weights:
$\lam{c}{n^*+j}\to\mathbf{1}_{j<0}+\frac12\mathbf{1}_{j=0}$
(Paper~XVII Lem.~2.1).
Weighted sum $\sum_{|j|\leq K}\lam{c}{n^*+j}|\tilde\psi_j\rangle\langle\tilde\psi_j|
\to\PK$ in operator norm.

\textbf{Step 3: Pointwise eigenfunction products.}
Lemma~\ref{lem:H2}:
$\cK^{(K)}(\xi,\eta)\to\cKA^{(K)}(\xi,\eta)$ for fixed $\xi,\eta\in[0,M]$.

\textbf{Step 4: $K\to\infty$.}
$|\cK-\cK^{(K)}|\leq Ce^{-\pi K/c^{1/3}}\to 0$;
$\cKA^{(K)}\to\cKA$ (Remark~\ref{rem:airykernel}).
\hfill$\square$
