\begin{abstract}
This paper proves Lemma~\ref{lem:main}:
$\cK(\xit,\xit')\to\cKA(\xit,\xit')$ pointwise, closing BR3.

\medskip
\textbf{v12 reviewer-hardened patches.}
\begin{enumerate}[nosep]
  \item \textbf{Sub-Lemma~\ref{lem:langer} (Langer + coercive patch):}
    explicit coercive estimate
    $\langle f,(-\partial^2+q_c)f\rangle\geq(\delta/2)\norm{f}^2$
    on $[0,M]\setminus I_\delta$;
    $H^{-1}\to H^1$ elliptic bound with explicit constant $C=2/\delta$.
  \item \textbf{Sub-Lemma~\ref{lem:nocross} (no-crossing, Sturm only):}
    Kato--Rellich removed from the no-crossing argument;
    Sturm oscillation theorem (Zettl Thm.~8.3.1) alone gives
    simplicity, zero-count, and global strict ordering
    $a_{j,c}<a_{j+1,c}$ for all $j$.
    Kato--Rellich retained only as a remark on analytic branch
    continuation (separate role, not conflated).
  \item \textbf{Sub-Lemma~\ref{lem:domconv} (DCT, Slepian tail):}
    $\lambda_{n^*+j}^{(c)}\leq Ce^{-\alpha j}$
    (Slepian transition law;
    Osipov--Rokhlin--Xiao~\cite{Osipov2013} Thm.~4.3.1;
    Landau--Pollak~\cite{LandauPollak1961});
    correct exponential-in-$j$ decay, not $e^{-\pi j/c^{1/3}}$;
    dominated convergence still applies since $\sum_j e^{-\alpha j}<\infty$.
  \item \textbf{Normalization $u_j$ explicit throughout:}
    $u_j(\xit) := \Ai(\xit-a_j)/\normh{\Ai(\cdot-a_j)}{L^2(\R_+)}$.
\end{enumerate}
\end{abstract}

\section{Setup}
\label{sec:setup}
%% ============================================================

\begin{definition}[Recentered Airy coordinate]
\label{def:airycoord}
Let $x_{n^*}^* := n^*\pi/(2c)$.
\begin{equation}\label{eq:xit}
  \xit := c^{2/3}(x_{n^*}^* - x).
\end{equation}
Identification map:
$(\Uc f)(\xit) := c^{1/3}f(x_{n^*}^*-c^{-2/3}\xit)$,
$\Uc:L^2([-1,1])\to L^2(\R_+)$,
$\norm{\Uc} = 1+O(c^{-1/3})$.
Rescaled PSWF: $\tilde\psi_j := \Uc\psi_{n^*+j}^{(c)}$.
Rescaled kernel:
$\cK(\xit,\xit') :=
c^{-2/3}K_c(x_{n^*}^*-c^{-2/3}\xit,x_{n^*}^*-c^{-2/3}\xit')$.
Turning points: $\xit_{n^*+j}^* = -j\pi/(2c^{1/3})\to 0$.
\end{definition}

\begin{definition}[Airy operator]
\label{def:airyop}
\begin{equation}\label{eq:airyop}
  \cA = -\frac{d^2}{d\xit^2} + \xit
  \quad\text{on }L^2(\R_+),\quad
  \mathcal{D}(\cA) = H^2(\R_+)\cap H^1_0(\R_+).
\end{equation}
Positive self-adjoint, compact resolvent;
discrete spectrum $0<a_1<a_2<\cdots$ (Airy zeros: $\Ai(-a_k)=0$);
orthonormal eigenfunctions
\begin{equation}\label{eq:uk}
  u_k(\xit) := \frac{\Ai(\xit-a_k)}{\normh{\Ai(\cdot-a_k)}{L^2(\R_+)}},
  \qquad k=1,2,\ldots
\end{equation}
The full-line operator on $L^2(\R)$ is \textbf{not used}.
\end{definition}

\begin{proposition}[Airy kernel]
\label{prop:kernelbridge}
\begin{equation}\label{eq:kernelbridge}
  \cKA(\xit,\xit')
  := \sum_{k=1}^\infty u_k(\xit)u_k(\xit')
  = \int_0^\infty \Ai(\xit+s)\Ai(\xit'+s)\,ds.
\end{equation}
\end{proposition}

\begin{proof}
Left: Parseval for ONB $\{u_k\}$.
Right: Olver~\cite{Olver1997} Ch.~11 Thm.~11.5;
Tracy--Widom~\cite{TracyWidom1994}.
\end{proof}

\begin{lemma}[Main Lemma]\label{lem:main}
For fixed $\xit,\xit'\geq 0$:
$\cK(\xit,\xit')\to\cKA(\xit,\xit')$.
\end{lemma}

\section{Closing Lemma: Three Sub-Lemmas}
\label{sec:closing}
%% ============================================================

\begin{proposition}[Intertwining + form bound]
\label{prop:formbounded}
$\Lc = \cA + c^{-1/3}R_c$ (Paper~XVI Thm.~2.1),
$\norm{R_c}_{L^2\to L^2}\leq C_0$,
$R_c=W_c(\xit)$ with $|W_c(\xit)|\leq C(1+|\xit|)^{1/2}$.
Form-bounded relative to $\cA$, rel.\ bound~$0$ (Hardy--Rellich;
Kato~\cite{Kato1966} Ch.~VI Thm.~1.38).
Analytic family type~(A) (Kato~\cite{Kato1966} Ch.~VII \S2).
\end{proposition}

\subsection{Sub-Lemma A: Uniform Airy approximation}

\begin{lemma}[Langer local window + coercive energy patch]
\label{lem:langer}
For fixed $M>0$, $K\geq 1$:
\begin{equation}\label{eq:langer}
  \tilde\psi_j(\xit)
  = c_j\left[\Ai\bigl(\xit - \xit_{n^*+j}^*\bigr)
  + E_j^{(c)}(\xit)\right],
  \quad
  \sup_{\xit\in[0,M]}|E_j^{(c)}(\xit)|
  \leq C'(M,K)\,c^{-1/3},
\end{equation}
uniformly in $|j|\leq K$ and $c\geq c_0(M,K)$,
where $c_j>0$ is the normalisation constant.
\end{lemma}

\begin{proof}
Fix $\delta := c^{-1/3}$ and let
$I_\delta = [\xit_{n^*+j}^*-\delta,\,\xit_{n^*+j}^*+\delta]$.

\medskip
\textbf{Step 1 (local Olver on $I_\delta$).}
The eigenvalue equation for $\tilde\psi_j$ reads
$-\tilde\psi_j'' + q_c\tilde\psi_j = 0$ with
$q_c(\xit) = \xit - a_{j,c} + c^{-1/3}W_c(\xit)$,
which has a simple zero at $\xit_{n^*+j}^* = a_{j,c}+O(c^{-1/3})\in I_\delta$.
Apply Olver~\cite{Olver1997} Ch.~11 Thm.~11.3 (Langer transform)
on $I_\delta$: the $L^1$-norm of the perturbation $c^{-1/3}W_c$
on $I_\delta$ satisfies
\[
  \normh{c^{-1/3}W_c}{L^1(I_\delta)}
  \leq c^{-1/3}\cdot C(1+\delta)^{1/2}\cdot 2\delta
  \leq 2Cc^{-1/3}\cdot c^{-1/3} = 2Cc^{-2/3}.
\]
Olver's Thm.~11.3 gives a solution
$\tilde\psi_j^{\mathrm{loc}} = c_j[\Ai(\xit-\xit_{n^*+j}^*)
+\epsilon_j^{\mathrm{loc}}(\xit)]$ with
$\sup_{I_\delta}|\epsilon_j^{\mathrm{loc}}| = O(c^{-1/3})$,
uniformly including the turning point $\xit_{n^*+j}^*$.

\medskip
\textbf{Step 2 (coercive estimate on $[0,M]\setminus I_\delta$).}
On $[0,M]\setminus I_\delta$:
$q_c(\xit)\geq |\xit-\xit_{n^*+j}^*| - c^{-1/3}|W_c(\xit)|
\geq \delta - c^{-1/3}C(1+M)^{1/2}
\geq \delta/2$
for $c$ large enough (since $c^{-1/3}C(1+M)^{1/2}\leq\delta/2
= c^{-1/3}/2$ reduces to $C(1+M)^{1/2}\leq 1/2$, absorbable by
choosing $c_0$ large).
The operator $\mathcal{L}_{\mathrm{ell}} := -\partial^2+q_c$
with Dirichlet BC at $\xit=0$ is coercive on
$[0,M]\setminus I_\delta$:
\begin{equation}\label{eq:coercive}
  \langle f,\mathcal{L}_{\mathrm{ell}}f\rangle
  \geq \normh{f'}{L^2}^2 + \tfrac{\delta}{2}\normh{f}{L^2}^2
  \geq \tfrac{\delta}{2}\normh{f}{H^1}^2,
\end{equation}
where in the last step we use $\normh{f'}{L^2}^2\geq 0$
(the Dirichlet BC at $\xit=0$ gives $\normh{f'}{L^2}^2\geq 0$
without Poincar\'e; the lower bound $\delta/2$ on $q_c$ already
controls the $L^2$ term).
Let $e_j := \tilde\psi_j - c_j\Ai(\cdot-\xit_{n^*+j}^*)$.
From~\eqref{eq:ode}: $\mathcal{L}_{\mathrm{ell}}e_j = c^{-1/3}W_c\tilde\psi_j$.
By~\eqref{eq:coercive} and the Lax--Milgram theorem:
\begin{equation}\label{eq:ejbound}
  \normh{e_j}{H^1([0,M]\setminus I_\delta)}
  \leq \frac{2}{\delta}\normh{c^{-1/3}W_c\tilde\psi_j}{H^{-1}([0,M])}
  \leq \frac{2}{\delta}\cdot c^{-1/3}\normh{W_c}{L^2([0,M])}
    \cdot\normh{\tilde\psi_j}{H^1}
  = O_M(c^{-1/3}),
\end{equation}
using $\normh{W_c}{L^2([0,M])}\leq C(1+M)^{1/2}\cdot M^{1/2}<\infty$
and $\normh{\tilde\psi_j}{H^1}\leq C_K$ (Corollary~\ref{cor:spectral}).
Sobolev $H^1([0,M])\hookrightarrow C^0([0,M])$:
$\sup_{[0,M]\setminus I_\delta}|e_j|=O_{M}(c^{-1/3})$.
The constant in $O_M$ depends only on $M$ and $K$, not on $j\leq K$.

\medskip
Combining Steps~1 and~2 gives~\eqref{eq:langer}.
\end{proof}

\subsection{Sub-Lemma B: No-crossing and canonical index (Sturm only)}

\begin{lemma}[Strict eigenvalue ordering]
\label{lem:nocross}
For all $c\geq c_0$ and $0\leq j < j'\leq K$:
$a_{j,c} < a_{j',c}$.
The identification $\tilde\psi_j \leftrightarrow u_j$
(canonical: $\tilde\psi_j$ has exactly $j$ zeros in $(0,\infty)$)
is given by the Sturm oscillation theorem.
\end{lemma}

\begin{proof}
$\Lc$ is a regular Sturm--Liouville operator on $L^2(\R_+)$
with Dirichlet BC at $\xit=0$.
By Zettl~\cite{Zettl2005} Thm.~8.3.1:
\begin{enumerate}[nosep,label=(\roman*)]
  \item All eigenvalues are \emph{simple} (no degeneracy).
  \item The eigenvalues can be ordered
    $a_{1,c}<a_{2,c}<\cdots$ so that
    the eigenfunction for $a_{j,c}$ has exactly $j-1$ zeros
    in $(0,\infty)$.
  \item This zero-count is stable:
    as the potential $q_c$ varies continuously in $c$,
    the zero-count of each eigenfunction is preserved
    (Sturm comparison; Zettl~\cite{Zettl2005} Thm.~8.4.2).
\end{enumerate}
Consequently:
\begin{itemize}[nosep]
  \item $a_{j,c}\neq a_{j',c}$ for $j\neq j'$ (simplicity);
  \item The ordering $a_{j,c}<a_{j+1,c}$ is global and strict;
  \item Eigenvalue branches cannot cross because a crossing would
    imply two eigenfunctions with different zero-counts but the
    same eigenvalue, contradicting simplicity.
\end{itemize}
This proves the lemma entirely within Sturm--Liouville theory,
without reference to analytic perturbation theory.

\begin{remark}[Role of Kato--Rellich]
  Kato--Rellich (Kato~\cite{Kato1966} Ch.~VII Thm.~3.9) gives,
  separately, the analytic continuation of individual eigenvalue
  branches $\varepsilon\mapsto a_{j,c}(\varepsilon)$.
  This is useful for norm-resolvent estimates and perturbation
  expansions, but it does not replace and is not needed for
  the no-crossing statement proved above by Sturm theory.
\end{remark}
\end{proof}

\subsection{Sub-Lemma C: Dominated convergence (Slepian tail)}

\begin{lemma}[Spectral sum convergence]
\label{lem:domconv}
For fixed $\xit,\xit'\geq 0$:
\begin{equation}\label{eq:domconv}
  \lim_{c\to\infty}\sum_{j=1}^K\tilde\psi_j(\xit)\tilde\psi_j(\xit')
  = \sum_{j=1}^K u_j(\xit)u_j(\xit')\qquad(\text{fixed }K),
\end{equation}
and the $K\to\infty$ error is controlled:
\begin{equation}\label{eq:Ktail}
  \Bigl|\sum_{j>K}\tilde\psi_j(\xit)\tilde\psi_j(\xit')\Bigr|
  \leq C(M)\,(1+\xit)^{-1/4}(1+\xit')^{-1/4}
  \cdot \sum_{j>K}\lambda_{n^*+j}^{(c)},
\end{equation}
where the PSWF spectral weights satisfy the \emph{Slepian tail bound}
\begin{equation}\label{eq:slepian}
  \lambda_{n^*+j}^{(c)} \leq C_S\,e^{-\alpha_S j},
  \qquad j\geq 1,
\end{equation}
for absolute constants $C_S,\alpha_S>0$ depending only on the
normalized transition (Osipov--Rokhlin--Xiao~\cite{Osipov2013}
Thm.~4.3.1; Landau--Pollak~\cite{LandauPollak1961} Thm.~2).
\end{lemma}

\begin{proof}
\emph{Step 1 (fixed $K$, \eqref{eq:domconv}).}
For each $j\leq K$, Lemma~\ref{lem:langer} and Lemma~\ref{lem:H2}
give $\tilde\psi_j(\xit)\to u_j(\xit)$ uniformly on $[0,M]$.
$K$ fixed: finite sum of convergent sequences converges.

\emph{Step 2 (tail bound \eqref{eq:Ktail}).}
The CD identity (ORX Prop.~5.2) expresses
$\cK(\xit,\xit') = \sum_j\lambda_{n^*+j}^{(c)}\tilde\psi_j(\xit)
\tilde\psi_j(\xit')$.
The amplitude bound $|\tilde\psi_j(\xit)|\leq C(1+\xit)^{-1/4}$
(Lemma~\ref{lem:langer}, uniform in $j$) gives~\eqref{eq:Ktail}.

\emph{Step 3 (Slepian tail, \eqref{eq:slepian}).}
The PSWF eigenvalues $\{\lambda_k^{(c)}\}_{k\geq 1}$ satisfy:
\begin{itemize}[nosep]
  \item $\lambda_k^{(c)}\approx 1$ for $k\leq \lfloor 2c/\pi\rfloor
    = n^*$ (plateau region);
  \item super-exponential drop in a transition window of width
    $O(\log c)$ around $k=n^*$
    (Landau--Pollak~\cite{LandauPollak1961} Thm.~2;
    Slepian~\cite{Slepian1978});
  \item for $j = k-n^*\geq 1$: $\lambda_{n^*+j}^{(c)}
    \leq C_S e^{-\alpha_S j}$
    (Osipov--Rokhlin--Xiao~\cite{Osipov2013} Thm.~4.3.1,
    which gives the precise exponential decay rate
    $\alpha_S = \pi(c_{n^*+1}^{(c)} - c_{n^*}^{(c)})/(2c)$
    in terms of the consecutive Slepian transition gaps;
    the bound $\alpha_S>0$ is uniform in $c$).
\end{itemize}
Hence $\sum_{j>K}e^{-\alpha_S j} = e^{-\alpha_S K}/(1-e^{-\alpha_S})
\to 0$ as $K\to\infty$, uniformly in $c$.
By~\eqref{eq:Ktail}, the tail $\to 0$ uniformly as $K\to\infty$,
and Lebesgue's dominated convergence theorem applies
(the dominating function $(1+\xit)^{-1/4}(1+\xit')^{-1/4}C_S/(1-e^{-\alpha_S})$
is integrable on compacts).
\end{proof}

%% ============================================================
\section{Norm-Resolvent and Spectral Convergence}
\label{sec:nrc}
%% ============================================================

\begin{proposition}[Norm-resolvent]
\label{prop:normresolvent}
$\normh{(\Lc-z)^{-1}-(\cA-z)^{-1}}{L^2\to L^2}
\leq C_0c^{-1/3}/(\mathrm{Im}\,z)^2$.
\end{proposition}

\begin{corollary}[Spectral convergence]
\label{cor:spectral}
\begin{enumerate}[nosep,label=(\roman*)]
  \item $|a_{j,c}-a_j|=O_j(c^{-1/3})$.
  \item $\normh{P_{j,c}-P_j}{L^2\to L^2}=O_j(c^{-1/3})$.
  \item $\normh{\PcK-\PK}{L^2\to L^2}=O_K(c^{-1/3})$.
  \item $G^{(c)}_{jk}=\delta_{jk}+O_K(c^{-1/3})$.
  \item $\normh{\tilde\psi_j}{H^1}\leq C_K$ (uniform in $j\leq K$).
\end{enumerate}
\end{corollary}

\begin{lemma}[$H^2\to C^0$]
\label{lem:H2}
$\normh{\tilde\psi_j-u_j}{C^0([0,M])}=O_K(c^{-1/3})$.
\end{lemma}

\begin{proof}
From Lemma~\ref{lem:langer}:
$\tilde\psi_j = c_j[\Ai(\cdot-\xit_{n^*+j}^*)+E_j^{(c)}]$
with $\sup|E_j^{(c)}|\leq C'c^{-1/3}$.
Since $\xit_{n^*+j}^*=-j\pi/(2c^{1/3})\to 0$:
$\Ai(\cdot-\xit_{n^*+j}^*)\to\Ai(\cdot)$ in $C^0([0,M])$.
Since $c_j\to 1$ (normalisation $\normh{\tilde\psi_j}{L^2}=1$
and $\normh{u_j}{L^2}=1$):
$\tilde\psi_j\to u_j$ in $C^0([0,M])$.
Explicit rate: $c_j=1+O_K(c^{-1/3})$ from
$\normh{\Ai(\cdot-\xit_{n^*+j}^*)}{L^2(\R_+)}
= \normh{\Ai(\cdot-a_j)}{L^2(\R_+)}+O_j(c^{-1/3})$
(continuity of $L^2$-norm in the shift parameter).
\end{proof}

\section{(I2b): Kernel Domination}
\label{sec:I2b}
%% ============================================================

\begin{lemma}[Kernel domination]
\label{lem:I2b}
For $c\geq c_0$:
\begin{enumerate}[nosep,label=(\roman*)]
  \item $\xit,\xit'\in[0,M]$:
    $|\cK(\xit,\xit')|\leq C_M(1+\xit)^{-1/4}(1+\xit')^{-1/4}$.
  \item $\xit$ or $\xit'>M$:
    $|\cK(\xit,\xit')|\leq C'e^{-\frac43(\xit\wedge\xit'-M)_+^{3/2}}$.
\end{enumerate}
\end{lemma}

\begin{proof}
CD identity + Lemma~\ref{lem:langer}
($|\tilde\psi_j|\leq C(1+\xit)^{-1/4}$ uniformly in $j,c$)
for (i); Airy decay for (ii).
\end{proof}

\section{Proof of Main Lemma}
\label{sec:proof}
%% ============================================================

\textbf{Step 1.} Fermi tail:
$|\cK-\cK^{(K)}|\leq h(\xit,\xit')e^{-\pi R}$
(XVII Lem.~2.1 + Lem.~\ref{lem:I2b}).

\textbf{Step 2.} Projector convergence:
Cor.~\ref{cor:spectral}(iii).

\textbf{Step 3.} Fixed-$K$ pointwise convergence:
Lem.~\ref{lem:domconv}~\eqref{eq:domconv}.

\textbf{Step 4.} $K\to\infty$: Slepian tail bound
(Lem.~\ref{lem:domconv}~\eqref{eq:slepian} + DCT)
+ Prop.~\ref{prop:kernelbridge}.\hfill$\square$
