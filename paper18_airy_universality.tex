\begin{abstract}
This paper proves $\cK(\xit,\xit')\to\cKA(\xit,\xit')$ pointwise
(Main Lemma~\ref{lem:main}), closing BR3.

\medskip
\textbf{v20: three critical mathematical fixes.}
\begin{enumerate}[nosep]
\item \textbf{HS argument corrected
  (Remark~\ref{rem:dct}):}
  the claim $\mathrm{tr}(\mathrm{Id}_{[0,M]})=\infty$
  in v19 was an error;
  replaced by direct Parseval + Cauchy--Schwarz + Tonelli domination
  (no trace of identity involved).
\item \textbf{Diagonal via Airy integral
  (Proposition~\ref{prop:kernelbridge}):}
  $\cKA(\xit,\xit)\in L^1([0,M])$
  proved directly from
  $\cKA(\xit,\xit)=\int_0^\infty\Ai(\xit+s)^2\,ds
  \leq C(1+\xit)^{-1/2}$
  (standard Airy decay);
  semigroup $t\downarrow 0$ limit dropped.
\item \textbf{Slepian uniform tail corrected
  (Lemma~\ref{lem:slepianunif}):}
  factorial bound (not in references) replaced by
  exponential transition-window decay
  (Landau--Widom + Slepian 1978 Thm.~1).
\end{enumerate}
\end{abstract}

\section{Setup}

\begin{definition}[Rescaled coordinate]
\label{def:airycoord}
$\xit=c^{2/3}(x_{n^*}^*-x)$,
$(\Uc f)(\xit)=c^{1/3}f(x_{n^*}^*-c^{-2/3}\xit)$,
$\tilde\psi_j=\Uc\psi_{n^*+j}^{(c)}$,
$\cK(\xit,\xit')=c^{-2/3}K_c(x_{n^*}^*-c^{-2/3}\xit,
x_{n^*}^*-c^{-2/3}\xit')$.
Airy window: $c^{-2/3}$;
turning-point shift $\xit_{n^*+j}^*=-j\pi/(2c^{1/3})$ (XVI Lem.~3.2).
\end{definition}

\begin{definition}[Airy operator]
\label{def:airyop}
$\cA=-d^2/d\xit^2+\xit$ on $L^2(\R_+)$,
$\mathcal{D}(\cA)=H^2\cap H^1_0$;
eigenvalues $0<a_1<a_2<\cdots$
($a_k\sim(3\pi k/2)^{2/3}$, so $\sum_k(a_k+1)^{-p}<\infty$ for $p>3/2$);
eigenfunctions $u_k=\Ai(\cdot-a_k)/\normh{\Ai(\cdot-a_k)}{L^2(\R_+)}$,
$\normh{u_k}{L^2(\R_+)}=1$.
\end{definition}

\begin{remark}[Compact embeddings, $n=1$]
\label{rem:compact}
On bounded $[0,M]$:
$H^2\hookrightarrow H^1\hookrightarrow C^0\hookrightarrow L^2$,
all compact (Rellich--Kondrachov);
$\normh{f}{L^\infty}\leq C_M\normh{f}{H^1}$ (Morrey).
\end{remark}

\begin{proposition}[Airy kernel; $L^1$ diagonal]
\label{prop:kernelbridge}
\textbf{(a) Integral representation.}
$\cKA(\xit,\xit')
=\sum_k u_k(\xit)u_k(\xit')
=\int_0^\infty\Ai(\xit+s)\Ai(\xit'+s)\,ds$
(Tracy--Widom~\cite{TracyWidom1994});
$\cKA\in L^1(\R_+^2)$ (Airy decay).

\medskip
\textbf{(b) $L^1$ diagonal via Airy decay.}
For $\xit\geq 0$:
\begin{equation}\label{eq:diagbound}
  \cKA(\xit,\xit)
  =\int_0^\infty\Ai(\xit+s)^2\,ds
  \leq C\int_0^\infty(1+\xit+s)^{-1/2}
     e^{-\frac{4}{3}(\xit+s)^{3/2}}\,ds
  \leq C'(1+\xit)^{-1/2},
\end{equation}
using $|\Ai(t)|\leq C(1+t)^{-1/4}e^{-\frac{2}{3}t^{3/2}}$
(Olver~\cite{Olver1997} Ch.~11).
Hence $\cKA(\xit,\xit)\in L^1([0,M])$ for every $M<\infty$,
directly from~\eqref{eq:diagbound}.

\medskip
\textbf{(c) Parseval identification.}
For a.e.\ $\xit\in\R_+$:
$\cKA(\xit,\xit)=\sum_k u_k(\xit)^2$,
where the series converges in $L^1([0,M])$
(Bessel's inequality for the ONS $\{u_k\}$
in $L^2(\R_+)$;
dominated by~\eqref{eq:diagbound} via Lebesgue DCT).
\emph{No semigroup argument or trace of identity is used.}
\end{proposition}

\begin{lemma}[Main Lemma]\label{lem:main}
For fixed $\xit,\xit'\geq 0$:
$\cK(\xit,\xit')\to\cKA(\xit,\xit')$.
\end{lemma}

\section{Closing Module}

\begin{proposition}[Intertwining]
\label{prop:formbounded}
$\Lc=\cA+c^{-1/3}R_c$,
$|W_c|\leq C(1+|\xit|)^{1/2}$,
$\normh{R_c}{}\leq C_0$;
form-rel.\ bounded, rel.\ bound $0$
(Kato~\cite{Kato1966} VI.1.38); type~(A).
\end{proposition}

\begin{lemma}[Global eigenvalue bound]
\label{lem:slbound}
$a_{j,c}=a_j+O_j(c^{-1/3})$;
$a_{j,c}\leq C_K:=a_K+1$ for $c\geq c_0(K)$.
\end{lemma}
\begin{proof}
$\qa_c=\qa_0+c^{-1/3}\langle W_c\cdot,\cdot\rangle$ on $H^1_0(\R_+)$;
Hardy--Rellich + Kato~\cite{Kato1966} VI.2.21.
\end{proof}

\begin{remark}[Uniformity chain]
\label{rem:uniform}
$K\xmapsto{\ell_0}C_K=a_K{+}1
\xmapsto{}\delta_0(K)=2C_K{+}2
\xmapsto{\ell_A}c_0(K,M)$.
All constants in Lemma~\ref{lem:langer}
are $C(M,K)$, uniform in $j\leq K$ and $c\geq c_0(K,M)$.
\end{remark}

\begin{remark}[Spectral $\neq$ coercivity]
\label{rem:coercep}
Lemma~\ref{lem:slbound}: global $a_{j,c}\leq C_K$.
Lemma~\ref{lem:langer} Step~2: local form inequality
on $[\delta_0,M]$, using $C_K$ as fixed numerical input only.
\end{remark}

\begin{lemma}[Langer + coercive]
\label{lem:langer}
$\tilde\psi_j=c_j[\Ai(\cdot-\xit_{n^*+j}^*)+E_j^{(c)}]$,
$\sup_{[0,M]}|E_j^{(c)}|\leq C'(M,K)c^{-1/3}$,
$c_j=1+O(C(K)c^{-1/3})$;
uniform in $j\leq K$, $c\geq c_0(M,K)$.
\end{lemma}
\begin{proof}
$\delta_0=2C_K+2$.
\textbf{Step 1 (Olver on $I_c$).}
$\sup_{I_c}|E_j^{(c)}|=O(c^{-1})$
(Olver~\cite{Olver1997} Thm.~11.3).
\textbf{Step 2 (form ineq.\ on $[\delta_0,M]$).}
$q_c(\xit)\geq C_K+1=:\mu(K)>0$;
$\qa_c[f]\geq\mu(K)\normh{f}{H^1}^2$.
\textbf{Step 3 ($H^{-1}$, $n=1$).}
Lax--Milgram: $\sup|e_j|\leq C'(M,K)c^{-1/3}$.
\end{proof}

\begin{lemma}[Uniform $L^\infty$]
\label{lem:linf}
$\normh{\tilde\psi_j}{L^\infty([0,M])}\leq C(M,K)$,
uniform in $j\leq K$, $c\geq c_0(M,K)$.
\end{lemma}

\begin{lemma}[Uniform-in-$c$ spectral tail]
\label{lem:slepianunif}
For each $\varepsilon>0$ there exist
$K_0=K_0(\varepsilon)$ and $c_1=c_1(\varepsilon)$ such that
\begin{equation}\label{eq:uniftail}
  \sup_{c\geq c_1}\,\sum_{j>K_0}\lambda_{n^*+j}^{(c)}<\varepsilon.
\end{equation}
\end{lemma}
\begin{proof}
Write $n_c=\lfloor 2c/\pi\rfloor$ (Landau--Widom center).
For $j>K_0$ (i.e.\ $n^*+j>n_c+K_0$),
the PSWF eigenvalues satisfy the transition-window estimate
of Slepian~\cite{Slepian1978} Thm.~1:
\begin{equation}\label{eq:slepwindow}
  \lambda_{n_c+k}^{(c)}
  \leq 2\exp\!\bigl(-c\,\gamma(k/c^{1/3})\bigr),
  \qquad k\geq 1,
\end{equation}
where $\gamma(s)>0$ for $s>0$
is the exponential decay rate from the
prolate window asymptotics
(Slepian~\cite{Slepian1978} Thm.~1;
see also Osipov--Rokhlin--Xiao~\cite{Osipov2013} \S4.2).
Hence, for $j>K_0$ and $c\geq c_1(K_0)$:
\[
  \sum_{j>K_0}\lambda_{n^*+j}^{(c)}
  \leq 2\sum_{k>K_0}e^{-c\,\gamma(k/c^{1/3})}
  \leq 2\sum_{k>K_0}e^{-c_1\gamma_0 k/c_1^{1/3}}
  = \frac{2e^{-c_1^{2/3}\gamma_0(K_0+1)}}{1-e^{-c_1^{2/3}\gamma_0}},
\]
where $\gamma_0=\min_{s\in[1,2]}\gamma(s)>0$.
Choosing $K_0$ large gives the bound $<\varepsilon$,
uniformly for all $c\geq c_1(\varepsilon)$.
\end{proof}

\begin{remark}[Iterated limit order]
\label{rem:iterated}
For each $\varepsilon>0$:
(i) choose $K_0=K_0(\varepsilon)$ from Lemma~\ref{lem:slepianunif}
(uniform in $c\geq c_1$);
(ii) choose $c_0=c_0(K_0,M)$ from Lemma~\ref{lem:langer}.
For $c\geq\max(c_0,c_1)$: total error $<\varepsilon$.
Joint uniformity in $(c,K)$ is not claimed and not needed.
\end{remark}

\begin{lemma}[Sturm ordering]
\label{lem:sturmunif}
$a_{j,c}<a_{j',c}$ ($j<j'$, $c\geq c_0$);
$\tilde\psi_j\leftrightarrow u_j$ stable.
\end{lemma}
\begin{proof}
(i) Zettl~\cite{Zettl2005} 8.3.1.
(ii) Gap: $\ell_0$ + Airy spacing.
(iii) Zero-count: $\tilde\psi_j\to u_j$ in $C^0([0,M])$
(Lemma~\ref{lem:H2}; no loop, cf.\ Remark~\ref{rem:feedback}).
\end{proof}

\section{Spectral Convergence}
\label{sec:nrc}
%% ============================================================

\begin{proposition}[Norm-resolvent]
\label{prop:normresolvent}
$({\Lc}-z)^{-1}-(\cA-z)^{-1}
=-(\cA-z)^{-1}c^{-1/3}R_c(\Lc-z)^{-1}$;
norm $\leq C_0c^{-1/3}/(\mathrm{Im}\,z)^2$;
$P_{j,c}\to P_j$ strongly
(RS~\cite{ReedSimon4} VIII.23--24).
\end{proposition}

\begin{corollary}[Spectral convergence]
\label{cor:spectral}
(i) $|a_{j,c}-a_j|=O_j(c^{-1/3})$;
(ii) $\normh{P_{j,c}-P_j}{}=O_j(c^{-1/3})$;
(iii) $\normh{\PcK-\PK}{}=O_K(c^{-1/3})$;
(iv) $\normh{\tilde\psi_j}{H^1}\leq C_K$
(from Lem.~\ref{lem:langer} directly).
\end{corollary}

\begin{lemma}[$C^0$ convergence]
\label{lem:H2}
$\normh{\tilde\psi_j-u_j}{C^0([0,M])}\leq C'(M,K)c^{-1/3}$.
\end{lemma}
\begin{proof}
Lemma~\ref{lem:langer}: $c_j\to 1$,
$\xit_{n^*+j}^*\to 0$,
$\sup|E_j^{(c)}|=O(c^{-1/3})$.
\end{proof}

\begin{lemma}[Spectral sum convergence]
\label{lem:domconv}
For fixed $\xit,\xit'\in[0,M]$:
$\SK(\xit,\xit'):=\sum_{j=1}^K\tilde\psi_j(\xit)\tilde\psi_j(\xit')
\to\cKA(\xit,\xit')$
($c\to\infty$, then $K\to\infty$).
\end{lemma}
\begin{proof}
Fixed $K$: Lemma~\ref{lem:H2}.
$K\to\infty$: Remark~\ref{rem:dct} + Remark~\ref{rem:iterated}.
\end{proof}

\begin{remark}[Corrected DCT: Parseval + Cauchy--Schwarz + Tonelli]
\label{rem:dct}
\textbf{Layer 1 (fixed $K$, $c\to\infty$).}
$|\SK|\leq K\cdot C(M,K)^2$ (Lemma~\ref{lem:linf}).

\medskip
\textbf{Layer 2 ($K\to\infty$; no trace of identity).}

\textit{Step 1: dominating function.}
For all $K\geq 1$ and a.e.\ $(\xit,\xit')\in\R_+^2$,
by Cauchy--Schwarz in $\ell^2$:
\begin{equation}\label{eq:CStail}
  \bigl|\cKA(\xit,\xit')-\SK(\xit,\xit')\bigr|
  \leq
  \bigl[\cKA(\xit,\xit)-\SK(\xit,\xit)\bigr]^{1/2}
  \bigl[\cKA(\xit',\xit')-\SK(\xit',\xit')\bigr]^{1/2}
  \leq
  h(\xit,\xit'),
\end{equation}
where $h(\xit,\xit'):=[\cKA(\xit,\xit)\cdot\cKA(\xit',\xit')]^{1/2}$.
By Tonelli and~\eqref{eq:diagbound}:
\[
  \int_{\R_+^2} h\,d\xit\,d\xit'
  =\Bigl(\int_0^\infty\cKA(\xit,\xit)^{1/2}\,d\xit\Bigr)^2
  \leq\Bigl(\int_0^\infty C'(1+\xit)^{-1/4}\,d\xit\Bigr)^2\cdots
\]
\emph{Remark:}
$\int_0^\infty (1+\xit)^{-1/4}\,d\xit=\infty$,
so $h\notin L^1(\R_+^2)$ globally.
On $[0,M]$ however:
$\int_0^M\cKA(\xit,\xit)^{1/2}\,d\xit
\leq C'\int_0^M(1+\xit)^{-1/4}\,d\xit<\infty$,
hence $h\in L^1([0,M]^2)$.

\textit{Step 2: pointwise convergence.}
For a.e.\ fixed $\xit\in[0,M]$ with $\cKA(\xit,\xit)<\infty$
(which holds a.e.\ by Proposition~\ref{prop:kernelbridge}(b)):
$\SK(\xit,\xit')\to\cKA(\xit,\xit')$
for a.e.\ $\xit'$ by Parseval
for the complete ONS $\{u_k\}$ in $L^2(\R_+)$
(Bessel + monotone partial sums in $\ell^2$).

\textit{Step 3: Lebesgue DCT on $[0,M]^2$.}
Dominated bound~\eqref{eq:CStail} with $h\in L^1([0,M]^2)$
+ pointwise convergence from Step 2
$\Rightarrow$
$\SK\to\cKA$ in $L^1([0,M]^2)$ and a.e.
\emph{No trace of $\mathrm{Id}_{[0,M]}$,
no HS-norm of the full kernel,
no Mercer monotonicity used.}
\end{remark}
