\end{enumerate}

\section{Setup and Logical Architecture}
\label{sec:setup}
%% ============================================================

\begin{definition}[Rescaled coordinate]
\label{def:airycoord}
$\xit = c^{2/3}(x_{n^*}^*-x)$,
$(\Uc f)(\xit) = c^{1/3}f(x_{n^*}^*-c^{-2/3}\xit)$,
$\tilde\psi_j = \Uc\psi_{n^*+j}^{(c)}$,
$\cK(\xit,\xit') = c^{-2/3}K_c(x_{n^*}^*-c^{-2/3}\xit,
x_{n^*}^*-c^{-2/3}\xit')$.

Airy window: $c^{-2/3}$ in the $\xit$-variable.
Turning-point shift: $\xit_{n^*+j}^* = -j\pi/(2c^{1/3})$
($\xit$-image of PSWF turning point;
Paper~XVI Lem.~3.2);
$\xit_{n^*+j}^*\to 0$ for fixed $j$.
\end{definition}

\begin{definition}[Airy operator]
\label{def:airyop}
$\cA = -d^2/d\xit^2+\xit$ on $L^2(\R_+)$,
$\mathcal{D}(\cA)=H^2\cap H^1_0$;
eigenvalues $0<a_1<a_2<\cdots$,
eighenfunctions
$u_k = \Ai(\cdot-a_k)/\normh{\Ai(\cdot-a_k)}{L^2(\R_+)}$.
\end{definition}

\begin{remark}[Compact embeddings used throughout]
\label{rem:compact}
All function spaces are defined on bounded intervals $[0,M]$
or $[\delta_0,M]\subset\R_+$.
On bounded intervals in $\R$ ($n=1$):
\begin{enumerate}[nosep,label=(\roman*)]
  \item $H^2([0,M])\hookrightarrow H^1([0,M])\hookrightarrow
    C^0([0,M])\hookrightarrow L^2([0,M])$
    (Sobolev--Morrey, $n=1$: $H^1\hookrightarrow L^\infty$;
    all embeddings compact by Rellich--Kondrachov).
  \item In particular:
    $\normh{f}{L^\infty([0,M])}\leq C_M\normh{f}{H^1([0,M])}$
    for all $f\in H^1([0,M])$.
  \item These embeddings are used in:
    Lemma~\ref{lem:langer} (Step~3),
    Lemma~\ref{lem:linf},
    Lemma~\ref{lem:H2},
    without further comment.
\end{enumerate}
\end{remark}

\begin{proposition}[Airy kernel]
\label{prop:kernelbridge}
$\cKA(\xit,\xit') = \sum_k u_k(\xit)u_k(\xit')
= \int_0^\infty\Ai(\xit+s)\Ai(\xit'+s)\,ds$;
$\cKA\in L^1(\R_+^2)\cap L^2(\R_+^2)$.
\end{proposition}
\begin{proof}
Parseval + Olver~\cite{Olver1997} Ch.~11 Thm.~11.5
+ Tracy--Widom~\cite{TracyWidom1994};
$L^1$: $|\Ai(s)|\leq Ce^{-\frac23s^{3/2}}$.
\end{proof}

\begin{lemma}[Main Lemma]\label{lem:main}
For fixed $\xit,\xit'\geq 0$:
$\cK(\xit,\xit')\to\cKA(\xit,\xit')$.
\end{lemma}

\begin{proposition}[Intertwining]
\label{prop:formbounded}
$\Lc=\cA+c^{-1/3}R_c$;
$R_c=W_c(\xit)$, $|W_c|\leq C(1+|\xit|)^{1/2}$,
$\normh{R_c}{}\leq C_0$;
form-relatively-bounded, rel.\ bound $0$
(Kato~\cite{Kato1966} Ch.~VI Thm.~1.38);
type~(A) (Kato~\cite{Kato1966} Ch.~VII \S2).
\end{proposition}

\subsection{Lemma~0: Global SL eigenvalue bound}

\begin{lemma}[Global eigenvalue bound]
\label{lem:slbound}
$a_{j,c} = a_j+O_j(c^{-1/3})$, hence $a_{j,c}\leq C_K:=a_K+1$
for $c\geq c_0(K)$.
\end{lemma}

\begin{proof}
Closed form of $\Lc$:
$\qa_c[f]=\qa_0[f]+c^{-1/3}\langle W_cf,f\rangle$, $f\in H^1_0(\R_+)$.
Hardy--Rellich: for any $\varepsilon>0$,
$|\langle W_cf,f\rangle|\leq\varepsilon\qa_0[f]+C_\varepsilon\normh{f}{}^2$.
Global form comparison (Kato~\cite{Kato1966} Ch.~VI Thm.~2.21)
+ min-max (Reed--Simon~\cite{ReedSimon4} Thm.~XIII.1):
$a_{j,c}=a_j+O_j(c^{-1/3})$ on $H^1_0(\R_+)$.
\end{proof}

\begin{remark}[Spectral bound vs.\ coercivity: logical separation]
\label{rem:coercep}
Lemma~\ref{lem:slbound} yields \emph{spectral information}:
$a_{j,c}\leq C_K$.
This is a statement about eigenvalues of $\Lc$,
obtained via global quadratic form comparison on $H^1_0(\R_+)$.

\medskip
Coercivity in Lemma~\ref{lem:langer} Step~2 is a separate,
\emph{local form inequality}:
$q_c(\xit)\geq\delta_0/2$ for $\xit\in[\delta_0,M]$,
where $\delta_0:=2C_K+2$.
This uses $a_{j,c}\leq C_K$ only as a \emph{numerical input}
to the pointwise estimate
$q_c(\xit)=\xit-a_{j,c}+c^{-1/3}W_c(\xit)\geq\delta_0-C_K-\text{(small)}
\geq\delta_0/2$.

\medskip
\textbf{No circularity:}
Lemma~\ref{lem:slbound} does not use coercivity;
Lemma~\ref{lem:langer} Step~2 uses only the output of
Lemma~\ref{lem:slbound} as a fixed constant.
The two statements live on different logical levels
(global spectrum vs.\ local potential).
\end{remark}

\subsection{Lemma~A: Uniform Airy approximation}

\begin{lemma}[Langer $+$ coercive patch]
\label{lem:langer}
$\tilde\psi_j=c_j[\Ai(\cdot-\xit_{n^*+j}^*)+E_j^{(c)}]$,
$\sup_{[0,M]}|E_j^{(c)}|\leq C'(M,K)c^{-1/3}$,
$c_j=1+O_K(c^{-1/3})$;
uniform in $|j|\leq K$, $c\geq c_0(M,K)$.
\end{lemma}

\begin{proof}
$I_c = [a_{j,c}^*-c^{-2/3},a_{j,c}^*+c^{-2/3}]$;
$\delta_0=2C_K+2$ (Remark~\ref{rem:coercep}).

\textbf{Step 1 (Olver on $I_c$, Airy scale $c^{-2/3}$).}
$\normh{c^{-1/3}W_c}{L^1(I_c)}\leq 2Cc^{-1}$;
Olver~\cite{Olver1997} Ch.~11 Thm.~11.3:
$\sup_{I_c}|E_j^{(c)}|=O(c^{-1})$.

\textbf{Step 2 (Local form inequality on $[\delta_0,M]$).}
For $\xit\geq\delta_0=2C_K+2$:
$q_c(\xit)\geq\xit-C_K-c^{-1/3}C(1+M)^{1/2}
\geq 2C_K+2-C_K-c^{-1/3}C(1+M)^{1/2}
\geq C_K+1>0$.
Taking $\mu:=\min(1,C_K+1)>0$,
the form $\qa_c[f]\geq\mu\normh{f}{H^1}^2$
(Hilbert-space form inequality on $H^1_0([\delta_0,M])$;
cf.\ Remark~\ref{rem:coercep}).

\textbf{Step 3 ($H^{-1}$ via $H^1_0$-duality, $n=1$).}
For $\phi\in H^1_0([\delta_0,M])$:
$|\langle W_c\tilde\psi_j,\phi\rangle|
\leq\normh{W_c}{L^2}\normh{\tilde\psi_j}{L^2}\normh{\phi}{L^\infty}
\leq C_M\cdot C_{\delta_0,M}\normh{\phi}{H^1}$
(Sobolev $H^1\hookrightarrow L^\infty$ in 1D;
Remark~\ref{rem:compact}(ii)).
Hence $\normh{W_c\tilde\psi_j}{H^{-1}}\leq C_{M,K}$;
Lax--Milgram: $\normh{e_j}{H^1}\leq C_{M,K}c^{-1/3}/\mu$;
Sobolev: $\sup|e_j|=O_{M,K}(c^{-1/3})$.

\textbf{Normalization.}
$N'(a)=-2\Ai(-a)^2$;
$|N(a_{j,c}^*)-N(a_j)|=O_K(c^{-1/3})$;
$c_j=1+O_K(c^{-1/3})$.
\end{proof}

\subsection{Lemma~$L^\infty$: Uniform eigenfunction bound}

\begin{lemma}[Uniform $L^\infty$ bound]
\label{lem:linf}
$\normh{\tilde\psi_j}{L^\infty([0,M])}\leq C_{M,K}$,
uniformly in $|j|\leq K$, $c\geq c_0$.
\end{lemma}

\begin{proof}
$-\tilde\psi_j''+q_c\tilde\psi_j=0$, $\normh{\tilde\psi_j}{L^2}=1$.
$H^2$ regularity (1D SL):
$\normh{\tilde\psi_j}{H^2([0,M])}
\leq C(1+\normh{q_c}{L^\infty([0,M])})\leq C_{M,K}$
(Lemma~\ref{lem:slbound}: $a_{j,c}\leq C_K$, so
$\normh{q_c}{L^\infty([0,M])}\leq M+C_K+c^{-1/3}C(1+M)^{1/2}\leq C'_{M,K}$).
Rellich--Kondrachov compact embedding
$H^2([0,M])\hookrightarrow C^0([0,M])$
(Remark~\ref{rem:compact}):
$\normh{\tilde\psi_j}{L^\infty([0,M])}\leq C_{M,K}$.
\end{proof}

\subsection{Lemma~B: Sturm ordering}

\begin{lemma}[Eigenvalue ordering]
\label{lem:sturmunif}
$a_{j,c}<a_{j',c}$ for $j<j'\leq K$ and $c\geq c_0$;
identification $\tilde\psi_j\leftrightarrow u_j$ stable.
\end{lemma}

\begin{proof}
(i) Zettl~\cite{Zettl2005} Thm.~8.3.1: simplicity, strict order.
(ii) Uniform gap: Lemma~\ref{lem:slbound} + Airy spacing
$a_{j+1}-a_j\geq\eta_K>0$ + $|a_{j,c}-a_j|=O_j(c^{-1/3})$.
(iii) Stable zero-count: $\tilde\psi_j\to u_j$ in $C^0([0,M])$
(Lemma~\ref{lem:H2}), so zero-count stable for large $c$.
\end{proof}

\subsection{Lemma~C: Trace-class tail}

\begin{lemma}[Trace-class spectral tail]
\label{lem:slepian}
$\sum_{j>K}\lambda_{n^*+j}^{(c)}\to 0$ as $K\to\infty$,
for each fixed $c$
(trace class: Landau--Widom~\cite{LandauWidom1980},
Simon~\cite{Simon2005}).
\end{lemma}

\section{Spectral Convergence and Resolvent}
\label{sec:nrc}
%% ============================================================

\begin{proposition}[Norm-resolvent]
\label{prop:normresolvent}
$(\Lc-z)^{-1}-(\cA-z)^{-1}
= -(\cA-z)^{-1}c^{-1/3}R_c(\Lc-z)^{-1}$;
$\normh{\cdot}{}\leq C_0c^{-1/3}/(\mathrm{Im}\,z)^2$.
Norm-resolvent convergence $\Rightarrow$ strong convergence of
spectral projections $P_{j,c}\to P_j$
(Reed--Simon~\cite{ReedSimon4} Thm.~VIII.23--24).
\end{proposition}

\begin{corollary}[Spectral convergence]
\label{cor:spectral}
(i) $|a_{j,c}-a_j|=O_j(c^{-1/3})$;
(ii) $\normh{P_{j,c}-P_j}{}=O_j(c^{-1/3})$;
(iii) $\normh{\PcK-\PK}{}=O_K(c^{-1/3})$;
(iv) $\normh{\tilde\psi_j}{H^1}\leq C_K$.
\end{corollary}

\begin{lemma}[$C^0$ convergence]
\label{lem:H2}
$\normh{\tilde\psi_j-u_j}{C^0([0,M])}=O_K(c^{-1/3})$.
\end{lemma}
\begin{proof}
Lemma~\ref{lem:langer}: $c_j\to 1$,
$\xit_{n^*+j}^*\to 0$, $\sup|E_j^{(c)}|=O(c^{-1/3})$.
\end{proof}

\begin{lemma}[Spectral sum DCT]
\label{lem:domconv}
For fixed $\xit,\xit'\in[0,M]$:
$\sum_{j=1}^K\tilde\psi_j(\xit)\tilde\psi_j(\xit')
\to\sum_{j=1}^K u_j(\xit)u_j(\xit')
\to\cKA(\xit,\xit')$.
\end{lemma}

\begin{proof}
Fixed $K$: Lemma~\ref{lem:H2} (pointwise).
$K\to\infty$: Lemma~\ref{lem:slepian} + Remark~\ref{rem:dct} below.
\end{proof}

\begin{remark}[Two-layer dominated convergence]
\label{rem:dct}
The $K\to\infty$ limit involves two separate DCT applications
operating at different levels:

\medskip
\textbf{Layer 1 (domination in $K$, uniform in $c$).}
For fixed $c\geq c_0$ and $\xit,\xit'\in[0,M]$:
\[
  \Bigl|\sum_{j=1}^K\tilde\psi_j(\xit)\tilde\psi_j(\xit')\Bigr|
  \leq K\cdot C_{M,K}^2 < \infty
\]
(Lemma~\ref{lem:linf}, uniform in $c$).
This ensures the $c\to\infty$ limit exists term-by-term for each $K$.

\medskip
\textbf{Layer 2 (domination in $c$, for the $K\to\infty$ step).}
In the limit $c\to\infty$, we have
$\sum_{j=1}^K u_j(\xit)u_j(\xit')\nearrow\cKA(\xit,\xit')\in L^1(\R_+^2)$
(Proposition~\ref{prop:kernelbridge}).
Since $\cKA\geq 0$ and $\sum_{j=1}^K u_ju_j\leq\cKA$ (spectral),
Lebesgue's monotone convergence theorem applies;
no additional domination function is needed beyond $\cKA$ itself.

\medskip
The two layers are logically independent:
Layer~1 controls uniformity in $c$;
Layer~2 controls the $K\to\infty$ passage in the limit system.
There is no circularity.
\end{remark}

\begin{lemma}[Kernel domination]
\label{lem:I2b}
$|\cK|\leq C_M(1+\xit)^{-1/4}(1+\xit')^{-1/4}$
on $[0,M]^2$;

\end{lemma}

\textbf{Proof of Main Lemma~\ref{lem:main}.}
