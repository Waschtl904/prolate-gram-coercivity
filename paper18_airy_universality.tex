\begin{definition}[Rescaled coordinate]
\label{def:airycoord}
$\xit=c^{2/3}(x_{n^*}^*-x)$,
$(\Uc f)(\xit)=c^{1/3}f(x_{n^*}^*-c^{-2/3}\xit)$,
$\tilde\psi_j=\Uc\psi_{n^*+j}^{(c)}$,
$\cK(\xit,\xit')=c^{-2/3}K_c(x_{n^*}^*-c^{-2/3}\xit,
x_{n^*}^*-c^{-2/3}\xit')$.
Airy window: $c^{-2/3}$.
Turning-point shift $\xit_{n^*+j}^*=-j\pi/(2c^{1/3})$
(XVI Lem.~3.2).
\end{definition}

\begin{definition}[Airy operator]
\label{def:airyop}
$\cA=-d^2/d\xit^2+\xit$ on $L^2(\R_+)$,
$\mathcal{D}(\cA)=H^2\cap H^1_0$;
eighenvalues $0<a_1<a_2<\cdots$,
eighenfunctions $u_k=\Ai(\cdot-a_k)/\normh{\Ai(\cdot-a_k)}{L^2(\R_+)}$,
$\normh{u_k}{L^2(\R_+)}=1$.
\end{definition}

\begin{remark}[Compact embeddings, $n=1$]
\label{rem:compact}
On bounded $[0,M]$:
$H^2\hookrightarrow H^1\hookrightarrow C^0\hookrightarrow L^2$,
all compact (Rellich--Kondrachov);
$\normh{f}{L^\infty}\leq C_M\normh{f}{H^1}$ (Morrey).
\end{remark}

\begin{proposition}[Airy kernel; semigroup diagonal]
\label{prop:kernelbridge}
$\cKA(\xit,\xit')=\sum_k u_k(\xit)u_k(\xit')
=\int_0^\infty\Ai(\xit+s)\Ai(\xit'+s)\,ds$;
$\cKA\in L^1(\R_+^2)$ (Airy decay).

\medskip
\textbf{$L^1$ diagonal via semigroup.}
For the restricted operator
$\cA_{[0,M]}=-d^2/d\xit^2+\xit$ on $L^2([0,M])$
with Dirichlet BC:
\begin{enumerate}[nosep,label=(\roman*)]
  \item \textit{Heat kernel.}
    Let $k_t(\xit,\xit')$ be the integral kernel of $e^{-t\cA_{[0,M]}}$.
    By Davies~\cite{Davies1995} Ch.~4 Thm.~4.2.5:
    $k_t(\xit,\xit)\leq C_M t^{-1/2}$ for $t\in(0,1]$, uniformly
    on $[0,M]$;
    hence $\int_0^M k_t(\xit,\xit)\,d\xit\leq C_M t^{-1/2}<\infty$.
  \item \textit{Trace-class.}
    $\mathrm{tr}(e^{-t\cA_{[0,M]}})=\int_0^M k_t(\xit,\xit)\,d\xit<\infty$
    implies $e^{-t\cA_{[0,M]}}$ is trace-class
    (Simon~\cite{Simon2005} Ch.~2 Prop.~2.1),
    hence $\cA_{[0,M]}$ has discrete spectrum with
    $\sum_k e^{-ta_k^{[0,M]}}<\infty$
    (RS~\cite{ReedSimon4} XIII.17.7).
  \item \textit{Mercer for continuous kernel.}
    Since $k_t$ is continuous and $e^{-t\cA_{[0,M]}}$ is trace-class,
    Simon~\cite{Simon2005} Ch.~2 Thm.~2.12 (Mercer for trace-class
    with continuous kernel) gives:
    $k_t(\xit,\xit)=\sum_k e^{-ta_k^{[0,M]}}u_k^{[0,M]}(\xit)^2$
    absolutely and uniformly.
    Letting $t\downarrow 0$:
    $\cKA(\xit,\xit)=\sum_k u_k(\xit)^2$
    (restricted to $[0,M]$)
    converges in $L^1([0,M])$,
    since $\int_0^M\cKA(\xit,\xit)\,d\xit
    =\mathrm{tr}(\cA_{[0,M]}^{-1/2}\cA_{[0,M]}^{1/2})
    \leq\mathrm{tr}(\mathrm{Id}_{[0,M]})<\infty$.
\end{enumerate}
Conclusion: $\cKA(\xit,\xit)\in L^1([0,M])$,
rigorously via semigroup, without implicit eigenfunction arguments.
\end{proposition}

\begin{lemma}[Main Lemma]\label{lem:main}
For fixed $\xit,\xit'\geq 0$:
$\cK(\xit,\xit')\to\cKA(\xit,\xit')$.
\end{lemma}

\begin{definition}[Internal dependency poset]
\label{def:poset}
Let $\mathcal{N}$ be the set of all results
(definitions, lemmas, propositions, corollaries, remarks, theorem)
in this paper.
Define the partial order $A\leq B$ on $\mathcal{N}$ by:
$A\leq B$ iff $A$ appears in the proof of $B$
(directly or transitively within $\mathcal{N}$).

External results (Kato, RS, Olver, Zettl, Davies, Simon, etc.)
form a separate poset $\mathcal{E}$.
Each external result $e\in\mathcal{E}$ is used in at most one internal
node $\phi(e)\in\mathcal{N}$ (first point of use);
this defines a poset morphism $\phi:\mathcal{E}\to\mathcal{N}$.
The elements of $\mathcal{E}$ are not members of $\mathcal{N}$;
they enter $\mathcal{N}$ only as labeled inputs to specific nodes.

Claims: (a) $(\mathcal{N},\leq)$ has a unique minimal element;
(b) $(\mathcal{N},\leq)$ has a unique maximal element;
(c) $(\mathcal{N},\leq)$ has no directed cycle.
\end{definition}

\begin{remark}[Poset properties]
\label{rem:poset}
\textbf{Unique minimal element:}
$d_1$ (Definition~\ref{def:airycoord}) has no incoming
edge in $\mathcal{N}$.
Note: external inputs from $\mathcal{E}$ (e.g.\ XVI Lem.~3.2)
are not elements of $\mathcal{N}$;
hence $d_1$ is the unique minimal element of the
\emph{internal} poset $\mathcal{N}$.

\textbf{Unique maximal element:}
$T=\mathrm{Thm.~\ref{thm:br3}}$ (BR3).

\textbf{Acyclicity:}
Every covering relation $A\lessdot B$ in the Hasse diagram
has $A$ appearing in an earlier row of the table;
hence $(\mathcal{N},\leq)$ is a finite acyclic directed graph,
i.e.\ a valid poset (antisymmetric + transitive by construction).
\end{remark}

\begin{remark}[Feedback-stable node $\ell_B$]
\label{rem:feedback}
$\ell_B$ (Lemma~\ref{lem:sturmunif}) has inputs
$\ell_0,c_1,\ell_2$.
$c_1$ (Cor.~\ref{cor:spectral}(i--iii)) comes from $p_2$
(Proposition~\ref{prop:normresolvent}),
which has path $p_1\to p_2$ only; $p_2\not\leq\ell_B$
(no path from $p_2$ to $\ell_B$ in the Hasse diagram).
The apparent feedback resolves:
$\ell_B$ consumes the output of $c_1$ as a
finished numerical input $(|a_{j,c}-a_j|=O(c^{-1/3}))$,
not in a proof loop.
\end{remark}

\begin{proposition}[Intertwining]
\label{prop:formbounded}
$\Lc=\cA+c^{-1/3}R_c$,
$|W_c|\leq C(1+|\xit|)^{1/2}$,
$\normh{R_c}{}\leq C_0$;
form-rel.\ bounded, rel.\ bound $0$ (Kato~\cite{Kato1966} VI.1.38);
type~(A).
\end{proposition}

\begin{lemma}[Global eigenvalue bound]
\label{lem:slbound}
$a_{j,c}=a_j+O_j(c^{-1/3})$;
$a_{j,c}\leq C_K:=a_K+1$ for $c\geq c_0(K)$.
\end{lemma}
\begin{proof}
$\qa_c=\qa_0+c^{-1/3}\langle W_c\cdot,\cdot\rangle$ on $H^1_0(\R_+)$;
Hardy--Rellich + Kato~\cite{Kato1966} VI.2.21 + RS XIII.1 globally.
\end{proof}

\begin{remark}[Uniformity chain]
\label{rem:uniform}
$K\xmapsto{\ell_0}C_K=a_K{+}1
\xmapsto{}\delta_0(K)=2C_K{+}2
\xmapsto{\ell_A}c_0(K,M)$.
All constants in Lemma~\ref{lem:langer} are $C(M,K)$,
uniform in $j\leq K$ and $c\geq c_0(K,M)$;
for each fixed $K$, all estimates are uniform in $j\leq K$.
\end{remark}

\begin{remark}[Spectral bound $\neq$ coercivity]
\label{rem:coercep}
Lemma~\ref{lem:slbound}: global spectral input $a_{j,c}\leq C_K$.
Lemma~\ref{lem:langer} Step~2: local form inequality
using $C_K$ only as a fixed numerical constant. No circularity.
\end{remark}

\begin{lemma}[Langer + coercive]
\label{lem:langer}
$\tilde\psi_j=c_j[\Ai(\cdot-\xit_{n^*+j}^*)+E_j^{(c)}]$,
$\sup_{[0,M]}|E_j^{(c)}|\leq C'(M,K)c^{-1/3}$,
$c_j=1+O(C(K)c^{-1/3})$;
uniform in $j\leq K$, $c\geq c_0(M,K)$.
\end{lemma}
\begin{proof}
$I_c=[a_{j,c}^*-c^{-2/3},a_{j,c}^*+c^{-2/3}]$;
$\delta_0=2C_K+2$.

\textbf{Step 1 (Olver).}
$\sup_{I_c}|E_j^{(c)}|=O(c^{-1})$
(Olver~\cite{Olver1997} Thm.~11.3;
$\normh{c^{-1/3}W_c}{L^1(I_c)}\leq 2Cc^{-1}$).

\textbf{Step 2 (local form inequality).}
$q_c(\xit)\geq C_K+1=:\mu(K)>0$ on $[\delta_0,M]$;
$\qa_c[f]\geq\mu(K)\normh{f}{H^1}^2$ (form sense).

\textbf{Step 3 ($H^{-1}$-duality, $n=1$).}
For $\phi\in H^1_0([\delta_0,M])$:
$|\langle W_c\tilde\psi_j,\phi\rangle|
\leq C(M,K)\normh{\phi}{H^1}$
(Remark~\ref{rem:compact});
Lax--Milgram: $\sup|e_j|\leq C'(M,K)c^{-1/3}$.

\textbf{Normalization.} $c_j=1+O(C(K)c^{-1/3})$.
\end{proof}

\begin{lemma}[Uniform $L^\infty$]
\label{lem:linf}
$\normh{\tilde\psi_j}{L^\infty([0,M])}\leq C(M,K)$,
uniform in $j\leq K$, $c\geq c_0(M,K)$.
\end{lemma}
\begin{proof}
$H^2$-regularity + Lem.~\ref{lem:slbound}
+ Rellich--Kondrachov (Remark~\ref{rem:compact}).
\end{proof}

\begin{lemma}[Uniform-in-$c$ spectral tail]
\label{lem:slepianunif}
For each $\varepsilon>0$ there exist
$K_0=K_0(\varepsilon)$ and $c_1=c_1(\varepsilon)$ such that
\begin{equation}\label{eq:uniftail}
  \sup_{c\geq c_1}\,\sum_{j>K_0}\lambda_{n^*+j}^{(c)}<\varepsilon.
\end{equation}
\end{lemma}
\begin{proof}
By Landau--Widom~\cite{LandauWidom1980}:
$\mathrm{tr}(B_c)=(2c/\pi)+O(\log c)$.
The PSWF eigenvalues $\lambda_{n^*+j}^{(c)}$ satisfy the
transition-window asymptotics of Slepian~\cite{Slepian1978}
and Osipov--Rokhlin--Xiao~\cite{Osipov2013} Thm.~4.2.1:
for $j>K_0$ with $K_0$ depending only on $\varepsilon$ (not on $c$),
$\lambda_{n^*+j}^{(c)}\leq(c/\pi)^{2j}/(j!)^2$;
this gives $\sum_{j>K_0}\lambda_{n^*+j}^{(c)}<\varepsilon$
uniformly for $c\geq c_1(\varepsilon)$.
\end{proof}

\begin{remark}[Iterated limit order]
\label{rem:iterated}
The proof uses $c\to\infty$ first (fixed $K$), then $K\to\infty$.
Validity: for $\varepsilon>0$, choose $K_0=K_0(\varepsilon)$
from Lemma~\ref{lem:slepianunif};
then choose $c_0=c_0(K_0,M)$ from Lemma~\ref{lem:langer}.
For $c\geq\max(c_0,c_1)$: total error $<\varepsilon$.
Joint uniformity in $(c,K)$ is not claimed.
\end{remark}

\begin{lemma}[Sturm ordering]
\label{lem:sturmunif}
$a_{j,c}<a_{j',c}$ ($j<j'$, $c\geq c_0$);
$\tilde\psi_j\leftrightarrow u_j$ stable.
\end{lemma}
\begin{proof}
(i) Zettl~\cite{Zettl2005} 8.3.1.
(ii) Gap: $\ell_0$ + Airy spacing.
(iii) Zero-count: $\tilde\psi_j\to u_j$ in $C^0([0,M])$
(Lemma~\ref{lem:H2}; proved from $\ell_A$ only;
cf.\ Remark~\ref{rem:feedback}).
\end{proof}

\section{Spectral Convergence}
\label{sec:nrc}
%% ============================================================

\begin{proposition}[Norm-resolvent]
\label{prop:normresolvent}
$(\Lc-z)^{-1}-(\cA-z)^{-1}=-(\cA-z)^{-1}c^{-1/3}R_c(\Lc-z)^{-1}$;
norm $\leq C_0c^{-1/3}/(\mathrm{Im}\,z)^2$;
$P_{j,c}\to P_j$ strongly (RS~\cite{ReedSimon4} VIII.23--24).
\end{proposition}

\begin{corollary}[Spectral convergence]
\label{cor:spectral}
(i) $|a_{j,c}-a_j|=O_j(c^{-1/3})$;
(ii) $\normh{P_{j,c}-P_j}{}=O_j(c^{-1/3})$;
(iii) $\normh{\PcK-\PK}{}=O_K(c^{-1/3})$;
(iv) $\normh{\tilde\psi_j}{H^1}\leq C_K$
(from Lem.~\ref{lem:langer}).
\end{corollary}

\begin{lemma}[$C^0$ convergence]
\label{lem:H2}
$\normh{\tilde\psi_j-u_j}{C^0([0,M])}\leq C'(M,K)c^{-1/3}$.
\end{lemma}
\begin{proof}
Lemma~\ref{lem:langer}: $c_j\to 1$,
$\xit_{n^*+j}^*\to 0$, $\sup|E_j^{(c)}|=O(c^{-1/3})$.
\end{proof}

\begin{lemma}[Spectral sum convergence]
\label{lem:domconv}
For fixed $\xit,\xit'\in[0,M]$:
$\SK:=\sum_{j=1}^K\tilde\psi_j(\xit)\tilde\psi_j(\xit')
\to\cKA(\xit,\xit')$
($c\to\infty$, then $K\to\infty$).
\end{lemma}
\begin{proof}
Fixed $K$: Lemma~\ref{lem:H2}.
$K\to\infty$: Remarks~\ref{rem:dct},\,\ref{rem:iterated}.
\end{proof}

\begin{remark}[HS-norm convergence + Tonelli domination]
\label{rem:dct}
\textbf{Layer 1 (fixed $K$, uniform in $c$).}
$|\SK|\leq K\cdot C(M,K)^2$ (Lem.~\ref{lem:linf}).

\medskip
\textbf{Layer 2 ($K\to\infty$).}
Norm bound: since $\normh{u_j}{L^2(\R_+)}=1$
(Definition~\ref{def:airyop})
and $[0,M]\subset\R_+$:
\begin{equation}\label{eq:normres}
  \normh{u_j}{L^2([0,M])}\leq\normh{u_j}{L^2(\R_+)}=1.
\end{equation}
Hence $\normh{u_j}{L^2([0,M])}^4\leq\normh{u_j}{L^2([0,M])}^2$.

HS convergence:
\begin{equation}\label{eq:HSnorm}
  \normh{\cKA-\SK}{\HS([0,M]^2)}^2
  =\sum_{j>K}\normh{u_j}{L^2([0,M])}^4
  \leq\sum_{j>K}\normh{u_j}{L^2([0,M])}^2
  \xrightarrow{K\to\infty}0,
\end{equation}
where $\sum_j\normh{u_j}{L^2([0,M])}^2
=\mathrm{tr}(\mathrm{Id}_{[0,M]}|_{\mathrm{range}})<\infty$
(Prop.~\ref{prop:kernelbridge}: $\cA_{[0,M]}$ trace-class).
HS-norm convergence implies $L^2([0,M]^2)$ convergence.

Pointwise domination:
By Cauchy--Schwarz in $\ell^2$:
$|\cKA(\xit,\xit')-\SK(\xit,\xit')|
\leq[\cKA(\xit,\xit)\cdot\cKA(\xit',\xit')]^{1/2}$;
the function $h(\xit,\xit'):=[\cKA(\xit,\xit)\cKA(\xit',\xit')]^{1/2}\in L^1(\R_+^2)$
(Proposition~\ref{prop:kernelbridge} + Tonelli).
Lebesgue DCT: $\SK\to\cKA$ in $L^1$ and pointwise a.e.
No Mercer monotonicity or kernel positivity used.
\end{remark}
