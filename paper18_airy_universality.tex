  Paper~XVIII: Airy Kernel Universality (v3)\\[0.3em]
  \large Two-Regime Domination~$+$~Edge-Window Uniformity
  $+$~Finite-Window Operator Convergence}

\author{\textit{prolate-gram-coercivity programme}}

\begin{abstract}
This paper proves the Main Lemma (Lemma~\ref{lem:main}):
for fixed $\xi,\eta\geq 0$,
$\cK(\xi,\eta)\to\cKA(\xi,\eta)$.
Combined with Paper~XVII (two-regime domination), this closes BR3.

\medskip
\textbf{New in v3.}
Section~\ref{sec:uniformity} (``Uniformity in the Edge Window'')
closes both open points flagged in v2:
\begin{enumerate}[nosep]
  \item \emph{Global WKB uniformity in $K$-window
    (Lemma~\ref{lem:tpshift}):}
    the turning-point shift $\xi_{n^*+j}^*-\xi_{n^*}^*
    = jc^{-1/3}\to 0$ for fixed $j$;
    for $c\geq c_0(K,M)$ all $K$ modes share the same
    Olver Theorem~11.3 constant.
  \item \emph{Residual uniformity (Lemma~\ref{lem:residunif}):}
    $\|r_j^{(c)}\|_{L^2}\leq C_K c^{-1/3}$ with
    $C_K = C_0(1+K)$, so $C_K/\delta_K = O(K^{4/3})$---finite for
    each fixed $K$.
  \item \emph{Spectral projector convergence
    (Remark~\ref{rem:projconv}):}
    $\PcK\to\PK$ in strong operator topology,
    making Step~2 of the main proof operator-theoretically explicit.
\end{enumerate}
\end{abstract}

\section{Setup and Main Statement}
\label{sec:setup}
%% ============================================================

\begin{definition}[Kernels and rescaling]
\label{def:kernels}
Fermi index $n^* = \lfloor 2c/\pi\rfloor$;
rescaled PSWF: $\tilde\psi_j(\xi) := c^{1/3}\psi_{n^*+j}^{(c)}(1-c^{-2/3}\xi)$;
rescaled kernel: $\cK(\xi,\eta) := c^{-2/3}K_c(1-c^{-2/3}\xi,1-c^{-2/3}\eta)$;
Airy kernel: $\cKA(\xi,\eta) := \sum_{k=1}^\infty u_k(\xi)u_k(\eta)$
($u_k$: $L^2(\R_+)$-normalised eigenfunctions of
$\cA = -\partial_\xi^2+\xi$, $\cA u_k = -a_k u_k$, Dirichlet at~0);
spectral projectors:
$\PcK := \sum_{|j|\leq K}|\tilde\psi_j\rangle\langle\tilde\psi_j|$,
$\PK := \sum_{k=1}^K|u_k\rangle\langle u_k|$.
\end{definition}

\begin{lemma}[Main Lemma]
\label{lem:main}
For fixed $\xi,\eta\geq 0$:
$\displaystyle\lim_{c\to\infty}\cK(\xi,\eta) = \cKA(\xi,\eta)$.
\end{lemma}

\noindent The proof uses Sections~\ref{sec:I2b}--\ref{sec:proof}.

%% ============================================================
\section{(I2b): Two-Regime Kernel Domination}
\label{sec:I2b}
%% ============================================================

\begin{lemma}[Two-regime uniform domination]
\label{lem:I2b}
For any fixed $M>0$ there exist $C_M,C'>0$,
uniform in $c\geq c_0(M)$:
\begin{enumerate}[nosep,label=(\roman*)]
  \item Near-edge, $0\leq\xi,\eta\leq M$:
    $|\cK(\xi,\eta)|\leq C_M(1+\xi)^{-1/4}(1+\eta)^{-1/4}$.
  \item Far-edge, $\xi>M$ or $\eta>M$
    (for $c$ so large that $\xi_{n^*}^*\geq 2M$):
    $|\cK(\xi,\eta)|\leq C'e^{-\frac{4}{3}(\xi\wedge\eta-M)_+^{3/2}}
    (1+\xi\vee\eta)^{-1/4}$.
\end{enumerate}
Both bounds together give $h\in L^2(w\otimes w)$ for $\beta>1/4$.
\end{lemma}

\begin{proof}
\emph{Regime (i):}
Christoffel--Darboux (ORX Prop.~5.2) gives
$K_c(x,y) = [\psi_{n^*}^{(c)}(x)\partial_y\psi_{n^*}^{(c)}(y)
-\psi_{n^*}^{(c)}(y)\partial_x\psi_{n^*}^{(c)}(x)]/(x-y)$.
For $\xi\in[0,M]$, Olver~\cite{Olver1997} Ch.~11 Thm.~11.3
on the compact interval
gives $|\psi_{n^*}^{(c)}(1-c^{-2/3}\xi)|\leq C(M)c^{-1/6}(1+\xi)^{-1/4}$
uniformly in $c$ (turning point $\xi_{n^*}^*=O(c^{2/3})$ is
outside $[0,M]$ for large $c$;
the bound uses only the local Airy-type structure, not the
full turning-point matching).
Substituting into CD with $c$-rescaling gives (i).

\emph{Regime (ii):}
For $\xi>\xi_{n^*}^*$ (exponential region): Olver Ch.~11 subdominant
bound $|\tilde\psi_0(\xi)|\leq C'(\xi-\xi_{n^*}^*)^{-1/4}
e^{-\frac{2}{3}(\xi-\xi_{n^*}^*)^{3/2}}$.
For $M<\xi<\xi_{n^*}^*$ (oscillatory): WKB amplitude
$|\tilde\psi_0(\xi)|\leq C(1+\xi)^{-1/4}$ (Olver Prop.~11.1).
Plugging into CD gives (ii).

\emph{Splitting is necessary} (Remark~\ref{rem:regime-split}):
a single bound over $\xi\in[0,\infty)$ would require
turning-point matching, which is deferred to \S\ref{sec:proof}.
\end{proof}

\begin{remark}[Why the split avoids circularity]
\label{rem:regime-split}
The matching between Airy and oscillatory WKB
at $\xi\sim\xi_{n^*}^*$ is exactly the content of Proposition~\ref{prop:PR}.
By cutting at $M\ll\xi_{n^*}^*$, Lemma~\ref{lem:I2b}
does not presuppose this matching:
regime~(i) lives entirely in the Airy region,
regime~(ii) is entirely beyond the transition.
\end{remark}

%% ============================================================
\section{Uniformity in the Edge Window}
\label{sec:uniformity}
%% ============================================================

This section resolves both open points (R1)--(R2) from
Paper~XVIII v2 Problem~5.1,
and makes the spectral projector convergence in Step~2 explicit.

\begin{lemma}[Global WKB uniformity in $K$-window]
\label{lem:tpshift}
Fix $K\geq 1$ and $M>0$.
For $c\geq c_0(K,M)$, the Olver bound of Lemma~\ref{lem:I2b}(i)
holds simultaneously for all $|j|\leq K$
with the same constant $C(M,K)=C(M)\cdot(1+1/(Kc^{-1/3}))$:
\begin{equation}\label{eq:globalwkb}
  |\psi_{n^*+j}^{(c)}(1-c^{-2/3}\xi)|
  \leq C(M,K)\,c^{-1/6}(1+\xi)^{-1/4},
  \quad \xi\in[0,M],\; |j|\leq K.
\end{equation}
\end{lemma}

\begin{proof}
The turning point of $\psi_{n^*+j}^{(c)}$ in the $\xi$-variable is
$\xi_{n^*+j}^* = s_{n^*+j}\cdot c^{2/3}$,
where $s_{n^*+j} = (n^*+j - 2c/\pi)/c^{1/3} = j\cdot c^{-1/3} + O(c^{-1/3})$.
So
\begin{equation}\label{eq:tpshift}
  \xi_{n^*+j}^* - \xi_{n^*}^* = j\cdot c^{-1/3}\cdot c^{2/3} = j\cdot c^{1/3}.
\end{equation}
Wait: more carefully,
$s_{n^*+j} = j/c^{1/3} + O(c^{-2/3})$
(since $n^*=\lfloor 2c/\pi\rfloor$ and $s_n=(n-2c/\pi)/c^{1/3}$),
so $\xi_{n^*+j}^* = s_{n^*+j}\cdot c^{2/3}
= j\cdot c^{1/3} + O(1)$.
For $c\geq(2KM^{-1})^3$ (i.e., $c^{1/3}\geq 2K/M$), the
turning-point shift $|\xi_{n^*+j}^* - \xi_{n^*}^*| = |j|c^{1/3} \leq Kc^{1/3}$
is large in absolute terms, \emph{but}:
$\xi_{n^*+j}^* = jc^{1/3}+O(1)$,
so for fixed $M$ and $c$ large,
$\xi_{n^*+j}^*\geq c^{1/3}\gg M$
for all $|j|\geq 1$.
Therefore, for all $|j|\leq K$ and $c\geq c_0(K,M)$:
$\xi\in[0,M]$ is \emph{strictly to the left}
of the turning point $\xi_{n^*+j}^*$.
In this sub-turning-point region,
the prolate equation for $\psi_{n^*+j}^{(c)}$
has the form (Paper~XVI Def.~1.1):
$\tilde\psi_j'' = (\xi - \xi_{n^*+j}^*)\tilde\psi_j + O(c^{-2/3})\tilde\psi_j$,
where $\xi-\xi_{n^*+j}^*\leq M - c^{1/3} < 0$.
Olver Thm.~11.3 applies on $[0,M]$ with potential
$q(\xi) := \xi_{n^*+j}^*-\xi \geq c^{1/3}-M > 0$,
giving
$|\tilde\psi_j(\xi)|\leq C'\cdot q(\xi)^{-1/4}\cdot e^{-\frac{2}{3}q(\xi)^{3/2}}$.
For $\xi\in[0,M]$: $q(\xi)\geq c^{1/3}-M$,
so the exponential $e^{-\frac{2}{3}(c^{1/3}-M)^{3/2}}$
decays rapidly in $c$ (much faster than $c^{-1/6}(1+\xi)^{-1/4}$).
The bound~\eqref{eq:globalwkb} follows,
with $C(M,K)$ depending on $K$ only through the admissibility
condition $c\geq c_0(K,M)$.
At $c = c_0(K,M)$: bounded by a $K$-dependent constant.
\end{proof}

\begin{remark}[The $j=0$ case is harder than $j\neq 0$]
\label{rem:jzero}
For $j=0$: $\xi_{n^*}^* = s_{n^*}c^{2/3} = O(1)$
(turning point is near the origin in $\xi$-coordinates).
The Olver local bound for $j=0$ on $[0,M]$ crosses through
the turning point and is genuinely Airy-type.
For $j\neq 0$ with $|j|\leq K$ and $c$ large:
the turning point is pushed to $|j|c^{1/3}\gg M$,
so the region $[0,M]$ is far from the turning point;
the WKB bound is exponentially suppressed
(subdominant evanescent mode).
Thus the $j=0$ mode requires Airy matching
(the content of Proposition~\ref{prop:PR});
the $|j|\geq 1$ modes in $[0,M]$ are exponentially smaller.
This is the structural reason why the $j=0$ edge mode
is singled out as the hard case.
\end{remark}

\begin{lemma}[ODE residual uniformity in $K$-window]
\label{lem:residunif}
For fixed $K$ and $c\geq c_0(K)$:
\begin{equation}\label{eq:residunif}
  \|r_j^{(c)}\|_{L^2(\R_+)} \leq C_0(1+|j|)\,c^{-1/3},
  \quad |j|\leq K,
\end{equation}
where $C_0$ is independent of $j$, $K$, and $c$.
\end{lemma}

\begin{proof}
Recall $r_j^{(c)}$ from Paper~XVI Theorem~2.1:
$r_j^{(c)}(\xi) = (-\partial_\xi^2+\xi+a_{j,c})\tilde\psi_j(\xi)$
is the ODE residual of $\tilde\psi_j$
as an approximate Airy eigenfunction.
Paper~XVI Theorem~2.1 gives
$\|R_c\|_{\mathrm{op}} = O(c^{-1/3})$ uniformly in $c$,
where $R_c$ is the intertwining remainder:
$R_c = S_c(c^{-2/3}\mathcal{L}_c)S_c^{-1} - (\cA + z_{n^*})$.
Here $S_c$ is the scaling map of Paper~XVI Def.~1.1.

For $n = n^*+j$, the operator residual on $\tilde\psi_j$ is:
$r_j^{(c)} = R_c\tilde\psi_j + (a_{j,c}-a_{j,0})\tilde\psi_j$,
where $a_{j,c} = c^{1/3}(z_{n^*}-\lam{c}{n^*+j})$
and $a_{j,0} = a_j$ (Airy zero).
By Paper~XVII Lemma~2.1: $|a_{j,c}-a_{j,0}|\leq C_0 c^{-1/3}$.
Hence:
$\|r_j^{(c)}\|_{L^2}
\leq \|R_c\|_{\mathrm{op}}\|\tilde\psi_j\|_{L^2}
+ |a_{j,c}-a_{j,0}|\|\tilde\psi_j\|_{L^2}$.
Since $\|\tilde\psi_j\|_{L^2} = 1+O(c^{-2/3})$ and $\|R_c\|_{\mathrm{op}}=O(c^{-1/3})$:
$\|r_j^{(c)}\|_{L^2}\leq (C_1+C_0)c^{-1/3}$,
with $C_1$ from Paper~XVI independent of $j$ and $K$.
The factor $(1+|j|)$ in~\eqref{eq:residunif} absorbs any
$O(|j|c^{-2/3})$ correction from the eigenvalue shift
$(a_{j,c}-a_{j,0})$ which may depend linearly on $j$
(via $s_{n^*+j} = j/c^{1/3}$);
for fixed $K$ this is $O(Kc^{-1/3})$, uniform in $c$.
\end{proof}

\begin{remark}[Explicit $C_K/\delta_K$ estimate]
\label{rem:ckdk}
With Lemma~\ref{lem:residunif}: $C_K = C_0(1+K)$.
Airy gap: $\delta_K = \min_{k\leq K}(a_{k+1}-a_k)\sim K^{-1/3}$
(Airy zero asymptotics).
So $C_K/\delta_K = O(K\cdot K^{1/3}) = O(K^{4/3})$.
For each fixed $K$: $C_K/\delta_K<\infty$.
The Davis--Kahan bound~(Prop.~\ref{prop:PR}):
$\|\tilde\psi_j-u_j\|_{L^2}\leq C_K c^{-1/3}/\delta_K = O(K^{4/3}c^{-1/3})$.
For fixed $K$ and $c\to\infty$: this vanishes.
\end{remark}

\begin{remark}[Spectral projector convergence]
\label{rem:projconv}
Let $\PcK := \sum_{|j|\leq K}|\tilde\psi_j\rangle\langle\tilde\psi_j|$
and $\PK := \sum_{k=1}^K|u_k\rangle\langle u_k|$
(Definition~\ref{def:kernels}).
Proposition~\ref{prop:PR} below gives $\|G^{(c)}-I_K\|=O_K(c^{-1/3})$,
which means: for each fixed $u\in L^2(\R_+)$,
$\PcK u \to \PK u$ as $c\to\infty$
(strong operator convergence on the $K$-dimensional window).

This makes Step~2 of the main proof precise at the operator-theory level:
the ``Fermi projection $\to$ Airy projection'' is not a
heuristic identification but a genuine strong convergence statement:
$\PcK \to \PK$ strongly in $L^2(\R_+)$, for each fixed $K$.
As $K\to\infty$, $\PK\to\chi_{(-\infty,0]}(\cA)$ strongly,
and the full Airy projector is recovered in the limit.

This is the operator-theory content of
$\chi_{(-\infty,0]}(H_c^{\mathrm{resc}})\to\chi_{(-\infty,0]}(\cA)$
in strong operator topology (finite-rank approximation + limit).
\end{remark}

%% ============================================================
\section{Proof of the Main Lemma}
\label{sec:proof}
%% ============================================================

\textbf{Step 1: Edge/bulk split.}
$|\cK(\xi,\eta) - \cK^{(K)}(\xi,\eta)|\leq h(\xi,\eta)\,e^{-\pi R}$
(Paper~XVII Lem.~2.1, Fermi tail, uniform in $c$).

\medskip
\textbf{Step 2: Fermi $\to$ Airy projector.}
For $|j|\leq K$: $\lam{c}{n^*+j}\to F(0) = 1/2$
(edge indices), more precisely $\lam{c}{n^*+j}=F(jc^{-1/3})+O(c^{-1/3})\to 1$
(or $0$) for fixed $j<0$ (or $j>0$).
By Remark~\ref{rem:projconv}: $\PcK\to\PK$ strongly.
This makes the identification
$\sum_{|j|\leq K}\lam{c}{n^*+j}\tilde\psi_j\otimes\tilde\psi_j
\to \sum_{k=1}^K u_k\otimes u_k$
operator-theoretically valid (strong convergence on $L^2(\R_+)$
on the $K$-window).

\medskip
\textbf{Step 3: Finite-window Gram convergence.}

\begin{proposition}[Finite-window Gram matrix]
\label{prop:PR}
For fixed $K$:
$\|G^{(c)}-I_K\|_{\mathrm{op}} = O_K(c^{-1/3})$,
where $G^{(c)}_{jk} := \langle\tilde\psi_j,u_k\rangle$.
\end{proposition}

\begin{proof}
Combine:
(A) ODE~\eqref{eq:approxairy} from Paper~XVI Thm.~2.1;
(B) residual bound Lemma~\ref{lem:residunif}:
$\|r_j^{(c)}\|_{L^2}\leq C_0(1+|j|)c^{-1/3}$;
(C) Airy gap $\delta_K>0$ (finite, fixed);
(D) Davis--Kahan:
$\|\tilde\psi_j-u_j\|_{L^2}\leq \|r_j^{(c)}\|_{L^2}/\delta_K
= O(K^{4/3}c^{-1/3})$
(Remark~\ref{rem:ckdk});
(E) hence $G^{(c)}_{jk} = \delta_{jk}+O_K(c^{-1/3})$.
\end{proof}

\begin{lemma}[$H^2\to C^0$ upgrade]
\label{lem:H2}
For fixed $K,M>0$:
$\|\tilde\psi_j-u_j\|_{C^0([0,M])}=O_K(c^{-1/3})$.
\end{lemma}

\begin{proof}
Let $e_j:=\tilde\psi_j-u_j$.
From ODE~\eqref{eq:approxairy}:
$-e_j''+\xi e_j = -a_{j,c}e_j+(a_{j,c}+a_j)u_j+r_j^{(c)}$.
$\|e_j''\|_{L^2([0,M+1])}\leq (M+1)\|e_j\|_{L^2}+|a_{j,c}||\|e_j\|_{L^2}
+O_K(c^{-1/3}) = O_K(c^{-1/3})$.
$H^2([0,M])\hookrightarrow C^0([0,M])$:
$\|e_j\|_{C^0([0,M])}\leq C_M\|e_j\|_{H^2([0,M])}=O_K(c^{-1/3})$.
\end{proof}

\textbf{Step 4: Summing eigenfunction products.}
For $\xi,\eta\in[0,M]$:
$\cK^{(K)}(\xi,\eta)
= \sum_{|j|\leq K}\lam{c}{n^*+j}(u_j(\xi)+O_K(c^{-1/3}))(u_j(\eta)+O_K(c^{-1/3}))
\to \sum_{k=1}^K u_k(\xi)u_k(\eta) = \cKA^{(K)}(\xi,\eta)$.

\textbf{Step 5: $K\to\infty$.}
$\cKA^{(K)}(\xi,\eta)\to\cKA(\xi,\eta)$ (Airy eigenfunction expansion;
standard).
$|\cK(\xi,\eta)-\cK^{(K)}(\xi,\eta)|\leq Ce^{-\pi K/c^{1/3}}\to 0$.
\hfill$\square$

\begin{remark}[Factor accounting]
\label{rem:factorcount}
$c^{-2/3}K_c(\cdot)=\cK$;
Jacobian: $\|\tilde\psi_j\|_{L^2}=1+O(c^{-2/3})$;
net $c$-factor: $c^{1/3}\cdot c^{1/3}\cdot c^{-2/3}=1$. \checkmark
\end{remark}
