\section{Setup}
\label{sec:setup}

\begin{definition}[Rescaled coordinate]
\label{def:airycoord}
$\xit = c^{2/3}(x_{n^*}^*-x)$,
$(\Uc f)(\xit) = c^{1/3}f(x_{n^*}^*-c^{-2/3}\xit)$,
$\tilde\psi_j = \Uc\psi_{n^*+j}^{(c)}$,
$\cK(\xit,\xit') = c^{-2/3}K_c(x_{n^*}^*-c^{-2/3}\xit,
x_{n^*}^*-c^{-2/3}\xit')$.
Airy window: $c^{-2/3}$.
Turning-point shift $\xit_{n^*+j}^* = -j\pi/(2c^{1/3})$
(XVI Lem.~3.2); $\xit_{n^*+j}^*\to 0$ for fixed $j$.
\end{definition}

\begin{definition}[Airy operator]
\label{def:airyop}
$\cA = -d^2/d\xit^2+\xit$ on $L^2(\R_+)$,
$\mathcal{D}(\cA)=H^2\cap H^1_0$;
eigenvalues $0<a_1<a_2<\cdots$,
eighenfunctions
$u_k = \Ai(\cdot-a_k)/\normh{\Ai(\cdot-a_k)}{L^2(\R_+)}$.
\end{definition}

\begin{remark}[Compact embeddings]
\label{rem:compact}
On bounded $I=[0,M]\subset\R$, $n=1$:
$H^2(I)\hookrightarrow H^1(I)\hookrightarrow C^0(I)\hookrightarrow L^2(I)$,
all compact (Rellich--Kondrachov);
$\normh{f}{L^\infty(I)}\leq C_M\normh{f}{H^1(I)}$ (Morrey, $n=1$).
Used in Lemmas \ref{lem:langer},\,\ref{lem:linf},\,\ref{lem:H2}
without further comment.
\end{remark}

\begin{proposition}[Airy kernel + Mercer diagonal]
\label{prop:kernelbridge}
$\cKA(\xit,\xit') = \sum_k u_k(\xit)u_k(\xit')
= \int_0^\infty\Ai(\xit+s)\Ai(\xit'+s)\,ds$.
Diagonal: $\cKA(\xit,\xit) = \sum_k u_k(\xit)^2\in L^1(\R_+)$
(since $\cA|_{[0,M]}$ is trace-class for each $M$;
Simon~\cite{Simon2005}).
$\cKA\in L^1(\R_+^2)$ (Airy decay $|\Ai(s)|\leq Ce^{-\frac23s^{3/2}}$).
\end{proposition}

\begin{lemma}[Main Lemma]\label{lem:main}
For fixed $\xit,\xit'\geq 0$:
$\cK(\xit,\xit')\to\cKA(\xit,\xit')$.
\end{lemma}

\begin{remark}[Feedback-stable dependency in Lemma~\ref{lem:sturmunif}]
\label{rem:feedback}
Lemma~\ref{lem:sturmunif} depends on Corollary~\ref{cor:spectral}(i--iii),
which depends on Proposition~\ref{prop:normresolvent}.
Lemma~\ref{lem:sturmunif} also depends on Lemma~\ref{lem:H2},
which depends on Lemma~\ref{lem:langer}.
This gives the path:
\[
  \text{Lem.~\ref{lem:slbound}}
  \to\text{Lem.~\ref{lem:langer}}
  \to\text{Lem.~\ref{lem:H2}}
  \to\text{Lem.~\ref{lem:sturmunif}}
  \to\text{Cor.~\ref{cor:spectral}(i--iii)}
  \to\text{Prop.~\ref{prop:normresolvent}}
  \not\to\text{Lem.~\ref{lem:sturmunif}}.
\]
This is not a cycle:
Proposition~\ref{prop:normresolvent} depends only on
Proposition~\ref{prop:formbounded} and external references;
it does not depend on Lemma~\ref{lem:sturmunif}.
The path terminates.
The ``feedback'' appearance arises because Lemma~\ref{lem:sturmunif}
consumes the output of the spectral convergence it logically
\emph{stabilizes} (zero-count identification),
but the proof-order dependence is strictly acyclic:
$\text{Lem.~\ref{lem:H2}}$ (proved first)
$\Rightarrow$ $\text{zero-count stable}$
$\Rightarrow$ $\text{Lem.~\ref{lem:sturmunif}}$ holds
without requiring Corollary~\ref{cor:spectral}(iv).
\end{remark}

\subsection{Lemma~0: Global eigenvalue bound}

\begin{lemma}[Global eigenvalue bound]
\label{lem:slbound}
$a_{j,c}=a_j+O_j(c^{-1/3})$;
$a_{j,c}\leq C_K:=a_K+1$ for $c\geq c_0(K)$.
\end{lemma}

\begin{proof}
$\qa_c[f]=\qa_0[f]+c^{-1/3}\langle W_cf,f\rangle$ on $H^1_0(\R_+)$;
Hardy--Rellich: $|\langle W_cf,f\rangle|\leq\varepsilon\qa_0[f]+C_\varepsilon\normh{f}{}^2$;
Kato~\cite{Kato1966} VI.2.21 + RS XIII.1:
$a_{j,c}=a_j+O_j(c^{-1/3})$, uniformly on $H^1_0(\R_+)$.
\end{proof}

\begin{remark}[Uniformity chain]
\label{rem:uniform}
All parameter dependencies are explicit and one-directional:
\[
  K
  \;\xmapsto{\text{Lem.~\ref{lem:slbound}}}\;
  C_K = a_K+1
  \;\xmapsto{\text{Rem.~\ref{rem:coercep}}}\;
  \delta_0(K) = 2C_K+2
  \;\xmapsto{\text{Lem.~\ref{lem:langer}}}\;
  c_0(K,M).
\]
Consequence:
\begin{enumerate}[nosep,label=(\roman*)]
  \item $C_K$ depends on $K$ only.
  \item $\delta_0=\delta_0(K)$ depends on $K$ only;
    the coercivity region $[\delta_0(K),M]$ grows with $K$.
    This is not a gap: for \emph{fixed} $K$,
    all estimates in Lemma~\ref{lem:langer} are
    uniform in $j\leq K$ and $c\geq c_0(K,M)$.
  \item All constants in Lemma~\ref{lem:langer}
    (Langer error, $H^{-1}$ bound, $H^1$ bound)
    are written $C(M,K)$, $C'(M,K)$, $c_0(M,K)$;
    they do not depend on the individual index $j\leq K$.
  \item The Main Lemma holds for each fixed
    $\xit,\xit'$ after choosing $K=K(\varepsilon)$;
    the chain terminates.
\end{enumerate}
\end{remark}

\begin{remark}[Spectral bound vs.\ coercivity]
\label{rem:coercep}
Lemma~\ref{lem:slbound}: \emph{global spectral information}
$a_{j,c}\leq C_K$, from form comparison on $H^1_0(\R_+)$.
Lemma~\ref{lem:langer} Step~2: \emph{local form inequality}
$q_c(\xit)\geq\delta_0/2$ on $[\delta_0,M]$,
using $a_{j,c}\leq C_K$ as a fixed numerical input only.
These are logically separate; no circularity.
\end{remark}

\subsection{Lemma~A: Uniform Airy approximation}

\begin{lemma}[Langer + coercive patch]
\label{lem:langer}
$\tilde\psi_j=c_j[\Ai(\cdot-\xit_{n^*+j}^*)+E_j^{(c)}]$,
$\sup_{[0,M]}|E_j^{(c)}|\leq C'(M,K)c^{-1/3}$,
$c_j=1+O(C(M,K)\cdot c^{-1/3})$;
uniform in $j\leq K$ and $c\geq c_0(M,K)$.
\end{lemma}

\begin{proof}
$\delta_0=\delta_0(K)=2C_K+2$, $I_c=[a_{j,c}^*-c^{-2/3},a_{j,c}^*+c^{-2/3}]$.

\textbf{Step 1 (Olver on $I_c$).}
$\normh{c^{-1/3}W_c}{L^1(I_c)}\leq 2Cc^{-1}$;
Olver~\cite{Olver1997} Thm.~11.3:
$\sup_{I_c}|E_j^{(c)}|=O(c^{-1})$, uniform in $j\leq K$.

\textbf{Step 2 (Local form inequality on $[\delta_0(K),M]$).}
$q_c(\xit)\geq\xit-C_K-c^{-1/3}C(1+M)^{1/2}
\geq\delta_0-C_K-c^{-1/3}C(1+M)^{1/2}\geq C_K+1>0$
for $c\geq c_0(K,M)$.
Hence $\qa_c[f]\geq\mu(K)\normh{f}{H^1}^2$,
$\mu(K)=\min(1,C_K+1)>0$, as Hilbert-space form inequality
(Remark~\ref{rem:coercep}).

\textbf{Step 3 ($H^{-1}$ via $H^1_0$-$L^\infty$-duality, $n=1$).}
For $\phi\in H^1_0([\delta_0,M])$:
$|\langle W_c\tilde\psi_j,\phi\rangle|
\leq\normh{W_c}{L^2}\normh{\tilde\psi_j}{L^2}\normh{\phi}{L^\infty}
\leq C_M\cdot C_{M,K}\normh{\phi}{H^1}$;
hence $\normh{W_c\tilde\psi_j}{H^{-1}}\leq C(M,K)$;
Lax--Milgram: $\normh{e_j}{H^1}\leq C(M,K)c^{-1/3}/\mu(K)$;
Sobolev: $\sup|e_j|\leq C'(M,K)c^{-1/3}$.

\textbf{Normalization.}
$N'(a)=-2\Ai(-a)^2$;
$c_j=1+O(C(K)\cdot c^{-1/3})$, uniform in $j\leq K$.
\end{proof}

\begin{lemma}[Uniform $L^\infty$ bound]
\label{lem:linf}
$\normh{\tilde\psi_j}{L^\infty([0,M])}\leq C(M,K)$,
uniform in $j\leq K$, $c\geq c_0(M,K)$.

$\normh{\tilde\psi_j}{H^2([0,M])}\leq C(M,K)$

Rellich--Kondrachov + Sobolev:
$\normh{\tilde\psi_j}{L^\infty([0,M])}\leq C(M,K)$.
\end{proof}

\section{Spectral Convergence and Resolvent}
\label{sec:nrc}
%% ============================================================

\begin{proposition}[Norm-resolvent]
\label{prop:normresolvent}
$(\Lc-z)^{-1}-(\cA-z)^{-1}
= -(\cA-z)^{-1}c^{-1/3}R_c(\Lc-z)^{-1}$;
$\normh{\cdot}{}\leq C_0c^{-1/3}/(\mathrm{Im}\,z)^2$;
$P_{j,c}\to P_j$ strongly (RS~\cite{ReedSimon4} VIII.23--24).
\end{proposition}

\begin{corollary}[Spectral convergence]
\label{cor:spectral}
(i) $|a_{j,c}-a_j|=O_j(c^{-1/3})$;
(ii) $\normh{P_{j,c}-P_j}{}=O_j(c^{-1/3})$;
(iii) $\normh{\PcK-\PK}{}=O_K(c^{-1/3})$;
(iv) $\normh{\tilde\psi_j}{H^1}\leq C_K$
(from Lem.~\ref{lem:langer} directly, not from norm-resolvent).
\end{corollary}

\begin{lemma}[$C^0$ convergence]
\label{lem:H2}
$\normh{\tilde\psi_j-u_j}{C^0([0,M])}=O(C(M,K)\cdot c^{-1/3})$.
\end{lemma}
\begin{proof}
Lemma~\ref{lem:langer}: $c_j\to 1$,
$\xit_{n^*+j}^*\to 0$, $\sup|E_j^{(c)}|=O_{M,K}(c^{-1/3})$.
\end{proof}

\begin{lemma}[Spectral sum DCT]
\label{lem:domconv}
For fixed $\xit,\xit'\in[0,M]$:
$\sum_{j=1}^K\tilde\psi_j(\xit)\tilde\psi_j(\xit')
\xrightarrow{c\to\infty}\sum_{j=1}^K u_j(\xit)u_j(\xit')
\xrightarrow{K\to\infty}\cKA(\xit,\xit')$.
\end{lemma}

\begin{proof}
Fixed $K$: Lemma~\ref{lem:H2}.
$K\to\infty$: Lemma~\ref{lem:slepian} + Remark~\ref{rem:dct}.
\end{proof}

\begin{remark}[Two-layer DCT: Tonelli/Mercer-safe]
\label{rem:dct}
Set $\SK(\xit,\xit'):=\sum_{j=1}^K u_j(\xit)u_j(\xit')$.

\medskip
\textbf{Layer 1 (domination in $K$, uniform in $c$).}
$|\sum_{j=1}^K\tilde\psi_j(\xit)\tilde\psi_j(\xit')|\leq K\cdot C(M,K)^2$
(Lemma~\ref{lem:linf});
controls the $c\to\infty$ limit for each fixed $K$.

\medskip
\textbf{Layer 2 ($K\to\infty$; Tonelli/Mercer domination).}
We need: $\SK\to\cKA$ in $L^1([0,M]^2)$ with an integrable majorant.
By Cauchy--Schwarz applied to the spectral sum:
\begin{equation}\label{eq:CSkernel}
  |\SK(\xit,\xit')|^2
  \leq\SK(\xit,\xit)\cdot\SK(\xit',\xit')
  \leq\cKA(\xit,\xit)\cdot\cKA(\xit',\xit'),
\end{equation}
where $\SK(\xit,\xit)=\sum_{j=1}^K u_j(\xit)^2
\leq\cKA(\xit,\xit)\in L^1(\R_+)$
(Proposition~\ref{prop:kernelbridge};
diagonal $\cKA(\xit,\xit)\in L^1$ since $\cA|_{[0,M]}$ trace-class).
Set $h(\xit,\xit'):=[\cKA(\xit,\xit)\cdot\cKA(\xit',\xit')]^{1/2}$.
Then by Tonelli:
\[
  \int_{\R_+^2}h\,d\xit\,d\xit'
  = \Bigl(\int_{\R_+}\cKA(\xit,\xit)^{1/2}\,d\xit\Bigr)^2
  < \infty,
\]
hence $h\in L^1(\R_+^2)$.
By~\eqref{eq:CSkernel}: $|\SK(\xit,\xit')|\leq h(\xit,\xit')$
for all $K$, pointwise.

\medskip
Since $\SK(\xit,\xit')\nearrow\cKA(\xit,\xit')$ pointwise
(Mercer~\cite{Mercer1909}, or Parseval for $\cA$),
Lebesgue dominated convergence with dominant $h\in L^1$
gives $\SK\to\cKA$ in $L^1(\R_+^2)$.

\medskip
This argument is Tonelli-safe:
it uses only the $L^1$-integrability of the diagonal
$\cKA(\xit,\xit)\in L^1(\R_+)$,
not any positivity of $\cKA$ as a majorant;
no Mercer theorem beyond the Parseval identity for $\cA$
is required.
\end{remark}

  +\,
  \underbrace{\text{Sturm}_{\mathrm{fs}}}_{{\text{Lem.~\ref{lem:sturmunif}}}}
  +\,
  \underbrace{h\in L^1}_{\text{Rem.~\ref{rem:dct}}}
  \Longrightarrow\mathrm{BR3}
}
