$\cK(\xi,\eta)\to\cKA(\xi,\eta)$ pointwise,

    explicit separation $n^* = 2c/\pi + \delta_c$, $\delta_c\in[-1,0)$;

  \item \emph{LaTeX fix:} \verb|\\eta| $\to$ \verb|\eta|

\end{enumerate}

\section{Setup and Coordinate Systems}
\label{sec:setup}
%% ============================================================

\begin{definition}[Airy coordinate and identification map]
\label{def:setup}
\emph{Airy coordinate:} $\xi = c^{2/3}(1-x)\geq 0$,
so $x = 1-c^{-2/3}\xi$.\quad
\emph{Identification map:}
$({\Uc} f)(\xi) := c^{1/3}f(1-c^{-2/3}\xi)$;
${\Uc}:L^2([-1,1])\to L^2(\R_+)$ with
$\|{\Uc}\| = 1+O(e^{-\pi c})$ (Jacobian substitution).\quad
\emph{Rescaled PSWF:} $\tilde\psi_j := {\Uc}\psi_{n^*+j}^{(c)}$.\quad
\emph{Rescaled kernel (LaTeX v7 fix):}
\begin{equation}\label{eq:cKdef}
  \cK(\xi,\eta) := c^{-2/3}K_c(1-c^{-2/3}\xi,\,1-c^{-2/3}\eta).
\end{equation}
\emph{Airy operator:} $\cA = -\partial_\xi^2+\xi$ on $L^2(\R_+)$,
domain $H^2(\R_+)\cap H^1_0(\R_+)$ (Dirichlet BC at~$0$);
$\cA$ is positive self-adjoint with compact resolvent.
Eigenvalues: $0<a_1<a_2<\cdots$,
$a_k\sim(\frac{3\pi}{2}(k-\frac14))^{2/3}$ (Weyl).
Eigenfunctions: $\cA u_k = a_k u_k$, $\{u_k\}$ ONB of $L^2(\R_+)$.
\end{definition}

\begin{definition}[Fermi index and floor separation]
\label{def:tp}
Write $n^* = \lfloor 2c/\pi\rfloor =: 2c/\pi + \delta_c$
where $\delta_c\in[-1,0)$ is the \emph{floor residual}.
Note $|\delta_c|<1$ uniformly in $c$.

The physical turning point of $\psi_{n^*+j}^{(c)}$ is at
$x_{n^*+j}^* = (n^*+j)\pi/(2c)$ (PSWF Sturm--Liouville,
Paper~XVI Def.~1.1).
In Airy coordinates:
\begin{align}
  \xi_{n^*+j}^*
  &= c^{2/3}(1-x_{n^*+j}^*)
  = c^{2/3}\Bigl(1 - \frac{(n^*+j)\pi}{2c}\Bigr)
  = c^{2/3}\cdot\frac{2c/\pi - n^* - j}{c}\cdot\frac{\pi}{2}
    \notag\\
  &= \frac{\pi}{2}\cdot\frac{2c/\pi - n^* - j}{c^{1/3}}
  = \frac{\pi}{2}\cdot\frac{-\delta_c - j}{c^{1/3}}
  \cdot c^{0}
  = -\frac{\pi}{2}(\delta_c+j)\cdot c^{-1/3}
  \cdot c^{2/3-1/3}\notag
\end{align}
More carefully:
\begin{equation}\label{eq:tp}
  \xi_{n^*+j}^*
  = \frac{\pi}{2}\cdot\frac{-\delta_c - j}{c^{1/3}}
  + O(c^{-4/3}),
\end{equation}
where:
\begin{itemize}[nosep]
  \item The term $\delta_c\in[-1,0)$ is $O(1)$ (not $O(c^{-1/3})$).
  \item For $j=0$: $\xi_{n^*}^* = -\frac\pi2\delta_c/c^{1/3} = O(c^{-1/3})$.
  \item For $j\neq 0$: $\xi_{n^*+j}^* = -\frac\pi2(j+\delta_c)/c^{1/3}+O(c^{-4/3})$.
\end{itemize}
In all cases $\xi_{n^*+j}^* = O(c^{-1/3})\to 0$ for fixed $j$.
But the $O(1)$ part of $\delta_c$ is divided by $c^{1/3}$,
so $|\xi_{n^*+j}^*|\leq \frac\pi2(|j|+1)c^{-1/3}+O(c^{-4/3})$,
which $\to 0$ as $c\to\infty$ for fixed $j$.
\end{definition}

\begin{remark}[Turning-point spacing --- corrected claim]
\label{rem:cluster}
Equation~\eqref{eq:tp} shows that for each fixed $j$,
$\xi_{n^*+j}^*\to 0$ in Airy coordinates.
However, the \emph{spacing} between adjacent turning points is
$\xi_{n^*+j+1}^*-\xi_{n^*+j}^* = -\frac\pi2 c^{-1/3}+O(c^{-4/3})\to 0$.
The turning points are $O(c^{-1/3})$-spaced, not $O(1)$-spaced.
In the Airy zero picture: the $k$-th Airy zero $a_k = O(k^{2/3})$
is $O(1)$-spaced for fixed $k$; this $O(1)$-spacing is in the
\emph{spectral} (eigenvalue) variable, not in the geometric ($\xi$) variable.

\textbf{Corrected formulation:}
\begin{quote}
  The turning points $\xi_{n^*+j}^*\to 0$ for fixed $j$,
  with pairwise spacing $O(c^{-1/3})\to 0$,
  consistent with the Airy eigenvalue spacing $a_{k+1}-a_k = O(k^{-1/3})$
  after rescaling by $c^{1/3}$.
  There is no degenerate multi-turning-point structure;
  each mode has a simple turning point.
\end{quote}
\end{remark}

\begin{proposition}[Airy kernel via Green's function]
\label{prop:airykernel}
The Airy kernel equals the eigenfunction expansion:
\begin{equation}\label{eq:airykernel}
  \cKA(\xi,\eta) = \sum_{k=1}^\infty u_k(\xi)u_k(\eta),
\end{equation}
with uniform convergence on compact sets
and $L^2(\R_+\times\R_+)$-convergence.
\end{proposition}

\begin{proof}
Let $R_z := (\cA-z)^{-1}$ for $z\notin\R_+$.
The resolvent kernel $R_z(\xi,\eta)$ (Green's function of $\cA-z$)
is given explicitly by two linearly independent Airy solutions
matched at $\xi=\eta$ with the correct Wronskian:
\begin{equation}\label{eq:greensfunction}
  R_z(\xi,\eta) = \frac{\Ai(\xi-z)\Bi(\eta-z) - \Bi(\xi-z)\Ai(\eta-z)}{W_z}
  \quad (\xi<\eta),
\end{equation}
where $W_z$ is the Wronskian (Olver~\cite{Olver1997} Ch.~11 \S11.5;
the Dirichlet BC at $0$ selects the solution vanishing at $0$).
By the spectral theorem for compact-resolvent operators
(Reed--Simon~\cite{ReedSimon4} Thm.~VI.16):
\begin{equation}\label{eq:spectralresolution}
  R_z(\xi,\eta) = \sum_{k=1}^\infty \frac{u_k(\xi)u_k(\eta)}{a_k - z}.
\end{equation}
The spectral projection $P_{[0,\Lambda]}$ onto eigenvalues $\leq\Lambda$
is recovered by the contour integral
\begin{equation}\label{eq:contourint}
  P_{[0,\Lambda]}(\xi,\eta)
  = \frac{1}{2\pi i}\oint_\gamma R_z(\xi,\eta)\,dz
  = \sum_{a_k\leq\Lambda} u_k(\xi)u_k(\eta),
\end{equation}
where $\gamma$ encircles $\{a_1,\ldots,a_k:a_k\leq\Lambda\}$.
Taking $\Lambda\to\infty$ and using completeness
($\sum_{k=1}^\infty u_k\otimes u_k = \mathrm{Id}$ in $L^2(\R_+)$):
\eqref{eq:airykernel} holds in $L^2$ and, by the Sobolev bound
$\|u_k\|_{C^0([0,M])}\leq C\|u_k\|_{H^1}$, uniformly on compacts.
\end{proof}

\begin{lemma}[Main Lemma]
\label{lem:main}
For fixed $\xi,\eta\geq 0$:
$\displaystyle\lim_{c\to\infty}\cK(\xi,\eta) = \cKA(\xi,\eta)$.
\end{lemma}

\noindent Proof in Sections~\ref{sec:nrc}--\ref{sec:proof}.

\section{(I2b): Two-Regime Domination}
\label{sec:I2b}
%% ============================================================

\begin{lemma}[Kernel domination]
\label{lem:I2b}
For fixed $M>0$, $K\geq 1$, there exist $C_{M,K},C'>0$
uniform in $c\geq c_0(M,K)$:
\begin{enumerate}[nosep,label=(\roman*)]
  \item $\xi,\eta\in[0,M]$:\quad
    $|\cK(\xi,\eta)|\leq C_{M,K}(1+\xi)^{-1/4}(1+\eta)^{-1/4}$.
  \item $\xi>M$ or $\eta>M$:\quad
    $|\cK(\xi,\eta)|\leq C'e^{-\frac{4}{3}(\xi\wedge\eta-M)_+^{3/2}}$.
\end{enumerate}
\end{lemma}

\begin{proof}
CD identity (ORX Prop.~5.2, algebraic):
$K_c(x,y) = [\psi_{n^*}^{(c)}(x)\partial_y\psi_{n^*}^{(c)}(y)
- \psi_{n^*}^{(c)}(y)\partial_x\psi_{n^*}^{(c)}(x)]/(x-y)$.

\emph{Regime (i).}
The rescaled potential $V_c(\xi)=\xi+O(c^{-1/3})$
on $[0,M]$ (from~\eqref{eq:tp}: turning points at $O(c^{-1/3})\to 0$,
so $V_c(\xi)=\xi-\xi_{n^*+j}^*\geq\xi-O(c^{-1/3})$ for $\xi\in[0,M]$).
WKB energy conservation on the compact set $[0,M]$:
$|\tilde\psi_j|^2\cdot|V_c'|/|V_c|^{3/2}\sim\text{const}$
(Olver~\cite{Olver1997} Ch.~11 \S11.3, oscillatory regime),
giving $|\tilde\psi_j(\xi)|\leq C(1+\xi)^{-1/4}$
uniformly in $c\geq c_0$ and $|j|\leq K$.
Substituting into CD: (i) holds with $C_{M,K}=2KC^2$.

\emph{Regime (ii).}
For $\xi>M\gg 1$ and all turning points at $O(c^{-1/3})\ll M$:
$V_c(\xi)=\xi+O(c^{-1/3})>\xi/2>0$,
subdominant Airy solution decays as
$e^{-\frac23\xi^{3/2}}$ (Olver Ch.~11); (ii) follows.
\end{proof}

\section{Norm-Resolvent Convergence}
\label{sec:nrc}
%% ============================================================

\begin{proposition}[Norm-resolvent convergence]
\label{prop:normresolvent}
Let $\Lc := {\Uc}\mathcal{L}_c^{\mathrm{resc}}{\Uc}^{-1}$ on $L^2(\R_+)$.
For $z\in\C\setminus\R$:
\begin{equation}\label{eq:nrc}
  \|(\Lc-z)^{-1} - (\cA-z)^{-1}\|_{L^2\to L^2}
  \leq \frac{C_0\,c^{-1/3}}{(\mathrm{Im}\,z)^2}.
\end{equation}
\end{proposition}

\begin{proof}
\emph{Intertwining identity} (Paper~XVI Thm.~2.1):
$\Lc = \cA + c^{-1/3}R_c$, $\|R_c\|_{L^2\to L^2}\leq C_0$.
The operator $R_c = -(V_c(\xi)-\xi)/c^{-1/3}$
is the rescaled potential perturbation;
$\|V_c(\xi)-\xi\|_{L^\infty([0,M])}=O(c^{-1/3})$
from Definition~\ref{def:tp} (all $\xi_{n^*+j}^*=O(c^{-1/3})$)
and Taylor expansion of the PSWF potential in Paper~XVI.

\emph{Second resolvent identity:}
$(\Lc-z)^{-1}-(\cA-z)^{-1}
= -(\Lc-z)^{-1}\cdot c^{-1/3}R_c\cdot(\cA-z)^{-1}$.

\emph{Norm bound:}
$\|(\Lc-z)^{-1}\|\leq|\mathrm{Im}\,z|^{-1}$,
$\|(\cA-z)^{-1}\|\leq|\mathrm{Im}\,z|^{-1}$ (self-adjointness).
Hence~\eqref{eq:nrc} with $C_0$ from Paper~XVI Thm.~2.1.
\end{proof}

\begin{lemma}[Min-max index identification]
\label{lem:minmax}
For each $j\geq 1$:
\begin{equation}\label{eq:minmax}
  a_{j,c} = \min_{\substack{V\subset L^2(\R_+)\\\dim V = j}}
  \max_{\substack{v\in V\\\|v\|=1}}
  \langle v, \Lc v\rangle
  \quad\xrightarrow{c\to\infty}\quad
  a_j = \min_{\substack{V\subset L^2(\R_+)\\\dim V = j}}
  \max_{\substack{v\in V\\\|v\|=1}}
  \langle v, \cA v\rangle.
\end{equation}
In particular, $a_{1,c}<a_{2,c}<\cdots$ (strictly increasing,
same ordering as $a_1<a_2<\cdots$), and $|a_{j,c}-a_j|=O(c^{-1/3})$.
\end{lemma}

\begin{proof}
Courant--Fischer min-max theorem
(Reed--Simon~\cite{ReedSimon1} Thm.~XIII.1):
$a_{j,c}$ and $a_j$ are both given by the min-max formula
for $\Lc$ and $\cA$ respectively (both self-adjoint, positive,
compact resolvent).
Since $\Lc = \cA + c^{-1/3}R_c$ with $\|R_c\|\leq C_0$:
for any $j$-dimensional $V$ and unit $v\in V$,
$|\langle v,\Lc v\rangle - \langle v,\cA v\rangle|
= c^{-1/3}|\langle v,R_c v\rangle|\leq C_0 c^{-1/3}$.
Hence the min-max values satisfy
$|a_{j,c}-a_j|\leq C_0 c^{-1/3}$
(standard perturbation of min-max values;
Kato~\cite{Kato1966} Ch.~IV Thm.~3.4 or
Weyl's inequality for self-adjoint operators).
Strict monotonicity $a_{1,c}<a_{2,c}<\cdots$ follows because
$\Lc$ has simple eigenvalues (simple turning point, Sturm--Liouville theory)
for all $c$ large; this can be seen from the positivity of the
Wronskian of the Airy solutions (Paper~XVI Rem.~1.2).
\end{proof}

\begin{corollary}[Spectral convergence]
\label{cor:spectral}
From Propositions~\ref{prop:normresolvent}
and Lemma~\ref{lem:minmax}
(and Kato~\cite{Kato1966} Ch.~IV; Reed--Simon~\cite{ReedSimon4} Thm.~VIII.23):
\begin{enumerate}[nosep,label=(\roman*)]
  \item $|a_{j,c}-a_j| = O(c^{-1/3})$, same ordering, no crossing.
  \item $\|P_{j,c}-P_j\|_{L^2\to L^2} = O_j(c^{-1/3})$
    (eigenprojector, operator norm).
  \item $\|\PcK-\PK\|_{L^2\to L^2} = O_K(c^{-1/3})$.
  \item $G^{(c)}_{jk}:=\langle\tilde\psi_j,u_k\rangle
    = \delta_{jk}+O_K(c^{-1/3})$.
\end{enumerate}
Index identification $\tilde\psi_j\leftrightarrow u_j$
is canonical by Lemma~\ref{lem:minmax} (same min-max index $j$).
\end{corollary}

\section{Proof of the Main Lemma}
\label{sec:proof}
%% ============================================================

\begin{lemma}[$H^2\to C^0$]
\label{lem:H2}
For fixed $K,M$:
$\|\tilde\psi_j-u_j\|_{C^0([0,M])}=O_K(c^{-1/3})$.
\end{lemma}

\begin{proof}
Let $e_j:=\tilde\psi_j-u_j$.
From $\Lc = \cA+c^{-1/3}R_c$:
$\cA e_j = a_j e_j - (a_{j,c}-a_j)\tilde\psi_j - c^{-1/3}R_c\tilde\psi_j$,
so $\|\cA e_j\|_{L^2} = O_j(c^{-1/3})$.
Since $\cA=-\partial_\xi^2+\xi$:
$\|e_j''\|_{L^2([0,M])}\leq\|\cA e_j\|_{L^2}+(M+1)\|e_j\|_{L^2}
= O_K(c^{-1/3})$.
$H^2([0,M])\hookrightarrow C^0([0,M])$: result follows.
\end{proof}

\textbf{Step 1: Edge/bulk split.}
$|\cK(\xi,\eta)-\cK^{(K)}(\xi,\eta)|\leq h(\xi,\eta)e^{-\pi R}$
(Fermi tail, Paper~XVII Lem.~2.1, dominated by Lemma~\ref{lem:I2b}).

\textbf{Step 2: Projector convergence.}
Corollary~\ref{cor:spectral}(iii) gives $\PcK\to\PK$;
eigenvalue weights $\lam{c}{n^*+j}\to\mathbf{1}_{j<0}$
(Paper~XVII Lem.~2.1).

\textbf{Step 3: Pointwise products.}
Lemma~\ref{lem:H2}:
$\cK^{(K)}(\xi,\eta)
\to\sum_{|j|\leq K}u_j(\xi)u_j(\eta)
= \cKA^{(K)}(\xi,\eta)$.

\textbf{Step 4: $K\to\infty$.}
$|\cK-\cK^{(K)}|\leq Ce^{-\pi K/c^{1/3}}\to 0$;
$\cKA^{(K)}\to\cKA$ (Proposition~\ref{prop:airykernel}).
\hfill$\square$
