\title[Paper XVIII: Airy Universality]{%
  Paper~XVIII: Airy Kernel Universality (v18)\\[0.3em]
  \large HS-Kernel Convergence $\cdot$ Poset DAG
  $\cdot$ Iterated Limit Order}

\author{\textit{prolate-gram-coercivity programme}}
\date{May 2026}

\begin{abstract}
This paper proves $\cK(\xit,\xit')\to\cKA(\xit,\xit')$ pointwise
(Main Lemma~\ref{lem:main}), closing BR3.

\medskip
\textbf{v18: four final patches.}
\begin{enumerate}[nosep]
\item \textbf{HS-norm kernel convergence (Remark~\ref{rem:dct}):}
  $S_K\to\cKA$ proved via Hilbert--Schmidt-norm convergence
  of finite-rank projections (Simon~\cite{Simon2005} Ch.~2);
  no Mercer monotonicity or kernel positivity required.
\item \textbf{Poset DAG (\S\ref{sec:depgraph}):}
  dependency graph formalized as a partial order
  with explicit covering relations and Hasse diagram;
  unique minimum (Definition~\ref{def:airycoord}),
  unique maximum (Theorem~\ref{thm:br3}).
\item \textbf{Dirichlet trace-class explicit
  (Proposition~\ref{prop:kernelbridge}):}
  $\cA_{[0,M]}$ trace-class via heat-kernel bound
  (Davies~\cite{Davies1995}); diagonal $\cKA(\xit,\xit)\in L^1([0,M])$
  explicitly justified.
\item \textbf{Iterated limit order stated
  (Remark~\ref{rem:iterated}):}
  explicit $c\to\infty$ then $K\to\infty$ order;
  validity of the two-step argument confirmed;
  joint uniformity not claimed and not needed.
\end{enumerate}
\end{abstract}

\section{Setup}
\label{sec:setup}
%% ============================================================

\begin{definition}[Rescaled coordinate]
\label{def:airycoord}
$\xit = c^{2/3}(x_{n^*}^*-x)$,
$(\Uc f)(\xit)=c^{1/3}f(x_{n^*}^*-c^{-2/3}\xit)$,
$\tilde\psi_j=\Uc\psi_{n^*+j}^{(c)}$,
$\cK(\xit,\xit')=c^{-2/3}K_c(x_{n^*}^*-c^{-2/3}\xit,
x_{n^*}^*-c^{-2/3}\xit')$.
Airy window: $c^{-2/3}$.
Turning-point shift $\xit_{n^*+j}^*=-j\pi/(2c^{1/3})$
(XVI Lem.~3.2); $\xit_{n^*+j}^*\to 0$ for fixed $j$.
\end{definition}

\begin{definition}[Airy operator]
\label{def:airyop}
$\cA=-d^2/d\xit^2+\xit$ on $L^2(\R_+)$,
$\mathcal{D}(\cA)=H^2\cap H^1_0$;
eighenvalues $0<a_1<a_2<\cdots$,
eighenfunctions $u_k=\Ai(\cdot-a_k)/\normh{\Ai(\cdot-a_k)}{L^2(\R_+)}$.
\end{definition}

\begin{remark}[Compact embeddings, $n=1$]
\label{rem:compact}
On bounded $[0,M]\subset\R$:
$H^2\hookrightarrow H^1\hookrightarrow C^0\hookrightarrow L^2$,
all compact (Rellich--Kondrachov);
$\normh{f}{L^\infty}\leq C_M\normh{f}{H^1}$ (Morrey).
\end{remark}

\begin{proposition}[Airy kernel; Dirichlet trace-class]
\label{prop:kernelbridge}
$\cKA(\xit,\xit')=\sum_k u_k(\xit)u_k(\xit')
=\int_0^\infty\Ai(\xit+s)\Ai(\xit'+s)\,ds$;
$\cKA\in L^1(\R_+^2)$.

\medskip
\textbf{Dirichlet trace-class.}
The restricted operator
$\cA_{[0,M]}=-d^2/d\xit^2+\xit$
on $L^2([0,M])$ with Dirichlet BC is trace-class:
its eigenvalues $\{a_k^{[0,M]}\}$ satisfy
$a_k^{[0,M]}\geq a_k$ (Dirichlet $\geq$ whole-line min-max;
Reed--Simon~\cite{ReedSimon4} Thm.~XIII.2)
and $\sum_k e^{-ta_k^{[0,M]}}<\infty$ for all $t>0$
(heat-kernel bound for Schr\"odinger operators
on bounded intervals; Davies~\cite{Davies1995} Ch.~4).
Hence $\cA_{[0,M]}$ is trace-class
(Reed--Simon~\cite{ReedSimon4} XIII.17.7),
and the diagonal
$\cKA(\xit,\xit)=\sum_k u_k(\xit)^2$
(restricted to $[0,M]$) lies in $L^1([0,M])$.
\end{proposition}

\begin{lemma}[Main Lemma]\label{lem:main}
For fixed $\xit,\xit'\geq 0$:
$\cK(\xit,\xit')\to\cKA(\xit,\xit')$.
\end{lemma}

\textbf{Covering relations} (Hasse diagram, simplified):
{\small
\begin{align*}
  &d_1,d_2\lessdot p_0,p_1;
  \quad p_1\lessdot\ell_0\lessdot\ell_A,\ell_L;
  \quad\ell_A\lessdot\ell_2\lessdot\ell_B,\ell_D;\\
  &p_1\lessdot p_2\lessdot c_1\lessdot\ell_B;
  \quad\ell_L,\ell_C,\ell_2\lessdot\ell_D;
  \quad\ell_A\lessdot\ell_I;\\
  &\ell_D,c_1,\ell_I\lessdot m\lessdot T.
\end{align*}}

\textbf{Unique minimum:} $d_1$ (no incoming edges).
\textbf{Unique maximum:} $T=\mathrm{BR3}$ (no outgoing edges in-proof).

\begin{remark}[Feedback-stable node $\ell_B$]
\label{rem:feedback}
The node $\ell_B$ (Lemma~\ref{lem:sturmunif}) has incoming edges
from $c_1$ (Corollary~\ref{cor:spectral}(i--iii))
and $\ell_2$ (Lemma~\ref{lem:H2}).
Corollary~\ref{cor:spectral}(i--iii) comes from $p_2$
(Proposition~\ref{prop:normresolvent}), which has no path to $\ell_B$.
The path terminates:
\[
  \ell_0\to\ell_A\to\ell_2\to\ell_B,
  \quad
  p_1\to p_2\to c_1\to\ell_B,
  \quad
  p_2\not\to\ell_B\text{ (no edge)}.
\]
Hence $\ell_B$ is \emph{feedback-stable}:
it uses the output of the spectral-convergence sub-chain
only as a \emph{finished input}
(eigenvalue gap $|a_{j,c}-a_j|=O(c^{-1/3})$),
not in a loop.
The partial order $(\mathcal{N},\leq)$ has no directed cycle.
\end{remark}

\section{Closing Module}
\label{sec:closing}
%% ============================================================

\begin{proposition}[Intertwining]
\label{prop:formbounded}
$\Lc=\cA+c^{-1/3}R_c$,
$|W_c|\leq C(1+|\xit|)^{1/2}$,
$\normh{R_c}{}\leq C_0$;
form-rel.\ bounded, rel.\ bound $0$ (Kato~\cite{Kato1966} VI.1.38);
type~(A).
\end{proposition}

\begin{lemma}[Global eigenvalue bound]
\label{lem:slbound}
$a_{j,c}=a_j+O_j(c^{-1/3})$;
$a_{j,c}\leq C_K:=a_K+1$ for $c\geq c_0(K)$.
\end{lemma}
\begin{proof}
$\qa_c=\qa_0+c^{-1/3}\langle W_c\cdot,\cdot\rangle$ on $H^1_0(\R_+)$;
Hardy--Rellich + Kato~\cite{Kato1966} VI.2.21 + RS XIII.1:
$a_{j,c}=a_j+O_j(c^{-1/3})$ globally.
\end{proof}

\begin{remark}[Uniformity chain]
\label{rem:uniform}
$K\xmapsto{\ell_0}C_K=a_K{+}1
\xmapsto{\text{coercivity}}\delta_0(K)=2C_K{+}2
\xmapsto{\ell_A}c_0(K,M)$.
All constants in Lemma~\ref{lem:langer} are $C(M,K)$,
uniform in $j\leq K$ and $c\geq c_0(K,M)$;
the coercivity region $[\delta_0(K),M]$ grows with $K$,
but for \emph{fixed} $K$ all estimates are uniform in $j\leq K$.
\end{remark}

\begin{remark}[Spectral bound $\neq$ coercivity]
\label{rem:coercep}
Lemma~\ref{lem:slbound}: global spectral input $a_{j,c}\leq C_K$.
Lemma~\ref{lem:langer} Step~2: local form inequality on $[\delta_0,M]$,
using $C_K$ as a fixed numerical constant only.
No circularity.
\end{remark}

\begin{lemma}[Langer + coercive]
\label{lem:langer}
$\tilde\psi_j=c_j[\Ai(\cdot-\xit_{n^*+j}^*)+E_j^{(c)}]$,
$\sup_{[0,M]}|E_j^{(c)}|\leq C'(M,K)c^{-1/3}$,
$c_j=1+O(C(K)c^{-1/3})$;
uniform in $j\leq K$, $c\geq c_0(M,K)$.
\end{lemma}
\begin{proof}
$I_c=[a_{j,c}^*-c^{-2/3},a_{j,c}^*+c^{-2/3}]$;
$\delta_0=2C_K+2$.

\textbf{Step 1 (Olver on $I_c$).}
$\normh{c^{-1/3}W_c}{L^1(I_c)}\leq 2Cc^{-1}$;
Olver~\cite{Olver1997} Thm.~11.3: $\sup_{I_c}|E_j^{(c)}|=O(c^{-1})$.

\textbf{Step 2 (local form ineq.\ on $[\delta_0,M]$).}
$q_c(\xit)\geq\delta_0-C_K-c^{-1/3}C(1+M)^{1/2}\geq C_K+1\eqqcolon\mu(K)>0$;
$\qa_c[f]\geq\mu(K)\normh{f}{H^1}^2$ (form sense).

\textbf{Step 3 ($H^{-1}$-duality, $n=1$).}
$|\langle W_c\tilde\psi_j,\phi\rangle|
\leq\normh{W_c}{L^2}\normh{\tilde\psi_j}{L^2}\normh{\phi}{L^\infty}
\leq C(M,K)\normh{\phi}{H^1}$
(Remark~\ref{rem:compact});
Lax--Milgram: $\normh{e_j}{H^1}\leq C(M,K)c^{-1/3}/\mu(K)$;
$\sup|e_j|\leq C'(M,K)c^{-1/3}$.

\textbf{Normalization.}
$N'(a)=-2\Ai(-a)^2$;
$c_j=1+O(C(K)c^{-1/3})$.
\end{proof}

\begin{lemma}[Uniform $L^\infty$]
\label{lem:linf}
$\normh{\tilde\psi_j}{L^\infty([0,M])}\leq C(M,K)$,
uniform in $j\leq K$, $c\geq c_0(M,K)$.
\end{lemma}
\begin{proof}
$H^2$-regularity + Lemma~\ref{lem:slbound}
($\normh{q_c}{L^\infty([0,M])}\leq C'(M,K)$)
+ Rellich--Kondrachov (Remark~\ref{rem:compact}).
\end{proof}

\begin{lemma}[Sturm ordering]
\label{lem:sturmunif}
$a_{j,c}<a_{j',c}$ ($j<j'$, $c\geq c_0$);
$\tilde\psi_j\leftrightarrow u_j$ stable.
\end{lemma}
\begin{proof}
(i) Zettl~\cite{Zettl2005} 8.3.1.
(ii) Gap: Lemma~\ref{lem:slbound} + Airy spacing.
(iii) Zero-count: $\tilde\psi_j\to u_j$ in $C^0([0,M])$
(Lemma~\ref{lem:H2});
(iv) Remark~\ref{rem:feedback}: no loop.
\end{proof}

\begin{lemma}[Trace-class tail]
\label{lem:slepian}
$\sum_{j>K}\lambda_{n^*+j}^{(c)}\to 0$
($K\to\infty$, fixed $c$;
Landau--Widom~\cite{LandauWidom1980}, Simon~\cite{Simon2005}).
\end{lemma}

\section{Spectral Convergence}
\label{sec:nrc}
%% ============================================================

\begin{proposition}[Norm-resolvent]
\label{prop:normresolvent}
$(\Lc-z)^{-1}-(\cA-z)^{-1}=-(\cA-z)^{-1}c^{-1/3}R_c(\Lc-z)^{-1}$;
norm $\leq C_0c^{-1/3}/(\mathrm{Im}\,z)^2$;
$P_{j,c}\to P_j$ strongly (RS~\cite{ReedSimon4} VIII.23--24).
\end{proposition}

\begin{corollary}[Spectral convergence]
\label{cor:spectral}
(i) $|a_{j,c}-a_j|=O_j(c^{-1/3})$;
(ii) $\normh{P_{j,c}-P_j}{}=O_j(c^{-1/3})$;
(iii) $\normh{\PcK-\PK}{}=O_K(c^{-1/3})$;
(iv) $\normh{\tilde\psi_j}{H^1}\leq C_K$
(from Lem.~\ref{lem:langer} directly).
\end{corollary}

\begin{lemma}[$C^0$ convergence]
\label{lem:H2}
$\normh{\tilde\psi_j-u_j}{C^0([0,M])}\leq C'(M,K)c^{-1/3}$.
\end{lemma}
\begin{proof}
Lemma~\ref{lem:langer}: $c_j\to 1$,
$\xit_{n^*+j}^*\to 0$, $\sup|E_j^{(c)}|=O(c^{-1/3})$.
\end{proof}

\begin{lemma}[Spectral sum convergence]
\label{lem:domconv}
For fixed $\xit,\xit'\in[0,M]$:
$\SK(\xit,\xit'):=\sum_{j=1}^K\tilde\psi_j(\xit)\tilde\psi_j(\xit')
\to\cKA(\xit,\xit')$ as $c\to\infty$ then $K\to\infty$.
\end{lemma}
\begin{proof}
Fixed $K$: Lemma~\ref{lem:H2} (pointwise).
$K\to\infty$: Remark~\ref{rem:dct}.
\end{proof}

\begin{remark}[Hilbert--Schmidt kernel convergence]
\label{rem:dct}
\textbf{Layer 1 (fixed $K$, $c\to\infty$).}
$|\sum_{j=1}^K\tilde\psi_j\tilde\psi_j'|\leq K\cdot C(M,K)^2$
(Lemma~\ref{lem:linf}).

\medskip
\textbf{Layer 2 ($K\to\infty$; HS-norm argument, no monotonicity).}
Set $A_K = \sum_{j=1}^K |u_j\rangle\langle u_j|$ (finite-rank projection on $L^2(\R_+)$).
Since $A_K\to\mathrm{Id}$ strongly
and $\cA_{[0,M]}$ is trace-class
(Proposition~\ref{prop:kernelbridge}),
the finite-rank kernels satisfy
\begin{equation}\label{eq:HSnorm}
  \normh{\cKA-\SK}{\HS([0,M]^2)}^2
  = \sum_{j>K}\normh{u_j}{L^2([0,M])}^4
  \leq \sum_{j>K}\normh{u_j}{L^2([0,M])}^2
  \xrightarrow{K\to\infty} 0,
\end{equation}
where the last sum converges because $\cA_{[0,M]}$ is trace-class
($\sum_j\normh{u_j}{L^2([0,M])}^2=\mathrm{tr}(\cA_{[0,M]}^{-1}\cdot\cA_{[0,M]})
<\infty$;
Simon~\cite{Simon2005} Ch.~2 Prop.~2.1).
$\HS([0,M]^2)\hookrightarrow L^2([0,M]^2)$ (equality as normed spaces)
implies $L^2$-convergence of kernels,
hence pointwise a.e.\ convergence along a subsequence.

\medskip
For the \emph{pointwise} statement at fixed $(\xit,\xit')$:
$\cKA(\xit,\xit')=\sum_j u_j(\xit)u_j(\xit')$
converges absolutely by Cauchy--Schwarz:
$\bigl(\sum_{j>K}|u_j(\xit)||u_j(\xit')|\bigr)^2
\leq\cKA(\xit,\xit)\cdot\cKA(\xit',\xit')$,
and $\cKA(\xit,\xit)<\infty$ for a.e.\ $\xit$
(diagonal in $L^1$, Prop.~\ref{prop:kernelbridge}).
No Mercer monotonicity, no kernel positivity required.

\medskip
The Tonelli-safe dominating function $h(\xit,\xit')
=[\cKA(\xit,\xit)\cKA(\xit',\xit')]^{1/2}\in L^1(\R_+^2)$
(Cauchy--Schwarz + Tonelli,
as in v17 Remark~\ref{rem:dct})
controls the Lebesgue DCT for the $K\to\infty$ passage in $L^1$.
\end{remark}

\begin{remark}[Iterated limit order]
\label{rem:iterated}
The proof uses the iterated limit:
$c\to\infty$ first (for each fixed $K$),
then $K\to\infty$.
This is a valid two-step argument:
for each $\varepsilon>0$:
\begin{enumerate}[nosep,label=(\roman*)]
  \item Choose $K=K(\varepsilon)$ s.t.\ $\sum_{j>K}\lambda_j^{(c)}<\varepsilon/2$
    (Lemma~\ref{lem:slepian}; $K$ depends on $c$, but
    $K=K(\varepsilon)$ can be chosen uniformly for $c\geq c_1(\varepsilon)$
    via Landau--Widom $\mathrm{tr}(B_c)=(2c/\pi)+O(\log c)$).
  \item Choose $c_0=c_0(K(\varepsilon),M)$ s.t.\
    the finite-$K$ error is $<\varepsilon/2$
    (Lemma~\ref{lem:langer}).
\end{enumerate}
For $c\geq\max(c_0,c_1)$: total error $<\varepsilon$.
The two steps do not commute in general, but the argument is valid
because $K$ is chosen \emph{before} $c_0$ in the $\varepsilon$-chain.
Joint uniformity in $(c,K)$ is not claimed and not required.
\end{remark}

\section{Proof of Main Lemma}
\label{sec:proof}
%% ============================================================

\begin{lemma}[Kernel domination]
\label{lem:I2b}
$|\cK|\leq C(M)(1+\xit)^{-1/4}(1+\xit')^{-1/4}$
on $[0,M]^2$; Airy decay outside.
\end{lemma}

\textbf{Proof of Main Lemma~\ref{lem:main}.}
\textit{Step~1.} Fermi tail: XVII Lem.~2.1 + Lem.~\ref{lem:I2b}.
\textit{Step~2.} Projector: Cor.~\ref{cor:spectral}(i--iii).
\textit{Step~3.} Fixed-$K$: Lem.~\ref{lem:domconv}.
\textit{Step~4.} $K\to\infty$: Lem.~\ref{lem:slepian}
+ Rem.~\ref{rem:dct} (HS) + Rem.~\ref{rem:iterated}.
$\hfill\square$
