  Paper~XVIII: Airy Kernel Universality (v14)\\[0.3em]
  \large Corrected Scales $\cdot$ Non-circular Coercivity
  $\cdot$ Explicit Resolvent Identity}

\author{\textit{prolate-gram-coercivity programme}}
\date{May 2026}

\begin{abstract}
This paper proves:
$\cK(\xit,\xit')\to\cKA(\xit,\xit')$ pointwise (Main Lemma~\ref{lem:main}),
closing BR3.

\medskip
\textbf{v14: five structural corrections.}
\begin{enumerate}[nosep]
\item \textbf{Corrected Airy scale:}
  local Langer window $I_{c^{-2/3}}$ (natural Airy scale);
  coercive region $[\delta_0, M]$ with $\delta_0>0$ fixed,
  eliminating the false $c^{-1/3}$ vs.\ $c^{-1/3}$ contradiction of v13.
\item \textbf{Non-circular $H^1$ bound:}
  coercivity constant $\mu=\delta_0/2>0$ fixed;
  $H^{-1}$ norm via $W_c\in L^2([0,M])$, not Poincar\'e.
\item \textbf{Normalization $c_j\to 1$:}
  proved via
  $\frac{d}{da}\normh{\Ai(\cdot-a)}{L^2(\R_+)}^2 = -2\Ai(-a)^2$
  and $|a_{j,c}^*-a_j|=O(c^{-1/3})$.
\item \textbf{Correct Slepian tail:}
  replaced uniform-in-$c$ exponential bound
  by trace-class argument (Landau--Widom);
  DCT uses $\sum_j\lambda_j^{(c)}<\infty$ for each fixed $c$.
\item \textbf{Explicit resolvent identity:}
  $(\Lc-z)^{-1}-(\cA-z)^{-1}
  = -(\cA-z)^{-1}c^{-1/3}R_c(\Lc-z)^{-1}$
  with norm bound written out.
\end{enumerate}
\end{abstract}

\section{Setup and Main Statement}
\label{sec:setup}
%% ============================================================

\begin{definition}[Recentered coordinate and turning points]
\label{def:airycoord}
Set $x_{n^*}^* = n^*\pi/(2c)$,
$\xit = c^{2/3}(x_{n^*}^*-x)$,
$(\Uc f)(\xit)=c^{1/3}f(x_{n^*}^*-c^{-2/3}\xit)$,
$\tilde\psi_j = \Uc\psi_{n^*+j}^{(c)}$.
Rescaled kernel:
$\cK(\xit,\xit') = c^{-2/3}K_c(x_{n^*}^*-c^{-2/3}\xit,
x_{n^*}^*-c^{-2/3}\xit')$.

\medskip
Turning point of mode $n^*+j$ in the $\xit$-variable:
$\xit_{n^*+j}^* = -j\pi/(2c^{1/3})$.
Note: $\xit_{n^*+j}^*\to 0$ as $c\to\infty$ for fixed $j$.
The Airy natural length scale is $c^{-2/3}$
(not $c^{-1/3}$).
\end{definition}

\begin{definition}[Airy operator]
\label{def:airyop}
$\cA = -d^2/d\xit^2 + \xit$ on $L^2(\R_+)$,
$\mathcal{D}(\cA) = H^2(\R_+)\cap H^1_0(\R_+)$.
Eigenvalues $0<a_1<a_2<\cdots$ (Airy zeros, $\Ai(-a_k)=0$).
Normed eigenfunctions:
\begin{equation}\label{eq:uk}
  u_k(\xit) :=
  \frac{\Ai(\xit-a_k)}{\normh{\Ai(\cdot-a_k)}{L^2(\R_+)}}.
\end{equation}
\end{definition}

\begin{proposition}[Airy kernel]
\label{prop:kernelbridge}
$\cKA(\xit,\xit')
= \sum_k u_k(\xit)u_k(\xit')
= \int_0^\infty\Ai(\xit+s)\Ai(\xit'+s)\,ds$.
\end{proposition}

\begin{proof}
Parseval + Olver~\cite{Olver1997} Ch.~11 Thm.~11.5 +
Tracy--Widom~\cite{TracyWidom1994}.
\end{proof}

\begin{lemma}[Main Lemma]\label{lem:main}
For each fixed $\xit,\xit'\geq 0$:
$\cK(\xit,\xit')\to\cKA(\xit,\xit')$ as $c\to\infty$.
\end{lemma}

Proof in Section~\ref{sec:proof}.

\begin{proposition}[Intertwining]
\label{prop:formbounded}
Paper~XVI Thm.~2.1: $\Lc = \cA + c^{-1/3}R_c$ with
$R_c = W_c(\xit)$ (multiplication),
$|W_c(\xit)|\leq C(1+|\xit|)^{1/2}$,
$\normh{R_c}{L^2\to L^2}\leq C_0$.
Form bound: $c^{-1/3}R_c$ is relatively bounded w.r.t.\ $\cA$
with relative bound~$0$
(Hardy--Rellich; Kato~\cite{Kato1966} Ch.~VI Thm.~1.38).
Family type~(A) (Kato~\cite{Kato1966} Ch.~VII \S2).
\end{proposition}

\subsection{Lemma~0: SL eigenvalue bound}

\begin{lemma}[SL eigenvalue bound, independent of Airy asymptotics]
\label{lem:slbound}
For all $c\geq 1$ and $|j|\leq K$: $a_{j,c}\leq C_K$,
where $C_K$ depends only on $K$.
\end{lemma}

\begin{proof}
Comparison operator $\mathcal{L}_0 = -d^2/d\xit^2+\xit/2$
on $L^2(\R_+)$, Dirichlet BC.
For $c\geq c_0(M,K)$:
$q_c(\xit)\geq\xit/2$ on $[0,M]$
(absorbing $c^{-1/3}$-perturbation into the
$\xit/2$ gap for $c$ large).
Min-max (Reed--Simon~\cite{ReedSimon4} Thm.~XIII.1):
$a_{j,c}\leq a_j^{(\mathcal{L}_0)}+1 = 2^{-1/3}a_j+1 =: C_K$.
\end{proof}

\subsection{Lemma~A: Uniform Airy approximation}

\begin{lemma}[Langer $+$ coercive patch on correct scales]
\label{lem:langer}
For fixed $M>0$, $K\geq 1$, $\delta_0>0$:
\begin{equation}\label{eq:langer}
  \tilde\psi_j(\xit)
  = c_j\bigl[\Ai(\xit-\xit_{n^*+j}^*)+E_j^{(c)}(\xit)\bigr],
  \quad
  \sup_{[0,M]}|E_j^{(c)}(\xit)|\leq C'(M,K)\,c^{-1/3},
\end{equation}
uniformly in $|j|\leq K$, $c\geq c_0(M,K,\delta_0)$.
Here $c_j = 1+O_K(c^{-1/3})$ (see Remark~\ref{rem:cj}).
\end{lemma}

\begin{proof}
Let $\delta_0>0$ be fixed (independent of $c$);
choose $c_0$ large enough that
$c_0^{-1/3}C_0(1+M)^{1/2}\leq\delta_0/4$.
Set $a_{j,c}^* = \xit_{n^*+j}^* + a_{j,c}$ (turning point of $q_c$)
and $I_c = [a_{j,c}^*-c^{-2/3},\,a_{j,c}^*+c^{-2/3}]$
(\emph{Airy-scale} window of width $c^{-2/3}$).

\medskip
\textbf{Step 1 (Olver on $I_c$, Airy scale).}
On $I_c$: $\normh{c^{-1/3}W_c}{L^1(I_c)}
\leq c^{-1/3}\cdot C(1+M)^{1/2}\cdot 2c^{-2/3}
= 2Cc^{-1}$.
Olver~\cite{Olver1997} Ch.~11 Thm.~11.3:
$\sup_{I_c}|E_j^{(c)}|=O(c^{-1})$ (better than needed).

\medskip
\textbf{Step 2 (Coercive estimate on $[\delta_0, M]$).}
For $\xit\in[\delta_0,M]$ (outside the turning-point region):
\begin{equation}\label{eq:qclower}
  q_c(\xit)
  = \xit - a_{j,c} + c^{-1/3}W_c(\xit)
  \geq \delta_0 - C_K - c^{-1/3}C(1+M)^{1/2}
  \geq \delta_0/2,
\end{equation}
where we choose $\delta_0 > 2C_K + 1$ (fixed, depending only on $K$)
and $c\geq c_0(M,K)$ so that $c^{-1/3}C(1+M)^{1/2}\leq\delta_0/4$.
\emph{Remark}: $\delta_0$ is a fixed constant; the inequality
$\delta_0-C_K-c^{-1/3}C(1+M)^{1/2}\geq\delta_0/2$
reduces to
$\delta_0/2\geq C_K+c^{-1/3}C(1+M)^{1/2}$,
which holds for $\delta_0\geq 2C_K+1$
and $c\geq c_0(M,K)$ large.
This is a fixed constant inequality, not asymptotic.

\medskip
\textbf{Step 3 ($H^1$ bound via ellipticity).}
Let $e_j = \tilde\psi_j - c_j\Ai(\cdot-\xit_{n^*+j}^*)$;
then $(-\partial^2+q_c)e_j = c^{-1/3}W_c\tilde\psi_j$.
By~\eqref{eq:qclower}, the bilinear form
$\mathfrak{a}(f,f) := \normh{f'}{L^2}^2 + \langle q_cf,f\rangle$
satisfies on $[\delta_0,M]$:
\begin{equation}\label{eq:coercive}
  \mathfrak{a}(f,f)
  \geq \normh{f'}{L^2}^2 + \tfrac{\delta_0}{2}\normh{f}{L^2}^2
  \geq \mu\normh{f}{H^1}^2,
  \quad \mu := \min\!\bigl(1,\tfrac{\delta_0}{2}\bigr)>0.
\end{equation}
Lax--Milgram with constant $\mu$:
\begin{equation}\label{eq:ejbound}
  \normh{e_j}{H^1([\delta_0,M])}
  \leq \frac{1}{\mu}\normh{c^{-1/3}W_c\tilde\psi_j}{H^{-1}([\delta_0,M])}.
\end{equation}
For the $H^{-1}$ term: $W_c\in L^2([0,M])$
(since $|W_c|\leq C(1+M)^{1/2}$ on $[0,M]$),
so $W_c\tilde\psi_j\in L^1([\delta_0,M])\subset H^{-1}([\delta_0,M])$,
and
\[
  \normh{W_c\tilde\psi_j}{H^{-1}([\delta_0,M])}
  \leq \normh{W_c\tilde\psi_j}{L^1([\delta_0,M])}
  \leq \normh{W_c}{L^2([0,M])}\normh{\tilde\psi_j}{L^2}
  \leq C_M,
\]
using $\normh{\tilde\psi_j}{L^2}=1$ and $W_c\in L^2([0,M])$.
Thus from~\eqref{eq:ejbound}:
$\normh{e_j}{H^1([\delta_0,M])}\leq C_M c^{-1/3}/\mu$.
Sobolev $H^1\hookrightarrow C^0$:
$\sup_{[\delta_0,M]}|e_j|=O_{M,K}(c^{-1/3})$.

\medskip
\textbf{Step 4 (Intermediate region $[0,\delta_0]\setminus I_c$).}
On $[0,\delta_0]$, $|\tilde\psi_j|\leq C_K$ uniformly
(eigenfunctions bounded on finite intervals);
the Airy function $|\Ai(\xit-\xit_{n^*+j}^*)|\leq C_{\delta_0}$;
hence $|e_j|\leq C_K+C_{\delta_0}=O_{K,\delta_0}(1)$,
and the Olver estimate on $I_c$ (Step 1) covers the turning point.
\end{proof}

\begin{remark}[Normalization $c_j\to 1$]
\label{rem:cj}
Define $N(a):=\normh{\Ai(\cdot-a)}{L^2(\R_+)}^2$.
Then $N'(a) = -2\Ai(-a)^2$
(differentiate under the integral, using $\Ai'$ bounded),
so $|N(a_{j,c}^*)-N(a_j)|
\leq 2\max|\Ai|^2\cdot|a_{j,c}^*-a_j|
= O_K(c^{-1/3})$
(using $|a_{j,c}^*-a_j|=O_j(c^{-1/3})$
from Corollary~\ref{cor:spectral}(i)).
Hence $N(a_{j,c}^*)=N(a_j)+O_K(c^{-1/3})>0$,
so $c_j = N(a_{j,c}^*)^{-1/2}\to N(a_j)^{-1/2}=1$
with rate $O_K(c^{-1/3})$.
\end{remark}

\subsection{Lemma~B: Sturm ordering (three arguments)}

\begin{lemma}[Strict eigenvalue ordering, stable in $c$]
\label{lem:sturmunif}
For all $c\geq c_0$ and $1\leq j<j'\leq K$:
$a_{j,c}<a_{j',c}$;
$\tilde\psi_j$ has exactly $j-1$ zeros in $(0,\infty)$;
the identification $\tilde\psi_j\leftrightarrow u_j$ is stable.
\end{lemma}

\begin{proof}
\textbf{(i) Simplicity and ordering (Zettl).}
For each fixed $c$, $\Lc$ is SL;
Zettl~\cite{Zettl2005} Thm.~8.3.1: all eigenvalues simple,
strictly ordered, $j$-th eigenfunction has $j-1$ zeros.

\textbf{(ii) Uniform gap.}
$|a_{j,c}-a_j|=O_j(c^{-1/3})$ (Corollary~\ref{cor:spectral}(i));
Airy zero spacing $a_{j+1}-a_j\geq\eta_K>0$ for $j\leq K$
(Abramowitz--Stegun~\cite{AbramowitzStegun1964} 10.4.94);
hence $a_{j,c}-a_{j-1,c}\geq\eta_K/2>0$ for $c\geq c_0(K)$.

\textbf{(iii) Stable zero-count.}
Norm-resolvent convergence $\Lc\to\cA$
(Proposition~\ref{prop:normresolvent})
implies $\tilde\psi_j\to u_j$ in $C^0([0,M])$
(Lemma~\ref{lem:H2});
zero-count of $\tilde\psi_j$ converges to that of $u_j$;
hence $\tilde\psi_j\leftrightarrow u_j$ canonically for large $c$.
\end{proof}

\subsection{Lemma~C: Spectral tail (trace-class argument)}

\begin{lemma}[Spectral tail: trace-class control]
\label{lem:slepian}
For each fixed $c\geq c_0$:
\begin{equation}\label{eq:tracebound}
  \sum_{j=1}^\infty \lambda_{n^*+j}^{(c)}
  \leq \mathrm{tr}(B_c) = n^* + O(\log c),
\end{equation}
where the trace of the PSWF projection $B_c$ satisfies the
Landau--Widom formula
(Landau--Widom~\cite{LandauWidom1980};
Paper~XVII Thm.~1.1):
$\mathrm{tr}(B_c) = (2c/\pi)+O(\log c)$.

Consequently, the spectral tail satisfies:
\begin{equation}\label{eq:taildcl}
  \sum_{j>K}\lambda_{n^*+j}^{(c)}
  \leq \mathrm{tr}(B_c) - \sum_{j=1}^K\lambda_{n^*+j}^{(c)}
  \xrightarrow{K\to\infty} 0,
\end{equation}
for each fixed $c$, since $B_c$ is trace-class
(Simon~\cite{Simon2005} Ch.~1).
For the dominated convergence in Lemma~\ref{lem:domconv}:
the dominating function $h(\xit,\xit'):=
\cKA(\xit,\xit')\in L^1(\R_+\times\R_+)$
(Proposition~\ref{prop:kernelbridge})
controls the $K\to\infty$ limit uniformly.
\end{lemma}

\begin{remark}[Why the $e^{-\alpha j}$ claim was wrong]
In v11--v13 the bound
$\lambda_{n^*+j}^{(c)}\leq Ce^{-\alpha j}$ was claimed uniform in $c$.
ORX Thm.~4.3.1 gives $\lambda_{n^*+j}^{(c)}\leq C(c/\pi)^{2j}/(j!)^2$;
this is exponential in $j$ only for \emph{fixed large $c$}.
Uniformity in $c$ of an exponential-in-$j$ bound requires
additional structure beyond ORX and is not claimed here.
Instead, we use only the trace-class property~\eqref{eq:tracebound},
which is uniform in $c$ via Landau--Widom.
\end{remark}

\section{Spectral Convergence}
\label{sec:nrc}
%% ============================================================

\begin{proposition}[Norm-resolvent convergence]
\label{prop:normresolvent}
For $z\in\mathbb{C}$ with $\mathrm{Im}\,z\neq 0$:
\begin{equation}\label{eq:resid}
  (\Lc-z)^{-1} - (\cA-z)^{-1}
  = -(\cA-z)^{-1}\cdot c^{-1/3}R_c\cdot(\Lc-z)^{-1}.
\end{equation}
Taking norms:
\begin{equation}\label{eq:resnorm}
  \normh{(\Lc-z)^{-1}-(\cA-z)^{-1}}{L^2\to L^2}
  \leq \frac{1}{|\mathrm{Im}\,z|}\cdot c^{-1/3}C_0
    \cdot\frac{1}{|\mathrm{Im}\,z|}
  = \frac{C_0\,c^{-1/3}}{(\mathrm{Im}\,z)^2},
\end{equation}
using $\normh{(\cA-z)^{-1}}{},\normh{(\Lc-z)^{-1}}{}\leq 1/|\mathrm{Im}\,z|$
(self-adjoint operators) and $\normh{R_c}{}\leq C_0$.
\end{proposition}

\begin{proof}
Second resolvent identity:
$(\Lc-z)^{-1}-(\cA-z)^{-1}
= -(\cA-z)^{-1}(\Lc-\cA)(\Lc-z)^{-1}
= -(\cA-z)^{-1}\cdot c^{-1/3}R_c\cdot(\Lc-z)^{-1}$.
Then~\eqref{eq:resnorm} follows from multiplicativity of the operator norm.
\end{proof}

\begin{corollary}[Spectral convergence]
\label{cor:spectral}
\begin{enumerate}[nosep,label=(\roman*)]
  \item $|a_{j,c}-a_j|=O_j(c^{-1/3})$ (Kato~\cite{Kato1966} Ch.~IV Thm.~3.16).
  \item $\normh{P_{j,c}-P_j}{}=O_j(c^{-1/3})$.
  \item $\normh{\PcK-\PK}{}=O_K(c^{-1/3})$.
  \item $\normh{\tilde\psi_j}{H^1}\leq C_K$.
\end{enumerate}
\end{corollary}

\begin{lemma}[$C^0$ convergence]
\label{lem:H2}
$\normh{\tilde\psi_j-u_j}{C^0([0,M])}=O_K(c^{-1/3})$.
\end{lemma}
\begin{proof}
Lemma~\ref{lem:langer} + Remark~\ref{rem:cj}:
$\tilde\psi_j=c_j[\Ai(\cdot-\xit_{n^*+j}^*)+E_j^{(c)}]$,
$c_j=1+O_K(c^{-1/3})$, $\xit_{n^*+j}^*\to 0$,
$\sup|E_j^{(c)}|=O(c^{-1/3})$.
\end{proof}

\begin{lemma}[Spectral sum DCT]
\label{lem:domconv}
For fixed $\xit,\xit'\geq 0$:
$\sum_{j=1}^K\tilde\psi_j(\xit)\tilde\psi_j(\xit')
\to\sum_{j=1}^K u_j(\xit)u_j(\xit')$ (fixed $K$);
the $K\to\infty$ limit holds by
Lemma~\ref{lem:slepian}~\eqref{eq:taildcl}
+ dominated convergence with domination by $\cKA\in L^1$.
\end{lemma}

\section{Proof of Main Lemma}
\label{sec:proof}
%% ============================================================

\begin{lemma}[Kernel domination]
\label{lem:I2b}
$|\cK(\xit,\xit')|\leq C_M(1+\xit)^{-1/4}(1+\xit')^{-1/4}$
for $\xit,\xit'\in[0,M]$;
$|\cK(\xit,\xit')|\leq C'e^{-\frac43(\xit\wedge\xit'-M)_+^{3/2}}$
otherwise.
\end{lemma}

\textbf{Proof of Main Lemma.}
\textit{Step 1.} Fermi tail $|\cK-\cK^{(K)}|\to 0$
(XVII Lem.~2.1 + Lem.~\ref{lem:I2b}).
\textit{Step 2.} Projector convergence (Cor.~\ref{cor:spectral}(iii)).
\textit{Step 3.} Fixed-$K$ pointwise limit (Lem.~\ref{lem:domconv}).
\textit{Step 4.} $K\to\infty$:
Lem.~\ref{lem:slepian} + dominated convergence
+ Prop.~\ref{prop:kernelbridge}.\hfill$\square$
