\begin{abstract}
This paper proves Lemma~\ref{lem:main}:
$\cK(\xit,\xit')\to\cKA(\xit,\xit')$ pointwise, closing BR3.

\medskip
\textbf{v11 (publication-ready patch).}
The Closing Lemma~\ref{lem:closing} is split into three
independent sub-lemmas, each addressing one reviewer objection:
\begin{enumerate}[nosep]
  \item \textbf{Lemma~\ref{lem:langer}} (Langer local + energy patch):
    Olver Thm.~11.3 is applied only on a local window
    $|\xit-\xit_{n^*+j}^*|\leq\delta$;
    a weighted $H^1$ energy estimate closes the global bound
    $\sup_{[0,M]}|E_j^{(c)}|=O(c^{-1/3})$ uniformly in $|j|\leq K$.
  \item \textbf{Lemma~\ref{lem:nocross}} (no-crossing via Kato--Rellich):
    for an analytic family of type~(A), simple eigenvalues
    at $\varepsilon=0$ remain simple for all $\varepsilon\in[0,\varepsilon_0]$
    (Kato Ch.~VII Thm.~3.9);
    Sturm oscillation then identifies the canonical index.
  \item \textbf{Lemma~\ref{lem:domconv}} (dominated convergence for spectral sum):
    the $K\to\infty$ limit is controlled by an explicit
    evanescent tail bound uniform in $j$,
    giving a rigorous dominated convergence argument.
\end{enumerate}
\end{abstract}

\section{Setup}
\label{sec:setup}
%% ============================================================

\begin{definition}[Recentered Airy coordinate]
\label{def:airycoord}
Let $x_{n^*}^* := n^*\pi/(2c)$ (turning point of mode $j=0$;
Paper~XVI Def.~1.1).
\begin{equation}\label{eq:xit}
  \xit := c^{2/3}(x_{n^*}^* - x), \qquad x\in[-1,1].
\end{equation}
Identification map:
$(\Uc f)(\xit) := c^{1/3}f(x_{n^*}^*-c^{-2/3}\xit)$,
$\Uc:L^2([-1,1])\to L^2(\R_+)$,
$\|\Uc\| = 1+O(c^{-1/3})$.
Rescaled PSWF: $\tilde\psi_j := \Uc\psi_{n^*+j}^{(c)}\in L^2(\R_+)$.
Rescaled kernel:
\begin{equation}\label{eq:cKdef}
  \cK(\xit,\xit') :=
  c^{-2/3}K_c\bigl(x_{n^*}^*-c^{-2/3}\xit,\;
  x_{n^*}^*-c^{-2/3}\xit'\bigr).
\end{equation}
Turning points:
$\xit_{n^*+j}^* = -j\pi/(2c^{1/3})\to 0$
(no floor residual; Definition absorbs $\delta_c$).
\end{definition}

\begin{definition}[Airy operator: half-line, unique model]
\label{def:airyop}
\begin{equation}\label{eq:airyop}
  \cA = -\frac{d^2}{d\xit^2} + \xit
  \quad\text{on }L^2(\R_+),\quad
  \mathcal{D}(\cA) = H^2(\R_+)\cap H^1_0(\R_+),\quad
  u(0)=0.
\end{equation}
$\cA$ is positive self-adjoint, compact resolvent,
discrete spectrum $0<a_1<a_2<\cdots$ (Airy zeros: $\Ai(-a_k)=0$),
ONB $\{u_k\}_{k\geq 1}$ of $L^2(\R_+)$.
The full-line operator on $L^2(\R)$ has purely continuous spectrum
and is \textbf{not used}.
\end{definition}

\begin{proposition}[Airy kernel: spectral $=$ integral form]
\label{prop:kernelbridge}
\begin{equation}\label{eq:kernelbridge}
  \cKA(\xit,\xit')
  := \sum_{k=1}^\infty u_k(\xit)u_k(\xit')
  = \int_0^\infty \Ai(\xit+s)\Ai(\xit'+s)\,ds,
\end{equation}
uniformly on compacts and in $L^2(\R_+\times\R_+)$.
\end{proposition}

\begin{proof}
Left: Parseval + Sobolev $H^1\hookrightarrow C^0$ + Weyl.
Right: Airy completeness on $\R_+$
(Olver~\cite{Olver1997} Ch.~11 Thm.~11.5;
Tracy--Widom~\cite{TracyWidom1994}).
\end{proof}

\begin{lemma}[Main Lemma]\label{lem:main}
For fixed $\xit,\xit'\geq 0$:
$\displaystyle\lim_{c\to\infty}\cK(\xit,\xit') = \cKA(\xit,\xit')$.
\end{lemma}

\noindent Proof in Sections~\ref{sec:closing}--\ref{sec:proof}.

\subsection{Intertwining}

\begin{proposition}[Intertwining + form bound]
\label{prop:formbounded}
Paper~XVI Thm.~2.1 in coordinate~\eqref{eq:xit}:
$\Lc = \cA + c^{-1/3}R_c$, $\|R_c\|_{L^2\to L^2}\leq C_0$.
$R_c = W_c(\xit)$ (multiplication, $|W_c(\xit)|\leq C(1+|\xit|)^{1/2}$).
Form-bounded relative to $\cA$ with relative bound~$0$
(Hardy--Rellich; Kato~\cite{Kato1966} Ch.~VI Thm.~1.38).
$\Lc(\varepsilon)=\cA+\varepsilon R_c$ analytic type~(A)
(Kato~\cite{Kato1966} Ch.~VII \S2).
\end{proposition}

\subsection{Sub-Lemma A: Langer local + energy patch}

\begin{lemma}[Uniform Airy approximation]
\label{lem:langer}
For fixed $M>0$, $K\geq 1$, and $c\geq c_0(M,K)$:
\begin{equation}\label{eq:langer}
  \tilde\psi_j(\xit)
  = c_j\Bigl[\Ai\bigl(\xit - \xit_{n^*+j}^*\bigr)
  + E_j^{(c)}(\xit)\Bigr],
  \qquad
  \sup_{\xit\in[0,M]}\bigl|E_j^{(c)}(\xit)\bigr|
  \leq C'(M,K)\,c^{-1/3},
\end{equation}
uniformly in $|j|\leq K$.
\end{lemma}

\begin{proof}
Fix $\delta>0$ (to be chosen) and $|j|\leq K$.

\medskip\noindent
\emph{Step 1: local Olver window.}
Let $I_\delta := [\xit_{n^*+j}^*-\delta, \xit_{n^*+j}^*+\delta]
\subset[-\delta,\delta+O(c^{-1/3})]$.
The eigenvalue equation reads
\begin{equation}\label{eq:ode}
  -\tilde\psi_j'' + \bigl(\xit + c^{-1/3}W_c(\xit) - a_{j,c}\bigr)\tilde\psi_j = 0.
\end{equation}
Setting $q_c(\xit) := \xit + c^{-1/3}W_c(\xit) - a_{j,c}$,
which has a simple zero at $\xit_{n^*+j}^*\in[-O(c^{-1/3}),0]$,
and applying the Langer transform on $I_\delta$
(Olver~\cite{Olver1997} Ch.~11 Thm.~11.3):
\[
  \|c^{-1/3}W_c\|_{L^1(I_\delta)}
  \leq c^{-1/3}\cdot C(1+\delta)^{1/2}\cdot 2\delta
  = O(c^{-1/3}\delta),
\]
so Olver's error term satisfies
$\sup_{I_\delta}|E_j^{(c)}| = O(c^{-1/3}\delta^{1/2})$
(the Olver $L^1$-bound on perturbation).
Choosing $\delta = c^{-1/3}$ gives
$\sup_{I_\delta}|E_j^{(c)}| = O(c^{-1/3})$ on the local window.

\medskip\noindent
\emph{Step 2: energy estimate on $[0,M]\setminus I_\delta$.}
On $[0,M]\setminus I_\delta$ the potential
$q_c(\xit)\geq \delta/2$ (since $|\xit-\xit_{n^*+j}^*|\geq\delta$
and $|c^{-1/3}W_c|\leq c^{-1/3}C(1+M)^{1/2}\ll\delta/4$ for $c$ large).
Let $e_j := \tilde\psi_j - c_j\Ai(\cdot-\xit_{n^*+j}^*)$; then
\[
  -e_j'' + q_c e_j = c^{-1/3}W_c\tilde\psi_j,
\]
so $\|e_j\|_{H^1([0,M]\setminus I_\delta)}
\leq C\|c^{-1/3}W_c\tilde\psi_j\|_{H^{-1}([0,M])}
\leq Cc^{-1/3}\|W_c\|_{L^2([0,M])}\|\tilde\psi_j\|_{H^1}
= O_M(c^{-1/3})$,
using $W_c\in L^2([0,M])$ (since $|W_c|\leq C(1+M)^{1/2}$ on $[0,M]$)
and $\|\tilde\psi_j\|_{H^1}=O_K(1)$ (Corollary~\ref{cor:spectral}).
By Sobolev $H^1([0,M])\hookrightarrow C^0([0,M])$:
$\sup_{[0,M]\setminus I_\delta}|e_j|=O_M(c^{-1/3})$.

\medskip\noindent
Combining Steps~1 and~2:
$\sup_{[0,M]}|E_j^{(c)}|\leq C'(M,K)c^{-1/3}$
with $C'$ independent of $j\leq K$.
(Uniformity in $j$: $\|\tilde\psi_j\|_{H^1}\leq C_K$ uniformly,
and $a_{j,c}\leq C_K$ uniformly by Corollary~\ref{cor:spectral}(i).)
\end{proof}

\subsection{Sub-Lemma B: No-crossing via Kato--Rellich}

\begin{lemma}[No-crossing and canonical index]
\label{lem:nocross}
For all $c\geq c_0$:
\begin{equation}\label{eq:nocross}
  a_{1,c} < a_{2,c} < \cdots < a_{K,c},
\end{equation}
and $\tilde\psi_j \leftrightarrow u_j$ canonically
($\tilde\psi_j$ has exactly $j$ zeros in $(0,\infty)$).
\end{lemma}

\begin{proof}
\emph{Simplicity via Kato--Rellich.}
Since $\Lc(\varepsilon)=\cA+\varepsilon R_c$ is an analytic family of type~(A)
(Proposition~\ref{prop:formbounded}),
the eigenvalues $a_{j,c}(\varepsilon)$ are analytic in $\varepsilon$ near~$0$
(Kato~\cite{Kato1966} Ch.~VII Thm.~2.6).
At $\varepsilon=0$, all eigenvalues $a_j$ are simple
(Sturm--Liouville; Zettl~\cite{Zettl2005} Thm.~8.3.1(ii)).
By Kato~\cite{Kato1966} Ch.~VII Thm.~3.9
(\emph{Kato--Rellich theorem for analytic families}):
simple eigenvalues persist under analytic deformation;
specifically, there exists $\varepsilon_0>0$ such that
$a_{j,c}(\varepsilon)$ is a simple isolated eigenvalue
for all $\varepsilon\in[0,\varepsilon_0]$.
Since $\varepsilon_c=c^{-1/3}\to 0$,
we have $\varepsilon_c\in[0,\varepsilon_0]$ for all large $c$,
proving~\eqref{eq:nocross}.

\emph{Canonical index via Sturm.}
By the Sturm--Liouville oscillation theorem
(Zettl~\cite{Zettl2005} Thm.~8.3.1):
the eigenfunction of $\Lc$ with eigenvalue $a_{j,c}$
has exactly $j$ zeros in $(0,\infty)$.
Since $u_j$ also has exactly $j$ zeros,
the identification $\tilde\psi_j\leftrightarrow u_j$ by zero-count is canonical.

\begin{quote}\itshape
  Note: the Sturm argument proves canonical identification,
  not no-crossing.  No-crossing is proved by Kato--Rellich.
  The two arguments are complementary, not redundant.
\end{quote}
\end{proof}

\subsection{Sub-Lemma C: Dominated convergence for spectral sum}

\begin{lemma}[Dominated convergence: spectral sum]
\label{lem:domconv}
For fixed $\xit,\xit'\geq 0$:
\begin{equation}\label{eq:domconv}
  \sum_{j=1}^K \tilde\psi_j(\xit)\tilde\psi_j(\xit')
  \;\xrightarrow{c\to\infty}\;
  \sum_{j=1}^K u_j(\xit)u_j(\xit'),
\end{equation}
and the $K\to\infty$ limit is controlled:
\begin{equation}\label{eq:Ktail}
  \Bigl|\sum_{j>K}\tilde\psi_j(\xit)\tilde\psi_j(\xit')\Bigr|
  \leq C(1+\xit)^{-1/4}(1+\xit')^{-1/4}e^{-\pi K/c^{1/3}},
\end{equation}
uniformly in $c$.
\end{lemma}

\begin{proof}
\emph{Fixed $K$ convergence~\eqref{eq:domconv}.}
For each $j\leq K$, Lemma~\ref{lem:langer} gives
$\tilde\psi_j\to c_j\Ai(\cdot-\xit_{n^*+j}^*)$ uniformly on $[0,M]$;
since $\xit_{n^*+j}^*\to 0$, $c_j\to 1$,
and $\Ai(\cdot-\xit_{n^*+j}^*)\to\Ai(\cdot)=u_j/\|u_j\|_{C^0}$
(Lemma~\ref{lem:H2}):
$\tilde\psi_j(\xit)\tilde\psi_j(\xit')\to u_j(\xit)u_j(\xit')$
pointwise.
Since $K$ is fixed, a finite sum of convergent sequences converges.

\emph{Tail bound~\eqref{eq:Ktail}.}
The PSWF eigenvalues satisfy $\lambda_{n^*+j}^{(c)}\leq e^{-\pi j/c^{1/3}}$
for $j>K$ (Paper~XVII Lem.~2.1: exponential Fermi decay in spectral weights).
Combined with the $L^\infty$ bound from Lemma~\ref{lem:langer}
($|\tilde\psi_j(\xit)|\leq C(1+\xit)^{-1/4}$
uniformly in $j$ and $c$, from~\eqref{eq:langer}),
the CD identity gives~\eqref{eq:Ktail}.

\emph{Dominated convergence.}
The tail bound~\eqref{eq:Ktail} provides
an integrable domination for the spectral sum in $K$
(the right-hand side is summable in $K$ uniformly in $c$,
and $\to 0$ as $K\to\infty$).
Hence Lebesgue's dominated convergence theorem applies
and $\sum_{j=1}^\infty\tilde\psi_j(\xit)\tilde\psi_j(\xit')
\to\sum_{j=1}^\infty u_j(\xit)u_j(\xit') = \cKA(\xit,\xit')$.
\end{proof}

\section{(I2b): Kernel Domination}
\label{sec:I2b}
%% ============================================================

\begin{lemma}[Kernel domination]
\label{lem:I2b}
For fixed $M>0$, $\exists\,C_M,C'>0$ uniform in $c\geq c_0$:
\begin{enumerate}[nosep,label=(\roman*)]
  \item $\xit,\xit'\in[0,M]$:
    $|\cK(\xit,\xit')|\leq C_M(1+\xit)^{-1/4}(1+\xit')^{-1/4}$.
  \item $\xit$ or $\xit'>M$:
    $|\cK(\xit,\xit')|\leq C'e^{-\frac43(\xit\wedge\xit'-M)_+^{3/2}}$.
\end{enumerate}
\end{lemma}

\begin{proof}
CD identity + Lemma~\ref{lem:langer} for (i)
($|\tilde\psi_j(\xit)|\leq C''(1+\xit)^{-1/4}$ uniformly in $j,c$);
Airy decay $\Ai(\xit)\sim C\xit^{-1/4}e^{-\frac23\xit^{3/2}}$ for (ii)
(Olver~\cite{Olver1997} Ch.~11 \S11.2).
\end{proof}

\section{Norm-Resolvent and Spectral Convergence}
\label{sec:nrc}
%% ============================================================

\begin{proposition}[Norm-resolvent convergence]
\label{prop:normresolvent}
$\|({\Lc}-z)^{-1}-({\cA}-z)^{-1}\|_{L^2\to L^2}
\leq C_0 c^{-1/3}/(\mathrm{Im}\,z)^2$.
\end{proposition}

\begin{proof}
Second resolvent identity + Prop.~\ref{prop:formbounded}.
\end{proof}

\begin{corollary}[Spectral convergence]
\label{cor:spectral}
\begin{enumerate}[nosep,label=(\roman*)]
  \item $|a_{j,c}-a_j|=O_j(c^{-1/3})$ (Kato~\cite{Kato1966} Ch.~IV).
  \item $\|P_{j,c}-P_j\|_{L^2\to L^2}=O_j(c^{-1/3})$.
  \item $\|\PcK-\PK\|_{L^2\to L^2}=O_K(c^{-1/3})$.
  \item $G^{(c)}_{jk}=\delta_{jk}+O_K(c^{-1/3})$.
  \item Index identification: Lemma~\ref{lem:nocross}.
\end{enumerate}
\end{corollary}

\section{Proof of the Main Lemma}
\label{sec:proof}
%% ============================================================

\begin{lemma}[$H^2\to C^0$]
\label{lem:H2}
$\|\tilde\psi_j-u_j\|_{C^0([0,M])}=O_K(c^{-1/3})$.
\end{lemma}

\begin{proof}
Lemma~\ref{lem:langer}: $\tilde\psi_j=c_j[\Ai(\cdot-\xit_{n^*+j}^*)+E_j^{(c)}]$,
$|E_j^{(c)}|\leq C'c^{-1/3}$;
$\Ai(\cdot-\xit_{n^*+j}^*)\to\Ai(\cdot)$ uniformly on $[0,M]$
(since $\xit_{n^*+j}^*\to 0$);
$c_j\to 1$ (normalization);
$u_j = \Ai(\cdot-a_j)/\|\cdot\|$.
\end{proof}

\textbf{Step 1: Fermi tail.}
$|\cK-\cK^{(K)}|\leq h(\xit,\xit')e^{-\pi R}$
(Paper~XVII Lem.~2.1 + Lemma~\ref{lem:I2b}).

\textbf{Step 2: Projector.}
Cor.~\ref{cor:spectral}(iii) + weights
$\lam{c}{n^*+j}\to\mathbf{1}_{j<0}$.

\textbf{Step 3: Pointwise products.}
Lemma~\ref{lem:domconv}~\eqref{eq:domconv} (fixed $K$):
$\cK^{(K)}(\xit,\xit')\to\cKA^{(K)}(\xit,\xit')$.

\textbf{Step 4: $K\to\infty$.}
Lemma~\ref{lem:domconv}~\eqref{eq:Ktail}
(tail bound uniform in $c$) + dominated convergence:
$\cKA^{(K)}\to\cKA$ (Proposition~\ref{prop:kernelbridge}).\hfill$\square$
