  Paper~XVIII: Airy Kernel Universality (v4)\\[0.3em]

\author{\textit{prolate-gram-coercivity programme}}

\begin{abstract}
This paper proves the Main Lemma~\ref{lem:main}:
for fixed $\xi,\eta\geq 0$,
$\cK(\xi,\eta)\to\cKA(\xi,\eta)$ as $c\to\infty$.
Combined with Paper~XVII, this closes BR3.

\medskip
\textbf{New in v4 (three consistency fixes).}
\begin{enumerate}[nosep]
  \item \emph{Three-scale dictionary (Definition~\ref{def:scales}):}
    explicit maps between physical $x$-scale,
    Airy $\xi$-scale, and spectral $s$-scale.
    All subsequent arguments are stated in \emph{one} scale at a time.
  \item \emph{Lemma~\ref{lem:tpshift} corrected:}
    the turning point $\xi_{n^*+j}^* = jc^{1/3}+O(1)$ in $\xi$-coordinates
    \emph{diverges} for $j\neq 0$;
    hence $|j|\geq 1$ modes are exponentially suppressed on $[0,M]$
    (evanescent regime), not ``same Olver constant as $j=0$''.
    The $j=0$ mode remains the sole Airy-window mode.
  \item \emph{Identification map $\Uc$ (Definition~\ref{def:Uc}):}
    explicit approximate isometry
    $\Uc:\mathrm{span}\{\psi_{n^*+j}^{(c)}:|j|\leq K\}\to L^2(\R_+)$
    with $\|\Uc f - f\|_{L^2} = O(c^{-2/3})\|f\|$.
    Projector convergence $\PcK\to\PK$ is now a statement
    in a single Hilbert space.
\end{enumerate}
\end{abstract}

\section{Setup: Three-Scale Dictionary}
\label{sec:setup}
%% ============================================================

\begin{definition}[Three coordinate scales]
\label{def:scales}
\mbox{}
\begin{enumerate}[nosep,label=(\Roman*)]
  \item \emph{Physical scale:} $x\in[-1,1]$,
    the native domain of $\psi_n^{(c)}$.
  \item \emph{Airy/rescaled scale:}
    $\xi = c^{2/3}(1-x)\in[0,\infty)$,
    so $x = 1-c^{-2/3}\xi$.
    Fixed window $[0,M]$ in $\xi$ corresponds to
    $x\in[1-Mc^{-2/3},1]$ in physical scale.
  \item \emph{Spectral scale:}
    $s_n := (n-2c/\pi)/c^{1/3}\in\R$.
    The Fermi index is $n^* = \lfloor 2c/\pi\rfloor$,
    so $s_{n^*} = O(c^{-1/3})\to 0$.
    For $n = n^*+j$: $s_{n^*+j} = j\,c^{-1/3} + O(c^{-2/3})$.
\end{enumerate}
The turning point of $\psi_{n^*+j}^{(c)}$ in each scale:
\begin{align}
  x_{n^*+j}^* &= 1 - s_{n^*+j}^2/(4c^{2/3}) + O(c^{-4/3})
    \approx 1\quad\text{(physical)},\label{eq:tp-x}\\
  \xi_{n^*+j}^* &= c^{2/3}(1-x_{n^*+j}^*)
    = s_{n^*+j}\,c^{2/3} = j\,c^{1/3} + O(1)
    \quad\text{(Airy scale)},\label{eq:tp-xi}\\
  s_{n^*+j} &= j\,c^{-1/3} + O(c^{-2/3})
    \quad\text{(spectral scale)}.\label{eq:tp-s}
\end{align}
\end{definition}

\begin{remark}[Scale separation — the key observation]
\label{rem:scales}
Equation~\eqref{eq:tp-xi} shows:
\begin{itemize}[nosep]
  \item $j=0$: $\xi_{n^*}^* = O(1)$ in Airy coordinates.
    The Airy window $[0,M]$ \emph{contains} the turning point.
  \item $j\neq 0$, fixed: $\xi_{n^*+j}^* = |j|c^{1/3}+O(1)\to\infty$.
    The turning point \emph{leaves the Airy window} as $c\to\infty$.
\end{itemize}
This is the fundamental asymmetry between the $j=0$ edge mode
and all $j\neq 0$ modes in the spectral window.
The $j=0$ mode is the unique ``Airy mode'';
all others are evanescent on $[0,M]$ for large $c$.
\end{remark}

\begin{definition}[Rescaled PSWFs and kernels]
\label{def:kernels}
Rescaled PSWF: $\tilde\psi_j(\xi) := c^{1/3}\psi_{n^*+j}^{(c)}(1-c^{-2/3}\xi)$
(scale~(II) applied to scale~(I)).\\
Rescaled kernel: $\cK(\xi,\eta) := c^{-2/3}K_c(1-c^{-2/3}\xi,1-c^{-2/3}\eta)$.\\
Airy kernel: $\cKA(\xi,\eta) := \sum_{k\geq 1}u_k(\xi)u_k(\eta)$
($u_k\in L^2(\R_+)$: eigenfunctions of $\cA=-\partial_\xi^2+\xi$,
$\cA u_k = -a_k u_k$, Dirichlet at~$0$).
\end{definition}

\begin{definition}[Identification map $\Uc$]
\label{def:Uc}
Let $V_K^{(c)} := \mathrm{span}\{\psi_{n^*+j}^{(c)} : |j|\leq K\}
\subset L^2([-1,1])$.
Define $\Uc : V_K^{(c)} \to L^2(\R_+)$ by
\begin{equation}\label{eq:Uc}
  (\Uc f)(\xi) := c^{1/3} f(1-c^{-2/3}\xi),
  \quad \xi\geq 0.
\end{equation}
\emph{Norm control.}
For $f\in V_K^{(c)}$ supported near $x=1$
(which holds for the $K$-window modes by WKB decay):
\begin{equation}\label{eq:Uciso}
  \|\Uc f\|_{L^2(\R_+)}^2
  = c^{2/3}\int_0^\infty |f(1-c^{-2/3}\xi)|^2\,d\xi
  = \int_{-\infty}^1 |f(x)|^2\,dx
  = \|f\|_{L^2([-1,1])}^2\bigl(1+O(e^{-\pi c})\bigr),
\end{equation}
where the error $O(e^{-\pi c})$ comes from the PSWF mass outside
$[1-\varepsilon,1]$ (exponentially small by Paper~XVII Lemma~2.1).
Thus $\Uc$ is an \emph{approximate isometry}: $\|\Uc\|\leq 1+O(e^{-\pi c})$.
\end{definition}

\begin{remark}[Operator convergence in a single space]
\label{rem:singlespace}
Via $\Uc$, both the PSWF projector
$P_{n^*,K}^{(c)} := \sum_{|j|\leq K}|\psi_{n^*+j}^{(c)}\rangle\langle\psi_{n^*+j}^{(c)}|$
(on $L^2([-1,1])$) and the Airy projector
$\PK := \sum_{k=1}^K|u_k\rangle\langle u_k|$
(on $L^2(\R_+)$) can be compared in a single Hilbert space $L^2(\R_+)$:
\begin{equation}\label{eq:conjprojector}
  \PcK := \Uc P_{n^*,K}^{(c)} \Uc^* \in \mathcal{B}(L^2(\R_+)).
\end{equation}
\emph{Strong convergence} $\PcK\to\PK$ now means:
for each fixed $v\in L^2(\R_+)$,
$\|\PcK v - \PK v\|_{L^2(\R_+)}\to 0$.
This is well-posed since both operators act on the same space.
\end{remark}

\begin{lemma}[Main Lemma]
\label{lem:main}
For fixed $\xi,\eta\geq 0$:
$\displaystyle\lim_{c\to\infty}\cK(\xi,\eta) = \cKA(\xi,\eta)$.
\end{lemma}

\noindent Proof in Sections~\ref{sec:I2b}--\ref{sec:proof}.

\section{(I2b): Two-Regime Kernel Domination}
\label{sec:I2b}
%% ============================================================

\begin{lemma}[Two-regime domination — corrected]
\label{lem:I2b}
For fixed $M>0$, there exist $C_M,C'>0$ uniform in $c\geq c_0(M)$:
\begin{enumerate}[nosep,label=(\roman*)]
  \item \emph{Airy regime, $\xi\in[0,M]$:}
    For $j=0$: Olver Airy-type bound (turning point inside $[0,M]$):
    \begin{equation}\label{eq:airybound}
      |\tilde\psi_0(\xi)| \leq C_M(1+\xi)^{-1/4},\quad \xi\in[0,M].
    \end{equation}
    For $|j|\geq 1$, $|j|\leq K$: turning point at
    $\xi_{n^*+j}^* = |j|c^{1/3}+O(1)\geq c^{1/3}-C$,
    so $[0,M]$ is \emph{evanescent} (Lemma~\ref{lem:tpshift}):
    \begin{equation}\label{eq:evanbound}
      |\tilde\psi_j(\xi)| \leq C_M'\,e^{-\frac{2}{3}(|j|c^{1/3}-M-C)^{3/2}},
      \quad \xi\in[0,M].
    \end{equation}
    Combining: $|\cK(\xi,\eta)|\leq C_M(1+\xi)^{-1/4}(1+\eta)^{-1/4}$.
  \item \emph{Far-field regime, $\xi>M$:}
    For all $j$: $|\tilde\psi_j(\xi)|\leq C'e^{-\frac{2}{3}(\xi-M)^{3/2}}$
    (either sub-turning-point decay for $j=0$, or evanescent for $j\neq 0$).
    Hence $|\cK(\xi,\eta)|\leq C'e^{-\frac{4}{3}(\xi\wedge\eta-M)_+^{3/2}}$.
\end{enumerate}
\end{lemma}

\begin{proof}
\emph{Regime (i), $j=0$:}
In Airy coordinates, $\xi_{n^*}^* = s_{n^*}c^{2/3} = O(c^{-1/3})\cdot c^{2/3} = O(c^{1/3})$.
Wait: more carefully, $s_{n^*} = (n^*-2c/\pi)/c^{1/3}$;
since $n^*=\lfloor 2c/\pi\rfloor$, $|s_{n^*}|\leq c^{-1/3}$.
So $\xi_{n^*}^* = s_{n^*}c^{2/3}\in(-1,1)$, i.e., $\xi_{n^*}^*=O(1)$
and lies \emph{inside} $[0,M]$ for any $M\geq 2$.
Olver~\cite{Olver1997} Ch.~11 Thm.~11.3 gives the Airy-type bound~\eqref{eq:airybound}
in the neighbourhood of the turning point; the bound is uniform in $c$
with constant depending on $M$ (the size of the Airy window).
No matching with far-field is needed on $[0,M]$.

\emph{Regime (i), $|j|\geq 1$:}
By~\eqref{eq:tp-xi}, $\xi_{n^*+j}^*\geq c^{1/3}-C$.
For $c\geq(M+C)^3$: $\xi_{n^*+j}^*\geq M+1$,
so all of $[0,M]$ is to the left of the turning point (evanescent side).
Olver subdominant solution: $|\tilde\psi_j(\xi)|\leq
C(\xi_{n^*+j}^*-\xi)^{-1/4}e^{-\frac{2}{3}(\xi_{n^*+j}^*-\xi)^{3/2}}
\leq C(c^{1/3}-M-C)^{-1/4}e^{-\frac{2}{3}(c^{1/3}-M-C)^{3/2}}$,
giving~\eqref{eq:evanbound} (exponentially small in $c^{1/2}$).
Note: this is \emph{not} the same bound as for $j=0$;
the $|j|\geq 1$ contributions to $\cK$ on $[0,M]$
are exponentially suppressed relative to $j=0$.
The domination bound is then driven entirely by $j=0$:
\[|\cK(\xi,\eta)|\leq |\tilde\psi_0(\xi)||\tilde\psi_0(\eta)|+
O(e^{-c^{1/2}})\leq C_M(1+\xi)^{-1/4}(1+\eta)^{-1/4}.\]

\emph{Regime (ii), $\xi>M$, all $j$:}
For $j=0$: $\xi>M>\xi_{n^*}^*$ (oscillatory or decaying);
WKB amplitude $C(1+\xi)^{-1/4}$ for $\xi\in(M,\xi_{n^*,\mathrm{osc}})$,
then exponential decay.
For $j\neq 0$: still evanescent for $\xi<\xi_{n^*+j}^*$,
exponentially decaying.
Product bound: $(ii)$ follows.
\end{proof}

\section{Uniformity in the Edge Window}
\label{sec:uniformity}
%% ============================================================

\begin{lemma}[Turning-point separation — corrected]
\label{lem:tpshift}
Fix $K\geq 1$ and $M>0$. For $c\geq (M+2)^3$:
\begin{enumerate}[nosep,label=(\roman*)]
  \item \emph{$j=0$ (Airy mode):}
    $\xi_{n^*}^*\in[0,2]$ (in Airy coordinates).
    The turning point lies inside $[0,M]$ for $M\geq 2$.
  \item \emph{$1\leq|j|\leq K$ (evanescent modes):}
    $\xi_{n^*+j}^* = |j|c^{1/3}+O(1)\geq c^{1/3}-C$.
    For $c^{1/3}\geq M+C+1$: the interval $[0,M]$
    lies entirely to the left of the turning point $\xi_{n^*+j}^*$.
\end{enumerate}
Consequently, the Olver bound on $[0,M]$ applies with
\emph{different characters}:
Airy-type for $j=0$, exponential-evanescent for $|j|\geq 1$.
There is no ``uniform Olver constant'' for all $|j|\leq K$ simultaneously
on $[0,M]$; instead the $|j|\geq 1$ contributions are
$O(e^{-c^{1/2}})$ relative to $j=0$.
\end{lemma}

\begin{proof}
Equation~\eqref{eq:tp-xi} with $j=0$:
$\xi_{n^*}^* = s_{n^*}c^{2/3}$,
$|s_{n^*}|\leq c^{-1/3}$,
so $|\xi_{n^*}^*|\leq 1$.
For $j\neq 0$:
$s_{n^*+j} = jc^{-1/3}+O(c^{-2/3})$,
so $\xi_{n^*+j}^* = jc^{1/3}+O(1)$.
For $c^{1/3}\geq M+C+1$: $\xi_{n^*+j}^*\geq c^{1/3}-C > M$.
\end{proof}

\begin{remark}[$j=0$ is the unique Airy mode]
\label{rem:j0unique}
Lemma~\ref{lem:tpshift} shows that the spectral window $|j|\leq K$
splits asymptotically (for large $c$) into:
one genuine Airy mode ($j=0$) with turning point in $[0,M]$,
and $2K$ evanescent modes ($|j|\geq 1$) with turning points
at distance $\sim c^{1/3}\to\infty$ from $[0,M]$.
The Airy universality (kernel convergence) is thus entirely
driven by the $j=0$ mode in the near-field $[0,M]$.
\end{remark}

\begin{lemma}[ODE residual uniformity]
\label{lem:residunif}
For $|j|\leq K$ and $c\geq c_0(K)$:
$\|r_j^{(c)}\|_{L^2(\R_+)}\leq C_0(1+|j|)\,c^{-1/3}$,
$C_0$ independent of $j$, $K$, $c$.
\end{lemma}

\begin{proof}
Paper~XVI Theorem~2.1 gives $\|R_c\|_{\mathrm{op}} = O(c^{-1/3})$
uniformly in $c$.
The residual $r_j^{(c)} = R_c\tilde\psi_j + (a_{j,c}-a_j)\tilde\psi_j$.
$|a_{j,c}-a_j| = O(|j|c^{-1/3})$
(from $s_{n^*+j} = jc^{-1/3}+O(c^{-2/3})$ and Paper~XVII Lem.~2.1).
$\|r_j^{(c)}\|_{L^2}\leq \|R_c\|_{\mathrm{op}} + |a_{j,c}-a_j|
= O(c^{-1/3}) + O(|j|c^{-1/3}) = O((1+|j|)c^{-1/3})$.
\end{proof}

\section{Proof of the Main Lemma}
\label{sec:proof}
%% ============================================================

\textbf{Step 1: Edge/bulk split.}
$|\cK(\xi,\eta)-\cK^{(K)}(\xi,\eta)|\leq h(\xi,\eta)e^{-\pi R}$
(Fermi tail, Paper~XVII, uniform in $c$).

\medskip
\textbf{Step 2: Spectral projector convergence.}

By Definition~\ref{def:Uc} and Remark~\ref{rem:singlespace},
both $\PcK$ and $\PK$ act on $L^2(\R_+)$.
Proposition~\ref{prop:PR} gives $\|G^{(c)}-I_K\|_{\mathrm{op}}=O_K(c^{-1/3})$,
where $G^{(c)}_{jk} = \langle\tilde\psi_j,u_k\rangle_{L^2(\R_+)}$.
Since $\tilde\psi_j = \Uc\psi_{n^*+j}^{(c)}$ by definition,
$G^{(c)}_{jk} = \langle\Uc\psi_{n^*+j}^{(c)},u_k\rangle$,
and $G^{(c)}\to I_K$ implies $\PcK\to\PK$ strongly
(a standard finite-dimensional argument:
$\|\PcK v-\PK v\|\leq \|G^{(c)}-I_K\|_{\mathrm{op}}\|v\|+O_K(c^{-1/3})\to 0$).

For the eigenvalue weights: $\lam{c}{n^*+j}\to 1$ for $j<0$
(below Fermi level) and $\to 0$ for $j>0$
(Paper~XVII Lem.~2.1); the transition at $j=0$ contributes $1/2$.
The weighted sum $\sum_{|j|\leq K}\lam{c}{n^*+j}\PcK^{(j)}$
converges to $\PK$ in the same strong sense.

\medskip
\textbf{Step 3: Finite-window Gram convergence.}

\begin{proposition}
\label{prop:PR}
For fixed $K$:
$\|G^{(c)}-I_K\|_{\mathrm{op}} = O_K(c^{-1/3})$,
where $G^{(c)}_{jk} = \langle\tilde\psi_j,u_k\rangle_{L^2(\R_+)}$.
\end{proposition}

\begin{proof}
(A) ODE: $(-\partial_\xi^2+\xi)\tilde\psi_j = -a_{j,c}\tilde\psi_j+r_j^{(c)}$
(Paper~XVI Thm.~2.1).
(B) Residual: $\|r_j^{(c)}\|_{L^2}\leq C_0(1+K)c^{-1/3}$
(Lemma~\ref{lem:residunif}).
(C) Gap: $\delta_K:=\min_{k\leq K}(a_{k+1}-a_k)>0$ (fixed $K$).
(D) Davis--Kahan:
$\|\tilde\psi_j-u_j\|_{L^2}\leq \|r_j^{(c)}\|_{L^2}/\delta_K
= O(K^{4/3}c^{-1/3})$.
(E) $G^{(c)}_{jk}=\delta_{jk}+O_K(c^{-1/3})$.
\end{proof}

\begin{lemma}[$H^2\to C^0$ upgrade]
\label{lem:H2}
For fixed $K,M$:
$\|\tilde\psi_j-u_j\|_{C^0([0,M])}=O_K(c^{-1/3})$.
\end{lemma}

\begin{proof}
$e_j:=\tilde\psi_j-u_j$ satisfies
$-e_j''+\xi e_j = -a_{j,c}e_j+(a_{j,c}+a_j)u_j+r_j^{(c)}$.
$\|e_j''\|_{L^2([0,M+1])}\leq (M+1)\|e_j\|_{L^2}+|a_{j,c}|\|e_j\|_{L^2}+O_K(c^{-1/3})
= O_K(c^{-1/3})$.
$H^2\hookrightarrow C^0$:
$\|e_j\|_{C^0([0,M])}\leq C_M\|e_j\|_{H^2([0,M])}=O_K(c^{-1/3})$.
\end{proof}

\textbf{Step 4: Summing products.}
For $\xi,\eta\in[0,M]$:
$\cK^{(K)}(\xi,\eta)
= \sum_{|j|\leq K}\lam{c}{n^*+j}(u_j(\xi)+O_K(c^{-1/3}))(u_j(\eta)+O_K(c^{-1/3}))
\to\sum_{k=1}^K u_k(\xi)u_k(\eta) = \cKA^{(K)}(\xi,\eta)$.

\textbf{Step 5: $K\to\infty$.}
$|\cK(\xi,\eta)-\cK^{(K)}(\xi,\eta)|\leq Ce^{-\pi K/c^{1/3}}\to 0$;
$\cKA^{(K)}\to\cKA$ pointwise. \hfill$\square$
