\title[Paper XVIII: Airy Universality]{%
  Paper~XVIII: Airy Kernel Universality (v8)\\[0.3em]
  \large Recentered Airy Scaling, Form-Bounded Perturbation,
  Analytic Family}

\author{\textit{prolate-gram-coercivity programme}}
\date{May 2026}

\begin{abstract}
This paper proves Lemma~\ref{lem:main}:
$\cK(\xit,\xit')\to\cKA(\xit,\xit')$ pointwise in the recentered
Airy coordinate $\xit = c^{2/3}(x_{n^*}-x)$,
closing BR3 with Paper~XVII.

\medskip
\textbf{Structural fix in v8 (central).}
\begin{enumerate}[nosep]
  \item \emph{Recentered Airy coordinate} (Definition~\ref{def:airycoord}):
    $\xit = c^{2/3}(x_{n^*}-x)$
    (not $c^{2/3}(1-x)$).
    The turning point of mode $j=0$ is exactly at $\xit=0$
    \emph{by construction}; all $|j|\leq K$ turning points
    are at $\xit_{n^*+j}^* = O(c^{-1/3})\to 0$.
    WKB regime (i) is now the two-sided interval $[-M,M]$
    (oscillatory for $\xit<0$, evanescent for $\xit>0$),
    restoring the proper edge-limit geometry.
  \item \emph{Form-bounded perturbation and analytic family}
    (Proposition~\ref{prop:formbounded}):
    $R_c$ is form-bounded relative to $\cA$ with relative bound~$0$;
    the family $\Lc(\varepsilon) = \cA + \varepsilon R_c$
    is analytic of type~(A) in the sense of Kato~\cite{Kato1966} Ch.~VII;
    no-crossing of eigenvalues follows rigorously.
\end{enumerate}
\end{abstract}

\section{Recentered Airy Coordinate and Setup}
\label{sec:setup}
%% ============================================================

\begin{definition}[Recentered Airy coordinate]
\label{def:airycoord}
Let $x_{n^*}^* = n^*\pi/(2c)$ be the physical turning point of
modus $j=0$ (PSWF Sturm--Liouville, Paper~XVI Def.~1.1).
The \emph{recentered Airy coordinate} is
\begin{equation}\label{eq:xit}
  \xit := c^{2/3}(x_{n^*}^* - x),
  \qquad x\in[-1,1],
\end{equation}
so that $x_{n^*}^*\mapsto 0$ exactly.
Comparison with the boundary scaling:
$c^{2/3}(1-x) = \xit + c^{2/3}(1-x_{n^*}^*)$;
the \emph{boundary offset}
$\beta_c := c^{2/3}(1-x_{n^*}^*) = c^{2/3}\bigl(1-\tfrac{n^*\pi}{2c}\bigr)$
satisfies $|\beta_c|\leq c^{2/3}/(2c)\cdot\pi = O(c^{-1/3})$
(using $|1-n^*\pi/(2c)|\leq \pi/(2c)$ from $n^*=\lfloor 2c/\pi\rfloor$),
so $c^{2/3}(1-x) = \xit + O(c^{-1/3})$.

\emph{Identification map:}
$(\Uc f)(\xit) := c^{1/3}f(x_{n^*}^*-c^{-2/3}\xit)$;
$\Uc:L^2([-1,1])\to L^2(\R)$ is an approximate isometry with
$\|\Uc\| = 1+O(c^{-1/3})$.

\emph{Rescaled kernel:}
\begin{equation}\label{eq:cKdef}
  \cK(\xit,\xit') :=
  c^{-2/3}K_c(x_{n^*}^*-c^{-2/3}\xit,\; x_{n^*}^*-c^{-2/3}\xit').
\end{equation}

\emph{Airy operator:} $\cA = -\partial_{\xit}^2 + \xit$
on $L^2(\R)$ with decay at $\pm\infty$
(or $L^2(\R_+)$ with Dirichlet BC if working on the half-line;
see Remark~\ref{rem:halfline}).
\end{definition}

\begin{remark}[Half-line vs.\ full-line]
\label{rem:halfline}
In the recentered coordinate $\xit\in\R$:
$\xit>0$ corresponds to $x < x_{n^*}^*$ (evanescent region),
$\xit<0$ corresponds to $x > x_{n^*}^*$ (oscillatory region).
The physical domain $x\in[-1,1]$ maps to
$\xit\in[c^{2/3}(x_{n^*}^*+1), c^{2/3}(x_{n^*}^*-1)]$,
which contains $[-M,M]$ for fixed $M$ and large $c$.
For the Airy operator limit, we work on $L^2(\R)$
(Schr\"odinger operator with linear potential);
the restriction to $L^2(\R_+)$ with Dirichlet BC at $\xit=0$
is the Tracy--Widom convention and is equivalent by odd/even symmetry.
\end{remark}

\begin{definition}[Turning points in recentered coordinate]
\label{def:tprecentered}
The physical turning point of $\psi_{n^*+j}^{(c)}$ is
$x_{n^*+j}^* = (n^*+j)\pi/(2c)$.
In recentered Airy coordinate:
\begin{align}
  \xit_{n^*+j}^*
  &:= c^{2/3}(x_{n^*}^* - x_{n^*+j}^*)
  = c^{2/3}\cdot\frac{(n^*-(n^*+j))\pi}{2c}
  = -\frac{j\pi}{2c^{1/3}}.
  \label{eq:tprecentered}
\end{align}
In particular:
\begin{itemize}[nosep]
  \item $\xit_{n^*}^* = 0$ (exact, no approximation).
  \item $\xit_{n^*+j}^* = -j\pi/(2c^{1/3})\to 0$ for fixed $j$.
  \item Spacing: $\xit_{n^*+j+1}^* - \xit_{n^*+j}^* = -\pi/(2c^{1/3})\to 0$.
  \item \emph{No floor residual $\delta_c$:}
    the recentering absorbs $\delta_c$ exactly.
\end{itemize}
\end{definition}

\begin{remark}[Why recentering eliminates $\delta_c$]
In the boundary coordinate $c^{2/3}(1-x)$,
one has $x_{n^*}^* = 1+O(c^{-1})$ (not exactly $1$),
producing the floor residual $\delta_c$.
In the recentered coordinate~\eqref{eq:xit},
$x_{n^*}^*\mapsto 0$ \emph{by definition},
so the $\delta_c$ term does not appear in~\eqref{eq:tprecentered}.
The two coordinates differ by $O(c^{-1/3})$
(the boundary offset $\beta_c$), which is absorbed into the $O(c^{-1/3})$
operator perturbation $R_c$.
\end{remark}

\begin{proposition}[Airy kernel via Green's function]
\label{prop:airykernel}
$\cKA(\xit,\xit') = \sum_{k=1}^\infty u_k(\xit)u_k(\xit')$,
where $u_k$ are the $L^2$-normalised eigenfunctions of
$\cA = -\partial_{\xit}^2+\xit$ on $L^2(\R_+)$ (Dirichlet BC),
with convergence uniform on compacts and in $L^2$.
\end{proposition}

\begin{proof}
The resolvent kernel $R_z(\xit,\xit') = (\cA-z)^{-1}(\xit,\xit')$
is given by the Green's function of $-\partial^2+\xit-z$:
matching two Airy solutions at $\xit=\xit'$
with the correct Wronskian gives the explicit formula
(Olver~\cite{Olver1997} Ch.~11 \S11.5).
Spectral theorem (Reed--Simon~\cite{ReedSimon4} Thm.~VI.16):
$R_z(\xit,\xit') = \sum_k u_k(\xit)u_k(\xit')/(a_k-z)$.
Spectral projection via contour integral around $\{a_k:a_k\leq\Lambda\}$
gives $P_{[0,\Lambda]}(\xit,\xit') = \sum_{a_k\leq\Lambda}u_k(\xit)u_k(\xit')$.
Completeness ($\sum_k u_k\otimes u_k = \mathrm{Id}$) and
Sobolev $H^1\hookrightarrow C^0$ give uniform convergence on compacts.
\end{proof}

\begin{lemma}[Main Lemma]
\label{lem:main}
For fixed $\xit,\xit'\in\R$:
$\displaystyle\lim_{c\to\infty}\cK(\xit,\xit') = \cKA(\xit,\xit')$.
\end{lemma}

\noindent Proof in Sections~\ref{sec:I2b}--\ref{sec:proof}.

\section{(I2b): Two-Regime Domination}
\label{sec:I2b}
%% ============================================================

\begin{lemma}[Two-regime domination]
\label{lem:I2b}
For fixed $M>0$, there exist $C_M,C'>0$
uniform in $c\geq c_0(M)$:
\begin{enumerate}[nosep,label=(\roman*)]
  \item \emph{Airy window} $\xit,\xit'\in[-M,M]$:
    $|\cK(\xit,\xit')|
    \leq C_M(1+|\xit|)^{-1/4}(1+|\xit'|)^{-1/4}$.
  \item \emph{Far field} $|\xit|>M$ or $|\xit'|>M$:
    $|\cK(\xit,\xit')|
    \leq C'e^{-\frac43(|\xit|\wedge|\xit'|-M)_+^{3/2}}$.
\end{enumerate}
\end{lemma}

\begin{proof}
CD identity (ORX Prop.~5.2, algebraic, Paper~XVI).

\emph{Regime (i): $\xit,\xit'\in[-M,M]$.}
In the recentered coordinate~\eqref{eq:xit},
the turning point of mode $j=0$ is exactly at $\xit=0$.
The rescaled PSWF potential in these coordinates is
$V_c(\xit) = \xit + O(c^{-1/3})$
(from the intertwining identity, Definition~\ref{def:airycoord}
and Paper~XVI Thm.~2.1):
\begin{itemize}[nosep]
  \item For $\xit\in[-M,0)$: oscillatory region
    ($V_c(\xit) < 0$ for small $|\xit|$,
    oscillatory WKB applies).
  \item For $\xit\in(0,M]$: evanescent region
    ($V_c(\xit) > 0$).
\end{itemize}
Olver~\cite{Olver1997} Ch.~11 \S11.3 gives the WKB amplitude bound:
$|\tilde\psi_j(\xit)|\leq C(1+|\xit|)^{-1/4}$,
uniformly in $c$ and $|j|\leq K$
(the potential $V_c(\xit)=\xit+O(c^{-1/3})$ is uniformly bounded
on $[-M,M]$ with bounded derivative,
and $|V_c(\xit)-\xit|\leq Cc^{-1/3}\to 0$ uniformly).

\emph{Regime (ii): $|\xit|>M$.}
For $\xit\gg M$: $V_c(\xit)>0$,
exponentially decaying region;
Olver subdominant solution:
$|\tilde\psi_j(\xit)|\leq Ce^{-\frac23\xit^{3/2}}$.
CD gives (ii) with $C'=C_{M,K}^2/4$.
\end{proof}

\section{Norm-Resolvent Convergence and Form Bound}
\label{sec:nrc}
%% ============================================================

\begin{proposition}[Intertwining and form-bounded perturbation]
\label{prop:formbounded}
In the recentered coordinate~\eqref{eq:xit}:
\begin{equation}\label{eq:intertwine}
  \Lc = \cA + \varepsilon_c\, R_c,
  \qquad \varepsilon_c := c^{-1/3},
\end{equation}
where:
\begin{enumerate}[nosep,label=(\alph*)]
  \item \emph{$L^2$-bound:} $\|R_c\|_{L^2\to L^2}\leq C_0$ (Paper~XVI Thm.~2.1).
  \item \emph{Form bound:} $R_c$ is form-bounded relative to $\cA$
    with relative bound $0$:
    \begin{equation}\label{eq:formbound}
      |\langle f, R_c f\rangle|
      \leq \alpha\langle f,\cA f\rangle + C(\alpha)\|f\|^2
      \quad\forall\, f\in H^1(\R_+),\;\forall\,\alpha>0.
    \end{equation}
    Proof of~\eqref{eq:formbound}:
    $R_c = W_c(\xit)$ is a multiplication operator by
    $W_c(\xit) = (V_c(\xit)-\xit)/c^{-1/3}$,
    which satisfies
    $|W_c(\xit)|\leq C(1+|\xit|)^{1/2}$
    (from the Taylor expansion of the PSWF potential;
    Paper~XVI Def.~1.1).
    The Hardy--Rellich inequality gives
    $\|(1+|\xit|)^{1/2}f\|^2
    \leq C_\alpha\langle f,\cA f\rangle + C'_\alpha\|f\|^2$,
    which is~\eqref{eq:formbound}.
  \item \emph{Analytic family of type~(A):}
    $\Lc(\varepsilon) = \cA + \varepsilon R_c$
    is an analytic family of type~(A) in the sense of
    Kato~\cite{Kato1966} Ch.~VII \S2.1,
    because $R_c$ is form-bounded relative to $\cA$
    with relative bound~$0$
    (Kato~\cite{Kato1966} Ch.~VII Thm.~2.6).
\end{enumerate}
\end{proposition}

\begin{proposition}[Norm-resolvent convergence]
\label{prop:normresolvent}
For $z\in\C\setminus\R$:
\begin{equation}\label{eq:nrc}
  \|(\Lc-z)^{-1} - (\cA-z)^{-1}\|_{L^2\to L^2}
  \leq \frac{C_0\,c^{-1/3}}{(\mathrm{Im}\,z)^2}.
\end{equation}
\end{proposition}

\begin{proof}
Second resolvent identity:
$(\Lc-z)^{-1}-(\cA-z)^{-1}
= -(\Lc-z)^{-1}\cdot c^{-1/3}R_c\cdot(\cA-z)^{-1}$.
Norm bound: $\|(\Lc-z)^{-1}\|\leq|\mathrm{Im}\,z|^{-1}$,
$\|(\cA-z)^{-1}\|\leq|\mathrm{Im}\,z|^{-1}$
(both self-adjoint, Proposition~\ref{prop:formbounded}c).
$\|R_c\|\leq C_0$ from Proposition~\ref{prop:formbounded}a.
Hence~\eqref{eq:nrc}.
\end{proof}

\begin{lemma}[Min-max index identification and no-crossing]
\label{lem:minmax}
For $\varepsilon_c = c^{-1/3}\in[0,1]$:
\begin{enumerate}[nosep,label=(\roman*)]
  \item $a_{j,c} = a_j + O_j(c^{-1/3})$
    (Weyl inequality for form-bounded perturbations;
    Kato~\cite{Kato1966} Ch.~VII Thm.~3.4).
  \item \emph{No-crossing:}
    since $\Lc(\varepsilon)$ is an analytic family of type~(A)
    (Proposition~\ref{prop:formbounded}c),
    the eigenvalues $a_{j,c}(\varepsilon)$ are analytic in $\varepsilon$
    (Kato~\cite{Kato1966} Ch.~VII Thm.~2.6);
    since they are distinct at $\varepsilon=0$ (Airy zeros, simple)
    and the family is analytic, they remain distinct for
    $\varepsilon = c^{-1/3}\in[0,1]$ (analytic continuation).
  \item \emph{Canonical index identification:}
    $a_{1,c}<a_{2,c}<\cdots$ (same ordering as $a_1<a_2<\cdots$);
    mode $\tilde\psi_j$ corresponds to $u_j$ by eigenvalue ordering.
\end{enumerate}
\end{lemma}

\begin{corollary}[Spectral convergence]
\label{cor:spectral}
From Propositions~\ref{prop:normresolvent}--\ref{prop:formbounded}
and Lemma~\ref{lem:minmax}
(Kato~\cite{Kato1966} Ch.~IV Thm.~3.16;
Reed--Simon~\cite{ReedSimon4} Thm.~VIII.23):
\begin{enumerate}[nosep,label=(\roman*)]
  \item $|a_{j,c}-a_j| = O_j(c^{-1/3})$.
  \item $\|P_{j,c}-P_j\|_{L^2\to L^2} = O_j(c^{-1/3})$.
  \item $\|\PcK-\PK\|_{L^2\to L^2} = O_K(c^{-1/3})$.
  \item $G^{(c)}_{jk} = \delta_{jk}+O_K(c^{-1/3})$.
\end{enumerate}
\end{corollary}

\section{Proof of the Main Lemma}
\label{sec:proof}
%% ============================================================

\begin{lemma}[$H^2\to C^0$]
\label{lem:H2}
For fixed $K,M$:
$\|\tilde\psi_j-u_j\|_{C^0([-M,M])}=O_K(c^{-1/3})$.
\end{lemma}

\begin{proof}
$e_j := \tilde\psi_j-u_j$;
$\cA e_j = a_j e_j - (a_{j,c}-a_j)\tilde\psi_j - c^{-1/3}R_c\tilde\psi_j$;
$\|\cA e_j\|_{L^2}=O_j(c^{-1/3})$ (Corollary~\ref{cor:spectral}i +
Proposition~\ref{prop:formbounded}a).
$\|e_j''\|_{L^2}\leq\|\cA e_j\|+M\|e_j\|=O_K(c^{-1/3})$;
$H^2([-M,M])\hookrightarrow C^0([-M,M])$ gives the result.
\end{proof}

\textbf{Step 1: Fermi tail.}
$|\cK(\xit,\xit')-\cK^{(K)}(\xit,\xit')|\leq h(\xit,\xit')e^{-\pi R}$
(Paper~XVII Lem.~2.1, Lemma~\ref{lem:I2b}).

\textbf{Step 2: Projector convergence.}
Corollary~\ref{cor:spectral}(iii);
weights $\lam{c}{n^*+j}\to\mathbf{1}_{j<0}$
(Paper~XVII Lem.~2.1).

\textbf{Step 3: Pointwise products.}
Lemma~\ref{lem:H2}:
$\cK^{(K)}(\xit,\xit')\to\cKA^{(K)}(\xit,\xit')$.

\textbf{Step 4: $K\to\infty$.}
$|\cK-\cK^{(K)}|\leq Ce^{-\pi K/c^{1/3}}\to 0$;
$\cKA^{(K)}\to\cKA$ (Proposition~\ref{prop:airykernel}).
\hfill$\square$
