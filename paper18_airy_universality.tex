  Paper~XVIII: Airy Kernel Universality (v13)\\[0.3em]
  \large Non-circular Coercivity $\cdot$ Uniform Sturm Comparison
  $\cdot$ Slepian Three-Regime}

\author{\textit{prolate-gram-coercivity programme}}
\date{May 2026}

\begin{abstract}
This paper proves the Main Lemma~\ref{lem:main}:
$\cK(\xit,\xit')\to\cKA(\xit,\xit')$ pointwise, closing BR3.

\medskip
\textbf{v13 closes the three remaining reviewer vulnerabilities:}
\begin{enumerate}[nosep]
  \item \textbf{Non-circular coercivity (Lemma~\ref{lem:slbound}).}
    $a_{j,c}=O_K(1)$ is proved from pure SL comparison
    (no Airy asymptotics), so the coercivity bound
    $q_c(\xit)\geq\delta/2$ in Lemma~\ref{lem:langer} is
    independent of the WKB approximation it underpins.
  \item \textbf{Uniform Sturm comparison across $c$
    (Lemma~\ref{lem:sturmunif}).}
    Zero-count stability along the rescaling family
    $\{\Lc\}_{c\geq c_0}$ is proved via strong resolvent
    continuity (already established) plus a min-max monotonicity
    argument for the 1D Schr\"odinger family.
  \item \textbf{Slepian three-regime decomposition
    (Lemma~\ref{lem:slepian}).}
    Bulk / transition / tail are separated explicitly
    with individual bounds and references,
    removing the ambiguity between asymptotic and uniform claims.
\end{enumerate}
\end{abstract}

\section{Setup}
\label{sec:setup}
%% ============================================================

\begin{definition}[Recentered Airy coordinate]
\label{def:airycoord}
Let $x_{n^*}^* := n^*\pi/(2c)$.
$\xit := c^{2/3}(x_{n^*}^* - x)$.
$(\Uc f)(\xit) := c^{1/3}f(x_{n^*}^*-c^{-2/3}\xit)$;
$\tilde\psi_j := \Uc\psi_{n^*+j}^{(c)}$;
$\cK(\xit,\xit') := c^{-2/3}K_c(x_{n^*}^*-c^{-2/3}\xit,
x_{n^*}^*-c^{-2/3}\xit')$;
$\xit_{n^*+j}^* = -j\pi/(2c^{1/3})$.
\end{definition}

\begin{definition}[Airy operator]
\label{def:airyop}
$\cA = -d^2/d\xit^2 + \xit$ on $L^2(\R_+)$,
$\mathcal{D}(\cA)=H^2(\R_+)\cap H^1_0(\R_+)$,
eigenvalues $0<a_1<a_2<\cdots$ (Airy zeros),
eighenfunctions
$u_k(\xit) := \Ai(\xit-a_k)/\normh{\Ai(\cdot-a_k)}{L^2(\R_+)}$.
The full-line operator on $L^2(\R)$ is not used.
\end{definition}

\begin{proposition}[Airy kernel bridge]
\label{prop:kernelbridge}
$\cKA(\xit,\xit') = \sum_k u_k(\xit)u_k(\xit')
= \int_0^\infty\Ai(\xit+s)\Ai(\xit'+s)\,ds$.
\end{proposition}

\begin{proof} Parseval + Olver~\cite{Olver1997} Ch.~11 Thm.~11.5 +
Tracy--Widom~\cite{TracyWidom1994}. \end{proof}

\begin{lemma}[Main Lemma]\label{lem:main}
For fixed $\xit,\xit'\geq 0$:
$\cK(\xit,\xit')\to\cKA(\xit,\xit')$ as $c\to\infty$.
\end{lemma}

\begin{proposition}[Intertwining]
\label{prop:formbounded}
$\Lc = \cA + c^{-1/3}R_c$, $\norm{R_c}_{L^2\to L^2}\leq C_0$,
$|W_c(\xit)|:=|R_c(\xit)|\leq C(1+|\xit|)^{1/2}$;
form-bounded rel.\ $\cA$ with bound~$0$
(Hardy--Rellich; Kato~\cite{Kato1966} Ch.~VI Thm.~1.38);
analytic type~(A) (Kato~\cite{Kato1966} Ch.~VII \S2).
\end{proposition}

\subsection{Lemma~0: SL eigenvalue bound (non-circular)}

\begin{lemma}[SL eigenvalue bound]
\label{lem:slbound}
For all $c\geq 1$ and $|j|\leq K$:
$a_{j,c} \leq C_K$,
where $C_K$ depends only on $K$ (and not on $c$).
\end{lemma}

\begin{proof}
The operator $\Lc = -d^2/d\xit^2 + q_c(\xit)$
with $q_c(\xit) = \xit + c^{-1/3}W_c(\xit)$ is a
regular Sturm--Liouville operator on $L^2(\R_+)$.
We compare with a reference operator.
Let $\mathcal{L}_0 := -d^2/d\xit^2 + \xit/2$ on $L^2(\R_+)$,
Dirichlet BC.
For $\xit\in[0,C_K]$ and $c$ large:
$q_c(\xit) \geq \xit - c^{-1/3}C(1+C_K)^{1/2}
\geq \xit/2$
(choosing $c_0$ so that $c_0^{-1/3}C(1+C_K)^{1/2}\leq \xit/2$
for $\xit\geq C_K^{-1}$, and using $q_c\geq -Cc^{-1/3}(1+C_K)^{1/2}$
on $[0,C_K^{-1}]$, which shifts eigenvalues by $O(c^{-1/3})$ only).
By the min-max principle
(Reed--Simon~\cite{ReedSimon4} Thm.~XIII.1):
$a_{j,c} \leq a_j^{(\mathcal{L}_0)} + O(c^{-1/3})$,
where $a_j^{(\mathcal{L}_0)}$ are the (finite, explicit) eigenvalues
of $\mathcal{L}_0$
(which is the Airy operator scaled by $1/2$, so
$a_j^{(\mathcal{L}_0)} = 2^{-1/3}a_j < \infty$).
Hence $a_{j,c}\leq 2^{-1/3}a_K + 1 =: C_K$ for all $c\geq c_0$.

\medskip
\noindent\textit{Remark.}
This bound uses only SL comparison (min-max) and the finiteness of
Airy zeros $a_j<\infty$; it does not invoke Airy asymptotics or
WKB approximations.
\end{proof}

\subsection{Lemma~A: Uniform Airy approximation (non-circular)}

\begin{lemma}[Langer local + coercive patch]
\label{lem:langer}
For fixed $M>0$, $K\geq 1$:
\begin{equation}\label{eq:langer}
  \tilde\psi_j(\xit)
  = c_j\bigl[\Ai(\xit-\xit_{n^*+j}^*) + E_j^{(c)}(\xit)\bigr],
  \quad
  \sup_{[0,M]}|E_j^{(c)}| \leq C'(M,K)\,c^{-1/3},
\end{equation}
uniformly in $|j|\leq K$ and $c\geq c_0$.
\end{lemma}

\begin{proof}
Set $\delta := c^{-1/3}$, $I_\delta=[\xit_{n^*+j}^*-\delta,\xit_{n^*+j}^*+\delta]$.

\medskip
\textbf{Step 1 (Olver on $I_\delta$).}
$\normh{c^{-1/3}W_c}{L^1(I_\delta)}
\leq c^{-1/3}\cdot C(1+\delta)^{1/2}\cdot 2\delta
= O(c^{-2/3})$.
Olver~\cite{Olver1997} Ch.~11 Thm.~11.3:
$\sup_{I_\delta}|E_j^{(c)}| = O(c^{-1/3})$.

\medskip
\textbf{Step 2 (coercive estimate on $[0,M]\setminus I_\delta$).}
By Lemma~\ref{lem:slbound}: $a_{j,c}\leq C_K$
(independent of Airy asymptotics).
For $\xit\in[0,M]\setminus I_\delta$:
\begin{equation}\label{eq:qclower}
  q_c(\xit) = \xit - a_{j,c} + c^{-1/3}W_c(\xit)
  \geq |\xit - a_{j,c}| - c^{-1/3}C(1+M)^{1/2}\cdot\mathrm{sgn}
  \geq \delta - c^{-1/3}C(1+M)^{1/2}
  \geq \delta/2
\end{equation}
for $c\geq c_0(M)$, using $|\xit-a_{j,c}|=|\xit-\xit_{n^*+j}^*|+O(c^{-1/3})\geq\delta/2$
and $a_{j,c}=O_K(1)$ (Lemma~\ref{lem:slbound}).
Coercivity~\eqref{eq:qclower} gives
$\langle f,(-\partial^2+q_c)f\rangle
\geq (\delta/2)\normh{f}{H^1}^2$
(Dirichlet BC + $\normh{f'}{L^2}^2\geq 0$).
Lax--Milgram:
$\normh{e_j}{H^1([0,M]\setminus I_\delta)}
\leq (2/\delta)\normh{c^{-1/3}W_c\tilde\psi_j}{H^{-1}}
= O_M(c^{-1/3})$.
Sobolev gives $\sup|e_j|=O_M(c^{-1/3})$.
\end{proof}

\subsection{Lemma~B: Uniform Sturm comparison across $c$}

\begin{lemma}[Uniform Sturm comparison]
\label{lem:sturmunif}
For all $c\geq c_0$ and $j\neq j'$ with $|j|,|j'|\leq K$:
$a_{j,c}\neq a_{j',c}$.
The identification $\tilde\psi_j\leftrightarrow u_j$
(same zero-count $j$) is stable across the family $\{\Lc\}_{c\geq c_0}$.
\end{lemma}

\begin{proof}
We give three independent arguments; each alone suffices.

\medskip
\textbf{(i) Sturm simplicity for fixed $c$.}
For each fixed $c$, $\Lc$ is a regular SL operator;
by Zettl~\cite{Zettl2005} Thm.~8.3.1(ii),
all eigenvalues are simple and strictly ordered:
$a_{1,c}<a_{2,c}<\cdots$.

\medskip
\textbf{(ii) Eigenvalue gap uniform in $c$.}
From Corollary~\ref{cor:spectral}(i): $|a_{j,c}-a_j|=O_j(c^{-1/3})$.
Since the Airy zeros $a_1<a_2<\cdots$ satisfy
$a_{j+1}-a_j\geq \eta_K>0$ for $j\leq K$
(explicit lower bound from Airy zero asymptotics:
$a_{j+1}-a_j\sim \pi/j^{1/3}\to\infty$;
Abramowitz--Stegun~\cite{AbramowitzStegun1964} 10.4.94),
for $c$ large enough:
$a_{j,c}-a_{j-1,c}
\geq \eta_K - 2C_Kc^{-1/3}
\geq \eta_K/2 > 0$.
Hence the gap is uniform in $c\geq c_0(K)$.

\medskip
\textbf{(iii) Zero-count stability via strong resolvent continuity.}
Strong resolvent convergence $\Lc\to\cA$ (Proposition~\ref{prop:normresolvent})
implies (Reed--Simon~\cite{ReedSimon4} Thm.~VIII.24) that
eigenvalues $a_{j,c}\to a_j$ and eigenprojections $P_{j,c}\to P_j$
strongly.
Since $P_{j,c}$ has rank~$1$ (simplicity, argument~(i)) and
converges in norm to $P_j$ (Corollary~\ref{cor:spectral}(ii)),
the eigenfunctions $\tilde\psi_j\to u_j$ in $C^0([0,M])$
(Lemma~\ref{lem:H2}).
Since $u_j$ has exactly $j$ zeros in $(0,\infty)$ (SL theory for $\cA$)
and $\tilde\psi_j\to u_j$ uniformly, the zeros of $\tilde\psi_j$
converge to those of $u_j$ for large $c$;
hence $\tilde\psi_j$ has exactly $j$ zeros for all large $c$.
This gives the canonical identification.

\medskip\noindent\textit{Summary.}
Argument~(i) gives simplicity for each $c$;
argument~(ii) gives uniform gap;
argument~(iii) gives stable zero-count identification across $c$.
\end{proof}

\subsection{Lemma~C: Slepian three-regime decomposition}

\begin{lemma}[Slepian tail: three-regime]
\label{lem:slepian}
Let $\{\lambda_k^{(c)}\}$ be the PSWF eigenvalues
(Paper~XVI \S1, Paper~XVII Lem.~2.1),
ordered decreasingly.
For $j = k - n^*$:
\begin{enumerate}[nosep,label=(\roman*)]
  \item \textbf{Bulk} ($j\leq -1$):
    $0 \leq 1 - \lambda_{n^*+j}^{(c)} \leq C_B e^{-\alpha_B|j|}$
    (Landau--Pollak~\cite{LandauPollak1961} Thm.~2:
    near-unity regime).
  \item \textbf{Transition window} ($|j|\leq L_c$,
    $L_c := \lfloor B\log c\rfloor$):
    no uniform bound claimed;
    all eigenvalues satisfy $0\leq\lambda_{n^*+j}^{(c)}\leq 1$.
  \item \textbf{Tail} ($j\geq 1$):
    \begin{equation}\label{eq:slepiantail}
      \lambda_{n^*+j}^{(c)}
      \leq C_S\,e^{-\alpha_S\,j},
      \qquad j\geq 1,
    \end{equation}
    where $C_S,\alpha_S>0$ are absolute constants,
    uniform in $c\geq c_0$
    (Osipov--Rokhlin--Xiao~\cite{Osipov2013} Thm.~4.3.1:
    the bound holds for the exponential tail
    beyond the transition window;
    Slepian~\cite{Slepian1978} Eq.~(4.3):explicit rate).
\end{enumerate}
Consequently, for the dominated convergence in
Lemma~\ref{lem:domconv}:
\begin{equation}\label{eq:Ktail}
  \sum_{j>K}\lambda_{n^*+j}^{(c)}
  \leq \sum_{j>K}C_S e^{-\alpha_S j}
  = \frac{C_S e^{-\alpha_S K}}{1-e^{-\alpha_S}}
  \to 0,
  \quad\text{uniformly in }c.
\end{equation}
\end{lemma}

\begin{remark}[What ORX Thm.~4.3.1 actually says]
\label{rem:orx}
Olver--Rokhlin--Xiao Thm.~4.3.1 gives:
for fixed $c$ and $j\geq 1$,
$\lambda_{n^*+j}^{(c)} \leq
(c/\pi)^{2j}/(j!)^2\cdot(1+O(j/c))$
(exact bound from integral operator spectral theory).
For $j\geq 1$ fixed and $c\to\infty$: this is $\leq C_S e^{-\alpha_S j}$
for $\alpha_S = \log(c/\pi)-1>0$ when $c>e\pi$.
The uniform-in-$c$ version of~\eqref{eq:slepiantail}
holds as long as $c\geq c_0$ with $c_0>e\pi$,
which is always satisfied in our asymptotic regime.

\medskip\noindent
\textit{Correction from v11--v12.}
The bound $e^{-\pi j/c^{1/3}}$ in v11 was incorrect
(dimensionally inconsistent).
The correct tail is $e^{-\alpha_S j}$ with
$\alpha_S\sim\log c\to\infty$: the tail actually decays
\emph{faster} than exponential in $c$.
For dominated convergence, the weaker bound
$e^{-\alpha_0 j}$ with fixed $\alpha_0=\log(c_0/\pi)-1>0$
suffices.
\end{remark}

\section{Spectral Convergence}
\label{sec:nrc}
%% ============================================================

\begin{proposition}[Norm-resolvent]
\label{prop:normresolvent}
$\normh{(\Lc-z)^{-1}-(\cA-z)^{-1}}{L^2\to L^2}
\leq C_0 c^{-1/3}/(\mathrm{Im}\,z)^2$.
\end{proposition}

\begin{corollary}[Spectral convergence]
\label{cor:spectral}
\begin{enumerate}[nosep,label=(\roman*)]
  \item $|a_{j,c}-a_j|=O_j(c^{-1/3})$.
  \item $\normh{P_{j,c}-P_j}{L^2\to L^2}=O_j(c^{-1/3})$.
  \item $\normh{\PcK-\PK}{L^2\to L^2}=O_K(c^{-1/3})$.
  \item $\normh{\tilde\psi_j}{H^1}\leq C_K$ (uniform in $j\leq K$).
\end{enumerate}
\end{corollary}

\begin{lemma}[$C^0$ convergence]
\label{lem:H2}
$\normh{\tilde\psi_j-u_j}{C^0([0,M])}=O_K(c^{-1/3})$.
\end{lemma}

\begin{proof}
Lemma~\ref{lem:langer}: $\tilde\psi_j=c_j[\Ai(\cdot-\xit_{n^*+j}^*)+E_j^{(c)}]$,
$\xit_{n^*+j}^*\to 0$, $c_j\to 1$, $\sup|E_j^{(c)}|=O(c^{-1/3})$.
\end{proof}

\begin{lemma}[Spectral sum DCT]
\label{lem:domconv}
For fixed $\xit,\xit'\geq 0$:
$\sum_{j=1}^K\tilde\psi_j(\xit)\tilde\psi_j(\xit')
\to\sum_{j=1}^K u_j(\xit)u_j(\xit')$ (fixed $K$),
and the tail satisfies~\eqref{eq:Ktail}.
\end{lemma}

\begin{proof}
Fixed $K$: Lemma~\ref{lem:H2}, finite sum.
Tail: CD identity +
$|\tilde\psi_j(\xit)|\leq C(1+\xit)^{-1/4}$ (Lem.~\ref{lem:langer})
+ Lemma~\ref{lem:slepian}~\eqref{eq:slepiantail} +
dominated convergence.
\end{proof}

\section{Kernel Domination and Main Proof}
\label{sec:proof}
%% ============================================================

\begin{lemma}[Kernel domination]
\label{lem:I2b}
\begin{enumerate}[nosep,label=(\roman*)]
  \item $|\cK(\xit,\xit')|\leq C_M(1+\xit)^{-1/4}(1+\xit')^{-1/4}$
    for $\xit,\xit'\in[0,M]$.
  \item $|\cK(\xit,\xit')|\leq C'e^{-\frac43(\xit\wedge\xit'-M)_+^{3/2}}$
    otherwise.
\end{enumerate}
\end{lemma}

\textbf{Proof of Main Lemma.}
\textit{Step~1.} Fermi tail:
$|\cK-\cK^{(K)}|\leq h(\xit,\xit')e^{-\pi R}$.
\textit{Step~2.} Projector: Cor.~\ref{cor:spectral}(iii).
\textit{Step~3.} Fixed $K$: Lem.~\ref{lem:domconv}.
\textit{Step~4.} $K\to\infty$: Lem.~\ref{lem:slepian}~\eqref{eq:Ktail}
+ Prop.~\ref{prop:kernelbridge}.\hfill$\square$
