\title[Paper XVIII: Airy Universality]{%
  Paper~XVIII: Airy Kernel Universality (v15)\\[0.3em]
  \large Form Ordering $\cdot$ $H^{-1}$ Duality $\cdot$
  Uniform $L^\infty$ Majorization}

\author{\textit{prolate-gram-coercivity programme}}
\date{May 2026}

\begin{abstract}
This paper proves: $\cK(\xit,\xit')\to\cKA(\xit,\xit')$ pointwise
(Main Lemma~\ref{lem:main}), closing BR3.

\medskip
\textbf{v15: three functional-analytic patches.}
\begin{enumerate}[nosep]
\item \textbf{Global SL bound via form ordering (Lemma~\ref{lem:slbound}):}
  quadratic form comparison on all of $\R_+$
  (Kato~\cite{Kato1966} Ch.~VI Thm.~2.21)
  gives $a_{j,c}\leq a_j+O_j(c^{-1/3})$ globally,
  replacing the local-only comparison of v13--v14.
  Coercivity constant $\delta_0=2C_K+1$ now
  follows from this global bound.
\item \textbf{$H^{-1}$ via $H^1_0$ duality (Lemma~\ref{lem:langer} Step~3):}
  $\|W_c\tilde\psi_j\|_{H^{-1}}$
  bounded via $H^1_0$-$L^\infty$ duality in 1D
  (Sobolev $H^1\hookrightarrow L^\infty$, $n=1$),
  replacing the incorrect $L^1\hookrightarrow H^{-1}$ claim.
\item \textbf{Uniform $L^\infty$ majorization for DCT
  (Lemma~\ref{lem:linf}):}
  $\|\tilde\psi_j\|_{L^\infty([0,M])}\leq C_{M,K}$
  uniformly in $j\leq K$ and $c\geq c_0$,
  from $H^2$ regularity of SL eigenfunctions + Sobolev.
  This closes the dominated convergence argument for spectral sums.
\end{enumerate}
\end{abstract}

\section{Setup}
\label{sec:setup}
%% ============================================================

\begin{definition}[Rescaled coordinate]
\label{def:airycoord}
$\xit = c^{2/3}(x_{n^*}^*-x)$,
$(\Uc f)(\xit) = c^{1/3}f(x_{n^*}^*-c^{-2/3}\xit)$,
$\tilde\psi_j = \Uc\psi_{n^*+j}^{(c)}$,
$\cK(\xit,\xit') = c^{-2/3}K_c(x_{n^*}^*-c^{-2/3}\xit,
x_{n^*}^*-c^{-2/3}\xit')$.

Airy natural window: $c^{-2/3}$ in the $\xit$-variable.
Turning-point shift:
$\xit_{n^*+j}^* = -j\pi/(2c^{1/3})$;
this is the $\xit$-image of the PSWF turning point
$x_{n^*+j}^* = x_{n^*}^* + j\pi/(2c)$ under the map $\Uc$
(Paper~XVI Lem.~3.2);
note $\xit_{n^*+j}^*\to 0$ as $c\to\infty$ for fixed $j$.
\end{definition}

\begin{definition}[Airy operator]
\label{def:airyop}
$\cA = -d^2/d\xit^2+\xit$ on $L^2(\R_+)$,
$\mathcal{D}(\cA)=H^2\cap H^1_0$;
eigenvalues $0<a_1<a_2<\cdots$,
normed eigenfunctions
$u_k = \Ai(\cdot-a_k)/\normh{\Ai(\cdot-a_k)}{L^2(\R_+)}$.
\end{definition}

\begin{proposition}[Airy kernel bridge]
\label{prop:kernelbridge}
$\cKA(\xit,\xit') = \sum_k u_k(\xit)u_k(\xit')
= \int_0^\infty\Ai(\xit+s)\Ai(\xit'+s)\,ds$;
$\cKA\in L^1(\R_+\times\R_+)\cap L^2(\R_+\times\R_+)$.
\end{proposition}
\begin{proof}
Parseval + Olver~\cite{Olver1997} Ch.~11 Thm.~11.5
+ Tracy--Widom~\cite{TracyWidom1994};
$L^1$: Airy decay $|\Ai(s)|\leq Ce^{-\frac23 s^{3/2}}$.
\end{proof}

\begin{lemma}[Main Lemma]\label{lem:main}
For fixed $\xit,\xit'\geq 0$:
$\cK(\xit,\xit')\to\cKA(\xit,\xit')$ as $c\to\infty$.
\end{lemma}

Proof in Section~\ref{sec:proof}.

\begin{proposition}[Intertwining]
\label{prop:formbounded}
$\Lc=\cA+c^{-1/3}R_c$;
$R_c=W_c(\xit)$, $|W_c|\leq C(1+|\xit|)^{1/2}$,
$\normh{R_c}{L^2\to L^2}\leq C_0$;
form-relatively-bounded w.r.t.\ $\cA$, relative bound $0$
(Hardy--Rellich; Kato~\cite{Kato1966} Ch.~VI Thm.~1.38);
type~(A) family (Kato~\cite{Kato1966} Ch.~VII \S2).
\end{proposition}

\subsection{Lemma~0: Global SL eigenvalue bound via form ordering}

\begin{lemma}[Global eigenvalue bound]
\label{lem:slbound}
For all $c\geq 1$ and $|j|\leq K$:
$a_{j,c} = a_j + O_j(c^{-1/3})$.
In particular, $a_{j,c}\leq C_K := a_K+1$ for $c\geq c_0(K)$.
\end{lemma}

\begin{proof}
Denote by $\qa_c[f] := \normh{f'}{L^2}^2+\langle q_c f,f\rangle$
the closed quadratic form of $\Lc$ on $H^1_0(\R_+)$,
and $\qa_0[f] := \normh{f'}{L^2}^2+\langle \xit f,f\rangle$
the form of $\cA$.
From Proposition~\ref{prop:formbounded}:
\begin{equation}\label{eq:formperturb}
  \qa_c[f] = \qa_0[f] + c^{-1/3}\langle W_c f,f\rangle,
  \quad f\in H^1_0(\R_+).
\end{equation}
Form-relative bound of $W_c$:
by Hardy--Rellich (Kato~\cite{Kato1966} Ch.~VI Thm.~1.38),
for any $\varepsilon>0$ there exists $C_\varepsilon$ such that
$|\langle W_c f,f\rangle|\leq\varepsilon\qa_0[f]+C_\varepsilon\normh{f}{L^2}^2$.
Hence on $H^1_0(\R_+)$ globally:
\[
  \qa_0[f]-c^{-1/3}(\varepsilon\qa_0[f]+C_\varepsilon\normh{f}{L^2}^2)
  \leq\qa_c[f]
  \leq\qa_0[f]+c^{-1/3}(\varepsilon\qa_0[f]+C_\varepsilon\normh{f}{L^2}^2).
\]
Choosing $\varepsilon=c^{-1/3}/2$ (which requires $c$ large):
$(1-c^{-2/3}/2)\qa_0[f] - c^{-1/3}C_\varepsilon\normh{f}{L^2}^2
\leq\qa_c[f]$.
By the min-max principle for closed forms
(Reed--Simon~\cite{ReedSimon4} Thm.~XIII.1)
applied globally on $H^1_0(\R_+)$:
$a_{j,c} = a_j+O_j(c^{-1/3})$ for each $j$
(this is Kato~\cite{Kato1966} Ch.~VI Thm.~2.21 and Ch.~IV Thm.~3.16).
The constant $C_K=a_K+1$ works for all $c\geq c_0(K)$.
\end{proof}

\subsection{Lemma~A: Uniform Airy approximation}

\begin{lemma}[Langer + coercive, correct scales]
\label{lem:langer}
$\sup_{[0,M]}|E_j^{(c)}|\leq C'(M,K)c^{-1/3}$, uniformly in $|j|\leq K$,
where $\tilde\psi_j = c_j[\Ai(\cdot-\xit_{n^*+j}^*)+E_j^{(c)}]$,
$c_j=1+O_K(c^{-1/3})$.
\end{lemma}

\begin{proof}
Set $I_c = [a_{j,c}^*-c^{-2/3},a_{j,c}^*+c^{-2/3}]$
(Airy-scale window; $a_{j,c}^*$ = turning point of $q_c$);
choose $\delta_0 = 2C_K+2$ (fixed, from Lemma~\ref{lem:slbound}).

\medskip
\textbf{Step 1 (Olver on $I_c$).}
$\normh{c^{-1/3}W_c}{L^1(I_c)}\leq 2Cc^{-1}$;
Olver~\cite{Olver1997} Ch.~11 Thm.~11.3:
$\sup_{I_c}|E_j^{(c)}|=O(c^{-1})$.

\medskip
\textbf{Step 2 (Coercivity on $[\delta_0,M]$ as form inequality).}
For $\xit\in[\delta_0,M]$:
$q_c(\xit) = \xit-a_{j,c}+c^{-1/3}W_c(\xit)
\geq\delta_0-C_K-c^{-1/3}C(1+M)^{1/2}
\geq\delta_0/2$
(Lemma~\ref{lem:slbound}: $a_{j,c}\leq C_K$;
and $c^{-1/3}C(1+M)^{1/2}\leq\delta_0/4$ for $c\geq c_0(M,K)$).
As a Hilbert space form inequality on $H^1_0([\delta_0,M])$:
\begin{equation}\label{eq:formineq}
  \qa_c[f]\geq\normh{f'}{L^2}^2+(\delta_0/2)\normh{f}{L^2}^2
  \geq\mu\normh{f}{H^1}^2,
  \quad \mu=\min(1,\delta_0/2)>0.
\end{equation}
This is an operator inequality $-\partial^2+q_c\geq\mu\cdot\mathrm{Id}$
on $L^2([\delta_0,M])$ (form sense; Kato~\cite{Kato1966} Ch.~VI \S2.1).

\medskip
\textbf{Step 3 ($H^1$ bound via $H^1_0$-duality, $n=1$).}
Let $e_j=\tilde\psi_j-c_j\Ai(\cdot-\xit_{n^*+j}^*)$;
$(-\partial^2+q_c)e_j = c^{-1/3}W_c\tilde\psi_j$ on $[\delta_0,M]$.
By Lax--Milgram with constant $\mu$:
$\normh{e_j}{H^1([\delta_0,M])}
\leq\mu^{-1}\normh{c^{-1/3}W_c\tilde\psi_j}{H^{-1}([\delta_0,M])}$.

\noindent$H^{-1}$ via duality with $H^1_0([\delta_0,M])$:
For $\phi\in H^1_0([\delta_0,M])$, $\dim=1$,
Sobolev $H^1([\delta_0,M])\hookrightarrow L^\infty([\delta_0,M])$
(Morrey, $n=1$: $\normh{\phi}{L^\infty}\leq C_{\delta_0,M}\normh{\phi}{H^1}$).
Hence:
\begin{equation}\label{eq:Hm1}
  \bigl|\langle W_c\tilde\psi_j,\phi\rangle\bigr|
  \leq\normh{W_c}{L^2([\delta_0,M])}\normh{\tilde\psi_j}{L^2}
    \normh{\phi}{L^\infty}
  \leq C_M\cdot C_{\delta_0,M}\normh{\phi}{H^1},
\end{equation}
so $\normh{W_c\tilde\psi_j}{H^{-1}([\delta_0,M])}\leq C_M C_{\delta_0,M}$,
and
$\normh{e_j}{H^1([\delta_0,M])}\leq C_{M,K}c^{-1/3}/\mu$.
Sobolev $H^1\hookrightarrow C^0$:
$\sup_{[\delta_0,M]}|e_j|=O_{M,K}(c^{-1/3})$, uniform in $j\leq K$.

\medskip
\textbf{Normalization.}
$N'(a)=-2\Ai(-a)^2$ (differentiate under $\int$);
$|N(a_{j,c}^*)-N(a_j)|\leq 2(\sup|\Ai|)^2|a_{j,c}^*-a_j|
=O_K(c^{-1/3})$;
$c_j=1+O_K(c^{-1/3})$.
\end{proof}

\subsection{Lemma~$L^\infty$: Uniform eigenfunction bound}

\begin{lemma}[Uniform $L^\infty$ bound]
\label{lem:linf}
For fixed $M>0$, $K\geq 1$:
$\normh{\tilde\psi_j}{L^\infty([0,M])}\leq C_{M,K}$,
uniformly in $|j|\leq K$ and $c\geq c_0$.
\end{lemma}

\begin{proof}
$\tilde\psi_j$ satisfies $-\tilde\psi_j''+q_c\tilde\psi_j=0$
on $[0,M]$ with $\normh{\tilde\psi_j}{L^2}=1$.
Standard $H^2$ regularity for 1D SL equations:
$\normh{\tilde\psi_j}{H^2([0,M])}\leq C(1+\normh{q_c}{L^\infty([0,M])})
\normh{\tilde\psi_j}{L^2}$.
Since $\normh{q_c}{L^\infty([0,M])}
\leq M+C_K+c^{-1/3}C(1+M)^{1/2}\leq C'_{M,K}$
(using Lemma~\ref{lem:slbound}: $a_{j,c}\leq C_K$):
$\normh{\tilde\psi_j}{H^2([0,M])}\leq C_{M,K}$.
Sobolev $H^2([0,M])\hookrightarrow H^1([0,M])\hookrightarrow L^\infty([0,M])$
($n=1$, Morrey):
$\normh{\tilde\psi_j}{L^\infty([0,M])}\leq C_{M,K}$.
\end{proof}

\subsection{Lemma~B: Sturm ordering}

\begin{lemma}[Eigenvalue ordering, stable in $c$]
\label{lem:sturmunif}
For all $c\geq c_0$, $1\leq j<j'\leq K$: $a_{j,c}<a_{j',c}$;
$\tilde\psi_j\leftrightarrow u_j$ canonically (same zero-count).
\end{lemma}

\begin{proof}
\textbf{(i)} Zettl~\cite{Zettl2005} Thm.~8.3.1: for each fixed $c$,
all eigenvalues simple, strictly ordered.
\textbf{(ii)} Uniform gap: Lemma~\ref{lem:slbound} +
$a_{j+1}-a_j\geq\eta_K>0$ (Airy zero spacing) +
$|a_{j,c}-a_j|=O_j(c^{-1/3})$
$\Rightarrow a_{j,c}-a_{j-1,c}\geq\eta_K/2$ for $c\geq c_0(K)$.
\textbf{(iii)} Stable zero-count: $\tilde\psi_j\to u_j$ in
$C^0([0,M])$ (Lemma~\ref{lem:H2}), so
zero-count converges and identification is stable for large $c$.
\end{proof}

\subsection{Lemma~C: Trace-class tail}

\begin{lemma}[Spectral tail]
\label{lem:slepian}
For each fixed $c\geq c_0$:
$\sum_{j>K}\lambda_{n^*+j}^{(c)}\to 0$ as $K\to\infty$
(trace-class: $\mathrm{tr}(B_c)=(2c/\pi)+O(\log c)$,
Landau--Widom~\cite{LandauWidom1980}).
For dominated convergence: the bound
\begin{equation}\label{eq:dombound}
  \Bigl|\sum_{j=1}^K\tilde\psi_j(\xit)\tilde\psi_j(\xit')\Bigr|
  \leq K\cdot C_{M,K}^2
\end{equation}
holds uniformly in $c$ (Lemma~\ref{lem:linf}),
and in the limit $c\to\infty$:
$\Bigl|\sum_{j=1}^K u_j(\xit)u_j(\xit')\Bigr|
\leq\cKA(\xit,\xit')\in L^1(\R_+^2)$
(Proposition~\ref{prop:kernelbridge}).
\end{lemma}

\section{Spectral Convergence and Resolvent}
\label{sec:nrc}
%% ============================================================

\begin{proposition}[Norm-resolvent convergence]
\label{prop:normresolvent}
For $\mathrm{Im}\,z\neq 0$, the second resolvent identity gives:
\begin{equation}\label{eq:resid}
  (\Lc-z)^{-1}-(\cA-z)^{-1}
  = -(\cA-z)^{-1}\cdot c^{-1/3}R_c\cdot(\Lc-z)^{-1},
\end{equation}
hence
$\normh{(\Lc-z)^{-1}-(\cA-z)^{-1}}{}
\leq C_0c^{-1/3}/(\mathrm{Im}\,z)^2$.

Norm-resolvent convergence implies
strong convergence of spectral projections
(Reed--Simon~\cite{ReedSimon4} Thm.~VIII.23--24):
$P_{j,c}\to P_j$ strongly as $c\to\infty$.
This is what BR3 (via XVII Thm.~4.2) requires.
\end{proposition}

\begin{corollary}[Spectral convergence]
\label{cor:spectral}
(i) $|a_{j,c}-a_j|=O_j(c^{-1/3})$;
(ii) $\normh{P_{j,c}-P_j}{}=O_j(c^{-1/3})$;
(iii) $\normh{\PcK-\PK}{}=O_K(c^{-1/3})$;
(iv) $\normh{\tilde\psi_j}{H^1}\leq C_K$.
\end{corollary}

\begin{lemma}[$C^0$ convergence]
\label{lem:H2}
$\normh{\tilde\psi_j-u_j}{C^0([0,M])}=O_K(c^{-1/3})$.
\end{lemma}
\begin{proof}
Lemma~\ref{lem:langer}: $c_j\to 1$,
$\xit_{n^*+j}^*\to 0$, $\sup|E_j^{(c)}|=O(c^{-1/3})$.
\end{proof}

\begin{lemma}[Spectral sum DCT]
\label{lem:domconv}
For fixed $\xit,\xit'\in[0,M]$:
\begin{equation}
  \sum_{j=1}^K\tilde\psi_j(\xit)\tilde\psi_j(\xit')
  \xrightarrow{c\to\infty}
  \sum_{j=1}^K u_j(\xit)u_j(\xit')
  \xrightarrow{K\to\infty}
  \cKA(\xit,\xit').
\end{equation}
\end{lemma}
\begin{proof}
Fixed $K$: Lemma~\ref{lem:H2} (pointwise convergence of products).
$K\to\infty$: Lemma~\ref{lem:slepian}.
Domination for the $K\to\infty$ step in $c$:
bounded by $\cKA\in L^1$ (Lemma~\ref{lem:slepian}~\eqref{eq:dombound}
and Proposition~\ref{prop:kernelbridge}).
Lebesgue DCT applies.
\end{proof}

\section{Proof of Main Lemma}
\label{sec:proof}
%% ============================================================

\begin{lemma}[Kernel domination]
\label{lem:I2b}
$|\cK(\xit,\xit')|\leq C_M(1+\xit)^{-1/4}(1+\xit')^{-1/4}$
for $\xit,\xit'\in[0,M]$;
Airy decay outside.
\end{lemma}

\textbf{Proof of Main Lemma.}
\textit{Step~1.} Fermi tail (XVII Lem.~2.1 + Lem.~\ref{lem:I2b}).
\textit{Step~2.} Projector convergence (Cor.~\ref{cor:spectral}(iii)).
\textit{Step~3.} Fixed-$K$: Lem.~\ref{lem:domconv}.
\textit{Step~4.} $K\to\infty$: Lem.~\ref{lem:slepian} + Prop.~\ref{prop:kernelbridge}.
$\hfill\square$
