\begin{abstract}
This paper proves (I2a): the rescaled PSWF kernel $\cK(\xi,\eta)$
converges pointwise to the Airy kernel $\cKA(\xi,\eta)$.
Combined with Paper~XVII, this closes BR3.

\medskip
\textbf{Corrections in v2 (addressing reviewer concerns).}
\begin{enumerate}[nosep]
  \item \emph{(I2b) / Lemma~\ref{lem:I2b}}:
    replaced by a two-regime domination lemma
    (near-edge $\xi\in[0,M]$ via local Olver;
    far-edge $\xi>M$ via exponential WKB decay).
    Uniformity in $c$ and $\xi$ is now explicit in each regime.
  \item \emph{Proposition~\ref{prop:PR}}:
    reformulated as a \emph{finite-window operator convergence statement}:
    for fixed $K$, the $K\times K$ Gram matrix
    $G^{(c)}_{jk} = \langle\tilde\psi_{n^*+j},u_k\rangle$
    satisfies $\|G^{(c)}-I_K\|=O(c^{-1/3}/\delta_K)$,
    with explicit constant $\delta_K>0$.
    This is \emph{not} uniform in $K$.
  \item \emph{Residual uniformity}:
    the bound $\|r_n\|_{L^2}=O(c^{-1/3})$
    is stated with explicit $K$-dependence;
    the required $H^2$-control
    is separated as Sub-Lemma~\ref{lem:H2}.
\end{enumerate}
\end{abstract}

\section{Setup and Main Statement}
\label{sec:setup}
%% ============================================================

\begin{definition}[Kernels and rescaling]
\label{def:kernels}
Fermi index $n^* = \lfloor 2c/\pi\rfloor$.
Rescaled PSWF:
$\tilde\psi_j(\xi) := c^{1/3}\psi_{n^*+j}^{(c)}(1-c^{-2/3}\xi)$,
$\xi\geq 0$, $|j|\leq K$.\\
Rescaled kernel:
$\cK(\xi,\eta) := c^{-2/3}K_c(1-c^{-2/3}\xi,1-c^{-2/3}\eta)$.\\
Airy kernel:
$\cKA(\xi,\eta) := \int_0^\infty\Ai(\xi+t)\Ai(\eta+t)\,dt$
$= \sum_{k=1}^\infty u_k(\xi)u_k(\eta)$
($u_k$: $L^2(\R_+)$-normalised eigenfunctions of
$\cA=-\partial_\xi^2+\xi$, $\cA u_k=-a_k u_k$, $a_k>0$, $u_k(0)=0$).\\
Spectral window: $\Sedge := \sum_{|j|\leq K}|\tilde\psi_j\rangle\langle\tilde\psi_j|$,
$\PK := \sum_{k=1}^K|u_k\rangle\langle u_k|$.
\end{definition}

\begin{lemma}[Main Lemma]
\label{lem:main}
For fixed $\xi,\eta\geq 0$:
$\lim_{c\to\infty}\cK(\xi,\eta) = \cKA(\xi,\eta)$.
\end{lemma}

The proof occupies Sections~\ref{sec:I2b}--\ref{sec:proof}.

%% ============================================================
\section{(I2b): Two-Regime Kernel Domination}
\label{sec:I2b}
%% ============================================================

\begin{lemma}[Two-regime uniform domination]
\label{lem:I2b}
For any fixed $M>0$ there exist $C_M,C'>0$,
uniform in $c\geq c_0(M)$, such that:
\begin{enumerate}[nosep,label=(\roman*)]
  \item \emph{Near-edge regime, $0\leq\xi,\eta\leq M$:}
    \begin{equation}\label{eq:near}
      |\cK(\xi,\eta)|\leq C_M(1+\xi)^{-1/4}(1+\eta)^{-1/4}.
    \end{equation}
  \item \emph{Far-edge regime, $\xi>M$ or $\eta>M$:}
    for $c$ large enough that
    the turning point $\xi_{n^*}^* = s_{n^*}c^{2/3}\geq 2M$:
    \begin{equation}\label{eq:far}
      |\cK(\xi,\eta)| \leq C'\,e^{-\frac{4}{3}(\xi\wedge\eta - M)^{3/2}}\,
      (1+\xi\vee\eta)^{-1/4}.
    \end{equation}
\end{enumerate}
In both regimes $h(\xi,\eta) := C_M(1+\xi)^{-1/4}(1+\eta)^{-1/4}
+ C'e^{-\frac{4}{3}(\xi\wedge\eta-M)_+^{3/2}}$
is in $L^2(w\otimes w)$ for $w=(1+\xi)^{-\beta}$, $\beta>1/4$.
\end{lemma}

\begin{proof}
\emph{Regime (i): $0\leq\xi,\eta\leq M$.}

\textit{Christoffel--Darboux.}
By ORX Proposition~5.2:
\begin{equation}\label{eq:CD}
  K_c(x,y) =
  \frac{\psi_{n^*}^{(c)}(x)(\partial_y\psi_{n^*}^{(c)})(y)
    - \psi_{n^*}^{(c)}(y)(\partial_x\psi_{n^*}^{(c)})(x)}{x-y}.
\end{equation}
This is the classical CD identity for orthogonal projections,
a purely algebraic consequence of the PSWF recurrence (ORX Ch.~3).
No Airy or scaling input.

\textit{Local Olver bound.}
At the Fermi index $n^*$, the rescaled PSWF $\tilde\psi_0(\xi)$
is at its turning point ($s_{n^*}=O(c^{-1/3})\approx 0$).
For $\xi\in[0,M]$ and $M$ fixed:
the potential $q(\xi) = \xi_{n^*}^*-\xi > -M$ is bounded below.
Olver Ch.~11 Theorem~11.3 applies on the compact interval $[0,M]$:
\begin{equation}\label{eq:olver-local}
  |\psi_{n^*}^{(c)}(1-c^{-2/3}\xi)|
  \leq C(M)\,c^{-1/6}(1+\xi)^{-1/4},
  \quad \xi\in[0,M],
\end{equation}
with constant $C(M)$ depending on $M$ but independent of $c\geq c_0(M)$.
The key point: on $[0,M]$ the Airy function $\Ai$ and its
perturbation are bounded uniformly (no transition to exponential)
because the turning point $\xi_{n^*}^*=O(1)$ is at most $O(1)$ away.

\textit{Rescaled CD bound.}
Substituting~\eqref{eq:olver-local} into~\eqref{eq:CD}
and tracking the $c$-rescaling
($x-y = c^{-2/3}(\xi-\eta)$,
$\partial_x = -c^{2/3}\partial_\xi$,
$c^{-2/3}K_c(\cdot)=\cK$):
the $c$-powers cancel as in Paper~XVIII v1
(two factors of $c^{-1/6}$ times $c^{2/3}$ from the derivative
times $c^{-2/3}$ from the definition of $\cK$),
giving~\eqref{eq:near} with $C_M = C(M)^2$.

\emph{Regime (ii): $\xi > M$.}

For $\xi > M$ and $c$ large so $\xi_{n^*}^* = s_{n^*}c^{2/3}\geq 2M$:
if $\xi > \xi_{n^*}^*$, then $\tilde\psi_0(\xi)$ is in the
exponentially decaying region (Olver Ch.~11 subdominant solution):
\begin{equation}\label{eq:olver-far}
  |\tilde\psi_0(\xi)| \leq C'\,(\xi-\xi_{n^*}^*)^{-1/4}
  \exp\!\bigl(-\tfrac{2}{3}(\xi-\xi_{n^*}^*)^{3/2}\bigr).
\end{equation}
For $M < \xi < \xi_{n^*}^*$ (oscillatory region between $M$ and turning point):
$|\tilde\psi_0(\xi)|\leq C(1+\xi)^{-1/4}$ by the WKB oscillatory amplitude
(Olver Ch.~11 Proposition~11.1).
Together, plugging into the CD formula,
using that at least one of $\xi,\eta$ exceeds $M$:
the exponentially decaying factor gives~\eqref{eq:far}.

\emph{$L^2(w\otimes w)$-integrability of $h$.}
Near-edge: $\int_0^M(1+\xi)^{-1/4-\beta}\,d\xi<\infty$ for $\beta>-3/4$.
Far-edge: $\int_M^\infty e^{-\frac{4}{3}(\xi-M)^{3/2}}(1+\xi)^{-\beta}\,d\xi<\infty$
for all $\beta$. Sum is in $L^2(w\otimes w)$ for $\beta>1/4$.
\end{proof}

\begin{remark}[Why regime splitting is necessary]
\label{rem:regime-split}
A single Olver bound ``$\xi\in[0,\infty)$'' would require:
(a) matching the turning-point Airy form to the
oscillatory WKB form at $\xi\sim\xi_{n^*}^*$, and
(b) a global uniform constant across regimes.
This matching at the turning point is exactly the hard content of Proposition~\ref{prop:PR};
it cannot be assumed in the domination lemma without circularity.
Splitting at $\xi=M\ll\xi_{n^*}^*$ avoids the turning-point matching
in Lemma~\ref{lem:I2b}: near-edge uses only the
compact Olver bound (before any matching), far-edge uses only
the exponential decay (after the turning point is crossed).
\end{remark}

%% ============================================================
\section{Proof of the Main Lemma}
\label{sec:proof}
%% ============================================================

\textbf{Step 1: Edge/bulk split and $R\to\infty$.}
Identical to Paper~XVIII v1 \S3 Step~1.
Bulk error $O(e^{-\pi R})$ uniformly in $c$.

\medskip
\textbf{Step 2: Fermi projection $\to$ Airy projection.}

For $|j|\leq K$, $\lam{c}{n^*+j}\to 1$ as $c\to\infty$
(those indices are below $n^*$: $s_{n^*+j}=j/c^{1/3}+O(c^{-1/3})\to 0^-$ for $j<0$,
or near $0$ for $j\sim 0$).
More precisely (by Paper~XVII Lemma~2.1):
$\lam{c}{n^*+j} = F(s_{n^*+j})+O(c^{-1/3})$.
For the $K$-window sum:
$\sum_{|j|\leq K}\lam{c}{n^*+j}\tilde\psi_j\otimes\tilde\psi_j
\to\sum_{|j|\leq K}\tilde\psi_j\otimes\tilde\psi_j$
(using $F(s)\to 1$ for $s<0$ and $F(s)\to 0$ for $s>0$;
for $s_{n^*+j}$ near $0$: $F(s)=1/2+O(s)$, and the
corresponding $\tilde\psi_j$ are controlled by Proposition~\ref{prop:PR}).

\begin{remark}[Fermi $\to$ Airy projector: explicit statement]
\label{rem:projector}
The Airy kernel $\cKA = \sum_{k\geq 1}u_k\otimes u_k$
projects onto the $L^2(\R_+)$-closure of the
negative spectral subspace of $\cA$.
For fixed $K$: $\PK\cKA\PK = \sum_{k=1}^K u_k\otimes u_k$.
The convergence statement in Step~4 gives:
$\Sedge\cK\Sedge \to \PK\cKA\PK$ (as operators on the $K$-window),
with error $O(c^{-1/3}/\delta_K)$.
This is strong operator convergence on the finite-dimensional window,
not uniform-in-$K$.
\end{remark}

\medskip
\textbf{Step 3: Finite-window operator convergence (Proposition~\ref{prop:PR}).}

\begin{proposition}[Finite-window Gram convergence]
\label{prop:PR}
Fix $K\geq 1$. Let $G^{(c)}\in\R^{K\times K}$ be the Gram matrix:
$G^{(c)}_{jk} := \langle\tilde\psi_j, u_k\rangle_{L^2(\R_+)}$,
$1\leq j,k\leq K$.
Then:
\begin{equation}\label{eq:gramconv}
  \|G^{(c)} - I_K\|_{\mathrm{op}} = O_K(c^{-1/3}),
\end{equation}
where the constant depends on $K$ (via $\delta_K$)
but not on $c$.
\end{proposition}

\begin{proof}
We use the framework of Paper~XVI Theorem~2.1
(intertwining + Davis--Kahan).

\emph{Step A: ODE for $\tilde\psi_j$.}
From Paper~XVI Definition~1.1 and Theorem~2.1:
$\tilde\psi_j$ satisfies
\begin{equation}\label{eq:approxairy}
  (-\partial_\xi^2 + \xi)\tilde\psi_j = -a_{j,c}\tilde\psi_j + r_j^{(c)},
\end{equation}
where $a_{j,c} := -c^{1/3}(\lam{c}{n^*+j}-z_{n^*})\to a_j$ (Airy zero),
and the ODE residual satisfies:
\begin{equation}\label{eq:resid}
  \|r_j^{(c)}\|_{L^2(\R_+)} \leq C_K\,c^{-1/3},
  \quad |j|\leq K,
\end{equation}
with $C_K$ depending on $K$ but uniform in $c\geq c_0(K)$.

\emph{Uniformity in $j$, $|j|\leq K$:}
This follows from Paper~XVI Theorem~2.1 applied to $n = n^*+j$
(the intertwining remainder $\|R_c\|_{\mathrm{op}} = O(c^{-1/3})$
holds for all $n$ in a fixed spectral window
$|n-n^*|\leq K$ by the stability of the WKB approximation:
the turning-point shift by $j$ is at most $O(Kc^{-1/3})$
in the $\xi$-variable, remaining within the Airy regime).

\emph{Step B: Airy spectral gap.}
The Airy operator $\cA = -\partial_\xi^2+\xi$ on $L^2(\R_+)$
has eigenvalues $-a_1>-a_2>\ldots$
with $a_k\sim(\frac{3\pi}{2}(k-\frac14))^{2/3}$.
For the $K$-window, the minimum gap is
$\delta_K := \min_{1\leq j\leq K}(a_{j+1}-a_j) > 0$.
Note: $\delta_K\to 0$ as $K\to\infty$;
we fix $K$ throughout.

\emph{Step C: Davis--Kahan.}
Since $\tilde\psi_j$ is an approximate eigenfunction of $\cA$
with residual~\eqref{eq:resid} and gap $\delta_K$:
\[
  \|\tilde\psi_j - u_j\|_{L^2} \leq \frac{\|r_j^{(c)}\|_{L^2}}{\delta_K}
  = O(C_K\,c^{-1/3}/\delta_K).
\]
Therefore $G^{(c)}_{jk} = \langle\tilde\psi_j,u_k\rangle
= \delta_{jk} + O(C_K c^{-1/3}/\delta_K)$,
giving~\eqref{eq:gramconv}.
\end{proof}

\begin{lemma}[$H^2$ control and pointwise convergence]
\label{lem:H2}
For fixed $K$ and $M>0$:
$\|\tilde\psi_j - u_j\|_{C^0([0,M])} = O_K(c^{-1/3})$.
\end{lemma}

\begin{proof}
Let $e_j := \tilde\psi_j - u_j$. From~\eqref{eq:approxairy}:
$-e_j'' + \xi e_j = -a_{j,c}e_j + (a_{j,c}+a_j)u_j + r_j^{(c)}$.
Since $|a_{j,c}+a_j|=O(c^{-1/3})$ and $\|u_j\|_{H^2([0,M+1])}$ is bounded
(fixed $j,M$):
$\|e_j''\|_{L^2([0,M+1])}
\leq \|\xi e_j\|_{L^2}+|a_{j,c}|\|e_j\|_{L^2}
+O(c^{-1/3})$.
Since $\xi\leq M+1$ on $[0,M+1]$ and $\|e_j\|_{L^2}=O(c^{-1/3})$:
$\|e_j''\|_{L^2([0,M+1])}=O_K(c^{-1/3})$.
Thus $\|e_j\|_{H^2([0,M])}=O_K(c^{-1/3})$.
Sobolev embedding $H^2([0,M])\hookrightarrow C^0([0,M])$:
$\|e_j\|_{C^0([0,M])}\leq C_M\|e_j\|_{H^2([0,M])}=O_K(c^{-1/3})$.
\end{proof}

\begin{remark}[$H^2$ step is separate from Davis--Kahan]
\label{rem:H2-separate}
Proposition~\ref{prop:PR} gives $L^2$-closeness.
Lemma~\ref{lem:H2} upgrades to $C^0([0,M])$ via $H^2$.
The $H^2$ step requires controlling $e_j''$,
which needs the full ODE~\eqref{eq:approxairy},
not just the $L^2$-residual.
This is the reason the two results are separated:
Proposition~\ref{prop:PR} (operator level) $\to$
Lemma~\ref{lem:H2} (function level).
\end{remark}

\medskip
\textbf{Step 4: Summing eigenfunction products.}

For fixed $K$ and $\xi,\eta\in[0,M]$:
\begin{align*}
  \cK^{(K)}(\xi,\eta)
  &:= \sum_{j=1}^K\lam{c}{n^*+j}\,
    c^{-1/3}\tilde\psi_j(\xi)\cdot c^{-1/3}\tilde\psi_j(\eta)\\
  &= \sum_{j=1}^K\lam{c}{n^*+j}\,
    \bigl(u_j(\xi)+O_K(c^{-1/3})\bigr)
    \bigl(u_j(\eta)+O_K(c^{-1/3})\bigr)\\
  &\to \sum_{j=1}^K u_j(\xi)u_j(\eta)
  = \cKA^{(K)}(\xi,\eta),
\end{align*}
where we used $\lam{c}{n^*+j}\to 1$ (Step~2) and
Lemma~\ref{lem:H2} for pointwise convergence.
The $c^{-2/3}$ normalisation factor cancels as in Remark~\ref{rem:factorcount}.

\medskip
\textbf{Step 5: Remove the cutoff $K\to\infty$.}

$|\cK(\xi,\eta)-\cK^{(K)}(\xi,\eta)|\leq C e^{-\pi K/c^{1/3}}\to 0$
(Fermi tail, Paper~XVII Lemma~2.1).
$\cKA^{(K)}(\xi,\eta)\to\cKA(\xi,\eta)$ as $K\to\infty$
(convergence of Airy eigenfunction expansion, standard).
\hfill$\square$

\begin{remark}[Factor accounting]
\label{rem:factorcount}
$\cK(\xi,\eta)=c^{-2/3}K_c(1-c^{-2/3}\xi,1-c^{-2/3}\eta)$;
the Jacobian $dx=c^{-2/3}d\xi$ gives
$\|\tilde\psi_j\|_{L^2(\R_+)} = 1+O(c^{-2/3})$;
the product $c^{1/3}\psi\cdot c^{1/3}\psi$ times $c^{-2/3}$ from $\cK$
gives net factor $1$.
\end{remark}

\section{Open Points and BR3 Status}
\label{sec:open}
%% ============================================================

\begin{theorem}[BR3 under Lemma~\ref{lem:main}]
\label{thm:br3}
Lemma~\ref{lem:main} + Paper~XVII Theorem~4.2
$\Rightarrow$ $\|\cK-\cKA\|_{L^2(w\otimes w)}\to 0$.
\end{theorem}

\begin{problem}[Two residual checks]
\label{prob:residual}
The following two estimates are asserted but require verification
beyond what is currently in the programme:
\begin{enumerate}[nosep,label=(R\arabic*)]
  \item \emph{ODE residual uniformity in $K$-window:}
    the constant $C_K$ in~\eqref{eq:resid}
    must not grow faster than $\delta_K^{-1}$
    as $K\to\infty$, or equivalently,
    the ratio $C_K/\delta_K$ must be bounded.
    Currently: $C_K=O(K)$ (linear in window size) and
    $\delta_K=\Theta(K^{-1/3})$ (Airy gap asymptotics),
    so $C_K/\delta_K = O(K^{4/3})$. This is fine for each fixed $K$,
    and the $K\to\infty$ limit is controlled by Step~5
    separately; the residual check is only needed for fixed $K$.
  \item \emph{Turning-point stability for $|j|\leq K$:}
    the turning-point shift $\xi_{n^*+j}^*-\xi_{n^*}^*
    = jc^{-1/3}\to 0$ for fixed $j$,
    so all $K$ modes share the same Airy regime for large $c$.
    This needs: the Olver local bound~\eqref{eq:olver-local}
    holds with the \emph{same} constant $C(M)$
    for all $|j|\leq K$ (a finite check for each $K$).
\end{enumerate}
Both are expected to hold from WKB stability;
none requires new analytic ideas.
\end{problem}

\begin{remark}[Honest status: what is conditional]
\label{rem:honest}
Paper~XVIII asserts BR3 conditionally on
Problem~\ref{prob:residual}~(R1)--(R2).
The structure is:
\[
  \underbrace{(I1)+(I3)+\text{Bridge}+(I2b)\;\text{(XVII)}}_{\text{unconditional}}
  +\underbrace{(R1)+(R2)}_{\text{finite WKB checks}}
  \Longrightarrow \text{BR3}.
\]
The remaining conditions (R1)--(R2) are
\emph{finite checks on WKB constants},
not conceptually new inputs;
they do not require additional mathematical ideas.
\end{remark}
