\begin{abstract}
We prove that the spectral projector $P_N$ associated with
prolate spheroidal wave functions (PSWFs) satisfies a uniform
$L^2 \to L^2$ bound on bulk sub-intervals, for all truncation
levels $N$ in the bulk regime $N \le \gamma N_{\mathrm{Sh}}$,
$0 < \gamma < 1$.
This establishes Assumption~2.4 of \cite{PaperIII}
unconditionally, completing the bulk stability program
initiated in \cite{PaperI,PaperII,PaperIII,PaperIV}.
The proof combines the Pr\"ufer-based amplitude control
of \cite{PaperIV} with a kernel splitting of the
Christoffel--Darboux projector and the classical Schur test.
\end{abstract}

\section{Introduction}
\label{sec:intro}
% =============================================================

\subsection{Position in the bulk stability program}

This paper completes the bulk stability program for prolate
spheroidal wave functions (PSWFs) initiated in
\cite{PaperI,PaperII,PaperIII,PaperIV}.

The program aims to establish quantitative control of
spectral projectors associated with PSWFs in the bulk regime
\[
  n \le \gamma N, \qquad N = \frac{cT}{\pi}, \quad 0 < \gamma < 1,
\]
with applications to coercivity, trace formulas, and
spectral tail bounds.

In \cite{PaperIII}, the main bulk tail bound is reduced to two
analytic hypotheses:
\begin{itemize}
  \item a semiclassical equidistribution statement for PSWF
  densities, and
  \item a uniform $L^2 \to L^2$ bound for the truncated
  spectral projector $P_N$ on bulk intervals.
\end{itemize}

The first hypothesis was established in \cite{PaperIV} via a
Pr\"ufer phase analysis, yielding an $O(1/n)$ rate of
equidistribution for $\psi_n^2$.

The second hypothesis (Assumption~2.4 of \cite{PaperIII})
remains the final open input.
The purpose of the present paper is to prove this assumption
unconditionally.

% --------------------------------------------------------------
\subsection{Main result}

Let
\[
  K_N(x,y) := \sum_{n=0}^{N-1} \psi_n(x)\psi_n(y)
\]
denote the Christoffel--Darboux kernel of the PSWF system,
and let $P_N$ be the associated integral operator on $[-T,T]$.

\begin{theorem}[Uniform bulk Schur bound]
\label{thm:schur-main}
Fix $0 < \delta < 1$ and $0 < \gamma < 1$, and let
\[
  I_\delta := [-(1-\delta)T,\,(1-\delta)T].
\]
Then for all $N$ and all bandwidth parameters $c > 0$,
the operator $P_N$ satisfies
\[
  \|P_N\|_{L^2 \to L^2(I_\delta)} \le C_{\delta,\gamma},
\]
uniformly for all truncation levels $N$ in the bulk regime
$N \le \gamma\, N_{\mathrm{Sh}}$, where $N_{\mathrm{Sh}} = 2c/\pi$.
\end{theorem}

As an immediate consequence, Assumption~2.4 of
\cite{PaperIII} holds.

\begin{corollary}[Unconditional bulk stability]
\label{cor:bulk-stability}
The bulk tail bound of \cite[Theorem~6.1]{PaperIII}
holds unconditionally for all $n \le \gamma N$.
\end{corollary}

% --------------------------------------------------------------
\subsection{Strategy of proof}

The proof combines three ingredients of different origin.

\medskip\noindent
\textbf{(i) Diagonal control via Pr\"ufer analysis.}
From \cite{PaperIV}, the squared amplitudes satisfy
\[
  \psi_n(x)^2
  = \lambda_n\,\rho_n^{\mathrm{cl}}(x)
    \bigl(1 + O(1/n)\bigr)
\]
uniformly on $I_\delta$. Summation over $n < N$
yields the diagonal bound
\[
  K_N(x,x) \le C_{\delta,\gamma}\,\frac{Nc}{\pi T}.
\]

\medskip\noindent
\textbf{(ii) Off-diagonal decay via kernel splitting.}
Since $N \le \gamma\,\NSh$, the projector $P_N$ is a
sub-projection of $P_M$ with $M := \lfloor\gamma\,\NSh\rfloor$.
We decompose
\[
  K_M = K_{\mathrm{BL}} + \widetilde{E}^- - \widetilde{E}^+,
\]
where $\widetilde{E}^- = \sum_{n<M}(1-\lambda_n)\psi_n\psi_n$
and $\widetilde{E}^+ = \sum_{n \ge M}\lambda_n\psi_n\psi_n$.
The term $\widetilde{E}^-$ is exponentially small ($1-\lambda_n \le e^{-c_1 c}$
for all $n < M \le \gamma\NSh$).
For $\widetilde{E}^+$ we split at $\NSh$:
the true tail $n \ge \NSh$ is exponentially small,
and the near-bulk block $M \le n < \NSh$ is controlled by
the positive semi-definiteness of $\widetilde{E}^+_{\mathrm{near}}$
and its dominance by $\KBL$.

\medskip\noindent
\textbf{(iii) Schur test.}
Combining the diagonal bound with the off-diagonal decay,
we verify the Schur integrability condition
\[
  \sup_{x \in I_\delta}
  \int_{I_\delta} |K_N(x,y)|\,dy \le C_{\delta,\gamma},
\]
which implies Theorem~\ref{thm:schur-main}
via the classical Schur test \cite[Thm.~VI.23]{ReedSimonI}.

% --------------------------------------------------------------
\subsection{Conceptual significance}

The argument shows that no additional microlocal or
pseudodifferential machinery is required to control the PSWF
projector in the bulk regime.
The key structural feature is that the operator $\KBL$
positive-semi-definiteness dominates the near-bulk error:
no estimate on individual eigenvalues $\lambda_n$ in the
transition region $M \le n < \NSh$ is required.

\section{The Diagonal Bound}
\label{sec:diagonal}
% =============================================================

\begin{proposition}[Diagonal bound]
\label{prop:diagonal}
Fix $0 < \delta < 1$ and $0 < \gamma < 1$.
Then for all $x \in I_\delta$ and all
$N \le \gamma\,\NSh$,
\begin{equation}
\label{eq:diagonal}
  K_N(x,x)
  = \sum_{n=0}^{N-1} \psi_n(x)^2
  \le C_{\delta,\gamma}\,\frac{Nc}{\pi T},
\end{equation}
uniformly in $c > 0$.
\end{proposition}

\begin{proof}
The proof proceeds in three steps.

\subsubsection*{Step 1: Pointwise bound via Pr\"ufer amplitudes}

By the Pr\"ufer decomposition, $\psi_n(x)^2 = r_n(x)^2$.
From \cite[Lemma~4.2 (\texttt{lem:amplitude-drift})]{PaperIV},
\begin{equation}
\label{eq:amplitude-drift}
  \frac{r_n(x)^2}{\rWKB(x)^2}
  = 1 + O\!\left(\tfrac{1}{n}\right),
\end{equation}
uniformly for $x \in I_\delta$ and $n \le \gamma\,\NSh$.
From \cite[Lemma~4.3 (\texttt{lem:normalization})]{PaperIV},
\begin{equation}
\label{eq:normalization}
  \frac{\rWKB(x)^2}{2}
  = \lambda_n(c)\,\rhocl(x)
    \left(1 + O\!\left(\tfrac{1}{n}\right)\right).
\end{equation}
Combining \eqref{eq:amplitude-drift} and \eqref{eq:normalization},
\begin{equation}
\label{eq:psi-squared-bound}
  \psi_n(x)^2
  \le C_{\delta,\gamma}\,\lambda_n(c)\,\rhocl(x).
\end{equation}

\subsubsection*{Step 2: Classical density bound in the bulk}

We claim that for all $n \le \gamma\,\NSh$ and $x \in I_\delta$,
\begin{equation}
\label{eq:density-bound}
  \rhocl(x) \le \frac{C_{\delta,\gamma}\,c}{T}.
\end{equation}

By definition (see \cite{PaperIV}, Definition~2.4), the
classical density takes the form
\[
  \rhocl(x)
  = \frac{1/\wcl(x)}
    {\displaystyle\int_{-x_+(n)}^{x_+(n)}
      \frac{dt}{\wcl(t)}},
\]
where the classical frequency is
\begin{equation}\label{eq:wcl-def}
  \wcl(x) := \sqrt{\chi_n(c) - V_c(x)},
  \qquad
  V_c(x) := c^2\!\left(1 - \frac{x^2}{T^2}\right)^{\!-1}\chi_n(c)\,
  \frac{x^2}{T^2},
\end{equation}
and $x_+(n)$ is the classical turning point at which
$\wcl(x_+(n)) = 0$.

\medskip\noindent
\textit{Step 2a: Upper bound on the numerator.}
By \cite[Lemma~2.5(i) (\texttt{lem:G-regularity})]{PaperIV},
\begin{equation}\label{eq:wcl-lower}
  \wcl(x) \ge c_{\delta,\gamma}\,n
  \qquad\text{for all } x \in I_\delta,
\end{equation}
hence $1/\wcl(x) \le C_{\delta,\gamma}/n$.

\medskip\noindent
\textit{Step 2b: Lower bound on the denominator.}
Since the integrand $1/\wcl(t)$ is non-negative,
we restrict to the bulk interval to avoid the turning
points $\pm x_+(n)$, where $\wcl(t) \to 0$:
\[
  \int_{-x_+(n)}^{x_+(n)} \frac{dt}{\wcl(t)}
  \;\ge\;
  \int_{I_\delta} \frac{dt}{\wcl(t)}.
\]
Since $V_c(x) \ge 0$, the definition~\eqref{eq:wcl-def} gives
$\wcl(x) \le \sqrt{\chi_n(c)}$.
For the eigenvalue we use the standard PSWF bound
(see \cite[Chs.~2--4]{OsipovRokhlinXiao}):
\begin{equation}\label{eq:chi-bound}
  \chi_n(c) \le c^2 + C n^2.
\end{equation}
Combining $\wcl(x) \le \sqrt{\chi_n(c)}$
with \eqref{eq:chi-bound} and $n \le \gamma\,\NSh \le c$:
\begin{equation}\label{eq:wcl-upper}
  \wcl(x) \le C_{\delta,\gamma}\,c
  \qquad\text{for all } x \in I_\delta.
\end{equation}
Therefore:
\begin{equation}\label{eq:denom-lower}
  \int_{I_\delta} \frac{dt}{\wcl(t)}
  \;\ge\;
  \frac{c'_{\delta,\gamma}\,T}{c}.
\end{equation}

\medskip\noindent
\textit{Conclusion of Step 2.}
Combining the numerator bound $1/\wcl(x) \le C_{\delta,\gamma}/n$
with \eqref{eq:denom-lower} and $n \ge 1$:
\[
  \rhocl(x)
  \le
  \frac{C_{\delta,\gamma}\,c}{n\,T}
  \le
  \frac{C_{\delta,\gamma}\,c}{T},
\]
establishing~\eqref{eq:density-bound}.

\subsubsection*{Step 3: Summation}

Inserting \eqref{eq:density-bound} into
\eqref{eq:psi-squared-bound} and summing:
\[
  K_N(x,x)
  \le C_{\delta,\gamma}\,\frac{c}{T}
       \sum_{n=0}^{N-1} \lambda_n(c)
  \le C_{\delta,\gamma}\,\frac{Nc}{\pi T},
\]
using $\lambda_n \le 1$.
The $O(1/n)$ corrections from \eqref{eq:psi-squared-bound}
contribute $O(c\log N/T) = o(Nc/T)$,
completing the proof.
\end{proof}

\begin{remark}[Relation to Landau--Widom]
\label{rem:landau-widom}
Proposition~\ref{prop:diagonal} recovers the classical
Landau--Widom diagonal estimate \cite{Widom64}.
The diagonal estimate requires no reproducing kernel
identities or the Landau--Widom trace formula;
it follows directly from the Pr\"ufer amplitude control
of \cite{PaperIV} with constants uniform in all parameters.
\end{remark}

\section{Off-diagonal Decay via Kernel Splitting}
\label{sec:offdiag}
% =============================================================

In this section we establish the off-diagonal bound
\begin{equation}
\label{eq:offdiag-main}
  |K_N(x,y)|
  \;\le\;
  \frac{C_{\delta,\gamma}}{1 + c|x-y|/T}
  \;+\;
  C_{\delta,\gamma}\,\frac{c}{\pi T}
  \;+\;
  C_{\delta,\gamma}\,\frac{c}{T}\,e^{-c_3 c},
\end{equation}
for all $x,y \in I_\delta$ and all $N \le \gamma\,\NSh$,
where $c_3 > 0$ depends only on $\delta$ and $\gamma$.
The second term, of order $c/T$, arises from the near-bulk
block $M \le n < \NSh$ via the $\KBL$-dominance argument;
it is controlled in the Schur test
(Section~\ref{sec:schur}) by a separate integration argument.

% --------------------------------------------------------------
\subsection{The band-limited kernel}
\label{subsec:BL-kernel}

Define the band-limited Christoffel--Darboux kernel as
\[
  \KBL(x,y)
  := \sum_{n=0}^{\infty} \lambda_n\,\psi_n(x)\psi_n(y).
\]
By the reproducing kernel identity for band-limited functions
of bandwidth $c/T$ (see \cite[Prop.~3.1]{OsipovRokhlinXiao}),
this kernel admits the explicit Fourier representation
\begin{equation}
\label{eq:KBL-sinc}
  \KBL(x,y)
  = \frac{c}{\pi T}
    \frac{\sin\!\bigl(c(x-y)/T\bigr)}{c(x-y)/T},
\end{equation}
where the right-hand side is interpreted as $c/(\pi T)$
when $x = y$.

\begin{lemma}[Band-limited kernel bound]
\label{lem:BL-bound}
For all $x, y \in \mathbb{R}$,
\begin{equation}
\label{eq:BL-bound}
  |\KBL(x,y)|
  \;\le\;
  \frac{2c/(\pi T)}{1 + c|x-y|/T}
  \;\le\;
  \frac{C}{1 + c|x-y|/T}.
\end{equation}
\end{lemma}

\begin{proof}
At $x = y$: $\KBL(x,x) = c/(\pi T)$ and the bound holds trivially.
For $x \neq y$: set $u := c|x-y|/T > 0$.
Using $|\sin(u)| \le 1$ and $|\sin(u)/u| \le 1/u$,
\[
  \left|\frac{\sin u}{u}\right|
  \le \min\!\left(1,\,\frac{1}{u}\right)
  \le \frac{2}{1 + u},
\]
where the last step follows by checking both cases $u \le 1$
($2/(1+u) \ge 1$) and $u > 1$ ($2/(1+u) > 1/u$).
Therefore
\[
  |\KBL(x,y)|
  \le \frac{c}{\pi T}\cdot\frac{2}{1 + u}
  = \frac{2c/(\pi T)}{1 + c|x-y|/T}. \qedhere
\]
\end{proof}

% --------------------------------------------------------------
\subsection{Algebraic decomposition}
\label{subsec:decomp}

Since $N \le \gamma\,\NSh$, the projector $P_N$ is a
sub-projection of $P_M$, where $M := \lfloor\gamma\,\NSh\rfloor$,
so $\|P_N\|_{L^2 \to L^2(\Ish)} \le \|P_M\|_{L^2 \to L^2(\Ish)}$.
It therefore suffices to bound the kernel $K_M$.

We write
\begin{equation}
\label{eq:split}
  K_M = \KBL + \widetilde{E}^- - \widetilde{E}^+,
\end{equation}
where
\[
  \widetilde{E}^-(x,y) := \sum_{n < M}(1-\lambda_n)\psi_n(x)\psi_n(y),
  \qquad
  \widetilde{E}^+(x,y) := \sum_{n \ge M}\lambda_n\psi_n(x)\psi_n(y).
\]
We further split $\widetilde{E}^+$ at $\NSh$:
\begin{equation}
\label{eq:Eplus-split-def}
  \widetilde{E}^+ = \widetilde{E}^+_{\mathrm{near}} + \widetilde{E}^+_{\mathrm{tail}},
\end{equation}
where
\[
  \widetilde{E}^+_{\mathrm{near}}(x,y)
  := \sum_{n=M}^{\NSh-1}\lambda_n\psi_n(x)\psi_n(y),
  \qquad
  \widetilde{E}^+_{\mathrm{tail}}(x,y)
  := \sum_{n \ge \NSh}\lambda_n\psi_n(x)\psi_n(y).
\]
The three terms are controlled as follows:
\begin{itemize}
  \item $\widetilde{E}^-$: every index satisfies $n < M \le \gamma\,\NSh$,
  so $1-\lambda_n \le e^{-c_1 c}$ (Lemma~\ref{lem:eigenvalue-decay}(i))
  $\Rightarrow$ exponentially small.
  \item $\widetilde{E}^+_{\mathrm{tail}}$: every index satisfies $n \ge \NSh$,
  so $\lambda_n \le e^{-c_2 c}$ (Lemma~\ref{lem:eigenvalue-decay}(ii))
  $\Rightarrow$ exponentially small.
  \item $\widetilde{E}^+_{\mathrm{near}}$: the near-bulk block $M \le n < \NSh$
  requires no eigenvalue bound in the Airy-scale transition region;
  it is controlled instead by $\KBL$-dominance via positive
  semi-definiteness (Subsection~\ref{subsec:diff-bound}).
\end{itemize}

% --------------------------------------------------------------
\subsection{Spectral decay of PSWF eigenvalues}
\label{subsec:spectral-decay}

\begin{lemma}[Eigenvalue concentration]
\label{lem:eigenvalue-decay}
There exist constants $c_1, c_2 > 0$, depending only on
$\gamma$, such that:
\begin{enumerate}[label=(\roman*)]
  \item \textup{(Bulk eigenvalues near 1.)}
  For all $n \le \gamma\,\NSh$:
  \begin{equation}
  \label{eq:lambda-bulk}
    1 - \lambda_n(c) \;\le\; e^{-c_1 c}.
  \end{equation}
  \item \textup{(Tail eigenvalues near 0.)}
  For all $n \ge \NSh$:
  \begin{equation}
  \label{eq:lambda-tail}
    \lambda_n(c) \;\le\; e^{-c_2 c}.
  \end{equation}
\end{enumerate}
Both bounds are uniform in $c > 0$.
\end{lemma}

\begin{proof}
Part~(i): \cite[Theorem~4.6]{OsipovRokhlinXiao} gives
$1 - \lambda_n \le e^{-c_1(c - n\pi/2)}$;
since $c - n\pi/2 \ge c(1 - \gamma\pi/4) \ge c_\gamma c$,
bound \eqref{eq:lambda-bulk} follows.
Part~(ii): \cite[Theorem~4.9]{OsipovRokhlinXiao} gives
$\lambda_n \le e^{-c_2(n\pi/2 - c)}$;
since $n\pi/2 - c \ge (\pi/2 - 1)c > 0$ for $n \ge \NSh = 2c/\pi$,
bound \eqref{eq:lambda-tail} follows.
\end{proof}

\begin{remark}[Transition region]
\label{rem:transition-region}
For indices $M \le n < \NSh$ in the near-bulk block, neither
bulk bound~\eqref{eq:lambda-bulk} nor tail
boundary~\eqref{eq:lambda-tail} applies:
the eigenvalues $\lambda_n$ transition on the Airy scale
$|n - \NSh| = O(c^{1/3})$ and are not individually small.
This is why $\widetilde{E}^+_{\mathrm{near}}$ requires the
PSD-dominance argument of Subsection~\ref{subsec:diff-bound},
rather than a direct eigenvalue bound.
\end{remark}

% --------------------------------------------------------------
\subsection{Pointwise $L^\infty$ bound for PSWF eigenfunctions}
\label{subsec:Linfty}

\begin{lemma}[$L^\infty$ bound for $\psi_n$]
\label{lem:Linfty}
For all $n \le \gamma\,\NSh$ and all $x \in I_\delta$,
\begin{equation}
\label{eq:Linfty}
  |\psi_n(x)|^2 \;\le\; C_{\delta,\gamma}\,\frac{c}{T}.
\end{equation}
\end{lemma}

\begin{proof}
From \eqref{eq:psi-squared-bound} and \eqref{eq:density-bound}:
$\psi_n(x)^2 \le C_{\delta,\gamma}\,\lambda_n(c)\,\rhocl(x)
\le C_{\delta,\gamma}\,c/T$.
\end{proof}

% --------------------------------------------------------------
\subsection{Bounding the error kernels}
\label{subsec:diff-bound}

\medskip\noindent
\textbf{Bounding $\widetilde{E}^-$.}
For $n < M \le \gamma\,\NSh$, Lemma~\ref{lem:eigenvalue-decay}(i)
and Lemma~\ref{lem:Linfty} give:
\begin{equation}
\label{eq:Eminus-final}
  |\widetilde{E}^-(x,y)|
  \le \frac{C_{\delta,\gamma}\,c}{T}\sum_{n<M}(1-\lambda_n)
  \le C_{\delta,\gamma}\,\frac{c^2}{T}\,e^{-c_1 c},
\end{equation}
using $M \le \gamma\,\NSh \asymp c$.

\medskip\noindent
\textbf{Bounding $\widetilde{E}^+_{\mathrm{tail}}$.}
For $n \ge \NSh$, Lemma~\ref{lem:eigenvalue-decay}(ii)
and Lemma~\ref{lem:Linfty} give:
\begin{equation}
\label{eq:Eplus-tail-final}
  |\widetilde{E}^+_{\mathrm{tail}}(x,y)|
  \le \frac{c}{T}\sum_{n \ge \NSh}\lambda_n
  \le C_{\delta,\gamma}\,\frac{c}{T}\,e^{-c_2 c},
\end{equation}
where we used $\sum_{n \ge \NSh}\lambda_n \le C\,e^{-c_2 c}$
(from \eqref{eq:lambda-tail} and geometric summation).

\medskip\noindent
\textbf{Bounding $\widetilde{E}^+_{\mathrm{near}}$ via $\KBL$-dominance.}
The key observation is that
$\widetilde{E}^+_{\mathrm{near}}$, $\widetilde{E}^+_{\mathrm{tail}}$,
and $\KBL$ are all positive semi-definite (PSD) kernels,
and
\begin{equation}
\label{eq:PSD-order}
  0 \preceq \widetilde{E}^+_{\mathrm{near}}
  \preceq \widetilde{E}^+
  \preceq \KBL,
\end{equation}
since $\KBL - \widetilde{E}^+ = \sum_{n < M}\lambda_n\psi_n\psi_n \succeq 0$
and $\widetilde{E}^+ - \widetilde{E}^+_{\mathrm{near}}
= \widetilde{E}^+_{\mathrm{tail}} \succeq 0$.

By the Cauchy--Schwarz inequality for PSD kernels:
\begin{equation}
\label{eq:PSD-CS}
  |A(x,y)|^2 \le A(x,x)\,A(y,y)
  \quad\text{for any PSD kernel } A.
\end{equation}
Applying \eqref{eq:PSD-CS} to $\widetilde{E}^+_{\mathrm{near}}$
and using \eqref{eq:PSD-order}:
\begin{equation}
\label{eq:Eplus-near-final}
  |\widetilde{E}^+_{\mathrm{near}}(x,y)|^2
  \le \widetilde{E}^+_{\mathrm{near}}(x,x)\,
      \widetilde{E}^+_{\mathrm{near}}(y,y)
  \le \KBL(x,x)\,\KBL(y,y)
  = \left(\frac{c}{\pi T}\right)^{\!2},
\end{equation}
hence
\begin{equation}
\label{eq:Eplus-near-pointwise}
  |\widetilde{E}^+_{\mathrm{near}}(x,y)|
  \;\le\;
  \frac{c}{\pi T}.
\end{equation}
No eigenvalue estimate in the Airy transition region
$M \le n < \NSh$ is required.

\begin{remark}[Why this is the right bound]
\label{rem:KBL-dominance}
The bound \eqref{eq:Eplus-near-pointwise} gives
$|\widetilde{E}^+_{\mathrm{near}}(x,y)| \le c/(\pi T)$,
which matches the diagonal $\KBL(x,x) = c/(\pi T)$.
This is not a \emph{small} bound in the sense of
being $o(1)$ as $c \to \infty$; rather, it is of the
same order as the main term $|\KBL(x,y)|$ at $x = y$.
The Schur test in Section~\ref{sec:schur} handles this
by integrating the kernel:
$\int_{\Ish}(c/\pi T)\,dy \le 2(1-\delta)c/\pi$,
which is absorbed into the final uniform constant
$C_{\delta,\gamma}$.
\end{remark}

\medskip\noindent
\textbf{Combined error bound.}
Collecting \eqref{eq:Eminus-final}, \eqref{eq:Eplus-tail-final},
and \eqref{eq:Eplus-near-pointwise}:
\begin{equation}
\label{eq:KM-KBL-bound}
  |K_M(x,y) - \KBL(x,y)|
  \;\le\;
  C_{\delta,\gamma}\,\frac{c}{\pi T}
  \;+\;
  C_{\delta,\gamma}\,\frac{c}{T}\,e^{-c_3 c},
\end{equation}
where $c_3 := \min(c_1, c_2) > 0$.

% --------------------------------------------------------------
\subsection{Final off-diagonal estimate}

\begin{proposition}[Off-diagonal bound]
\label{prop:offdiag}
For all $x, y \in I_\delta$ and all $N \le \gamma\,\NSh$,
\begin{equation}
\label{eq:offdiag-final}
  |K_N(x,y)|
  \;\le\;
  \frac{C_{\delta,\gamma}}{1 + c|x-y|/T}
  \;+\;
  C_{\delta,\gamma}\,\frac{c}{\pi T}
  \;+\;
  C_{\delta,\gamma}\,\frac{c}{T}\,e^{-c_3 c}.
\end{equation}
All constants are uniform in $c > 0$, $T > 0$,
and $N \le \gamma\,\NSh$.
\end{proposition}

\begin{proof}
Since $K_N$ is a PSD kernel and $N \le M$,
the kernel $K_M - K_N = \sum_{N \le n < M}\psi_n\psi_n \succeq 0$,
so by the PSD Cauchy--Schwarz inequality~\eqref{eq:PSD-CS}:
\[
  |K_N(x,y)|
  \le K_N(x,x)^{1/2}\,K_N(y,y)^{1/2}
  \le K_M(x,x)^{1/2}\,K_M(y,y)^{1/2}.
\]
Moreover, by the triangle inequality and \eqref{eq:split}:
\[
  |K_M(x,y)|
  \le |\KBL(x,y)|
     + |\widetilde{E}^-(x,y)|
     + |\widetilde{E}^+_{\mathrm{near}}(x,y)|
     + |\widetilde{E}^+_{\mathrm{tail}}(x,y)|.
\]
Applying Lemma~\ref{lem:BL-bound},
\eqref{eq:Eminus-final}, \eqref{eq:Eplus-tail-final},
and \eqref{eq:Eplus-near-pointwise}
gives \eqref{eq:offdiag-final}.
\end{proof}
