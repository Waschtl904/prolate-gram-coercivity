\begin{abstract}
We establish two results that together complete the bulk stability
program for prolate spheroidal wave functions (PSWFs) in the strict
bulk regime $0 < \gamma < 1/2$.

First, we prove that the out-of-band spectral energy
$\Eout(f_{mn}) := \int_{|\xi|>\omega}|\widehat{f}_{mn}(\xi)|^2\,d\xi$
of the PSWF product $f_{mn} = \psi_m \psi_n$ decays exponentially
in the bandwidth parameter $c$, uniformly for $m,n \le \gamma \NSh$
with $\gamma < 1/2$.
The proof uses a geometric forbidden-zone cover argument
(Theorem~\ref{thm:bridge}):
for $\gamma < 1/2$, the classical forbidden zones of $\psi_m$ and
$\psi_n$ together cover the entire convolution overlap interval,
so at every integration point at least one PSWF factor is
exponentially suppressed by WKB tunneling.
This proves Assumption~ass:bulkconv of \cite{PaperIII}
unconditionally.

Second, we use the Pr\"ufer-based amplitude control of
\cite{PaperIV}, a kernel splitting of the Christoffel--Darboux
projector, and the Schur test to establish a uniform
$L^2 \to L^2$ bound for the spectral projector $P_N$ on bulk
sub-intervals (Theorem~\ref{thm:schur-main}), for all
$N \le \gamma \NSh$ with $\gamma < 1/2$.
This establishes Assumption~2.4 of \cite{PaperIII}
unconditionally in that range.

Both results rely on two ingredients only: the WKB forbidden-zone
decay of PSWFs (from \cite{OsipovRokhlinXiao}) and the
geometry of the convolution overlap, requiring no phase-cancellation
or off-diagonal index decay.
\end{abstract}

\section{Introduction}
\label{sec:intro}
% =============================================================

\subsection{Position in the bulk stability program}

This paper closes two open inputs in the bulk stability program
for PSWFs initiated in \cite{PaperI,PaperII,PaperIII,PaperIV}.
Both results are established unconditionally, without any
hypothesis on quadrature schemes, oscillatory cancellation,
or index-dependent off-diagonal decay.

The program aims to establish quantitative control of the
spectral projector $P_N$ associated with the PSWF eigenbasis
$\{\psin\}_{n \ge 0}$ of the concentration operator
$\mathcal{K}_c$ on $L^2([-T,T])$, in the bulk regime
\[
  n \le \gamma N, \qquad N \le \gamma\,\NSh,
  \qquad \NSh := \frac{2c}{\pi},
  \quad 0 < \gamma < \tfrac{1}{2}.
\]
The time-bandwidth product is $c = \omega T$ with $\omega > 0$
the bandwidth and $T > 0$ the time-window half-length.

In \cite{PaperIII}, the main bulk tail bound is reduced to
two analytic hypotheses:
\begin{enumerate}[label=(H\arabic*)]
  \item\label{H1} A semiclassical equidistribution statement for
  PSWF densities $\psin^2$.
  \item\label{H2} A uniform $L^2 \to L^2$ bound for $P_N$ on
  bulk sub-intervals $I_\delta = [-(1-\delta)T, (1-\delta)T]$.
\end{enumerate}
Hypothesis~\ref{H1} was established in \cite{PaperIV} via a
Pr\"ufer phase analysis, yielding an $O(1/n)$ equidistribution
rate for $\psi_n^2$ uniformly on $I_\delta$.
Hypothesis~\ref{H2} is the subject of the present paper.

Additionally, the spectral tail estimate of \cite[Assumption
ass:bulkconv]{PaperIII} requires:
\begin{enumerate}[label=(H\arabic*),resume]
  \item\label{H3} Exponential decay of the out-of-band energy
  $\Eout(f_{mn}) \le C_\gamma\,e^{-\alpha_\gamma c}$ for all
  $m,n \le \gamma\,\NSh$.
\end{enumerate}
This paper proves both \ref{H2} and \ref{H3} unconditionally
for $\gamma < 1/2$.

\subsection{Main results}
\label{subsec:main-results}

Let $K_N(x,y) := \sum_{n=0}^{N-1}\psin(x)\psin(y)$ denote
the Christoffel--Darboux kernel and $P_N$ the associated
integral operator on $L^2([-T,T])$.

\begin{theorem}[Bridge Lemma: exponential out-of-band decay]
\label{thm:bridge-intro}
Fix $0 < \gamma < 1/2$.
There exist $C_\gamma > 0$ and $K_\gamma > 0$ such that for
all $m, n \le \gamma\,\NSh$ and all $c \ge 1$:
\[
  \Eout(\fmn) \;\le\; C_\gamma\,e^{-K_\gamma c}.
\]
In particular, Assumption~ass:bulkconv of \cite{PaperIII}
holds unconditionally.
\end{theorem}

\begin{theorem}[Uniform bulk Schur bound]
\label{thm:schur-main}
Fix $0 < \delta < 1$ and $0 < \gamma < 1/2$.
Then for all $N \le \gamma\,\NSh$ and all $c \ge 1$:
\[
  \|P_N\|_{L^2 \to L^2(I_\delta)} \le C_{\delta,\gamma},
\]
with $C_{\delta,\gamma}$ independent of $N$ and $c$.
\end{theorem}

\begin{corollary}[Unconditional bulk tail bound]
\label{cor:bulk-stability}
The bulk spectral tail bound of \cite[Theorem~3.2]{PaperIII}
holds unconditionally for all $m,n,N \le \gamma\,\NSh$
with $\gamma < 1/2$:
\[
  \|(I-P_N)f_{mn}\|_{L^2} \;\le\; C_\gamma\,e^{-\alpha_\gamma' N}.
\]
\end{corollary}

\subsection{Key ideas}
\label{subsec:ideas}

\medskip\noindent
\textbf{(I) The WKB forbidden-zone cover (Section~\ref{sec:bridge}).}
For $m, n \le \gamma\,\NSh$ with $\gamma < 1/2$, the Fourier
transform $\widehat{f}_{mn}(\omega+\delta)$ at out-of-band
frequencies $\xi = \omega + \delta$ can be expressed as a
convolution integral over $x \in [\delta T/\omega, T]$
(via the PSWF self-duality relation).
For every $x$ in this interval, the two WKB forbidden zones
\[
  \{x > \gamma T\} \quad\text{(for } \psi_n\text{)}
  \qquad\text{and}\qquad
  \{T + \delta T/\omega - x > \gamma T\} \quad\text{(for } \psi_m\text{)}
\]
together cover $[\delta T/\omega, T]$ entirely, because
$\gamma T + \gamma T = 2\gamma T < T$ for $\gamma < 1/2$.
At every integration point, at least one PSWF factor is
exponentially small (WKB tunneling), so the integral itself
is exponentially small.

\medskip\noindent
\textbf{(II) Kernel splitting and Schur test
(Sections~\ref{sec:offdiag}--\ref{sec:schur}).}
We decompose $K_N = K_M = \KBL + \widetilde{E}^- - \widetilde{E}^+$
around the sinc-type band-limited kernel $\KBL$.
The error terms are controlled by eigenvalue concentration
(from \cite{OsipovRokhlinXiao}) and positive semi-definiteness.
The classical Schur test then gives the $L^2$ operator bound.

\medskip\noindent
\textbf{(III) Separation of mechanisms.}
The WKB cover (Idea~I) is a purely energetic argument;
it gives no off-diagonal decay in $|m-n|$.
The Schur argument (Idea~II) is a functional-analytic argument;
it requires only that the kernel is integrable, not that the
defect is Schur-summable with an index-dependent profile.
The DSTP off-diagonal structure of \cite{PaperI} is a
strictly additional statement requiring phase-gap cancellation;
it is not used here.

\section{Preliminaries}
\label{sec:prelim}
% =============================================================

\subsection{PSWF notation and basic properties}

The PSWFs $\{\psin\}_{n \ge 0}$ form an orthonormal basis of
$L^2([-T,T])$ and are eigenfunctions of the concentration
operator
\[
  \mathcal{K}_c f(x)
  = \int_{-T}^{T} \frac{\sin(\omega(x-y))}{\pi(x-y)}\,f(y)\,dy
  \colon L^2(\mathbb{R}) \to L^2(\mathbb{R}),
\]
with eigenvalues $1 > \lambda_0(c) > \lambda_1(c) > \cdots > 0$
\cite{Slepian64,OsipovRokhlinXiao}.
The spectral parameter $\chi_n(c)$ is the eigenvalue of the
associated Sturm--Liouville operator
\[
  \mathcal{L}_c \psi
  := -\frac{d}{dx}\!\left[(1-x^2/T^2)\frac{d\psi}{dx}\right]
     + \frac{c^2 x^2}{T^2}\,\psi
  = \chi_n(c)\,\psi.
\]
The Slepian number is $\NSh := 2c/\pi$.
For $n \le \gamma\,\NSh$, the classical turning point satisfies
$x_+(n) \le \gamma\,T$ \cite[\S4.4]{OsipovRokhlinXiao}.

\subsection{Eigenvalue concentration}

\begin{lemma}[Eigenvalue concentration]
\label{lem:eigenvalue-decay}
There exist constants $c_1, c_2 > 0$, depending only on $\gamma$,
such that for all $c \ge 1$:
\begin{enumerate}[label=(\roman*)]
  \item For all $n \le \gamma\,\NSh$:
  $\displaystyle 1 - \lambda_n(c) \le e^{-c_1 c}$.
  \item For all $n \ge \NSh$:
  $\displaystyle \lambda_n(c) \le e^{-c_2 c}$.
\end{enumerate}
\end{lemma}

\begin{proof}
(i) \cite[Theorem~4.6]{OsipovRokhlinXiao}:
$1-\lambda_n \le e^{-c_1(c-n\pi/2)}$;
for $n \le \gamma\,\NSh$: $c - n\pi/2 \ge c(1-\gamma) > 0$.
(ii) \cite[Theorem~4.9]{OsipovRokhlinXiao}:
$\lambda_n \le e^{-c_2(n\pi/2-c)}$;
for $n \ge \NSh$: $n\pi/2 - c \ge (\pi/2-1)c > 0$.
\end{proof}

\subsection{Uniform $L^\infty$ bound for PSWFs}

\begin{lemma}[$L^\infty$ bound]
\label{lem:Linfty-global}
For all $n \ge 0$, $c \ge 1$, and $x \in \mathbb{R}$:
\[
  |\psin(x)| \le C_0,
\]
with $C_0 > 0$ independent of $n$, $c$, and $x$.
\end{lemma}

\begin{proof}
\cite[Corollary~4.3]{OsipovRokhlinXiao}.
\end{proof}

\subsection{The Schur test}

\begin{lemma}[Schur test]
\label{lem:schur-test}
Let $A$ be an integral operator on $L^2(\Omega, \mu)$ with
kernel $A(x,y)$. If
\[
  \sup_{x} \int |A(x,y)|\,d\mu(y) \le M
  \quad\text{and}\quad
  \sup_{y} \int |A(x,y)|\,d\mu(x) \le M,
\]
then $\|A\|_{L^2 \to L^2} \le M$.
\end{lemma}

\begin{proof}
\cite[Theorem~VI.23]{ReedSimonI}.
\end{proof}

\section{The Diagonal Bound}
\label{sec:diagonal}
% =============================================================

\begin{proposition}[Diagonal bound]
\label{prop:diagonal}
Fix $0 < \delta < 1$ and $0 < \gamma < 1/2$.
For all $x \in I_\delta$ and all $N \le \gamma\,\NSh$:
\begin{equation}
\label{eq:diagonal}
  K_N(x,x) = \sum_{n=0}^{N-1}\psin(x)^2
  \;\le\; C_{\delta,\gamma}\,\frac{Nc}{\pi T},
\end{equation}
uniformly in $c \ge 1$.
\end{proposition}

\begin{proof}
\textbf{Step 1 (Pr\"ufer amplitude control).}
From \cite[Lemma~4.2]{PaperIV}:
\begin{equation}\label{eq:amplitude-drift}
  \frac{r_n(x)^2}{\rWKB(x)^2} = 1 + O(1/n),
\end{equation}
uniformly for $x \in I_\delta$, $n \le \gamma\,\NSh$.
From \cite[Lemma~4.3]{PaperIV}:
\begin{equation}\label{eq:normalization}
  \tfrac{1}{2}\rWKB(x)^2 = \lambda_n(c)\,\rhocl(x)(1+O(1/n)).
\end{equation}
Combining:
\begin{equation}\label{eq:psi-sq}
  \psin(x)^2 \le C_{\delta,\gamma}\,\lambda_n(c)\,\rhocl(x).
\end{equation}

\textbf{Step 2 (Density bound).}
The classical density satisfies
\begin{equation}\label{eq:density-bound}
  \rhocl(x) \le C_{\delta,\gamma}\,c/T
  \qquad\text{for all } x \in I_\delta,\ n \le \gamma\,\NSh.
\end{equation}
This follows from the explicit formula
$\rhocl(x) = (1/\wcl(x)) / \int_{-x_+(n)}^{x_+(n)} (1/\wcl(t))\,dt$,
combined with the lower bound $\wcl(x) \ge c_{\delta,\gamma}\,n$
\cite[Lemma~2.5(i)]{PaperIV} and the lower bound on the
denominator $\int_{I_\delta} (1/\wcl(t))\,dt \ge c'_{\delta,\gamma} T/c$
(from $\wcl \le C_{\delta,\gamma}\,c$ on $I_\delta$).

\textbf{Step 3 (Summation).}
Inserting \eqref{eq:density-bound} into \eqref{eq:psi-sq}
and summing over $n < N$:
\[
  K_N(x,x) \le C_{\delta,\gamma}\,\frac{c}{T}\sum_{n=0}^{N-1}\lambda_n
  \le C_{\delta,\gamma}\,\frac{Nc}{\pi T}. \qedhere
\]
\end{proof}

\begin{remark}[Landau--Widom]
\label{rem:LW}
The bound \eqref{eq:diagonal} recovers the classical Landau--Widom
diagonal estimate \cite{Widom64}, with constants uniform in $n$ and $c$.
\end{remark}

\section{Off-diagonal Decay via Kernel Splitting}
\label{sec:offdiag}
% =============================================================

\subsection{The band-limited kernel and its sinc bound}

The concentration operator $\mathcal{K}_c$ has a reproducing
kernel representation:
\begin{equation}\label{eq:KBL-sinc}
  \KBL(x,y) := \sum_{n \ge 0}\lambda_n\,\psin(x)\psin(y)
  = \frac{c}{\pi T}\,\frac{\sin(c(x-y)/T)}{c(x-y)/T}.
\end{equation}

\begin{lemma}[Band-limited kernel bound]
\label{lem:BL-bound}
For all $x,y \in \mathbb{R}$:
\begin{equation}\label{eq:BL-bound}
  |\KBL(x,y)| \le \frac{C}{1+c|x-y|/T}.
\end{equation}
\end{lemma}

\begin{proof}
At $x=y$: $\KBL(x,x) = c/(\pi T)$. For $x\ne y$, set $u=c|x-y|/T>0$:
$|\sin u/u| \le 2/(1+u)$ (check $u \le 1$ and $u > 1$ separately).
\end{proof}

\subsection{Algebraic kernel decomposition}
\label{subsec:decomp}

Let $M := \lfloor\gamma\,\NSh\rfloor$. Since $N \le M$,
it suffices to bound the kernel $K_M$. Write:
\begin{equation}\label{eq:split}
  K_M = \KBL + \widetilde{E}^- - \widetilde{E}^+,
\end{equation}
where
\[
  \widetilde{E}^-\!:= \sum_{n<M}(1-\lambda_n)\psin\psin,
  \quad
  \widetilde{E}^+\!:= \sum_{n \ge M}\lambda_n\psin\psin
  = \widetilde{E}^+_{\mathrm{near}} + \widetilde{E}^+_{\mathrm{tail}},
\]
with
$\widetilde{E}^+_{\mathrm{near}} := \sum_{n=M}^{\NSh-1}\lambda_n\psin\psin$
and
$\widetilde{E}^+_{\mathrm{tail}} := \sum_{n \ge \NSh}\lambda_n\psin\psin$.

\begin{remark}[Transition region]
\label{rem:transition}
For $M \le n < \NSh$, the eigenvalues $\lambda_n$ pass through
the Airy-scale transition and are not individually small.
Therefore $\widetilde{E}^+_{\mathrm{near}}$ is controlled by a
PSD-dominance argument (Subsection~\ref{subsec:diff-bound})
rather than by an eigenvalue bound.
\end{remark}

\subsection{Bounding the error terms}
\label{subsec:diff-bound}

\begin{proposition}[Error term bounds]
\label{prop:error-bounds}
For all $x, y \in I_\delta$ and $N \le \gamma\,\NSh$,
with $c_3 := \min(c_1,c_2) > 0$ from
Lemma~\ref{lem:eigenvalue-decay}:
\begin{align}
  |\widetilde{E}^-(x,y)| &\le C_{\delta,\gamma}\,\frac{c^2}{T}\,e^{-c_1 c},
  \label{eq:Eminus-final}\\
  |\widetilde{E}^+_{\mathrm{tail}}(x,y)| &\le C_{\delta,\gamma}\,\frac{c}{T}\,e^{-c_2 c},
  \label{eq:Etail-final}\\
  |\widetilde{E}^+_{\mathrm{near}}(x,y)| &\le \frac{c}{\pi T}.
  \label{eq:Enear-final}
\end{align}
\end{proposition}

\begin{proof}
\textbf{Bound~\eqref{eq:Eminus-final}.}
By Lemma~\ref{lem:eigenvalue-decay}(i) and \eqref{eq:psi-sq}:
$|\psin(x)\psin(y)| \le C_{\delta,\gamma}\,c/T$ (using
\eqref{eq:density-bound}).
Hence $|\widetilde{E}^-(x,y)| \le (c/T)\sum_{n<M}(1-\lambda_n)
\le C_{\delta,\gamma}(c/T)\cdot c\,e^{-c_1 c}$.

\textbf{Bound~\eqref{eq:Etail-final}.}
By Lemma~\ref{lem:eigenvalue-decay}(ii):
$|\widetilde{E}^+_{\mathrm{tail}}(x,y)| \le (c/T)\sum_{n \ge \NSh}\lambda_n
\le C_{\delta,\gamma}(c/T)\,e^{-c_2 c}$
(geometric summation).

\textbf{Bound~\eqref{eq:Enear-final}.}
All four kernels $\widetilde{E}^+_{\mathrm{near}}$,
$\widetilde{E}^+_{\mathrm{tail}}$, $\widetilde{E}^+$, and $\KBL$
are positive semi-definite (PSD), and
\[
  0 \preceq \widetilde{E}^+_{\mathrm{near}}
  \preceq \widetilde{E}^+ \preceq \KBL.
\]
The PSD Cauchy--Schwarz inequality $|A(x,y)|^2 \le A(x,x)A(y,y)$
applied to $A = \widetilde{E}^+_{\mathrm{near}}$ gives:
\[
  |\widetilde{E}^+_{\mathrm{near}}(x,y)|^2
  \le \KBL(x,x)\,\KBL(y,y) = (c/(\pi T))^2.
  \qedhere
\]
\end{proof}

\begin{proposition}[Off-diagonal bound]
\label{prop:offdiag}
For all $x,y \in I_\delta$ and $N \le \gamma\,\NSh$:
\begin{equation}\label{eq:offdiag}
  |K_N(x,y)|
  \le \frac{C_{\delta,\gamma}}{1+c|x-y|/T}
  + C_{\delta,\gamma}\,\frac{c}{\pi T}
  + C_{\delta,\gamma}\,\frac{c^2}{T}\,e^{-c_3 c}.
\end{equation}
\end{proposition}

\begin{proof}
Combine Lemma~\ref{lem:BL-bound} with the decomposition
\eqref{eq:split} and Proposition~\ref{prop:error-bounds},
using the triangle inequality.
\end{proof}
