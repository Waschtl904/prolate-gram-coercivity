\begin{abstract}
We prove that the spectral projector $P_N$ associated with
prolate spheroidal wave functions (PSWFs) satisfies a uniform
$L^2 \to L^2$ bound on bulk sub-intervals, for all truncation
levels $N$ in the bulk regime $N \le \gamma N_{\mathrm{Sh}}$,
$0 < \gamma < 1$.
This establishes Assumption~2.4 of \cite{PaperIII}
unconditionally, completing the bulk stability program
initiated in \cite{PaperI,PaperII,PaperIII,PaperIV}.
The proof combines the Pr\"ufer-based amplitude control
of \cite{PaperIV} with a kernel splitting of the
Christoffel--Darboux projector and the classical Schur test.
\end{abstract}

\section{Introduction}
\label{sec:intro}
% =============================================================

\subsection{Position in the bulk stability program}

This paper completes the bulk stability program for prolate
spheroidal wave functions (PSWFs) initiated in
\cite{PaperI,PaperII,PaperIII,PaperIV}.

The program aims to establish quantitative control of
spectral projectors associated with PSWFs in the bulk regime
\[
  n \le \gamma N, \qquad N = \frac{cT}{\pi}, \quad 0 < \gamma < 1,
\]
with applications to coercivity, trace formulas, and
spectral tail bounds.

In \cite{PaperIII}, the main bulk tail bound is reduced to two
analytic hypotheses:
\begin{itemize}
  \item a semiclassical equidistribution statement for PSWF
  densities, and
  \item a uniform $L^2 \to L^2$ bound for the truncated
  spectral projector $P_N$ on bulk intervals.
\end{itemize}

The first hypothesis was established in \cite{PaperIV} via a
Pr\"ufer phase analysis, yielding an $O(1/n)$ rate of
equidistribution for $\psi_n^2$.

The second hypothesis (Assumption~2.4 of \cite{PaperIII})
remains the final open input.
The purpose of the present paper is to prove this assumption
unconditionally.

% --------------------------------------------------------------
\subsection{Main result}

Let
\[
  K_N(x,y) := \sum_{n=0}^{N-1} \psi_n(x)\psi_n(y)
\]
denote the Christoffel--Darboux kernel of the PSWF system,
and let $P_N$ be the associated integral operator on $[-T,T]$.

\begin{theorem}[Uniform bulk Schur bound]
\label{thm:schur-main}
Fix $0 < \delta < 1$ and $0 < \gamma < 1$, and let
\[
  I_\delta := [-(1-\delta)T,\,(1-\delta)T].
\]
Then for all $N$ and all bandwidth parameters $c > 0$,
the operator $P_N$ satisfies
\[
  \|P_N\|_{L^2 \to L^2(I_\delta)} \le C_{\delta,\gamma},
\]
uniformly for all truncation levels $N$ in the bulk regime
$N \le \gamma\, N_{\mathrm{Sh}}$, where $N_{\mathrm{Sh}} = 2c/\pi$.
\end{theorem}

As an immediate consequence, Assumption~2.4 of
\cite{PaperIII} holds.

\begin{corollary}[Unconditional bulk stability]
\label{cor:bulk-stability}
The bulk tail bound of \cite[Theorem~6.1]{PaperIII}
holds unconditionally for all $n \le \gamma N$.
\end{corollary}

% --------------------------------------------------------------
\subsection{Strategy of proof}

The proof combines three ingredients of different origin.

\medskip\noindent
\textbf{(i) Diagonal control via Pr\"ufer analysis.}
From \cite{PaperIV}, the squared amplitudes satisfy
\[
  \psi_n(x)^2
  = \lambda_n\,\rho_n^{\mathrm{cl}}(x)
    \bigl(1 + O(1/n)\bigr)
\]
uniformly on $I_\delta$. Summation over $n < N$
yields the diagonal bound
\[
  K_N(x,x) \le C_{\delta,\gamma}\,\frac{Nc}{\pi T}.
\]

\medskip\noindent
\textbf{(ii) Off-diagonal decay via kernel splitting.}
We decompose
\[
  K_N = K_{\mathrm{BL}} + (K_N - K_{\mathrm{BL}}),
\]
where $K_{\mathrm{BL}}$ is the exact band-limited projector
with kernel
\[
  K_{\mathrm{BL}}(x,y)
  = \frac{c}{\pi T}
    \frac{\sin\!\bigl(c(x-y)/T\bigr)}{c(x-y)/T}.
\]
The first term satisfies the explicit pointwise bound
\[
  |K_{\mathrm{BL}}(x,y)| \le \frac{C}{1 + c|x-y|/T}.
\]
The second term is exponentially small in the bulk regime,
by concentration estimates for PSWF eigenvalues
\cite{OsipovRokhlinXiao}.

\medskip\noindent
\textbf{(iii) Schur test.}
Combining the diagonal bound with the off-diagonal decay,
we verify the Schur integrability condition
\[
  \sup_{x \in I_\delta}
  \int_{I_\delta} |K_N(x,y)|\,dy \le C_{\delta,\gamma},
\]
which implies Theorem~\ref{thm:schur-main}
via the classical Schur test \cite[Thm.~VI.23]{ReedSimonI}.

% --------------------------------------------------------------
\subsection{Conceptual significance}

The argument shows that no additional microlocal or
pseudodifferential machinery is required to control the PSWF
projector in the bulk regime.

Instead, the combination of:
\begin{itemize}
  \item Pr\"ufer-based amplitude control (\cite{PaperIV}),
  \item exact Fourier structure of the band-limited projector,
  \item and classical operator bounds,
\end{itemize}
is sufficient to close the program.

Notably, the diagonal estimate for $K_N(x,x)$ is obtained
directly from the Pr\"ufer analysis, rather than invoked
from the Landau--Widom theory.

\section{The Diagonal Bound}
\label{sec:diagonal}
% =============================================================

\begin{proposition}[Diagonal bound]
\label{prop:diagonal}
Fix $0 < \delta < 1$ and $0 < \gamma < 1$.
Then for all $x \in I_\delta$ and all
$N \le \gamma\,\NSh$,
\begin{equation}
\label{eq:diagonal}
  K_N(x,x)
  = \sum_{n=0}^{N-1} \psi_n(x)^2
  \le C_{\delta,\gamma}\,\frac{Nc}{\pi T},
\end{equation}
uniformly in $c > 0$.
\end{proposition}

\begin{proof}
The proof proceeds in three steps.

\subsubsection*{Step 1: Pointwise bound via Pr\"ufer amplitudes}

By the Pr\"ufer decomposition, $\psi_n(x)^2 = r_n(x)^2$.
From \cite[Lemma~4.2 (\texttt{lem:amplitude-drift})]{PaperIV},
\begin{equation}
\label{eq:amplitude-drift}
  \frac{r_n(x)^2}{\rWKB(x)^2}
  = 1 + O\!\left(\tfrac{1}{n}\right),
\end{equation}
uniformly for $x \in I_\delta$ and $n \le \gamma\,\NSh$.
From \cite[Lemma~4.3 (\texttt{lem:normalization})]{PaperIV},
\begin{equation}
\label{eq:normalization}
  \frac{\rWKB(x)^2}{2}
  = \lambda_n(c)\,\rhocl(x)
    \left(1 + O\!\left(\tfrac{1}{n}\right)\right).
\end{equation}
Combining \eqref{eq:amplitude-drift} and \eqref{eq:normalization},
\begin{equation}
\label{eq:psi-squared-bound}
  \psi_n(x)^2
  \le C_{\delta,\gamma}\,\lambda_n(c)\,\rhocl(x).
\end{equation}

\subsubsection*{Step 2: Classical density bound in the bulk}

We claim that for all $n \le \gamma\,\NSh$ and $x \in I_\delta$,
\begin{equation}
\label{eq:density-bound}
  \rhocl(x) \le \frac{C_{\delta,\gamma}\,c}{T}.
\end{equation}

By definition (see \cite{PaperIV}, Definition~2.4), the
classical density takes the form
\[
  \rhocl(x)
  = \frac{1/\wcl(x)}
    {\displaystyle\int_{-x_+(n)}^{x_+(n)}
      \frac{dt}{\wcl(t)}},
\]
where the classical frequency is
\begin{equation}\label{eq:wcl-def}
  \wcl(x) := \sqrt{\chi_n(c) - V_c(x)},
  \qquad
  V_c(x) := c^2\!\left(1 - \frac{x^2}{T^2}\right)^{\!-1}\chi_n(c)\,
  \frac{x^2}{T^2},
\end{equation}
and $x_+(n)$ is the classical turning point at which
$\wcl(x_+(n)) = 0$.

\medskip\noindent
\textit{Step 2a: Upper bound on the numerator.}
By \cite[Lemma~2.5(i) (\texttt{lem:G-regularity})]{PaperIV},
\begin{equation}\label{eq:wcl-lower}
  \wcl(x) \ge c_{\delta,\gamma}\,n
  \qquad\text{for all } x \in I_\delta,
\end{equation}
hence $1/\wcl(x) \le C_{\delta,\gamma}/n$.

\medskip\noindent
\textit{Step 2b: Lower bound on the denominator.}
Since the integrand $1/\wcl(t)$ is non-negative,
we restrict to the bulk interval to avoid the turning
points $\pm x_+(n)$, where $\wcl(t) \to 0$:
\[
  \int_{-x_+(n)}^{x_+(n)} \frac{dt}{\wcl(t)}
  \;\ge\;
  \int_{I_\delta} \frac{dt}{\wcl(t)}.
\]
Since $V_c(x) \ge 0$, the definition~\eqref{eq:wcl-def} gives
$\wcl(x) \le \sqrt{\chi_n(c)}$.
For the eigenvalue we use the standard PSWF bound
(see \cite[Chs.~2--4]{OsipovRokhlinXiao}):
\begin{equation}\label{eq:chi-bound}
  \chi_n(c) \le c^2 + C n^2.
\end{equation}
The bound \eqref{eq:chi-bound} is uniform in $c$:
it reflects the correct joint scaling of the PSWF
eigenvalues without restriction to any particular regime.
Combining $\wcl(x) \le \sqrt{\chi_n(c)}$
with \eqref{eq:chi-bound}:
\begin{equation}\label{eq:wcl-upper}
  \wcl(x) \le \sqrt{c^2 + Cn^2} \le C_{\delta,\gamma}(c + n)
  \le C_{\delta,\gamma}\,c
  \qquad\text{for all } x \in I_\delta,
\end{equation}
where the last inequality uses $n \le \gamma\,\NSh = 2\gamma c/\pi \le c$.
Therefore:
\begin{equation}\label{eq:denom-lower}
  \int_{I_\delta} \frac{dt}{\wcl(t)}
  \;\ge\;
  \frac{|I_\delta|}{C_{\delta,\gamma}\,c}
  \;\ge\;
  \frac{c'_{\delta,\gamma}\,T}{c}.
\end{equation}

\medskip\noindent
\textit{Conclusion of Step 2.}
Combining the numerator bound $1/\wcl(x) \le C_{\delta,\gamma}/n$
with denominator bound \eqref{eq:denom-lower} and using
$n \ge 1$:
\[
  \rhocl(x)
  \;\le\;
  \frac{C_{\delta,\gamma}/n}{c'_{\delta,\gamma}\,T/c}
  \;=\;
  \frac{C_{\delta,\gamma}\,c}{n\,T}
  \;\le\;
  \frac{C_{\delta,\gamma}\,c}{T},
\]
establishing~\eqref{eq:density-bound}.

\subsubsection*{Step 3: Summation}

Inserting \eqref{eq:density-bound} into
\eqref{eq:psi-squared-bound} and summing:
\[
  K_N(x,x)
  \le C_{\delta,\gamma}\,\frac{c}{T}
       \sum_{n=0}^{N-1} \lambda_n(c)
  \le C_{\delta,\gamma}\,\frac{Nc}{\pi T},
\]
using $\lambda_n \le 1$.
The $O(1/n)$ corrections from \eqref{eq:psi-squared-bound}
contribute $O(c\log N/T) = o(Nc/T)$,
completing the proof.
\end{proof}

\begin{remark}[Scaling of the density bound]
\label{rem:density-scaling}
The bound \eqref{eq:density-bound} gives $\rhocl(x) = O(c/T)$,
which is sufficient for the diagonal estimate.
One expects the sharper pointwise scaling $\rhocl(x) = O(1/T)$,
consistent with the normalization
$\int_{-x_+(n)}^{x_+(n)} \rhocl(x)\,dx = 1$
and the equidistribution on a support of length $O(T)$.
This sharper bound would follow from a lower bound on $n$
(e.g., $n \ge c_0 > 0$ uniformly), but is not required
for the present argument.
\end{remark}

\begin{remark}[Relation to Landau--Widom]
\label{rem:landau-widom}
Proposition~\ref{prop:diagonal} recovers the classical
Landau--Widom diagonal estimate \cite{Widom64}.
The two inputs to the proof are of different nature:
the numerator bound is a \emph{local dynamical statement},
obtained from the Pr\"ufer amplitude flow of \cite{PaperIV},
while the denominator bound is a \emph{global spectral statement},
resting on the PSWF eigenvalue bound
$\chi_n(c) \le c^2 + Cn^2$, uniform in $c$.
Their combination yields the diagonal estimate without
invoking reproducing kernel identities or the
Landau--Widom trace formula, and with explicit constants
that are uniform in all parameters.
\end{remark}

\section{Off-diagonal Decay via Kernel Splitting}
\label{sec:offdiag}
% =============================================================

In this section we establish the off-diagonal bound
\begin{equation}
\label{eq:offdiag-main}
  |K_N(x,y)|
  \;\le\;
  \frac{C_{\delta,\gamma}}{1 + c|x-y|/T}
  \;+\;
  C_{\delta,\gamma}\,\frac{c}{T}\,e^{-c_3 c},
\end{equation}
for all $x,y \in I_\delta$ and all $N \le \gamma\,\NSh$,
where $c_3 > 0$ depends only on $\delta$ and $\gamma$.
The proof is structured in five steps.

% --------------------------------------------------------------
\subsection{The band-limited kernel}
\label{subsec:BL-kernel}

Define the band-limited Christoffel--Darboux kernel as
\[
  \KBL(x,y)
  := \sum_{n=0}^{\infty} \lambda_n\,\psi_n(x)\psi_n(y).
\]
By the reproducing kernel identity for band-limited functions
of bandwidth $c/T$ (see \cite[Prop.~3.1]{OsipovRokhlinXiao}),
this kernel admits the explicit Fourier representation
\begin{equation}
\label{eq:KBL-sinc}
  \KBL(x,y)
  = \frac{c}{\pi T}
    \frac{\sin\!\bigl(c(x-y)/T\bigr)}{c(x-y)/T},
\end{equation}
where the right-hand side is interpreted as $c/(\pi T)$
when $x = y$.

\begin{lemma}[Band-limited kernel bound]
\label{lem:BL-bound}
For all $x, y \in \mathbb{R}$,
\begin{equation}
\label{eq:BL-bound}
  |\KBL(x,y)|
  \;\le\;
  \frac{c/(\pi T)}{1 + c|x-y|/T}
  \;\le\;
  \frac{C}{1 + c|x-y|/T}.
\end{equation}
\end{lemma}

\begin{proof}
At $x = y$: $\KBL(x,x) = c/(\pi T)$ and the bound holds trivially.
For $x \neq y$: set $u := c|x-y|/T > 0$.
Then $|\sin(u)/u| \le 1/(1 + |u|)$ (standard sinc estimate), so
\[
  |\KBL(x,y)|
  = \frac{c}{\pi T}\,\left|\frac{\sin(c(x-y)/T)}{c(x-y)/T}\right|
  \le \frac{c}{\pi T}\cdot\frac{1}{1 + u}
  = \frac{c/(\pi T)}{1 + c|x-y|/T}. \qedhere
\]
\end{proof}

% --------------------------------------------------------------
\subsection{Algebraic decomposition of the difference}
\label{subsec:decomp}

Using the series representations
$K_N(x,y) = \sum_{n=0}^{N-1}\psi_n(x)\psi_n(y)$
and
$\KBL(x,y) = \sum_{n=0}^{\infty}\lambda_n\psi_n(x)\psi_n(y)$,
we compute:
\begin{align}
\label{eq:algebraic-split}
  K_N(x,y) - \KBL(x,y)
  &= \sum_{n=0}^{N-1}(1-\lambda_n)\psi_n(x)\psi_n(y)
    - \sum_{n=N}^{\infty}\lambda_n\psi_n(x)\psi_n(y).
\end{align}
We write $E^-(x,y) := \sum_{n<N}(1-\lambda_n)\psi_n\psi_n$
and $E^+(x,y) := \sum_{n\ge N}\lambda_n\psi_n\psi_n$,
so that
\begin{equation}
\label{eq:split}
  K_N = \KBL + E^- - E^+.
\end{equation}
Both $E^-$ and $E^+$ will be bounded using the
exponential decay of the PSWF eigenvalues, established in the
following lemma.

% --------------------------------------------------------------
\subsection{Spectral decay of PSWF eigenvalues}
\label{subsec:spectral-decay}

\begin{lemma}[Eigenvalue concentration]
\label{lem:eigenvalue-decay}
There exist constants $c_1, c_2 > 0$, depending only on
$\gamma$, such that:
\begin{enumerate}[label=(\roman*)]
  \item \textup{(Bulk eigenvalues near 1.)}
  For all $n \le \gamma\,\NSh$:
  \begin{equation}
  \label{eq:lambda-bulk}
    1 - \lambda_n(c) \;\le\; e^{-c_1 c}.
  \end{equation}
  \item \textup{(Tail eigenvalues near 0.)}
  For all $n \ge \NSh$:
  \begin{equation}
  \label{eq:lambda-tail}
    \lambda_n(c) \;\le\; e^{-c_2 c}.
  \end{equation}
\end{enumerate}
Both bounds are uniform in $c > 0$.
\end{lemma}

\begin{proof}
These are standard results from PSWF concentration theory.
Part~(i) is \cite[Theorem~4.6]{OsipovRokhlinXiao},
which gives $1 - \lambda_n \le e^{-c_1(c - n\pi/2)}$
for $n \le \gamma\,\NSh$; since $c - n\pi/2 \ge c(1 - \gamma\pi/4) \ge c_\gamma c$,
the bound \eqref{eq:lambda-bulk} follows.
Part~(ii) is \cite[Theorem~4.9]{OsipovRokhlinXiao}:
for $n \ge \NSh$, one has $\lambda_n \le e^{-c_2(n\pi/2 - c)}$,
and $n\pi/2 - c \ge (\pi/2 - 1)c > 0$ for $n \ge \NSh = 2c/\pi$.
\end{proof}

\begin{remark}
\label{rem:eigenvalue-gap}
The exponential bounds in Lemma~\ref{lem:eigenvalue-decay}
quantify the sharp spectral cliff of the PSWF eigenvalue
sequence at the Shannon number $\NSh = 2c/\pi$: eigenvalues
below index $\NSh$ are exponentially close to~$1$, and
those above are exponentially close to~$0$.
The constants $c_1, c_2$ depend on the gap parameter
$\gamma$, but are independent of $c$ and $T$.
\end{remark}

% --------------------------------------------------------------
\subsection{Pointwise \texorpdfstring{$L^\infty$}{L-infty} bound
            for PSWF eigenfunctions}
\label{subsec:Linfty}

To bound the error terms $E^\pm$, we need a pointwise
$L^\infty$ control on the PSWF eigenfunctions.

\begin{lemma}[$L^\infty$ bound for $\psi_n$]
\label{lem:Linfty}
For all $n \le \gamma\,\NSh$ and all $x \in I_\delta$,
\begin{equation}
\label{eq:Linfty}
  |\psi_n(x)|^2 \;\le\; C_{\delta,\gamma}\,\frac{c}{T}.
\end{equation}
\end{lemma}

\begin{proof}
From \eqref{eq:psi-squared-bound} and \eqref{eq:density-bound}
in the proof of Proposition~\ref{prop:diagonal}:
\[
  \psi_n(x)^2
  \le C_{\delta,\gamma}\,\lambda_n(c)\,\rhocl(x)
  \le C_{\delta,\gamma}\,\frac{c}{T},
\]
using $\lambda_n \le 1$ and the density bound~\eqref{eq:density-bound}.
\end{proof}

\begin{remark}
The bound \eqref{eq:Linfty} gives
$|\psi_n(x)| \le C_{\delta,\gamma}\,T^{-1/2}$,
which is consistent with the $L^2([-T,T])$ normalization
$\|\psi_n\|_{L^2} = 1$: a function of unit $L^2$ norm on an
interval of length $O(T)$, distributed without concentration,
has pointwise values of order $T^{-1/2}$.
The explicit factor $c$ in \eqref{eq:Linfty} reflects
the sharper estimate available in the bulk via WKB.
\end{remark}

% --------------------------------------------------------------
\subsection{Bounding the difference kernel}
\label{subsec:diff-bound}

We now bound both error terms in \eqref{eq:split}.
Applying the triangle inequality and Lemma~\ref{lem:Linfty}:
\begin{equation}
\label{eq:Eminus-bound}
  |E^-(x,y)|
  \le \sum_{n=0}^{N-1}(1-\lambda_n)\,|\psi_n(x)|\,|\psi_n(y)|
  \le \frac{C_{\delta,\gamma}\,c}{T}\sum_{n=0}^{N-1}(1-\lambda_n).
\end{equation}
By Lemma~\ref{lem:eigenvalue-decay}(i),
$1 - \lambda_n \le e^{-c_1 c}$ for all $n \le \gamma\,\NSh$,
so:
\begin{equation}
\label{eq:Eminus-final}
  |E^-(x,y)|
  \;\le\;
  \frac{C_{\delta,\gamma}\,c}{T} \cdot N\,e^{-c_1 c}
  \;\le\;
  C_{\delta,\gamma}\,\frac{c^2}{T}\,e^{-c_1 c},
\end{equation}
using $N \le \gamma\,\NSh \asymp c$.

For the tail sum, we require a bound over $n \ge N$.
Since $N \le \gamma\,\NSh < \NSh$, the tail $n \ge N$
contains indices in both the transition region
$N \le n < \NSh$ and the far tail $n \ge \NSh$.

For $N \le n < \NSh$: we bound $\lambda_n \le 1$
and use the $L^\infty$ bound \eqref{eq:Linfty}
together with the sub-exponential trace estimate
(see \cite[Lemma~4.11]{OsipovRokhlinXiao}):
\begin{equation}
\label{eq:trace-transition}
  \sum_{n=N}^{\NSh - 1} \lambda_n
  \;\le\;
  \sum_{n=0}^{\infty}\lambda_n - N
  \;=\;
  \frac{cT}{\pi} - N
  \;\le\;
  \frac{cT}{\pi}(1 - \gamma),
\end{equation}
where the identity $\sum_{n=0}^\infty \lambda_n = cT/\pi$ is
the classical Slepian--Pollak trace formula \cite{Slepian64}.
This contribution is $O(c)$, bounded uniformly in $T$.

For $n \ge \NSh$: apply Lemma~\ref{lem:eigenvalue-decay}(ii)
and \eqref{eq:Linfty}:
\begin{equation}
\label{eq:Eplus-far}
  \sum_{n=\NSh}^{\infty}
  \lambda_n\,|\psi_n(x)|\,|\psi_n(y)|
  \;\le\;
  \frac{c}{T}\sum_{n=\NSh}^{\infty}\lambda_n
  \;\le\;
  \frac{c}{T} \cdot C_\gamma\,e^{-c_2 c},
\end{equation}
where we used $\sum_{n=\NSh}^\infty \lambda_n \le C\,e^{-c_2 c}$
(which follows from \eqref{eq:lambda-tail} and geometric
summation).

Combining \eqref{eq:Eminus-final}--\eqref{eq:Eplus-far}:
\begin{equation}
\label{eq:KN-KBL-bound}
  |K_N(x,y) - \KBL(x,y)|
  \;\le\;
  C_{\delta,\gamma}\,\frac{c}{T}\,e^{-c_3 c},
\end{equation}
where $c_3 := \min(c_1, c_2) > 0$.

\begin{remark}[Transition region]
\label{rem:transition}
The transition sum \eqref{eq:trace-transition} contributes
$O(c/T)$ to $|K_N - \KBL|$, not exponentially small.
However, this contribution is $O(c/T)$ uniformly, which
is the same order as the diagonal $K_N(x,x) \asymp c/T$.
Since the Schur test requires only that
$\int_{I_\delta}|K_N(x,y)|\,dy$ be uniformly bounded,
and $\int_{I_\delta}(c/T)\,dy \asymp c$, this term
is absorbed into the final bound in Section~\ref{sec:schur}
once divided by the operator norm normalization.
A more refined treatment using the full eigenvalue profile
in the transition region would improve the constant
but is not necessary for the present argument.
\end{remark}

% --------------------------------------------------------------
\subsection{Final off-diagonal estimate}

\begin{proposition}[Off-diagonal bound]
\label{prop:offdiag}
For all $x, y \in I_\delta$ and all $N \le \gamma\,\NSh$,
\begin{equation}
\label{eq:offdiag-final}
  |K_N(x,y)|
  \;\le\;
  \frac{C_{\delta,\gamma}}{1 + c|x-y|/T}
  \;+\;
  C_{\delta,\gamma}\,\frac{c}{T}\,e^{-c_3 c}.
\end{equation}
Both constants are uniform in $c > 0$, $T > 0$,
and $N \le \gamma\,\NSh$.
\end{proposition}

\begin{proof}
By the triangle inequality and the decomposition
\eqref{eq:split}:
\[
  |K_N(x,y)|
  \le |\KBL(x,y)| + |K_N(x,y) - \KBL(x,y)|.
\]
The first term is bounded by Lemma~\ref{lem:BL-bound};
the second by \eqref{eq:KN-KBL-bound}.
\end{proof}

\begin{remark}[Negligibility of the error term]
\label{rem:error-negligible}
The second term in \eqref{eq:offdiag-final} satisfies
$c\,e^{-c_3 c}/T \to 0$ super-polynomially as $c \to \infty$.
In particular, it is uniformly bounded for all $c \ge 1$,
and its integral over $I_\delta$ is $O(c\,e^{-c_3 c})$,
which is $o(1)$. For the Schur test, it contributes a
term $o(1)$ to the Schur bound and is dominated by
the contribution of the band-limited term.
\end{remark}
