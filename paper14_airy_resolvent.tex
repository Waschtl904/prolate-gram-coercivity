\title[Paper XIV: Airy Resolvent for Sinc Kernel]{%
  Paper~XIV: Airy Resolvent Asymptotics for the Sinc Kernel Operator\\[0.3em]
  \large Edge Universality and the Local Spectral Counting Condition}

\author{\textit{prolate-gram-coercivity programme}}
\date{May 2026}

\begin{abstract}
Paper~XIII identified condition~(Q5) ---
the local spectral counting statement in a $c^{-1/3}$-scale window ---
as the primary obstruction to Gap-S.
In its binary form $N_\varphi(c) = 2$, condition~(Q5) is
too fragile as a deterministic claim:
it depends sensitively on window calibration and index matching,
and has no universal character.

\medskip\noindent
\textbf{Revised target.}
This paper replaces the binary count by a structural convergence statement:
\begin{equation*}
  \text{(Q5$'$):}\quad
  \text{The local eigenvalue process of $\Kc$ converges,
  in the Airy double-scaling limit, to the Airy point process.}
\end{equation*}
From this convergence, the binary count $N_\varphi(c) = 2 + O(c^{-1/3})$
follows as a \emph{secondary consequence} for windows calibrated to
contain exactly two Airy eigenvalues.

\medskip\noindent
\textbf{Analytic programme.}
The approach uses the Helffer--Sj\"ostrand functional calculus:
\[
  N_\varphi(c) = -\frac{1}{\pi}\int_\C \partial_{\bar z}\tilde\varphi_c(z)\,
  \Tr[(\Kc-z)^{-1}]\,d^2z.
\]
The central conjecture (Conjecture~\ref{conj:resolvent}) asserts
that $\Tr[(\Kc - z)^{-1}]$ converges, in the double-scaling regime,
to $\Tr[(\cA - \zeta)^{-1}]$ plus controllable error.
This converts (Q5$'$) into a \emph{resolvent trace asymptotics problem}.

\medskip\noindent
\textbf{Three main gaps} (precisely identified):
\begin{enumerate}[nosep]
  \item Bridge Lemma (Lemma~\ref{lem:bridge}): kernel convergence
    in $L^2 \Rightarrow$ resolvent convergence in trace norm
    (the analytic core, currently a placeholder).
  \item Regularisation of $\det(\cA - \zeta)$ for the Fredholm route
    (Section~\ref{sec:fredholm}).
  \item Scale separation: why exactly three scales
    $c, c^{2/3}, c^{1/3}$ dominate (Section~\ref{sec:scales}).
\end{enumerate}
\end{abstract}

\section{Setup: From Q5 to a Structural Convergence Target}
\label{sec:setup}
%% ============================================================

\subsection{The operator and transition regime}

$(\Kc f)(x) = \int_{-1}^1 \frac{\sin(c(x-y))}{\pi(x-y)} f(y)\,dy$,
eigenvalues $1 > \lam{c}{0} \geq \lam{c}{1} \geq \cdots > 0$,
transition index $n^* \sim 2c/\pi$.

\subsection{Why \texorpdfstring{$N_\varphi(c) = 2$}{N=2} is fragile}

\begin{remark}[Fragility of the binary count]
\label{rem:fragility}
The Airy operator $\cA = -d^2/d\xi^2 + \xi$ on $L^2(\R_+)$
has eigenvalues $\{-a_k\}_{k\geq 1}$
with spacings $a_{k+1} - a_k \sim \pi/\sqrt{a_k}$ for large $k$.
For the first zeros: $a_1 \approx 2.338$, $a_2 \approx 4.088$,
$a_3 \approx 5.521$, with spacing $a_2 - a_1 \approx 1.75$.
The claim $N_\varphi(c) = 2$ requires:
\begin{itemize}[nosep]
  \item \emph{Window calibration:}
    $\zeta_0$ chosen so that $[-\zeta_0, \zeta_0]$ contains
    $\{-a_1, -a_2\}$ but not $-a_3$.
    This is possible for $1.75 < 2\zeta_0 < 3.18$, a finite interval
    of valid $\zeta_0$-values.
  \item \emph{Index matching:}
    The transition index $n^*$ corresponds to the $k$-th Airy eigenvalue
    for the correct $k$. This matching is not automatic and depends on
    sub-leading terms in the Weyl law.
\end{itemize}
\emph{Conclusion:}
The deterministic claim $N_\varphi(c) = 2$ is not a robust asymptotic
principle but a calibration-dependent consequence.
The robust statement is Conjecture~\ref{conj:pointprocess} below.
\end{remark}

\subsection{The revised target: Airy point process convergence}

\begin{definition}[Airy point process]
The \emph{Airy point process} $\mathrm{APP}$ is the determinantal
point process on $\R$ with kernel
\begin{equation}\label{eq:Airykernel}
  K_{\Ai}(s,t) =
  \frac{\Ai(s)\Ai'(t) - \Ai'(s)\Ai(t)}{s-t}
  = \int_0^\infty \Ai(s+u)\Ai(t+u)\,du.
\end{equation}
Its support is the set of Airy zeros $\{-a_k\}_{k\geq 1}$.
\end{definition}

\begin{conjecture}[Airy point process convergence for $\Kc$]
\label{conj:pointprocess}
Let $\xi_j^{(c)} := c^{1/3}(\lam{c}{n^*+j} - z_{n^*})$ be the
rescaled eigenvalues near the transition index $n^*$.
Then, as $c \to \infty$, the point process
$\sum_j \delta_{\xi_j^{(c)}}$ converges weakly to the Airy point process.
\end{conjecture}

\begin{remark}[Status of Conjecture~\ref{conj:pointprocess}]
This is the \emph{analogue of the GUE edge universality} for the
prolate concentration operator.
For the GUE, this follows from Riemann--Hilbert analysis
(Deift--Kriecherbauer--McLaughlin--Venakides--Zhou~\cite{DKMVZ1999}).
For the sinc kernel on $[-1,1]$, the corresponding Riemann--Hilbert
analysis is \emph{not known}: the kernel is not of convolution type
on the full line, and direct universality transfer fails.
Conjecture~\ref{conj:pointprocess} is the deepest open problem
of this paper.
\end{remark}

\begin{corollary}[Q5 as secondary consequence]
\label{cor:Q5}
Assume Conjecture~\ref{conj:pointprocess}.
Then for any $\zeta_0 \in (1.75/2,\, 3.18/2)$ and $c$ large,
\[
  N_\varphi(c)
  := \#\bigl\{j : \xi_j^{(c)} \in [-\zeta_0,\zeta_0]\bigr\}
  \to \#\bigl\{k : -a_k \in [-\zeta_0,\zeta_0]\bigr\} = 2.
\]
In particular $N_\varphi(c) = 2$ for all large $c$,
which is condition~(Q5) of Paper~XIII.
\end{corollary}

\section{The Double-Scaling Limit and Resolvent Expansion}
\label{sec:doublescaling}
%% ============================================================

\subsection{Scaling ansatz}

\begin{equation}\label{eq:scaling}
  x = 1 - c^{-2/3}\xi, \quad
  y = 1 - c^{-2/3}\eta, \quad
  z = z_{n^*} + c^{-1/3}\zeta,
\end{equation}
with $\xi,\eta \in [0,c^{2/3}]$ (the scaled interval $[-1,1]$)
and $\zeta \in \C \setminus \R$.

\subsection{Why exactly three scales: the scale separation argument}
\label{sec:scales}

\begin{remark}[Scale structure from PSWE]
\label{rem:scales}
The three scales $c, c^{2/3}, c^{1/3}$ in the resolvent expansion
are not assumed but \emph{derived} from the PSWE turning-point structure.
The prolate spheroidal wave equation is
\begin{equation}\label{eq:PSWE}
  \frac{d}{dx}[(1-x^2)\psi'] + [\chi_n - c^2 x^2]\psi = 0,
\end{equation}
with turning point at $x = x_0 = \sqrt{\chi_n}/c \to 1^-$
as $n \to n^*$.
Near $x = 1$, substituting $x = 1 - c^{-2/3}\xi$ and
$\chi_n = c^2 - 2c^{4/3}\tau_n + O(c^{2/3})$ (where
$\tau_n \sim (n - 2c/\pi)\cdot c^{-1/3}$) transforms~\eqref{eq:PSWE} into
\begin{equation}\label{eq:Airyasymp}
  \psi'' = (\xi + \tau_n)\psi + O(c^{-1/3}),
\end{equation}
the Airy equation shifted by $\tau_n$.
\emph{The three scales arise as:}
\begin{itemize}[nosep]
  \item $c$: Weyl-level bulk density $\Tr[\Kc] \sim 2c/\pi$;
  \item $c^{2/3}$: spatial scale of turning-point region ($x \sim 1 - c^{-2/3}$);
  \item $c^{1/3}$: spectral scale of transition ($\lambda_{n^*} \sim z_{n^*} \pm c^{-1/3}$).
\end{itemize}
These are the unique scales compatible with~\eqref{eq:Airyasymp};
no other scales arise to leading order.
This derivation justifies the ansatz in
Conjecture~\ref{conj:resolvent} below.
\end{remark}

\subsection{The central conjecture}

\begin{conjecture}[Resolvent trace expansion]
\label{conj:resolvent}
In the double-scaling regime~\eqref{eq:scaling},
\begin{equation}\label{eq:resolventexp}
  \Tr\bigl[(\Kc - z_{n^*} - c^{-1/3}\zeta)^{-1}\bigr]
  \;=\;
  c\cdot T_0 \;+\;
  c^{2/3}\cdot \Tr\bigl[(\cA - \zeta)^{-1}\bigr]
  \;+\; O(c^{1/3})
\end{equation}
uniformly for $\zeta$ in compact subsets of $\C\setminus\R$,
where $T_0$ is the bulk trace density (independent of $\zeta$)
and $\cA = -d^2/d\xi^2 + \xi$ on $L^2(\R_+)$ with Dirichlet BC.
\end{conjecture}

\begin{remark}[What the conjecture says and does not say]
\label{rem:conjstatus}
Conjecture~\ref{conj:resolvent} is \emph{not} a formal consequence
of the Airy approximation~\eqref{eq:Airyasymp} alone.
It requires additionally:
\begin{itemize}[nosep]
  \item that bulk and edge contributions to the trace are
    \emph{analytically separable} (not automatic for kernels on
    compact intervals; see Remark~\ref{rem:bulkedge});
  \item that the error in~\eqref{eq:Airyasymp} propagates
    to an $O(c^{1/3})$ error in the trace (Bridge Lemma, below);
  \item that $T_0$ is holomorphic in $\zeta$ (bulk separation).
\end{itemize}
Each of these is a genuine open sub-problem.
\end{remark}

\begin{remark}[Bulk--edge separation for compact kernels]
\label{rem:bulkedge}
For translation-invariant kernels on $\R$ (sine kernel, Bessel kernel),
bulk and edge contributions are analytically separated by
the support of the kernel.
For the sinc kernel on $[-1,1]$, the kernel is \emph{not}
translation-invariant: contributions from $x,y$ near $\pm 1$
(edge) and $x,y$ near $0$ (bulk) are coupled through the
Fourier restriction to $[-1,1]$.
A rigorous bulk--edge separation requires an explicit
\emph{microlocal decomposition} of the kernel, which is the
analytic content of step~(S1).
\end{remark}

\subsection{From resolvent expansion to trace asymptotics}

Assuming Conjecture~\ref{conj:resolvent} and applying
Helffer--Sj\"ostrand with $z = z_{n^*} + c^{-1/3}\zeta$:
\begin{align}\label{eq:tracedecomp}
  N_\varphi(c)
  &= -\frac{c^{-1/3}}{\pi}\int_\C
    \partial_{\bar\zeta}\tilde\varphi(\zeta)\,
    \Tr[(\Kc - z_{n^*} - c^{-1/3}\zeta)^{-1}]\,d^2\zeta \notag\\
  &= \underbrace{-\frac{c^{-1/3}\cdot c}{\pi}
    \int_\C \partial_{\bar\zeta}\tilde\varphi\cdot T_0\,d^2\zeta}_{=\,0}
  \;-\;
  \frac{c^{-1/3}\cdot c^{2/3}}{\pi}
  \int_\C \partial_{\bar\zeta}\tilde\varphi(\zeta)\,
  \Tr[(\cA-\zeta)^{-1}]\,d^2\zeta
  \;+\; O(c^{-1/3}).
\end{align}
The $c$-term vanishes because $T_0$ is holomorphic in $\zeta$
(if bulk separation holds).
The remaining $c^{1/3}$-prefactor cancels to $1$:
\begin{equation}\label{eq:NAifinal}
  N_\varphi(c)
  \;=\;
  \#\bigl\{\text{eigenvalues of }\cA\text{ in }\mathrm{supp}(\varphi)\bigr\}
  \;+\; O(c^{-1/3}).
\end{equation}
Corollary~\ref{cor:Q5} then follows from window calibration.

\section{The Bridge Lemma: Kernel $\to$ Trace-Norm Convergence}
\label{sec:bridge}
%% ============================================================

\begin{remark}[Why operator-norm convergence is insufficient]
Strong resolvent convergence $\Kc^{(\mathrm{sc})} \to \cA$
(which may follow from form convergence of the PSWE to the Airy ODE)
gives operator-norm convergence of the resolvent.
However, the Helffer--Sj\"ostrand formula requires
\emph{trace-norm} convergence.
The gap between the two is:
\[
  \|(B-\zeta)^{-1} - (\cA-\zeta)^{-1}\|_{\mathrm{tr}}
  \leq \|(B-\cA)\|_{\mathrm{tr}} \cdot \|(B-\zeta)^{-1}\|_{\op}
  \cdot \|(\cA-\zeta)^{-1}\|_{\op},
\]
so trace-norm resolvent convergence follows from
\emph{trace-norm convergence of the operators themselves}.
The quantitative form needed is:
$\|\Kc^{(\mathrm{sc})} - \cA\|_{\mathrm{tr}} = O(c^{-1/3})$.
\end{remark}

\begin{lemma}[Bridge Lemma --- placeholder]
\label{lem:bridge}
Let $\Kc^{(\mathrm{sc})}$ be the scaled operator obtained from $\Kc$
by the substitution~\eqref{eq:scaling}, acting on $L^2([0,c^{2/3}])$,
extended by zero to $L^2(\R_+)$.
Assume:
\begin{enumerate}[label=(B\arabic*),nosep]
  \item The kernel of $\Kc^{(\mathrm{sc})}$ converges in $L^2(\R_+^2)$
    to the kernel of the spectral projection $P_{(-\infty,0]}(\cA)$;
  \item The $L^2$-error satisfies $\|K_c^{(\mathrm{sc})} - K_{\cA}\|_{L^2} = O(c^{-1/3})$.
\end{enumerate}
Then:
\begin{equation}\label{eq:bridgeconclusion}
  \|\Kc^{(\mathrm{sc})} - P_{(-\infty,0]}(\cA)\|_{\mathrm{tr}}
  = O(c^{-1/3}),
\end{equation}
and consequently
\begin{equation}\label{eq:resolventtracebound}
  \bigl|\Tr[(\Kc^{(\mathrm{sc})}-\zeta)^{-1}] - \Tr[(\cA-\zeta)^{-1}]\bigr|
  = O\bigl(c^{-1/3} |\mathrm{Im}(\zeta)|^{-2}\bigr)
\end{equation}
for $\zeta \in \C\setminus\R$.
\end{lemma}

\begin{remark}[Status of Lemma~\ref{lem:bridge}]
\textbf{This is a placeholder lemma.}
The stated form is correct as an implication chain:
$(B1) + (B2) \Rightarrow \eqref{eq:bridgeconclusion}
\Rightarrow \eqref{eq:resolventtracebound}$.
The content, which is not yet proved, is conditions~(B1) and~(B2).
Verifying~(B2) is the analytic core of this paper.
It requires: precise asymptotics of the sinc kernel
$K_c(1-c^{-2/3}\xi, 1-c^{-2/3}\eta)$ near $\xi,\eta = 0$,
and an $L^2$-rate of convergence to the Airy spectral projection kernel.
This is step~(S1) of the programme; it is currently open.
\end{remark}

\section{The Fredholm Determinant Route}
\label{sec:fredholm}
%% ============================================================

\subsection{The regularisation problem}

\begin{remark}[Why $\det(\cA - \zeta)$ requires regularisation]
\label{rem:detreg}
$\cA$ is an unbounded self-adjoint operator with
$\sigma(\cA) = \{-a_k\}_{k\geq 1} \to +\infty$.
The expression $\det(\cA - \zeta)$ is therefore not a
classical Fredholm determinant (which requires a trace-class perturbation).
Three standard regularisation strategies are available:
\begin{enumerate}[label=(R\arabic*),nosep]
  \item \textbf{Relative Fredholm determinant.}
    Fix a reference operator $\cA_0$ with known determinant
    (e.g.\ harmonic oscillator) and define
    $\det\bigl[(\cA-\zeta)(\cA_0-\zeta)^{-1}\bigr]$,
    which is a legitimate Fredholm determinant if
    $(\cA-\cA_0)(\cA_0-\zeta)^{-1}$ is trace class.
  \item \textbf{Zeta-regularised determinant.}
    $\det_\zeta(\cA - \zeta) := \exp(-\partial_s|_{s=0} \zeta_{\cA}(s,\zeta))$,
    where $\zeta_{\cA}(s,\zeta) := \sum_k (-a_k - \zeta)^{-s}$.
    This is the standard approach in spectral geometry.
  \item \textbf{Simon's modified determinant.}
    $\det_2(I + B) := \det((I+B)e^{-B})$ for $B$ Hilbert--Schmidt
    (Simon~\cite{Simon2005}).
\end{enumerate}
For the Fredholm route of Section~\ref{sec:fredholmroute},
strategy~(R1) is most natural: it connects the zero structure
of $\det[(\Kc - \lambda)P_{\mathrm{edge}}]$ directly to
the relative determinant via the scaling limit.
\end{remark}

\subsection{The Fredholm route to Q5}
\label{sec:fredholmroute}

Let $P_{\mathrm{edge}}$ be the spectral projection of $\Kc$
onto eigenvalues near $z_{n^*}$ (within $2\zeta_0 c^{-1/3}$).
Define the \emph{edge Fredholm determinant}
\begin{equation}\label{eq:edgedet}
  D_{\mathrm{edge}}(c,\lambda)
  := \det\bigl(I - \lambda \Kc P_{\mathrm{edge}}\bigr).
\end{equation}
This is a polynomial in $\lambda$ of degree $N_{\mathrm{edge}}$
(the rank of $P_{\mathrm{edge}}$).

\begin{conjecture}[Fredholm convergence]
\label{conj:fredholm}
In the double-scaling limit, $D_{\mathrm{edge}}(c,\cdot)$
converges (after appropriate normalisation) to the
Airy Fredholm determinant
\begin{equation}\label{eq:AiryFred}
  D_{\Ai}(\zeta) :=
  \det\bigl(I - P_{(-\infty,\zeta]} K_{\Ai}\bigr),
\end{equation}
where $K_{\Ai}$ is the Airy kernel operator on $L^2(\R)$
and $P_{(-\infty,\zeta]}$ is multiplication by
$\mathbf{1}_{(-\infty,\zeta]}$.
Here $D_{\Ai}(\zeta)$ is the \emph{Tracy--Widom distribution function},
which is well-defined as a genuine Fredholm determinant
(trace-class perturbation).
\end{conjecture}

\begin{remark}[Rouch\'{e} argument --- conditional]
If Conjecture~\ref{conj:fredholm} holds with error $O(c^{-1/3})$
in Fredholm norm, then by Rouch\'{e}'s theorem
for operator-valued analytic functions (Gohberg--Krein~\cite{GohKre1969}),
the zeros of $D_{\mathrm{edge}}(c,\cdot)$ near $z_{n^*}$
converge to the zeros of $D_{\Ai}(\cdot)$,
which are exactly the Airy eigenvalues.
Local counting of PSWF eigenvalues then follows
from counting zeros of the Airy Fredholm determinant.
\end{remark}

\subsection{Open problems}

\begin{problem}[Kernel convergence rate]
\label{prob:S1}
Prove $\|K_c^{(\mathrm{sc})}(\xi,\eta) - K_{\cA}(\xi,\eta)\|_{L^2(\R_+^2)} = O(c^{-1/3})$.
\end{problem}

\begin{problem}[Bridge Lemma: conditions (B1)--(B2)]
\label{prob:bridge}
Verify the hypotheses of Lemma~\ref{lem:bridge}.
\end{problem}

\begin{problem}[Bulk--edge microlocal separation]
\label{prob:bulkedge}
Construct a microlocal decomposition separating bulk
and edge contributions to $\Tr[(\Kc-z)^{-1}]$
for $z$ near $z_{n^*}$.
\end{problem}

\begin{problem}[Airy point process universality for $\Kc$]
\label{prob:APP}
Prove Conjecture~\ref{conj:pointprocess}.
This is the structural analogue of GUE edge universality
for the prolate concentration operator.
\end{problem}

\begin{problem}[(Q5-lower) via quasimode injection]
\label{prob:lower}
Prove $N_\varphi(c) \geq 2 - O(c^{-1/3})$
by constructing two approximate eigenvectors
$f_n^{(c)}, f_{n+1}^{(c)}$ with centres in $W_c$
and residual $o(c^{-1/3})$.
\end{problem}
