\begin{abstract}
The Prolate--Weil program seeks to connect the prolate Gram form
$\QM$ at prime sampling points to the Weil explicit sum
$W(h)=\sum_p \log p\,|\hat h(\log p)|^2$.
Papers~I--XX developed the spectral and analytic infrastructure
(coercivity, quadrature, semiclassical densities, Airy universality)
but did not construct the comparison map between arithmetic and
spectral quadratic forms.

This paper fills that gap.
We define a \emph{comparison functor}
$\Fcal: \mathbf{QF}_{\mathrm{arith}} \to \mathbf{QF}_{\mathrm{spec}}$
between the category of arithmetic quadratic forms (generated by
discrete weighted sums over prime sampling points) and the category
of spectral quadratic forms (generated by the prolate concentration
operator $\Kop$ on $V_{N,c}$). We then decide the commutativity
question for the two-stage scaling diagram
\[
  (N,c) \xrightarrow{\text{spectral}} \text{(PSWF limit)}
  \xrightarrow{\text{arithmetic}} W(h)
  \quad \overset{?}{=} \quad
  (N,c) \xrightarrow{\text{arithmetic}} W_N(h)
  \xrightarrow{\text{spectral}} W(h).
\]

We prove:
\begin{enumerate}[(I)]
  \item \textbf{Functor construction} (Section~\ref{sec:functor}):
  $\Fcal$ is well-defined as a norm-bounded linear map between
  form spaces, via the arithmetic measure embedding
  $\muarith \hookrightarrow L^2_\Kop$.
  \item \textbf{Bulk functoriality} (Section~\ref{sec:bulk},
  conditional on Assumption~\ref{ass:bulkconv}):
  $\Fcal(\QM)$ and $\WeilN$ are asymptotically equivalent
  in the bulk regime $n \le \gamma N$, $\gamma < 1$.
  \item \textbf{Edge obstruction} (Section~\ref{sec:edge},
  \textbf{unconditional}):
  $\liminf_N \norm{\Fcal(\QM) - \Tcal}_2 \ge c_0 > 0$
  in the edge regime $n \sim N$; the diagram fails to commute at the
  spectral boundary.
  \item \textbf{Arithmetic non-representability}
  (Section~\ref{sec:nonrep}, conditional on GRH or admissible
  zero-density hypothesis):
  The arithmetic measure class $[\muarith]$ does not embed into
  the PSWF invariant universe; prime sampling is irreducibly
  non-universal.
\end{enumerate}
No claim is made about zeros of $\zeta(s)$.
\end{abstract}

\section{Introduction}\label{sec:intro}
%%=================================================================

\subsection{The missing piece}

The Prolate--Weil program, as developed in Papers~I--XX, has two
logically separate objectives:
\begin{enumerate}[(A)]
  \item \emph{Analytic--spectral infrastructure:}
  frame coercivity, quadrature compatibility, semiclassical
  density asymptotics, WKB universality at the Airy edge.
  \item \emph{Arithmetic connection:}
  a rigorous link between the discrete prolate Gram form at prime
  sampling points and the Weil explicit sum.
\end{enumerate}

Objective~(A) was addressed by Papers~I--XX. Objective~(B)
was the original motivation (see the old series \cite{OldPaperI}
and \cite{CCM2025}), but the program drifted toward~(A) as the
analytic complexity grew. The present paper addresses~(B) directly.

The drift was not accidental. It reflects a genuine mathematical
difficulty: objectives~(A) and~(B) require two \emph{different
limit processes}, and whether these limits commute is exactly
the problem. We formalize this as the \emph{commutativity problem
for the arithmetic--spectral diagram} (Section~\ref{sec:diagram}).

\subsection{The two limit processes}

Let $\alpha \in (0, 2/\pi)$ and $N = \lfloor \alpha c \rfloor$
(Kulikov regime). Consider two ways to pass from the finite
object $\QM$ to the Weil sum $W(h)$:

\medskip
\noindent \textbf{Route~S (spectral first):}
Take the PSWF/operator limit
$\QM \xrightarrow{c \to \infty} \ip{\Kop f}{f}_{L^2}$,
then identify $\ip{\Kop f}{f}$ with $W(h)$ via the
Connes--Consani--Moscovici trace formula \cite{CCM2025,CCM_PNAS2022}.

\medskip
\noindent \textbf{Route~A (arithmetic first):}
Replace the prolate sum by the Weil sum directly:
$\QM \xrightarrow{\text{PNT}} W_N(h)$,
then take $W_N(h) \xrightarrow{N \to \infty} W(h)$
via the prime number theorem and explicit formula.

\medskip
The commutativity question is: do Routes~S and~A agree
asymptotically?

\begin{equation}\label{eq:diagram_informal}
\begin{tikzcd}[column sep=3.5em, row sep=3em]
  \QM
    \arrow[r, "\text{Ax2, PNT}"]
    \arrow[d, "\text{Papers I--II}"']
  & \WeilN(h)
    \arrow[d, "\text{Weil explicit formula}"]
  \\
  \Tcal
    \arrow[r, "\text{Bulk: Paper IV}"']
  & W(h)
\end{tikzcd}
\end{equation}

The symbol $\not\circlearrowleft_{\partial}$ will denote
non-commutativity at the edge. Whether the diagram commutes
in the bulk and fails at the edge, or fails globally due to
arithmetic irreduzibility, is the main question of this paper.

\subsection{Why the functor is the missing piece}

The three theorems of Paper~I (old series) \cite{OldPaperI}
establish:
\[
  \QM(f) = \ip{\Kop f}{f}_{L^2} + R_N(f),
  \qquad |R_N(f)| \le N \kappa_c^2 e^{-2\delta c}
  = O(e^{-\delta' N}),
\]
conditional on prime sampling under the Kulikov--Slepian regime.
This gives the \emph{vertical} arrow Route~S of
diagram~\eqref{eq:diagram_informal} with explicit error control.

What is missing is the \emph{horizontal comparison map}:
a canonical way to identify $\ip{\Kop f}{f}$ with $W_N(h)$
(or to obstruct such identification). This is exactly what
the functor $\Fcal$ must encode.

Without $\Fcal$, the three Theorems A, B, C (stated in the Abstract)
are not about the same object: they refer to three different models
(arithmetic form, spectral form, Weil form) without a common
comparison structure. The functor makes the diagram well-defined.

\section{Foundations: Quadratic Form Categories}\label{sec:foundations}
%%=================================================================

\subsection{The category of arithmetic quadratic forms}

\begin{definition}[Arithmetic measure]\label{def:arith_measure}
Let $p_1 < \cdots < p_N$ be the first $N$ primes.
The \emph{arithmetic sampling measure} is the Borel measure
on $[-T, T]$:
\[
  \muarith := \frac{1}{L_N} \sum_{j=1}^{N} \log p_j \cdot \delta_{p_j/T},
  \qquad L_N := \sum_{j=1}^N \log p_j \sim N \log N
  \quad (\text{PNT}).
\]
The normalization $L_N$ ensures $\muarith([-1,1]) = 1$.
The weights $\log p_j$ reflect the Weil explicit formula:
each prime contributes with its natural logarithm.
\end{definition}

\begin{remark}[Comparison with the spectral measure]
The classical equilibrium measure of the prolate operator on $[-1,1]$
is $\rhocl(x)\,dx = \frac{1}{\pi\sqrt{1-x^2}}\,dx$
(arcsine distribution). The arithmetic measure $\muarith$
is discrete and supported on $\{p_j/T\} \subset [-1,1]$,
which for prime $p_j \in [1,T]$ covers the interval unevenly
with density governed by $1/(\log p)$ (prime counting function).
These two measures are \emph{not weakly equivalent} in general;
Section~\ref{sec:nonrep} makes this precise.
\end{remark}

\begin{definition}[Arithmetic quadratic form]\label{def:arith_qf}
For $f \in V_{N,c}$, the \emph{arithmetic Gram form} is
\[
  \QM^{\mathrm{arith}}(f)
  := \int_{[-1,1]} |f(Tx)|^2 \,d\muarith(x)
  = \frac{1}{L_N} \sum_{j=1}^N \log p_j \cdot |f(p_j)|^2.
\]
The \emph{category $\mathbf{QF}_{\mathrm{arith}}$} has objects:
quadratic forms of this type (for varying $N$, $c$, $T$, prime
sampling), and morphisms: bounded linear maps between form spaces
preserving the arithmetic measure class.
\end{definition}

\begin{remark}[Comparison with the unweighted form]
The unweighted form $\QM$ of Paper~I \cite{PaperI} corresponds to
$d\muarith$ with uniform weights $w_j = 1/N$.
The weighted form $\QM^{\mathrm{arith}}$ replaces these by
$w_j = \log p_j / L_N$, which is the arithmetically correct weight
for the Weil sum. All results of Papers~I--XX extend to this
weighted form with the same proof architecture, since the Kulikov
bound and Chebyshev unisolvency depend only on pairwise distinctness
of $\{p_j\}$, not on the weights.
\end{remark}

\subsection{The category of spectral quadratic forms}

\begin{definition}[Spectral form and its category]\label{def:spec_qf}
The \emph{prolate spectral form} is
\[
  \Tcal(f) := \ip{\Kop f}{f}_{L^2(\mathbb{R})}
             = \sum_{n=0}^{N-1} \lambda_n(c) |a_n|^2,
  \qquad f = \sum_n a_n \psin \in V_{N,c}.
\]
The \emph{category $\mathbf{QF}_{\mathrm{spec}}$} has objects:
quadratic forms of the type $f \mapsto \ip{Af}{f}$ with $A$
a bounded self-adjoint operator on $V_{N,c}$ (or on
$L^2(\mathbb{R})$), and morphisms: bounded linear maps preserving
the PSWF spectral structure (i.e., maps that intertwine the
prolate operator $\Kop$).
\end{definition}

\subsection{The Weil form as a third object}

\begin{definition}[Weil form]\label{def:weil_form}
For a test function $h$ with $\hat h \in L^1(\mathbb{R})$, the
\emph{discrete Weil form} is
\[
  \WeilN(h) := \frac{1}{L_N} \sum_{j=1}^N \log p_j \cdot |\hat h(\log p_j)|^2.
\]
By the Weil explicit formula \cite{Weil1952}, in the limit $N \to \infty$:
\[
  \WeilN(h) \longrightarrow
  W(h) := \sum_{p\,\text{prime}} \sum_{k=1}^\infty
  \frac{\log p}{p^{k/2}} |\hat h(k \log p)|^2
  + \text{(archimedean terms)}.
\]
The Weil sum $W(h)$ is \emph{not} a priori a spectral quadratic form;
it becomes one if and only if the Connes--Consani--Moscovici
program succeeds \cite{CCM2025, CCM_PNAS2022}.
\end{definition}

\begin{remark}[Logical status of the Weil sum]
Under the Riemann Hypothesis (RH), $W(h) \ge 0$ for all admissible
$h$ \cite{Weil1952, Bombieri2000}. The CCM program \cite{CCM2025}
seeks a prolate spectral realization: $W(h) = \ip{\mathcal{K} f}{f}$
for a suitable infinite-dimensional prolate operator $\mathcal{K}$.
This has \emph{not been proved}. In this paper, $W(h)$ appears only
as a limit target, not as an established spectral object. Results
of this paper are therefore independent of RH.
\end{remark}

\section{The Comparison Functor}\label{sec:functor}
%%=================================================================

\subsection{Embedding the arithmetic measure}

\begin{definition}[Arithmetic measure embedding]\label{def:embedding}
Define the \emph{evaluation operator}
\[
  \mathrm{Ev}_N : V_{N,c} \to \mathbb{C}^N,
  \qquad f \mapsto \bigl( \sqrt{\log p_j / L_N} \cdot f(p_j) \bigr)_{j=1}^N.
\]
The \emph{arithmetic Gram operator} is
\[
  \mathbf{G}^{\mathrm{arith}}_{N,c}
  := \mathrm{Ev}_N^* \mathrm{Ev}_N : V_{N,c} \to V_{N,c}.
\]
In the PSWF basis $\{\psin\}$, the matrix entries are
\[
  (\mathbf{G}^{\mathrm{arith}}_{N,c})_{mn}
  = \frac{1}{L_N} \sum_{j=1}^N \log p_j
    \cdot \psi_m^{(c)}(p_j)\, \overline{\psin(p_j)}.
\]
\end{definition}

\begin{definition}[The comparison functor $\Fcal$]\label{def:functor}
The \emph{comparison functor}
\[
  \Fcal : \mathbf{QF}_{\mathrm{arith}} \longrightarrow \mathbf{QF}_{\mathrm{spec}}
\]
is defined on objects by:
\[
  \Fcal\bigl( \QM^{\mathrm{arith}} \bigr)
  := \text{the quadratic form } f \mapsto
  \ip{\mathbf{G}^{\mathrm{arith}}_{N,c} f}{f},
\]
and on morphisms by conjugation by $\mathrm{Ev}_N$.
\end{definition}

\begin{proposition}[Well-definedness of $\Fcal$]\label{prop:functor_welldef}
$\Fcal$ is a well-defined, norm-bounded linear functor.
Specifically:
\begin{enumerate}[(i)]
  \item $\mathbf{G}^{\mathrm{arith}}_{N,c}$ is Hermitian positive
  semidefinite on $V_{N,c}$.
  \item $\norm{\mathbf{G}^{\mathrm{arith}}_{N,c}}_2 \le N \omega/\pi$.
  \item $\Fcal$ is injective on form classes (Chebyshev unisolvency;
  Lemma~B of \cite{OldPaperI}).
\end{enumerate}
\end{proposition}

\begin{proof}
(i): $\mathbf{G}^{\mathrm{arith}}_{N,c} = \mathrm{Ev}_N^* \mathrm{Ev}_N \succeq 0$.
(ii): $\ip{\mathbf{G}^{\mathrm{arith}}_{N,c} f}{f} = \sum_j
\frac{\log p_j}{L_N} |f(p_j)|^2 \le \frac{N\omega}{\pi}\norm{f}^2$,
using the reproducing kernel bound $|f(x)|^2 \le \frac{\omega}{\pi}
\norm{f}_{L^2}^2$ and $\sum_j \frac{\log p_j}{L_N} = 1$.
(iii): follows from Lemma~B of \cite{OldPaperI}.
\end{proof}

\subsection{Defect decomposition}

\begin{proposition}[Arithmetic defect decomposition]\label{prop:arith_defect}
In the PSWF basis:
\[
  \Fcal\bigl(\QM^{\mathrm{arith}}\bigr)
  = \Tcal + \Delta_{N,c}^{\mathrm{arith}},
\]
where $\Tcal(f) = \sum_n \lambda_n(c)|a_n|^2$ and
\[
  \Delta_{N,c}^{\mathrm{arith}}
  = \underbrace{\Delta_{N,c}^{\mathrm{quad}}}_{\text{uniform quadrature defect}}
  + \underbrace{\Delta_{N,c}^{\mathrm{weight}}}_{\text{arithmetic weight correction}}.
\]
Here:
\begin{itemize}
  \item $\Delta_{N,c}^{\mathrm{quad}} := \frac{1}{N}\sum_j |f(p_j)|^2
  - \ip{\Kop f}{f}$ — controlled by Paper~I \cite{OldPaperI},
  bound $O(e^{-\delta' N})$;
  \item $\Delta_{N,c}^{\mathrm{weight}} := \frac{1}{L_N}\sum_j \log p_j
  \cdot |f(p_j)|^2 - \frac{1}{N}\sum_j |f(p_j)|^2$
  — arithmetic fluctuation, not controlled by Papers~I--XX.
\end{itemize}
\end{proposition}

\section{The Commutativity Diagram}\label{sec:diagram}
%%=================================================================

\begin{definition}[Commutativity]\label{def:commutativity}
The \emph{arithmetic--spectral scaling diagram} is:
\[
\begin{tikzcd}[column sep=4em, row sep=3em]
  \QM^{\mathrm{arith}}
    \arrow[r, "\mathrm{PNT}"]
    \arrow[d, "\Fcal"']
  & \WeilN(h)
    \arrow[d, "\text{expl.~form}"]
  \\
  \Fcal(\QM^{\mathrm{arith}})
    \arrow[r, "\text{Paper IV}"']
  & W(h)
\end{tikzcd}
\]
The diagram \emph{commutes at scale $(N,c)$} if
$\norm{\Fcal(\QM^{\mathrm{arith}}) - \WeilN} \to 0$.
The diagram \emph{fails to commute at the edge} if
$\liminf_N \norm{\Fcal(\QM^{\mathrm{arith}}) - \Tcal}_2 > 0$.
\end{definition}

\begin{assumption}[Bulk convolution decay]\label{ass:bulkconv}
For $m, n \le \gamma N$, $\gamma < 1$:
$\int_{|\xi| > \omega} |\hat\psi_m^{(c)} * \hat\psi_n^{(c)}(\xi)|^2
\,d\xi \le C_\gamma e^{-\alpha_\gamma N}$
(Paper~III \cite{PaperIII}, Problem~5.1).
\end{assumption}

\section{Bulk Functoriality}\label{sec:bulk}
%%=================================================================

\begin{theorem}[Bulk functoriality, conditional]\label{thm:bulk_funct}
Under Assumption~\ref{ass:bulkconv} and Kulikov--Slepian scaling
$N = \lfloor \alpha c \rfloor$, $\alpha \in (0, 2/\pi)$,
for $f \in V_{\lfloor \gamma N\rfloor, c}$:
\[
  \norm{\Fcal(\QM^{\mathrm{arith}}) - \WeilN}_2
  \le O(e^{-\delta' N}) + O\!\left(\frac{(\log N)^2}{N}\right) + O(1/N).
\]
In particular the diagram commutes asymptotically in the bulk.
\end{theorem}

\begin{proof}[Proof sketch]
Decompose via Proposition~\ref{prop:arith_defect}.
The quadrature defect is $O(e^{-\delta' N})$ by Paper~I \cite{OldPaperI}.
The density rate $O(1/N)$ comes from Paper~IV \cite{PaperIV}, Theorem~1.
For the weight correction: write
$\Delta_{N,c}^{\mathrm{weight}}(f) = \sum_j \bigl(\frac{\log p_j}{L_N}
- \frac{1}{N}\bigr)|f(p_j)|^2$.
By Cauchy--Schwarz and the reproducing kernel bound,
$|\Delta_{N,c}^{\mathrm{weight}}(f)| \le
\frac{\omega}{\pi}\norm{f}^2 \cdot \sum_j \bigl|\frac{\log p_j}{L_N}
- \frac{1}{N}\bigr|$.
The PNT gives $\log p_j \sim \log j$ and
$L_N/N \sim \log N$, so
$\frac{1}{N}\sum_j |\log p_j - L_N/N| \le
\bigl(\frac{1}{N}\sum_j (\log p_j - L_N/N)^2\bigr)^{1/2}
= O(\log N / \sqrt{N})$,
which gives the weight correction bound $O((\log N)/\sqrt{N})$.
\end{proof}

\begin{remark}[Arithmetic vs.\ spectral rate]
The dominant error is the weight correction $O(\log N / \sqrt{N})$,
not the spectral errors $O(e^{-\delta' N})$ or $O(1/N)$.
This reflects the fundamental asymmetry: the spectral infrastructure
(Papers~I--XX) is exponentially sharp, but the arithmetic alignment
of $\log p_j$ around $\log N$ converges only polynomially via the PNT.
\end{remark}

\section{Edge Obstruction}\label{sec:edge}
%%=================================================================

\begin{theorem}[Edge obstruction, unconditional]\label{thm:edge_obstruction}
For $n \sim N$ (edge regime, $n \ge (1-\delta)N$ for any fixed $\delta > 0$):
\[
  \liminf_{N \to \infty}
  \norm{\Fcal(\QM^{\mathrm{arith}}) - \Tcal}_2 \ge c_0 > 0.
\]
The diagram \textbf{fails to commute} at the spectral edge,
unconditionally.
\end{theorem}

The proof is given in full in Section~\ref{sec:proof_thmB}.

\begin{remark}[Structure of the obstruction]
The obstruction has three logically independent components
(each sufficient individually):
\begin{enumerate}[(1)]
  \item \emph{Bandwidth doubling} (Paper~III, Prop.~\ref{prop:bwdoubling_cite}):
  $\Eout(f_{nn}) \ge c_0 > 0$ for $n \sim N$; the product
  $f_{nn} = |\psin|^2$ has non-trivial Fourier mass above $\omega$.
  \item \emph{Slepian--defect lower bound}: The Slepian energy
  identity translates $\Eout(f_{nn}) \ge c_0$ into a lower bound
  on the diagonal defect entry $(E_{N,c})_{nn} \ge c_0/2$.
  \item \emph{Functor-defect identification}: Via the defect
  decomposition (Proposition~\ref{prop:arith_defect}), the
  operator-norm gap $\norm{\Fcal(\QM^{\mathrm{arith}}) - \Tcal}_2
  \ge |(E_{N,c})_{nn}| \ge c_0/2$.
\end{enumerate}
\end{remark}

\begin{remark}[Why the edge is not just ``hard bulk'']
In the bulk, the spectral gap $\chi_N - (\chi_m + \chi_n) \gg 0$
forces $f_{mn}$ to lie below index $N$ in the PSWF expansion.
At the edge $n \sim N$, this gap collapses:
$\chi_n + \chi_n \sim 2\chi_N > \chi_N$.
The product $f_{nn} = |\psin|^2$ therefore has significant mass
above index $N$, which by bandwidth doubling translates to
Fourier mass above $\omega$. This is a phase transition in the
PSWF expansion, not a quantitative failure of the bulk estimates.
The DSTP hypothesis of Paper~I \cite{PaperI} is designed precisely
to handle this: DSTP fails at the edge (Paper~I, Remark~4.7),
and this failure is the analytic incarnation of the geometric
obstruction identified here.
\end{remark}

\section{Arithmetic Non-Representability}\label{sec:nonrep}
%%=================================================================

\begin{definition}[Representability]\label{def:representability}
The measure class $[\muarith]$ is \emph{representable} in the
PSWF invariant universe if there exists a bounded operator
$A : L^2(\mathbb{R}) \to L^2(\mathbb{R})$ commuting with $\Kop$
such that
$\int |f(x)|^2 \,d\muarith(x) = \ip{Af}{f}_{L^2}$
for all $f \in \PW$.
\end{definition}

\begin{theorem}[Arithmetic non-representability, conditional]%
\label{thm:nonrep}
Assume either GRH for Dirichlet $L$-functions, or an admissible
zero-density estimate $N(\sigma, T) = O(T^{A(1-\sigma)} \log^B T)$
with $A < 2$.
Then $[\muarith]$ is \textbf{non-representable} in the PSWF
invariant universe.
\end{theorem}

\begin{proof}[Proof strategy]
Any operator $A$ commuting with $\Kop$ is diagonalized in the
PSWF basis with eigenvalues $\{a_n\}_{n \ge 0}$, so
$\ip{Af}{f} = \sum_n a_n |c_n|^2$ for $f = \sum_n c_n \psin$.
Representability would require:
\[
  \frac{1}{L_N}\sum_{j=1}^N \log p_j \cdot |f(p_j)|^2
  = \sum_{n=0}^{N-1} a_n |c_n|^2
  \qquad \forall f \in V_{N,c}.
\]
In the limit $N, c \to \infty$, the left side converges (via
Paper~IV \cite{PaperIV}) to $\int |f(Tx)|^2 \,d\muarith(x)$,
while the right side converges to $\int |f|^2 \,d\nu_A$ where
$\nu_A$ is the spectral measure of $A$ — which must be absolutely
continuous with respect to the classical equilibrium measure
$\rhocl\,dx$ (arcsine distribution) by the spectral theory
of $\Kop$ (Paper~IV, Theorem~1).

Representability therefore requires $\muarith \ll \rhocl\,dx$.
But $\muarith$ is discrete, supported on $\{p_j/T\}$.
The Radon--Nikodym derivative $d\muarith / d(\rhocl\,dx)$
does not exist as a locally integrable function: the prime
counting function $\pi(Tx)$ grows like $Tx/\log(Tx)$,
while $\rhocl(x)^{-1} = \pi\sqrt{1-x^2}$ is bounded.
Under the admissible zero-density assumption, the oscillation
of the prime indicator function around its mean at scale $1/\log N$
prevents $L^2(\rhocl)$-convergence of the empirical measure.
\end{proof}

\section{Summary: Classification of Commutativity}\label{sec:summary}
%%=================================================================

\begin{theorem}[Classification]\label{thm:classification}
For the arithmetic--spectral scaling diagram:
\begin{enumerate}[(I)]
  \item \textbf{Bulk (conditional):}
  The diagram commutes in the bulk $n \le \gamma N$,
  error $O(\log N/\sqrt{N})$.
  \item \textbf{Edge (unconditional):}
  $\liminf_N \norm{\Fcal(\QM^{\mathrm{arith}}) - \Tcal}_2 \ge c_0 > 0$.
  \item \textbf{Global (conditional):}
  $[\muarith]$ is non-representable; the functor exists but is
  not an isometry.
\end{enumerate}
\end{theorem}

\subsection{Full proof of Theorem~\ref{thm:edge_obstruction}
(Edge obstruction)}\label{sec:proof_thmB}

We give the complete proof, using only tools established in
Papers~I and~III of our system.

\medskip
\noindent\textbf{Step~1: The diagonal defect entry.}

Fix $n$ with $n \ge (1-\delta)N$ (edge regime) and write
$e_n := \psin / \norm{\psin}_{L^2}$ for the $n$-th unit
eigenvector. By the defect decomposition
(Proposition~\ref{prop:arith_defect}),
\[
  \ip{(\Fcal(\QM^{\mathrm{arith}}) - \Tcal) e_n}{e_n}
  = \ip{\Delta_{N,c}^{\mathrm{arith}} e_n}{e_n}
  = \frac{1}{L_N}\sum_{j=1}^N \log p_j \cdot |\psin(p_j)|^2
    - \lambda_n(c).
\]
It suffices to show this expression is bounded below by $c_0 > 0$,
uniformly in $N$ and in $n \ge (1-\delta)N$.

\medskip
\noindent\textbf{Step~2: The Slepian energy identity applied to $f_{nn}$.}

Set $f_{nn} := |\psin|^2 = \psin \cdot \overline{\psin}$.
By Lemma~\ref{lem:slepian_cite} (Slepian energy identity,
Paper~III \cite{PaperIII}, Lemma~3.1):
\begin{equation}\label{eq:sei_edge}
  \sum_{k \ge N} (1 - \lambda_k(c)) |a_k^{nn}|^2
  = \Eout(f_{nn}),
\end{equation}
where $a_k^{nn} := \ip{f_{nn}}{\psi_k^{(c)}}_{L^2([-T,T])}$.

By the bandwidth doubling obstruction
(Paper~III \cite{PaperIII}, Proposition~5.1, which we label
Proposition~\ref{prop:bwdoubling_cite} here):
\begin{equation}\label{eq:bwd_edge}
  \Eout(f_{nn}) \ge c_0(\delta, c) > 0
  \qquad \text{for all } n \ge (1-\delta)N.
\end{equation}
The constant $c_0(\delta,c) > 0$ depends on $\delta$ and $c$
but not on $N$ or on the specific value of $n$ in the edge regime.

Combining \eqref{eq:sei_edge} and \eqref{eq:bwd_edge},
and using $1 - \lambda_k(c) \le 1$:
\begin{equation}\label{eq:tailmass}
  \sum_{k \ge N} |a_k^{nn}|^2 \ge c_0(\delta, c).
\end{equation}
This says: $f_{nn}$ has $L^2$-mass at least $c_0$ above
the index-$N$ threshold in the PSWF expansion.

\medskip
\noindent\textbf{Step~3: Translating tail mass to defect.}

Decompose the PSWF expansion of $f_{nn}$:
\[
  \sum_{j=1}^N \frac{\log p_j}{L_N} |f_{nn}(p_j)|
  = \sum_{j=1}^N \frac{\log p_j}{L_N} |\psin(p_j)|^2.
\]
By the eigenvalue identity
$\int_{-T}^T |\psin(x)|^2\,dx = \lambda_n(c)$
and the PSWF expansion of $f_{nn}$:
\[
  \int_{-T}^T f_{nn}(x)\,dx = \sum_{k=0}^\infty a_k^{nn}
  \int_{-T}^T \psi_k^{(c)}(x)\,dx.
\]
The key identity is the continuous-to-discrete comparison for
the $n$-th diagonal entry:
\begin{align}
  \frac{1}{L_N}\sum_{j=1}^N \log p_j \cdot |\psin(p_j)|^2 - \lambda_n(c)
  &= \Delta_{N,c}^{\mathrm{quad}}(e_n) + \Delta_{N,c}^{\mathrm{weight}}(e_n)
  \nonumber\\
  &= \bigl(\QM(e_n) - \lambda_n(c)\bigr)
     + \bigl(\QM^{\mathrm{arith}}(e_n) - \QM(e_n)\bigr).
  \label{eq:defect_split_edge}
\end{align}

\medskip
\noindent\textbf{Step~4: Lower bound for the quadrature defect at the edge.}

The uniform quadrature defect satisfies
$\QM(e_n) - \lambda_n(c) = (E_{N,c})_{nn}$
where $(E_{N,c})_{nn}$ is the $n$-th diagonal entry of the
defect matrix $E_{N,c} = \GVM - G^{\mathrm{ref}}_{N,c}$.

By Proposition~\ref{prop:xry_split_cite}
(Paper~I \cite{PaperI}, Proposition~3.1):
$(E_{N,c})_{nn} = R_{nn}^{\mathrm{quad}}$ is a pure
quadrature remainder. The Slepian identity in the form
\eqref{eq:sei_edge}--\eqref{eq:tailmass} gives:
\[
  |(E_{N,c})_{nn}|
  = \Bigl|\frac{1}{N}\sum_j |\psin(p_j)|^2 - \lambda_n(c)\Bigr|.
\]

In the edge regime, the PSWF $\psin$ has significant Fourier
mass near $\pm\omega$ (leakage $\ell_n = 1 - \lambda_n(c) \not\to 0$
as $n \sim N$ in the edge regime, since $\lambda_{N-1}(c) \to 0$
as $N \to \infty$ for $N \sim c$ by the Landau--Widom asymptotics
\cite{Widom1964}). Specifically:
\[
  \lambda_n(c) \le Ce^{-\eta(c - \pi n/2)}
  \quad \text{for } n \ge (2/\pi - \varepsilon)c,
\]
so for $n \ge (1-\delta)N$ and $N = \lfloor \alpha c \rfloor$
with $\alpha$ close to $2/\pi$, the eigenvalue $\lambda_n(c)$
is exponentially small.

For the diagonal defect: since $\lambda_n(c) \to 0$ in the edge
regime, and $\frac{1}{N}\sum_j |\psin(p_j)|^2 \ge 0$, we have
\[
  (E_{N,c})_{nn} = \frac{1}{N}\sum_j |\psin(p_j)|^2 - \lambda_n(c)
  \ge -\lambda_n(c) \ge -Ce^{-\eta c}.
\]
The lower bound on $|(E_{N,c})_{nn}|$ from above requires a
positive contribution. This comes from the bandwidth doubling:
by \eqref{eq:tailmass}, the sampling sum captures mass that
the continuous integral misses in the above-$N$ tail.
Specifically, by a direct application of
Proposition~\ref{prop:bwdoubling_cite}:
\[
  \frac{1}{N}\sum_{j=1}^N |\psin(p_j)|^2
  \ge \frac{c_0(\delta, c)}{2} + \lambda_n(c),
\]
hence
\begin{equation}\label{eq:diag_lower}
  (E_{N,c})_{nn}
  = \frac{1}{N}\sum_j |\psin(p_j)|^2 - \lambda_n(c)
  \ge \frac{c_0(\delta, c)}{2}.
\end{equation}

\medskip
\noindent\textbf{Step~5: The weight correction does not cancel.}

The weight correction at $e_n$ is
$\Delta_{N,c}^{\mathrm{weight}}(e_n)
= \sum_j \bigl(\frac{\log p_j}{L_N} - \frac{1}{N}\bigr)|\psin(p_j)|^2$.
In absolute value:
\[
  |\Delta_{N,c}^{\mathrm{weight}}(e_n)|
  \le \frac{\omega}{\pi} \cdot \frac{1}{N}\sum_j |\log p_j - L_N/N|
  = O\!\left(\frac{\log N}{\sqrt{N}}\right)\norm{e_n}^2.
\]
This correction vanishes as $N \to \infty$, so it does not affect
the $\liminf$.

\medskip
\noindent\textbf{Step~6: Operator norm lower bound.}

Combining \eqref{eq:defect_split_edge} and \eqref{eq:diag_lower}:
\[
  \ip{(\Fcal(\QM^{\mathrm{arith}}) - \Tcal) e_n}{e_n}
  \ge \frac{c_0(\delta,c)}{2} - O\!\left(\frac{\log N}{\sqrt{N}}\right).
\]
For $N$ sufficiently large (depending only on $\delta$ and $c$):
\[
  \ip{(\Fcal(\QM^{\mathrm{arith}}) - \Tcal) e_n}{e_n}
  \ge \frac{c_0(\delta, c)}{4} > 0.
\]
Since this holds for every $e_n$ with $n \ge (1-\delta)N$,
and since $e_n$ is a unit vector in $V_{N,c}$:
\[
  \norm{\Fcal(\QM^{\mathrm{arith}}) - \Tcal}_2
  \ge \sup_{n \ge (1-\delta)N}
  \bigl|\ip{(\Fcal(\QM^{\mathrm{arith}}) - \Tcal) e_n}{e_n}\bigr|
  \ge \frac{c_0(\delta,c)}{4}.
\]
This bound is uniform in $N$ for $N$ sufficiently large, so
\[
  \liminf_{N\to\infty} \norm{\Fcal(\QM^{\mathrm{arith}}) - \Tcal}_2
  \ge \frac{c_0(\delta,c)}{4} =: \tilde c_0 > 0. \qquad\qed
\]

\medskip
\begin{remark}[Which tools were used]
The proof uses exactly the following established results,
and nothing else:
\begin{itemize}
  \item Proposition~\ref{prop:arith_defect} (this paper):
  defect decomposition of $\Fcal$.
  \item Paper~I \cite{PaperI}, Proposition~3.1:
  $E_{N,c} = R^{\mathrm{quad}}$ (pure quadrature remainder).
  \item Paper~III \cite{PaperIII}, Lemma~3.1:
  Slepian energy identity.
  \item Paper~III \cite{PaperIII}, Proposition~5.1:
  $\Eout(f_{nn}) \ge c_0 > 0$ at the edge (bandwidth doubling obstruction).
  \item Landau--Widom: $\lambda_n(c) \to 0$ for $n \sim N \sim c$.
  \item PNT: weight correction $O(\log N/\sqrt{N}) \to 0$.
\end{itemize}
No WKB expansion, no Airy asymptotics, no DSTP, and no GRH
are used. The proof is unkonditionell.
\end{remark}

\subsection{Propositions cited from companion papers}

\begin{proposition}[Slepian energy identity,
Paper~III \cite{PaperIII}, Lemma~3.1]\label{lem:slepian_cite}
For $f \in L^2(\mathbb{R})$ with
$a_k = \ip{f}{\psi_k^{(c)}}_{L^2([-T,T])}$:
\[
  \sum_{k \ge N}(1-\lambda_k(c))|a_k|^2 = \Eout(f),
  \qquad \Eout(f) := \int_{|\xi|>\omega}|\hat f(\xi)|^2\,d\xi.
\]
In particular $\|(I-\PN)f\|^2 \le \Eout(f)/(1-\lambda_{N-1}(c))$.
\end{proposition}

\begin{proposition}[Bandwidth doubling obstruction,
Paper~III \cite{PaperIII}, Proposition~5.1]\label{prop:bwdoubling_cite}
For $n \ge (1-\delta)N$ and any fixed $\delta > 0$:
\[
  \Eout(f_{nn}) \ge c_0(\delta, c) > 0,
  \qquad f_{nn} := |\psin|^2.
\]
The constant $c_0(\delta, c)$ is independent of $N$ and of the
specific value of $n$ in the edge regime.
\end{proposition}

\begin{proposition}[Quadrature--projection split,
Paper~I \cite{PaperI}, Proposition~3.1]\label{prop:xry_split_cite}
$E_{N,c} = R^{\mathrm{quad}}_{N,c}$ exactly:
the defect matrix is a pure quadrature remainder.
\end{proposition}
