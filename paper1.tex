\begin{abstract}
Let $\PW$ denote the Paley--Wiener space of $\omega$-bandlimited
functions and let
\[
  V_{N,c}:=\operatorname{span}\{\psi_0^{(c)},\dots,\psi_{N-1}^{(c)}\}\subset \PW
\]
be the $N$-dimensional prolate subspace associated with the
time-bandwidth product $c=\omega T$.
For $N$ distinct sampling points $\mathbf{p}\subset[-T,T]$
we consider the prolate Gram form
$\QM(f)=\sum_{j=1}^N |f(p_j)|^2$ on $V_{N,c}$.

Layer~A proves, under the unisolvence assumption on $\mathbf{p}$,
coercivity and continuity of $\QM$ by finite-dimensional linear algebra.
Layer~B proves a conditional spectral decomposition under an abstract
sampling-quality condition~\textup{(T.2)}.
Layer~C is explicitly heuristic and discusses prime sampling.

No claim is made here about zeros of $\zeta(s)$, about prime sampling
satisfying~\textup{(T.2)}, or about coupled limits in which both $N$
and $c$ vary.
\end{abstract}

\section{Introduction}

The Weil positivity criterion \cite{Weil1952,Bombieri2000,CC2019}
connects positivity of an explicit quadratic form over primes with the
Riemann Hypothesis. The prolate program of
Connes--Consani--Moscovici \cite{CCM_PNAS2022,CCM2025}
seeks a spectral realization of this form using the prolate operator
$\Kop$ on Paley--Wiener space.

The present paper isolates a finite-dimensional coercivity statement.
Its logical architecture is deliberately stratified:
Layer~A is unconditional up to unisolvence,
Layer~B is conditional on an abstract stability hypothesis,
and Layer~C is heuristic only.

\section{Setup and notation}\label{sec:setup}

\subsection{Basic objects}

We use the engineering Fourier convention
$\widehat{f}(\xi)=\int_{\mathbb{R}}f(x)e^{-2\pi i x\xi}\,dx$.
Throughout, $N\ge 1$, $\omega>0$, $T>0$, and $c=\omega T$.

The Paley--Wiener space $\PW$ consists of all
$f\in L^2(\mathbb{R})$ with Fourier support in
$[-\omega/(2\pi),\omega/(2\pi)]$.
Its reproducing kernel satisfies
\[
  f(x)=\ip{f}{K_x}_{L^2},\qquad K_{\PW}(x,x)=\omega/\pi
\]
for all $f\in\PW$ \cite[Eq.~(3)]{BonamiKaroui}.

Let $\Kop$ be the prolate concentration operator
\[
  \Kop f(x)=\int_{-T}^{T}\frac{\sin\omega(x-y)}{\pi(x-y)}f(y)\,dy.
\]
Its eigenvalues satisfy
$1>\lambda_0(c)>\lambda_1(c)>\cdots>0$ and the corresponding
PSWFs $\{\psi_n^{(c)}\}_{n\ge 0}$ form an orthonormal basis of $\PW$
compatible with the spectral decomposition of $\Kop$
\cite{Slepian1978,BonamiKaroui}.
For each fixed $c$ and $N$, we define
\[
  V_{N,c}:=\operatorname{span}\{\psi_0^{(c)},\dots,\psi_{N-1}^{(c)}\}.
\]
This is an $N$-dimensional subspace for each fixed value of $c$.
There is no canonical identification of $V_{N,c}$ across different
values of $c$ unless one is explicitly chosen.

\subsection{Gram matrix and reference matrix}

\begin{definition}[Prolate Gram matrix and form]\label{def:GramForm}
For distinct points $\mathbf{p}=(p_1,\dots,p_N)\subset[-T,T]$, define
\[
  M(\mathbf{p},c)_{jn}:=\psi_n^{(c)}(p_j),
\]
and
\[
  \GVM := M(\mathbf{p},c)^*M(\mathbf{p},c).
\]
If $f=\sum_{n=0}^{N-1} a_n \psi_n^{(c)}\in V_{N,c}$, define
\[
  \QM(f):=\sum_{j=1}^N |f(p_j)|^2
  =\ip{\GVM \mathbf a}{\mathbf a},
\qquad \norm{f}_{L^2}^2=\norm{\mathbf a}_2^2.
\]
We write
$\AM:=\sqrt{\lambda_{\min}(\GVM)}$ and
$\BM:=\sqrt{\lambda_{\max}(\GVM)}$.
\end{definition}

\begin{definition}[Reference matrix and condition \textup{(T.2)}]\label{def:T2}
In the PSWF basis, define the diagonal matrix
\[
  G^{\mathrm{ref}}_{N,c}:=\operatorname{diag}(\lambda_0(c),\dots,\lambda_{N-1}(c)).
\]
This matrix is a \emph{basis-dependent reference matrix} induced by the
continuous quadratic form
\[
  f\mapsto \ip{\Kop f}{f}_{L^2}

after expanding $f$ in the PSWF basis.
It is not a canonical discrete analog of the Gram matrix; rather,
it is the natural comparison object once the PSWF basis has been fixed.
Set
\[
  \EN := \GVM - G^{\mathrm{ref}}_{N,c}.
\]
We say that $\mathbf p$ satisfies \textup{(T.2)} if
\[
  \norm{\EN}_2 \le \varepsilon_N
  \qquad\text{and}\qquad
  \varepsilon_N < \lambda_{N-1}(c).
\]
\end{definition}

\begin{remark}[Discrete versus continuous objects]\label{rem:identification}
The Gram matrix $\GVM$ comes from discrete point evaluation,
while $G^{\mathrm{ref}}_{N,c}$ comes from the continuous quadratic form
$f\mapsto \ip{\Kop f}{f}$ after choosing the PSWF basis.
Their comparison is therefore not canonical; it is mediated by the
choice of basis in $V_{N,c}$.
This basis dependence is harmless for the finite-dimensional statements
in Layer~B, but should be kept conceptually distinct from any intrinsic
operator-theoretic identification.
\end{remark}

\section{Layer A: operator core}\label{sec:layerA}

\subsection{Unisolvent sets}

\begin{definition}[Unisolvent set]\label{def:unisolvent}
A set $\mathbf p=(p_1,\dots,p_N)\subset[-T,T]$ of $N$ distinct points is
called \emph{unisolvent for $V_{N,c}$} if the only $f\in V_{N,c}$
with $f(p_j)=0$ for all $j=1,\dots,N$ is $f\equiv 0$.
Equivalently, the evaluation map
\[
  \mathcal E_{\mathbf p}:V_{N,c}\to \mathbb C^N,
  \qquad f\mapsto (f(p_1),\dots,f(p_N))
\]
is injective.
Since $\dim V_{N,c}=N$, injectivity is equivalent to bijectivity.
\end{definition}

\begin{assumption}[Unisolvence]\label{ass:A0}
Throughout Layer~A, the sampling set $\mathbf p$ is assumed to be
unisolvent for $V_{N,c}$.
\end{assumption}

\begin{lemma}[Gram positivity under unisolvence]\label{lem:B}
Under Assumption~\ref{ass:A0}, the matrix $\GVM$ is Hermitian positive
definite and
\[
  \lambda_{\min}(\GVM)\norm{f}_{L^2}^2
  \le \QM(f)
  \le \lambda_{\max}(\GVM)\norm{f}_{L^2}^2
\]
for all $f\in V_{N,c}$.
\end{lemma}

\begin{proof}
If $f=\sum a_n\psi_n^{(c)}$, then
$\QM(f)=\ip{\GVM\mathbf a}{\mathbf a}=\norm{M(\mathbf p,c)\mathbf a}_2^2$.
Unisolvence means $M(\mathbf p,c)$ has trivial kernel, hence full rank,
so $\GVM=M^*M$ is positive definite. The bounds follow from the min-max
principle for Hermitian matrices.
\end{proof}

\begin{lemma}[Pointwise upper bound]\label{lem:upper}
For every $f\in V_{N,c}$ and every $x\in\mathbb R$,
\[
  |f(x)|^2 \le K_{\PW}(x,x)\norm{f}_{L^2}^2
  = (\omega/\pi)\norm{f}_{L^2}^2.
\]
\end{lemma}

\begin{proof}
By the reproducing-kernel property,
$f(x)=\ip{f}{K_x}_{L^2}$. Hence by Cauchy--Schwarz,
$|f(x)|^2\le \norm{f}_{L^2}^2\norm{K_x}_{L^2}^2$.
Since $\norm{K_x}_{L^2}^2=K_{\PW}(x,x)=\omega/\pi$,
the claim follows.
\end{proof}

\begin{theorem}[Operator core]\label{thm:layerA}
Under Assumption~\ref{ass:A0}:
\begin{enumerate}[(I)]
\item \textbf{Coercivity.}
For all $f\in V_{N,c}$,
\[
  \QM(f) \ge \AM^2\norm{f}_{L^2}^2.
\]
\item \textbf{Continuity.}
For all $f\in V_{N,c}$,
\[
  \QM(f) \le N(\omega/\pi)\norm{f}_{L^2}^2.
\]
\end{enumerate}
\end{theorem}

\begin{proof}
Part~(I) is Lemma~\ref{lem:B}.
For part~(II), sum the pointwise estimate of Lemma~\ref{lem:upper}
over $j=1,\dots,N$.
\end{proof}

\section{Layer B: conditional sampling theory}\label{sec:layerB}

All results in this section are conditional on \textup{(T.2)}.

\begin{lemma}[Spectral stability via Weyl]\label{lem:C}
Assume \textup{(T.2)}. Then
\[
  \lambda_{\min}(\GVM)
  \ge \lambda_{N-1}(c)-\norm{\EN}_2
  \ge \lambda_{N-1}(c)-\varepsilon_N
  >0.
\]
In particular,
\[
  \AM \ge \sqrt{\lambda_{N-1}(c)-\varepsilon_N}.
\]
\end{lemma}

\begin{proof}
This is Weyl's inequality for Hermitian perturbations
\cite[Thm.~4.3.1]{HornJohnson2013}.
\end{proof}

\begin{theorem}[Spectral decomposition with remainder]\label{thm:layerB}
Assume \textup{(T.2)}. For every
$f=\sum_{n=0}^{N-1} a_n \psi_n^{(c)}\in V_{N,c}$,
\[
  \QM(f)=\ip{\Kop f}{f}_{L^2}+R_N(f),
  \qquad |R_N(f)|\le \varepsilon_N\norm{f}_{L^2}^2.
\]
\end{theorem}

\begin{proof}
Because we are working in the PSWF eigenbasis,
\[
  \ip{\Kop f}{f}_{L^2}
  =\sum_{n=0}^{N-1}\lambda_n(c)|a_n|^2
  =\ip{G^{\mathrm{ref}}_{N,c}\mathbf a}{\mathbf a}.
\]
Hence
\[
  \QM(f)=\ip{\GVM\mathbf a}{\mathbf a}
  =\ip{G^{\mathrm{ref}}_{N,c}\mathbf a}{\mathbf a}
   +\ip{\EN\mathbf a}{\mathbf a},
\]
so the remainder is
$R_N(f)=\ip{\EN\mathbf a}{\mathbf a}$ and satisfies
$|R_N(f)|\le \norm{\EN}_2\norm{\mathbf a}_2^2
\le \varepsilon_N\norm{f}_{L^2}^2$.
\end{proof}

\section{Layer C: heuristics for prime sampling}\label{sec:layerC}

This section is explicitly heuristic and not part of the main theorem.
Prime sampling is included only as motivation for later work.

\begin{heuristic}[Prime sampling stability]
Let $\mathbf p=(p_1,\dots,p_N)$ be prime points with $p_N\le T$.
One may hope that structured cancellation, together with PSWF decay,
yields
\[
  \norm{\EN}_2 = O(e^{-\delta' N})
\]
in a suitable regime. No proof of this claim is given here.
\end{heuristic}

\begin{remark}[Status of asymptotics]
The $c\to\infty$ behavior of PSWFs is subtle and basis-dependent.
No rigorous asymptotic lemma comparing $\GVM$ to the identity matrix is
used anywhere in the main theorems of this paper.
Any such claim belongs to later work and must be formulated only after
an explicit identification of the varying spaces $V_{N,c}$ has been fixed.
\end{remark}

\section{Open problems}\label{sec:open}

\begin{problem}[Unisolvence]
Characterize unisolvent configurations for $V_{N,c}$.
In particular, determine whether prime sampling is unisolvent.
\end{problem}

\begin{problem}[Sampling stability]
Prove or disprove condition \textup{(T.2)} for structured sets,
including prime sampling.
\end{problem}

\begin{problem}[Coupled limits]
Develop a rigorous treatment of Regimes~II and~III, where $N$ and $c$
are no longer fixed and norm equivalence arguments cease to be trivial.
\end{problem}
