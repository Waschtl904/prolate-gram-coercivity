\begin{abstract}
Let $\PW$ denote the Paley--Wiener space of $\omega$-bandlimited
functions and $V_N\subset\PW$ the $N$-dimensional prolate subspace
with time-bandwidth product $c=\omega T$.
For $N$ distinct sampling points $\mathbf{p}\subset[-T,T]$
the \emph{prolate Gram form} is
$\QM(f)=\sum_{j=1}^N|f(p_j)|^2$, $f\in V_N$.

We prove, in three logical layers of increasing conditionality:

\textbf{Layer~A (unconditional).}
Assuming only that the evaluation map
$\mathcal{E}_{\mathbf{p}}:V_N\to\mathbb{C}^N$, $f\mapsto(f(p_j))$,
is injective (Assumption~\ref{ass:A0}),
$\QM$ is coercive:
$\QM(f)\geq\AM^2\norm{f}_{L^2}^2$ for all $f\in V_N$,
together with a matching continuity upper bound.
The limit $\QM(f)\to\norm{f}_{L^2}^2$ as $c\to\infty$ holds
for fixed $N$ and $\mathbf{p}$.

\textbf{Layer~B (conditional on (T.2)).}
Under the abstract sampling-quality condition (T.2)
--- which requires $\norm{\EN}_2$ to be small relative to
$\lambda_{N-1}(c)$ --- we obtain explicit spectral stability and
a spectral decomposition $\QM(f)=\ip{\Kop f}{f}+R_N(f)$
with $|R_N(f)|\leq\varepsilon_N\norm{f}^2$.

\textbf{Layer~C (heuristic).}
For prime sampling points, we sketch why (T.2) is plausible
and why an exponential rate $\varepsilon_N=O(e^{-\delta'N})$
may hold. These claims are \emph{not} part of the main theorem.

No claim is made regarding zeros of $\zeta(s)$
or the passage $N\to\infty$.
Prime sampling is introduced for motivational reasons only.
\end{abstract}

\section{Introduction}
%%% =================================================================

The \emph{Weil positivity criterion} \cite{Weil1952,Bombieri2000,CC2019}
states that the Riemann Hypothesis is equivalent to positivity of
\[
  W(h)\;=\;
  \sum_{\substack{p\,\text{prime}\\k\geq1}}
  \log p\cdot\bigl|\widehat{h}(\log p^k)\bigr|^2
  +\text{(archimedean terms)}
\]
for all admissible test functions $h$.
The program of Connes--Consani--Moscovici \cite{CCM_PNAS2022,CCM2025}
seeks a spectral realization of this form via the prolate operator
$\Kop$ and its eigenstructure in $\PW$.

The present paper contributes a rigorous finite-dimensional building
block: coercivity of the prolate Gram form $\QM$ on $V_N$.
We structure the paper into four logical layers to make transparent
exactly which conclusions are unconditional, which are conditional,
and which remain heuristic.

\medskip\noindent\textbf{Four-layer architecture.}
\begin{enumerate}[(A)]
\item \emph{Operator core (Layer~A, \S\ref{sec:layerA}).}
Coercivity and continuity from finite-dimensional linear algebra;
requires only Assumption~\ref{ass:A0} (evaluation injectivity).
Fully rigorous.

\item \emph{Conditional sampling theory (Layer~B, \S\ref{sec:layerB}).}
Spectral stability and operator decomposition under the abstract
sampling-quality condition~(T.2). Rigorous given (T.2).

\item \emph{Heuristics (Layer~C, \S\ref{sec:layerC}).}
Motivation for why prime sampling satisfies (T.2) and why
exponential rates may hold. Explicitly not proven.

\item \emph{Open problems (Layer~D, \S\ref{sec:open}).}
Rigorous verification of (T.2), exponential rates, and the
large-$N$ limit.
\end{enumerate}

\medskip\noindent\textbf{What this paper does not claim.}
No statement is made about zeros of $\zeta(s)$.
No claim is made that prime sampling satisfies (T.2).
The operator-theoretic limit $N\to\infty$ is the open problem
of \cite{CCM2025}, addressed in Paper~II.

\section{Setup and Notation}\label{sec:setup}
%%% =================================================================

We use the engineering Fourier convention
$\widehat{f}(\xi)=\int_{\mathbb{R}}f(x)e^{-2\pi ix\xi}dx$
\cite{BonamiKaroui}.
Throughout: $N\geq1$, $\omega>0$, $T>0$, $c=\omega T$.

The \emph{Paley--Wiener space} $\PW$ consists of all
$f\in L^2(\mathbb{R})$ with $\operatorname{supp}\widehat{f}\subseteq
[-\omega/(2\pi),\omega/(2\pi)]$;
reproducing kernel $K_{\PW}(x,x)=\omega/\pi$ \cite[Eq.~(3)]{BonamiKaroui}.

The \emph{prolate concentration operator}
\[
  \Kop f(x)\;=\;
  \int_{-T}^{T}\frac{\sin\omega(x-y)}{\pi(x-y)}\,f(y)\,dy,
  \qquad \Kop:\PW\to\PW,
\]
has eigenvalues $1>\lambda_0(c)>\lambda_1(c)>\cdots>0$ \cite{Slepian1978},
eigenfunctions PSWFs $\{\psin\}_{n\geq0}$.
Set $V_N=\operatorname{span}\{\psi_0^{(c)},\ldots,\psi_{N-1}^{(c)}\}$.

The \emph{Kulikov constant}
$\Kul:=\sup_{n,c}(1-\lam)^{1/2}\norm{\psin}_{L^\infty}<\infty$
\cite[Thm.~1.1]{Kulikov2023}.

\begin{definition}[Prolate Gram matrix and form]\label{def:GramForm}
For $N$ distinct $\mathbf{p}=(p_1,\ldots,p_N)\subset[-T,T]$:
evaluation matrix $M(\mathbf{p})_{jn}=\psi_n^{(c)}(p_j)$;
prolate Gram matrix $\GVM=M(\mathbf{p})^*M(\mathbf{p})$;
prolate Gram form $\QM(f)=\sum_{j=1}^N|f(p_j)|^2$ for $f\in V_N$.
Write $\AM=\sqrt{\lambda_{\min}(\GVM)}$,
$\BM=\sqrt{\lambda_{\max}(\GVM)}$.
\end{definition}

\begin{definition}[Gram splitting and condition~(T.2)]\label{def:T2}
Define the \emph{diagonal reference matrix}
$(G_c^{(N)})_{mn}:=\lambda_n(c)\delta_{mn}$
and the \emph{error (discretization) matrix} $\EN:=\GVM-G_c^{(N)}$.

\medskip\noindent\textbf{Interpretation.}
Since $\int_{-T}^T\psi_m^{(c)}\overline{\psi_n^{(c)}}dx
=\lambda_n(c)\delta_{mn}$,
the matrix $G_c^{(N)}$ is the \emph{continuous analog} of $\GVM$
(replace discrete sum by integral). Thus $\EN$ is a
\emph{discretization error matrix}; no further quantitative claim
about $\norm{\EN}$ is made in Layer~A.

\medskip\noindent\textbf{(T.2) Sampling-quality condition.}
$\mathbf{p}$ \emph{satisfies~(T.2)} with $\varepsilon_N\geq0$ if
\[
  \norm{\EN}_2\;\leq\;\varepsilon_N
  \quad\text{and}\quad
  \varepsilon_N\;<\;\lambda_{N-1}(c).
\]
\end{definition}

\begin{remark}[Status of (T.2)]\label{rem:T2}
Condition~(T.2) is an \emph{abstract hypothesis} on the sampling set
$\mathbf{p}$. Whether specific families of points (e.g.\ primes)
satisfy (T.2) is \emph{not proved} in this paper; see
Problem~\ref{prob:T2} and the heuristic discussion in
\S\ref{sec:layerC}.
\end{remark}

\section{Layer~A: Operator Core (Unconditional)}\label{sec:layerA}
%%% =================================================================

\begin{assumption}[Evaluation injectivity]\label{ass:A0}
The evaluation map
\[
  \mathcal{E}_{\mathbf{p}}:V_N\to\mathbb{C}^N,
  \qquad f\mapsto(f(p_1),\ldots,f(p_N)),
\]
is injective, i.e.\ the only $f\in V_N$ with $f(p_j)=0$
for all $j=1,\ldots,N$ is $f\equiv0$.
\end{assumption}

\begin{remark}[Discussion of Assumption~\ref{ass:A0}]\label{rem:A0}
Assumption~\ref{ass:A0} holds whenever the $N$ points are
\emph{generic} (no special resonance condition). For PSWFs,
it is related to the Chebyshev-system / zero-counting properties
of $V_N$ \cite{KarlinStudden1966,Slepian1978}, but a complete
unconditional proof for all $N$ distinct points is not available
in the present paper. Rigorously, Assumption~\ref{ass:A0} should
be treated as an explicit hypothesis; its verification for
concrete families is Problem~\ref{prob:A0}.
\end{remark}

\begin{lemma}[Gram positivity under~\ref{ass:A0}]\label{lem:B}
Under Assumption~\ref{ass:A0}:
\begin{enumerate}[(i)]
\item $\operatorname{rank}M(\mathbf{p})=N$.
\item $\GVM=M(\mathbf{p})^*M(\mathbf{p})$ is Hermitian positive definite;
$\lambda_{\min}(\GVM)>0$.
\item Frame property: $\lambda_{\min}(\GVM)\norm{f}^2\leq\QM(f)
\leq\lambda_{\max}(\GVM)\norm{f}^2$ for all $f\in V_N$.
\end{enumerate}
\end{lemma}

\begin{proof}
(i) $\mathcal{E}_{\mathbf{p}}$ injective $\Leftrightarrow$ $\ker M=\{0\}$
$\Leftrightarrow$ $\operatorname{rank}M=N$.
(ii) $\GVM=M^*M$ and $\ker M=\{0\}$ give $\GVM\succ0$.
(iii) Standard: $\ip{\GVM\mathbf{a}}{\mathbf{a}}=\norm{M\mathbf{a}}^2$.
\end{proof}

\begin{lemma}[Pointwise upper bound]\label{lem:upper}
For any $f\in V_N$:
$|f(x)|^2\leq(\omega/\pi)\norm{f}_{L^2}^2$
for all $x\in\mathbb{R}$,
by the reproducing-kernel inequality $K_{\PW}(x,x)=\omega/\pi$
\cite[Eq.~(3)]{BonamiKaroui}.
\end{lemma}

\begin{lemma}[Spectral limit]\label{lem:limit}
For fixed $N$ and $\mathbf{p}$: $\lambda_n(c)\to1$ for all
$n\leq N-1$ as $c\to\infty$ \cite{Slepian1978},
so $G_c^{(N)}\to\mathrm{Id}_N$.
Since $\GVM=G_c^{(N)}+\EN$ and
$\norm{\EN}_2\to0$ (the pointwise values $|\psin(p_j)|^2$
converge to $\lambda_n(c)\to1$ as $c\to\infty$),
we obtain $\GVM\to\mathrm{Id}_N$ in operator norm.
\end{lemma}

\begin{theorem}[Operator Core]\label{thm:layerA}
Under Assumption~\ref{ass:A0}:
\begin{enumerate}[(I)]
\item \textbf{Coercivity (unconditional).}
$\QM(f)\geq\AM^2\norm{f}_{L^2}^2$ for all $f\in V_N$.

\item \textbf{Continuity (unconditional).}
$\QM(f)\leq N(\omega/\pi)\norm{f}_{L^2}^2$
for all $f\in V_N$.

\item \textbf{Limit.}
For fixed $N$ and $\mathbf{p}$, as $c\to\infty$:
$\QM(f)\to\norm{f}_{L^2}^2$ uniformly in
$f\in V_N$, $\norm{f}_{L^2}=1$.
\end{enumerate}
\end{theorem}

\begin{proof}
(I) By Lemma~\ref{lem:B}(iii), $\QM(f)\geq\lambda_{\min}(\GVM)\norm{f}^2=\AM^2\norm{f}^2$.
(II) By Lemma~\ref{lem:upper}: $\QM(f)=\sum_j|f(p_j)|^2\leq N(\omega/\pi)\norm{f}^2$.
(III) By Lemma~\ref{lem:limit}, $\GVM\to\mathrm{Id}_N$, so
$\QM(f)=\ip{\GVM\mathbf{a}}{\mathbf{a}}\to\norm{\mathbf{a}}^2=\norm{f}_{L^2}^2$
uniformly for $\norm{f}=1$.
\end{proof}

\section{Layer~B: Conditional Sampling Theory}\label{sec:layerB}
%%% =================================================================

All results in this section are \emph{conditional on (T.2)}.
Condition~(T.2) is an abstract hypothesis on $\mathbf{p}$;
no specific family of points is claimed to satisfy it here.

\begin{lemma}[Spectral stability via Weyl]\label{lem:C}
Assume (T.2) with parameter $\varepsilon_N$.
\begin{enumerate}[(i)]
\item By Weyl's inequality \cite[Thm.~4.3.1]{HornJohnson2013}:
\[
  \lambda_{\min}(\GVM)
  \;\geq\;
  \lambda_{N-1}(c)-\norm{\EN}_2
  \;\geq\;
  \lambda_{N-1}(c)-\varepsilon_N
  \;>\;0.
\]
\item In particular, $\AM\geq\AM^{\mathrm{stab}}:=
\sqrt{\lambda_{N-1}(c)-\varepsilon_N}>0$.
\end{enumerate}
\end{lemma}

\begin{theorem}[Spectral decomposition with remainder]\label{thm:layerB}
Assume (T.2) with parameter $\varepsilon_N<\lambda_{N-1}(c)$.
For every $f=\sum_na_n\psin\in V_N$:
\[
  \QM(f)\;=\;\ip{\Kop f}{f}_{L^2}+R_N(f),
  \qquad
  |R_N(f)|\;\leq\;\varepsilon_N\norm{f}_{L^2}^2.
\]
\end{theorem}

\begin{proof}
By the eigenvalue equation:
$\ip{\Kop f}{f}_{L^2}=\sum_n\lambda_n(c)|a_n|^2
=\ip{G_c^{(N)}\mathbf{a}}{\mathbf{a}}$.
Hence
$\QM(f)=\ip{\GVM\mathbf{a}}{\mathbf{a}}
=\ip{G_c^{(N)}\mathbf{a}}{\mathbf{a}}+\ip{\EN\mathbf{a}}{\mathbf{a}}
=\ip{\Kop f}{f}+R_N(f)$,
where $|R_N(f)|=|\ip{\EN\mathbf{a}}{\mathbf{a}}|
\leq\norm{\EN}_2\norm{\mathbf{a}}^2\leq\varepsilon_N\norm{f}^2$.
\end{proof}

\begin{remark}[Layer~B summary]
Theorem~\ref{thm:layerB} is logically clean:
given the abstract hypothesis (T.2), the spectral decomposition
and stability bound are fully rigorous.
The quality of the result depends entirely on how small $\varepsilon_N$ is,
which is a question about the sampling set $\mathbf{p}$ --- addressed
heuristically in Layer~C and rigorously in Layer~D.
\end{remark}

\section{Layer~C: Heuristics for Prime Sampling}\label{sec:layerC}
%%% =================================================================

\textbf{This section is explicitly heuristic.}
No result here is part of the main theorems (Theorems~\ref{thm:layerA}
and~\ref{thm:layerB}). The goal is to motivate why prime sampling
is natural in this context and to outline what a rigorous proof of
(T.2) for primes would require.

\subsection{Motivation: prime sampling and the Weil formula}

The Weil explicit formula involves the sum
$\sum_p\log p\cdot|\widehat{h}(\log p)|^2$ over primes.
Prime sampling $\mathbf{p}=(p_1,\ldots,p_N)$ is the natural
finite-dimensional approximation. Coercivity of $\QM$ at prime
points is thus the finite-dimensional analog of Weil positivity.
No formal connection between Gram form coercivity and Weil positivity
is established here.

\subsection{Heuristic: why (T.2) might hold for primes}

\begin{heuristic}[Prime sampling stability]\label{heur:primes}
Let $p_1<\cdots<p_N$ be primes with $p_N\leq T$ and
$N\leq(2/\pi-\varepsilon)c$ (Kulikov--Slepian regime).
We expect
\[
  \norm{\EN}_2\;=\;O\bigl(N\Kul^2 e^{-2\delta c}\bigr)
  \;=\;O(e^{-\delta'N})
\]
for some $\delta'>0$.
Consequently (T.2) should hold with an exponentially small parameter.
\end{heuristic}

\textbf{Sketch of the argument (not a proof).}
The diagonal entries $({\EN})_{nn}$ equal
$\sum_j|\psin(p_j)|^2 - \lambda_n(c)$,
which is a quadrature error for $|\psin|^2$.
Using Bernstein's inequality and prime-gap estimates
($|I_j|\lesssim\log^2 p_j$ for Voronoi cells $I_j$),
one heuristically obtains a polynomial factor from the gaps and
an exponential decay from $1-\lambda_n(c)\leq Ce^{-\eta c}$
(Landau--Widom).
\emph{However, the key step --- that the exponential decay
$e^{-\eta c}$ overcomes the polynomial growth from prime gaps ---
is not rigorously established}, because:
\begin{enumerate}[(a)]
\item The transition from the naive bound
$\norm{\EN}_F = O(Ne^{2\delta c})$
to the improved bound $O(Ne^{-2\delta c})$
requires a cancellation mechanism that is currently not proven.
\item The prime-gap estimate $|I_j|\leq C\log^2 p_j$
follows from Bertrand's postulate for $j\geq2$,
but the implied constant in the quadrature bound
$\sum_j|I_j|^2\norm{(|\psin|^2)'}_{L^\infty}$
needs careful treatment.
\item No Marcinkiewicz--Zygmund inequality for PSWFs at prime
nodes has been established.
\end{enumerate}
A rigorous proof would require one of the approaches in Problem~\ref{prob:T2}.

\subsection{What a rigorous proof of (T.2) for primes would need}

\begin{enumerate}[(i)]
\item \emph{Approach A (sampling theory).}
Establish a Marcinkiewicz--Zygmund theorem for $V_N$ at prime nodes:
find weights $w_j>0$ such that
$\sum_j w_j|f(p_j)|^2\sim\norm{f}_{L^2[-T,T]}^2$
uniformly for $f\in V_N$.
This would give a weighted version of (T.2) directly.

\item \emph{Approach B (PSWF-specific estimates).}
Derive uniform quadrature estimates for $|\psin|^2$ at prime nodes,
using the specific oscillation properties of PSWFs near the
transition region $n\approx(2/\pi)c$.

\item \emph{Approach C (large-$c$ perturbation).}
Use the fact that for large $c$ and fixed $N$,
$\GVM\to\mathrm{Id}_N$ (Lemma~\ref{lem:limit})
and therefore (T.2) holds trivially for large enough $c$.
This gives (T.2) for fixed $N$ but does not control the rate
or the coupled limit $N,c\to\infty$.
\end{enumerate}

\section{Open Problems}\label{sec:open}
%%% =================================================================

\begin{problem}[Injectivity of evaluation map]\label{prob:A0}
Prove unconditionally (for all $N\geq1$ and all $N$ pairwise
distinct $\mathbf{p}\subset[-T,T]$) that
$\mathcal{E}_{\mathbf{p}}:V_N\to\mathbb{C}^N$ is injective.
Is this a consequence of known Chebyshev-system / oscillation
theory for PSWFs?
\end{problem}

\begin{problem}[Condition (T.2) for prime sampling]\label{prob:T2}
Prove rigorously that prime sampling satisfies (T.2) with a
quantitative bound on $\varepsilon_N$.
Does a Marcinkiewicz--Zygmund inequality hold for PSWFs at prime nodes?
What is the optimal rate of $\varepsilon_N\to0$?
\end{problem}

\begin{problem}[Exponential rate]\label{prob:exp}
Establish rigorously a bound of the form
$\norm{\EN}_2\leq C(N,\varepsilon)e^{-\delta c}$
for prime (or any structured) sampling, identifying the
structural cancellation mechanism.
\end{problem}

\begin{problem}[Large-dimensional limit]\label{prob:largeN}
Determine the behavior of $\lambda_{\min}(\GVM)$ and $\QM$
in the coupled limit $N,c\to\infty$ with $N/c\to\rho\in(0,2/\pi)$.
Does the normalized Gram form detect spectral information
about $\zeta(s)$? Addressed in Paper~II.
\end{problem}

\begin{problem}[Characterization of (T.2) sets]\label{prob:T2char}
Characterize all $\mathbf{p}\subset[-T,T]$ of cardinality $N$
satisfying (T.2) with $\varepsilon_N\to0$.
Is Beurling density sufficient?
\end{problem}
