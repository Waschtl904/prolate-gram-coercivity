\begin{abstract}
Let $\PW$ be the Paley--Wiener space of $\omega$-bandlimited functions
and $V_{N,c}:=\operatorname{span}\{\psi_0^{(c)},\dots,\psi_{N-1}^{(c)}\}$
the $N$-dimensional prolate subspace at time-bandwidth product $c=\omega T$.
For $N$ distinct sampling points $\mathbf{p}\subset[-T,T]$ we study
the prolate Gram form $\QM(f)=\sum_{j=1}^N|f(p_j)|^2$ on $V_{N,c}$.

The nontrivial content of this paper is two-fold.
First, we prove unisolvence for random configurations and for weighted sampling models
satisfying a Marcinkiewicz--Zygmund-type lower frame bound
(Propositions~\ref{prop:random} and~\ref{prop:mz_unisolvent}).
For uniform grids we explain the associated zero-counting problem,
but we do not claim a theorem because $V_{N,c}$ is not known to be
a Chebyshev system in general.
Second, under an abstract sampling-quality condition~(T.2),
we prove an explicit quantitative comparison between discrete sampling and the prolate operator,
namely
$\QM(f)=\ip{\Kop f}{f}_{L^2}+R_N(f)$ with $|R_N(f)|\le\varepsilon_N\norm{f}^2$,
where the remainder measures how well discrete sampling approximates
the continuous quadratic form of~$\Kop$.
For weighted sampling models we further isolate the discrete operator error into a weighted-frame defect
and a spectral mismatch term, thereby reducing the comparison problem to two explicit perturbative components.

No claim is made about zeros of $\zeta(s)$, prime sampling, or limits
in which both $N$ and $c$ vary.
\end{abstract}

\section{Introduction}

The Weil positivity criterion \cite{Weil1952,Bombieri2000,CC2019}
expresses the Riemann Hypothesis as positivity of an explicit
quadratic form over primes.
The prolate program of Connes--Consani--Moscovici
\cite{CCM_PNAS2022,CCM2025} seeks a spectral realization of this form
using the prolate concentration operator $\Kop$ on Paley--Wiener space.

This paper isolates a finite-dimensional coercivity principle that is
independent of any number-theoretic input.
The linear-algebraic core is elementary: coercivity is equivalent to
invertibility of the evaluation matrix $M(\mathbf{p},c)$.
The real content lies in two issues:
(i) identifying concrete families of sampling sets for which
unisolvence can actually be proved, and
(ii) quantifying the gap between the discrete sampling form and the
continuous quadratic form of $\Kop$.

Layer~A proves unisolvence for random sampling and for weighted sampling models
satisfying a lower frame inequality, and records the uniform-grid case as an open zero-counting problem.
Layer~B proves the spectral decomposition under condition~(T.2).
Layer~C contains heuristic discussion of prime sampling.

No statement is made about zeros of $\zeta(s)$.
The limit $N\to\infty$ and coupled limits in which $N$ and $c$ vary
are outside the scope of this paper.

\section{Setup and notation}\label{sec:setup}

\subsection{Basic objects}

We use the engineering Fourier convention
$\widehat{f}(\xi)=\int_{\mathbb{R}}f(x)e^{-2\pi ix\xi}\,dx$.

\begin{remark}[Fourier convention and bandwidth]
Under this convention the Paley--Wiener space of bandwidth $\omega$
corresponds to Fourier support in
$[-\omega/(2\pi),\omega/(2\pi)]$.
Many references use the convention
$\widehat f(\xi)=\int f(x)e^{-i\xi x}\,dx$,
for which the same space corresponds to $[-\omega,\omega]$.
\end{remark}

The \emph{Paley--Wiener space} $\PW$ consists of all
$f\in L^2(\mathbb{R})$ with Fourier support in
$[-\omega/(2\pi),\omega/(2\pi)]$.
It is a reproducing-kernel Hilbert space with kernel
$K_{\PW}(x,y)=\tfrac{\omega}{\pi}\operatorname{sinc}(\omega(x-y)/\pi)$
and $K_{\PW}(x,x)=\omega/\pi$ \cite[Eq.~(3)]{BonamiKaroui}.
Every $f\in\PW$ extends to an entire function of exponential type,
by the Paley--Wiener theorem.

The \emph{prolate concentration operator} is
\[
  \Kop f(x)=\int_{-T}^{T}\frac{\sin\omega(x-y)}{\pi(x-y)}f(y)\,dy,
  \qquad \Kop:\PW\to\PW.
\]
It is self-adjoint and compact on $\PW$, with simple eigenvalues
$1>\lambda_0(c)>\lambda_1(c)>\cdots>0$ and eigenfunctions
(PSWFs) $\{\psin\}_{n\ge 0}$ forming an orthonormal basis of $\PW$
\cite{Slepian1978,BonamiKaroui}.
Throughout, $c=\omega T$ and
\[
  V_{N,c}:=\operatorname{span}\{\psi_0^{(c)},\dots,\psi_{N-1}^{(c)}\}.
\]
This is an $N$-dimensional subspace of $\PW$ for each fixed pair $(N,c)$.
There is no canonical identification of $V_{N,c}$ across different values
of $c$.

\subsection{Gram matrix and reference matrix}

\begin{definition}[Prolate Gram matrix and form]\label{def:GramForm}
For distinct $\mathbf{p}=(p_1,\dots,p_N)\subset[-T,T]$, the
\emph{evaluation matrix} is $M(\mathbf{p},c)_{jn}:=\psi_n^{(c)}(p_j)$,
and the \emph{prolate Gram matrix} is
$\GVM:=M(\mathbf{p},c)^*M(\mathbf{p},c)$.

If $f=\sum_{n=0}^{N-1}a_n\psin\in V_{N,c}$, then
$\norm{f}_{L^2}^2=\norm{\mathbf{a}}_2^2$ and
$\QM(f):=\sum_{j=1}^N|f(p_j)|^2=\ip{\GVM\mathbf{a}}{\mathbf{a}}$.
Write $\AM:=\sqrt{\lambda_{\min}(\GVM)}$ and
$\BM:=\sqrt{\lambda_{\max}(\GVM)}$.
\end{definition}

\begin{definition}[Reference matrix and condition~(T.2)]\label{def:T2}
Define the diagonal matrix
\[
  G^{\mathrm{ref}}_{N,c}:=\operatorname{diag}(\lambda_0(c),\dots,\lambda_{N-1}(c)).
\]
This is the matrix of $\Kop|_{V_{N,c}}$ in the PSWF eigenbasis:
for $f=\sum_n a_n\psin$,
\[
  \ip{\Kop f}{f}_{L^2}
  =\sum_n\lambda_n(c)|a_n|^2
  =\ip{G^{\mathrm{ref}}_{N,c}\mathbf{a}}{\mathbf{a}}.
\]
Set $\EN:=\GVM-G^{\mathrm{ref}}_{N,c}$.
We say that $\mathbf{p}$ satisfies~(T.2) if
\[
  \norm{\EN}_2\le\varepsilon_N
  \qquad\text{and}\qquad
  \varepsilon_N<\lambda_{N-1}(c).
\]
\end{definition}

\begin{remark}[Coefficient-space identification]\label{rem:coeff_ident}
The matrices $\GVM$ and $G^{\mathrm{ref}}_{N,c}$ are both viewed as operators on
$\mathbb C^N$ through the coefficient map
$f=\sum_{n=0}^{N-1}a_n\psi_n^{(c)}\longleftrightarrow \mathbf a=(a_0,\dots,a_{N-1})$.
Under this identification, $\GVM$ encodes the discrete sampling quadratic form
and $G^{\mathrm{ref}}_{N,c}$ encodes the restriction of $\Kop$ to $V_{N,c}$.
The notation \textup{(T.2)} is chosen to remain compatible with the numbering
used later in the scaling-limit continuation of this project.
\end{remark}

\section{Layer~A: Unisolvence and operator core}\label{sec:layerA}

\begin{definition}[Unisolvent set]\label{def:unisolvent}
A set $\mathbf{p}=(p_1,\dots,p_N)\subset[-T,T]$ of $N$ distinct points
is \emph{unisolvent for $V_{N,c}$} if the only $f\in V_{N,c}$ satisfying
$f(p_j)=0$ for all $j$ is $f\equiv0$.
Since $\dim V_{N,c}=N$, this is equivalent to bijectivity of the evaluation map
$\mathcal{E}_{\mathbf{p}}:V_{N,c}\to\mathbb{C}^N$.
\end{definition}

\begin{definition}[Weighted frame model on $V_{N,c}$]\label{def:mz_model}
A family of nodes and positive weights
\[
  \{(p_j,w_j)\}_{j=1}^N, \qquad -T\le p_1<\cdots<p_N\le T, \qquad w_j>0,
\]
is said to satisfy a \emph{Marcinkiewicz--Zygmund frame condition} on $V_{N,c}$
with defect $\delta\in[0,1)$ if
\[
  (1-\delta)\norm{f}_{L^2(-T,T)}^2
  \le \sum_{j=1}^N w_j |f(p_j)|^2
  \le (1+\delta)\norm{f}_{L^2(-T,T)}^2
\]
for all $f\in V_{N,c}$.
The case $\delta=0$ corresponds to exact weighted discretization of the
$L^2(-T,T)$ geometry of $V_{N,c}$.
\end{definition}

\begin{remark}[Weighted-frame viewpoint]
Definition~\ref{def:mz_model} treats discrete sampling on $V_{N,c}$ as a weighted frame problem.
The geometry is therefore encoded not by an exact quadrature oracle but by a stability defect $\delta$.
Gauss--PSWF constructions should be regarded as motivation for such models, not as an automatic source of exact identities.
\end{remark}

\begin{remark}[Why unisolvence is non-trivial for $V_{N,c}$]
For a Chebyshev system, every $N$-tuple of distinct points is unisolvent
\cite[Ch.~1]{KarlinStudden1966}.
The space $V_{N,c}$ is not known to be a Chebyshev system in general.
Hence unisolvence must be proved separately for specific families.
In particular, the natural zero-counting route for uniform grids remains open.
\end{remark}

\begin{lemma}[Identity principle for Paley--Wiener functions]\label{lem:identity}
If $f\in\PW$ vanishes on a nonempty interval, then $f\equiv 0$ on
$\mathbb R$.
\end{lemma}

\begin{proof}
Every $f\in\PW$ extends to an entire function of exponential type.
If it vanishes on an interval, it has an accumulation set of zeros.
By the identity theorem for holomorphic functions, the entire extension
must vanish identically, hence $f\equiv 0$ on $\mathbb R$.
\end{proof}

\begin{proposition}[Weighted MZ lower frame implies unisolvence]\label{prop:mz_unisolvent}
Assume that $\{(p_j,w_j)\}_{j=1}^N$ satisfies the lower frame bound
\[
  A\norm{f}_{L^2(-T,T)}^2\le \sum_{j=1}^N w_j |f(p_j)|^2
\]
for all $f\in V_{N,c}$, with some constant $A>0$.
Then $\mathbf p=(p_1,\dots,p_N)$ is unisolvent for $V_{N,c}$.
In particular, every Marcinkiewicz--Zygmund frame condition with defect $\delta<1$
implies unisolvence.
\end{proposition}

\begin{proof}
If $f\in V_{N,c}$ satisfies $f(p_j)=0$ for all $j$, then
\[
  0=\sum_{j=1}^N w_j|f(p_j)|^2\ge A\norm{f}_{L^2(-T,T)}^2.
\]
Hence $f=0$ almost everywhere on $[-T,T]$, thus everywhere on that interval
by continuity, and therefore $f\equiv0$ on $\mathbb R$
by Lemma~\ref{lem:identity}.
\end{proof}

\begin{proposition}[Random sampling is almost surely unisolvent]
\label{prop:random}
Let $p_1,\dots,p_N$ be i.i.d. with a distribution absolutely continuous
with respect to Lebesgue measure on $[-T,T]$.
Then $\mathbf{p}$ is unisolvent for $V_{N,c}$ almost surely.
\end{proposition}

\begin{proof}
Non-unisolvence means $\det M(\mathbf{p},c)=0$.
Since each PSWF $\psi_n^{(c)}$ is a real-analytic function, the map
$(p_1,\dots,p_N)\mapsto\det M(\mathbf{p},c)$ is real analytic.
It is not identically zero because the family of point-evaluation functionals on
$V_{N,c}$ is separating, hence one can choose $N$ points giving an invertible evaluation map.
The zero set of a nonzero real-analytic function has Lebesgue measure zero,
so absolute continuity implies that non-unisolvence occurs with probability zero.
\end{proof}

\begin{lemma}[Pointwise bound via reproducing kernel]\label{lem:upper}
For every $f\in\PW$ and every $x\in\mathbb R$,
\[
  |f(x)|^2
  =|\ip{f}{K_x}_{L^2}|^2
  \le \norm{f}_{L^2}^2\,\norm{K_x}_{L^2}^2
  = K_{\PW}(x,x)\,\norm{f}_{L^2}^2
  = \frac{\omega}{\pi}\,\norm{f}_{L^2}^2.
\]
\end{lemma}

\begin{theorem}[Operator core]\label{thm:layerA}
Let $(N,c)$ be fixed and let $\mathbf{p}$ be unisolvent for $V_{N,c}$.
Then:
\begin{enumerate}[(I)]
\item $\QM(f)\ge \AM^2\norm{f}_{L^2}^2$ for all $f\in V_{N,c}$.
\item $\QM(f)\le N(\omega/\pi)\norm{f}_{L^2}^2$ for all $f\in V_{N,c}$.
\end{enumerate}
In particular, these bounds hold for every weighted frame model satisfying the hypotheses of
Proposition~\ref{prop:mz_unisolvent}, and by Proposition~\ref{prop:random} for random sampling almost surely.
\end{theorem}

\begin{proof}
Unisolvence is equivalent to $\ker M(\mathbf{p},c)=\{0\}$,
hence to positive definiteness of $\GVM=M(\mathbf{p},c)^*M(\mathbf{p},c)$.
Therefore
\[
  \QM(f)=\ip{\GVM\mathbf a}{\mathbf a}
  \ge \lambda_{\min}(\GVM)\norm{\mathbf a}_2^2
  = \AM^2\norm{f}_{L^2}^2.
\]
The upper bound follows by summing Lemma~\ref{lem:upper} over the $N$ sample points.
\end{proof}

\begin{lemma}[Rayleigh-quotient form of the upper bound]\label{lem:lambda_max_bound}
For every set of distinct points $\mathbf p\subset[-T,T]$,
\[
  \lambda_{\max}(\GVM)=\sup_{\norm{\mathbf a}_2=1}\ip{\GVM\mathbf a}{\mathbf a}
  =\sup_{\substack{f\in V_{N,c}\\ \norm{f}_{L^2}=1}} \QM(f).
\]
Consequently, if $\QM(f)\le B\norm{f}_{L^2}^2$ for all $f\in V_{N,c}$,
then $\lambda_{\max}(\GVM)\le B$.
In particular,
\[
  \lambda_{\max}(\GVM)\le N\omega/\pi.
\]
\end{lemma}

\begin{proof}
Under the coefficient-space identification from Remark~\ref{rem:coeff_ident},
one has $\norm{f}_{L^2}=\norm{\mathbf a}_2$ and
$\QM(f)=\ip{\GVM\mathbf a}{\mathbf a}$.
The first identity is therefore just the Rayleigh-quotient characterization of the largest eigenvalue
of the Hermitian matrix $\GVM$.
The implication with constant $B$ is immediate.
Applying Theorem~\ref{thm:layerA}(II) gives the final bound.
\end{proof}

\section{Layer~B: Comparison Theorem}\label{sec:layerB}

All results here are conditional on~(T.2).

\begin{definition}[Weighted discrete reference matrix]\label{def:weighted_ref}
Given positive weights $w_1,\dots,w_N$, let $D:=\operatorname{diag}(w_1,\dots,w_N)$ and define
\[
  G^{\mathrm{w}}_{N,c}(\mathbf p,D):=M(\mathbf p,c)^* D\,M(\mathbf p,c).
\]
This is the weighted discrete Gram matrix associated with the sampling model $(\mathbf p,D)$.
\end{definition}

\begin{proposition}[Two-step error decomposition]\label{prop:two_step_error}
Let $(\mathbf p,D)$ be a weighted sampling model with positive diagonal weight matrix $D$.
Then
\[
  \EN
  =\GVM-G^{\mathrm{ref}}_{N,c}
  =M(\mathbf p,c)^*(I-D)M(\mathbf p,c)
   +\bigl(G^{\mathrm{w}}_{N,c}(\mathbf p,D)-G^{\mathrm{ref}}_{N,c}\bigr).
\]
Consequently,
\[
  \norm{\EN}_2
  \le \norm{I-D}_2\,\norm{M(\mathbf p,c)}_2^2
   + \norm{G^{\mathrm{w}}_{N,c}(\mathbf p,D)-G^{\mathrm{ref}}_{N,c}}_2.
\]
\end{proposition}

\begin{proof}
Insert and subtract the weighted discrete Gram matrix:
\[
  \EN
  =\GVM-G^{\mathrm{w}}_{N,c}(\mathbf p,D)
   +G^{\mathrm{w}}_{N,c}(\mathbf p,D)-G^{\mathrm{ref}}_{N,c}.
\]
Since
\[
  \GVM-G^{\mathrm{w}}_{N,c}(\mathbf p,D)
  =M(\mathbf p,c)^*(I-D)M(\mathbf p,c),
\]
the claimed identity follows. The norm bound is immediate from subadditivity and submultiplicativity.
\end{proof}

\begin{remark}[Interpretation of Proposition~\ref{prop:two_step_error}]
The error now splits into two independent effects: a weighted-frame defect
$M^*(I-D)M$ and a spectral mismatch term
$G^{\mathrm{w}}_{N,c}(\mathbf p,D)-G^{\mathrm{ref}}_{N,c}$.
Thus condition~\textup{(T.2)} is not reduced to pure weight flatness alone;
it also requires compatibility between the weighted discrete model and the prolate spectrum.
\end{remark}

\begin{corollary}[Sufficient criterion for \textup{(T.2)} from two defects]\label{cor:two_defects}
Assume that the weighted sampling model $(\mathbf p,D)$ satisfies
\[
  \norm{I-D}_2\le \delta_N,
  \qquad
  \norm{G^{\mathrm{w}}_{N,c}(\mathbf p,D)-G^{\mathrm{ref}}_{N,c}}_2\le \eta_N,
\]
with
\[
  \delta_N\,\frac{N\omega}{\pi}+\eta_N<\lambda_{N-1}(c).
\]
Then condition~\textup{(T.2)} holds with
\[
  \varepsilon_N:=\delta_N\,\frac{N\omega}{\pi}+\eta_N.
\]
\end{corollary}

\begin{proof}
By Proposition~\ref{prop:two_step_error},
\[
  \norm{\EN}_2
  \le \norm{I-D}_2\,\norm{M(\mathbf p,c)}_2^2
   +\norm{G^{\mathrm{w}}_{N,c}(\mathbf p,D)-G^{\mathrm{ref}}_{N,c}}_2.
\]
Using Lemma~\ref{lem:lambda_max_bound},
$\norm{M(\mathbf p,c)}_2^2=\lambda_{\max}(\GVM)\le N\omega/\pi$.
Therefore
\[
  \norm{\EN}_2\le \delta_N\,\frac{N\omega}{\pi}+\eta_N=\varepsilon_N.
\]
The displayed hypothesis gives $\varepsilon_N<\lambda_{N-1}(c)$, so Definition~\ref{def:T2} applies.
\end{proof}

\begin{lemma}[Spectral stability via Weyl]\label{lem:C}
Assume~(T.2) with parameter $\varepsilon_N$.
Then
\[
  \lambda_{\min}(\GVM)
  \ge \lambda_{\min}(G^{\mathrm{ref}}_{N,c})-\norm{\EN}_2
  = \lambda_{N-1}(c)-\norm{\EN}_2
  \ge \lambda_{N-1}(c)-\varepsilon_N
  >0,
\]
where the first inequality is the standard Hermitian perturbation bound
$\lambda_{\min}(A+E)\ge \lambda_{\min}(A)-\norm{E}_2$ applied on $\mathbb C^N$
under the coefficient-space identification of Remark~\ref{rem:coeff_ident}.
\end{lemma}

\begin{theorem}[Spectral decomposition with remainder]\label{thm:layerB}
Assume~(T.2) with $\varepsilon_N<\lambda_{N-1}(c)$.
For every $f=\sum_n a_n\psin\in V_{N,c}$,
\[
  \QM(f)=\ip{\Kop f}{f}_{L^2}+R_N(f),
  \qquad |R_N(f)|\le\varepsilon_N\norm{f}_{L^2}^2.
\]
\end{theorem}

\begin{proof}
In the PSWF basis,
\[
  \ip{\Kop f}{f}_{L^2}
  =\sum_n \lambda_n(c)|a_n|^2
  =\ip{G^{\mathrm{ref}}_{N,c}\mathbf a}{\mathbf a}.
\]
Hence
\[
  \QM(f)=\ip{\GVM\mathbf a}{\mathbf a}
  =\ip{G^{\mathrm{ref}}_{N,c}\mathbf a}{\mathbf a}+\ip{\EN\mathbf a}{\mathbf a}.
\]
Set $R_N(f):=\ip{\EN\mathbf a}{\mathbf a}$.
Then
\[
  |R_N(f)|\le \norm{\EN}_2\norm{\mathbf a}_2^2
  \le \varepsilon_N\norm{f}_{L^2}^2.
\]
\end{proof}

\begin{remark}[Structural bottleneck]
The key quantitative issue for any future application is to prove that
$\norm{\EN}_2$ is small relative to $\lambda_{N-1}(c)$ for structured
sampling sets.
By Proposition~\ref{prop:two_step_error}, this requires simultaneous control of a weighted-frame defect
and a spectral mismatch defect.
Without such control, Layer~B remains a formal comparison principle.
\end{remark}

\section{Layer~C: Heuristics for prime sampling}\label{sec:layerC}

This section is explicitly heuristic: statements in the environment
below indicate motivation and possible future directions, not proved theorems.
The Weil explicit formula involves prime sums of the form
$\sum_p \log p\,|\widehat h(\log p)|^2$,
which motivates considering prime sampling sets.
To obtain coercivity from prime sampling one would need both
unisolvence and condition~(T.2), and both problems are open.

\begin{heuristic}[Prime sampling stability]
In the regime $N\le (2/\pi-\varepsilon)c$, one may hope that PSWF
concentration and prime-gap information combine to yield estimates simultaneously controlling
weighted-frame stability and the spectral mismatch term.
This is currently speculative.
\end{heuristic}

\section{Open problems}\label{sec:open}

\begin{problem}[Uniform-grid unisolvence]
Determine whether uniform grids are unisolvent for $V_{N,c}$.
More generally, decide whether nontrivial zero-counting bounds hold for
linear combinations of the first $N$ PSWFs.
\end{problem}

\begin{problem}[Weighted frame models for PSWF sampling]
Construct explicit structured sampling sets and weights satisfying a Marcinkiewicz--Zygmund frame condition on $V_{N,c}$.
\end{problem}

\begin{problem}[Condition~(T.2) from two defects]
Derive quantitative estimates on the weighted-frame defect and the spectral mismatch defect
strong enough to trigger Corollary~\ref{cor:two_defects}.
Does a Marcinkiewicz--Zygmund inequality hold for PSWFs at prime nodes?
\end{problem}

\begin{problem}[Coupled limits]
Analyze $\lambda_{\min}(\GVM)$ and the competition between
$\norm{\EN}_2$ and $\lambda_{N-1}(c)$ in regimes where $N$ and $c$
vary simultaneously.
\end{problem}
