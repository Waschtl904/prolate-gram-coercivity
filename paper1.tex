\begin{abstract}
Let $\PW$ be the Paley--Wiener space of $\omega$-bandlimited functions
and $V_{N,c}:=\operatorname{span}\{\psi_0^{(c)},\dots,\psi_{N-1}^{(c)}\}$
the $N$-dimensional prolate subspace at time-bandwidth product $c=\omega T$.
We study the discrete sampling form
$\QM(f)=\sum_{j=1}^N w_j|f(p_j)|^2$ as a finite frame form on $V_{N,c}$.

We establish coercivity results for random sampling configurations and
for weighted sampling models satisfying a Marcinkiewicz--Zygmund lower
frame bound. Our main structural result is an exact defect decomposition
reducing frame analysis to simultaneous control of a weighted-frame
defect and a quadrature defect.

Assuming a standard first-order quadrature consistency estimate, we
derive a global quadrature defect bound in the PSWF eigenbasis:
\[
  |E_{mn}| \le C\,h c
  \qquad \text{for all } 0\le m,n\le N-1,
\]
where $h$ denotes the mesh size of the quadrature rule. The exact
identity $E_{mn} = 0 + R_{mn}^{\mathrm{quad}}$
(Proposition~\ref{prop:xry_split}) clarifies that the first term
vanishes by PSWF orthogonality, so the quadrature defect is fully
isolated. Separately, the continuous PSWF product integral exhibits
algebraic decay via a quantified phase-gap estimate and iterated
integration by parts.

The bulk--turning analysis is developed as an independent structural
component and is not used in the proof of the main stability theorem.
It identifies the oscillatory mechanism underlying decay in continuous
PSWF interactions; the transfer of this structure to the discrete
sampling level requires an additional hypothesis beyond first-order
quadrature consistency, encapsulated in the DSTP interface axiom and
addressed in a companion paper.
\end{abstract}

\section{Introduction}
%% ==================================================================

The prolate spheroidal wave functions (PSWFs) form an orthonormal basis
for the finite-dimensional subspace $V_{N,c}$ of the Paley--Wiener
space $\PW$ that is optimally concentrated in a time--frequency window
\cite{Slepian1978,BonamiKaroui}. From the perspective of frame theory
and sampling in reproducing kernel Hilbert spaces, the weighted
evaluation map $f\mapsto(\sqrt{w_j}f(p_j))_j$ defines a finite frame
operator on $V_{N,c}$, whose associated Gram matrix encodes the
stability of the sampling scheme.

This paper addresses two questions. First, which classes of sampling
configurations yield coercive Gram operators on $V_{N,c}$? Second, how
does the discrete sampling Gram matrix compare to the reference spectral
matrix $G^{\mathrm{ref}}_{N,c}=\Kop|_{V_{N,c}}$ in the PSWF
eigenbasis?

For the first question we prove coercivity for random sampling and for
weighted MZ frames. For the second, we establish an exact algebraic
defect decomposition, derive a global first-order quadrature bound for
the discrete defect, and separately analyse the oscillatory decay
mechanism of the continuous PSWF product integral via a WKB--bulk /
Airy--turning decomposition. A companion paper studies a PSWF-adapted
XRY quadrature regime and conditional spectral compatibility;
see~\cite{PaperII}.

\subsection*{Main results}
The key new ingredient is the replacement of the earlier conditional
quadrature-compatibility hypothesis by a verifiable first-order
quadrature estimate. The proof architecture has four explicit
components:
\begin{enumerate}
\item \textbf{Quadrature--projection split}
  (Proposition~\ref{prop:xry_split}): $E_{mn}$ is a pure quadrature
  remainder; the PSWF orthogonality term is exactly zero.
\item \textbf{Spectral parameter control}
  (Proposition~\ref{prop:spectral_gap}): the transition-regime
  asymptotics for $\chi_n(c)$ are stated explicitly and used in the
  phase-gap argument.
\item \textbf{Uniform phase gap and IBP}
  (Lemmas~\ref{lem:bulk_spectral_sep}, \ref{lem:phase_gap_final},
  \ref{lem:bulk_derivative}, \ref{lem:ibp_uniform},
  and~\ref{lem:bulk_decay_continuous}): bulk-to-spectral separation
  makes the phase-gap estimate explicit; bulk-restricted semiclassical
  derivative bounds control PSWF derivatives away from turning regions;
  bulk-localised iterated integration by parts yields algebraic decay
  for the continuous PSWF product integral.
\item \textbf{Quadrature reduction by derivative control}
  (Assumption~\ref{ass:quad_first_order},
  Lemma~\ref{lem:derivative_bound}, Theorem~\ref{thm:banded_decay}):
  the quadrature remainder is controlled by a standard first-order
  error term involving $\int|f'_{mn}|$, estimated via PSWF regularity.
\end{enumerate}

The oscillatory bulk--turning analysis in component~3 is developed as
an independent structural unit. It is \emph{not} used in the proof of
the main stability theorem (Theorem~\ref{thm:conditional_frame_schur});
its role is to isolate the analytic decay mechanism of continuous PSWF
interactions and to identify the structural gap that must be closed by
a PSWF-adapted quadrature framework. That gap is formalised by the
Discrete Spectral Transfer Property (DSTP), which serves as the
explicit axiomatic interface between the present paper and~\cite{PaperII}.

\section{Setup and notation}\label{sec:setup}
%% ==================================================================

\subsection{Paley--Wiener space and concentration operator}

We use the Fourier convention
$\widehat{f}(\xi)=\int_{\mathbb{R}}f(x)e^{-2\pi ix\xi}\,dx$.
The Paley--Wiener space $\PW$ consists of all $f\in L^2(\mathbb{R})$
with Fourier support in $[-\omega/(2\pi),\omega/(2\pi)]$. It is a
reproducing-kernel Hilbert space with kernel
\[
  K_{\PW}(x,y)
  =\tfrac{\omega}{\pi}\operatorname{sinc}(\omega(x-y)/\pi),
  \qquad K_{\PW}(x,x)=\omega/\pi
\]
\cite{BonamiKaroui}.

The prolate concentration operator is
\[
  \Kop f(x)=\int_{-T}^{T}\frac{\sin\omega(x-y)}{\pi(x-y)}f(y)\,dy,
  \qquad \Kop:\PW\to\PW,
\]
with eigenvalues $1>\lambda_0(c)>\lambda_1(c)>\cdots>0$, PSWF
eigenfunctions $\{\psin\}_{n\ge 0}$ forming an orthonormal basis of
$\PW$ \cite{Slepian1978}, and associated spectral parameters $\chi_n(c)$
(eigenvalues of the PSWF differential operator; see
\cite{OsipovRokhlinXiao,BonamiKaroui2014}). We set $c=\omega T$ and
$V_{N,c}:=\operatorname{span}\{\psi_0^{(c)},\dots,\psi_{N-1}^{(c)}\}$.
The transition regime $N\sim c$ is the natural operating range.

\subsection{Frame operator and defect matrix}

\begin{definition}[Weighted evaluation frame and Gram matrix]%
\label{def:GramForm}
For distinct $\mathbf{p}=(p_1,\dots,p_N)\subset[-T,T]$ and weights
$w_j>0$, the evaluation matrix is
$M(\mathbf{p},c)_{jn}:=\psi_n^{(c)}(p_j)$ and
$D:=\operatorname{diag}(w_1,\dots,w_N)$. The weighted Gram matrix is
\[
  G^{\mathrm{w}}_{N,c}(\mathbf{p},D)
  :=M(\mathbf{p},c)^*D\,M(\mathbf{p},c).
\]
For $f=\sum_{n=0}^{N-1}a_n\psin\in V_{N,c}$,
$\QM(f):=\sum_{j=1}^N w_j|f(p_j)|^2
=\ip{G^{\mathrm{w}}_{N,c}\mathbf{a}}{\mathbf{a}}$.
\end{definition}

\begin{definition}[Reference spectral matrix and defect]%
\label{def:ref_matrix}
\[
  G^{\mathrm{ref}}_{N,c}
  :=\operatorname{diag}(\lambda_0(c),\dots,\lambda_{N-1}(c)),
  \qquad
  \EN:=G^{\mathrm{w}}_{N,c}(\mathbf{p},D)-G^{\mathrm{ref}}_{N,c}.
\]
The defect entries are
\[
  E_{mn}
  :=\sum_{j=1}^N w_j\,\psi_m^{(c)}(p_j)\overline{\psi_n^{(c)}(p_j)}
    -\lambda_n(c)\,\delta_{mn}.
\]
\end{definition}

\begin{definition}[Frame-stable regime]\label{def:framestable}
$\mathbf{p}$ is in the frame-stable regime if
$\norm{\EN}_2<\lambda_{N-1}(c)$.
\end{definition}

\begin{proposition}[Spectral parameter asymptotics]%
\label{prop:spectral_gap}
Let $N\sim c$. Then there exists $C>0$ such that
\[
  \chi_n(c) = (2n+1)c + r_n(c),
  \qquad |r_n(c)| \le C c^{4/3},
\]
uniformly for $0\le n\le N-1$.
\end{proposition}
\begin{proof}
See \cite[Section~2.5]{OsipovRokhlinXiao} together with the uniform
transition-regime bounds in \cite{BonamiKaroui2014}.
\end{proof}

\section{Unisolvence and coercivity}\label{sec:coercivity}
%% ==================================================================

\begin{definition}[Frame condition]\label{def:frame}
$\{(p_j,w_j)\}_{j=1}^N$ is a weighted frame for $V_{N,c}$ if
$A\norm{f}_{L^2}^2\le\sum_j w_j|f(p_j)|^2\le B\norm{f}_{L^2}^2$
for all $f\in V_{N,c}$ with $0<A\le B<\infty$.
\end{definition}

\begin{definition}[Unisolvent set]
$\mathbf p$ is unisolvent for $V_{N,c}$ if
$f\mapsto(f(p_1),\dots,f(p_N))$ is injective on $V_{N,c}$.
\end{definition}

\begin{lemma}[Identity principle]\label{lem:identity}
If $f\in\PW$ vanishes on a set with a limit point in $\mathbb R$,
then $f\equiv 0$.
\end{lemma}
\begin{proof}
Every $f\in\PW$ is entire of exponential type; apply the identity
theorem.
\end{proof}

\begin{proposition}[MZ lower bound implies unisolvence]%
\label{prop:mz_unisolvent}
If $\{(p_j,w_j)\}$ is a weighted frame with lower bound $A>0$, then
$\mathbf p$ is unisolvent for $V_{N,c}$.
\end{proposition}
\begin{proof}
$f(p_j)=0$ for all $j$ implies $A\norm{f}^2\le 0$, hence $f\equiv 0$.
\end{proof}

\begin{proposition}[Random sampling]\label{prop:random}
Let $p_1,\dots,p_N$ be i.i.d.\ absolutely continuous on $[-T,T]$.
Then $\mathbf{p}$ is unisolvent for $V_{N,c}$ almost surely.
\end{proposition}
\begin{proof}
$(p_1,\dots,p_N)\mapsto\det M(\mathbf{p},c)$ is real-analytic and
non-identically zero, so its zero set has measure zero.
\end{proof}

\begin{lemma}[Pointwise RKHS bound]\label{lem:upper}
For $f\in\PW$ and $x\in\mathbb R$,
$|f(x)|^2\le(\omega/\pi)\norm{f}_{L^2}^2$.
\end{lemma}
\begin{proof}
Cauchy--Schwarz with the reproducing kernel.
\end{proof}

\begin{theorem}[Frame coercivity]\label{thm:coercivity}
Let $\mathbf{p}$ be unisolvent for $V_{N,c}$ and $D=I$. Then
\[
  \sqrt{\lambda_{\min}(\GVM)}\,\norm{f}_{L^2}
  \le\QM(f)^{1/2}
  \le\sqrt{N\omega/\pi}\,\norm{f}_{L^2}
  \qquad\forall f\in V_{N,c}.
\]
\end{theorem}
\begin{proof}
Lower bound: Rayleigh quotient. Upper bound: sum Lemma~\ref{lem:upper}
over nodes.
\end{proof}

\begin{lemma}[Operator norm bound]\label{lem:lambda_max_bound}
$\lambda_{\max}(G^{\mathrm{w}}_{N,c})\le N\omega/\pi$.
\end{lemma}

\section{Quadrature decomposition and oscillatory decay}\label{sec:decay}
%% ==================================================================

\begin{proposition}[Quadrature--Projection Decomposition]%
\label{prop:xry_split}
Let $M(\mathbf{p},c)$ be the evaluation matrix and
$D=\mathrm{diag}(w_j)$. Then
\[
  E_{mn}
  =
  \underbrace{
  \Bigl(
  \int_{-T}^{T}\psi_m^{(c)}(x)\overline{\psi_n^{(c)}(x)}\,dx
  -\lambda_n(c)\delta_{mn}
  \Bigr)}_{=\,0}
  +
  R_{mn}^{\mathrm{quad}},
\]
where the quadrature remainder is
\[
  R_{mn}^{\mathrm{quad}}
  =
  \sum_{j=1}^N w_j f_{mn}(p_j)
  -\int_{-T}^{T}f_{mn}(x)\,dx,
  \qquad f_{mn}(x)
  :=\psi_m^{(c)}(x)\overline{\psi_n^{(c)}(x)}.
\]
Since $\Kop$ is self-adjoint on $L^2([-T,T])$ and $\{\psi_n^{(c)}\}$
is an orthonormal eigenbasis of the concentration operator restricted
to $L^2([-T,T])$, we have
\[
\langle \Kop \psi_n^{(c)}, \psi_m^{(c)} \rangle_{L^2(\mathbb{R})}
= \lambda_n(c)\,\delta_{mn}.
\]
Using the kernel representation of $\Kop$ and the restriction of the
integration to $[-T,T]$, this inner product rewrites explicitly as
\[
\int_{-T}^{T} \psi_n^{(c)}(x)\overline{\psi_m^{(c)}(x)}\,dx
= \langle \Kop \psi_n^{(c)},
  \psi_m^{(c)} \rangle_{L^2(\mathbb{R})}
= \lambda_n(c)\,\delta_{mn},
\]
which proves that the bracketed term vanishes. Consequently,
\[
  E_{mn} = -R_{mn}^{\mathrm{quad}}
\]
under the sign convention of Assumption~\ref{ass:quad_first_order}, so
all later estimates use only $|R_{mn}^{\mathrm{quad}}|$.
\end{proposition}

\begin{remark}
Proposition~\ref{prop:xry_split} corrects an implicit error in earlier
drafts: $E_{mn}$ is \emph{not} an algebraically exact split arising
from some identity; it is purely a quadrature error for the PSWF
product integral.
\end{remark}

\begin{definition}[Bulk and turning regions]\label{def:bulk}
Let $x_n^*>0$ denote the positive turning point of $\psi_n^{(c)}$ and
fix $\rho>0$. The Airy neighbourhood is
\[
  U_n
  :=\{x\in[-T,T]:|x-x_n^*|\le\rho c^{-1/3}\}
   \cup\{x\in[-T,T]:|x+x_n^*|\le\rho c^{-1/3}\}.
\]
For a pair $(m,n)$:
\[
  \Rturning^{(m,n)}:=U_m\cup U_n,
  \qquad
  \Rbulk^{(m,n)}:=[-T,T]\setminus\Rturning^{(m,n)}.
\]
\end{definition}

\begin{remark}
The width $O(c^{-1/3})$ is the standard Airy transition scale
\cite{Olver1974}: beyond the neighbourhoods of the turning points
$\pm x_n^*$ the WKB phase is non-stationary for $|m-n|\ge 1$.
\end{remark}

\begin{lemma}[Bulk spectral separation via turning-point geometry]%
\label{lem:bulk_spectral_sep}
Let $x\in\Rbulk^{(m,n)}$. Then there exists $\delta_0>0$, depending
only on $\rho$, such that
\[
  \left|\chi_m(c)-\frac{c^2x^2}{T^2}\right|\ge\delta_0 c^{2/3},
  \qquad
  \left|\chi_n(c)-\frac{c^2x^2}{T^2}\right|\ge\delta_0 c^{2/3}.
\]
\end{lemma}
\begin{proof}
For $k\in\{m,n\}$, $\chi_k(c)-c^2(x_k^*)^2/T^2=0$ gives
$\chi_k(c)-c^2x^2/T^2 = (c^2/T^2)(x_k^*-x)(x_k^*+x)$.
On $\Rbulk^{(m,n)}$, Definition~\ref{def:bulk} ensures
$|x_k^*\pm x|\ge\rho c^{-1/3}$, hence
\[
  \left|\chi_k(c)-\frac{c^2x^2}{T^2}\right|
  \ge\frac{c^2}{T^2}\,\rho^2 c^{-2/3}
  =\frac{\rho^2}{T^2}c^{2/3}.
\]
Set $\delta_0:=\rho^2/T^2$.
\end{proof}

\begin{lemma}[Uniform phase gap in the bulk, two-regime form]%
\label{lem:phase_gap_final}
Let $N\sim c$ and fix $\rho>0$. There exist $c_0>0$ and $\kappa>0$
such that for all $c\ge c_0$, uniformly in $0\le m\ne n\le N-1$:
\begin{align*}
  |\Phi_{mn}'(x)|&\ge\tfrac{\kappa}{T}|m-n|,
  &&|m-n|\le c^{1/3},\\
  |\Phi_{mn}'(x)|&\ge\tfrac{\kappa}{T}c^{1/3},
  &&|m-n|\ge c^{1/3},
\end{align*}
for all $x\in\Rbulk^{(m,n)}$. Consequently,
\[
  |\Phi_{mn}'(x)|
  \ge\frac{\kappa}{T}\cdot\frac{|m-n|}{1+c^{-2/3}|m-n|}
  \qquad\forall x\in\Rbulk^{(m,n)}.
\]
The final bound is a unified lower envelope and is not claimed sharp.
\end{lemma}
\begin{proof}
Set $\mu:=\chi_m(c)-c^2x^2/T^2$, $\nu:=\chi_n(c)-c^2x^2/T^2$.
By Lemma~\ref{lem:bulk_spectral_sep}, $\mu,\nu\ge\delta_0 c^{2/3}$.
Using $|\sqrt\mu-\sqrt\nu|=|\mu-\nu|/(\sqrt\mu+\sqrt\nu)$:
\[
  |\Phi_{mn}'(x)|
  =\frac{|\chi_m(c)-\chi_n(c)|}{T(\sqrt\mu+\sqrt\nu)}.
\]
\textbf{Step~1 (numerator).} By Proposition~\ref{prop:spectral_gap},
$|\chi_m-\chi_n|\ge 2c|m-n|-2C_*c^{4/3}$.
For $|m-n|\le c^{1/3}$ and $c\ge c_0$ large: $|\chi_m-\chi_n|\ge c|m-n|$.
For $|m-n|\ge c^{1/3}$: $|\chi_m-\chi_n|\ge c^{4/3}$.

\noindent\textbf{Step~2 (denominator).}
$\sqrt\mu+\sqrt\nu\le 2\sqrt{\chi_{N-1}(c)}\le C_3 c$
since $\chi_{N-1}(c)\lesssim c^2$ in the regime $N\sim c$.

\noindent\textbf{Step~3.} Combining yields the two-regime bounds. The
unified envelope follows from
$c|m-n|\mathbf1_{|m-n|\le c^{1/3}}+c^{4/3}\mathbf1_{|m-n|\ge c^{1/3}}
\gtrsim c|m-n|/(1+c^{-2/3}|m-n|)$.
\end{proof}

\begin{lemma}[Bulk semiclassical derivative control]%
\label{lem:bulk_derivative}
Fix $\rho>0$. For every integer $k\ge 0$ there exists $C_k>0$,
depending only on $k$ and $\rho$, such that
\[
  \|\partial_x^k\psi_n^{(c)}\|_{L^\infty(\Rbulk^{(m,n)})}
  \le C_k\,c^{k/2}
\]
uniformly for $0\le n\le N-1$ and $c\ge 1$ in the regime $N\sim c$.
\end{lemma}
\begin{proof}
This follows from uniform semiclassical estimates developed in
\cite{OsipovRokhlinXiao} and \cite{BonamiKaroui2014}, together with
classical WKB regularity \cite[Ch.~6]{Olver1974}.

On $\Rbulk^{(m,n)}$ the PSWF differential equation lies in the
non-stationary semiclassical regime. The WKB phase satisfies
$\Phi_n'(x)=(1/T)\sqrt{\chi_n(c)-c^2x^2/T^2}=O(c^{1/2})$ by
Lemma~\ref{lem:bulk_spectral_sep}.

The cited results provide uniform $C^k$ bounds for the PSWFs on
compact subsets of the bulk region separated from the turning points
by Airy-scale neighbourhoods of size $\rho c^{-1/3}$:
\[
  \|\partial_x^k\psi_n^{(c)}\|_{L^\infty(\Rbulk^{(m,n)})}
  \le C_k'\,c^{k/2},
\]
with constants uniform in $n$ for $N\sim c$. We do not require an
explicit amplitude-phase decomposition. Each application of $\partial_x$
contributes at most a factor $\Phi_n'(x)=O(c^{1/2})$, consistent
with the stated bound.
\end{proof}

\begin{lemma}[Bulk-localised uniform IBP control]\label{lem:ibp_uniform}
Let $a_{mn}(x)=A_m(x)\overline{A_n(x)}$ be the WKB amplitude product
and $L:=(i\Phi_{mn}'(x))^{-1}\partial_x$. For every $p\ge 1$,
\[
  \|(L^*)^p a_{mn}\|_{L^1(\Rbulk^{(m,n)})}
  \le C_p\Bigl(\frac{1+c^{-2/3}|m-n|}{|m-n|}\Bigr)^p.
\]
Moreover, $(L^*)^p a_{mn}$ is a finite linear combination of terms
$c_{p,k,\ell}(x)\,\partial_x^k a_{mn}(x)/(\Phi_{mn}'(x))^{\ell}$,
$0\le k\le p$, $1\le\ell\le p$, with coefficients depending only on
$p$ and finitely many derivatives of $\Phi_{mn}'$.
\end{lemma}
\begin{proof}
The formal adjoint is
$L^*=-(i\Phi_{mn}')^{-1}\partial_x+b_1(x)(\Phi_{mn}')^{-2}$,
with $b_1$ smooth on $\Rbulk^{(m,n)}$. By induction, $(L^*)^p a_{mn}$
has the stated structure. By Lemma~\ref{lem:phase_gap_final},
$|(\Phi_{mn}')^{-1}|\le(T/\kappa)(1+c^{-2/3}|m-n|)/|m-n|$.
Lemma~\ref{lem:bulk_derivative} gives uniform $C^k$ bounds for all
amplitude derivatives on $\Rbulk^{(m,n)}$. Integrating over
$\Rbulk^{(m,n)}\subset[-T,T]$ yields the stated $L^1$ bound.
\end{proof}

\begin{lemma}[Bulk decay for the continuous integral]\label{lem:bulk_decay}
Under the hypotheses of Lemma~\ref{lem:ibp_uniform}, for $m\neq n$,
\[
  \left|\int_{\Rbulk^{(m,n)}}f_{mn}(x)\,dx\right|
  \le C_p\Bigl(\frac{1+c^{-2/3}|m-n|}{|m-n|}\Bigr)^p
  \le C_p'(1+|m-n|)^{-p},
\]
with constants independent of $m,n,N,c$.
\end{lemma}
\begin{proof}
Apply $L$ iteratively on $\Rbulk^{(m,n)}$; oscillatory boundary terms
are absorbed into the bulk localisation. Lemma~\ref{lem:ibp_uniform}
controls the amplitude expressions in $L^1$. The simplification
$(1+c^{-2/3}|m-n|)/|m-n|\le 2/(1+|m-n|)$ yields the final bound.
\end{proof}

\begin{lemma}[Turning-point estimate]\label{lem:turning_revised}
For all $0\le m,n\le N-1$,
$\int_{U_m}|\psi_m^{(c)}(x)\psi_n^{(c)}(x)|\,dx\le C\,c^{-1/3}$.
\end{lemma}
\begin{proof}
By \cite{BonamiKaroui2014}, $\|\psi_k^{(c)}\|_{L^\infty}\le C_\infty$
uniformly. Since $|U_m|=O(c^{-1/3})$,
$\|\psi_k^{(c)}\|_{L^2(U_m)}\le C_\infty|U_m|^{1/2}=O(c^{-1/6})$.
Cauchy--Schwarz gives
$\int_{U_m}|\psi_m^{(c)}\psi_n^{(c)}|\le C|U_m|=O(c^{-1/3})$.
\end{proof}

\begin{assumption}[First-order quadrature consistency]%
\label{ass:quad_first_order}
Let $\{(p_j,w_j)\}_{j=1}^N$ be a quadrature rule on $[-T,T]$ with
mesh size $h:=\max_{j}|p_{j+1}-p_j|$. We assume there exists $C_Q>0$
such that for all $f\in W^{1,1}([-T,T])$,
\[
  \Bigl|\int_{-T}^{T}f(x)\,dx-\sum_{j=1}^N w_jf(p_j)\Bigr|
  \le C_Q\,h\int_{-T}^{T}|f'(x)|\,dx.
\]
\end{assumption}

\begin{remark}[Scope of the quadrature input]\label{rem:scope_quad}
Assumption~\ref{ass:quad_first_order} is intentionally coarse: it
controls only the absolute size of the quadrature remainder and encodes
no oscillation-sensitive cancellation, PSWF-adapted exactness, or
transfer of continuous off-diagonal decay to the discrete level.
\end{remark}

\begin{lemma}[Derivative bound for PSWF products]%
\label{lem:derivative_bound}
For $f_{mn}(x):=\psi_m^{(c)}(x)\overline{\psi_n^{(c)}(x)}$, if the
Bonami--Karoui bounds $\|\psi_k^{(c)}\|_{L^\infty}\le C_0$ and
$\|\psi_k^{(c)\,\prime}\|_{L^\infty}\le C_1 c$ hold uniformly for
$0\le k\le N-1$, then
\[
  \int_{-T}^{T}|f'_{mn}(x)|\,dx\le Cc,
\]
with a constant independent of $m,n,N,c$.
\end{lemma}
\begin{proof}
Differentiate: $|f'_{mn}|\le|\psi_m'\|\psi_n|+|\psi_m||\psi_n'|
\le 2C_0C_1 c$. Integration over $[-T,T]$ gives the claim.
\end{proof}

\begin{theorem}[Quadrature defect bound]\label{thm:banded_decay}
Let Assumption~\ref{ass:quad_first_order} hold. Then
$|E_{mn}|\le Chc$ for all $0\le m,n\le N-1$,
with $C>0$ independent of $m,n,N,c$.
\end{theorem}
\begin{proof}
By Proposition~\ref{prop:xry_split}, $|E_{mn}|=|R_{mn}^{\mathrm{quad}}|$.
Apply Assumption~\ref{ass:quad_first_order} and then
Lemma~\ref{lem:derivative_bound}:
$|E_{mn}|\le C_Q h\int|f'_{mn}|\le C_QCh c$.
\end{proof}

\begin{remark}[Sharpness]\label{rem:sharpness_quad}
The bound $|E_{mn}|\le Chc$ is structure-blind. Without additional
oscillation-sensitive quadrature structure, no off-diagonal decay in
$|m-n|$ is expected at the discrete level; the estimate is essentially
sharp under the present assumptions.
\end{remark}

\begin{lemma}[Continuous bulk--turning decay]%
\label{lem:bulk_decay_continuous}
For all $p\ge 1$ and $0\le m,n\le N-1$,
\[
  \left|\int_{-T}^{T}\psi_m^{(c)}(x)\overline{\psi_n^{(c)}(x)}\,dx\right|
  \le C_p(1+|m-n|)^{-p}+C\,c^{-1/3}.
\]
For $m=n$ the integral equals $\lambda_n(c)$.
\end{lemma}
\begin{proof}
For $m\neq n$ decompose over $\Rbulk^{(m,n)}$ and $\Rturning^{(m,n)}$.
Lemma~\ref{lem:bulk_decay} bounds the bulk term by
$C_p(1+|m-n|)^{-p}$; Lemma~\ref{lem:turning_revised} bounds the
turning term by $Cc^{-1/3}$.
\end{proof}

\begin{remark}
This decay describes the continuous PSWF kernel and does \emph{not}
transfer to $E_{mn}$ without additional oscillation-compatible
quadrature assumptions; see~\cite{PaperII}.
\end{remark}

\section{Defect decomposition and quantitative frame stability}%
\label{sec:stability}
%% ==================================================================

\begin{theorem}[Defect decomposition]\label{thm:main_reduction}
For any $(\mathbf p,D)$ with $D>0$,
\[
  \EN=M(\mathbf p,c)^*(I-D)M(\mathbf p,c)
     +(G^{\mathrm{w}}_{N,c}-G^{\mathrm{ref}}_{N,c}),
\]
and $\norm{\EN}_2\le\norm{I-D}_2\cdot N\omega/\pi
+\norm{G^{\mathrm{w}}_{N,c}-G^{\mathrm{ref}}_{N,c}}_2$.
\end{theorem}

\begin{lemma}[Weyl perturbation]\label{lem:weyl}
If $\norm{\EN}_2<\lambda_{N-1}(c)$ then
$\lambda_{\min}(G^{\mathrm{w}}_{N,c})
\ge\lambda_{N-1}(c)-\norm{\EN}_2>0$.
\end{lemma}
\begin{proof}
$\lambda_{\min}(A+E)\ge\lambda_{\min}(A)-\norm{E}_2$ for Hermitian
matrices.
\end{proof}

\begin{definition}[Discrete Spectral Transfer Property]\label{def:dstp}
Let $\{(p_j,w_j)\}_{j=1}^N$ satisfy Assumption~\ref{ass:quad_first_order}
with defect matrix $\EN$. We say the rule satisfies the \emph{Discrete
Spectral Transfer Property} (DSTP) on $V_{N,c}$ if there exist
$C>0$ and $p>1$ such that
\[
  |E_{mn}|\le Chc\,(1+|m-n|)^{-p}
  \qquad\text{for all }0\le m,n\le N-1.
\]
DSTP is an \emph{axiomatic interface hypothesis} that goes strictly
beyond Assumption~\ref{ass:quad_first_order}: it asserts that the
discrete defect inherits a Schur-summable off-diagonal profile at
scale $hc$, in a form that mirrors the continuous oscillatory decay
established in Section~\ref{sec:decay}. DSTP is \emph{not} a
consequence of first-order quadrature consistency, nor is it implied
by the continuous decay estimates of Section~\ref{sec:decay}. It is
an \emph{additional structural assumption} that precisely encodes the
missing transfer mechanism between continuous oscillatory decay and
discrete frame stability, and whose verification for concrete
quadrature schemes is deferred to~\cite{PaperII}.
\end{definition}

\begin{theorem}[Conditional frame stability under Schur-decay hypothesis]%
\label{thm:conditional_frame_schur}
Let $\{(p_j,w_j)\}_{j=1}^N$ satisfy Assumption~\ref{ass:quad_first_order}
with $N\sim c$, and assume DSTP (Definition~\ref{def:dstp}). If
\[
  Chc\sum_{k\in\mathbb Z}(1+|k|)^{-p}<\lambda_{N-1}(c),
\]
then $\{(p_j,w_j)\}$ is a weighted frame for $V_{N,c}$ with bounds
\begin{align*}
  A(N,c)&:=\lambda_{N-1}(c)-Chc\!\sum_{k}(1+|k|)^{-p}>0,\\
  B(N,c)&:=\lambda_0(c)+Chc\!\sum_{k}(1+|k|)^{-p}.
\end{align*}
\end{theorem}
\begin{proof}
DSTP and the Schur test give
$\norm{\EN}_2\le Chc\sum_k(1+|k|)^{-p}$.
Apply Lemma~\ref{lem:weyl} for the lower bound and the reference
diagonal for the upper bound.
\end{proof}

\begin{corollary}\label{cor:fixed_param_scope}
Theorem~\ref{thm:conditional_frame_schur} yields finite-dimensional
frame stability under DSTP and first-order quadrature control.
Assumption~\ref{ass:quad_first_order} alone does not imply the
Schur-summable structure required.
\end{corollary}

\begin{remark}[Separation of scales]
The continuous oscillatory decay of PSWF products is independent of
the quadrature-induced defect floor. Without additional structure in
the sampling rule, this decay does not propagate to $E_{mn}$.
\end{remark}

\begin{remark}[DSTP as interface axiom]\label{rem:dstp_interface}
The present paper does not establish DSTP. Once DSTP holds, frame
stability follows from the Schur-test argument above. Standard Riemann-
or trapezoidal-type rules do not in general satisfy DSTP. Identifying
concrete schemes for which DSTP holds is developed in~\cite{PaperII}.
\end{remark}

\begin{remark}[Scaling]\label{rem:scaling_stability}
Under DSTP, $\norm{\EN}_2=O(hc)$, so frame stability is governed
directly by $hc/\lambda_{N-1}(c)$, without additional polynomial
growth in $N$.
\end{remark}

\section{Boundary to the companion framework}\label{sec:bridge}
%% ==================================================================

The present paper proves robust finite-dimensional stability under a
standard first-order quadrature consistency assumption, separating
three distinct mechanisms: continuous PSWF orthogonality, oscillatory
decay of the continuous product kernel, and quadrature-induced defect
control.

The continuous bulk--turning analysis of Section~\ref{sec:decay}
explains why off-diagonal PSWF interactions decay analytically but is
\emph{not used} in the proof of Theorem~\ref{thm:conditional_frame_schur}.
The unconditional discrete estimate is governed by the coarse bound
$|E_{mn}|\le Chc$; the frame theorem requires the additional interface
hypothesis DSTP. Consequently the present framework is robust but not
spectrally sharp: it yields frame stability for fixed $(N,c)$ once the
defect is dominated by $\lambda_{N-1}(c)$, but does not provide an
asymptotically uniform theorem in the transition regime $N\sim c$.

The companion paper~\cite{PaperII} addresses whether oscillatory
structure of PSWF products can be preserved at the discrete level by
PSWF-adapted quadrature. Establishing DSTP for concrete schemes is
deferred entirely to that paper. DSTP is the explicit axiomatic
interface between the two papers.

\section{Open problems}\label{sec:open}
%% ==================================================================

\begin{problem}[Sharp decay rate]
Determine whether the algebraic decay in
Lemma~\ref{lem:bulk_decay_continuous} can be upgraded to exponential,
or whether $p\to\infty$ yields superalgebraic decay.
\end{problem}

\begin{problem}[Sharper turning bounds]
Refine the turning contribution from $O(c^{-1/3})$ via Airy asymptotics
and microlocal analysis near the turning points.
\end{problem}

\begin{problem}[Quantitative constants]
Make the constants in Lemmas~\ref{lem:bulk_spectral_sep},
\ref{lem:phase_gap_final}, \ref{lem:ibp_uniform},
and~\ref{lem:derivative_bound} fully explicit in $N$ and $c$.
\end{problem}

\begin{problem}[Oscillation-compatible quadrature]
Identify concrete quadrature schemes satisfying DSTP; a PSWF-Gauss/XRY
framework is developed in~\cite{PaperII}.
\end{problem}

\begin{problem}[Uniform-grid unisolvence]
Determine whether uniform grids are unisolvent for $V_{N,c}$.
\end{problem}

\begin{problem}[Coupled limits]
Analyse $\lambda_{\min}(G^{\mathrm{w}}_{N,c})$ when $N$ and $c$ vary
simultaneously.
\end{problem}

\section{Airy uniformisation near turning points}\label{app:airy}
%% ==================================================================

\begin{lemma}[PSWF Airy approximation]\label{lem:airy_approx}
For each $0\le n\le N-1$ and $x\in U_n$,
\[
  \psi_n^{(c)}(x)
  =\kappa_n^{(c)}\,\mathrm{Ai}\!\bigl(-\zeta_n(x)\bigr)+O(c^{-1/3}),
\]
where $\zeta_n(x)=c^{2/3}(x-x_n^*)/T+O(c^{-1/3})$ and
$|\kappa_n^{(c)}|\le C_\kappa$ uniformly in $n,c$
\cite{Olver1974,OsipovRokhlinXiao}.
\end{lemma}
\begin{proof}
Standard semiclassical analysis \cite[Ch.~11]{Olver1974}.
\end{proof}

\begin{lemma}[Turning-point integral via Airy]\label{lem:airy_bound}
For all $0\le m,n\le N-1$,
\[
  \left|\int_{U_m\cup U_n}
    \psi_m^{(c)}(x)\overline{\psi_n^{(c)}(x)}\,dx\right|
  \le C\,c^{-1/3}.
\]
\end{lemma}
\begin{proof}
By the uniform sup-norm bound and $|U_k|=O(c^{-1/3})$,
$\|\psi_k^{(c)}\|_{L^2(U_m\cup U_n)}=O(c^{-1/6})$.
Cauchy--Schwarz on $U_m\cup U_n$ gives $O(c^{-1/3})$.
\end{proof}
