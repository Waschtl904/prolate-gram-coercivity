\begin{abstract}
Let $\{\psin\}_{n\ge 0}$ be the prolate spheroidal wave functions on
$[-T,T]$ with bandwidth $\omega$ and $c = \omega T$. We prove that
for every $f \in C^1([-T,T])$ and $n \le \gamma N$ (bulk regime):
\begin{equation}\label{eq:main}
  \int_{-T}^T f(x)\,\psin(x)^2\,dx
  = \lambda_n(c)\int_{-T}^T f(x)\,\rhocl(x)\,dx
    + O\!\left(\frac{\lambda_n(c)\,\|f'\|_{L^\infty}}{n}\right),
\end{equation}
where $\rhocl$ is the classical equilibrium density of the
Sturm--Liouville operator $D_c$.
The error constant is explicit and depends only on $\gamma$.

The proof uses the Pr\"ufer transformation to separate amplitude
and phase, establishes a $O(1/n)$-bound on oscillatory integrals
against $\cos(2\theta_n)$ via a single integration by parts
(Lemma~\ref{lem:prufer-oscillation-control}), and controls the
amplitude drift by a coupled fixed-point argument on the Pr\"ufer ODE
(Lemma~\ref{lem:amplitude-drift}).
No Airy-function matching is needed; the turning points $x_\pm(n)$
are handled by a bulk cutoff and the exponential concentration of
$\psin$ outside the classically allowed region.
\end{abstract}

\section{Introduction}\label{sec:intro}
%%=================================================================

\subsection{Setting and goal}

This paper addresses Problem~\ref{prob:pswf-weak-limit} stated in
\cite{PaperIII}: the quantitative weak convergence of the normalised
PSWF density $\psin^2\,dx/\lambda_n(c)$ to the classical equilibrium
measure $\rhocl(x)\,dx$ on $[-T,T]$.

By Lemma~6.3 of \cite{PaperIII} (\texttt{lem:bulk-reduction}),
the main result of this paper implies the energy equidistribution
assertion of Problem~6.2 of \cite{PaperIII}
(\texttt{prob:comm-refined}),
and thereby resolves the central open question of the
Prolate--Weil program in the bulk regime.

\subsection{Why this is non-trivial for PSWFs}

For classical orthogonal polynomials (Legendre, Hermite, Laguerre),
the analogous equidistribution follows from the explicit
three-term recurrence and Szeg\H{o}'s classical theory \cite{Szego}.
For PSWFs, no clean algebraic recurrence is available: the
Chr\-is\-tof\-fel--Dar\-boux kernel is not of hypergeometric type,
and the connection coefficients between different $c$-values are
not explicit.
The argument must therefore be purely analytic, via the ODE structure
of the Sturm--Liouville operator $D_c$.

\subsection{Strategy}

The proof has four steps:
\begin{enumerate}[(1)]
  \item \emph{Bulk cutoff.}
  For $n \le \gamma N$, the turning points $x_\pm(n) = \pm T\sqrt{1 - 2n/c}$
  (approximately) lie strictly inside $(-T,T)$.
  The contribution of $\psin^2$ outside $[x_-(n)-\varepsilon, x_+(n)+\varepsilon]$
  is exponentially small (Lemma~\ref{lem:turning-point-cutoff}).
  \item \emph{Pr\"ufer decomposition.}
  On the bulk interval $I_\delta := [-(1-\delta)T, (1-\delta)T]$,
  write $\psin = r_n(x)\sin\theta_n(x)$ via the Pr\"ufer transformation
  (Section~\ref{sec:prufer}).
  The Pr\"ufer phase satisfies $\theta_n'(x) \ge c_\delta n$
  for all $x \in I_\delta$
  (Lemma~\ref{lem:phase-monotone}).
  \item \emph{Oscillation control.}
  The $\cos(2\theta_n)$-part of $\psin^2$ integrates to $O(1/n)$
  against any $C^1$ function (Lemma~\ref{lem:prufer-oscillation-control},
  the main technical lemma).
  \item \emph{Amplitude matching.}
  The averaged part $r_n^2/2$ of $\psin^2$ agrees with
  $\lambda_n(c)\rhocl$ up to $O(\lambda_n/n)$
  (Lemma~\ref{lem:amplitude-drift} + Lemma~\ref{lem:normalization}).
\end{enumerate}

\subsection{Relation to Paper III}

The hypothesis~(6.1) of Lemma~6.3 in \cite{PaperIII} is exactly
\eqref{eq:main} with $\omega_n = C_\gamma/n$.
The main Theorem~\ref{thm:weak-limit} of this paper supplies this
hypothesis, completing the bulk-decorrelation reduction chain:
\[
  \text{Thm.~\ref{thm:weak-limit}}
  \Rightarrow \texttt{lem:bulk-reduction}\,(\text{\cite{PaperIII}})
  \Rightarrow \texttt{prob:comm-refined}\,(\text{\cite{PaperIII}})
  \Rightarrow \text{Bulk tail bound}.
\]

\subsection{Organization}

Section~\ref{sec:prelim} fixes notation and states the
concentration property of PSWFs near turning points.
Section~\ref{sec:prufer} develops the Pr\"ufer system for $D_c$.
Section~\ref{sec:oscillation} proves the main oscillation lemma
(Lemma~\ref{lem:prufer-oscillation-control}).
Section~\ref{sec:amplitude} controls the amplitude drift
(Lemma~\ref{lem:amplitude-drift}).
Section~\ref{sec:normalization} establishes the normalization matching
(Lemma~\ref{lem:normalization}).
Section~\ref{sec:weaklimit} assembles the proof of
Theorem~\ref{thm:weak-limit}.

\section{Preliminaries}\label{sec:prelim}
%%=================================================================

\subsection{PSWF conventions}

We retain all conventions of \cite{PaperIII}, Section~2.
In particular $D_c = -(p(x)\cdot)'\,{}' + c^2x^2/T^2$ with
$p(x) = 1 - x^2/T^2$, and $\chi_n(c)$ denotes the $n$-th
eigenvalue.
We write $\lambda_n := \lambda_n(c)$ for brevity.

\subsection{Classical data}

\begin{definition}[Classical frequency and turning points]%
\label{def:classical-freq}
For $n \le \gamma N$, set:
\begin{align}
  \wncl(x)  &:= \sqrt{\frac{\chi_n(c) - c^2x^2/T^2}{p(x)}},
  \label{eq:wncl}\\
  x_+(n) &:= T\sqrt{1 - \chi_n(c)/c^2}
  \quad (\text{classical turning point, right}).
  \label{eq:turning}
\end{align}
By the eigenvalue asymptotics $\chi_n(c) = n(n+1) + c^2/2 + O(n^{-1}c^2)$,
for $n \le \gamma N$ with $\gamma < 1$ fixed:
$x_+(n) \le (1 - \eta_\gamma)T$ for some $\eta_\gamma > 0$.
\end{definition}

\begin{definition}[Classical equilibrium density]%
\label{def:rho-cl}
\begin{equation}\label{eq:rho-cl}
  \rhocl(x) := \frac{\wncl(x)^{-1}}{\int_{-x_+(n)}^{x_+(n)}\wncl(y)^{-1}\,dy}
  = \frac{1}{\pi}\cdot\frac{1}{\wncl(x)\cdot K_n},
\end{equation}
where $K_n := \frac{1}{\pi}\int_{-x_+(n)}^{x_+(n)}\wncl(x)^{-1}\,dx$
is the normalization constant.
\end{definition}

\begin{remark}[Classical period and normalization constant]%
\label{rem:period}
The integral $K_n$ equals the classical period of the Hamiltonian
flow for $D_c$ at energy $\chi_n(c)$:
$K_n = T_{\mathrm{cl}}(n) / (2\pi)$,
where $T_{\mathrm{cl}}(n) = 2\int_{-x_+(n)}^{x_+(n)}\wncl(x)^{-1}\,dx$.
Lemma~\ref{lem:normalization} shows $K_n = (\lambda_n\pi)^{-1}(1 + O(1/n))$,
which makes $\int\rhocl = 1$ consistent with $\int\psin^2 \approx \lambda_n$.
\end{remark}

\subsection{Turning point concentration}

\begin{lemma}[Exponential concentration in the bulk]%
\label{lem:turning-point-cutoff}
For $n \le \gamma N$ and $\delta \in (0,\eta_\gamma)$,
let $I_\delta := [-(1-\delta)T,(1-\delta)T]$.
Then:
\begin{equation}\label{eq:exp-cutoff}
  \int_{[-T,T]\setminus I_\delta} \psin(x)^2\,dx
  \le C_\gamma\,e^{-\beta_\gamma n}\,\lambda_n,
\end{equation}
for constants $C_\gamma, \beta_\gamma > 0$ depending only on $\gamma$.
\end{lemma}

\begin{proof}
On $[x_+(n)+\delta T/2,T]$, the SL equation gives
$D_c\psin = \chi_n\psin$ with $c^2x^2/T^2 > \chi_n$
(classically forbidden), so $\psin$ satisfies a differential
inequality that forces exponential decay away from $x_+(n)$.
The rate $\beta_\gamma$ follows from
$\min_{x \in [x_+(n)+\delta T/2,T]} (c^2x^2/T^2 - \chi_n) \ge c_\gamma\,c^2$.
For the explicit bound, see \cite[Ch.~4, Prop.~4.2]{OsipovRokhlinXiao}.
\end{proof}

\begin{remark}[No Airy matching needed]
Lemma~\ref{lem:turning-point-cutoff} suffices for our purposes:
we only need $\int_{\mathrm{outside}} f\psin^2 = O(e^{-\beta n}\lambda_n)$,
which is smaller than the $O(\lambda_n/n)$ target.
The precise Airy-function description of $\psin$ near $x_+(n)$
(see \cite[Ch.~5]{OsipovRokhlinXiao}) is not required.
\end{remark}

\section{Pr\"ufer System for $D_c$}\label{sec:prufer}
%%=================================================================

\subsection{Pr\"ufer transformation}

\begin{definition}[Pr\"ufer variables]\label{def:prufer}
For $x \in I_\delta$, define the Pr\"ufer amplitude $r_n(x) > 0$
and Pr\"ufer phase $\theta_n(x)$ by:
\begin{align}
  \psin(x) &= r_n(x)\sin\theta_n(x), \label{eq:prufer-psi}\\
  \sqrt{p(x)}\,\psin'(x) &= r_n(x)\wncl(x)\cos\theta_n(x). \label{eq:prufer-psip}
\end{align}
This is the standard Pr\"ufer transformation adapted to the
coefficient $p(x)$ of $D_c$; see \cite[\S4.1]{Levitan}.
\end{definition}

\begin{lemma}[Pr\"ufer ODEs]\label{lem:prufer-ode}
The functions $r_n$ and $\theta_n$ satisfy:
\begin{align}
  \theta_n'(x)
  &= \wncl(x) - \frac{p'(x)}{2p(x)}\sin\theta_n(x)\cos\theta_n(x)
     + \frac{c^2x/T^2}{p(x)\wncl(x)}\sin^2\theta_n(x),
  \label{eq:theta-ode}\\
  \frac{r_n'(x)}{r_n(x)}
  &= \left(\frac{(\wncl)'(x)}{2\wncl(x)}
     + \frac{p'(x)}{4p(x)}\right)\cos(2\theta_n(x))
     + \frac{p'(x)}{4p(x)}.
  \label{eq:r-ode}
\end{align}
\end{lemma}

\begin{proof}
Differentiate \eqref{eq:prufer-psi}--\eqref{eq:prufer-psip} and
use the SL equation $-(p\psin')' + (c^2x^2/T^2)\psin = \chi_n\psin$
to eliminate $\psin''$. The identity
$-(p(x)u')' = -p'u' - pu''$ gives, after substituting
the Pr\"ufer expressions:
\[
  \theta_n' = \wncl + (\text{correction terms involving }p', (\wncl)')
\]
The correction terms are $O(1/n)$ in $I_\delta$ since
$\wncl \asymp n$ and $p', (\wncl)' = O(1)$ there.
See \cite[Thm.~4.1]{Levitan} for the general Pr\"ufer system;
the specific form \eqref{eq:theta-ode} is obtained by choosing
the weight $\wncl$ as in \eqref{eq:prufer-psip}.
\end{proof}

\subsection{Phase monotonicity in the bulk}

\begin{lemma}[Phase lower bound]\label{lem:phase-monotone}
For $n \le \gamma N$, $x \in I_\delta$,
and $n$ sufficiently large:
\begin{equation}\label{eq:phase-lower}
  \theta_n'(x) \ge \frac{1}{2}\wncl(x) \ge \frac{c_{\delta,\gamma}}{T}\,n,
\end{equation}
where $c_{\delta,\gamma} > 0$ depends only on $\delta$ and $\gamma$.
\end{lemma}

\begin{proof}
From \eqref{eq:theta-ode}:
\begin{align*}
  \theta_n'(x)
  &= \wncl(x)\left[1
     - \frac{p'(x)}{2p(x)\wncl(x)}\sin\theta_n\cos\theta_n
     + \frac{c^2x/T^2}{p(x)\wncl(x)^2}\sin^2\theta_n\right].
\end{align*}
In $I_\delta$: $p(x) \ge \delta(2-\delta)$, $|p'(x)| = 2|x|/T^2 \le 2/T$,
and $\wncl(x) \ge c_{\delta,\gamma}n/T$ (from $\chi_n \asymp n^2$ and
boundedness of $c^2x^2/T^2$ away from $\chi_n$).
The correction term satisfies:
\[
  \left|\frac{p'}{2p\wncl}\sin\theta\cos\theta\right|
  \le \frac{|p'|}{2p\wncl}
  \le \frac{1/T}{\delta(2-\delta)\cdot c_{\delta,\gamma}n/T}
  = \frac{1}{c_{\delta,\gamma}\delta(2-\delta)\,n}
  \le \frac{1}{4}
\]
for $n \ge n_0(\delta,\gamma)$. The $\sin^2$ term is similarly small.
Hence $\theta_n'(x) \ge (1/2)\wncl(x) \ge (c_{\delta,\gamma}/T)n$.
\end{proof}

\begin{remark}[No stationary points in the bulk]
Lemma~\ref{lem:phase-monotone} asserts that $\theta_n$ has
\emph{no stationary points} in $I_\delta$.
This is the key property for the integration-by-parts argument
in Section~\ref{sec:oscillation}.
It fails at the turning points $x_\pm(n)$ where $\wncl(x_\pm) = 0$;
this is why the bulk cutoff of Section~\ref{sec:prelim} is necessary.
\end{remark}

\section{Oscillation Control Lemma}\label{sec:oscillation}
%%=================================================================

This is the technical heart of the paper.

\begin{lemma}[Pr\"ufer oscillation control]\label{lem:prufer-oscillation-control}
Let $h \in C^1(I_\delta)$. Then:
\begin{equation}\label{eq:osc-control}
  \left|\int_{I_\delta} h(x)\cos(2\theta_n(x))\,dx\right|
  \le \frac{1}{c_{\delta,\gamma}\,n}
      \left(\frac{2\|h\|_{L^\infty(I_\delta)}}{T}
        + \|h'\|_{L^1(I_\delta)}\right).
\end{equation}
\end{lemma}

\begin{proof}
Since $\theta_n' \ge (c_{\delta,\gamma}/T)n > 0$ on $I_\delta$
(Lemma~\ref{lem:phase-monotone}),
we may substitute $u = 2\theta_n(x)$ and integrate by parts:
\begin{equation}\label{eq:ibp1}
  \int_{I_\delta} h(x)\cos(2\theta_n(x))\,dx
  = \int_{I_\delta} h(x)\cdot\frac{\cos(2\theta_n(x))\cdot 2\theta_n'(x)}{2\theta_n'(x)}\,dx
  = \int_{I_\delta} \frac{h(x)}{2\theta_n'(x)}\,d(\sin(2\theta_n(x))).
\end{equation}
Integrating by parts (boundary $I_\delta = [-(1-\delta)T,(1-\delta)T]$):
\begin{equation}\label{eq:ibp2}
  \int_{I_\delta} \frac{h}{2\theta_n'}\,d(\sin(2\theta_n))
  = \left[\frac{h(x)\sin(2\theta_n(x))}{2\theta_n'(x)}\right]_{-(1-\delta)T}^{(1-\delta)T}
    - \int_{I_\delta} \sin(2\theta_n(x))\cdot\frac{d}{dx}\!\left(\frac{h(x)}{2\theta_n'(x)}\right)\,dx.
\end{equation}

\medskip
\noindent\textit{Boundary term.}
Since $|\sin(2\theta_n)| \le 1$ and $\theta_n'(\pm(1-\delta)T) \ge (c_{\delta,\gamma}/T)n$:
\begin{equation}\label{eq:boundary}
  \left|\left[\frac{h\sin(2\theta_n)}{2\theta_n'}\right]\right|
  \le \frac{\|h\|_{L^\infty}}{\theta_n'(\pm(1-\delta)T)}
  \le \frac{T\,\|h\|_{L^\infty}}{c_{\delta,\gamma}\,n}.
\end{equation}

\medskip
\noindent\textit{Interior integral.}
We expand:
\begin{equation}\label{eq:expand-deriv}
  \frac{d}{dx}\!\left(\frac{h}{2\theta_n'}\right)
  = \frac{h'}{2\theta_n'} - \frac{h\,\theta_n''}{2(\theta_n')^2}.
\end{equation}
For the first summand:
\begin{equation}\label{eq:int1}
  \left|\int_{I_\delta}\sin(2\theta_n)\cdot\frac{h'}{2\theta_n'}\,dx\right|
  \le \frac{1}{2}\cdot\frac{T}{c_{\delta,\gamma}\,n}\int_{I_\delta}|h'(x)|\,dx
  = \frac{T\,\|h'\|_{L^1}}{2c_{\delta,\gamma}\,n}.
\end{equation}
For the second summand, we need a bound on $|\theta_n''/(\theta_n')^2|$.
From \eqref{eq:theta-ode}, differentiating:
$\theta_n'' = (\wncl)' + O(1)$
(the correction terms contribute $O(n)\cdot O(1/n) = O(1)$ to $\theta_n'$,
so their $x$-derivatives contribute $O(n^{-1})\cdot O(n) = O(1)$ to $\theta_n''$).
Hence $|\theta_n''| \le C_\delta(|(\wncl)'| + 1)$.
Since $\wncl(x) = \sqrt{(\chi_n - c^2x^2/T^2)/p(x)}$,
a direct calculation gives $|(\wncl)'(x)| \le C_\delta\,\chi_n/n = O(n)$
(both $\chi_n \asymp n^2$ and $(\wncl) \asymp n$ contribute).
Therefore $|\theta_n''| = O(n)$, and:
\begin{equation}\label{eq:thetapp-bound}
  \frac{|\theta_n''|}{(\theta_n')^2}
  \le \frac{C_\delta\,n}{(c_{\delta,\gamma}n/T)^2}
  = \frac{C_\delta T^2}{c_{\delta,\gamma}^2\,n}.
\end{equation}
Hence:
\begin{equation}\label{eq:int2}
  \left|\int_{I_\delta}\sin(2\theta_n)\cdot\frac{h\,\theta_n''}{2(\theta_n')^2}\,dx\right|
  \le \frac{C_\delta T^2}{2c_{\delta,\gamma}^2\,n}\cdot |I_\delta|\cdot\|h\|_{L^\infty}
  \le \frac{C_\delta' T\,\|h\|_{L^\infty}}{c_{\delta,\gamma}^2\,n}.
\end{equation}

\medskip
\noindent\textit{Assembly.}
Combining \eqref{eq:boundary}, \eqref{eq:int1}, \eqref{eq:int2}:
\begin{equation*}
  \left|\int_{I_\delta} h\cos(2\theta_n)\,dx\right|
  \le \frac{1}{n}\left(\frac{(1+C_\delta')T}{c_{\delta,\gamma}}\,\|h\|_{L^\infty}
    + \frac{T}{2c_{\delta,\gamma}}\,\|h'\|_{L^1}\right).
\end{equation*}
Renaming the constants gives \eqref{eq:osc-control}.
\end{proof}

\begin{remark}[Sharpness]
The bound \eqref{eq:osc-control} is sharp in $n$:
for $h \equiv 1$ and a constant-frequency approximation
$\theta_n(x) \approx nx/T$, the left side
$|\int\cos(2nx/T)\,dx| = O(1/n)$.
The $\|h'\|$ term accounts for the non-constancy of the weight;
if $h$ is constant, only the boundary term contributes.
\end{remark}

\begin{corollary}[Oscillation control with slowly varying weight]%
\label{cor:osc-slowly-varying}
If $h(x) = a(x)b_n(x)$ where $a \in C^1(I_\delta)$ and
$b_n$ is a slowly varying function satisfying
$\|b_n\|_{L^\infty} \le C$,
$\|b_n'\|_{L^\infty} \le C/n$ (e.g., $b_n = r_n^2$, see Lemma~\ref{lem:amplitude-drift}),
then:
\begin{equation}\label{eq:osc-sv}
  \left|\int_{I_\delta} a(x)b_n(x)\cos(2\theta_n(x))\,dx\right|
  = O\!\left(\frac{\|a\|_{C^1}\,\|b_n\|_{L^\infty}}{n}\right).
\end{equation}
\end{corollary}

\begin{proof}
Apply Lemma~\ref{lem:prufer-oscillation-control} with $h = ab_n$:
$\|h\|_{L^\infty} \le \|a\|_{L^\infty}\|b_n\|_{L^\infty}$
and
$\|h'\|_{L^1} \le (\|a'\|_{L^\infty}\|b_n\|_{L^\infty} + \|a\|_{L^\infty}\|b_n'\|_{L^\infty})\cdot|I_\delta|
 = O(\|a\|_{C^1}\|b_n\|_{L^\infty})$.
\end{proof}

\section{Amplitude Drift Control}\label{sec:amplitude}
%%=================================================================

We now control how much the Pr\"ufer amplitude $r_n$ drifts
from its WKB value.

\begin{definition}[WKB amplitude]\label{def:wkb-amp}
Define the WKB (reference) amplitude:
\begin{equation}\label{eq:wkb-amp}
  r_n^{\mathrm{WKB}}(x)
  := \frac{r_n(x_0)}{\sqrt{\wncl(x)/\wncl(x_0)}}
  \quad\text{for a fixed } x_0 \in I_\delta,
\end{equation}
corresponding to $r_n'/r_n = -(\wncl)'/2\wncl + p'/(4p)$ with
no $\cos(2\theta)$ term (the ``averaged'' Pr\"ufer ODE).
\end{definition}

\begin{lemma}[Amplitude drift]\label{lem:amplitude-drift}
For $n \le \gamma N$ and $x \in I_\delta$:
\begin{equation}\label{eq:amp-drift}
  \left|\frac{r_n(x)^2}{r_n^{\mathrm{WKB}}(x)^2} - 1\right|
  \le \frac{C_{\delta,\gamma}}{n}.
\end{equation}
Consequently, $\|(r_n^2)' \|_{L^\infty(I_\delta)} \le C_{\delta,\gamma}\,r_n(x_0)^2/n$.
\end{lemma}

\begin{proof}
From the Pr\"ufer amplitude ODE \eqref{eq:r-ode}:
\begin{equation}\label{eq:r-ode2}
  (\log r_n)'(x)
  = \underbrace{\frac{(\wncl)'}{2\wncl} + \frac{p'}{4p}}_{=: A(x)}
    + \underbrace{\left(\frac{(\wncl)'}{2\wncl} + \frac{p'}{4p}\right)\cos(2\theta_n(x))}_{=: B_n(x)}.
\end{equation}
Note: the first term $A(x)$ alone gives the WKB amplitude;
the second term $B_n(x)$ is the oscillatory correction.

\medskip
\noindent\textit{Step 1: Bound on $B_n$ after integration.}
Let $G(x) := (\wncl)'/(2\wncl) + p'/(4p) \in C^1(I_\delta)$.
Then $B_n(x) = G(x)\cos(2\theta_n(x))$.
By Lemma~\ref{lem:prufer-oscillation-control} applied to $h = G$:
\begin{equation}\label{eq:B-int}
  \left|\int_{x_0}^x B_n(t)\,dt\right|
  = \left|\int_{x_0}^x G(t)\cos(2\theta_n(t))\,dt\right|
  \le \frac{C_{\delta,\gamma}}{n}
      \left(\|G\|_{L^\infty} + \|G'\|_{L^1}\right)
  \le \frac{C_{\delta,\gamma}'}{n},
\end{equation}
where we use $|G(x)| \le C_\delta/n$ on $I_\delta$
(since $(\wncl)'/\wncl = O(1/n)$ when $\wncl \asymp n$
and $(\wncl)' \asymp 1/n\cdot n = O(1)$; more precisely
$(\wncl)' = O(n)/\wncl^2 \cdot \wncl = O(1)$, so
$G(x) = O(1)/O(n) = O(1/n)$).
Thus $\int B_n = O(1/n)$.

\medskip
\noindent\textit{Step 2: Exponentiation.}
Integrate \eqref{eq:r-ode2} from $x_0$ to $x$:
\[
  \log r_n(x) - \log r_n(x_0)
  = \int_{x_0}^x A(t)\,dt + \int_{x_0}^x B_n(t)\,dt
  = \log\frac{r_n^{\mathrm{WKB}}(x)}{r_n(x_0)} + O(1/n).
\]
Exponentiating: $r_n(x)/r_n^{\mathrm{WKB}}(x) = e^{O(1/n)} = 1 + O(1/n)$,
so $r_n(x)^2/r_n^{\mathrm{WKB}}(x)^2 = 1 + O(1/n)$.

For the derivative bound: $(r_n^2)' = 2r_n r_n' = 2r_n^2(\log r_n)' = 2r_n^2(A + B_n)$.
Since $|A|, |G| = O(1/n)$ and $r_n^2 \asymp r_n(x_0)^2$
(from the $1+O(1/n)$ bound), we get
$|(r_n^2)'| \le C_{\delta,\gamma}\,r_n(x_0)^2/n$.
\end{proof}

\begin{remark}[The coupled structure]
The key step in Lemma~\ref{lem:amplitude-drift} is using
Lemma~\ref{lem:prufer-oscillation-control} on the function
$G(x) = (\wncl)'/(2\wncl) + p'/(4p)$, which is the coefficient
of $\cos(2\theta_n)$ in the amplitude ODE.
The crucial observation is that $G(x) = O(1/n)$ in $I_\delta$,
so the oscillation control gives $\int B_n = O(1/n)\cdot O(1/n)$
from \eqref{eq:osc-control} --- actually $O(1/n)\cdot O(\|G\|/n)$.
This coupled structure (amplitude drift controlled by its own
oscillation lemma) is what makes the argument circular-free:
the oscillation control is purely a consequence of phase
monotonicity, not of amplitude control.
\end{remark}

\section{Normalization Matching}\label{sec:normalization}
%%=================================================================

\begin{lemma}[WKB normalization]\label{lem:normalization}
Let $K_n$ be as in Definition~\ref{def:rho-cl}. Then:
\begin{equation}\label{eq:norm-match}
  \frac{1}{2}\int_{I_\delta} r_n^{\mathrm{WKB}}(x)^2\,dx
  = \lambda_n + O(e^{-\beta_\gamma n}\lambda_n),
\end{equation}
and consequently
\begin{equation}\label{eq:rn-vs-rhocl}
  \frac{r_n^{\mathrm{WKB}}(x)^2}{2} = \lambda_n\,\rhocl(x)\,(1 + O(1/n)).
\end{equation}
\end{lemma}

\begin{proof}
\textit{Step 1: Identity for $\int r_n^2$.}
From $\psin^2 = r_n^2\sin^2\theta_n = r_n^2(1/2 - \cos(2\theta_n)/2)$:
\[
  \int_{I_\delta}\psin^2\,dx
  = \frac{1}{2}\int_{I_\delta} r_n^2\,dx
  - \frac{1}{2}\int_{I_\delta} r_n^2\cos(2\theta_n)\,dx.
\]
By Corollary~\ref{cor:osc-slowly-varying} with $a \equiv 1$, $b_n = r_n^2$:
$\int r_n^2\cos(2\theta_n) = O(r_n(x_0)^2\cdot|I_\delta|/n) = O(\lambda_n/n)$.
By Lemma~\ref{lem:turning-point-cutoff}:
$\int_{I_\delta}\psin^2 = \lambda_n + O(e^{-\beta n}\lambda_n)$.
Combining: $\frac{1}{2}\int_{I_\delta}r_n^2 = \lambda_n + O(\lambda_n/n)$.
By Lemma~\ref{lem:amplitude-drift}: $\int r_n^2 = \int r_n^{\mathrm{WKB},2}(1+O(1/n))$,
giving \eqref{eq:norm-match}.

\textit{Step 2: Pointwise identification.}
The WKB amplitude $r_n^{\mathrm{WKB}}(x)^2$ is proportional to
$1/\wncl(x)$ by definition \eqref{eq:wkb-amp}:
$r_n^{\mathrm{WKB}}(x)^2 = r_n(x_0)^2\cdot(\wncl(x_0)/\wncl(x))$.
Integrating: $\int_{I_\delta}r_n^{\mathrm{WKB},2} = r_n(x_0)^2\wncl(x_0)\int_{I_\delta}1/\wncl$.
From Step~1, $r_n(x_0)^2\wncl(x_0) = 2\lambda_n/(\int_{I_\delta}1/\wncl) + O(\lambda_n/n)$.
Thus:
$r_n^{\mathrm{WKB}}(x)^2/2 = \lambda_n/((\int_{I_\delta}1/\wncl)\cdot\wncl(x)) + O(\lambda_n/n)$.
By Definition~\ref{def:rho-cl}: $(\int_{x_-(n)}^{x_+(n)}1/\wncl)^{-1}/\wncl(x) = \pi\,\rhocl(x)\cdot K_n^{-1}$.
Using Remark~\ref{rem:period} ($K_n \sim \pi/\lambda_n$):
$r_n^{\mathrm{WKB}}(x)^2/2 \approx \lambda_n\,\rhocl(x)$,
giving \eqref{eq:rn-vs-rhocl} after accounting for the $I_\delta$
vs.\ $[x_-(n),x_+(n)]$ discrepancy (exponentially small).
\end{proof}

\begin{remark}[The normalization circle]
Lemma~\ref{lem:normalization} uses the oscillation control
(Corollary~\ref{cor:osc-slowly-varying}) to identify $\int r_n^2$,
then uses $\int\psin^2 = \lambda_n$ to pin the global constant.
This is the only place where the eigenvalue $\lambda_n$ (the
time-concentration of $\psin$) appears in the amplitude matching.
The circular dependency is broken because the oscillation control
needs only phase monotonicity, not amplitude information.
\end{remark}

\section{Proof of the Main Theorem}\label{sec:weaklimit}
%%=================================================================

\begin{theorem}[Semiclassical equidistribution of PSWF densities]%
\label{thm:weak-limit}
For $n \le \gamma N$ and every $f \in C^1([-T,T])$:
\begin{equation}\label{eq:weak-limit}
  \int_{-T}^T f(x)\,\psin(x)^2\,dx
  = \lambda_n(c)\int_{-T}^T f(x)\,\rhocl(x)\,dx
    + R_n(f),
\end{equation}
where:
\begin{equation}\label{eq:remainder}
  |R_n(f)| \le \frac{C_{\delta,\gamma}\,\lambda_n(c)}{n}
               \left(\|f\|_{L^\infty} + T\,\|f'\|_{L^\infty}\right).
\end{equation}
The constant $C_{\delta,\gamma}$ is explicit
(depending on $\delta, \gamma$, $c$, $T$, but not on $n$ or $f$).
\end{theorem}

\begin{proof}
\textit{Step 1: Bulk reduction.}
By Lemma~\ref{lem:turning-point-cutoff}:
\[
  \int_{-T}^T f\psin^2 = \int_{I_\delta}f\psin^2 + O(e^{-\beta n}\lambda_n\|f\|_{L^\infty}).
\]
The exponential error is $\ll \lambda_n/n$ for large $n$.

\textit{Step 2: Pr\"ufer decomposition.}
On $I_\delta$, write $\psin^2 = r_n^2\sin^2\theta_n = \frac{1}{2}r_n^2 - \frac{1}{2}r_n^2\cos(2\theta_n)$.
Thus:
\begin{equation}\label{eq:decomp}
  \int_{I_\delta}f\psin^2
  = \frac{1}{2}\int_{I_\delta} f\,r_n^2
  - \frac{1}{2}\int_{I_\delta} f\,r_n^2\cos(2\theta_n).
\end{equation}

\textit{Step 3: Oscillation term.}
Apply Corollary~\ref{cor:osc-slowly-varying} with $a = f \in C^1$
and $b_n = r_n^2$ (which satisfies $\|b_n'\|_{L^\infty} \le C\lambda_n/n$
by Lemma~\ref{lem:amplitude-drift}):
\begin{equation}\label{eq:osc-term}
  \left|\int_{I_\delta}f\,r_n^2\cos(2\theta_n)\right|
  \le \frac{C_{\delta,\gamma}\,\lambda_n}{n}
      \left(\|f\|_{L^\infty} + T\|f'\|_{L^\infty}\right).
\end{equation}

\textit{Step 4: Mean term.}
The averaged term:
\[
  \frac{1}{2}\int_{I_\delta}f\,r_n^2
  = \frac{1}{2}\int_{I_\delta}f(x)\cdot r_n^{\mathrm{WKB}}(x)^2(1+O(1/n))\,dx
  = \lambda_n\int_{I_\delta}f\rhocl + O(\lambda_n\|f\|_{L^\infty}/n),
\]
using Lemma~\ref{lem:amplitude-drift} ($r_n^2 = r_n^{\mathrm{WKB},2}(1+O(1/n))$)
and Lemma~\ref{lem:normalization} (\eqref{eq:rn-vs-rhocl}).

\textit{Step 5: Extending to $[-T,T]$.}
$\int_{[-T,T]\setminus I_\delta}f\rhocl = O(e^{-\beta n}\|f\|_{L^\infty})$
by the same exponential concentration argument.

\textit{Assembly.}
Combining Steps 1--5:
\[
  \int_{-T}^T f\psin^2
  = \lambda_n\int_{-T}^T f\rhocl + O\!\left(\frac{\lambda_n(\|f\|_{L^\infty}+T\|f'\|_{L^\infty})}{n}\right),
\]
which is \eqref{eq:weak-limit}--\eqref{eq:remainder}.
\end{proof}

\begin{remark}[Uniformity and the constant $C_{\delta,\gamma}$]
The constant in \eqref{eq:remainder} is:
$C_{\delta,\gamma} = C_1/c_{\delta,\gamma} + C_2/c_{\delta,\gamma}^2$
where $c_{\delta,\gamma}$ is the phase lower bound of
Lemma~\ref{lem:phase-monotone} and $C_1, C_2$ depend on
$\|p'\|, \|(\wncl)'\|$ etc. (bounded by $C(c,T,\delta,\gamma)$).
The bound degrades as $\delta \to 0$ or $\gamma \to 1$ (bulk shrinks),
but is uniform in $n$ and $f$ for fixed $\delta, \gamma$.
\end{remark}

\begin{corollary}[Closure of the bulk program]
\label{cor:bulk-program-closed}
Theorem~\ref{thm:weak-limit} supplies hypothesis~(6.1) of
Lemma~6.3 (\texttt{lem:bulk-reduction}) of \cite{PaperIII}
with $\omega_n = C_{\delta,\gamma}/n$.
Therefore, by \cite{PaperIII} Lemma~6.3 and
Problems~6.2--6.3 (\texttt{prob:comm-refined}, \texttt{prob:cg}):
\begin{enumerate}[(i)]
  \item The energy equidistribution $\mathbb{E}_{\rho_{mn}}[x^2]
  \approx \mathbb{E}_{\rho_m}[x^2]$ holds with error $O(1/n)$.
  \item For $m,n \le \gamma N$:
  $|\mathbb{E}_{mn}[\chi_k]/\mu_{mn} - 1| = O(1/n)$.
  \item The full bulk spectral tail bound
  $\|(I-P_N)f_{mn}\|_{L^2} \le C_\gamma e^{-\alpha_\gamma N}$
  holds, conditional only on Assumption~2.4
  (bulk convolution decay) of \cite{PaperIII}.
\end{enumerate}
\end{corollary}

\section{Open Problems}\label{sec:open}
%%=================================================================

\begin{enumerate}[(P1)]
  \item \textbf{Explicit constant $K_n$.}
  Remark~\ref{rem:period} uses $K_n \sim \pi/\lambda_n$ without
  proof. This follows from the semiclassical Bohr--Sommerfeld
  quantization $\oint p\,dq = 2\pi n$, translated to $D_c$;
  an explicit proof for PSWFs using the integral formula for
  $\lambda_n(c)$ from \cite{Slepian1978} would close this gap.
  \item \textbf{Extension to $f \in C^0$.}
  Theorem~\ref{thm:weak-limit} requires $f \in C^1$.
  For $f \in C^0$, a density argument using $C^1$ approximation
  gives weak-* convergence without a rate.
  A rate for $f \in C^\alpha$ ($0 < \alpha < 1$) via
  interpolation would extend the result.
  \item \textbf{Off-diagonal regime.}
  The present paper handles the bulk ($m,n \le \gamma N$).
  Whether an analogue of Theorem~\ref{thm:weak-limit} holds
  for the off-diagonal regime ($|m-n| \ge \delta N$) is open;
  the classical density argument would need to be
  adapted for the product density $\rho_{mn}$ rather than
  the single-index density $\rho_m$.
\end{enumerate}
