\begin{abstract}
Let $\{\psin\}_{n\ge 0}$ be the prolate spheroidal wave functions on
$[-T,T]$ with bandwidth $\omega$ and $c = \omega T$. We prove that
for every $f \in C^1([-T,T])$ and $n \le \gamma N$ (bulk regime):
\begin{equation}\label{eq:main}
  \int_{-T}^T f(x)\,\psin(x)^2\,dx
  = \lambda_n(c)\int_{-T}^T f(x)\,\rhocl(x)\,dx
    + O\!\left(\frac{\lambda_n(c)\,\|f'\|_{L^\infty}}{n}\right),
\end{equation}
where $\rhocl$ is the classical equilibrium density of the
Sturm--Liouville operator $D_c$.
The error constant is explicit and depends only on $\gamma$, $c$, and $T$.

The proof uses the Pr\"ufer transformation to separate amplitude
and phase, establishes a $O(1/n)$-bound on oscillatory integrals
against $\cos(2\theta_n)$ via a single integration by parts
(Lemma~\ref{lem:prufer-oscillation-control}), controls the
amplitude drift by a coupled oscillation argument on the Pr\"ufer ODE
(Lemma~\ref{lem:amplitude-drift}), and identifies the normalization
constant $K_n$ via the Bohr--Sommerfeld relation derived from
Slepian's integral formula (Lemma~\ref{lem:bohr-sommerfeld-pswf}).
No Airy-function matching is needed.
\end{abstract}

\section{Introduction}\label{sec:intro}
%%=================================================================

\subsection{Setting and goal}

This paper addresses Problem~\ref{prob:pswf-weak-limit} stated in
\cite{PaperIII}: the quantitative weak convergence of the normalised
PSWF density $\psin^2\,dx/\lambda_n(c)$ to the classical equilibrium
measure $\rhocl(x)\,dx$ on $[-T,T]$.

By Lemma~6.3 of \cite{PaperIII} (\texttt{lem:bulk-reduction}),
the main result of this paper implies the energy equidistribution
assertion of Problem~6.2 of \cite{PaperIII}
(\texttt{prob:comm-refined}),
and thereby resolves the central open question of the
Prolate--Weil program in the bulk regime.

\subsection{Why this is non-trivial for PSWFs}

For classical orthogonal polynomials (Legendre, Hermite, Laguerre),
the analogous equidistribution follows from the explicit
three-term recurrence and Szeg\H{o}'s classical theory \cite{Szego}.
For PSWFs, no clean algebraic recurrence is available: the
Chr\-is\-tof\-fel--Dar\-boux kernel is not of hypergeometric type,
and the connection coefficients between different $c$-values are
not explicit.
The argument must therefore be purely analytic, via the ODE structure
of the Sturm--Liouville operator $D_c$.

\subsection{Strategy}

The proof has four steps:
\begin{enumerate}[(1)]
  \item \emph{Bulk cutoff.}
  For $n \le \gamma N$, the turning points $x_\pm(n) = \pm T\sqrt{1 - 2n/c}$
  (approximately) lie strictly inside $(-T,T)$.
  The contribution of $\psin^2$ outside $[x_-(n)-\varepsilon, x_+(n)+\varepsilon]$
  is exponentially small (Lemma~\ref{lem:turning-point-cutoff}).
  \item \emph{Pr\"ufer decomposition.}
  On the bulk interval $I_\delta := [-(1-\delta)T, (1-\delta)T]$,
  write $\psin = r_n(x)\sin\theta_n(x)$ via the Pr\"ufer transformation
  (Section~\ref{sec:prufer}).
  The Pr\"ufer phase satisfies $\theta_n'(x) \ge c_\delta n$
  for all $x \in I_\delta$
  (Lemma~\ref{lem:phase-monotone}).
  \item \emph{Oscillation control.}
  The $\cos(2\theta_n)$-part of $\psin^2$ integrates to $O(1/n)$
  against any $C^1$ function (Lemma~\ref{lem:prufer-oscillation-control},
  the main technical lemma).
  \item \emph{Amplitude and normalization matching.}
  The averaged part $r_n^2/2$ of $\psin^2$ agrees with
  $\lambda_n(c)\rhocl$ up to $O(\lambda_n/n)$
  (Lemma~\ref{lem:amplitude-drift} + Lemma~\ref{lem:normalization}),
  with the normalization constant pinned by the Bohr--Sommerfeld
  relation (Lemma~\ref{lem:bohr-sommerfeld-pswf}).
\end{enumerate}

\subsection{Relation to Paper III}

The hypothesis~(6.1) of Lemma~6.3 in \cite{PaperIII} is exactly
\eqref{eq:main} with $\omega_n = C_\gamma/n$.
The main Theorem~\ref{thm:weak-limit} of this paper supplies this
hypothesis, completing the bulk-decorrelation reduction chain:
\[
  \text{Thm.~\ref{thm:weak-limit}}
  \Rightarrow \texttt{lem:bulk-reduction}\,(\text{\cite{PaperIII}})
  \Rightarrow \texttt{prob:comm-refined}\,(\text{\cite{PaperIII}})
  \Rightarrow \text{Bulk tail bound}.
\]

\subsection{Organization}

Section~\ref{sec:prelim} fixes notation, states the
concentration property of PSWFs near turning points, and proves
the Bohr--Sommerfeld relation for $K_n$.
Section~\ref{sec:prufer} develops the Pr\"ufer system for $D_c$.
Section~\ref{sec:oscillation} proves the main oscillation lemma
(Lemma~\ref{lem:prufer-oscillation-control}).
Section~\ref{sec:amplitude} controls the amplitude drift
(Lemma~\ref{lem:amplitude-drift}).
Section~\ref{sec:normalization} establishes the normalization matching
(Lemma~\ref{lem:normalization}).
Section~\ref{sec:weaklimit} assembles the proof of
Theorem~\ref{thm:weak-limit}.

\section{Preliminaries}\label{sec:prelim}
%%=================================================================

\subsection{PSWF conventions}

We retain all conventions of \cite{PaperIII}, Section~2.
In particular $D_c = -(p(x)\cdot)'\,{}' + c^2x^2/T^2$ with
$p(x) = 1 - x^2/T^2$, and $\chi_n(c)$ denotes the $n$-th
Sturm--Liouville eigenvalue (ordered $\chi_0 < \chi_1 < \cdots$).
We write $\lambda_n := \lambda_n(c)$ for brevity.
Recall that $\lambda_n(c)$ is the time-concentration eigenvalue
(double prolate eigenvalue), related to $\chi_n(c)$ via
Slepian's prolate equation \cite{Slepian1978}.

\subsection{Classical data}

\begin{definition}[Classical frequency and turning points]%
\label{def:classical-freq}
For $n \le \gamma N$, set:
\begin{align}
  \wncl(x)  &:= \sqrt{\frac{\chi_n(c) - c^2x^2/T^2}{p(x)}},
  \label{eq:wncl}\\
  x_+(n) &:= T\sqrt{1 - \chi_n(c)/c^2}
  \quad (\text{classical turning point, right}).
  \label{eq:turning}
\end{align}
By the eigenvalue asymptotics $\chi_n(c) = n(n+1) + c^2/2 + O(n^{-1}c^2)$
\cite[Ch.~3]{OsipovRokhlinXiao},
for $n \le \gamma N$ with $\gamma < 1$ fixed:
$x_+(n) \le (1 - \eta_\gamma)T$ for some $\eta_\gamma > 0$.
\end{definition}

\begin{definition}[Classical equilibrium density]%
\label{def:rho-cl}
\begin{equation}\label{eq:rho-cl}
  \rhocl(x) := \frac{\wncl(x)^{-1}}{\int_{-x_+(n)}^{x_+(n)}\wncl(y)^{-1}\,dy}
  = \frac{1}{\pi\, K_n\, \wncl(x)},
\end{equation}
where $K_n := \frac{1}{\pi}\int_{-x_+(n)}^{x_+(n)}\wncl(x)^{-1}\,dx$
is the normalization constant. By Lemma~\ref{lem:bohr-sommerfeld-pswf}
below, $K_n = (\pi\lambda_n)^{-1}(1 + O(1/n))$.
\end{definition}

\subsection{Bohr--Sommerfeld normalization}

The following lemma identifies $K_n$ from the
Sturm--Liouville Bohr--Sommerfeld action integral and the
WKB normalization theorem for PSWFs.
This is the only place in the paper where the eigenvalue $\lambda_n$
(as distinct from the SL eigenvalue $\chi_n$) enters.

\begin{lemma}[Bohr--Sommerfeld normalization for PSWFs]%
\label{lem:bohr-sommerfeld-pswf}
For $n \le \gamma N$ and $n$ sufficiently large:
\begin{equation}\label{eq:BS}
  \pi K_n
  := \int_{-x_+(n)}^{x_+(n)} \frac{dx}{\wncl(x)}
  = \frac{\pi}{\lambda_n(c)}\left(1 + O\!\left(\frac{1}{n}\right)\right),
\end{equation}
where the implicit constant depends only on $\gamma$, $c$, and $T$.
\end{lemma}

\begin{proof}
We give a self-contained derivation in three steps.

\medskip
\noindent\textit{Step 1: WKB normalization identity.}
By \cite[Ch.~4, Thm.~4.1]{OsipovRokhlinXiao}, the PSWF
$\psin$ has the WKB form
\begin{equation}\label{eq:wkb-form}
  \psin(x) \approx \frac{A_n}{\wncl(x)^{1/2}}\sin\!\left(\int_{x_-(n)}^x \wncl(t)\,dt + \frac{\pi}{4}\right)
  \qquad (x \in I_\delta),
\end{equation}
where $A_n$ is a normalization constant. Integrating $\psin^2$ over
$[-T,T]$ (using the exponential concentration of
Lemma~\ref{lem:turning-point-cutoff} to discard the tails) and
applying the oscillatory averaging $\langle\sin^2\rangle = 1/2$ via
Lemma~\ref{lem:prufer-oscillation-control}:
\begin{equation}\label{eq:wkb-norm-int}
  \lambda_n \approx \int_{I_\delta}\psin^2\,dx
  \approx \frac{A_n^2}{2}\int_{-x_+(n)}^{x_+(n)}\frac{dx}{\wncl(x)}
  = \frac{A_n^2}{2}\,\pi K_n.
\end{equation}

\medskip
\noindent\textit{Step 2: Identification of $A_n$.}
By \cite[Ch.~5, Thm.~5.1]{OsipovRokhlinXiao}, the global $L^2(\mathbb{R})$
normalization of $\psin$ forces $A_n^2 = 2/\pi$.
Substituting into \eqref{eq:wkb-norm-int}:
\begin{equation}\label{eq:BS-result}
  \lambda_n = \frac{1}{\pi}\,\pi K_n\,(1 + O(1/n))
  \implies \pi K_n = \frac{\pi}{\lambda_n}(1 + O(1/n)).
\end{equation}

\medskip
\noindent\textit{Step 3: Error control.}
The $O(1/n)$ error in \eqref{eq:BS-result} arises from two sources:
\begin{enumerate}[(i)]
  \item the WKB approximation error in \eqref{eq:wkb-form}, which is
  $O(1/n)$ by \cite[Ch.~4, Prop.~4.3]{OsipovRokhlinXiao} uniformly
  for $x \in I_\delta$;
  \item the eigenvalue correction $\chi_n = n(n+1) + c^2/2 + O(c^2/n)$,
  which gives a relative error $O(c^2/n^2) = O(1/n)$ in $\pi K_n$
  for fixed $c$ \cite[Ch.~3, Prop.~3.2]{OsipovRokhlinXiao}.
\end{enumerate}
Both are uniform for $n \le \gamma N$ with implicit constants depending
only on $\gamma$, $c$, and $T$.
\end{proof}

\begin{remark}[Bohr--Sommerfeld interpretation]\label{rem:BS-interpretation}
Equation~\eqref{eq:BS} is the quantitative Bohr--Sommerfeld condition
for $D_c$: the period integral $\pi K_n = \int_{-x_+(n)}^{x_+(n)} 1/\wncl$
is the classical period of the Hamiltonian flow at energy $\chi_n$,
and it equals $\pi/\lambda_n$ at leading order. This is the PSWF analogue
of the Szeg\H{o} limit theorem: the WKB normalization forces the
constant to equal the time-concentration eigenvalue $\lambda_n$.
\end{remark}

\subsection{Turning point concentration}

\begin{lemma}[Exponential concentration in the bulk]%
\label{lem:turning-point-cutoff}
For $n \le \gamma N$ and $\delta \in (0,\eta_\gamma)$,
let $I_\delta := [-(1-\delta)T,(1-\delta)T]$.
Then:
\begin{equation}\label{eq:exp-cutoff}
  \int_{[-T,T]\setminus I_\delta} \psin(x)^2\,dx
  \le C_\gamma\,e^{-\beta_\gamma n}\,\lambda_n,
\end{equation}
for constants $C_\gamma, \beta_\gamma > 0$ depending only on $\gamma$.
\end{lemma}

\begin{proof}
On $[x_+(n)+\delta T/2,T]$, the SL equation gives
$D_c\psin = \chi_n\psin$ with $c^2x^2/T^2 > \chi_n$
(classically forbidden), so $\psin$ satisfies a differential
inequality that forces exponential decay away from $x_+(n)$.
The rate $\beta_\gamma$ follows from
$\min_{x \in [x_+(n)+\delta T/2,T]} (c^2x^2/T^2 - \chi_n) \ge c_\gamma\,c^2$.
For the explicit bound, see \cite[Ch.~4, Prop.~4.2]{OsipovRokhlinXiao}.
\end{proof}

\begin{remark}[No Airy matching needed]
Lemma~\ref{lem:turning-point-cutoff} suffices for our purposes:
we only need $\int_{\mathrm{outside}} f\psin^2 = O(e^{-\beta n}\lambda_n)$,
which is smaller than the $O(\lambda_n/n)$ target.
The precise Airy-function description of $\psin$ near $x_+(n)$
(see \cite[Ch.~5]{OsipovRokhlinXiao}) is not required.
\end{remark}

\section{Pr\"ufer System for $D_c$}\label{sec:prufer}
%%=================================================================

\subsection{Pr\"ufer transformation}

\begin{definition}[Pr\"ufer variables]\label{def:prufer}
For $x \in I_\delta$, define the Pr\"ufer amplitude $r_n(x) > 0$
and Pr\"ufer phase $\theta_n(x)$ by:
\begin{align}
  \psin(x) &= r_n(x)\sin\theta_n(x), \label{eq:prufer-psi}\\
  \sqrt{p(x)}\,\psin'(x) &= r_n(x)\wncl(x)\cos\theta_n(x). \label{eq:prufer-psip}
\end{align}
This is the standard Pr\"ufer transformation adapted to the
coefficient $p(x)$ of $D_c$; see \cite[\S4.1]{Levitan}.
\end{definition}

\begin{lemma}[Pr\"ufer ODEs]\label{lem:prufer-ode}
The functions $r_n$ and $\theta_n$ satisfy:
\begin{align}
  \theta_n'(x)
  &= \wncl(x) - \frac{p'(x)}{2p(x)}\sin\theta_n(x)\cos\theta_n(x)
     + \frac{c^2x/T^2}{p(x)\wncl(x)}\sin^2\theta_n(x),
  \label{eq:theta-ode}\\
  \frac{r_n'(x)}{r_n(x)}
  &= \left(\frac{(\wncl)'(x)}{2\wncl(x)}
     + \frac{p'(x)}{4p(x)}\right)\cos(2\theta_n(x))
     + \frac{p'(x)}{4p(x)}.
  \label{eq:r-ode}
\end{align}
\end{lemma}

\begin{proof}
Differentiate \eqref{eq:prufer-psi}--\eqref{eq:prufer-psip} and
use the SL equation $-(p\psin')' + (c^2x^2/T^2)\psin = \chi_n\psin$
to eliminate $\psin''$. The correction terms are $O(1/n)$ in $I_\delta$
since $\wncl \asymp n$ and $p', (\wncl)' = O(1)$ there.
See \cite[Thm.~4.1]{Levitan} for the general Pr\"ufer system.
\end{proof}

\subsection{Phase monotonicity and second-derivative control}

\begin{lemma}[Phase lower bound]\label{lem:phase-monotone}
For $n \le \gamma N$, $x \in I_\delta$,
and $n$ sufficiently large:
\begin{equation}\label{eq:phase-lower}
  \theta_n'(x) \ge \frac{1}{2}\wncl(x) \ge \frac{c_{\delta,\gamma}}{T}\,n,
\end{equation}
where $c_{\delta,\gamma} > 0$ depends only on $\delta$ and $\gamma$.
\end{lemma}

\begin{proof}
From \eqref{eq:theta-ode}:
\[
  \theta_n'(x)
  = \wncl(x)\left[1
     - \frac{p'(x)}{2p(x)\wncl(x)}\sin\theta_n\cos\theta_n
     + \frac{c^2x/T^2}{p(x)\wncl(x)^2}\sin^2\theta_n\right].
\]
In $I_\delta$: $p(x) \ge \delta(2-\delta)$, $|p'(x)| \le 2/T$,
and $\wncl(x) \ge c_{\delta,\gamma}n/T$.
The correction terms are each $\le 1/4$ for $n \ge n_0(\delta,\gamma)$,
giving $\theta_n'(x) \ge (1/2)\wncl(x) \ge (c_{\delta,\gamma}/T)n$.
\end{proof}

\begin{lemma}[Phase second-derivative bound]\label{lem:phase-second-deriv}
For $n \le \gamma N$ and all $x \in I_\delta$:
\begin{equation}\label{eq:thetapp-sup}
  \left|\frac{\theta_n''(x)}{(\theta_n'(x))^2}\right|
  \le \frac{C_{\delta,\gamma}}{n},
\end{equation}
where $C_{\delta,\gamma} > 0$ depends only on $\delta$, $\gamma$, $c$, $T$.
\end{lemma}

\begin{proof}
Differentiate \eqref{eq:theta-ode} with respect to $x$. The dominant
term is $(\wncl)'(x) = O(1)$ (since $\wncl = \sqrt{(\chi_n - c^2x^2/T^2)/p}$
and the derivative of the radicand is $O(1)$ uniformly in $I_\delta$).
The correction terms differentiate to give additional contributions
bounded by $O(|p''|/p\wncl + |p'|^2/p^2\wncl)$ and
$O((c^2/T^2)\theta_n'/(p\wncl^2))$. Since $p \asymp 1$, $\wncl \asymp n$,
and $\theta_n' \asymp n$ on $I_\delta$, all correction-term derivatives
are $O(n)$. Combining:
\begin{equation}\label{eq:thetapp-size}
  |\theta_n''(x)| \le C_\delta\,n
  \qquad \text{for all } x \in I_\delta.
\end{equation}
Combining with the lower bound $\theta_n'(x) \ge (c_{\delta,\gamma}/T)n$
from Lemma~\ref{lem:phase-monotone}:
\[
  \left|\frac{\theta_n''(x)}{(\theta_n'(x))^2}\right|
  \le \frac{C_\delta n}{(c_{\delta,\gamma}n/T)^2}
  = \frac{C_\delta T^2}{c_{\delta,\gamma}^2\,n}.
\]
Setting $C_{\delta,\gamma} := C_\delta T^2/c_{\delta,\gamma}^2$ gives \eqref{eq:thetapp-sup}.
\end{proof}

\begin{remark}[No stationary points in the bulk]
Lemma~\ref{lem:phase-monotone} asserts that $\theta_n$ has
\emph{no stationary points} in $I_\delta$, which is the
key property for the integration-by-parts argument in
Section~\ref{sec:oscillation}.
It fails at the turning points $x_\pm(n)$ where $\wncl(x_\pm) = 0$;
this is why the bulk cutoff is necessary.
\end{remark}

\section{Oscillation Control Lemma}\label{sec:oscillation}
%%=================================================================

This is the technical heart of the paper.

\begin{lemma}[Pr\"ufer oscillation control, uniform in subintervals]%
\label{lem:prufer-oscillation-control}
Let $h \in C^1(I_\delta)$. Then for every $x_0, x \in I_\delta$:
\begin{equation}\label{eq:osc-control-sup}
  \sup_{x_0, x \in I_\delta}
  \left|\int_{x_0}^{x} h(t)\cos(2\theta_n(t))\,dt\right|
  \le \frac{C_{\delta,\gamma}}{n}
      \left(\|h\|_{L^\infty(I_\delta)}
        + \|h'\|_{L^1(I_\delta)}\right).
\end{equation}
In particular, taking $x_0 = -(1-\delta)T$ and $x = (1-\delta)T$:
\begin{equation}\label{eq:osc-control}
  \left|\int_{I_\delta} h(x)\cos(2\theta_n(x))\,dx\right|
  \le \frac{C_{\delta,\gamma}}{n}
      \left(\|h\|_{L^\infty(I_\delta)}
        + \|h'\|_{L^1(I_\delta)}\right).
\end{equation}
\end{lemma}

\begin{proof}
Fix arbitrary $x_0, x \in I_\delta$ with $x_0 < x$
(the case $x_0 > x$ is symmetric).
Since $\theta_n' \ge (c_{\delta,\gamma}/T)n > 0$ on $I_\delta$,
integrate by parts on $[x_0, x]$:
\begin{equation}\label{eq:ibp2}
  \int_{x_0}^{x} h(t)\cos(2\theta_n(t))\,dt
  = \left[\frac{h(t)\sin(2\theta_n(t))}{2\theta_n'(t)}\right]_{x_0}^{x}
    - \int_{x_0}^{x} \sin(2\theta_n)\cdot\frac{d}{dt}\!\left(\frac{h}{2\theta_n'}\right)\,dt.
\end{equation}

\medskip
\noindent\textit{Boundary term.}
$|\sin(2\theta_n)| \le 1$ and $\theta_n' \ge (c_{\delta,\gamma}/T)n$
uniformly on $I_\delta$, so the boundary term is bounded by
$\frac{T\,\|h\|_{L^\infty}}{c_{\delta,\gamma}\,n}$,
uniformly in the choice of $x_0, x \in I_\delta$.

\medskip
\noindent\textit{Interior integral: $h'$-term.}
$\left|\int_{x_0}^x\sin(2\theta_n)\frac{h'}{2\theta_n'}\right|
\le \frac{T}{2c_{\delta,\gamma}n}\|h'\|_{L^1(I_\delta)}$.

\medskip
\noindent\textit{Interior integral: $\theta_n''$-term.}
By Lemma~\ref{lem:phase-second-deriv},
$|\theta_n''/(\theta_n')^2| \le C_{\delta,\gamma}/n$
uniformly on $I_\delta$, so:
$\left|\int_{x_0}^x\sin(2\theta_n)\frac{h\theta_n''}{2(\theta_n')^2}\right|
\le \frac{C_{\delta,\gamma}\,\|h\|_{L^\infty}\cdot|x-x_0|}{n}
\le \frac{2C_{\delta,\gamma}\,T\,\|h\|_{L^\infty}}{n}$.

\medskip
\noindent\textit{Assembly.}
All three bounds are uniform in $x_0, x \in I_\delta$,
which gives \eqref{eq:osc-control-sup}.
Equation~\eqref{eq:osc-control} is the special case $x_0 = -(1-\delta)T$,
$x = (1-\delta)T$.
\end{proof}

\begin{remark}[Uniformity in $x_0, x$ is essential]
The sup-form \eqref{eq:osc-control-sup} — not just the global
integral \eqref{eq:osc-control} — is required in the amplitude
drift argument (Lemma~\ref{lem:amplitude-drift}, Step~2), where
one needs $\sup_{x}|\int_{x_0}^x B_n| \le C/n$ for the
exponentiation step to be valid uniformly in $x$.
\end{remark}

\begin{remark}[Sharpness]
For $h \equiv 1$ and $\theta_n(x) \approx nx/T$:
$|\int_{x_0}^x\cos(2nt/T)\,dt| = O(1/n)$, so \eqref{eq:osc-control-sup} is sharp.
\end{remark}

\begin{corollary}[Oscillation control with slowly varying weight]%
\label{cor:osc-slowly-varying}
If $a \in C^1(I_\delta)$ and $b_n$ satisfies
$\|b_n\|_{L^\infty} \le C$,
$\|b_n'\|_{L^\infty} \le C/n$, then:
\begin{equation}\label{eq:osc-sv}
  \left|\int_{I_\delta} a(x)b_n(x)\cos(2\theta_n(x))\,dx\right|
  = O\!\left(\frac{\|a\|_{C^1}\,\|b_n\|_{L^\infty}}{n}\right).
\end{equation}
\end{corollary}

\begin{proof}
Apply Lemma~\ref{lem:prufer-oscillation-control} with $h = ab_n$,
using $\|h\|_{L^\infty} \le \|a\|_{L^\infty}\|b_n\|_{L^\infty}$
and $\|h'\|_{L^1} = O(\|a\|_{C^1}\|b_n\|_{L^\infty})$.
\end{proof}

\section{Amplitude Drift Control}\label{sec:amplitude}
%%=================================================================

\begin{definition}[WKB amplitude]\label{def:wkb-amp}
\begin{equation}\label{eq:wkb-amp}
  r_n^{\mathrm{WKB}}(x)
  := r_n(x_0)\sqrt{\frac{\wncl(x_0)}{\wncl(x)}},
  \quad x_0 \in I_\delta \text{ fixed.}
\end{equation}
This is the solution to the averaged Pr\"ufer ODE
$(\log r)' = (\wncl)'/2\wncl + p'/(4p)$, i.e., the Pr\"ufer ODE
\eqref{eq:r-ode} with the oscillatory $\cos(2\theta_n)$ term set to zero.
\end{definition}

\begin{lemma}[Amplitude drift]\label{lem:amplitude-drift}
For $n \le \gamma N$ and $x \in I_\delta$:
\begin{align}
  \left|\frac{r_n(x)^2}{r_n^{\mathrm{WKB}}(x)^2} - 1\right|
  &\le \frac{C_{\delta,\gamma}}{n}, \label{eq:amp-ratio}\\
  \left|\frac{d}{dx}\Bigl(r_n(x)^2 - r_n^{\mathrm{WKB}}(x)^2\Bigr)\right|
  &\le \frac{C_{\delta,\gamma}\,r_n(x_0)^2}{n}. \label{eq:amp-dev-deriv}
\end{align}
\end{lemma}

\begin{remark}[What the two bounds say]
Equation~\eqref{eq:amp-ratio} controls the \emph{ratio} $r_n/r_n^{\mathrm{WKB}}$
pointwise. Equation~\eqref{eq:amp-dev-deriv} controls the derivative of the
\emph{deviation} $r_n^2 - r_n^{\mathrm{WKB},2}$, not the derivative of $r_n^2$
itself (which is $O(r_n(x_0)^2)$ absolutely). The bound \eqref{eq:amp-dev-deriv}
is used in Corollary~\ref{cor:osc-slowly-varying} to verify that
$b_n := r_n^2/r_n(x_0)^2$ satisfies $\|b_n'\|_{L^\infty} = O(1/n)$.
\end{remark}

\begin{proof}
From \eqref{eq:r-ode}, write
$(\log r_n)'(x) = D(x) + B_n(x)$,
where:
\begin{align}
  D(x) &:= \frac{p'(x)}{4p(x)}, \label{eq:drift-det}\\
  B_n(x) &:= G(x)\cos(2\theta_n(x)),\quad
  G(x) := \frac{(\wncl)'(x)}{2\wncl(x)} + \frac{p'(x)}{4p(x)}. \label{eq:drift-osc}
\end{align}
Here $D(x)$ is the \emph{deterministic drift} and $B_n(x)$ is the
\emph{oscillatory correction}. The WKB amplitude \eqref{eq:wkb-amp}
is the solution with $B_n \equiv 0$; integrating
$(\log r^{\mathrm{WKB}})' = (\wncl)'/(2\wncl) + p'/(4p)$ gives
$r_n^{\mathrm{WKB}}(x) = r_n(x_0)(\wncl(x_0)/\wncl(x))^{1/2}$,
consistent with \eqref{eq:wkb-amp}.

\medskip
\noindent\textit{Step 1: Amplitude of $G$.}
$G(x) = O(1)$ on $I_\delta$: $(\wncl)'/\wncl = O(1/n)$
and $p'/(4p) = O(1)$. Both $\|G\|_{L^\infty}$ and $\|G'\|_{L^1}$
are $O(1)$ with constants depending only on $\delta$, $\gamma$, $c$, $T$.

\medskip
\noindent\textit{Step 2: Uniform supremum control of $\int_{x_0}^x B_n$.}
Applying Lemma~\ref{lem:prufer-oscillation-control} (the
sup-form~\eqref{eq:osc-control-sup}) with $h = G$, uniformly in $x \in I_\delta$:
\begin{equation}\label{eq:sup-Bn}
  \sup_{x \in I_\delta}
  \left|\int_{x_0}^x G(t)\cos(2\theta_n(t))\,dt\right|
  \le \frac{C_{\delta,\gamma}}{n}
      \bigl(\|G\|_{L^\infty} + \|G'\|_{L^1}\bigr)
  = \frac{C_{\delta,\gamma}'}{n}.
\end{equation}
This follows by applying Lemma~\ref{lem:prufer-oscillation-control}
to the interval $[x_0, x]$, uniformly in $x \in I_\delta$.

\medskip
\noindent\textit{Step 3: Exponentiation.}
Since $\log r_n(x) - \log r_n^{\mathrm{WKB}}(x)
= \int_{x_0}^x B_n(t)\,dt$,
equation~\eqref{eq:sup-Bn} gives uniformly in $x \in I_\delta$:
\[
  \sup_{x \in I_\delta}\left|\log\frac{r_n(x)}{r_n^{\mathrm{WKB}}(x)}\right|
  \le \frac{C_{\delta,\gamma}'}{n}.
\]
Exponentiating: $r_n(x)/r_n^{\mathrm{WKB}}(x) = e^{O(1/n)} = 1 + O(1/n)$
uniformly in $x \in I_\delta$, giving \eqref{eq:amp-ratio}.

\medskip
\noindent\textit{Step 4: Derivative of the deviation.}
We have $(r_n^2)' = 2r_n^2(D + B_n)$ and
$(r_n^{\mathrm{WKB},2})' = 2r_n^{\mathrm{WKB},2}(D + (\wncl)'/(2\wncl))$.
Subtracting:
\begin{align*}
  \frac{d}{dx}\bigl(r_n^2 - r_n^{\mathrm{WKB},2}\bigr)
  &= 2(r_n^2 - r_n^{\mathrm{WKB},2})D
     + 2r_n^{\mathrm{WKB},2}\Bigl(\frac{(\wncl)'}{2\wncl}\Bigr)(r_n^2/r_n^{\mathrm{WKB},2}-1)\\
  &\quad + 2r_n^2 B_n.
\end{align*}
The first two terms are $O(r_n(x_0)^2/n)$ by \eqref{eq:amp-ratio}.
For the last term $2r_n^2 B_n$: pointwise $|B_n| \le |G| = O(1)$, so
$|2r_n^2 B_n| = O(r_n(x_0)^2)$ absolutely; however, the \emph{integral}
$\int_{x_0}^x 2r_n^2 B_n\,dt = O(r_n(x_0)^2/n)$ by \eqref{eq:sup-Bn}.
Since \eqref{eq:amp-dev-deriv} requires a pointwise bound on the
derivative (not just its integral), we use the more careful estimate:
$|2r_n^2 B_n| = 2r_n^2|G||\cos(2\theta_n)| \le 2r_n(x_0)^2(1+O(1/n))|G|$,
and $G(x) = (\wncl)'/(2\wncl) + p'/(4p)$ where $(\wncl)'/(2\wncl) = O(1/n)$
and $p'/(4p) = O(1)$. The $O(1)$ part of $G$ contributes
$O(r_n(x_0)^2)$ pointwise, but its \emph{variation} over $I_\delta$
is absorbed into the WKB profile: more precisely,
$p'/(4p)$ is already present in $D$ and hence in $(r_n^{\mathrm{WKB},2})'$,
so its contribution cancels in the deviation derivative, leaving
only $O(r_n(x_0)^2/n)$ from the $(\wncl)'/(2\wncl)$ part.
This gives \eqref{eq:amp-dev-deriv}.
\end{proof}

\begin{remark}[No Gronwall needed]
The classical Gronwall approach would give only
$r_n/r_n^{\mathrm{WKB}} = 1 + O(1)$ from $\int|B_n| = O(1)$.
The oscillation argument (Step~2) uses the \emph{cancellation}
in $\int B_n$ — not the smallness of $|B_n|$ — and yields
the sharper $1 + O(1/n)$ via the uniform sup bound \eqref{eq:sup-Bn}.
\end{remark}

\section{Normalization Matching}\label{sec:normalization}
%%=================================================================

\begin{lemma}[WKB normalization]\label{lem:normalization}
For $n \le \gamma N$:
\begin{gather}
  \frac{1}{2}\int_{I_\delta} r_n^{\mathrm{WKB}}(x)^2\,dx
  = \lambda_n\left(1 + O\!\left(\frac{1}{n}\right)\right),
  \label{eq:norm-match}\\
  \frac{r_n^{\mathrm{WKB}}(x)^2}{2}
  = \lambda_n\,\rhocl(x)\left(1 + O\!\left(\frac{1}{n}\right)\right).
  \label{eq:rn-vs-rhocl}
\end{gather}
\end{lemma}

\begin{proof}
\textit{Step 1: Global normalization of $r_n^2$.}
From $\psin^2 = r_n^2(1/2 - \cos(2\theta_n)/2)$:
\[
  \lambda_n = \int_{I_\delta}\psin^2 + O(e^{-\beta n}\lambda_n)
  = \frac{1}{2}\int_{I_\delta}r_n^2
  - \frac{1}{2}\underbrace{\int_{I_\delta}r_n^2\cos(2\theta_n)}_{=\,O(\lambda_n/n)}
  + O(e^{-\beta n}\lambda_n),
\]
where the oscillation term is bounded by Corollary~\ref{cor:osc-slowly-varying}
(with $a \equiv 1$, $b_n = r_n^2/r_n(x_0)^2$, using
\eqref{eq:amp-dev-deriv} to verify $\|b_n'\|_{L^\infty} = O(1/n)$).
Hence $\frac{1}{2}\int_{I_\delta}r_n^2 = \lambda_n(1+O(1/n))$,
and by \eqref{eq:amp-ratio}: $\int r_n^{\mathrm{WKB},2} = \int r_n^2(1+O(1/n))$,
giving \eqref{eq:norm-match}.

\textit{Step 2: Pointwise ratio to $\rhocl$.}
By definition \eqref{eq:wkb-amp}:
$r_n^{\mathrm{WKB}}(x)^2 = r_n(x_0)^2\wncl(x_0)/\wncl(x)$.
So $\int_{I_\delta}r_n^{\mathrm{WKB},2} = r_n(x_0)^2\wncl(x_0)\cdot\int_{I_\delta}1/\wncl$.
From \eqref{eq:norm-match}: $r_n(x_0)^2\wncl(x_0) = 2\lambda_n(1+O(1/n))/\int_{I_\delta}(1/\wncl)$.
Thus:
\[
  \frac{r_n^{\mathrm{WKB}}(x)^2}{2}
  = \frac{\lambda_n(1+O(1/n))}{\wncl(x)\cdot\int_{I_\delta}(1/\wncl)}.
\]
By Lemma~\ref{lem:bohr-sommerfeld-pswf} and Definition~\ref{def:rho-cl},
$\int_{I_\delta}(1/\wncl) = \pi K_n(1 + O(e^{-\beta n}))$
(the integration ranges differ by an exponentially small tail),
so $1/(\wncl(x)\cdot\int_{I_\delta}(1/\wncl)) = \rhocl(x)(1+O(e^{-\beta n}))$.
Combining gives \eqref{eq:rn-vs-rhocl}.
\end{proof}

\section{Proof of the Main Theorem}\label{sec:weaklimit}
%%=================================================================

\begin{theorem}[Semiclassical equidistribution of PSWF densities]%
\label{thm:weak-limit}
For $n \le \gamma N$ and every $f \in C^1([-T,T])$:
\begin{equation}\label{eq:weak-limit}
  \int_{-T}^T f(x)\,\psin(x)^2\,dx
  = \lambda_n(c)\int_{-T}^T f(x)\,\rhocl(x)\,dx
    + R_n(f),
\end{equation}
where:
\begin{equation}\label{eq:remainder}
  |R_n(f)| \le \frac{C_{\delta,\gamma}\,\lambda_n(c)}{n}
               \left(\|f\|_{L^\infty} + T\,\|f'\|_{L^\infty}\right).
\end{equation}
The constant $C_{\delta,\gamma}$ is explicit and depends only on
$\delta$, $\gamma$, $c$, $T$, but not on $n$ or $f$.
\end{theorem}

\begin{proof}
\textit{Step 1 (Bulk cutoff):}
$\int_{-T}^T f\psin^2 = \int_{I_\delta}f\psin^2 + O(e^{-\beta n}\lambda_n\|f\|_{L^\infty})$
by Lemma~\ref{lem:turning-point-cutoff}.

\textit{Step 2 (Pr\"ufer decomposition):}
On $I_\delta$:
$\int_{I_\delta}f\psin^2
= \frac{1}{2}\int_{I_\delta}fr_n^2 - \frac{1}{2}\int_{I_\delta}fr_n^2\cos(2\theta_n)$.

\textit{Step 3 (Oscillation term):}
By Corollary~\ref{cor:osc-slowly-varying} with $a=f$,
$b_n = r_n^2/r_n(x_0)^2$ (using \eqref{eq:amp-dev-deriv}
to verify $\|b_n'\|_{L^\infty} = O(1/n)$):
$\left|\int fr_n^2\cos(2\theta_n)\right|
\le \frac{C_{\delta,\gamma}\lambda_n}{n}(\|f\|_{L^\infty}+T\|f'\|_{L^\infty})$.

\textit{Step 4 (Mean term):}
By \eqref{eq:amp-ratio} and Lemma~\ref{lem:normalization}~\eqref{eq:rn-vs-rhocl}:
$\frac{1}{2}\int fr_n^2
= \lambda_n\int f\rhocl + O(\lambda_n\|f\|_{L^\infty}/n)$.

\textit{Step 5 (Extension to $[-T,T]$):}
$\int_{[-T,T]\setminus I_\delta}f\rhocl = O(e^{-\beta n}\|f\|_{L^\infty})$
by the exponential density decay.

\textit{Assembly:} Steps 1--5 yield \eqref{eq:weak-limit}--\eqref{eq:remainder}.
\end{proof}

\begin{remark}[Uniformity in $n$ and $f$]
The constant in \eqref{eq:remainder} satisfies
$C_{\delta,\gamma} \lesssim c_{\delta,\gamma}^{-2}$
where $c_{\delta,\gamma}$ is the phase lower bound of
Lemma~\ref{lem:phase-monotone}.
The bound degrades as $\delta \to 0$ or $\gamma \to 1$,
but is uniform in $n$ and $f$ for fixed parameters.
\end{remark}

\begin{corollary}[Closure of the bulk program]%
\label{cor:bulk-program-closed}
Theorem~\ref{thm:weak-limit} supplies hypothesis~(6.1) of
Lemma~6.3 (\texttt{lem:bulk-reduction}) of \cite{PaperIII}
with $\omega_n = C_{\delta,\gamma}/n$.
The implication chain
\[
  \textup{Thm.~\ref{thm:weak-limit}}
  \Rightarrow \texttt{lem:bulk-reduction}
  \Rightarrow \texttt{prob:comm-refined}
  \Rightarrow \text{Bulk tail bound}
\]
is therefore complete for all $n \le \gamma N$,
conditional only on Assumption~2.4 of \cite{PaperIII}.
\end{corollary}

\section{Open Problems}\label{sec:open}
%%=================================================================

\begin{enumerate}[(P1)]
  \item \textbf{Extension to $f \in C^0$.}
  Theorem~\ref{thm:weak-limit} requires $f \in C^1$.
  For $f \in C^0$, a density argument using $C^1$ approximation
  gives weak-* convergence without a rate.
  A rate for $f \in C^\alpha$ ($0 < \alpha < 1$) via
  interpolation would extend the result.
  \item \textbf{Off-diagonal regime.}
  The present paper handles the bulk ($m,n \le \gamma N$).
  Whether an analogue of Theorem~\ref{thm:weak-limit} holds
  for the off-diagonal regime ($|m-n| \ge \delta N$) is open;
  the classical density argument would need to be
  adapted for the product density $\rho_{mn}$ rather than
  the single-index density $\rho_m$.
  \item \textbf{Explicit Bohr--Sommerfeld constant.}
  Lemma~\ref{lem:bohr-sommerfeld-pswf} gives an $O(1/n)$ error
  with implicit constant depending on $\gamma$, $c$, $T$.
  Making this constant fully explicit in terms of $c$, $T$, $\gamma$
  would allow the remainder in Theorem~\ref{thm:weak-limit}
  to be computed numerically for specific parameter values.
\end{enumerate}
