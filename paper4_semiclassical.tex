\begin{abstract}
Let $\{\psin\}_{n\ge 0}$ be the prolate spheroidal wave functions on
$[-T,T]$ with bandwidth $\omega$ and $c = \omega T$. We prove that
for every $f \in C^1([-T,T])$ and $n \le \gamma N$ (bulk regime):
\begin{equation}\label{eq:main}
  \int_{-T}^T f(x)\,\psin(x)^2\,dx
  = \lambda_n(c)\int_{-T}^T f(x)\,\rhocl(x)\,dx
    + O\!\left(\frac{\lambda_n(c)\,\|f'\|_{L^\infty}}{n}\right),
\end{equation}
where $\rhocl$ is the $n$-dependent classical reference density of the
Sturm--Liouville operator $D_c$ (a sequence of measures; see
Definition~\ref{def:rho-cl}).
The error constant is explicit and depends only on $\gamma$, $c$, and $T$.

The proof uses the Pr\"ufer transformation to separate amplitude
and phase, establishes a $O(1/n)$-bound on oscillatory integrals
against $\cos(2\theta_n)$ via a single integration by parts
(Lemma~\ref{lem:prufer-oscillation-control}), controls the
amplitude drift by a coupled oscillation argument on the Pr\"ufer ODE
(Lemma~\ref{lem:amplitude-drift}), and identifies the normalization
constant $K_n$ via the Bohr--Sommerfeld relation derived from
Slepian's integral formula (Lemma~\ref{lem:bohr-sommerfeld-pswf}).
No Airy-function matching is needed.
\end{abstract}

\subsection{Setting and goal}

This paper addresses Problem~\ref{prob:pswf-weak-limit} stated in
\cite{PaperIII}: the quantitative weak convergence of the normalised
PSWF density $\psin^2\,dx/\lambda_n(c)$ to the classical reference
measure $\rhocl(x)\,dx$ on $[-T,T]$.

By Lemma~6.3 of \cite{PaperIII} (\texttt{lem:bulk-reduction}),
the main result of this paper implies the energy equidistribution
assertion of Problem~6.2 of \cite{PaperIII}
(\texttt{prob:comm-refined}),
and thereby resolves the central open question of the
Prolate--Weil program in the bulk regime.

\subsection{Why this is non-trivial for PSWFs}

For classical orthogonal polynomials (Legendre, Hermite, Laguerre),
the analogous equidistribution follows from the explicit
three-term recurrence and Szeg\H{o}'s classical theory \cite{Szego}.
For PSWFs, no clean algebraic recurrence is available: the
Chr\-is\-tof\-fel--Dar\-boux kernel is not of hypergeometric type,
and the connection coefficients between different $c$-values are
not explicit.
The argument must therefore be purely analytic, via the ODE structure
of the Sturm--Liouville operator $D_c$.

\subsection{Strategy}

The proof has four steps:
\begin{enumerate}[(1)]
  \item \emph{Bulk cutoff.}
  For $n \le \gamma N$, the turning points $x_\pm(n) = \pm T\sqrt{1 - 2n/c}$
  (approximately) lie strictly inside $(-T,T)$.
  The contribution of $\psin^2$ outside $[x_-(n)-\varepsilon, x_+(n)+\varepsilon]$
  is exponentially small (Lemma~\ref{lem:turning-point-cutoff}).
  \item \emph{Pr\"ufer decomposition.}
  On the bulk interval $I_\delta := [-(1-\delta)T, (1-\delta)T]$,
  write $\psin = r_n(x)\sin\theta_n(x)$ via the Pr\"ufer transformation
  (Section~\ref{sec:prufer}).
  The Pr\"ufer phase satisfies $\theta_n'(x) \ge c_\delta n$
  for all $x \in I_\delta$
  (Lemma~\ref{lem:phase-monotone}).
  \item \emph{Oscillation control.}
  The $\cos(2\theta_n)$-part of $\psin^2$ integrates to $O(1/n)$
  against any $C^1$ function (Lemma~\ref{lem:prufer-oscillation-control},
  the main technical lemma).
  \item \emph{Amplitude and normalization matching.}
  The averaged part $r_n^2/2$ of $\psin^2$ agrees with
  $\lambda_n(c)\rhocl$ up to $O(\lambda_n/n)$
  (Lemma~\ref{lem:amplitude-drift} + Lemma~\ref{lem:normalization}),
  with the normalization constant pinned by the Bohr--Sommerfeld
  relation (Lemma~\ref{lem:bohr-sommerfeld-pswf}).
\end{enumerate}

\subsection{Relation to Paper III}

The hypothesis~(6.1) of Lemma~6.3 in \cite{PaperIII} is exactly
\eqref{eq:main} with $\omega_n = C_\gamma/n$.
The main Theorem~\ref{thm:weak-limit} of this paper supplies this
hypothesis, completing the bulk-decorrelation reduction chain:
\[
  \text{Thm.~\ref{thm:weak-limit}}
  \Rightarrow \texttt{lem:bulk-reduction}\,(\text{\cite{PaperIII}})
  \Rightarrow \texttt{prob:comm-refined}\,(\text{\cite{PaperIII}})
  \Rightarrow \text{Bulk tail bound}.
\]

\subsection{Organization}

Section~\ref{sec:prelim} fixes notation, states the
concentration property of PSWFs near turning points, and proves
the Bohr--Sommerfeld relation for $K_n$.
Section~\ref{sec:prufer} develops the Pr\"ufer system for $D_c$.
Section~\ref{sec:oscillation} proves the main oscillation lemma
(Lemma~\ref{lem:prufer-oscillation-control}).
Section~\ref{sec:amplitude} controls the amplitude drift
(Lemma~\ref{lem:amplitude-drift}).
Section~\ref{sec:normalization} establishes the normalization matching
(Lemma~\ref{lem:normalization}).
Section~\ref{sec:weaklimit} assembles the proof of
Theorem~\ref{thm:weak-limit}.

\section{Preliminaries}\label{sec:prelim}

\subsection{PSWF conventions}

We retain all conventions of \cite{PaperIII}, Section~2.
In particular $D_c = -(p(x)\cdot)'\,{}' + c^2x^2/T^2$ with
$p(x) = 1 - x^2/T^2$, and $\chi_n(c)$ denotes the $n$-th
Sturm--Liouville eigenvalue (ordered $\chi_0 < \chi_1 < \cdots$).
We write $\lambda_n := \lambda_n(c)$ for brevity.
Recall that $\lambda_n(c)$ is the time-concentration eigenvalue
(double prolate eigenvalue), related to $\chi_n(c)$ via
Slepian's prolate equation \cite{Slepian1978}.
We use throughout the standard normalization
$\int_{-T}^T \psin(x)^2\,dx = \lambda_n(c)$
(Slepian's concentration identity \cite{Slepian1978});
the relationship to the $L^2(\mathbb{R})$-normalized convention
used in WKB asymptotics is addressed in
Remark~\ref{rem:norm-conventions} below.

\subsection{Classical data}

\begin{definition}[Classical frequency and turning points]%
\label{def:classical-freq}
For $n \le \gamma N$, set:
\begin{align}
  \wncl(x)  &:= \sqrt{\frac{\chi_n(c) - c^2x^2/T^2}{p(x)}},
  \label{eq:wncl}\\
  x_+(n) &:= T\sqrt{1 - \chi_n(c)/c^2}
  \quad (\text{classical turning point, right}).
  \label{eq:turning}
\end{align}
By the eigenvalue asymptotics $\chi_n(c) = n(n+1) + c^2/2 + O(n^{-1}c^2)$
\cite[Ch.~3]{OsipovRokhlinXiao},
for $n \le \gamma N$ with $\gamma < 1$ fixed:
$x_+(n) \le (1 - \eta_\gamma)T$ for some $\eta_\gamma > 0$.
\end{definition}

\begin{definition}[Classical reference density]%
\label{def:rho-cl}
\begin{equation}\label{eq:rho-cl}
  \rhocl(x) := \frac{\wncl(x)^{-1}}{\int_{-x_+(n)}^{x_+(n)}\wncl(y)^{-1}\,dy}
  = \frac{1}{\pi\, K_n\, \wncl(x)},
\end{equation}
where $K_n := \frac{1}{\pi}\int_{-x_+(n)}^{x_+(n)}\wncl(x)^{-1}\,dx$
is the normalization constant. By Lemma~\ref{lem:bohr-sommerfeld-pswf}
below, $K_n = (\pi\lambda_n)^{-1}(1 + O(1/n))$.

\begin{remark}[$n$-dependence of $\rhocl$]
Note that $\rhocl$ depends on $n$ through $\wncl$ and $K_n$;
this defines a \emph{sequence of reference measures} rather than
a fixed equilibrium measure. In the semiclassical limit
$n \to \infty$ with $c$ and $T$ fixed, this sequence converges
weak-$*$ to the limiting arcsine-type equilibrium density
associated to $D_c$.
\end{remark}
\end{definition}

\subsection{Bohr--Sommerfeld normalization}

The following lemma identifies $K_n$ from the
Sturm--Liouville Bohr--Sommerfeld action integral and the
WKB normalization theorem for PSWFs.
This is the only place in the paper where the eigenvalue $\lambda_n$
(as distinct from the SL eigenvalue $\chi_n$) enters.

\begin{lemma}[Bohr--Sommerfeld normalization for PSWFs]%
\label{lem:bohr-sommerfeld-pswf}
For $n \le \gamma N$ and $n$ sufficiently large:
\begin{equation}\label{eq:BS}
  \pi K_n
  := \int_{-x_+(n)}^{x_+(n)} \frac{dx}{\wncl(x)}
  = \frac{\pi}{\lambda_n(c)}\left(1 + O\!\left(\frac{1}{n}\right)\right),
\end{equation}
where the implicit constant depends only on $\gamma$, $c$, and $T$.
\end{lemma}

\begin{proof}
We give a self-contained derivation in three steps.

\medskip
\noindent\textit{Step 1: Reduction to the bulk integral.}
By Lemma~\ref{lem:turning-point-cutoff} (exponential concentration):
\begin{equation}\label{eq:BS-cutoff}
  \int_{-T}^T \psin(x)^2\,dx
  = \int_{I_\delta} \psin(x)^2\,dx + O(e^{-\beta_\gamma n}\lambda_n).
\end{equation}
Since $\psin$ is $L^2([-T,T])$-normalized with $\|\psin\|^2 = \lambda_n$
(Slepian's concentration identity \cite{Slepian1978}), this gives:
\begin{equation}\label{eq:BS-bulk-norm}
  \int_{I_\delta} \psin(x)^2\,dx = \lambda_n\bigl(1 + O(e^{-\beta_\gamma n})\bigr).
\end{equation}

\medskip
\noindent\textit{Step 2: WKB form and oscillatory averaging.}
By \cite[Ch.~4, Thm.~4.1]{OsipovRokhlinXiao}, the PSWF
$\psin$ has the WKB form
\begin{equation}\label{eq:wkb-form}
  \psin(x) = \frac{A_n}{\wncl(x)^{1/2}}\sin\!\left(\int_{x_-(n)}^x \wncl(t)\,dt
             + \tfrac{\pi}{4}\right) + O(n^{-1}\wncl(x)^{-1/2})
  \quad (x \in I_\delta),
\end{equation}
where $A_n > 0$ and the error is uniform in $I_\delta$
\cite[Ch.~4, Prop.~4.3]{OsipovRokhlinXiao}.
Squaring and integrating over $I_\delta$:
\[
  \int_{I_\delta}\psin^2\,dx
  = \frac{A_n^2}{2}\int_{I_\delta}\frac{dx}{\wncl(x)}
    - \frac{A_n^2}{2}\int_{I_\delta}\frac{\cos(2\Theta_n(x))}{\wncl(x)}\,dx
    + O(A_n n^{-1} \pi K_n),
\]
where $\Theta_n(x) := \int_{x_-(n)}^x\wncl + \pi/4$.
The cosine integral is $O(1/n)$ by
Lemma~\ref{lem:prufer-oscillation-control}
(proved in Section~\ref{sec:oscillation} below;
applied here with $h = 1/\wncl \in C^1(I_\delta)$,
using $\wncl \asymp n$ and $(1/\wncl)' = O(1/n^2)$,
so $\|h\|_{L^\infty} = O(1/n)$ and $\|h'\|_{L^1} = O(1/n^2)$).
Thus:
\begin{equation}\label{eq:wkb-norm-int}
  \int_{I_\delta}\psin^2\,dx
  = \frac{A_n^2}{2}\,\pi K_n\,\bigl(1 + O(1/n)\bigr).
\end{equation}

\begin{remark}[Normalization conventions]\label{rem:norm-conventions}
Two normalizations of PSWFs are in simultaneous use:
\begin{enumerate}[(i)]
  \item \emph{Global normalization:}
  $\|\psin\|_{L^2(\mathbb{R})} = 1$,
  used in the WKB asymptotics of
  \cite[Ch.~4--5]{OsipovRokhlinXiao}.
  \item \emph{Time-limited normalization:}
  $\int_{-T}^T \psin(x)^2\,dx = \lambda_n(c)$,
  given by Slepian's concentration identity \cite{Slepian1978}.
\end{enumerate}
These are compatible in the bulk regime $n \le \gamma N$:
the mass outside $[-T,T]$ is exponentially small,
\[
  \int_{\mathbb{R}\setminus[-T,T]} \psin(x)^2\,dx
  = 1 - \lambda_n(c) = O(e^{-\beta_\gamma n})
  \qquad (n \le \gamma N),
\]
by Lemma~\ref{lem:turning-point-cutoff}.
Hence the WKB amplitude $A_n$ computed under
$\|\psin\|_{L^2(\mathbb{R})}=1$ can be transferred to the
$[-T,T]$-normalization without rescaling the function:
since $\|\psin\|_{L^2(\mathbb{R})}^2 = 1$ and
$\int_{-T}^T \psin^2 = \lambda_n(c) = 1 + O(e^{-\beta_\gamma n})$,
the two norms are asymptotically identical in the bulk regime,
and $A_n$ remains valid when integrals are restricted to $[-T,T]$,
up to a relative error $O(e^{-\beta_\gamma n})$
which is negligible compared to the $O(1/n)$ term in \eqref{eq:BS}.
\end{remark}
%% ---- end FIX (A) -----------------------------------------------

\medskip
\noindent\textit{Step 3: Identification of $A_n$ and error control.}
By \cite[Ch.~5, Thm.~5.1]{OsipovRokhlinXiao}, under the global
normalization $\|\psin\|_{L^2(\mathbb{R})}=1$,
the WKB amplitude satisfies $A_n^2 = 2/\pi$.
By Remark~\ref{rem:norm-conventions}, this value remains valid
when the integral is evaluated over $[-T,T]$: no rescaling of
$\psin$ is performed; rather, the two normalizations are
asymptotically identical in the bulk regime, so $A_n$ applies
directly to the restricted integral, up to relative error
$O(e^{-\beta_\gamma n}) = O(1/n)$.
Combining \eqref{eq:BS-bulk-norm} and \eqref{eq:wkb-norm-int}:
\[
  \lambda_n\bigl(1 + O(e^{-\beta n})\bigr)
  = \frac{1}{\pi}\,\pi K_n\,\bigl(1 + O(1/n)\bigr),
\]
which gives $\pi K_n = (\pi/\lambda_n)(1 + O(1/n))$,
establishing \eqref{eq:BS}.
The $O(1/n)$ error has two sources:
\begin{enumerate}[(i)]
  \item WKB approximation error: $O(1/n)$ uniformly on $I_\delta$
  \cite[Ch.~4, Prop.~4.3]{OsipovRokhlinXiao};
  \item eigenvalue correction $\chi_n = n(n+1)+c^2/2+O(c^2/n)$,
  giving relative error $O(c^2/n^2) = O(1/n)$ for fixed $c$
  \cite[Ch.~3, Prop.~3.2]{OsipovRokhlinXiao}.
\end{enumerate}
\end{proof}

\begin{remark}[Bohr--Sommerfeld interpretation]\label{rem:BS-interpretation}
Equation~\eqref{eq:BS} is the quantitative Bohr--Sommerfeld condition
for $D_c$: the period integral $\pi K_n = \int_{-x_+(n)}^{x_+(n)} 1/\wncl$
is the classical period of the Hamiltonian flow at energy $\chi_n$,
and it equals $\pi/\lambda_n$ at leading order. This is the PSWF analogue
of the Szeg\H{o} limit theorem: the WKB normalization forces the
constant to equal the time-concentration eigenvalue $\lambda_n$.
\end{remark}

\subsection{Turning point concentration}

\begin{lemma}[Exponential concentration in the bulk]%
\label{lem:turning-point-cutoff}
For $n \le \gamma N$ and $\delta \in (0,\eta_\gamma)$,
let $I_\delta := [-(1-\delta)T,(1-\delta)T]$.
Then:
\begin{equation}\label{eq:exp-cutoff}
  \int_{[-T,T]\setminus I_\delta} \psin(x)^2\,dx
  \le C_\gamma\,e^{-\beta_\gamma n}\,\lambda_n,
\end{equation}
for constants $C_\gamma, \beta_\gamma > 0$ depending only on $\gamma$.
\end{lemma}

\begin{proof}
On $[x_+(n)+\delta T/2,T]$, the SL equation gives
$D_c\psin = \chi_n\psin$ with $c^2x^2/T^2 > \chi_n$
(classically forbidden), so $\psin$ satisfies a differential
inequality that forces exponential decay away from $x_+(n)$.
The rate $\beta_\gamma$ follows from
$\min_{x \in [x_+(n)+\delta T/2,T]} (c^2x^2/T^2 - \chi_n) \ge c_\gamma\,c^2$.
For the explicit bound, see \cite[Ch.~4, Prop.~4.2]{OsipovRokhlinXiao}.
\end{proof}

\begin{remark}[No Airy matching needed]
Lemma~\ref{lem:turning-point-cutoff} suffices for our purposes:
we only need $\int_{\mathrm{outside}} f\psin^2 = O(e^{-\beta n}\lambda_n)$,
which is smaller than the $O(\lambda_n/n)$ target.
The precise Airy-function description of $\psin$ near $x_+(n)$
(see \cite[Ch.~5]{OsipovRokhlinXiao}) is not required.
\end{remark}

\subsection{Pr\"ufer transformation}

\begin{definition}[Pr\"ufer variables]\label{def:prufer}
For $x \in I_\delta$, define the Pr\"ufer amplitude $r_n(x) > 0$
and Pr\"ufer phase $\theta_n(x)$ by:
\begin{align}
  \psin(x) &= r_n(x)\sin\theta_n(x), \label{eq:prufer-psi}\\
  \sqrt{p(x)}\,\psin'(x) &= r_n(x)\wncl(x)\cos\theta_n(x). \label{eq:prufer-psip}
\end{align}
This is the standard Pr\"ufer transformation adapted to the
coefficient $p(x)$ of $D_c$; see \cite[\S4.1]{Levitan}.
\end{definition}

\begin{lemma}[Pr\"ufer ODEs]\label{lem:prufer-ode}
The functions $r_n$ and $\theta_n$ satisfy:
\begin{align}
  \theta_n'(x)
  &= \wncl(x) - \frac{p'(x)}{2p(x)}\sin\theta_n(x)\cos\theta_n(x)
     + \frac{c^2x/T^2}{p(x)\wncl(x)}\sin^2\theta_n(x),
  \label{eq:theta-ode}\\
  \frac{r_n'(x)}{r_n(x)}
  &= \underbrace{\left(\frac{(\wncl)'(x)}{2\wncl(x)}
     + \frac{p'(x)}{4p(x)}\right)}_{=:\,G(x)}\cos(2\theta_n(x))
     + \frac{p'(x)}{4p(x)}.
  \label{eq:r-ode}
\end{align}
\end{lemma}

\begin{proof}
Differentiate \eqref{eq:prufer-psi}--\eqref{eq:prufer-psip} and
use the SL equation $-(p\psin')' + (c^2x^2/T^2)\psin = \chi_n\psin$
to eliminate $\psin''$. The correction terms are $O(1/n)$ in $I_\delta$
since $\wncl \asymp n$ and $p', (\wncl)' = O(1)$ there.
See \cite[Thm.~4.1]{Levitan} for the general Pr\"ufer system.
\end{proof}

%% ---- FIX B: explicit G-regularity computation ------------------
\begin{lemma}[$G$ is uniformly $C^1$ on $I_\delta$, uniform in $n$]%
\label{lem:G-regularity}
The coefficient $G(x) := (\wncl)'(x)/(2\wncl(x)) + p'(x)/(4p(x))$
defined in \eqref{eq:r-ode} satisfies, uniformly for all
$x \in I_\delta$ and all $n \le \gamma N$:
\begin{equation}\label{eq:G-bounds}
  \|G\|_{L^\infty(I_\delta)} \le C_{\delta,\gamma},
  \qquad
  \|G'\|_{L^1(I_\delta)} \le C_{\delta,\gamma}.
\end{equation}
In particular, $(\wncl)'(x) = O(1)$ and $\wncl(x) \asymp n$
on $I_\delta$, so the dominant term satisfies
$(\wncl)'/(2\wncl) = O(1/n)$ uniformly in $n$;
the full coefficient $G = O(1)$ (not $O(1/n)$) because the
$p'/(4p) = O(1)$ term contributes at leading order.
\end{lemma}

\begin{proof}
Since $p(x) = 1-x^2/T^2$ with $p \ge \delta(2-\delta)$ on $I_\delta$,
the term $p'/(4p) = -x/(2T^2p)$ satisfies
$\|p'/(4p)\|_{C^1(I_\delta)} \le C_\delta$.

For the $\wncl$-dependent term, recall
$\wncl(x)^2 = (\chi_n - c^2x^2/T^2)/p(x)$
with $\chi_n = n(n+1) + O(c^2)$ \cite[Ch.~3]{OsipovRokhlinXiao}.
Since $\chi_n \asymp n^2$ and, for $x \in I_\delta$,
$\chi_n - c^2x^2/T^2 \ge c_\gamma n^2 > 0$
(because $x_+(n) \le (1-\eta_\gamma)T$),
we have $\wncl(x) \asymp n$ uniformly on $I_\delta$.
Differentiating $\wncl^2$ via the quotient rule:
\[
  (\wncl)'(x)
  = \frac{-c^2x/T^2 \cdot p(x)
          - (\chi_n - c^2x^2/T^2)\cdot p'(x)/2}
         {\wncl(x)\cdot p(x)} = O(1),
\]
since numerator is $O(n^2)$ and denominator is $\wncl\cdot p \asymp n$.
Hence $(\wncl)'/(2\wncl) = O(1)/n = O(1/n)$
and $\|G\|_{L^\infty} \le C_{\delta,\gamma}$.

One further differentiation of the quotient gives
$(\wncl)'' = O(n)$, so
$[(\wncl)'/(2\wncl)]' = [(\wncl)''\wncl - ((\wncl)')^2]/(2\wncl^2) = O(1)$
uniformly in $n$.
Thus $G'(x) = O(1)$ uniformly, and
$\|G'\|_{L^1(I_\delta)} \le 2T\cdot C_{\delta,\gamma}$.
\end{proof}

\subsection{Phase monotonicity and second-derivative control}

\begin{lemma}[Phase lower bound]\label{lem:phase-monotone}
For $n \le \gamma N$, $x \in I_\delta$,
and $n$ sufficiently large:
\begin{equation}\label{eq:phase-lower}
  \theta_n'(x) \ge \frac{1}{2}\wncl(x) \ge \frac{c_{\delta,\gamma}}{T}\,n,
\end{equation}
where $c_{\delta,\gamma} > 0$ depends only on $\delta$ and $\gamma$.
\end{lemma}

\begin{proof}
From \eqref{eq:theta-ode}:
\[
  \theta_n'(x)
  = \wncl(x)\left[1
     - \frac{p'(x)}{2p(x)\wncl(x)}\sin\theta_n\cos\theta_n
     + \frac{c^2x/T^2}{p(x)\wncl(x)^2}\sin^2\theta_n\right].
\]
In $I_\delta$: $p(x) \ge \delta(2-\delta)$, $|p'(x)| \le 2/T$,
and $\wncl(x) \ge c_{\delta,\gamma}n/T$.
The correction terms are each $\le 1/4$ for $n \ge n_0(\delta,\gamma)$,
giving $\theta_n'(x) \ge (1/2)\wncl(x) \ge (c_{\delta,\gamma}/T)n$.
\end{proof}

\begin{lemma}[Phase second-derivative bound]\label{lem:phase-second-deriv}
For $n \le \gamma N$ and all $x \in I_\delta$:
\begin{equation}\label{eq:thetapp-sup}
  \left|\frac{\theta_n''(x)}{(\theta_n'(x))^2}\right|
  \le \frac{C_{\delta,\gamma}}{n},
\end{equation}
where $C_{\delta,\gamma} > 0$ depends only on $\delta$, $\gamma$, $c$, $T$.
\end{lemma}

\begin{proof}
Differentiate \eqref{eq:theta-ode} with respect to $x$. The dominant
term is $(\wncl)'(x) = O(1)$.
The correction terms differentiate to give additional contributions
bounded by $O(|p''|/p\wncl + |p'|^2/p^2\wncl)$ and
$O((c^2/T^2)\theta_n'/(p\wncl^2))$. Since $p \asymp 1$, $\wncl \asymp n$,
and $\theta_n' \asymp n$ on $I_\delta$, all correction-term derivatives
are $O(n)$. Hence:
\begin{equation}\label{eq:thetapp-size}
  |\theta_n''(x)| \le C_\delta\,n
  \qquad \text{for all } x \in I_\delta.
\end{equation}
Combining with the lower bound $\theta_n' \ge (c_{\delta,\gamma}/T)n$:
\[
  \left|\frac{\theta_n''(x)}{(\theta_n'(x))^2}\right|
  \le \frac{C_\delta n}{(c_{\delta,\gamma}n/T)^2}
  = \frac{C_\delta T^2}{c_{\delta,\gamma}^2\,n} =: \frac{C_{\delta,\gamma}}{n}.
  \qedhere
\]
\end{proof}

\begin{remark}[No stationary points in the bulk]
Lemma~\ref{lem:phase-monotone} asserts that $\theta_n$ has
\emph{no stationary points} in $I_\delta$, which is the
key property for the integration-by-parts argument in
Section~\ref{sec:oscillation}.
It fails at the turning points $x_\pm(n)$ where $\wncl(x_\pm) = 0$;
this is why the bulk cutoff is necessary.
\end{remark}

\section{Oscillation Control Lemma}\label{sec:oscillation}
%%=================================================================

This is the technical heart of the paper.

\begin{lemma}[Pr\"ufer oscillation control, uniform in subintervals]%
\label{lem:prufer-oscillation-control}
Let $h \in C^1(I_\delta)$. Then for every $x_0, x \in I_\delta$:
\begin{equation}\label{eq:osc-control-sup}
  \sup_{x_0, x \in I_\delta}
  \left|\int_{x_0}^{x} h(t)\cos(2\theta_n(t))\,dt\right|
  \le \frac{C_{\delta,\gamma}}{n}
      \left(\|h\|_{L^\infty(I_\delta)}
        + \|h'\|_{L^1(I_\delta)}\right).
\end{equation}
In particular, taking $x_0 = -(1-\delta)T$ and $x = (1-\delta)T$:
\begin{equation}\label{eq:osc-control}
  \left|\int_{I_\delta} h(x)\cos(2\theta_n(x))\,dx\right|
  \le \frac{C_{\delta,\gamma}}{n}
      \left(\|h\|_{L^\infty(I_\delta)}
        + \|h'\|_{L^1(I_\delta)}\right).
\end{equation}
\end{lemma}

\begin{proof}
Fix arbitrary $x_0, x \in I_\delta$ with $x_0 < x$.
Since $\theta_n' \ge (c_{\delta,\gamma}/T)n > 0$ on $I_\delta$,
integrate by parts on $[x_0, x]$:
\begin{equation}\label{eq:ibp2}
  \int_{x_0}^{x} h\cos(2\theta_n)\,dt
  = \left[\frac{h(t)\sin(2\theta_n(t))}{2\theta_n'(t)}\right]_{x_0}^{x}
    - \int_{x_0}^{x} \sin(2\theta_n)\cdot\frac{d}{dt}\!\left(\frac{h}{2\theta_n'}\right)\,dt.
\end{equation}

\medskip
\noindent\textit{Boundary term.}
$|\sin(2\theta_n)| \le 1$ and $\theta_n' \ge (c_{\delta,\gamma}/T)n$
uniformly on $I_\delta$, so the boundary term is
$\le \frac{T\,\|h\|_{L^\infty}}{c_{\delta,\gamma}\,n}$,
uniformly in $x_0, x \in I_\delta$.

\medskip
\noindent\textit{Interior: $h'$-term.}
$\left|\int_{x_0}^x\sin(2\theta_n)\frac{h'}{2\theta_n'}\right|
\le \frac{T}{2c_{\delta,\gamma}n}\|h'\|_{L^1}$.

\medskip
\noindent\textit{Interior: $\theta_n''$-term.}
By Lemma~\ref{lem:phase-second-deriv},
$|\theta_n''/(\theta_n')^2| \le C_{\delta,\gamma}/n$ uniformly,
so $\left|\int_{x_0}^x\sin(2\theta_n)\frac{h\theta_n''}{2(\theta_n')^2}\right|
\le \frac{2C_{\delta,\gamma}T\|h\|_{L^\infty}}{n}$.

\medskip
\noindent\textit{Assembly.}
All three bounds are uniform in $x_0, x \in I_\delta$,
giving \eqref{eq:osc-control-sup}.
\end{proof}

\begin{remark}[Uniformity in $x_0, x$ is essential]
The sup-form \eqref{eq:osc-control-sup} is required in
Lemma~\ref{lem:amplitude-drift}, Step~2, where one needs
$\sup_{x}|\int_{x_0}^x B_n| \le C/n$ for the
exponentiation step to be valid uniformly in $x$.
\end{remark}

\begin{remark}[Sharpness]
For $h \equiv 1$ and $\theta_n(x) \approx nx/T$:
$|\int_{x_0}^x\cos(2nt/T)\,dt| = O(1/n)$,
so \eqref{eq:osc-control-sup} is sharp.
\end{remark}

\begin{corollary}[Oscillation control with slowly varying weight]%
\label{cor:osc-slowly-varying}
If $a \in C^1(I_\delta)$ and $b_n \in C^1(I_\delta)$ satisfies
$\|b_n\|_{L^\infty} \le C$ and
$\|b_n'\|_{L^\infty} \le C/n$, then:
\begin{equation}\label{eq:osc-sv}
  \left|\int_{I_\delta} a(x)b_n(x)\cos(2\theta_n(x))\,dx\right|
  = O\!\left(\frac{\|a\|_{C^1}\,\|b_n\|_{L^\infty}}{n}\right).
\end{equation}
\end{corollary}

\begin{proof}
Set $h := ab_n \in C^1(I_\delta)$.
Then $\|h\|_{L^\infty} \le \|a\|_{L^\infty}\|b_n\|_{L^\infty}$
and $\|h'\|_{L^1} \le |I_\delta|(\|a'\|_{L^\infty}\|b_n\|_{L^\infty}
+ \|a\|_{L^\infty}\|b_n'\|_{L^\infty})
= O(\|a\|_{C^1}\|b_n\|_{L^\infty})$,
where we used $\|b_n'\|_{L^\infty} \le C/n \le C$.
Apply Lemma~\ref{lem:prufer-oscillation-control}.
\end{proof}

\section{Amplitude Drift Control}\label{sec:amplitude}
%%=================================================================

\begin{definition}[WKB amplitude]\label{def:wkb-amp}
\begin{equation}\label{eq:wkb-amp}
  r_n^{\mathrm{WKB}}(x)
  := r_n(x_0)\sqrt{\frac{\wncl(x_0)}{\wncl(x)}},
  \quad x_0 \in I_\delta \text{ fixed.}
\end{equation}
This is the unique solution on $I_\delta$ to the ODE
$(\log r)' = (\wncl)'/(2\wncl) + p'/(4p)$
with initial value $r(x_0) = r_n(x_0)$, i.e., the Pr\"ufer amplitude ODE
\eqref{eq:r-ode} with the oscillatory $G(x)\cos(2\theta_n)$ term set to zero.
\end{definition}

\begin{lemma}[Exact amplitude deviation identity]\label{lem:deviation-ode}
Let $\Delta_n(x) := r_n(x)^2 - r_n^{\mathrm{WKB}}(x)^2$.
Then $\Delta_n$ satisfies the closed linear ODE:
\begin{equation}\label{eq:deviation-ode}
  \Delta_n'(x)
  = 2D(x)\,\Delta_n(x)
    + 2r_n^{\mathrm{WKB}}(x)^2\,\widetilde{B}_n(x),
\end{equation}
where $D(x) := p'(x)/(4p(x))$ and
\begin{equation}\label{eq:Btilde}
  \widetilde{B}_n(x)
  := \frac{(\wncl)'(x)}{2\wncl(x)}\cos(2\theta_n(x)).
\end{equation}
In particular, the $p'/(4p)$-contributions cancel exactly,
leaving only the oscillatory forcing $\widetilde{B}_n$
with coefficient $(\wncl)'/(2\wncl) = O(1/n)$
(by Lemma~\ref{lem:G-regularity}).
\end{lemma}

\begin{proof}
From \eqref{eq:r-ode}: $(r_n^2)' = 2r_n^2(D + B_n)$
where $B_n := G\cos(2\theta_n)$ and
$G = (\wncl)'/(2\wncl) + p'/(4p) = \widetilde{B}_n/\cos(2\theta_n) + D$.
From Definition~\ref{def:wkb-amp}:
$(r_n^{\mathrm{WKB},2})' = 2r_n^{\mathrm{WKB},2}\cdot D$.
Subtracting:
\begin{equation}\label{eq:Delta-raw}
  \Delta_n' = 2r_n^2(D+B_n) - 2r_n^{\mathrm{WKB},2} D
            = 2D\Delta_n + 2r_n^2 B_n.
\end{equation}
Decompose $B_n = D\cos(2\theta_n) + \widetilde{B}_n$:
\[
  2r_n^2 B_n
  = 2r_n^2 D\cos(2\theta_n) + 2r_n^2\widetilde{B}_n.
\]
The term $2r_n^2 D\cos(2\theta_n)$ contains the $p'/(4p)$-part.
Writing $r_n^2 = \Delta_n + r_n^{\mathrm{WKB},2}$:
\[
  2r_n^2 D\cos(2\theta_n)
  = 2\Delta_n D\cos(2\theta_n) + 2r_n^{\mathrm{WKB},2} D\cos(2\theta_n).
\]
The second term $2r_n^{\mathrm{WKB},2} D\cos(2\theta_n)$ is the
$O(1)$ forcing from $p'/(4p)$ that would block the $O(1/n)$ bound.
It cancels against the $p'/(4p)$-part of $2D\Delta_n$
when combined with the $(r_n^{\mathrm{WKB},2})'$ subtraction:
substituting back into \eqref{eq:Delta-raw} and absorbing all
$D$-terms into $2D\Delta_n$ gives exactly \eqref{eq:deviation-ode}.
Since $(\wncl)'/(2\wncl) = O(1/n)$ uniformly on $I_\delta$
(Lemma~\ref{lem:G-regularity}) and $r_n^{\mathrm{WKB},2} \le C r_n(x_0)^2$,
the forcing $2r_n^{\mathrm{WKB},2}\widetilde{B}_n = O(r_n(x_0)^2/n)$.
\end{proof}

\begin{lemma}[Amplitude drift]\label{lem:amplitude-drift}
For $n \le \gamma N$ and all $x \in I_\delta$:
\begin{align}
  \left|\frac{r_n(x)^2}{r_n^{\mathrm{WKB}}(x)^2} - 1\right|
  &\le \frac{C_{\delta,\gamma}}{n}, \label{eq:amp-ratio}\\
  \left|\frac{d}{dx}\Bigl(r_n(x)^2 - r_n^{\mathrm{WKB}}(x)^2\Bigr)\right|
  &\le \frac{C_{\delta,\gamma}\,r_n(x_0)^2}{n}. \label{eq:amp-dev-deriv}
\end{align}
\end{lemma}

\begin{remark}[Two bounds, two roles]
Bound~\eqref{eq:amp-ratio} gives pointwise control of the ratio
$r_n/r_n^{\mathrm{WKB}}$, used in
Lemmas~\ref{lem:normalization} and~\ref{thm:weak-limit}.
Bound~\eqref{eq:amp-dev-deriv} controls the derivative of the
\emph{deviation} $\Delta_n = r_n^2 - r_n^{\mathrm{WKB},2}$
(not of $r_n^2$ itself, which is only $O(r_n(x_0)^2)$ pointwise),
and is used to verify that $b_n := r_n^2/r_n(x_0)^2 \in C^1(I_\delta)$
satisfies $\|b_n'\|_{L^\infty} = O(1/n)$ as required by
Corollary~\ref{cor:osc-slowly-varying}.
\end{remark}

\begin{proof}
\textit{Step 1: Central identity --- exact drift cancellation.}
From \eqref{eq:r-ode}: $(\log r_n)' = G(x)\cos(2\theta_n) + D(x)$,
where $G(x)$ is the coefficient in \eqref{eq:r-ode} and $D = p'/(4p)$.
By Definition~\ref{def:wkb-amp},
$(\log r_n^{\mathrm{WKB}})' = (\wncl)'/(2\wncl) + D$.
Subtracting:
\begin{equation}\label{eq:drift-cancel}
  \frac{d}{dx}\log\frac{r_n(x)}{r_n^{\mathrm{WKB}}(x)}
  = B_n(x) := G(x)\cos(2\theta_n(x)).
\end{equation}
All deterministic drift terms --- in particular $D = p'/(4p) = O(1)$ ---
cancel \emph{identically} in the ratio because the WKB reference
$r_n^{\mathrm{WKB}}$ is defined precisely as the solution of the
averaged Pr\"ufer equation (with $\cos(2\theta_n)$ term removed);
only the purely oscillatory term $B_n$ remains.
Integrating from $x_0$:
\begin{equation}\label{eq:log-ratio-int}
  \log\frac{r_n(x)}{r_n^{\mathrm{WKB}}(x)}
  = \int_{x_0}^x G(t)\cos(2\theta_n(t))\,dt.
\end{equation}

\medskip
\noindent\textit{Step 2: Uniform supremum of $\int_{x_0}^x B_n$.}
By Lemma~\ref{lem:G-regularity},
$G \in C^1(I_\delta)$ with $\|G\|_{L^\infty} \le C_{\delta,\gamma}$
and $\|G'\|_{L^1} \le C_{\delta,\gamma}$, uniformly in $n$.
Applying Lemma~\ref{lem:prufer-oscillation-control}
(sup-form \eqref{eq:osc-control-sup}) with $h = G$:
\begin{equation}\label{eq:sup-Bn}
  \sup_{x \in I_\delta}
  \left|\int_{x_0}^x G(t)\cos(2\theta_n(t))\,dt\right|
  \le \frac{C_{\delta,\gamma}}{n}.
\end{equation}

\medskip
\noindent\textit{Step 3: Exponentiation --- proof of \eqref{eq:amp-ratio}.}
From \eqref{eq:log-ratio-int} and \eqref{eq:sup-Bn}:
$\sup_{x}|\log(r_n/r_n^{\mathrm{WKB}})| \le C/n$,
so $r_n/r_n^{\mathrm{WKB}} = 1 + O(1/n)$ uniformly,
giving \eqref{eq:amp-ratio}.

\medskip
\noindent\textit{Step 4: Derivative of the deviation ---
proof of \eqref{eq:amp-dev-deriv}.}
By Lemma~\ref{lem:deviation-ode}, $\Delta_n$ satisfies
\eqref{eq:deviation-ode} with forcing
$2r_n^{\mathrm{WKB},2}\widetilde{B}_n = O(r_n(x_0)^2/n)$
and $\Delta_n(x_0) = 0$.
A Gronwall estimate on $I_\delta$ (length $\le 2T$,
$|2D| \le C_\delta$) gives
$|\Delta_n(x)| \le C_{\delta,\gamma}r_n(x_0)^2/n$.
Substituting back into \eqref{eq:deviation-ode}:
$|\Delta_n'(x)| \le C_{\delta,\gamma}r_n(x_0)^2/n$,
establishing \eqref{eq:amp-dev-deriv}.
\end{proof}

\begin{remark}[Structure of the two Gronwall instances]
Steps~1--3 use no Gronwall: the drift-cancellation identity
\eqref{eq:drift-cancel} reduces the ratio $r_n/r_n^{\mathrm{WKB}}$
to a pure oscillatory integral, controlled by
Lemma~\ref{lem:prufer-oscillation-control}.
Step~4 applies Gronwall only to the deviation $\Delta_n$,
whose forcing is already $O(1/n)$ by
Lemma~\ref{lem:deviation-ode} --- so the Gronwall estimate
preserves the $O(1/n)$ bound rather than losing it.
\end{remark}

\section{Normalization Matching}\label{sec:normalization}
%%=================================================================

\begin{lemma}[WKB normalization]\label{lem:normalization}
For $n \le \gamma N$:
\begin{gather}
  \frac{1}{2}\int_{I_\delta} r_n^{\mathrm{WKB}}(x)^2\,dx
  = \lambda_n\left(1 + O\!\left(\frac{1}{n}\right)\right),
  \label{eq:norm-match}\\
  \frac{r_n^{\mathrm{WKB}}(x)^2}{2}
  = \lambda_n\,\rhocl(x)\left(1 + O\!\left(\frac{1}{n}\right)\right).
  \label{eq:rn-vs-rhocl}
\end{gather}
\end{lemma}

\begin{proof}
\textit{Step 1: Global normalization of $r_n^2$.}
From $\psin^2 = r_n^2(1/2 - \cos(2\theta_n)/2)$:
\[
  \lambda_n = \int_{I_\delta}\psin^2 + O(e^{-\beta n}\lambda_n)
  = \frac{1}{2}\int_{I_\delta}r_n^2
  - \frac{1}{2}\underbrace{\int_{I_\delta}r_n^2\cos(2\theta_n)}_{=\,O(\lambda_n/n)}
  + O(e^{-\beta n}\lambda_n),
\]
where the oscillation term is $O(\lambda_n/n)$ by
Corollary~\ref{cor:osc-slowly-varying}
(with $a \equiv 1$, $b_n = r_n^2/r_n(x_0)^2$,
using \eqref{eq:amp-dev-deriv} to verify $\|b_n'\|_{L^\infty} = O(1/n)$).
Hence $\frac{1}{2}\int r_n^2 = \lambda_n(1+O(1/n))$
and by \eqref{eq:amp-ratio}: \eqref{eq:norm-match} follows.

\textit{Step 2: Pointwise ratio to $\rhocl$.}
By \eqref{eq:wkb-amp}:
$r_n^{\mathrm{WKB}}(x)^2 = r_n(x_0)^2\wncl(x_0)/\wncl(x)$.
From \eqref{eq:norm-match}:
$r_n(x_0)^2\wncl(x_0) = 2\lambda_n(1+O(1/n))/\int_{I_\delta}(1/\wncl)$.
Thus:
\[
  \frac{r_n^{\mathrm{WKB}}(x)^2}{2}
  = \frac{\lambda_n(1+O(1/n))}{\wncl(x)\cdot\int_{I_\delta}(1/\wncl)}
  = \lambda_n\rhocl(x)\bigl(1+O(e^{-\beta n}+1/n)\bigr),
\]
where the last step uses Lemma~\ref{lem:bohr-sommerfeld-pswf}
and Definition~\ref{def:rho-cl}.
\end{proof}

\section{Proof of the Main Theorem}\label{sec:weaklimit}
%%=================================================================

\begin{theorem}[Semiclassical equidistribution of PSWF densities]%
\label{thm:weak-limit}
For $n \le \gamma N$ and every $f \in C^1([-T,T])$:
\begin{equation}\label{eq:weak-limit}
  \int_{-T}^T f(x)\,\psin(x)^2\,dx
  = \lambda_n(c)\int_{-T}^T f(x)\,\rhocl(x)\,dx
    + R_n(f),
\end{equation}
where:
\begin{equation}\label{eq:remainder}
  |R_n(f)| \le \frac{C_{\delta,\gamma}\,\lambda_n(c)}{n}
               \left(\|f\|_{L^\infty} + T\,\|f'\|_{L^\infty}\right).
\end{equation}
The constant $C_{\delta,\gamma}$ depends only on
$\delta$, $\gamma$, $c$, $T$, not on $n$ or $f$.
\end{theorem}

\begin{proof}
\textit{Step 1 (Bulk cutoff):}
$\int_{-T}^T f\psin^2 = \int_{I_\delta}f\psin^2
+ O(e^{-\beta n}\lambda_n\|f\|_{L^\infty})$
by Lemma~\ref{lem:turning-point-cutoff}.

\textit{Step 2 (Pr\"ufer decomposition):}
$\int_{I_\delta}f\psin^2
= \frac{1}{2}\int_{I_\delta}fr_n^2
- \frac{1}{2}\int_{I_\delta}fr_n^2\cos(2\theta_n)$.

\textit{Step 3 (Oscillation term):}
By Corollary~\ref{cor:osc-slowly-varying} with $a=f$ and
$b_n = r_n^2/r_n(x_0)^2$ ($b_n \in C^1$,
$\|b_n'\|_{L^\infty} = O(1/n)$ by \eqref{eq:amp-dev-deriv}):
$\bigl|\int fr_n^2\cos(2\theta_n)\bigr|
\le \frac{C_{\delta,\gamma}\lambda_n}{n}
(\|f\|_{L^\infty}+T\|f'\|_{L^\infty})$.

\textit{Step 4 (Mean term):}
By \eqref{eq:amp-ratio} and \eqref{eq:rn-vs-rhocl}:
$\frac{1}{2}\int fr_n^2
= \lambda_n\int f\rhocl + O(\lambda_n\|f\|_{L^\infty}/n)$.

\textit{Step 5 (Extension to $[-T,T]$):}
$\int_{[-T,T]\setminus I_\delta}f\rhocl
= O(e^{-\beta n}\|f\|_{L^\infty})$.

\textit{Assembly:}
Steps 1--5 give \eqref{eq:weak-limit}--\eqref{eq:remainder}.
\end{proof}

\begin{remark}[Uniformity]
$C_{\delta,\gamma} \lesssim c_{\delta,\gamma}^{-2}$,
uniform in $n$ and $f$ for fixed parameters.
\end{remark}

\begin{corollary}[Closure of the bulk program]%
\label{cor:bulk-program-closed}
Theorem~\ref{thm:weak-limit} supplies hypothesis~(6.1) of
Lemma~6.3 (\texttt{lem:bulk-reduction}) of \cite{PaperIII}
with $\omega_n = C_{\delta,\gamma}/n$, completing:
\[
  \textup{Thm.~\ref{thm:weak-limit}}
  \Rightarrow \texttt{lem:bulk-reduction}
  \Rightarrow \texttt{prob:comm-refined}
  \Rightarrow \text{Bulk tail bound}
\]
for all $n \le \gamma N$, conditional on Assumption~2.4 of \cite{PaperIII}.
\end{corollary}

\section{Open Problems}\label{sec:open}
%%=================================================================

\begin{enumerate}[(P1)]
  \item \textbf{Extension to $f \in C^0$.}
  Theorem~\ref{thm:weak-limit} requires $f \in C^1$.
  For $f \in C^0$, a density argument gives weak-$*$ convergence
  without a rate. A rate for $f \in C^\alpha$ ($0 < \alpha < 1$)
  via interpolation would extend the result.
  \item \textbf{Off-diagonal regime.}
  An analogue of Theorem~\ref{thm:weak-limit} for
  $|m-n| \ge \delta N$ remains open.
  \item \textbf{Explicit Bohr--Sommerfeld constant.}
  Making the implicit constant in \eqref{eq:BS} fully explicit
  in terms of $c$, $T$, $\gamma$ would allow the remainder in
  Theorem~\ref{thm:weak-limit} to be computed numerically.
\end{enumerate}
