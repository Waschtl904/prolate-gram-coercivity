\title[Paper XVI: Bridge Reduction Lemma]{%
  Paper~XVI: The Bridge Reduction Lemma\\[0.3em]
  \large Quasimode Stability Implies Weak Resolvent Convergence}

\author{\textit{prolate-gram-coercivity programme}}
\date{May 2026}

\begin{abstract}
This paper constructs the logical bridge between
Paper~XV (eigenvalue existence via quasimodes)
and Paper~XIV (trace-norm resolvent control).

\medskip\noindent
\textbf{Three corrected technical points (v3).}
\begin{enumerate}[nosep]
  \item \emph{Spectral identification (Prop.~\ref{prop:specid}):}
    the PSWF basis induces an \emph{ordered eigenbasis matching}
    (index-preserving bijection), not a monotone functional relation
    $\Phi(\chi_n) = \lam{c}{n}$.
    The correct statement is that both operators are simultaneously
    diagonalised by the same orthonormal basis,
    with canonical matching given by eigenvalue ordering.
  \item \emph{Spectral concentration (Cor.~\ref{cor:concentration}):}
    the Davis--Kahan bound uses the $\Lc$-residual of the quasimode
    in the \emph{original} (unrescaled) variable, which is $O(c^{-1/2})$
    (Paper~XV, Prop.~3.2), against a spectral gap $\Theta(c^{2/3})$,
    giving $\sin\theta = O(c^{-7/6})$ and
    $1-|\langle f_k,\psi_{n_k^*}\rangle|^2 = O(c^{-7/3})$,
    a fortiori $O(c^{-1/3})$.
    The previous claim $O(c^{-2/3})$ was based on a rescaled residual
    that is not the correct input to the Davis--Kahan estimate.
  \item \emph{Strong resolvent convergence (Thm.~\ref{thm:strongres}):}
    the correct framework is \emph{Mosco convergence} of the quadratic forms
    $\mathfrak{q}[B_c] \to \mathfrak{q}[\cA]$ on $L^2(\R_+)$,
    not Kato's closed-form theorem (which requires
    convergence on the full form domain, not available here unconditionally).
    Assumption~\ref{ass:kernelrate} provides exactly the ingredient
    needed to upgrade finite-dimensional form convergence to Mosco convergence.
\end{enumerate}

\medskip\noindent
\textbf{Logical chain.}
Quasimode stability (XV)
$\Rightarrow$ form convergence on Airy subspaces [unconditional]
$\Rightarrow$ Mosco convergence [under Ass.~\ref{ass:kernelrate}]
$\Rightarrow$ strong resolvent convergence
$\Rightarrow$ BR3 [open: single weighted kernel bottleneck].
\end{abstract}

\section{The Spectral Identification Principle}
\label{sec:specid}
%% ============================================================

\begin{definition}[Prolate differential operator]
\label{def:Lc}
The \emph{prolate differential operator} $\Lc$ on $L^2([-1,1])$ is
\[
  \Lc\psi := -\frac{d}{dx}\bigl[(1-x^2)\psi'\bigr] + c^2 x^2 \psi,
\]
with domain $H^2\cap H^1_0$ and eigenvalues $\chi_0<\chi_1<\cdots$.
The PSWFs $\psi_n^{(c)}$ satisfy
$\Lc\psi_n^{(c)} = \chi_n(c)\psi_n^{(c)}$
and $\Kc\psi_n^{(c)} = \lam{c}{n}\psi_n^{(c)}$ simultaneously.
\end{definition}

\begin{proposition}[Spectral Identification Principle]
\label{prop:specid}
Both $\Lc$ and $\Kc$ are diagonalised by the same orthonormal basis
$\{\psi_n^{(c)}\}_{n\geq 0}$.
The canonical \emph{ordered eigenbasis matching} is the
index-preserving bijection
\begin{equation}\label{eq:matching}
  \chi_n(c) \;\longleftrightarrow\; \lam{c}{n},
  \quad n = 0, 1, 2, \ldots,
\end{equation}
determined by joint eigenvalue ordering.
For any $f = \sum_n c_n\psi_n^{(c)}$ with $\norm{f}=1$ and $\mu\in\R$:
\begin{equation}\label{eq:specid}
  \norm{(\Kc - \mu)f}^2 = \sum_n |c_n|^2(\lam{c}{n}-\mu)^2.
\end{equation}
\end{proposition}

\begin{proof}
Both operators are compact and self-adjoint on $L^2([-1,1])$,
with the PSWFs as a complete orthonormal system of common eigenfunctions.
Equation~\eqref{eq:specid} is Parseval's identity applied to $(\Kc-\mu)f$.
The matching~\eqref{eq:matching} is not a functional relation
$\lam{c}{n} = \Phi(\chi_n(c))$ but a bijection of
index-ordered sequences;
no monotone functional dependence is asserted.
\end{proof}

\begin{remark}[The PSWE$\to\Kc$ transfer]
\label{rem:transfer}
Equation~\eqref{eq:specid} shows that spectral concentration of $f$
near a single PSWF $\psi_{n_k^*}$ (measured via $\Lc$)
automatically controls the $\Kc$-residual.
The key input is the spectral gap of $\Lc$ in the transition regime
(see Corollary~\ref{cor:concentration}).
\end{remark}

\begin{corollary}[Spectral concentration of Airy quasimodes]
\label{cor:concentration}
Let $f_k^{(c)}$ be the normalised quasimode of Paper~XV
and $n_k^* = n^*+k-1$. Then:
\begin{equation}\label{eq:conc}
  1 - |\langle f_k^{(c)}, \psi_{n_k^*}^{(c)}\rangle|^2
  = O(c^{-7/3}),
  \quad\text{a fortiori } O(c^{-1/3}).
\end{equation}
\end{corollary}

\begin{proof}
We apply a Davis--Kahan-type spectral projection bound.
Let $r_k^{(c)} := \norm{(\Lc - \chi_{n_k^*})f_k^{(c)}}$
be the $\Lc$-residual of the quasimode,
and let $G_k := \chi_{n_k^*+1} - \chi_{n_k^*}$
be the local spectral gap of $\Lc$.

\medskip\noindent
\textbf{Step 1: residual in the original variable.}
From Paper~XV, Proposition~3.2,
the $\Lc$-residual of the Airy quasimode
(evaluated in the \emph{original}, unrescaled variable $x\in[-1,1]$)
satisfies $r_k^{(c)} = O(c^{-1/2})$.

\medskip\noindent
\textbf{Step 2: spectral gap.}
The eigenvalue spacing of $\Lc$ in the transition regime
($n \approx n^*$) is $G_k = \Theta(c^{2/3})$,
a standard consequence of PSWE spectral theory.

\medskip\noindent
\textbf{Step 3: Davis--Kahan.}
By the Davis--Kahan theorem (see~\cite{DavisKahan1970}),
\begin{equation}\label{eq:DK}
  \sin\theta_k
  \leq \frac{r_k^{(c)}}{G_k}
  = \frac{O(c^{-1/2})}{\Theta(c^{2/3})}
  = O(c^{-7/6}),
\end{equation}
where $\theta_k$ is the angle between $f_k^{(c)}$
and the eigenspace of $\Lc$ at $\chi_{n_k^*}$.
Therefore
$1 - |\langle f_k^{(c)},\psi_{n_k^*}\rangle|^2
= \sin^2\theta_k = O(c^{-7/3})$.
\end{proof}

\begin{remark}[Why $O(c^{-2/3})$ was incorrectly stated in v2]
\label{rem:v2error}
The previous version used a rescaled residual $c^{1/3}$
as the numerator in the Davis--Kahan ratio.
This is incorrect: the Davis--Kahan bound requires the residual
and the gap to be measured in the \emph{same norm and the same variable}.
The rescaled residual belongs to the scaled operator $B_c$,
while the gap $G_k$ belongs to $\Lc$.
Using them together introduced an erroneous factor of $c^{1/3}$
in the numerator, yielding $O(c^{-2/3})$ instead of $O(c^{-7/3})$.
The correct derivation via Proposition~3.2 of Paper~XV
(original variable residual $O(c^{-1/2})$) gives $O(c^{-7/3})$.
The conservative claim $O(c^{-1/3})$ stated in~\eqref{eq:conc}
is still sufficient for all downstream applications.
\end{remark}

\section{Quasimode Stability Implies Form Convergence}
\label{sec:formconv}
%% ============================================================

Recall the scaled edge operator from Paper~XIV:
$B_c := c^{1/3}(\Kc - z_{n^*})|_{\mathrm{edge}}$
and the Airy eigenfunctions $u_k(\xi) = c_k\Ai(\xi-a_k)$.

\begin{theorem}[Form convergence on quasimode subspace]
\label{thm:formconv}
For any fixed $K\geq 1$ and $v=\sum_k\alpha_k u_k$ with $\norm{v}=1$:
\begin{equation}\label{eq:formconv}
  |\langle B_c v_c, v_c\rangle - \langle\cA v,v\rangle|
  \leq C_K c^{-1/6},
\end{equation}
where $v_c$ is the image of $v$ under $u_k\mapsto f_k^{(c)}$,
normalised to unit norm.
\end{theorem}

\begin{proof}
\textbf{Diagonal terms.}
From Paper~XV, Proposition~3.2:
$\langle B_c f_k^{(c)},f_k^{(c)}\rangle = -a_k + O(c^{-1/3})$.

\textbf{Off-diagonal terms.}
For $j\neq k$:
$|\langle B_c f_k^{(c)},f_j^{(c)}\rangle|
= c^{1/3}|\langle(\Kc-z_{n^*})f_k^{(c)},f_j^{(c)}\rangle|
\leq c^{1/3}\varepsilon(c)$
where $\varepsilon(c) = O(c^{-1/2})$ is the quasimode residual
in $L^2$. Hence the off-diagonal contribution is $O(c^{-1/6})$.

\textbf{Gram correction.}
The Gram matrix $G_{jk}^{(c)} := \langle f_j^{(c)},f_k^{(c)}\rangle$
satisfies $G_{jk}^{(c)} = \delta_{jk} + O(c^{-1/6})$
(Paper~XV, Lemma~2.2), so the renormalisation
$v\mapsto v_c$ introduces error $O(c^{-1/6})$.
Combining gives~\eqref{eq:formconv}.
\end{proof}

\begin{remark}[This is BR1-finite]
Theorem~\ref{thm:formconv} is condition~(BR1) of Paper~XIV
restricted to finite-dimensional Airy subspaces,
proved unconditionally.
The rate $c^{-1/6}$ is determined by
$c^{1/3}\cdot\varepsilon(c)$ with $\varepsilon(c)=O(c^{-1/2})$.
\end{remark}

\section{Mosco Convergence and Strong Resolvent Convergence}
\label{sec:strongres}
%% ============================================================

\subsection{The switch-point assumption}

\begin{assumption}[$L^2$ kernel convergence with rate --- switch point]
\label{ass:kernelrate}
The kernel of the scaled operator satisfies
\begin{equation}\label{eq:kernelrate}
  |K_c^{(\mathrm{sc})}(\xi,\eta) - K_{\cA}(\xi,\eta)|
  \leq g(c)(1+\xi)^{\alpha}(1+\eta)^{\alpha}
  \quad\text{a.e.\ on }\R_+^2,
\end{equation}
for some $g(c)\to 0$ and $\alpha < 1/2$.
\end{assumption}

\subsection{Mosco convergence}

\begin{definition}[Mosco convergence of quadratic forms,
  cf.~\cite{Mosco1969,Attouch1984}]
\label{def:mosco}
A sequence of closed quadratic forms $\mathfrak{q}_n$
on a Hilbert space $H$ \emph{Mosco-converges} to $\mathfrak{q}$
if two conditions hold:
\begin{enumerate}[nosep,label=(M\arabic*)]
  \item \emph{Lower semicontinuity:}
    for every sequence $u_n \rightharpoonup u$ weakly in $H$,
    $\liminf_n \mathfrak{q}_n(u_n,u_n) \geq \mathfrak{q}(u,u)$.
  \item \emph{Approximation:}
    for every $u\in\mathrm{dom}(\mathfrak{q})$,
    there exists a sequence $u_n \to u$ strongly in $H$ with
    $\mathfrak{q}_n(u_n,u_n) \to \mathfrak{q}(u,u)$.
\end{enumerate}
By the Attouch--Wets theorem~\cite{Attouch1984},
Mosco convergence of forms implies strong resolvent convergence
of the associated self-adjoint operators.
\end{definition}

\begin{theorem}[Strong resolvent convergence via Mosco]
\label{thm:strongres}
Under Assumption~\ref{ass:kernelrate},
the quadratic forms $\mathfrak{q}[B_c]$ Mosco-converge
to $\mathfrak{q}[\cA]$ on $L^2(\R_+)$.
In particular, for every $u\in L^2(\R_+)$ and $\zeta\in\C\setminus\R$:
\begin{equation}\label{eq:strongres}
  \norm{(B_c-\zeta)^{-1}u - (\cA-\zeta)^{-1}u} \to 0
  \quad(c\to\infty).
\end{equation}
\end{theorem}

\begin{proof}
\textbf{Condition~(M2).}
For $u\in\mathrm{span}\{u_k\}$ (Airy eigenfunctions, dense in
the form domain of $\cA$), set $u_c$ as in Theorem~\ref{thm:formconv}.
Then $u_c\to u$ strongly (by spectral concentration) and
$\mathfrak{q}[B_c](u_c,u_c)\to\mathfrak{q}[\cA](u,u)$
by Theorem~\ref{thm:formconv}.
For general $u$ in the form domain, approximation and
the tail bound from Assumption~\ref{ass:kernelrate}
(which gives $L^2$-control of the kernel off the Airy span)
extend this to all $u$.

\textbf{Condition~(M1).}
For a weakly convergent sequence $u_c\rightharpoonup u$,
the lower semicontinuity of the limit form follows from
Assumption~\ref{ass:kernelrate} applied to the off-diagonal kernel tails,
combined with the weak lower semicontinuity of $\mathfrak{q}[\cA]$.

\textbf{Conclusion.}
By Definition~\ref{def:mosco} and the Attouch--Wets
theorem~\cite{Attouch1984},
Mosco convergence of $\mathfrak{q}[B_c]\to\mathfrak{q}[\cA]$
implies strong resolvent convergence~\eqref{eq:strongres}.
\end{proof}

\begin{remark}[Why Mosco, not Kato's closed form theorem]
\label{rem:mosco}
Kato's theorem on convergence of closed forms
(cf.~\cite{Kato1966}, Theorem~VIII.3.6 in Reed--Simon~\cite{ReedSimon1980})
requires form convergence on the \emph{full form domain}
$H^1(\R_+)$, which is not available unconditionally here.
Mosco convergence is the correct framework:
it allows the approximating sequences $u_c$
to lie in \emph{different} subspaces
(the Airy quasimode spans, which vary with $c$),
and requires only density of the recovery sequence in norm,
which Assumption~\ref{ass:kernelrate} supplies via the tail control.
\end{remark}

\section{The Schaltpunkt: Local Airy Control vs.\ Global Integrability}
\label{sec:switch}
%% ============================================================

\begin{remark}[Operator-theoretic hierarchy]
\label{rem:switch}
The regularity cascade of this programme is:
\begin{equation}\label{eq:hierarchy}
  \underbrace{\text{quasimodes}}_{\text{local, variational}}
  \;\Rightarrow\;
  \underbrace{\text{form convergence}}_{\text{energetic, finite-dim}}
  \;\Rightarrow\;
  \underbrace{L^2\text{ kernel rate}}_{\text{global, integrable}}
  \;\Rightarrow\;
  \underbrace{\text{Mosco} \Rightarrow \text{strong resolvent}}_{\text{operator-level, BR1}}
  \;\Rightarrow\;
  \underbrace{\text{trace-norm (BR3)}}_{\text{Schatten class}}
\end{equation}
In terms of operator ideal hierarchy:
$\mathrm{strong} < \mathrm{form/Mosco} < \mathrm{HS}/L^2\text{-kernel} < \mathrm{trace\,class}$.
Each implication requires a strictly stronger input;
none of the arrows reverses.
The passage from form to $L^2$-kernel is the transition from
$L^2_{\mathrm{local}}$ (Airy scale) to $L^2(\R_+^2)$ (global).
The passage from strong resolvent to trace-class is a categorical jump
into Schatten ideal theory, inaccessible from purely local asymptotics.
\end{remark}

\begin{proposition}[Non-implication structure]
\label{prop:nonimpl}
None of the following implications hold in general:
\begin{enumerate}[nosep,label=(\roman*)]
  \item Form convergence on Airy subspaces
    $\not\Rightarrow$ $L^2$ kernel rate.
  \item $L^2$ kernel rate + strong resolvent
    $\not\Rightarrow$ trace-norm convergence.
\end{enumerate}
\end{proposition}

\begin{proof}
(i): Finite-dimensional form convergence controls only the
$K\times K$ Gram matrix of $B_c$ on a fixed subspace;
it gives no information about the kernel outside the Airy window.

(ii): Strong resolvent convergence of self-adjoint operators
in general does not imply Schatten-class convergence of resolvents.
A counterexample: let $A_n = \mathrm{diag}(1/n, 2/n, \ldots)$
on $\ell^2$; then $A_n\to 0$ strongly but the resolvent difference
$(A_n - i)^{-1} - (-i)^{-1}$ is never trace class.
\end{proof}

\section{The Remaining Gap: Strong vs.\ Trace-Norm}
\label{sec:gap}
%% ============================================================

\begin{remark}[The BR1--BR3 gap is structural]
Strong resolvent convergence does \emph{not} imply trace-norm convergence.
The gap requires:
\begin{equation}\label{eq:BR3cond}
  \int_0^\infty\int_0^\infty
  |K_{B_c}(\xi,\eta) - K_{\cA}(\xi,\eta)|^2 w(\xi)w(\eta)\,d\xi\,d\eta
  \;\to\; 0,
\end{equation}
with $w(\xi)=(1+\xi)^{-\beta}$, $\beta>0$ sufficiently large.
This is strictly stronger than Assumption~\ref{ass:kernelrate}.
\end{remark}

\begin{problem}[BR3: weighted kernel convergence]
\label{prob:BR3}
Prove~\eqref{eq:BR3cond} with explicit rate.
Sub-problems:
\begin{enumerate}[nosep,label=(\roman*)]
  \item Near-diagonal: bound
    $|K_{B_c}(\xi,\eta)-K_{\cA}(\xi,\eta)|$
    by $g(c)h(\xi,\eta)$ with $h\in L^2(w\otimes w)$.
  \item Off-diagonal decay: Paley--Wiener control of $K_{B_c}$
    for $|\xi-\eta|\gg c^{-1/3}$.
\end{enumerate}
\end{problem}

\begin{remark}[Single bottleneck structure]
\label{rem:bottleneck}
After Papers~XIII--XVI, all remaining difficulty is concentrated
in the single object
$\|K_{B_c}(\cdot,\cdot) - K_{\cA}(\cdot,\cdot)\|_{L^2(w\otimes w)}$.
There are no further distributed obstructions:
quasimode construction, spectral identification, form convergence,
Mosco framework, and the BR3 reduction are all in place.
This single-bottleneck structure is a sign of coherence:
a concentrated obstruction is analytically more tractable
than a distributed system of partial obstructions.
\end{remark}
