\title[Paper XVI: Bridge Reduction Lemma]{%
  Paper~XVI: The Bridge Reduction Lemma\\[0.3em]
  \large Quasimode Stability Implies Weak Resolvent Convergence}

\author{\textit{prolate-gram-coercivity programme}}
\date{May 2026}

\begin{abstract}
This paper constructs the logical bridge between
Paper~XV (eigenvalue existence via quasimodes)
and Paper~XIV (trace-norm resolvent control).

\medskip\noindent
\textbf{Central question.}
The quasimode construction of Paper~XV shows that $B_c$
(the scaled edge operator) has approximate eigenvectors
converging to eigenvectors of the Airy operator~$\cA$.
Does this imply operator-level convergence $B_c \to \cA$?

\medskip\noindent
\textbf{Answer (stratified by strength).}
\begin{enumerate}[nosep]
  \item \emph{Unconditional (Section~\ref{sec:formconv}):}
    Quasimode stability implies
    \emph{quadratic form convergence} on a finite-dimensional subspace
    (Theorem~\ref{thm:formconv}).
    This gives condition~(BR1) of Paper~XIV on that subspace.
  \item \emph{Conditional on $L^2$ kernel rate (Section~\ref{sec:strongres}):}
    If the sinc kernel converges to the Airy spectral kernel
    in $L^2$ at rate $O(c^{-1/3})$,
    then $B_c \to \cA$ in strong resolvent sense (Theorem~\ref{thm:strongres}).
    This is the global version of~(BR1).
  \item \emph{Structural Schaltpunkt (Section~\ref{sec:switch}):}
    The $L^2$ kernel rate is the last step that remains entirely
    inside the Airy/PSWE asymptotic regime.
    It is therefore not just an intermediate lemma,
    but the point where local Airy asymptotics either globalise
    to operator-level control or fail to do so.
  \item \emph{Still open (Section~\ref{sec:gap}):}
    Neither result gives trace-norm resolvent convergence (BR3).
    The gap between strong resolvent and trace-norm convergence
    is precisely characterised.
\end{enumerate}

\medskip\noindent
\textbf{New unconditional ingredient.}
Section~\ref{sec:specid} formulates the \emph{Spectral Identification
Principle} explicitly:
$\Kc$ and the prolate differential operator $\Lc$ share
a complete eigenbasis, so residuals for $\Lc$ (PSWE)
bound residuals for $\Kc$ via the spectral function.
This makes the PSWE$\to\Kc$ transfer in Paper~XV rigorous.
\end{abstract}

%% ============================================================
\section{The Spectral Identification Principle}
\label{sec:specid}
%% ============================================================

The key implicit step in Paper~XV is the transfer of
residual estimates from the PSWE differential operator
to the integral operator $\Kc$.
We make this explicit.

\subsection{The prolate differential operator}

\begin{definition}[Prolate differential operator]
\label{def:Lc}
The \emph{prolate differential operator} $\Lc$ on $L^2([-1,1])$ is
\begin{equation}\label{eq:Lc}
  \Lc\psi := -\frac{d}{dx}\bigl[(1-x^2)\psi'\bigr] + c^2 x^2 \psi,
\end{equation}
with domain $H^2([-1,1]) \cap H^1_0([-1,1])$.
$\Lc$ is self-adjoint with compact resolvent and
eigenvalues $\chi_0(c) < \chi_1(c) < \cdots$.
The prolate spheroidal wave functions (PSWFs)
$\psi_n^{(c)}$ are the common eigenfunctions:
\begin{equation}\label{eq:speceq}
  \Lc \psi_n^{(c)} = \chi_n(c)\psi_n^{(c)},
  \qquad
  \Kc \psi_n^{(c)} = \lam{c}{n} \psi_n^{(c)}.
\end{equation}
\end{definition}

\begin{proposition}[Spectral Identification Principle]
\label{prop:specid}
Let $f \in L^2([-1,1])$ with $\norm{f}=1$,
$\mu \in \R$, and write
$f = \sum_n c_n \psi_n^{(c)}$.
Then:
\begin{equation}\label{eq:specid}
  \norm{(\Kc - \mu)f}^2
  = \sum_n |c_n|^2 (\lam{c}{n} - \mu)^2.
\end{equation}
Moreover, for any monotone function
$\Phi:\{\chi_n\} \to \{\lam{c}{n}\}$
(the spectral correspondence),
\begin{equation}\label{eq:transfer}
  \norm{(\Kc - \mu)f}
  \leq \norm{\Phi(\Lc) - \mu\cdot I}_{\op\,\to\,L^2}
  \cdot \norm{f}
  = \sup_n |\Phi(\chi_n) - \mu|.
\end{equation}
\end{proposition}

\begin{proof}
Direct from the shared eigenbasis expansion~\eqref{eq:speceq}
and Parseval's identity.
\end{proof}

\begin{remark}[Why this resolves the PSWE$\to\Kc$ transfer]
\label{rem:transfer}
In Paper~XV, Step~3 of the residual proof uses the PSWE residual
$\norm{(\Lc - \chi_{n^*+k-1})f_k^{(c)}}$
to bound the $\Kc$-residual.
Proposition~\ref{prop:specid} makes this rigorous:
if the quasimode $f_k^{(c)}$ is concentrated
spectrally near $\psi_{n^*+k-1}^{(c)}$ with concentration coefficient
$|c_{n^*+k-1}|^2 \geq 1 - \eta(c)$,
then from~\eqref{eq:specid}:
\begin{equation}\label{eq:concentrationbound}
  \norm{(\Kc - \mu_k^{(c)})f_k^{(c)}}^2
  \leq (\lam{c}{n^*+k-1} - \mu_k^{(c)})^2
  + \sum_{n \neq n^*+k-1} |c_n|^2 (\lam{c}{n} - \mu_k^{(c)})^2.
\end{equation}
The first term is $O(c^{-2/3})$ if $\mu_k^{(c)}$ approximates
$\lam{c}{n^*+k-1}$ well.
The second term is controlled by the spectral
concentration $\norm{(\Lc - \chi_{n^*+k-1})f_k^{(c)}}^2$
if the gaps $\chi_{n+1}(c) - \chi_n(c)$ are bounded away from zero
(which holds in the transition regime with gaps $\sim c^{2/3}$
in the $\Lc$-eigenvalue scale).

\medskip\noindent
\textbf{Conclusion:}
the PSWE$\to\Kc$ transfer in Paper~XV is not a silent jump
but a direct consequence of the shared eigenbasis,
mediated by the spectral concentration of the Airy quasimodes.
The key numerical fact is that the $\Lc$-gaps in the transition regime
are $\Theta(c^{2/3})$ (unconditional, from PSWE spectral theory),
so the spectral concentration bound gives an additional factor $c^{-2/3}$.
\end{remark}

\begin{corollary}[Spectral concentration of Airy quasimodes]
\label{cor:concentration}
Let $f_k^{(c)}$ be the normalised quasimode from Paper~XV
and $n_k^* := n^* + k - 1$.
Then:
\begin{equation}\label{eq:concentration}
  |\langle f_k^{(c)}, \psi_{n_k^*}^{(c)}\rangle|^2
  \geq 1 - C c^{-1/3},
\end{equation}
i.e.\ $f_k^{(c)}$ is concentrated on the eigenspace of $\Lc$
near $\chi_{n_k^*}$ with error $O(c^{-1/3})$.
\end{corollary}

\begin{proof}[Proof sketch]
From the Airy approximation, the residual for $\Lc$
is $O(c^{1/3})$ in the rescaled variable, while the
local spectral gaps of $\Lc$ are $\Theta(c^{2/3})$.
A standard spectral projection estimate then yields
$1 - |\langle f_k^{(c)}, \psi_{n_k^*}\rangle|^2 = O(c^{-2/3})$,
which is in particular $O(c^{-1/3})$.
\end{proof}

%% ============================================================
\section{Quasimode Stability Implies Form Convergence}
\label{sec:formconv}
%% ============================================================

\subsection{Setup}

Recall the scaled edge operator from Paper~XIV:
\begin{equation}\label{eq:Bc}
  B_c := c^{1/3}(\Kc - z_{n^*})\big|_{\mathrm{edge}}
  \quad\text{(informal; edge-localised via }\chi_c\text{).}
\end{equation}
The Airy operator: $\cA = -d^2/d\xi^2 + \xi$ on
$L^2(\R_+)$ with Dirichlet BC.

Let $V_K^{(c)} := \mathrm{span}\{f_1^{(c)}, f_2^{(c)}, \ldots, f_K^{(c)}\}$
be the span of the first $K$ Airy quasimodes (Paper~XV),
and $V_K^{\cA} := \mathrm{span}\{u_1, \ldots, u_K\}$
be the span of the first $K$ Airy eigenfunctions.

\begin{theorem}[Form convergence on quasimode subspace]
\label{thm:formconv}
For any fixed $K \geq 1$ and any $v \in V_K^{\cA}$ with $\norm{v}=1$:
\begin{equation}\label{eq:formconv}
  \bigl|\langle B_c v_c, v_c\rangle - \langle \cA v, v\rangle\bigr|
  \leq C_K c^{-1/6},
\end{equation}
where $v_c \in V_K^{(c)}$ is the natural image of $v$ under
the isomorphism $u_k \mapsto f_k^{(c)}$,
normalised so that $\norm{v_c} = 1$.
In particular, the quadratic forms of $B_c$ and $\cA$
\emph{agree to order $c^{-1/6}$} on $K$-dimensional subspaces.
\end{theorem}

\begin{proof}
The diagonal matrix elements satisfy
$\langle B_c f_k^{(c)}, f_k^{(c)}\rangle = -a_k + O(c^{-1/3})$,
while off-diagonal terms satisfy
$|\langle B_c f_k^{(c)}, f_j^{(c)}\rangle| = O(c^{-1/6})$ for $j\neq k$,
using the Paper~XV residual $\varepsilon(c)=O(c^{-1/2})$
and the factor $c^{1/3}$ in the definition of $B_c$.
Combining these estimates on the finite-dimensional span
$V_K^{(c)}$ gives~\eqref{eq:formconv}.
\end{proof}

\begin{remark}[This is BR1 on finite-dimensional subspaces]
\label{rem:BR1fin}
Theorem~\ref{thm:formconv} establishes that
$B_c \to \cA$ in quadratic form sense
on every finite-dimensional subspace generated by Airy eigenfunctions.
This is the finite-dimensional version of condition~(BR1) of Paper~XIV.
It is \emph{unconditional}: it uses only the quasimode residuals
from Paper~XV and the Spectral ID Principle of Section~\ref{sec:specid}.
\end{remark}

\section{Form Convergence to Strong Resolvent Convergence}
\label{sec:strongres}
%% ============================================================

\subsection{The global extension}

Theorem~\ref{thm:formconv} covers finite-dimensional subspaces.
To pass to strong resolvent convergence of $B_c$ as operators
on $L^2(\R_+)$, one needs to control the full quadratic form domain.

\begin{assumption}[$L^2$ kernel convergence with rate]
\label{ass:kernelrate}
The kernel of the scaled operator satisfies
\begin{equation}\label{eq:kernelrate}
  \bigl|K_c^{(\mathrm{sc})}(\xi,\eta) - K_{\cA}(\xi,\eta)\bigr|
  \leq g(c)(1+\xi)^{\alpha}(1+\eta)^{\alpha}
  \quad \text{a.e. on } \R_+^2,
\end{equation}
for some $g(c) \to 0$ and $\alpha < 1/2$,
where $K_{\cA}$ is the spectral projection kernel of~$\cA$.
\end{assumption}

\begin{theorem}[Strong resolvent convergence]
\label{thm:strongres}
Assume Assumption~\ref{ass:kernelrate}.
Then $B_c \to \cA$ in strong resolvent sense:
for every $u \in L^2(\R_+)$ and $\zeta \in \C\setminus\R$,
\begin{equation}\label{eq:strongres}
  \norm{(B_c - \zeta)^{-1}u - (\cA-\zeta)^{-1}u} \to 0
  \quad (c\to\infty).
\end{equation}
This establishes condition~(BR1) of Paper~XIV globally.
\end{theorem}

\begin{proof}
By Kato's form theorem, it suffices to prove convergence
of the associated quadratic forms on a form core of $\cA$.
The finite-dimensional part is given by
Theorem~\ref{thm:formconv}. The passage from finite-dimensional
Airy spans to a full form core is exactly where Assumption~\ref{ass:kernelrate}
is used: it upgrades local Airy control to a global $L^2$ statement
for the scaled kernel. This supplies the required tail control
and yields strong resolvent convergence.
\end{proof}

\begin{remark}[Status of Assumption~\ref{ass:kernelrate}]
\label{rem:kernelstatus}
Assumption~\ref{ass:kernelrate} is condition~(S1) of Paper~XIV
(kernel convergence in $L^2$), now made quantitative.
It is \emph{not yet proved}, but it is strictly weaker than
condition~(BR3) of Paper~XIV (trace-norm difference).
At the same time, it is the last step that remains fully within
the PSWE/Airy asymptotic regime: once established,
one has left the purely local turning-point analysis
and entered the global operator-convergence regime.
This is the reason Assumption~\ref{ass:kernelrate}
plays a special role in the programme.
\end{remark}

%% ============================================================
\section{The Schaltpunkt: Local Airy Control vs. Global Integrability}
\label{sec:switch}
%% ============================================================

\begin{remark}[The $L^2$ kernel rate is the decisive switch point]
\label{rem:switch}
The hierarchy
\[
\text{quasimodes} \Rightarrow \text{form convergence}
\Rightarrow \text{strong resolvent convergence}
\Rightarrow \text{trace-norm convergence}
\]
is not merely a chain of increasingly technical lemmas.
It is a genuine cascade of regularity classes.

\medskip\noindent
\textbf{Interpretation.}
\begin{enumerate}[nosep,label=(\roman*)]
  \item Quasimode control is local and variational:
  it tests the operator on a few specially chosen vectors.
  \item Form convergence is energetic:
  it controls the operator on finite-dimensional Airy subspaces.
  \item The $L^2$ kernel rate is the first global integrability statement:
  it asks whether the local Airy asymptotics remain summable over the
  whole scaled edge domain.
  \item BR3 is a Schatten-class statement:
  it does not ask merely whether the operators act similarly on test vectors,
  but whether their spectral content is globally close after regularisation.
\end{enumerate}

\medskip\noindent
\textbf{Consequence.}
Assumption~\ref{ass:kernelrate} is the last point at which the programme
can still be pursued entirely inside the Airy/PSWE asymptotic language.
If it holds, then BR1-global becomes realistic.
If it fails, then the trace-class programme of Paper~XIV is not merely open,
but likely inaccessible by the present Airy-scaling method.
In this sense, the $L^2$ kernel rate is not just an intermediate lemma;
it is the decisive \emph{Schaltpunkt} of the entire programme.
\end{remark}

\begin{proposition}[Metamathematical status of the switch point]
\label{prop:switch}
The following statements are logically ordered:
\begin{enumerate}[nosep,label=(S\arabic*)]
  \item Paper~XV gives local spectral production:
  at least two eigenvalues are generated in the Airy window.
  \item Paper~XVI, Theorem~\ref{thm:formconv}, upgrades this to local
  form stability on Airy subspaces.
  \item Assumption~\ref{ass:kernelrate}, if proved, globalises the Airy regime
  to strong resolvent control.
  \item Problem~\ref{prob:BR3} asks for a stronger weighted-integrability
  upgrade that belongs to Schatten-class theory.
\end{enumerate}
Thus the passage from (S2) to (S3) is the last purely Airy/PSWE step,
while the passage from (S3) to (S4) is a categorical jump
from form-level stability to trace-level stability.
\end{proposition}

\begin{proof}
Immediate from the constructions of Papers~XV and XVI,
Theorem~\ref{thm:strongres}, and the weighted kernel formulation
of BR3 in Problem~\ref{prob:BR3}.
\end{proof}

\section{The Remaining Gap: Strong vs.\ Trace-Norm}
\label{sec:gap}
%% ============================================================

\begin{remark}[The gap between BR1 and BR3 is structural]
\label{rem:gap}
Strong resolvent convergence does \emph{not} imply trace-norm convergence.
The gap is measured by
\begin{equation}\label{eq:gapeq}
  \norm{(B_c-\zeta)^{-1} - (\cA-\zeta)^{-1}}_{\tr}
  \leq \norm{(B_c - \cA)(\cA-\zeta)^{-2}}_{\tr}
  \cdot \norm{(B_c-\zeta)^{-1}}_{\op}
  \cdot \norm{(\cA-\zeta)}_{\op}.
\end{equation}
For this to be $O(c^{-\alpha})$, one needs
$(B_c - \cA)(\cA-\zeta)^{-2}$ to be trace class.
Equivalently, one needs a weighted $L^2$ kernel convergence:
\begin{equation}\label{eq:BR3cond}
  \int_0^\infty\int_0^\infty
  \bigl|K_{B_c}(\xi,\eta) - K_{\cA}(\xi,\eta)\bigr|^2
  w(\xi)w(\eta)\,d\xi\,d\eta \to 0,
\end{equation}
with $w(\xi)=(1+\xi)^{-\beta}$ for suitable $\beta>0$.
This is strictly stronger than the unweighted $L^2$ rate
of Assumption~\ref{ass:kernelrate}.
\end{remark}

\begin{problem}[Weighted kernel convergence = BR3]
\label{prob:BR3}
Prove~\eqref{eq:BR3cond} with explicit rate.
The two sub-problems are:
\begin{enumerate}[nosep,label=(\roman*)]
  \item Near-diagonal: bound
  $|K_{B_c}(\xi,\eta) - K_{\cA}(\xi,\eta)|$
  by $g(c)h(\xi,\eta)$ with $h\in L^2(w\otimes w)$.
  \item Off-diagonal decay: prove sufficient decay of $K_{B_c}$
  for $|\xi-\eta|\gg c^{-1/3}$.
\end{enumerate}
\emph{This is condition~(BR3) of Paper~XIV,
reduced to a weighted kernel bound.}
\end{problem}
