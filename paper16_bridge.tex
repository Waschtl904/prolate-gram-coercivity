\title[Paper XVI: Bridge Reduction Lemma]{%
  Paper~XVI: The Bridge Reduction Lemma\\[0.3em]
  \large Quasimode Stability Implies Weak Resolvent Convergence}

\author{\textit{prolate-gram-coercivity programme}}
\date{May 2026}

\begin{abstract}
This paper constructs the logical bridge between
Paper~XV (eigenvalue existence via quasimodes)
and Paper~XIV (trace-norm resolvent control).

\medskip\noindent
\textbf{Central question.}
The quasimode construction of Paper~XV shows that $B_c$
(the scaled edge operator) has approximate eigenvectors
converging to eigenvectors of the Airy operator~$\cA$.
Does this imply operator-level convergence $B_c \to \cA$?

\medskip\noindent
\textbf{Answer (stratified by strength).}
\begin{enumerate}[nosep]
  \item \emph{Unconditional (\S\ref{sec:formconv}):}
    Quasimode stability implies
    \emph{quadratic form convergence} on finite-dimensional Airy subspaces
    (Theorem~\ref{thm:formconv}), giving condition~(BR1) of Paper~XIV locally.
  \item \emph{Conditional on $L^2$ kernel rate (\S\ref{sec:strongres}):}
    If the sinc kernel converges to the Airy spectral kernel in~$L^2$
    at rate $O(c^{-1/3})$, then $B_c \to \cA$ in strong resolvent sense.
  \item \emph{Decisive switch point (\S\ref{sec:switch}):}
    The $L^2$ kernel rate is the last step inside the Airy/PSWE regime.
    Beyond it, a categorical jump in regularity (from energy level to Schatten class) is required.
  \item \emph{Still open (\S\ref{sec:gap}):}
    The trace-norm gap (BR3) is precisely characterised
    as a \emph{single bottleneck}: one weighted kernel convergence statement
    subsumes all remaining difficulty.
\end{enumerate}

\medskip\noindent
\textbf{Sharpened ingredient.}
Corollary~\ref{cor:concentration} gives the \emph{sharp} spectral
concentration rate $O(c^{-2/3})$ (not the conservative $O(c^{-1/3})$
of the previous version), using the exact cancellation between
the PSWE residual scale $c^{1/3}$ and the spectral gap scale $c^{2/3}$.
\end{abstract}

%% ============================================================
\section{The Spectral Identification Principle}
\label{sec:specid}
%% ============================================================

\begin{definition}[Prolate differential operator]
\label{def:Lc}
The \emph{prolate differential operator} $\Lc$ on $L^2([-1,1])$ is
\begin{equation}\label{eq:Lc}
  \Lc\psi := -\frac{d}{dx}\bigl[(1-x^2)\psi'\bigr] + c^2 x^2 \psi,
\end{equation}
with domain $H^2 \cap H^1_0$, and eigenvalues $\chi_0 < \chi_1 < \cdots$.
The PSWFs $\psi_n^{(c)}$ satisfy
$\Lc\psi_n = \chi_n\psi_n$ and $\Kc\psi_n = \lam{c}{n}\psi_n$
simultaneously.
\end{definition}

\begin{proposition}[Spectral Identification Principle]
\label{prop:specid}
For any $f = \sum_n c_n\psi_n^{(c)} \in L^2([-1,1])$ with $\norm{f}=1$
and $\mu\in\R$:
\begin{equation}\label{eq:specid}
  \norm{(\Kc - \mu)f}^2 = \sum_n |c_n|^2(\lam{c}{n}-\mu)^2.
\end{equation}
\end{proposition}

\begin{proof}
Parseval's identity applied to the shared PSWF basis.
\end{proof}

\begin{remark}[The PSWE$\to\Kc$ transfer is not a silent jump]
\label{rem:transfer}
Equation~\eqref{eq:specid} shows: controlling
$\norm{(\Lc-\chi_{n_k^*})f}$ (the PSWE residual)
automatically controls $\norm{(\Kc-\mu_k^{(c)})f}$,
provided the quasimode $f$ is spectrally concentrated
near $\psi_{n_k^*}$ and the eigenvalues $\lam{c}{n_k^*}$
approximate $\mu_k^{(c)}$ well.
The key numerical input is that the $\Lc$-gaps in the transition
regime are $\Theta(c^{2/3})$, which supplies the concentration bound
of Corollary~\ref{cor:concentration} below.
\end{remark}

\begin{corollary}[Sharp spectral concentration of Airy quasimodes]
\label{cor:concentration}
Let $f_k^{(c)}$ be the normalised quasimode of Paper~XV
and $n_k^* = n^*+k-1$.
Then:
\begin{equation}\label{eq:sharpconc}
   1 - \bigl|\langle f_k^{(c)}, \psi_{n_k^*}^{(c)}\rangle\bigr|^2
  = O(c^{-2/3}).
\end{equation}
\end{corollary}

\begin{proof}
Let $G_k := \chi_{n_k^*+1} - \chi_{n_k^*}$.
From PSWE spectral theory in the transition regime,
$G_k = \Theta(c^{2/3})$.
The PSWE residual of the Airy quasimode (in the $\Lc$-scaled variable)
is $\norm{(\Lc - \chi_{n_k^*})f_k^{(c)}} = O(c^{1/3})$
(the factor $c^{2/3}$ from rescaling $\Lc$ combined with
the $O(c^{-1/3})$ Airy-equation residual).
By the spectral projection bound,
\begin{equation}\label{eq:sharpderivation}
  1 - |\langle f_k^{(c)},\psi_{n_k^*}\rangle|^2
  \leq \frac{\norm{(\Lc-\chi_{n_k^*})f_k^{(c)}}^2}{G_k^2}
  = \frac{O(c^{2/3})}{\Theta(c^{4/3})}
  = O(c^{-2/3}).
\end{equation}
\end{proof}

\begin{remark}[Why the previous $O(c^{-1/3})$ was unnecessarily conservative]
\label{rem:sharpening}
The prior version of this corollary stated $O(c^{-1/3})$
as a conservative weakening of the same computation,
writing $O(c^{-2/3})$ as $O(c^{-1/3})$ for brevity.
However, the sharp rate $O(c^{-2/3})$ is actually stronger and
carries structural information: it reflects the precise cancellation
between the PSWE-residual scale $c^{1/3}$ and the spectral gap scale $c^{2/3}$,
with ratio $c^{1/3}/c^{2/3} = c^{-1/3}$ which after squaring
yields $c^{-2/3}$.
This is not just a numerical improvement: it shows that
the quasimode is spectrally concentrated with a rate
\emph{determined by the gap structure}, not merely by the
turning-point approximation alone.
\end{remark}

%% ============================================================
\section{Quasimode Stability Implies Form Convergence}
\label{sec:formconv}
%% ============================================================

Recall the scaled edge operator from Paper~XIV:
$B_c := c^{1/3}(\Kc - z_{n^*})|_{\mathrm{edge}}$
and the Airy eigenfunctions $u_k(\xi) = c_k\Ai(\xi-a_k)$.

\begin{theorem}[Form convergence on quasimode subspace]
\label{thm:formconv}
For any fixed $K \geq 1$ and $v = \sum_k \alpha_k u_k$ with $\norm{v}=1$:
\begin{equation}\label{eq:formconv}
  \bigl|\langle B_c v_c, v_c\rangle - \langle \cA v, v\rangle\bigr|
  \leq C_K c^{-1/6},
\end{equation}
where $v_c$ is the natural image of $v$ in
$\mathrm{span}\{f_1^{(c)},\ldots,f_K^{(c)}\}$,
normalised to unit norm.
\end{theorem}

\begin{proof}
\textbf{Diagonal terms.}
From the Spectral ID Principle and the
$O(c^{-1/2})$ quasimode residuals of Paper~XV:
\[
  \langle B_c f_k^{(c)}, f_k^{(c)}\rangle
  = c^{1/3}(\lam{c}{n_k^*} - z_{n^*}) + O(c^{1/3}\cdot c^{-1})
  = -a_k + O(c^{-1/3}).
\]

\textbf{Off-diagonal terms.}
For $j\neq k$:
$|\langle B_c f_k^{(c)}, f_j^{(c)}\rangle|
= c^{1/3}|\langle(\Kc-z_{n^*})f_k^{(c)}, f_j^{(c)}\rangle|
\leq c^{1/3}\varepsilon(c) = c^{1/3}\cdot O(c^{-1/2}) = O(c^{-1/6}).$

\textbf{Gram correction.}
The approximate orthonormality of $\{f_k^{(c)}\}$
(Paper~XV, Lemma~2.2, error $O(c^{-1/6})$)
enters the renormalisation $v \mapsto v_c$ with error $O(c^{-1/6})$.
Combining gives~\eqref{eq:formconv}.
\end{proof}

\begin{remark}[Unconditional BR1 on Airy subspaces]
Theorem~\ref{thm:formconv} is condition~(BR1) of Paper~XIV
restricted to finite-dimensional Airy subspaces, proved unconditionally.
The rate $c^{-1/6}$ is determined by the off-diagonal residual
$c^{1/3} \cdot \varepsilon(c)$ with $\varepsilon(c) = O(c^{-1/2})$.
\end{remark}

\section{Form Convergence to Strong Resolvent Convergence}
\label{sec:strongres}
%% ============================================================

\begin{assumption}[$L^2$ kernel convergence with rate --- the switch point]
\label{ass:kernelrate}
The kernel of the scaled operator satisfies
\begin{equation}\label{eq:kernelrate}
  \bigl|K_c^{(\mathrm{sc})}(\xi,\eta) - K_{\cA}(\xi,\eta)\bigr|
  \leq g(c)(1+\xi)^{\alpha}(1+\eta)^{\alpha}
  \quad \text{a.e.\ on } \R_+^2,
\end{equation}
for some $g(c) \to 0$ and $\alpha < 1/2$.
This is called the \emph{switch point assumption} (see Section~\ref{sec:switch}).
\end{assumption}

\begin{theorem}[Strong resolvent convergence --- conditional]
\label{thm:strongres}
Assume Assumption~\ref{ass:kernelrate}.
Then for every $u\in L^2(\R_+)$ and $\zeta\in\C\setminus\R$:
\begin{equation}\label{eq:strongres}
  \norm{(B_c-\zeta)^{-1}u - (\cA-\zeta)^{-1}u} \to 0
  \quad (c\to\infty).
\end{equation}
\end{theorem}

\begin{proof}
By Kato's form theorem~\cite{Kato1966},
convergence of sesquilinear forms on a form-core of $\cA$
implies strong resolvent convergence.
Theorem~\ref{thm:formconv} supplies form convergence on
the span of Airy eigenfunctions, which is dense in the form domain.
Assumption~\ref{ass:kernelrate} provides the tail estimate
that upgrades finite-dimensional convergence to the full
$L^2(\R_+)$ setting.
\end{proof}

%% ============================================================
\section{The Schaltpunkt: Local Airy Control vs.\ Global Integrability}
\label{sec:switch}
%% ============================================================

\begin{remark}[The $L^2$ kernel rate as decisive switch point]
\label{rem:switch}
The operator-theoretic hierarchy of this programme is:
\begin{equation}\label{eq:hierarchy}
  \underbrace{\text{quasimodes}}_{\text{local, variational}}
  \;\Rightarrow\;
  \underbrace{\text{form convergence}}_{\text{energetic, finite-dim}}
  \;\Rightarrow\;
  \underbrace{L^2\text{ kernel rate}}_{\text{global, integrable}}
  \;\Rightarrow\;
  \underbrace{\text{strong resolvent}}_{\text{operator-level, BR1}}
  \;\Rightarrow\;
  \underbrace{\text{trace-norm (BR3)}}_{\text{Schatten class}}
\end{equation}
This is not a chain of increasingly technical estimates.
Each arrow is a jump into a strictly larger function space:
\begin{itemize}[nosep]
  \item Quasimode $\to$ form: from pointwise vector tests
    to energy-level stability on finite subspaces.
  \item Form $\to$ $L^2$ kernel: from
    $L^2_{\mathrm{local}}$ (Airy scale $\xi\in[0,R_0]$) to
    $L^2(\R_+^2)$ (global integrability over the edge domain).
    This is the last step fully inside the PSWE/Airy regime.
  \item $L^2$ kernel $\to$ strong resolvent:
    global $L^2$ kernel control implies $H^0 \to H^0$ operator control.
  \item Strong resolvent $\to$ trace-norm:
    from $H^0 \to H^0$ to $\mathcal{S}_1$
    (first Schatten class), a categorical jump
    that is not available from purely local asymptotics.
\end{itemize}
\end{remark}

\begin{proposition}[Metamathematical status of the switch point]
\label{prop:switch}
The statements~(S1)--(S4) below are ordered strictly by logical strength:
\begin{enumerate}[nosep,label=(S\arabic*)]
  \item Paper~XV: $\exists$ at least two eigenvalues of $\Kc$ in the
    Airy window (local spectral production).
  \item Paper~XVI, Theorem~\ref{thm:formconv}:
    quadratic form convergence on finite Airy subspaces
    (local energy stability).
  \item Assumption~\ref{ass:kernelrate}, if proved:
    global $L^2$ kernel rate; BR1-global via Theorem~\ref{thm:strongres}.
  \item Problem~\ref{prob:BR3} below:
    weighted $L^2$ kernel convergence;
    trace-norm resolvent convergence (BR3).
\end{enumerate}
Moreover, (S2) does not imply (S3) and (S3) does not imply (S4).
Each implication requires a strictly stronger input.
The passage (S2)$\to$(S3) is the transition from local Airy asymptotics
to global integrability.
The passage (S3)$\to$(S4) is a categorical jump from energy to Schatten class.
\end{proposition}

\begin{proof}
The strict non-implications: (S2)$\not\Rightarrow$(S3)
because finite-dimensional form convergence does not control
the $L^2$ tail of the kernel (no summability in the global variable
follows from local Airy matching).
(S3)$\not\Rightarrow$(S4) because strong resolvent convergence
is a norm statement in $B(L^2,L^2)$,
strictly weaker than trace-norm convergence
(the identity operator on an infinite-dimensional Hilbert space
is a standard counterexample: it satisfies any eigenvalue equation
strongly but is not trace class).
\end{proof}

%% ============================================================
\section{The Remaining Gap: Strong vs.\ Trace-Norm}
\label{sec:gap}
%% ============================================================

\begin{remark}[The gap between BR1 and BR3 is structural]
\label{rem:gap}
Strong resolvent convergence does \emph{not} imply trace-norm convergence.
The gap is:
\begin{equation}\label{eq:gapeq}
  \norm{(B_c-\zeta)^{-1} - (\cA-\zeta)^{-1}}_{\tr}
  \leq \norm{(B_c-\cA)(\cA-\zeta)^{-2}}_{\tr}
  \cdot \norm{(B_c-\zeta)^{-1}}_{\op}
  \cdot \norm{\cA-\zeta}_{\op}.
\end{equation}
For this to be $O(c^{-\alpha})$, one needs a \emph{weighted $L^2$
kernel convergence}: for $w(\xi) = (1+\xi)^{-\beta}$, $\beta>0$
sufficiently large,
\begin{equation}\label{eq:BR3cond}
  \int_0^\infty\int_0^\infty
  |K_{B_c}(\xi,\eta) - K_{\cA}(\xi,\eta)|^2 w(\xi)w(\eta)\,d\xi\,d\eta
  \;\to\; 0.
\end{equation}
This is strictly stronger than the unweighted rate
of Assumption~\ref{ass:kernelrate}.
\end{remark}

\begin{problem}[BR3: weighted kernel convergence]
\label{prob:BR3}
Prove~\eqref{eq:BR3cond} with explicit rate.
\begin{enumerate}[nosep,label=(\roman*)]
  \item \emph{Near-diagonal:}
    $|K_{B_c}(\xi,\eta) - K_{\cA}(\xi,\eta)| \leq g(c)h(\xi,\eta)$
    with $h\in L^2(w\otimes w)$.
  \item \emph{Off-diagonal decay:}
    Paley--Wiener / Bessel-function control of $K_{B_c}$
    for $|\xi-\eta|\gg c^{-1/3}$.
\end{enumerate}
This subsumes all remaining difficulty in the Airy edge programme.
\end{problem}

\begin{remark}[Single bottleneck structure]
\label{rem:bottleneck}
After Papers~XIII--XVI, the full remaining difficulty of the
Airy edge programme is concentrated in a \emph{single object}:
the weighted kernel error term
$|K_{B_c}(\xi,\eta)-K_{\cA}(\xi,\eta)|\cdot w(\xi)^{1/2}w(\eta)^{1/2}$
and its $L^2(\R_+^2)$ norm.

There are no further distributed obstructions:
\begin{itemize}[nosep]
  \item Quasimode construction: complete (Paper~XV).
  \item Spectral ID principle: explicit (this paper, \S\ref{sec:specid}).
  \item Form convergence on Airy subspaces: proved (Theorem~\ref{thm:formconv}).
  \item Switch-point structure: identified and characterised
    (Section~\ref{sec:switch}).
  \item BR3 reduction to weighted kernel: explicit (Problem~\ref{prob:BR3}).
\end{itemize}
This is a single-bottleneck structure in the sense of
operator ideal theory:
all difficulty has been funnelled into one Schatten-class
integrability question.
This is a sign of structural coherence, not of difficulty:
a single concentrated obstruction is analytically more tractable
than a distributed system of partial obstructions.
\end{remark}
